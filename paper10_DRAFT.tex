\begin{abstract}
We study the concentration operators
$\Hc = P^*\mathcal{H}_c^{[0,1]}P$,
where $\mathcal{H}_c^{[0,1]}$ is the
classical Slepian--Pollak
time-band-limiting operator on
$L^2([0,1])$.
The family $\{\Hc\}_{c>0}$ consists
of uniformly bounded,
monotonically increasing,
compact self-adjoint operators on
$L^2(\mathbb{R})$.

The foundational identity
(Lemma~\ref{lem:ftb})
\[
  \ip{\Hc f}{f} = \|\Bc(P^*Pf)\|^2
\]
reduces the entire operator family to
a Fourier band projection conjugated
by time-restriction.
All results in this paper are
consequences of this single identity
and the inclusion
$[-c,c]\subset[-c',c']$ for $c\le c'$;
we call it the \emph{coordinatisation
lemma} of the theory.

\textbf{Dual convergence structure.}
The identity gives rise to two
concurrent convergence threads:
\begin{enumerate}[(A)]
\item \textbf{Operator thread
      (SOT):}
      Uniform boundedness and density
      yield, via the SOT extension
      principle (Lemma~\ref{lem:sc}),
      a unique
      $\Hc\xrightarrow{\mathrm{SOT}}
      \Hstrong$.
\item \textbf{Form thread
      (Mosco):}
      Operator monotonicity and
      Hilbert--Schmidt compactness
      yield
      $\qc\xrightarrow{\mathrm{M}}\qlim$.
\end{enumerate}
The threads share a common origin
but are methodologically independent:
Thread~(A) uses uniform boundedness
and Cauchy nets;
Thread~(B) uses monotonicity and
compactness.
Identification $T_{\qlim}=\Hstrong$
(Theorem~\ref{thm:friedrichs})
yields strong resolvent convergence
(Corollary~\ref{cor:resolvent}).

\textbf{Bridge theorem.}
For a spectrally constructed
$\Hspectral$ (Paper~IX),
$\Hstrong=\overline{\Hspectral}$:
both sides are SOT limits of the
same directed family $(\Hc)_{c>0}$;
uniqueness in the Hausdorff SOT
topology on $\mathcal{B}(L^2)$
gives equality without eigenfunction
analysis.

\textbf{Scope.}
This paper is pure operator-limit
theory: no properties of limiting
eigenfunctions are proved; no
unconditional connection to
$\zeta(s)$-zeros is made.
\end{abstract}

\section{Setup and Foundational Lemmas}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

Let $\Fc$ denote the Fourier transform
$(\Fc f)(\xi)=
\int_{-\infty}^\infty f(x)
e^{-i\xi x}\,dx$,
with Plancherel isometry
$\|\Fc f\|_{L^2}=\sqrt{2\pi}\|f\|_{L^2}$
(\cite{ReedSimon1975}, Ch.~IX).

\begin{conv}[Operators]
\label{conv:ops}
\begin{alignat}{2}
  \Bc&:=\Fc^{-1}\chi_{[-c,c]}\Fc,
  &&\quad\Bc^*=\Bc=\Bc^2,\\
  P&:L^2(\mathbb{R})\to L^2([0,1]),
  &&\quad\|P\|=1,\\
  P^*&:L^2([0,1])\to L^2(\mathbb{R}),
  &&\quad\|P^*\|=1,\\
  \mathcal{H}_c^{[0,1]}
  &:L^2([0,1])\to L^2([0,1]),
  &&\quad K_c(x,y)
  =\tfrac{\sin c(x-y)}{\pi(x-y)},\\
  \Hc&:=P^*\mathcal{H}_c^{[0,1]}P,
  &&\quad K_c(x,x):=c/\pi.
\end{alignat}
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
$\Hc$ self-adjoint,
$0\le\Hc\le I$,
$\|\Hc\|\le 1$
(\cite{Slepian1961};
\cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[Scope]
\label{rem:scope}
Sections~\ref{sec:setup}--%
\ref{sec:friedrichs} use only
\eqref{F1} and
Lemmas~\ref{lem:ftb}--\ref{lem:dense}.
\end{remark}

\begin{lemma}[Coordinatisation lemma]
\label{lem:ftb}
For all $c>0$, $f\in L^2(\mathbb{R})$:
\begin{align}
  \label{eq:ftb}
  \ip{\Hc f}{f}&=\|\Bc(P^*Pf)\|^2,\\
  \label{eq:ftb-op}
  \mathcal{H}_c^{[0,1]}&=P\Bc P^*.
\end{align}
Every subsequent property of $\Hc$
follows from \eqref{eq:ftb} and
$[-c,c]\subset[-c',c']$.
\end{lemma}

\begin{proof}
The Fourier pair
$K_c(u)=\frac{\sin cu}{\pi u}$,
$(\Fc K_c)(\xi)=\chi_{[-c,c]}(\xi)$
(\cite{Slepian1961}, Eq.~(3))
gives $K_c * h = \Bc h$ for all
$h\in L^2$
(convolution theorem,
\cite{ReedSimon1975}, Thm.~IX.2).
For $g\in L^2([0,1])$,
$\tilde g:=P^*g$ vanishes outside
$[0,1]$, so
$(K_c*\tilde g)(x)
=\int_0^1 K_c(x-y)g(y)\,dy$;
thus $\mathcal{H}_c^{[0,1]}g=P\Bc P^*g$,
giving \eqref{eq:ftb-op}.
For $g=Pf$:
$\ip{\Hc f}{f}=\ip{\Bc(P^*Pf)}{P^*Pf}
=\|\Bc(P^*Pf)\|^2$.
\end{proof}

\begin{lemma}[Monotonicity ---
  Thread~B input]
\label{lem:monotone}
$0<c\le c'\Rightarrow
\ip{\Hc f}{f}\le
\ip{\mathcal{H}_{c'}f}{f}$.
\end{lemma}
\begin{proof}
$[-c,c]\subset[-c',c']\Rightarrow
\Bc\le B_{c'}$
(\cite{ReedSimon1975}, Ch.~VIII.1);
apply \eqref{eq:ftb}.
\end{proof}

\begin{lemma}[Hilbert--Schmidt ---
  Thread~B input]
\label{lem:hs-compact}
For fixed $c_0>0$,
$\widetilde{\mathcal{H}}_{c_0}$
is Hilbert--Schmidt, hence compact.
\end{lemma}
\begin{proof}
$\|\mathcal{H}_{c_0}^{[0,1]}
\|_{\mathrm{HS}}^2
\le\int_{\mathbb{R}}|K_{c_0}|^2
=c_0/\pi$
(Fubini + Parseval);
compactness of $\widetilde{\mathcal{H}}_{c_0}
=P^*\mathcal{H}_{c_0}^{[0,1]}P$:
\cite{ReedSimon1975}, Thm.~VI.22.
\end{proof}

\begin{lemma}[Dense domain ---
  Thread~A input]
\label{lem:dense}
$\mathcal{D}:=\{f\in L^2:\Hc f
\text{ converges in }L^2\}$
contains $C_c^\infty$, hence is dense.
\end{lemma}
\begin{proof}
For $f\in C_c^\infty$,
$\tilde g:=P^*Pf\in L^2$:
$\|\Hc f - P^*P\tilde g\|
\le\|\Bc\tilde g-\tilde g\|\to 0$
by $\Bc\xrightarrow{\mathrm{SOT}}I$
(\cite{ReedSimon1975}, Thm.~IX.14)
and $\|P^*P\|\le 1$.
\end{proof}

\begin{remark}[Logical status]
\label{rem:dense-trivial}
Once
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$,
one has $\mathcal{D}=L^2$;
Lemma~\ref{lem:dense} serves only
as pre-input to Thread~A.
\end{remark}

\section{Thread~A: Strong Operator
  Limit}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Thread~A profile]
\label{rem:threadA}
Inputs: \eqref{F1}, Lem.~\ref{lem:dense}.
Tool: Cauchy nets in complete Banach
space.
Output:
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$.
\end{remark}

\begin{lemma}[SOT extension principle]
\label{lem:sc}
Let $(A_c)_{c>0}$ be a directed net
in $\mathcal{B}(\mathcal{H})$ with
$M:=\sup_c\|A_c\|<\infty$, and
$\mathcal{D}\subset\mathcal{H}$ dense
with $\lim_c A_c g$ existing for
every $g\in\mathcal{D}$.
Then $\exists!\,A\in\mathcal{B}(\mathcal{H})$
with $\|A\|\le M$ and
$A_c\xrightarrow{\mathrm{SOT}}A$.
\end{lemma}
\begin{proof}
Given $f$ and $\varepsilon>0$,
choose $g\in\mathcal{D}$ with
$\|f-g\|<\varepsilon/(3M)$.
For $c,c'>c_0(g,\varepsilon)$:
$\|A_cf-A_{c'}f\|
\le 2M\|f-g\|+\|A_cg-A_{c'}g\|
<\varepsilon$;
so $(A_cf)$ is Cauchy, hence convergent.
$Af:=\lim_cA_cf$ is linear,
$\|A\|\le M$, unique
(\cite{ReedSimon1975}, Thm.~I.26).
\end{proof}

\begin{lemma}[Thread~A conclusion]
\label{lem:hlim-bounded}
$\exists!\,\Hstrong\in\mathcal{B}(L^2)$
with $\|\Hstrong\|\le 1$ and
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$.
\end{lemma}
\begin{proof}
Lem.~\ref{lem:sc} with $M=1$
(\eqref{F1}) and $\mathcal{D}$ from
Lem.~\ref{lem:dense}.
\end{proof}

\begin{corollary}
\label{cor:qlim-op}
$\qlim(f,f)=\ip{\Hstrong f}{f}$.
\end{corollary}

\begin{corollary}
\label{cor:sa-strong}
$\Hstrong=\Hstrong^*\ge 0$;
$\sigma(\Hstrong)\subset[0,1]$.
\end{corollary}

\section{Thread~B: Mosco Convergence
  of Forms}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Thread~B profile]
\label{rem:threadB}
Inputs: Lem.~\ref{lem:monotone},
Lem.~\ref{lem:hs-compact}.
Tools: monotone convergence in
$\mathbb{R}$; compactness.
Output: $\qc\xrightarrow{\mathrm{M}}
\qlim$.
Methodologically independent of
Thread~A (no Cauchy nets,
no $\mathcal{D}$); same FTB origin.
\end{remark}

\begin{definition}[Limit form]
\label{def:qlim}
Since $c\mapsto\ip{\Hc f}{f}$ is
monotone increasing
(Lem.~\ref{lem:monotone})
and bounded above by $\|f\|^2$
(\eqref{F1}), it converges in
$\mathbb{R}$ for each $f$;
define
\[
  \qlim(f,f):=
  \sup_{c>0}\ip{\Hc f}{f}
  =\lim_{c\to\infty}\ip{\Hc f}{f}.
\]
Extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc)\xrightarrow{\mathrm{M}}\qlim$ iff:
\begin{enumerate}[\rm(M1)]
\item $c_n\to\infty$,
      $f_n\rightharpoonup f\Rightarrow
      \qlim(f,f)\le
      \liminf_n q_{c_n}(f_n,f_n)$.
\item $\forall f$,
      $\exists c_n\nearrow\infty$,
      $f_n\to f$:
      $q_{c_n}(f_n,f_n)\to\qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Compact$+$weak
  $\Rightarrow$ form continuity]
\label{lem:wc}
Let $T\in\mathcal{B}(\mathcal{H})$
be compact and
$f_n\rightharpoonup f$.
Then $\ip{Tf_n}{f_n}\to\ip{Tf}{f}$.
\end{lemma}
\begin{proof}
$Tf_n\to Tf$ strongly
(\cite{ReedSimon1975}, Thm.~VI.1).
Since weakly convergent sequences
are bounded
(\cite{ReedSimon1975}, Thm.~II.12),
$\ip{Tf_n}{f_n}\to\ip{Tf}{f}$
by strong--weak continuity of the
inner product.
\end{proof}

\begin{theorem}[Thread~B conclusion:
  Mosco convergence]
\label{thm:mosco}
$\qc\xrightarrow{\mathrm{M}}\qlim$.
\end{theorem}

\begin{proof}
\textit{(M2).} Take $f_n=f$;
trivial from Def.~\ref{def:qlim}.

\textit{(M1).}
Let $c_n\to\infty$,
$f_n\rightharpoonup f$;
fix $c_0>0$.
For $c_n\ge c_0$:
$q_{c_n}(f_n,f_n)\ge q_{c_0}(f_n,f_n)$
(Lem.~\ref{lem:monotone}).
$\widetilde{\mathcal{H}}_{c_0}$ compact
(Lem.~\ref{lem:hs-compact}),
so Lem.~\ref{lem:wc} gives
$q_{c_0}(f_n,f_n)\to q_{c_0}(f,f)$.
Hence $\liminf_n q_{c_n}(f_n,f_n)
\ge q_{c_0}(f,f)$;
taking $\sup_{c_0}$ yields
$\liminf_n q_{c_n}(f_n,f_n)
\ge\qlim(f,f)$.
\end{proof}

\section{Identification and Closedness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
Bounded symmetric forms are closed
(\cite{ReedSimon1975}, Thm.~VIII.15).
\end{proof}

\begin{theorem}[Unification:
  Riesz--Kato]
\label{thm:friedrichs}
$T_{\qlim}=\Hstrong$.
\end{theorem}
\begin{proof}
$\ip{\Hstrong f}{g}=\qlim(f,g)$
(Cor.~\ref{cor:qlim-op});
uniqueness of Riesz representative.
\end{proof}

\begin{remark}[Threads united]
Thread~A produced $\Hstrong$;
Thread~B produced $\qlim$;
Thm.~\ref{thm:friedrichs} identifies
them as the same object.
\end{remark}

\section{Quantitative Rate
  (External, Paper~VIII)}
\label{sec:rate}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}
No other section depends on
Thm.~\ref{thm:rate-ext}.
\end{remark}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
$0\le\qlim(f,f)-\qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$
for $c\ge c_0(\kappa)$.
\end{theorem}
\begin{proof}[Proof (external)]
Paper~VIII, Cor.~4.3:
$\|\Hstrong-\Hc\|
\le C_\kappa c^{-1/4}$
(operator-norm; strictly stronger
than Mosco).
\end{proof}

\begin{remark}[Obstruction]
\label{rem:no-indep-proof}
An internal proof requires uniform
eigenvector asymptotics;
mixing fixed-$c$ eigenbasis with
$c\to\infty$ weights is invalid.
See Problem~\ref{prob:rate}.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

Sole import point for Paper~IX.

\begin{enumerate}[(IX.a)]
\item\label{IXa}
      $\Hspectral f
      :=\sum_n
      \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}}
      \Phi_n^{(\infty)}$;
      densely defined, symmetric,
      closable
      (Paper~IX, Thms.~2.5,~3.2).
\item\label{IXb}
      $\|\Hspectral f\|^2\le\|f\|^2$
      on $\Dom(\Hspectral)$;
      hence
      $\overline{\Hspectral}
      \in\mathcal{B}(L^2)$,
      everywhere defined
      (\cite{ReedSimon1975},
      Thm.~I.7).
\item\label{IXc}
      Paper~IX, Rem.~2.6:
      $\Hc\xrightarrow{\mathrm{SOT}}
      \overline{\Hspectral}$,
      same directed family
      $(\Hc)_{c>0}$,
      same SOT topology on
      $\mathcal{B}(L^2)$.
\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong=\overline{\Hspectral}$
in $\mathcal{B}(L^2(\mathbb{R}))$.
\end{theorem}

\begin{proof}
By Lem.~\ref{lem:hlim-bounded} and
\eqref{IXc},
both $\Hstrong$ and
$\overline{\Hspectral}$ are SOT
limits of $(\Hc)_{c>0}$ in the same
topology.
The SOT topology on
$\mathcal{B}(L^2)$ is Hausdorff
(pointwise convergence on all
vectors; limits are unique);
hence $\Hstrong=\overline{\Hspectral}$.
\end{proof}

\begin{remark}[Proof structure]
No eigenfunction analysis or
dense-set comparison:
the argument is a categorical
identity in the Hausdorff SOT
topology on $\mathcal{B}(L^2)$.
Sole import from Paper~IX: \eqref{IXc}.
\end{remark}

\begin{remark}
$\Hlim:=\Hstrong
=\overline{\Hspectral}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For $z\notin[0,1]$:
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$.
\end{corollary}
\begin{proof}
Thm.~\ref{thm:mosco};
$\sigma(\Hlim)\subset[0,1]$
(Cor.~\ref{cor:sa-strong});
\cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
Without hypothesis:
$\Hlim^*=\Hlim\ge 0$;
$\|\Hlim\|\le 1$;
$T_{\qlim}=\Hlim$;
$\qc\xrightarrow{\mathrm{M}}\qlim$;
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$;
$\Hstrong=\overline{\Hspectral}$;
rate $O(c^{-1/4})$ (Paper~VIII).
\end{remark}

\begin{remark}[Speculative framework:
  Riemann zeros]
\label{rem:rh}
\emph{Speculative and programmatic;
no unconditional claim.}

Under two unverified hypotheses:
\begin{enumerate}[(H1)]
\item $\{\Phi_n^{(\infty)}\}$
      complete; $\Hlim$ ess.\
      self-adjoint
      (Paper~IX, Hyp.~4.1).
\item Zeros $\rho$ in
      $\sigma_{\mathrm{app}}(\Hlim)$
      (\cite{CCM2025}, Thm.~1.3).
\end{enumerate}
(H1)--(H2) give
$\rho\in\sigma(\Hlim)
\subset\mathbb{R}$,
hence $\mathrm{Re}(\rho)=1/2$.
(H2) has no proved operator
identification.
\end{remark}
