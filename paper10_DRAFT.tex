\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of compact
self-adjoint operators on $L^2(\mathbb{R})$.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent
stability (unconditional).}
The foundational identity
(Lemma~\ref{lem:ftb}, Slepian--Pollak
\cite{Slepian1961})
$\ip{\Hc f}{f} = \|\Bc(P^*Pf)\|^2$
reduces the PSWF operator to a Fourier
band projection conjugated by
time-restriction.
From this identity, operator monotonicity
and Hilbert--Schmidt compactness are
derived; the SOT extension principle
(Lemma~\ref{lem:sc}) then yields
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$
on all of $L^2(\mathbb{R})$.
Mosco convergence, closedness of
$\qlim$, and strong resolvent
convergence follow.
A quantitative rate $O(c^{-1/4})$
is stated as an external theorem
(Paper~VIII, operator-norm level).

\textbf{(B) Bridge theorem (unconditional).}
$\Hstrong = \overline{\Hspectral}$
follows from uniqueness of the SOT limit
of the family $(\Hc)$.

The conditional connection to zeros of
$\zeta(s)$ (Remark~\ref{rem:rh}) is
purely programmatic;
no unconditional claim is made.
\end{abstract}

\section{Setup and Foundational Lemmas}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

Let $\Fc$ denote the Fourier transform
$(\Fc f)(\xi)=
\int_{-\infty}^\infty f(x)
e^{-i\xi x}\,dx$,
with Plancherel isometry
$\|\Fc f\|_{L^2}=\sqrt{2\pi}\|f\|_{L^2}$
(\cite{ReedSimon1975}, Ch.~IX).

\begin{conv}[Operators]
\label{conv:ops}
\begin{alignat}{2}
  \Bc&:=\Fc^{-1}\chi_{[-c,c]}\Fc
  &&\;(\Bc^*=\Bc=\Bc^2),\\
  P&:L^2(\mathbb{R})\to L^2([0,1]),
  &&\;\text{restriction; }\|P\|=1,\\
  P^*&:L^2([0,1])\to L^2(\mathbb{R}),
  &&\;\text{extension by zero;
  }\|P^*\|=1,\\
  \mathcal{H}_c^{[0,1]}
  &:L^2([0,1])\to L^2([0,1]),
  &&\;K_c(x,y)
  =\tfrac{\sin c(x-y)}{\pi(x-y)},\\
  \Hc&:=P^*\mathcal{H}_c^{[0,1]}P
  &&\;\in\mathcal{B}(L^2(\mathbb{R})).
\end{alignat}
$K_c(x,x):=c/\pi$.
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
$\Hc$ is self-adjoint,
$0\le\Hc\le I$,
$\|\Hc\|\le 1$
(\cite{Slepian1961};
\cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[Scope]
\label{rem:scope}
Sections~\ref{sec:setup}--%
\ref{sec:friedrichs} use only
\eqref{F1} and
Lemmas~\ref{lem:ftb}--\ref{lem:dense}.
No results from Papers~VIII or~IX
appear.
\end{remark}

%── lem:ftb
\begin{lemma}[Fourier--time--band
  representation]
\label{lem:ftb}
For all $c>0$, $f\in L^2(\mathbb{R})$:
\begin{align}
  \label{eq:ftb}
  \ip{\Hc f}{f}
  &=\|\Bc(P^*Pf)\|^2,\\
  \label{eq:ftb-op}
  \mathcal{H}_c^{[0,1]}
  &=P\,\Bc\,P^*.
\end{align}
\end{lemma}

\begin{proof}
\textit{Step~1: Convolution identity.}
$K_c(u)=\frac{\sin cu}{\pi u}
\leftrightarrow
(\Fc K_c)(\xi)=\chi_{[-c,c]}(\xi)$
(\cite{Slepian1961}, Eq.~(3);
\cite{ReedSimon1975}, Ex.~IX.7).
Convolution theorem
(\cite{ReedSimon1975}, Thm.~IX.2):
\begin{equation}\label{eq:conv-bc}
  K_c*h=\Bc h
  \quad\forall h\in L^2(\mathbb{R}).
\end{equation}

\textit{Step~2: Restriction.}
For $g\in L^2([0,1])$,
$\tilde g:=P^*g$.
Since $\tilde g=0$ a.e.\ outside
$[0,1]$:
$(K_c*\tilde g)(x)
=\int_0^1 K_c(x-y)g(y)\,dy
=(\mathcal{H}_c^{[0,1]}g)(x)$
for $x\in[0,1]$.
With \eqref{eq:conv-bc}:
$\mathcal{H}_c^{[0,1]}g
=P(\Bc\tilde g)=P\Bc P^*g$,
giving \eqref{eq:ftb-op}.

\textit{Step~3.}
For $g=Pf$:
$\ip{\Hc f}{f}
=\ip{P\Bc P^*Pf}{Pf}
=\ip{\Bc\tilde g}{\tilde g}
=\|\Bc\tilde g\|^2$.
\end{proof}

%── lem:monotone
\begin{lemma}[Operator monotonicity]
\label{lem:monotone}
$0<c\le c'\Rightarrow
\ip{\Hc f}{f}\le
\ip{\mathcal{H}_{c'}f}{f}$.
\end{lemma}
\begin{proof}
$\|\Bc(P^*Pf)\|^2\le
\|B_{c'}(P^*Pf)\|^2$:
$\Bc\le B_{c'}$ since
$[-c,c]\subset[-c',c']$
(\cite{ReedSimon1975}, Ch.~VIII.1).
\end{proof}

%── lem:hs-compact
\begin{lemma}[Hilbert--Schmidt]
\label{lem:hs-compact}
For fixed $c_0>0$:
$\Hc[c_0]$ is Hilbert--Schmidt,
hence compact.
\end{lemma}
\begin{proof}
$\|\mathcal{H}_{c_0}^{[0,1]}
\|_{\mathrm{HS}}^2
=\int_0^1\int_0^1
|K_{c_0}(x-y)|^2\,dx\,dy
\le\int_{-\infty}^\infty
|K_{c_0}(u)|^2\,du
=c_0/\pi$
(Fubini + Parseval;
$u=x-y$, Jacobian $1$).
Compactness of $\Hc[c_0]=P^*
\mathcal{H}_{c_0}^{[0,1]}P$:
\cite{ReedSimon1975}, Thm.~VI.22.
\end{proof}

%── lem:dense (with remark on triviality)
\begin{lemma}[Density of $\mathcal{D}$]
\label{lem:dense}
Define
$\mathcal{D}:=\{f\in L^2(\mathbb{R}):
\Hc f \text{ converges in } L^2
\text{ as }c\to\infty\}$.
Then $C_c^\infty(\mathbb{R})
\subset\mathcal{D}$;
in particular $\mathcal{D}$ is dense.
\end{lemma}

\begin{proof}
For $f\in C_c^\infty$,
set $\tilde g:=P^*Pf$.
By \eqref{eq:ftb}:
$\Hc f = P^*P(\Bc\tilde g)$.
Since $\Bc\xrightarrow{\mathrm{SOT}}I$
(\cite{ReedSimon1975}, Thm.~IX.14)
and $P^*P\in\mathcal{B}(L^2)$:
$\|\Hc f-P^*P\tilde g\|
\le\|\Bc\tilde g-\tilde g\|\to 0$.
\end{proof}

\begin{remark}[Logical status of
  Lemma~\ref{lem:dense}]
\label{rem:dense-trivial}
Lemma~\ref{lem:dense} is used
only as input to
Lemma~\ref{lem:sc} (Step~1 of the
SOT-extension argument).
Once Lemma~\ref{lem:sc} is applied
and $\Hc\xrightarrow{\mathrm{SOT}}
\Hstrong$ is established on all
of $L^2(\mathbb{R})$, the set
$\mathcal{D}$ equals $L^2(\mathbb{R})$
and the density claim is
automatically superseded.
In particular, the lemma does not
add mathematical depth beyond what
the SOT-extension principle already
delivers; it is included to make
the logical steps fully explicit.
\end{remark}

\section{Strong-Limit Operator}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[SOT extension
  principle]
\label{lem:sc}
Let $(A_c)_{c>0}$ be a
\emph{directed net} (ordered by
$c\to\infty$) in
$\mathcal{B}(\mathcal{H})$
with $M:=\sup_c\|A_c\|<\infty$.
Let $\mathcal{D}\subset\mathcal{H}$
be dense with
$\lim_{c\to\infty}A_c g$ existing
for every $g\in\mathcal{D}$.
Then there exists a unique
$A\in\mathcal{B}(\mathcal{H})$ with
$\|A\|\le M$ and
$A_c\xrightarrow{\mathrm{SOT}}A$
(i.e., $A_c f\to Af$ for all
$f\in\mathcal{H}$ in the directed-net
sense).
\end{lemma}

\begin{proof}
Fix $f\in\mathcal{H}$ and
$\varepsilon>0$.
Choose $g\in\mathcal{D}$ with
$\|f-g\|<\varepsilon/(3M)$.
For $c,c'$ large:
\[
  \|A_c f-A_{c'}f\|
  \le 2M\|f-g\|
  +\|A_c g-A_{c'}g\|
  < \tfrac{2\varepsilon}{3}
  +\tfrac{\varepsilon}{3}=\varepsilon.
\]
(Here $c,c'$ are elements of the
directed set $(0,\infty)$;
``large'' means above some
$c_0(g,\varepsilon)$.)
Hence $(A_c f)$ is a
Cauchy net in the complete space
$\mathcal{H}$;
$Af:=\lim_c A_c f$ exists.
Linearity, $\|A\|\le M$, and
uniqueness:
\cite{ReedSimon1975}, Thm.~I.26.
\end{proof}

\begin{lemma}[Existence of
  $\Hstrong$]
\label{lem:hlim-bounded}
$\Hc\xrightarrow{\mathrm{SOT}}
\Hstrong$ for a unique
$\Hstrong\in\mathcal{B}(L^2)$
with $\|\Hstrong\|\le 1$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:sc} with
$\mathcal{D}$ from
Lemma~\ref{lem:dense},
$M=1$ from \eqref{F1}.
\end{proof}

\begin{corollary}
\label{cor:qlim-op}
$\qlim(f,f)=\ip{\Hstrong f}{f}$.
\end{corollary}

\begin{corollary}
\label{cor:sa-strong}
$\Hstrong=\Hstrong^*\ge 0$;
$\sigma(\Hstrong)\subset[0,1]$.
\end{corollary}

\section{Limit Form and Mosco
  Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
$\qlim(f,f):=
\sup_{c>0}\ip{\Hc f}{f}
=\lim_{c\to\infty}\ip{\Hc f}{f}$;
extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc)\xrightarrow{\mathrm{M}}\qlim$
iff:
\begin{enumerate}[\rm(M1)]
\item $c_n\to\infty$,
      $f_n\rightharpoonup f\Rightarrow
      \qlim(f,f)\le
      \liminf_n q_{c_n}(f_n,f_n)$.
\item $\forall f$,
      $\exists c_n\to\infty$,
      $f_n\to f$ strongly:
      $q_{c_n}(f_n,f_n)\to\qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Compact$+$weak
  $\Rightarrow$ form continuity]
\label{lem:wc}
Let $T\in\mathcal{B}(\mathcal{H})$
be compact,
$f_n\rightharpoonup f$.
Then:
(a) $Tf_n\to Tf$ strongly
(\cite{ReedSimon1975}, Thm.~VI.1);
(b) $\ip{Tf_n}{f_n}\to\ip{Tf}{f}$
(\cite{ReedSimon1975}, Lem.~II.3).
\end{lemma}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc\xrightarrow{\mathrm{M}}\qlim$.
\end{theorem}

\begin{proof}
\textit{(M2).}
$f_n:=f$: trivial from
Def.~\ref{def:qlim}.

\textit{(M1): two steps.}
Let $c_n\to\infty$,
$f_n\rightharpoonup f$.
For fixed $c_0>0$,
once $c_n\ge c_0$:
$q_{c_n}(f_n,f_n)
\ge q_{c_0}(f_n,f_n)$
(Lem.~\ref{lem:monotone}).
$\Hc[c_0]$ compact
(Lem.~\ref{lem:hs-compact}),
so by Lem.~\ref{lem:wc}(b):
$q_{c_0}(f_n,f_n)\to q_{c_0}(f,f)$.
Hence
$\liminf_n q_{c_n}(f_n,f_n)
\ge q_{c_0}(f,f)$.
Taking $\sup_{c_0}$:
$\liminf_n q_{c_n}(f_n,f_n)
\ge\qlim(f,f)$.
\end{proof}

\section{Closedness and
  Riesz--Kato Identification}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
Since $\Hstrong$ is a bounded
symmetric operator,
the form $\qlim(f,g)
=\ip{\Hstrong f}{g}$ is
sectilinear, symmetric, and
bounded;
bounded symmetric forms are
automatically closed
(\cite{ReedSimon1975},
Thm.~VIII.15).
Alternatively:
$\|f\|_{\qlim}^2
=\|f\|^2+\ip{\Hstrong f}{f}
\le 2\|f\|^2$
and $\|f\|^2\le\|f\|_{\qlim}^2$;
so form and $L^2$ norms are
equivalent, giving closedness
directly.
\end{proof}

\begin{theorem}[Riesz--Kato
  identification]
\label{thm:friedrichs}
$T_{\qlim}=\Hstrong$.
\end{theorem}
\begin{proof}
$\ip{\Hstrong f}{g}=\qlim(f,g)$
(Cor.~\ref{cor:qlim-op});
uniqueness of Riesz representative.
\end{proof}

\begin{remark}[Independence]
No other section depends on

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
For all $f$ and $c\ge c_0(\kappa)$:
$0\le\qlim(f,f)-\qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{theorem}
\begin{proof}[Proof (external)]
Lower bound: Lem.~\ref{lem:monotone}.
Upper bound: Paper~VIII, Cor.~4.3
provides $\|\Hstrong-\Hc\|
\le C_\kappa c^{-1/4}$
at the \emph{operator-norm} level;
apply to $\ip{(\Hstrong-\Hc)f}{f}$.
\end{proof}

\begin{remark}[Level of the rate]
\label{rem:rate-level}
Paper~VIII delivers
operator-norm control,
strictly stronger than Mosco-level.
Thm.~\ref{thm:mosco} is logically
independent of Thm.~\ref{thm:rate-ext}.
\end{remark}

\begin{remark}[Obstruction]
\label{rem:no-indep-proof}
An internal proof would require
uniform eigenvector asymptotics;
mixing fixed-$c$ basis with
$c\to\infty$ weights is invalid.
See Problem~\ref{prob:rate}.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the sole import
point for Paper~IX.

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      Paper~IX, Def.~2.3:
      $\Hspectral f
      :=\sum_n
      \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}}
      \Phi_n^{(\infty)}$;
      densely defined, symmetric,
      closable (Paper~IX,
      Thms.~2.5,~3.2).

\item\label{IXb}
      \textit{Bounded closure.}
      $\|\Hspectral f\|^2\le\|f\|^2$
      on $\Dom(\Hspectral)$
      (Bessel + $\lambda_n^{(\infty)}
      \in[0,1]$);
      hence $\overline{\Hspectral}
      \in\mathcal{B}(L^2)$,
      everywhere defined
      (\cite{ReedSimon1975},
      Thm.~I.7).

\item\label{IXc}
      \textit{SOT limit.}
      Paper~IX, Rem.~2.6:
      $\Hc\xrightarrow{\mathrm{SOT}}
      \overline{\Hspectral}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong=\overline{\Hspectral}$
in $\mathcal{B}(L^2(\mathbb{R}))$.
\end{theorem}

\begin{proof}
By Lem.~\ref{lem:hlim-bounded}:
$\Hc\xrightarrow{\mathrm{SOT}}
\Hstrong$.
By \eqref{IXc}:
$\Hc\xrightarrow{\mathrm{SOT}}
\overline{\Hspectral}$.
SOT limits are unique: SOT
convergence means pointwise
convergence on every $f\in L^2$;
if two operators $A,B$ satisfy
$A_c f\to Af$ and $A_c f\to Bf$
for all $f$, then $(A-B)f=0$
for all $f$, so $A=B$
(since $L^2$ is Hausdorff).
Hence $\Hstrong
=\overline{\Hspectral}$.
\end{proof}

\begin{remark}[Logical structure]
The only import from Paper~IX
is \eqref{IXc}.
The proof uses solely
SOT-uniqueness; no dense-set
agreement argument is needed.
\end{remark}

\begin{remark}
$\Hlim:=\Hstrong
=\overline{\Hspectral}$.
\end{remark}

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For $z\notin[0,1]$:
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$.
\end{corollary}
\begin{proof}
Mosco (Thm.~\ref{thm:mosco});
$\sigma(\Hlim)\subset[0,1]$
(Cor.~\ref{cor:sa-strong});
\cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional
  results]
Without any hypothesis:
$\Hlim=\Hlim^*\ge 0$;
$\|\Hlim\|\le 1$;
$T_{\qlim}=\Hlim$;
$\qc\xrightarrow{\mathrm{M}}\qlim$;
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$;
$\Hstrong=\overline{\Hspectral}$.
External (operator-norm):
rate $O(c^{-1/4})$.
\end{remark}

\begin{remark}[Speculative
  framework: Riemann zeros]
\label{rem:rh}
This remark is a
\emph{speculative, programmatic}
framework; it contains no proved
result and no implication from the
preceding theorems to zeta zeros
is claimed.

Two unverified hypotheses:
\begin{enumerate}[(H1)]
\item Paper~IX, Hyp.~4.1:
      $\{\Phi_n^{(\infty)}\}$
      complete;
      $\Hlim$ essentially
      self-adjoint.
\item \cite{CCM2025}, Thm.~1.3:
      non-trivial zeros $\rho$
      lie in
      $\sigma_{\mathrm{app}}(\Hlim)$.
\end{enumerate}
Under (H1)--(H2):
$\rho\in\sigma(\Hlim)
\subset\mathbb{R}$,
so $\mathrm{Re}(\rho)=1/2$.
(H2) has no rigorous operator
identification between $\Hlim$
and any zeta-spectral object;
it is borrowed from the CCM
program without a proved bridge.
Readers should treat this as
motivational context only.
\end{remark}
