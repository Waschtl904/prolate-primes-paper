\maketitle

\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of self-adjoint
integral operators on $L^2(\mathbb{R})$.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent
stability (unconditional).}
We prove that the quadratic forms $\qc$
converge to a limit form $\qlim$ in the
sense of Mosco \cite{Mosco1969}, that
$\qlim$ is closed, and that the associated
resolvents converge strongly:
$(\Hc-z)^{-1} \xrightarrow{\mathrm{s}}
(\Hlim-z)^{-1}$ for every $z\notin[0,1]$.
These results are proved entirely within
the form framework, using only monotone
convergence and compactness of $\Hc$;
no spectral data from Paper~VIII or
Paper~IX is used in this part.
A quantitative rate $O(c^{-1/4})$ is
cited as an external theorem from
Paper~VIII (see Theorem~\ref{thm:rate-ext})
and stated without independent proof;
it is a consequence for the Mosco
convergence, not a prerequisite.

\textbf{(B) Bridge theorem (unconditional).}
The strong-limit operator $\Hstrong$
(constructed via density extension)
and the spectral-closure operator
$\Hspectral$ (constructed in Paper~IX)
are identical:
$\Hstrong = \overline{\Hspectral}$.
This is proved once, in isolation;
it is the \emph{only} place where
Paper~IX results are imported.

Conditionally on the spectral triple
of Connes--Consani--Moscovici
\cite{CCM2025} and Paper~IX,
Hypothesis~4.1, every non-trivial
zero of $\zeta(s)$ lies on the
critical line.
\end{abstract}

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.

\begin{conv}[Ambient space and operators]
\label{conv:tilde}
All operators act on $L^2(\mathbb{R})$.
The PSWF concentration operator
$\mathcal{H}_c^{[0,1]}$ on $L^2([0,1])$
is extended by
$\Hc := P^*\mathcal{H}_c^{[0,1]}P$,
where $P : L^2(\mathbb{R}) \to L^2([0,1])$
is restriction and
$P^* : L^2([0,1]) \to L^2(\mathbb{R})$
is extension by zero.
Then $\Hc \in \mathcal{B}(L^2(\mathbb{R}))$
with $0 \le \Hc \le I$.
\end{conv}

\textbf{Facts used in the form thread
(Sections~\ref{sec:setup}--%
\ref{sec:friedrichs}).}

\begin{enumerate}[(F.1)]

\item\label{F1}
      $0 \le \ip{\Hc f}{f} \le \|f\|^2$
      for all $f \in L^2$, all $c > 0$
      (\cite{Slepian1961}).

\item\label{F2}
      \textit{Monotone form.}
      The map $c \mapsto \ip{\Hc f}{f}$
      is non-decreasing and bounded by
      $\|f\|^2$ for every fixed $f$;
      hence
      \[
        \qlim(f,f)
        := \sup_{c>0}\ip{\Hc f}{f}
        = \lim_{c\to\infty}\ip{\Hc f}{f}
        \in [0,\|f\|^2]
      \]
      exists for every $f$
      (Paper~V, Prop.~2.2).
      \emph{No operator is introduced here.}

\item\label{F3}
      \textit{Dense strong-convergence
      domain.}
      Let $\mathcal{D} := \{f \in L^2 :
      \lim_{c\to\infty}\Hc f
      \text{ exists in } L^2\}$.
      All compactly supported $f$ lie
      in $\mathcal{D}$, so $\mathcal{D}$
      is dense (Paper~V, Prop.~2.2).

\item\label{F4}
      $\Hc$ is self-adjoint, positive,
      and $\|\Hc\| \le 1$
      (\cite{Slepian1961};
      \cite{ReedSimon1975}, Thm.~VI.1).

\item\label{F5}
      \textit{Compactness of fixed $\Hc$.}
      For each fixed $c_0 > 0$,
      $\Hc[c_0]$ is compact on
      $L^2(\mathbb{R})$.
      \textit{Proof:}
      $\mathcal{H}_{c_0}^{[0,1]}$ acts
      on $L^2([0,1])$ via the kernel
      $K_{c_0}(x,y)
      = \frac{\sin c_0(x-y)}{\pi(x-y)}$,
      which satisfies
      $\|K_{c_0}\|_{L^2([0,1]^2)}
      < \infty$
      (bounded domain, bounded kernel);
      hence $\mathcal{H}_{c_0}^{[0,1]}$
      is Hilbert--Schmidt on $L^2([0,1])$,
      thus compact.
      Since $P$ and $P^*$ are bounded,
      $\Hc[c_0] = P^* \mathcal{H}_{c_0}^{[0,1]}
      P$ is compact on $L^2(\mathbb{R})$
      (bounded $\times$ compact $\times$
      bounded; \cite{ReedSimon1975},
      Thm.~VI.12).

\end{enumerate}

\begin{remark}[What is \emph{not} used
  before the Bridge]
\label{rem:no-spectral-input}
\emph{The following objects do not
appear in
Sections~\ref{sec:setup}--%

\end{remark}

\section{Strong-Limit Operator
  \texorpdfstring{$\Hstrong$}{H-str}}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Density-extension principle]
\label{lem:sc}
Let $(A_c)_{c>0} \subset \mathcal{B}(\mathcal{H})$
with $M := \sup_c\|A_c\| < \infty$,
and let $\mathcal{D} \subset \mathcal{H}$
be a dense subspace on which
$A_c f \to Bf$ for every $f \in \mathcal{D}$.
Then $A_c f$ converges for every
$f \in \mathcal{H}$, the limit defines
$A \in \mathcal{B}(\mathcal{H})$
with $\|A\| \le M$, and
$A_c \to A$ strongly.
\end{lemma}

\begin{proof}
For any $f \in \mathcal{H}$ and
$\varepsilon > 0$, choose
$g \in \mathcal{D}$ with
$\|f - g\| < \varepsilon/(3M)$.
For large enough $c, c'$:
$\|A_c f - A_{c'} f\|
\le \|A_c(f-g)\|
+ \|A_c g - A_{c'} g\|
+ \|A_{c'}(g-f)\|
\le \varepsilon$.
Completeness gives the limit
$Af$, and $A_c \to A$ strongly.
\end{proof}

\begin{lemma}[Existence of $\Hstrong$]
\label{lem:hlim-bounded}
There exists a unique
$\Hstrong \in \mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hstrong\| \le 1$ and
$\Hc \to \Hstrong$ strongly.
\end{lemma}

\begin{proof}
Apply Lemma~\ref{lem:sc} with
$A_c := \Hc$, $\mathcal{D}$ and the
dense strong-convergence domain from
\eqref{F3}, $M := 1$ from \eqref{F4}.
\end{proof}

\begin{corollary}[Form equals operator]
\label{cor:qlim-op}
For every $f \in L^2(\mathbb{R})$:
$\qlim(f,f) = \ip{\Hstrong f}{f}$.
\end{corollary}
\begin{proof}
$\qlim(f,f)
= \lim_c\ip{\Hc f}{f}
= \ip{\Hstrong f}{f}$
by strong convergence and continuity
of the inner product.
\end{proof}

\begin{corollary}[Self-adjointness]
\label{cor:sa-strong}
$\Hstrong$ is self-adjoint and positive,
and $\sigma(\Hstrong) \subset [0,1]$.
\end{corollary}
\begin{proof}
Symmetry: pass to the strong limit in
$\ip{\Hc f}{g} = \ip{f}{\Hc g}$.
Since $\Hstrong$ is bounded and symmetric
on all of $L^2$, it is self-adjoint
(BSA principle;
\cite{ReedSimon1975}, Thm.~III.6).
Positivity and $\sigma \subset [0,1]$:
$0 \le \qlim(f,f) \le \|f\|^2$
(Corollary~\ref{cor:qlim-op}).
\end{proof}

\section{Limit Form and Mosco
  Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
$\qlim(f,f) :=
\lim_{c\to\infty}\ip{\Hc f}{f}$
(exists by \eqref{F2}); extend by
polarisation to a Hermitian form on
$L^2(\mathbb{R})$.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc) \xrightarrow{\mathrm{M}} \qlim$ if:
\begin{enumerate}[\rm(M1)]
\item For every $c_n \to \infty$
      and $f_n \rightharpoonup f$:
      $\qlim(f,f) \le
      \liminf_n q_{c_n}(f_n,f_n)$.
\item For every $f$, there exist
      $c_n \to \infty$ and $f_n \to f$
      strongly with
      $q_{c_n}(f_n,f_n) \to \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Weak continuity of
  fixed form]
\label{lem:wc}
For each fixed $c_0 > 0$:
if $f_n \rightharpoonup f$, then
$q_{c_0}(f_n,f_n) \to q_{c_0}(f,f)$.
\end{lemma}
\begin{proof}
Write $q_{c_0}(f_n,f_n)
- q_{c_0}(f,f)
= I_n + II_n$ with
$I_n := \ip{\Hc[c_0](f_n-f)}{f_n}$
and
$II_n := \ip{\Hc[c_0]f}{f_n-f}$.
$II_n \to 0$ by weak convergence.
By \eqref{F5}, $\Hc[c_0]$ is compact;
hence $\Hc[c_0](f_n - f) \to 0$
strongly (compactness maps weak to
strong convergence on differences
converging weakly to $0$;
\cite{ReedSimon1975}, Thm.~VI.1).
Since $\sup_n\|f_n\| < \infty$
(Banach--Steinhaus), $|I_n| \to 0$.
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc \xrightarrow{\mathrm{M}} \qlim$.
\end{theorem}
\begin{proof}
\textit{(M2).}
Take $f_n := f$, $c_n \to \infty$:
$q_{c_n}(f,f) \to \qlim(f,f)$
by definition \eqref{F2}.

\textit{(M1).}
Let $c_n \to \infty$,
$f_n \rightharpoonup f$.
For every fixed $c_0 > 0$,
once $c_n \ge c_0$,
monotonicity \eqref{F2} gives
$q_{c_n}(f_n,f_n) \ge q_{c_0}(f_n,f_n)$.
Hence:
\[
  \liminf_n q_{c_n}(f_n,f_n)
  \ge \liminf_n q_{c_0}(f_n,f_n)
  = q_{c_0}(f,f)
\]
by Lemma~\ref{lem:wc}.
Since this bound holds for every
fixed $c_0$, and the left side is
independent of $c_0$:
\[
  \liminf_n q_{c_n}(f_n,f_n)
  \ge \sup_{c_0>0} q_{c_0}(f,f)
  = \qlim(f,f). \qedhere
\]
\end{proof}

\section{Closedness and Riesz--Kato
  Identification}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of $\qlim$]
\label{thm:form-closed}
$\qlim$ is a closed form.
\end{theorem}
\begin{proof}
$\qlim(f,f) = \ip{\Hstrong f}{f}$
(Corollary~\ref{cor:qlim-op}) and
$\Hstrong \in \mathcal{B}(L^2)$;
the form norm
$\|f\|_{\qlim}^2 =
\|f\|^2 + \qlim(f,f)$
is equivalent to $\|f\|^2$,
so $\qlim$ is closed.
\end{proof}

\begin{theorem}[Riesz--Kato identification]
\label{thm:friedrichs}
$T_{\qlim} = \Hstrong$,
where $T_{\qlim}$ is the unique
bounded self-adjoint operator with
$\qlim(f,g) = \ip{T_{\qlim}f}{g}$
for all $f,g$.
\end{theorem}
\begin{proof}
By Corollary~\ref{cor:qlim-op},
$\ip{\Hstrong f}{g} = \qlim(f,g)$
for all $f, g$.
Uniqueness of the Riesz
representative gives
$T_{\qlim} = \Hstrong$.
\end{proof}

\section{Quantitative Rate
  (External Input from Paper~VIII)}
\label{sec:rate}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Status of this section]
\label{rem:rate-status}
This section contains no independent
proof.
The rate $O(c^{-1/4})$ is a direct
consequence of Paper~VIII;
it is stated here for completeness
and to make explicit what the Mosco
convergence yields quantitatively.
\emph{No result in
Sections~\ref{sec:setup}--%
\ref{sec:bridge} depends on this
section.}
\end{remark}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
For all $f \in L^2(\mathbb{R})$
and $c \ge c_0(\kappa)$:
\[
  0 \le \qlim(f,f) - \qc(f,f)
  \le C_\kappa\, c^{-1/4}\, \|f\|^2.
\]
\end{theorem}

\begin{proof}[Proof (external)]
The lower bound follows from monotonicity
\eqref{F2}.
The upper bound is a consequence of
Paper~VIII, Cor.~4.3,
which provides the operator-norm bound
\[
  \|\Hstrong - \Hc\| \le C_\kappa c^{-1/4}
\]
for $c \ge c_0(\kappa)$.
This bound is established in Paper~VIII
via PSWF eigenvalue asymptotics and
does not depend on any result of this
paper.
Given this bound:
\[
  \qlim(f,f) - \qc(f,f)
  = \ip{(\Hstrong - \Hc)f}{f}
  \le \|\Hstrong - \Hc\| \cdot \|f\|^2
  \le C_\kappa c^{-1/4} \|f\|^2. \qedhere
\]
\end{proof}

\begin{remark}[Why no independent proof]
\label{rem:no-indep-proof}
Any attempt to prove the upper bound
in Theorem~\ref{thm:rate-ext}
independently from within Paper~X
runs into the following obstruction:
one must pass from scalar eigenvalue
asymptotics of $\Hc$ (as $c \to \infty$)
to a uniform operator-norm bound,
but the eigenbasis $\{\phi_n^{(c)}\}$
of $\Hc$ changes with $c$.
Equations of the form
$\qlim(f,f) =
\sum_n \lambda_n^{(\infty)}
|\langle f,\phi_n^{(c)}\rangle|^2$
(with $c$ fixed in the basis but
$c \to \infty$ in the weights)
mix two non-commuting limits and
are not valid.
The correct proof requires controlling
the eigenvector asymptotics alongside
the eigenvalue asymptotics, which
is the content of Paper~VIII;
it is not reproduced here.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the \emph{only} section
that imports results from Paper~IX.

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      Paper~IX, Definition~2.3 constructs
      the operator
      $\Hspectral f :=
      \sum_{n \ge 0}
      \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}}
      \Phi_n^{(\infty)}$
      on its natural domain in $L^2$.
      Here $\{\Phi_n^{(\infty)}\}$ are
      the limit PSWF eigenvectors and
      $\lambda_n^{(\infty)}$ their
      corresponding limit eigenvalues.
      $\Hspectral$ is symmetric,
      densely defined, and closable
      (Paper~IX, Thm.~2.5, Thm.~3.2).

\item\label{IXb}
      $\lambda_n^{(\infty)} \in [0,1]$
      for all $n$, so
      $\overline{\Hspectral}
      \in \mathcal{B}(L^2)$
      with norm $\le 1$.

\item\label{IXc}
      For every $f \in \mathcal{D}$:
      $\Hc f \to \Hstrong f$
      (definition of $\Hstrong$),
      and also
      $\Hc f \to \overline{\Hspectral} f$
      (Paper~IX, Remark~2.6).
      Uniqueness of $L^2$-limits:
      $\Hstrong = \overline{\Hspectral}$
      on $\mathcal{D}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong = \overline{\Hspectral}$
as bounded operators on
$L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Both are bounded
(\eqref{IXb} and
Lemma~\ref{lem:hlim-bounded}).
They agree on the dense set
$\mathcal{D}$ (\eqref{IXc}).
Hence they are equal on all of $L^2$.
\end{proof}

\begin{remark}[After the bridge]
\label{rem:bridge}
We write $\Hlim :=
\Hstrong = \overline{\Hspectral}$.
The symbols $\lambda_n^{(\infty)}$
and $\Phi_n^{(\infty)}$ now refer to
the spectral data of $\Hspectral$
as in Paper~IX.
\end{remark}

\begin{remark}[$\Hlim$ is not a
  multiplication operator]
\label{rem:hlim-not-mult}
$\sigma(\Hlim) \subset
\overline{\{\lambda_n^{(\infty)}
: n \ge 0\}}$.
The Landau--Pollak transition accumulates
densely in $(0,1)$ (\cite{Slepian1961}),
so $\sigma(\Hlim) \not\subset \{0,1\}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every
$z \in \mathbb{C} \setminus [0,1]$:
$(\Hc - z)^{-1}
\xrightarrow{\mathrm{s}}
(\Hlim - z)^{-1}$.
\end{corollary}
\begin{proof}
Mosco convergence
(Theorem~\ref{thm:mosco}),
$\sigma(\Hlim) \subset [0,1]$
(Corollary~\ref{cor:sa-strong}),
and \cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
\label{rem:unconditional}
Without any hypothesis:
$\Hlim \in \mathcal{B}(L^2)$;
$\Hlim = \Hlim^* \ge 0$;
$\sigma(\Hlim) \subset [0,1]$;
$T_{\qlim} = \Hlim$;
$(\Hc-z)^{-1} \xrightarrow{\mathrm{s}}
(\Hlim-z)^{-1}$;
$\qc \xrightarrow{\mathrm{M}} \qlim$;
$\Hstrong = \overline{\Hspectral}$.
With the external input of Paper~VIII:
quantitative rate $O(c^{-1/4})$.
\end{remark}

\begin{corollary}[Conditional RH]
\label{cor:rh}
Conditionally on CCM
(\cite{CCM2025}, Thm.~1.3)
and Paper~IX, Hypothesis~4.1,
every non-trivial zero $\rho$ of
$\zeta(s)$ satisfies
$\mathrm{Re}(\rho) = \tfrac{1}{2}$.
\textit{No unconditional claim about
zeta zeros is made.}
\end{corollary}
\begin{proof}
Under Paper~IX, Hyp.~4.1,
$\Hlim$ is essentially self-adjoint
(Paper~IX, Cor.~5.2).
Under CCM:
$\rho \in \sigma_{\mathrm{app}}(\Hlim)
= \sigma(\Hlim) \subset \mathbb{R}$;
the CCM embedding forces
$\mathrm{Re}(\rho) = 1/2$.
\end{proof}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Completeness of
  limit eigensystem]
\label{prob:completeness}
Prove Paper~IX, Hypothesis~4.1:
$\{\Phi_n^{(\infty)}\}$ is complete
in $L^2(\mathbb{R})$, equivalently,
$\overline{\Hspectral}$ has no
deficiency vectors.
\end{problem}

\begin{problem}[Independent proof
  of rate]
\label{prob:rate}
Give an independent proof within
Paper~X of the bound
$\qlim(f,f) - \qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$.
This requires a uniform estimate
for $\|\Hstrong - \Hc\|$ that does
not use the eigenbasis of $\Hc$ as
a proxy for the eigenbasis of $\Hstrong$.
The key difficulty is identified in
Remark~\ref{rem:no-indep-proof}.
\end{problem}

\begin{problem}[Quantitative resolvent]
\label{prob:resolvent}
Find an explicit rate $F(c,z) \to 0$
with
$\|(\Hc-z)^{-1}-(\Hlim-z)^{-1}\|
\le F(c,z)$.
\end{problem}

\begin{problem}[Spectrum of $\Hlim$]
\label{prob:spectrum}
Is $\sigma(\Hlim) = [0,1]$?
\end{problem}
