\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of self-adjoint
integral operators on $L^2(\mathbb{R})$.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent
stability (unconditional).}
We prove that the quadratic forms $\qc$
converge to a limit form $\qlim$ in the
sense of Mosco \cite{Mosco1969} at rate
$O(c^{-1/4})$, that $\qlim$ is closed,
and that the associated resolvents converge
strongly: $(\Hc-z)^{-1}
\xrightarrow{\mathrm{s}}(\Hlim-z)^{-1}$
for every $z\notin[0,1]$.
The Riesz--Kato representation
$T_{\qlim} = \Hlim$ is derived as an
independent identification.
The quantitative rate $O(c^{-1/4})$ is
proved entirely within the form framework:
no operator $\widetilde{\mathcal{H}}_k
- \Hc$ is formed at any point;
the argument uses only the spectral
theorem for $\Hc$ alone and the
eigenvalue-rate of Paper~VIII, Cor.~4.3.

\textbf{(B) Bridge theorem (unconditional).}
The strong-limit operator $\Hstrong$
(constructed here via density extension)
and the spectral-closure operator $\Hspectral$
(constructed in Paper~IX via eigenvalue
asymptotics and operator closure) are
identical as bounded operators:
$\Hstrong = \overline{\Hspectral}$.
This bridge theorem
(Theorem~\ref{thm:bridge})
is proved in isolation; it is the
\emph{only} place where Paper~IX results
are imported.

Conditionally on the spectral triple of
Connes--Consani--Moscovici \cite{CCM2025}
and the completeness hypothesis of
Paper~IX, every non-trivial zero of
$\zeta(s)$ lies on the critical line;
no unconditional statement about zeta
zeros is claimed.
\end{abstract}

\section*{Guide for the Referee}

\textbf{Architecture.}
The paper is organised around three
strictly separated threads:

\begin{enumerate}[(I)]
\item \textbf{Form-convergence thread}
      (Sections~\ref{sec:setup}--\ref{sec:friedrichs}):
      uses only the monotone form structure
      and the spectral theorem for $\Hc$.
      \emph{No operator
      $\widetilde{\mathcal{H}}_k - \Hc$
      is ever formed.
      No limit eigenvectors or limit
      eigenvalues are used.
      No results from Paper~IX are imported.}

\item \textbf{Bridge thread}
      (Section~\ref{sec:bridge}):
      only here are Paper~IX results used.

\item \textbf{Consequences thread}
      (Section~\ref{sec:consequences}):
      resolvent convergence, conditional
      connection to Riemann zeros.
\end{enumerate}

\textbf{Quantitative rate.}
The rate $O(c^{-1/4})$ in
Lemma~\ref{lem:form-rate} and
Theorem~\ref{thm:form-convergence}
is derived as follows:
(i) the spectral theorem for the
\emph{single} operator $\Hc$ yields
$\qc(f,f) = \sum_n \lambda_n^{(c)}
|\ip{f}{\phi_n^{(c)}}|^2$;
(ii) the eigenvalue monotonicity rate
$0 \le \lambda_n^{(\infty)}
- \lambda_n^{(c)} \le C_\kappa c^{-1/4}$
(Paper~VIII, Cor.~4.3) is applied
pointwise to this scalar series;
(iii) the result is a scalar bound on
$\qlim(f,f) - \qc(f,f)$,
obtained without forming any
operator difference.
The symbol $\lambda_n^{(\infty)}$
appears \emph{only} in this lemma
as a scalar limit of
$\lambda_n^{(c)}$ as $c \to \infty$;
it is not the eigenvalue of any
operator introduced before
Section~\ref{sec:bridge}.

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.

\begin{conv}[Ambient space and operators]
\label{conv:tilde}
All operators act on $L^2(\mathbb{R})$.
The PSWF operator $\mathcal{H}_c^{[0,1]}$
on $L^2([0,1])$ is extended by
$\Hc := P^*\mathcal{H}_c^{[0,1]}P$,
where $P$ is restriction and $P^*$ is
extension by zero.
Then $\Hc \in \mathcal{B}(L^2(\mathbb{R}))$
and $0 \le \Hc \le I$.
\end{conv}

\begin{enumerate}[(F.1)]

\item\label{F1}
      $0 \le \ip{\Hc f}{f} \le \|f\|^2$
      for all $f$, all $c$
      (\cite{Slepian1961}).

\item\label{F2}
      \textit{Monotone form and eigenvalue
      limit.}
      For every $f \in L^2(\mathbb{R})$,
      the map $c \mapsto \ip{\Hc f}{f}$
      is non-decreasing and bounded by
      $\|f\|^2$; hence
      \[
        \qlim(f,f)
        := \lim_{c\to\infty}\ip{\Hc f}{f}
        = \sup_{c>0}\ip{\Hc f}{f}
      \]
      exists for every $f$
      (Paper~V, Prop.~2.2).
      Likewise, for each $n \ge 0$,
      the map $c \mapsto \lambda_n^{(c)}$
      is non-decreasing and bounded by $1$;
      hence the scalar limit
      \[
        \lambda_n^{(\infty)}
        := \lim_{c\to\infty} \lambda_n^{(c)}
        \in [0,1]
      \]
      exists (Paper~VIII, Cor.~4.3).
      \emph{$\lambda_n^{(\infty)}$ is
      a scalar; no operator with this
      eigenvalue is introduced before
      Section~\ref{sec:bridge}.}

\item\label{F3}
      \textit{Dense strong-convergence
      domain.}
      Let
      \[
        \mathcal{D}
        := \{f \in L^2(\mathbb{R})
        : \lim_{c\to\infty}\Hc f
        \text{ exists in } L^2\}.
      \]
      All compactly supported $L^2$-functions
      lie in $\mathcal{D}$, so $\mathcal{D}$
      is dense
      (Paper~V, Prop.~2.2).

\item\label{F4}
      $\Hc$ is self-adjoint, positive,
      $\|\Hc\| \le 1$
      (\cite{Slepian1961};
      \cite{ReedSimon1975}, Thm.~VI.1).

\item\label{F6}
      \textit{Eigenvalue convergence rate.}
      For $c \ge c_0(\kappa)$ and
      $n \le \kappa c$:
      \[
        0 \le \lambda_n^{(\infty)}
        - \lambda_n^{(c)}
        \le C_\kappa\, c^{-1/4}.
      \]
      For $n > \kappa c$:
      $\lambda_n^{(c)},
      \lambda_n^{(\infty)}
      \le e^{-\gamma_\kappa c}$
      (Slepian tail;
      Paper~V, Thm.~2.1;
      Paper~VIII, Cor.~4.3).
      \emph{This is a scalar statement
      about eigenvalues of $\Hc$;
      no operator difference appears.}

\end{enumerate}

\begin{remark}[What is \emph{not} assumed]
\label{rem:no-spectral-input}
Sections~\ref{sec:setup}--\ref{sec:friedrichs}
use only \eqref{F1}--\eqref{F6}.
In particular:
\begin{itemize}
\item The scalar limits
      $\lambda_n^{(\infty)}$ from \eqref{F2}
      are used only in
      Lemma~\ref{lem:form-rate}.
      They are \emph{not} presented as
      eigenvalues of any limit operator.
\item No eigenvectors
      $\Phi_n^{(\infty)}$ appear.
\item No operator difference
      $\widetilde{\mathcal{H}}_k - \Hc$
      is formed.
\item No results from Paper~IX are used
      before Section~\ref{sec:bridge}.
\end{itemize}
\end{remark}

\section{Strong-Limit Operator
  \texorpdfstring{$\Hstrong$}{H-str}}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Density-extension principle]
\label{lem:sc}
Let $(A_c)_{c>0} \subset \mathcal{B}(\mathcal{H})$
with $M := \sup_c\|A_c\| < \infty$,
and let $\mathcal{D} \subset \mathcal{H}$
be a dense subspace on which
$A_c f \to Bf$ for every $f \in \mathcal{D}$.
Then $A_c f$ converges for every
$f \in \mathcal{H}$, the limit defines
$A \in \mathcal{B}(\mathcal{H})$
with $\|A\| \le M$, and $A_c \to A$
strongly.
\end{lemma}

\begin{proof}
Standard $3\varepsilon$ argument using
density of $\mathcal{D}$ and completeness
of $\mathcal{H}$; see e.g.\
\cite{ReedSimon1975}, Thm.~I.26.
\end{proof}

\begin{lemma}[Existence of $\Hstrong$]
\label{lem:hlim-bounded}
There exists a unique
$\Hstrong \in \mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hstrong\| \le 1$ and
$\Hc \to \Hstrong$ strongly.
\end{lemma}

\begin{proof}
Apply Lemma~\ref{lem:sc} with
$A_c := \Hc$, $\mathcal{D}$ as in \eqref{F3},
$M := 1$ (by \eqref{F4}).
\end{proof}

\begin{corollary}[Form equals operator]
\label{cor:qlim-op}
For every $f \in L^2(\mathbb{R})$:
\[
  \qlim(f,f) = \ip{\Hstrong f}{f}.
\]
\end{corollary}
\begin{proof}
$\qlim(f,f) = \lim_c\ip{\Hc f}{f}
= \ip{\Hstrong f}{f}$
by strong convergence
(Lemma~\ref{lem:hlim-bounded})
and continuity of the inner product.
\end{proof}

\begin{corollary}[Self-adjointness]
\label{cor:sa-strong}
$\Hstrong$ is self-adjoint and positive,
$\sigma(\Hstrong) \subset [0,1]$.
\end{corollary}
\begin{proof}
Symmetry follows from symmetry of each
$\Hc$ by passing to the strong limit.
Since $\Hstrong$ is bounded and symmetric
on all of $L^2(\mathbb{R})$, it is
self-adjoint (BSA principle;
\cite{ReedSimon1975}, Thm.~III.6).
Positivity and spectral inclusion follow
from $0 \le \qlim(f,f) \le \|f\|^2$
(Corollary~\ref{cor:qlim-op}).
\end{proof}

\section{Quantitative Form Rate}
\label{sec:form-rate}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Strategy: pure form argument]
\label{rem:form-rate-strategy}
The quantitative bound
$\qlim(f,f) - \qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$
is established \emph{without} forming
any operator $\widetilde{\mathcal{H}}_k
- \Hc$.
The proof has two steps:
\begin{enumerate}[\rm(S1)]
\item Apply the spectral theorem
      for $\Hc$ alone to write
      $\qlim(f,f) - \qc(f,f)$ as a
      weighted scalar series involving
      $\lambda_n^{(\infty)} - \lambda_n^{(c)}$.
\item Apply the scalar rate \eqref{F6}
      to bound each term uniformly.
\end{enumerate}
At no point is a second operator $k \ne c$
introduced.
\end{remark}

\begin{lemma}[Quantitative form rate]
\label{lem:form-rate}
For all $f \in L^2(\mathbb{R})$
and $c \ge c_0(\kappa)$:
\[
  0 \le \qlim(f,f) - \qc(f,f)
  \le C_\kappa\, c^{-1/4}\, \|f\|^2.
\]
\end{lemma}

\begin{proof}
\textit{Lower bound.}
$\qlim(f,f) \ge \qc(f,f)$ by
monotonicity \eqref{F2}.

\textit{Upper bound.}
Let $\{\phi_n^{(c)}\}$ be the orthonormal
eigenbasis of $\Hc$ with eigenvalues
$\lambda_n^{(c)}$
(spectral theorem for compact self-adjoint
operators; \cite{ReedSimon1975},
Thm.~VI.16).
Write $f = \sum_n a_n \phi_n^{(c)}$
with $a_n = \ip{f}{\phi_n^{(c)}}$
and $\sum_n |a_n|^2 = \|f\|^2$.

By definition and the spectral theorem
for $\Hc$:
\[
  \qc(f,f) = \ip{\Hc f}{f}
  = \sum_n \lambda_n^{(c)} |a_n|^2.
\]

For $\qlim(f,f)$: the scalar limits
$\lambda_n^{(\infty)} \in [0,1]$
(\eqref{F2}) satisfy
$\lambda_n^{(c)} \le \lambda_n^{(\infty)}$
for all $n$ (monotonicity in $c$).
We claim:
\begin{equation}\label{eq:qlim-series}
  \qlim(f,f)
  = \sum_n \lambda_n^{(\infty)} |a_n|^2.
\end{equation}

\textit{Proof of \eqref{eq:qlim-series}.}
Fix $N$. Since $\lambda_n^{(c)}
\nearrow \lambda_n^{(\infty)}$
monotonically, by the monotone convergence
theorem for series of non-negative terms:
\begin{align*}
  \qlim(f,f)
  &= \lim_{c\to\infty}
     \sum_n \lambda_n^{(c)} |a_n|^2
     \tag{spectral theorem $+$ \eqref{F2}}\\
  &= \sum_n \lim_{c\to\infty}
     \lambda_n^{(c)} |a_n|^2
     \tag{MCT: each term non-decreasing}\\
  &= \sum_n \lambda_n^{(\infty)} |a_n|^2.
\end{align*}
The interchange is justified because
$0 \le \lambda_n^{(c)} |a_n|^2
\le |a_n|^2$ and
$\sum_n |a_n|^2 = \|f\|^2 < \infty$;
dominated convergence applies.

\textit{Final estimate.}
Using \eqref{eq:qlim-series}:
\begin{align*}
  \qlim(f,f) - \qc(f,f)
  &= \sum_n
     (\lambda_n^{(\infty)} - \lambda_n^{(c)})
     |a_n|^2.
\end{align*}
Split into two parts:
\begin{align*}
  \sum_{n \le \kappa c}
  (\lambda_n^{(\infty)} - \lambda_n^{(c)})
  |a_n|^2
  &\le C_\kappa c^{-1/4}
       \sum_{n \le \kappa c} |a_n|^2
     \tag{\eqref{F6}}\\
  &\le C_\kappa c^{-1/4} \|f\|^2,
\end{align*}
and:
\begin{align*}
  \sum_{n > \kappa c}
  (\lambda_n^{(\infty)} - \lambda_n^{(c)})
  |a_n|^2
  &\le \sum_{n > \kappa c}
       2e^{-\gamma_\kappa c} |a_n|^2
     \tag{\eqref{F6}: tail bound}\\
  &\le 2e^{-\gamma_\kappa c} \|f\|^2
  \le C_\kappa c^{-1/4} \|f\|^2.
\end{align*}
Adding both parts gives the claim.
\end{proof}

\begin{remark}[No limit operator
  introduced]
\label{rem:no-limit-op}
Equation~\eqref{eq:qlim-series}
expresses $\qlim(f,f)$ purely as a
scalar series with coefficients
$a_n = \ip{f}{\phi_n^{(c)}}$
(eigenvectors of $\Hc$, fixed $c$)
and scalar weights $\lambda_n^{(\infty)}$
(limits of eigenvalues of $\Hc$).
No operator with eigenvectors
$\Phi_n^{(\infty)}$ is introduced or
needed here.
The identity \eqref{eq:qlim-series}
holds because $\{\phi_n^{(c)}\}$ is
an ONB for $L^2$ for each fixed $c$
(compactness of $\Hc$), and the
coefficients $a_n$ depend only on $f$
and this fixed basis.
\end{remark}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc)$ converge to $\qlim$ in the
sense of Mosco if:
\begin{enumerate}[\rm(M1)]
\item For every $c_n \to \infty$ and
      $f_n \rightharpoonup f$:
      $\qlim(f,f) \le
      \liminf_{n} q_{c_n}(f_n,f_n)$.
\item For every $f$ there exist
      $c_n \to \infty$, $f_n \to f$
      strongly with
      $q_{c_n}(f_n,f_n) \to \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Compactness and
  weak continuity]
\label{lem:hs}
For each fixed $c_0 > 0$:
\begin{enumerate}[\rm(a)]
\item $\Hc[c_0]$ is compact
      (Hilbert--Schmidt on $L^2([0,1])$,
      extended by zero;
      \cite{ReedSimon1975}, Thm.~VI.22).
\item $f_n \rightharpoonup f
      \Rightarrow
      \ip{\Hc[c_0] f_n}{f_n}
      \to \ip{\Hc[c_0] f}{f}$.
\end{enumerate}
\end{lemma}
\begin{proof}
(a) $\Hc[c_0]$ acts on $L^2([0,1])$
via the Sinc kernel
$K_{c_0}(x,y)
= \frac{\sin c_0(x-y)}{\pi(x-y)}$,
which satisfies
$\|K_{c_0}\|_{L^2([0,1]^2)} < \infty$
(bounded domain, bounded kernel).
Hence $\Hc[c_0]|_{L^2([0,1])}$ is
Hilbert--Schmidt, thus compact;
the extension to $L^2(\mathbb{R})$
by zero is compact.

(b) Write the difference as $I + II$
with $I := \ip{\Hc[c_0](f_n-f)}{f_n}$,
$II := \ip{\Hc[c_0]f}{f_n-f}$.
$II \to 0$ by weak convergence;
$I \to 0$ by compactness and
$\sup_n\|f_n\| < \infty$.
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc \to \qlim$ in the Mosco sense.
\end{theorem}
\begin{proof}
\textit{(M2).}
Take $f_n := f$, $c_n \to \infty$:
$q_{c_n}(f,f) \to \qlim(f,f)$
by definition \eqref{F2}.

\textit{(M1).}
Let $c_n \to \infty$,
$f_n \rightharpoonup f$.
For every fixed $c_0 > 0$, once
$c_n \ge c_0$:
$q_{c_n}(f_n,f_n) \ge q_{c_0}(f_n,f_n)$
by \eqref{F2}.
Hence $\liminf_n q_{c_n}(f_n,f_n)
\ge \liminf_n q_{c_0}(f_n,f_n)
= q_{c_0}(f,f)$
by Lemma~\ref{lem:hs}(b).
Since this holds for every fixed $c_0$,
taking the supremum (which does not
involve $n$):
$\liminf_n q_{c_n}(f_n,f_n)
\ge \sup_{c_0>0} q_{c_0}(f,f)
= \qlim(f,f)$.
\end{proof}

\section{Closedness and Quantitative Rate}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of $\qlim$]
\label{thm:form-closed}
$\qlim$ is a closed form.
\end{theorem}
\begin{proof}
Since $\qlim(f,f) = \ip{\Hstrong f}{f}$
and $\Hstrong \in \mathcal{B}(L^2)$,
the form norm
$\|f\|_{\qlim}^2 = \|f\|^2 + \qlim(f,f)$
is equivalent to $\|f\|^2$,
so any $\|\cdot\|_{\qlim}$-Cauchy
sequence is $L^2$-Cauchy, and $\qlim$
is continuous at its limit.
\end{proof}

\begin{theorem}[Quantitative form
  convergence]
\label{thm:form-convergence}
For all $f \in L^2(\mathbb{R})$,
$c \ge c_0(\kappa)$:
\[
  0 \le \qlim(f,f) - \qc(f,f)
  \le C_\kappa\, c^{-1/4}\, \|f\|^2.
\]
\end{theorem}
\begin{proof}
This is exactly Lemma~\ref{lem:form-rate}.
\end{proof}

\section{Riesz--Kato Form Identification}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Riesz--Kato identification]
\label{thm:friedrichs}
$T_{\qlim} = \Hstrong$,
where $T_{\qlim}$ is the unique
$T \in \mathcal{B}(L^2)$ with
$\qlim(f,g) = \ip{Tf}{g}$ for all
$f, g \in L^2$.
\end{theorem}
\begin{proof}
By Corollary~\ref{cor:qlim-op}:
$\ip{\Hstrong f}{g} = \qlim(f,g)$
for all $f, g$,
so $\Hstrong$ represents $\qlim$.
Uniqueness of the Riesz representative
gives $T_{\qlim} = \Hstrong$.
\end{proof}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the \emph{only} section
that imports results from Paper~IX.

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      Paper~IX, Definition~2.3 defines
      \[
        \Hspectral f
        := \sum_{n \ge 0}
           \lambda_n^{(\infty)}
           \ip{f}{\Phi_n^{(\infty)}}
           \Phi_n^{(\infty)}
      \]
      on its natural domain in $L^2$,
      where $\{\Phi_n^{(\infty)}\}$ are
      the limit PSWF eigenvectors.
      This operator is symmetric, densely
      defined, and closable
      (Paper~IX, Thm.~2.5, Thm.~3.2).

\item\label{IXb}
      Since $\lambda_n^{(\infty)} \in [0,1]$,
      the closure
      $\overline{\Hspectral}
      \in \mathcal{B}(L^2)$
      with $\|\overline{\Hspectral}\| \le 1$.

\item\label{IXc}
      For $f \in \mathcal{D}$:
      $\Hc f \to \Hstrong f$
      (by definition of $\Hstrong$)
      and $\Hc f \to
      \overline{\Hspectral} f$
      (Paper~IX, Remark~2.6).
      Uniqueness of $L^2$-limits:
      $\Hstrong f
      = \overline{\Hspectral} f$
      on all of $\mathcal{D}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong = \overline{\Hspectral}$
as bounded operators on $L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Both are bounded (\eqref{IXb};
Lemma~\ref{lem:hlim-bounded}).
They agree on the dense set $\mathcal{D}$
(\eqref{IXc}).
Hence they are equal on all of $L^2$.
\end{proof}

\begin{remark}[After the bridge]
\label{rem:bridge}
We now write $\Hlim := \Hstrong
= \overline{\Hspectral}$.
The scalar limits $\lambda_n^{(\infty)}$
used in Section~\ref{sec:form-rate}
are precisely the eigenvalues of
$\Hspectral$ (Paper~IX, Def.~2.3);
this identification is now legitimate
because the bridge has been established.
Note that using $\lambda_n^{(\infty)}$
as scalars in Lemma~\ref{lem:form-rate}
before the bridge was \emph{not} circular:
they were introduced as scalar limits
of $\lambda_n^{(c)}$ via monotone
convergence, independently of any
operator construction.
\end{remark}

\begin{remark}[$\Hlim$ is not a
  multiplication operator]
\label{rem:hlim-not-mult}
$\sigma(\Hlim) \subset
\overline{\{\lambda_n^{(\infty)}
: n \ge 0\}}$.
The Landau--Pollak transition accumulates
values densely in $(0,1)$
(\cite{Slepian1961}),
so $\sigma(\Hlim) \not\subset \{0,1\}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every $z \in \mathbb{C}
\setminus [0,1]$:
$(\Hc - z)^{-1}
\xrightarrow{\mathrm{s}}
(\Hlim - z)^{-1}$.
\end{corollary}
\begin{proof}
Mosco convergence
(Theorem~\ref{thm:mosco}),
$\sigma(\Hlim) \subset [0,1]$
(Corollary~\ref{cor:sa-strong}
+ Theorem~\ref{thm:bridge}),
and \cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
\label{rem:unconditional}
The following hold unconditionally:
$\Hlim \in \mathcal{B}(L^2)$,
$\Hlim = \Hlim^* \ge 0$,
$\sigma(\Hlim) \subset [0,1]$,
$T_{\qlim} = \Hlim$,
$(\Hc-z)^{-1} \xrightarrow{s}
(\Hlim-z)^{-1}$,
$\qc \xrightarrow{\mathrm{Mosco}} \qlim$
at rate $O(c^{-1/4})$,
and $\Hstrong = \overline{\Hspectral}$.
\end{remark}

\begin{corollary}[Conditional RH]
\label{cor:rh}
Conditionally on CCM
(\cite{CCM2025}, Thm.~1.3)
and Paper~IX, Hypothesis~4.1,
every non-trivial zero $\rho$ of
$\zeta(s)$ satisfies
$\mathrm{Re}(\rho) = \tfrac{1}{2}$.
\textit{No unconditional claim about
zeta zeros is made.}
\end{corollary}
\begin{proof}
Under Paper~IX, Hyp.~4.1,
$\Hlim$ is essentially self-adjoint
(Paper~IX, Cor.~5.2).
Under CCM:
$\rho \in \sigma_{\mathrm{app}}(\Hlim)
= \sigma(\Hlim) \subset \mathbb{R}$;
the CCM embedding forces
$\mathrm{Re}(\rho) = 1/2$.
\end{proof}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Completeness]
\label{prob:completeness}
Prove Paper~IX, Hypothesis~4.1:
$\{\Phi_n^{(\infty)}\}$ is complete
in $L^2(\mathbb{R})$, equivalently
$\overline{\Hspectral}$ has no
deficiency vectors.
\end{problem}

\begin{problem}[Sharp form rate]
\label{prob:rate}
Is $c^{-1/4}$ in
Theorem~\ref{thm:form-convergence}
sharp, or is the true rate
$c^{-1/2}$ or better?
(Paper~VIII gives $c^{-1/2}$ for the
individual eigenvalue gap
$|\lambda_n^{(c+1)} - \lambda_n^{(c)}|$;
the uniform-in-$n$ rate \eqref{F6}
from Cor.~4.3 is $c^{-1/4}$.)
\end{problem}

\begin{problem}[Quantitative resolvent]
\label{prob:resolvent}
Find an explicit rate $F(c,z) \to 0$
with
$\|(\Hc-z)^{-1}-(\Hlim-z)^{-1}\|
\le F(c,z)$.
\end{problem}

\begin{problem}[Spectrum of $\Hlim$]
\label{prob:spectrum}
Is $\sigma(\Hlim) = [0,1]$,
or does it have gaps?
\end{problem}
