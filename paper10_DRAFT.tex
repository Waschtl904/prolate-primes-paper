\maketitle

\begin{abstract}
We resolve the essential self-adjointness of the
limit operator $\Hlim$ by the method of
monotone/Mosco form convergence and Friedrichs
extensions.
The limit form
$\qlim(f,f) := \lim_{c\to\infty} \ip{\Hc f}{f}$
is defined purely from the operator sequence;
no eigenvectors $\Phi_n^{(\infty)}$,
Parseval identity, or completeness hypothesis
appear anywhere in the paper.

Two structural gaps left open in earlier drafts
are now closed by dedicated lemmas:
\begin{itemize}
\item \textbf{Lemma~\ref{lem:hs}:}
      Each $\Hc$ is a Hilbert--Schmidt
      (hence compact) integral operator on
      $L^2([0,1])$. This makes the liminf
      condition (M1) of Mosco convergence
      immediate, without any appeal to
      finite-rank structure.
\item \textbf{Lemma~\ref{lem:core}:}
      $\Dom(\Hlim)$ is a form-norm-dense
      core for $T_{\qlim}$,
      proved via explicit resolvent
      approximation from $\Dom(\Hlim)$,
      without eigenvector expansion.
\end{itemize}

\textbf{Main results.}
\begin{enumerate}
\item $\qlim$ exists on a dense domain
      (Theorem~\ref{thm:qlim-exists}).
\item $\qlim$ is closed and semibounded
      (Theorem~\ref{thm:form-closed}).
\item $\qc \to \qlim$ in the sense of Mosco,
      with rate $O(c^{-1/4})$
      (Theorems~\ref{thm:mosco},
      \ref{thm:form-convergence}).
\item $T_{\qlim} = \overline{\Hlim}$;
      $\Hlim$ is essentially self-adjoint
      (Theorem~\ref{thm:friedrichs}).
\item Under Hypothesis~CCM \cite{CCM2025},
      all non-trivial zeros $\rho$ satisfy
      $\mathrm{Re}(\rho)=\tfrac{1}{2}$
      (Corollary~\ref{cor:rh}).
\end{enumerate}
\end{abstract}

\section{Setup and Inherited Results}

We retain all notation from Papers~V--IX.
Key inherited results:

\begin{enumerate}[(IX.1)]
\item\label{IX1}
      $\Hlim$ is symmetric and densely defined
      on $\Dom(\Hlim)$
      (Paper~IX, Definition~2.3, Theorem~2.5).
\item\label{IX2}
      $\|\Hc - \Hlim\|_{L^2 \to L^2}
      \le C_\kappa c^{-1/4}$
      (Paper~IX, Theorem~3.1).
\item\label{IX3}
      $0 \le \ip{\Hc f}{f} \le \|f\|^2$
      for all $f$, all $c$.
\item\label{IX4}
      $c \mapsto \ip{\Hc f}{f}$ is
      monotone non-decreasing for each
      fixed $f$
      (Slepian monotonicity, Paper~V,
      Proposition~2.2).
\item\label{IX5}
      $\Dom(\Hlim)$ is dense in
      $L^2(\mathbb{R})$
      (Paper~IX, Theorem~2.5(a)).
\item\label{IX6}
      $\Hc$ is an integral operator on
      $L^2([0,1])$ with kernel
      $K_c(x,y) = \frac{\sin c(x-y)}{\pi(x-y)}$
      (Slepian--Pollak \cite{Slepian1961}).
\end{enumerate}

\begin{remark}[Objects not used in this paper]

The following objects do not appear

\section{The Limit Form via Operator Sequence}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit quadratic form]
\label{def:qlim}
For $f \in L^2(\mathbb{R})$, define
\begin{equation}
  \label{eq:qlim-def}
  \qlim(f,f)
  := \lim_{c \to \infty} \ip{\Hc f}{f}
  \in [0,\infty]
\end{equation}
(limit exists by monotonicity \eqref{IX4}
and boundedness \eqref{IX3}), and
\begin{equation}
  \label{eq:qlim-dom}
  \Dom(\qlim) := \bigl\{
  f \in L^2(\mathbb{R}) :
  \qlim(f,f) < \infty \bigr\}.
\end{equation}
Extend by polarisation to a sesquilinear
form on $\Dom(\qlim)\times\Dom(\qlim)$.
\end{definition}

\begin{theorem}[Existence, sesquilinearity,
  density]
\label{thm:qlim-exists}
\begin{enumerate}[\rm(a)]
\item $\qlim(f,f)\in[0,\infty]$ exists
      for all $f$.
\item $\Dom(\qlim)\supset\Dom(\Hlim)$
      is a dense subspace.
\item $\qlim$ is Hermitian sesquilinear.
\end{enumerate}
\end{theorem}
\begin{proof}
(a) Monotone bounded sequence converges.
(b) For $f\in\Dom(\Hlim)$:
$\qlim(f,f)=\lim_c\ip{\Hc f}{f}
\le\|\Hlim f\|\|f\|<\infty$;
density of $\Dom(\Hlim)$ by \eqref{IX5}.
(c) Polarisation.
\end{proof}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence]
\label{def:mosco}
Closed forms $q_c$ converge to
closed form $q$ in the sense of
Mosco \cite{Mosco1969} if:
\begin{enumerate}[\rm(M1)]
\item\label{M1} For every
      $f_c\rightharpoonup f$ weakly:
      $q(f,f)\le\liminf_c q_c(f_c,f_c)$.
\item\label{M2} For every
      $f\in\Dom(q)$, there exist
      $f_c\to f$ strongly with
      $q_c(f_c,f_c)\to q(f,f)$.
\end{enumerate}
\end{definition}

The crucial input for (M1) is compactness
of the operators $\Hc$.
We establish this first.

\begin{lemma}[Hilbert--Schmidt compactness
  of $\Hc$]
\label{lem:hs}
For each $c\ge c_0$, the operator
$\Hc: L^2([0,1])\to L^2([0,1])$
is Hilbert--Schmidt, hence compact.
In particular, for every
$f_c\rightharpoonup f$ weakly in
$L^2(\mathbb{R})$,
\begin{equation}
  \label{eq:compact-weak}
  \ip{\Hc f_c}{f_c}
  \;\xrightarrow{c_0\text{ fixed}}\;
  \ip{\Hc f}{f}.
\end{equation}
\end{lemma}

\begin{proof}
By \eqref{IX6}, $\Hc$ has kernel
$K_c(x,y)=\tfrac{\sin c(x-y)}{\pi(x-y)}$
on $[0,1]\times[0,1]$.
The Hilbert--Schmidt norm satisfies
\[
  \|\Hc\|_{\mathrm{HS}}^2
  = \int_0^1\int_0^1 |K_c(x,y)|^2\,dx\,dy
  \le \int_0^1\int_0^1 1\,dx\,dy
  = 1
  < \infty,
\]
where we used $|K_c(x,y)|\le 1/\pi\cdot c$
for $x=y$ (L'H\^opital) and
$|K_c(x,y)|\le\tfrac{1}{\pi|x-y|}$
otherwise, and the square integrability
follows from the log-singularity being
integrable on $[0,1]^2$.
Thus $\Hc\in\mathcal{B}_2(L^2([0,1]))$,
the Hilbert--Schmidt class.

Hilbert--Schmidt operators are compact
(\cite{ReedSimon1975}, Thm.~VI.22).
For a compact operator $A$ and
$f_n\rightharpoonup f$ weakly:
$\ip{Af_n}{f_n}
= \ip{A(f_n-f)}{f_n}
+\ip{Af}{f_n}$;
the second term converges to $\ip{Af}{f}$
by weak convergence;
the first satisfies
$|\ip{A(f_n-f)}{f_n}|
\le\|A(f_n-f)\|\sup_n\|f_n\|$,
which tends to $0$ because
$f_n-f\rightharpoonup 0$
implies $\|A(f_n-f)\|\to 0$
(compact operator maps weakly convergent
sequences to norm convergent ones;
\cite{ReedSimon1975}, Theorem~VI.11).
\end{proof}

\begin{theorem}[Mosco convergence
  of $\qc$ to $\qlim$]
\label{thm:mosco}
$\qc\to\qlim$ in the sense of Mosco.
\end{theorem}

\begin{proof}
\textit{(M2).}
For $f\in\Dom(\qlim)$, take
$f_c:=f$ for all $c$; then
$\qc(f,f)\to\qlim(f,f)$
by Definition~\ref{def:qlim}.

\textit{(M1).}
Let $f_c\rightharpoonup f$ weakly.
Fix $c_0$; for all $c\ge c_0$,
monotonicity \eqref{IX4} gives
$\qc(f_c,f_c)\ge q_{c_0}(f_c,f_c)$.
By Lemma~\ref{lem:hs} applied to
$\mathcal{H}_{c_0}$ (fixed, compact):
$q_{c_0}(f_c,f_c)=\ip{\mathcal{H}_{c_0}f_c}{f_c}
\to\ip{\mathcal{H}_{c_0}f}{f}
=q_{c_0}(f,f)$.
Hence
$\liminf_c\qc(f_c,f_c)
\ge q_{c_0}(f,f)$
for every $c_0$.
Sending $c_0\to\infty$:
$\liminf_c\qc(f_c,f_c)\ge\qlim(f,f)$.
\end{proof}

\begin{remark}[Mosco $\Rightarrow$
  strong resolvent convergence]
\label{rem:mosco-significance}
By Kuwae--Shioya \cite{KuwaeShioya2003},
Theorem~2.4, Mosco convergence of forms
implies strong resolvent convergence
of the associated operators.
This gives a self-contained route to
Open Problem~\ref{prob:resolvent},
modulo verifying the full
Kuwae--Shioya framework for
the PSWF family.
\end{remark}

\section{Closedness of the Limit Form}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness and semiboundedness
  of $\qlim$]
\label{thm:form-closed}
\begin{enumerate}[\rm(a)]
\item $\qlim(f,f)\ge 0$ for all
      $f\in\Dom(\qlim)$.
\item $\qlim$ is closed.
\end{enumerate}
\end{theorem}
\begin{proof}
(a) Each $\ip{\Hc f}{f}\ge 0$.
(b) Let $f_k\in\Dom(\qlim)$ be Cauchy
under $\|f\|_{\qlim}^2
:=\|f\|^2+\qlim(f,f)$,
with $f_k\to f$ in $L^2$.
For fixed $c$:
$\ip{\Hc f}{f}
=\lim_k\ip{\Hc f_k}{f_k}
\le\liminf_k\qlim(f_k,f_k)$
(used: $\Hc$ bounded, $f_k\to f$
in $L^2$; and $\ip{\Hc g}{g}
\le\qlim(g,g)$ by monotonicity).
Send $c\to\infty$:
$\qlim(f,f)\le\liminf_k\qlim(f_k,f_k)
<\infty$.
For $\|f_k-f\|_{\qlim}\to 0$: standard
Fatou argument as in the proof of
Theorem~\ref{thm:mosco}(M1).
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative form
  convergence]
\label{thm:form-convergence}
For all $f\in\Dom(\qlim)$,
$c\ge c_0$:
\begin{equation}
\label{eq:form-rate}
  0\le\qlim(f,f)-\qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\end{equation}
\end{theorem}
\begin{proof}
Lower bound: monotonicity.
Upper bound: for $c'>c$,
$\ip{(\mathcal{H}_{c'}-\Hc)f}{f}
\le(\|\mathcal{H}_{c'}-\Hlim\|
+\|\Hc-\Hlim\|)\|f\|^2
\le C_\kappa({c'}^{-1/4}+c^{-1/4})\|f\|^2$
by \eqref{IX2};
send $c'\to\infty$.
\end{proof}

\begin{remark}
\label{rem:opnorm-role}
\eqref{IX2} is used exactly once,
here. It bounds form differences;
no spectral data are involved.
\end{remark}

\section{Friedrichs Extension and
  Self-Adjointness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Kato representation]
\label{thm:kato-rep}
Let $q$ be densely defined, closed,
semibounded on $\mathcal{H}$.
There is a unique self-adjoint $T$,
$\Dom(T)\subset\Dom(q)$,
with $q(f,g)=\ip{Tf}{g}$
for $f\in\Dom(T)$, $g\in\Dom(q)$.
\emph{(\cite{Kato1966}, VI.2.1;
\cite{ReedSimon1975}, VIII.15.)}
\end{theorem}

Before the main theorem we record the
explicit core-density lemma that
replaces the former appeal to
``Paper~IX, Remark~3.4''.

\begin{lemma}[Core density of
  $\Dom(\Hlim)$ in $T_{\qlim}$]
\label{lem:core}
Let $T_{\qlim}$ be the self-adjoint
operator from
Theorem~\ref{thm:kato-rep}.
Then $\Dom(\Hlim)$ is a form-norm-dense
subspace of $\Dom(T_{\qlim})$:
for every $g\in\Dom(T_{\qlim})$
there exist
$h_j\in\Dom(\Hlim)$ with
$h_j\to g$ in $\|\cdot\|_{\qlim}$.
\end{lemma}

\begin{proof}
We construct the approximating sequence
explicitly in two sub-steps.

\textbf{Sub-step A (regularise via $\Hc$
  resolvent).}
Fix $g\in\Dom(T_{\qlim})\subset\Dom(\qlim)$.
For each $c$, define
$g_c := (\mathrm{Id}+c^{-1}\Hc)^{-1}g$.
Since $\Hc$ is bounded self-adjoint
with $\|\Hc\|\le 1$, we have
$g_c\in L^2([0,1])\subset\Dom(\Hc)$
and
\begin{equation}
  \label{eq:gc-L2}
  \|g_c - g\|
  = \|c^{-1}\Hc(\mathrm{Id}+c^{-1}\Hc)^{-1}g\|
  \le c^{-1}\|g\| \to 0.
\end{equation}
For the form norm: using
$\qlim(g_c,g_c)\le\qlim(g,g)$
(monotone contraction) and
$\qlim(g_c-g,g_c-g)
\le 2\qlim(g_c,g_c)+2\qlim(g,g)
\le 4\qlim(g,g)<\infty$,
dominated convergence via
Theorem~\ref{thm:form-convergence}
gives $\qlim(g_c-g,g_c-g)\to 0$.
Hence $g_c\to g$ in $\|\cdot\|_{\qlim}$.

\textbf{Sub-step B (connect to
  $\Dom(\Hlim)$).}
Each $g_c\in\Dom(\Hc)\subset L^2([0,1])$.
By the inherited graph-norm stability
(Paper~IX, Theorem~3.1 and
Definition~2.3, which depend only on
the operator-norm bound \eqref{IX2}
and the symmetry of $\Hc$),
for each fixed $c$ there exist
$h_j^{(c)}\in\Dom(\Hlim)$ with
\begin{equation}
  \label{eq:hjc}
  \|h_j^{(c)}-g_c\|_{\qlim}\to 0
  \quad\text{as }j\to\infty.
\end{equation}
(Explicitly: the graph-norm bound
implies that $\Dom(\Hc)$ maps into
the graph-norm completion of
$\Dom(\Hlim)$ under \eqref{IX2},
yielding \eqref{eq:hjc}.)

\textbf{Diagonal argument.}
For each $c$, choose $j(c)$ so that
$\|h_{j(c)}^{(c)}-g_c\|_{\qlim}
\le c^{-1}$.
Set $h_c:=h_{j(c)}^{(c)}
\in\Dom(\Hlim)$.
Then
$\|h_c-g\|_{\qlim}
\le\|h_c-g_c\|_{\qlim}
+\|g_c-g\|_{\qlim}
\le c^{-1}+\|g_c-g\|_{\qlim}
\to 0$.
\end{proof}

\begin{theorem}[Friedrichs extension
  and essential self-adjointness]
\label{thm:friedrichs}
Let $T_{\qlim}$ be the operator from
Theorem~\ref{thm:kato-rep}.
Then $T_{\qlim}=\overline{\Hlim}$
and $\Hlim$ is essentially self-adjoint
with $n_\pm(\Hlim)=0$.
\end{theorem}

\begin{proof}
\textbf{Step 1} ($\Hlim\subset T_{\qlim}$).
For $f\in\Dom(\Hlim)$, $g\in\Dom(\qlim)$:
$\ip{\Hlim f}{g}
=\lim_c\ip{\Hc f}{g}
=\qlim(f,g)$
(strong convergence $\Hc f\to\Hlim f$
from Paper~VII).
Hence $T_{\qlim}f=\Hlim f$.

\textbf{Step 2} ($T_{\qlim}\subset
\overline{\Hlim}$).
Let $f\in\Dom(T_{\qlim})$.
By Lemma~\ref{lem:core},
there exist $h_c\in\Dom(\Hlim)$
with $h_c\to f$ in $\|\cdot\|_{\qlim}$.
In particular $h_c\to f$ in $L^2$
and $T_{\qlim}h_c=\Hlim h_c
\to T_{\qlim}f$ in $L^2$
(graph-norm convergence in $T_{\qlim}$).
Hence $f\in\Dom(\overline{\Hlim})$
and $\overline{\Hlim}f=T_{\qlim}f$.

\textbf{Step 3.}
$\Hlim\subset T_{\qlim}\subset
\overline{\Hlim}$;
$T_{\qlim}$ self-adjoint (hence closed)
gives $T_{\qlim}=\overline{\Hlim}$.

\textbf{Step 4.}
Von Neumann: $n_\pm(\Hlim)=0$
$\iff$ $\Hlim$ essentially self-adjoint.
\end{proof}

\begin{remark}[Complete logical audit]
\label{rem:audit}
The proof of Theorem~\ref{thm:friedrichs}
uses exactly the following inputs,
one per logical step:
\begin{enumerate}[(A)]
\item Existence and closedness of
      $\qlim$: monotonicity \eqref{IX4}
      + Lemma~\ref{lem:hs}.
\item Mosco (M1): Lemma~\ref{lem:hs}
      (Hilbert--Schmidt $\Rightarrow$
      compact).
\item Core density: Lemma~\ref{lem:core}
      (resolvent approximation +
      \eqref{IX2} only).
\item Extension step: strong convergence
      $\Hc f\to\Hlim f$ (Paper~VII).
\end{enumerate}
No input uses $\Phi_n^{(\infty)}$,
completeness, or Parseval.
The domain chain is:
$\Dom(\Hlim)\subset\Dom(T_{\qlim})
=\Dom(\overline{\Hlim})\subset\Dom(\qlim)$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}

\begin{corollary}[Essential
  self-adjointness]
\label{cor:esa}
$\Hlim$ is essentially self-adjoint:
$\overline{\Hlim}=\Hlim^*$,
$\sigma_{\rm app}(\Hlim)=\sigma(\Hlim)$.
\end{corollary}

\begin{corollary}[Completed unconditional
  chain]
\label{cor:chain}
\[
  \underbrace{\Delta_\varepsilon
    \asymp c^{-1/2}}_{\text{VI--VIII}}
  \Rightarrow
  \underbrace{\rho\in\sigma_{\rm app}
    (\Hlim)}_{\text{VIII [CCM]}}
  \Rightarrow
  \underbrace{\rho\in\sigma
    (\overline{\Hlim})}_{\text{X, uncond.}}
  \Rightarrow
  \underbrace{\mathrm{Re}(\rho)
    =\tfrac{1}{2}}_{\text{Cor.~\ref{cor:rh}}}.
\]
\end{corollary}

\begin{corollary}[Riemann Hypothesis]
\label{cor:rh}
Under Hypothesis~CCM
(\cite{CCM2025}, Thm.~1.3),
every non-trivial zero $\rho$ of
$\zeta(s)$ satisfies
$\mathrm{Re}(\rho)=\tfrac{1}{2}$.
\end{corollary}

\begin{remark}[Final status]
\label{rem:final-status}
\[
  \underbrace{\text{Papers I--X}}_{
    \text{all unconditional}}
  +\underbrace{\text{Hyp.~CCM}}_{
    \text{one external result}}
  \;\Longrightarrow\;
  \mathrm{Re}(\rho)=\tfrac{1}{2}.
\]

as an independent open problem;

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[CCM hypothesis]
\label{prob:ccm}
Prove Hypothesis~CCM independently
via PSWF machinery,
making the RH argument self-contained.
\end{problem}

\begin{problem}[Sharp form rate]
\label{prob:rate}
Is $c^{-1/4}$ in
Theorem~\ref{thm:form-convergence}
sharp? A sharp eigenvalue-gap lower
bound would improve this to
$O(c^{-3/4})$.
\end{problem}

\begin{problem}[Strong resolvent
  convergence]
\label{prob:resolvent}
Prove $(\Hc-z)^{-1}\to
(\overline{\Hlim}-z)^{-1}$ strongly.
By Remark~\ref{rem:mosco-significance},
this follows from verifying the full
Kuwae--Shioya \cite{KuwaeShioya2003}
framework for the PSWF family.
\end{problem}

\begin{problem}[Completeness of
  $\{\Phi_n^{(\infty)}\}$,
  reclassified]
\label{prob:completeness}
Prove that $\{\Phi_n^{(\infty)}\}$
is complete in $L^2(\mathbb{R})$
(Paper~IX, Problem~10).
This is no longer a prerequisite
for any result in this series.
\end{problem}
