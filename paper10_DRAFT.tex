\title[Mosco Convergence and Resolvent Stability
  of PSWF Concentration Forms]{%
  Mosco Convergence, Resolvent Stability,
  and the Bridge Theorem for
  PSWF Concentration Operators}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\maketitle

\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of self-adjoint
integral operators on $L^2(\mathbb{R})$.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent
stability (unconditional).}
We prove that the quadratic forms $\qc$
converge to a limit form $\qlim$ in the
sense of Mosco \cite{Mosco1969}, that
$\qlim$ is closed, and that the associated
resolvents converge strongly:
$(\Hc-z)^{-1} \xrightarrow{\mathrm{s}}
(\Hlim-z)^{-1}$ for every $z\notin[0,1]$.
All results in this part are proved
entirely within the form framework.
The operator-norm monotonicity of $\Hc$
in $c$ (Lemma~\ref{lem:monotone}) and
the density of the strong-convergence
domain $\mathcal{D}$ (Lemma~\ref{lem:dense})
are stated and proved as explicit lemmas.
A quantitative rate $O(c^{-1/4})$ is
cited as an external theorem from
Paper~VIII (Theorem~\ref{thm:rate-ext})
and stated without independent proof.

\textbf{(B) Bridge theorem (unconditional).}
The strong-limit operator $\Hstrong$
and the spectral-closure operator
$\Hspectral$ are identical:
$\Hstrong = \overline{\Hspectral}$.
The boundedness of $\overline{\Hspectral}$
is derived explicitly from the spectrum
of $\Hspectral$ before the bridge is invoked.

Conditionally on the spectral triple
of Connes--Consani--Moscovici
\cite{CCM2025} and Paper~IX,
Hypothesis~4.1, every non-trivial
zero of $\zeta(s)$ lies on the
critical line.
\end{abstract}

\section{Setup, Structural Lemmas,
  and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.

\begin{conv}[Ambient space and operators]
\label{conv:tilde}
All operators act on $L^2(\mathbb{R})$.
The PSWF concentration operator on
$L^2([0,1])$ is
\[
  (\mathcal{H}_c^{[0,1]} g)(x)
  := \int_0^1
     \frac{\sin c(x-y)}{\pi(x-y)}\,
     g(y)\,dy,
  \quad g \in L^2([0,1]),
\]
and its extension to $L^2(\mathbb{R})$
is $\Hc := P^*\mathcal{H}_c^{[0,1]}P$,
where $P : L^2(\mathbb{R}) \to L^2([0,1])$
is the restriction map and $P^*$ is
extension by zero.
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
      $0 \le \Hc \le I$ as operators;
      equivalently
      $0 \le \ip{\Hc f}{f} \le \|f\|^2$
      for all $f$, $c$
      (\cite{Slepian1961}).
\item\label{F4}
      $\Hc$ is self-adjoint on
      $L^2(\mathbb{R})$, with
      $\|\Hc\| \le 1$
      (\cite{Slepian1961};
      \cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[What is \emph{not} used
  before the Bridge]
\label{rem:no-spectral-input}
Sections~\ref{sec:setup}--\ref{sec:friedrichs}
use only Lemmas~\ref{lem:monotone}--%
\ref{lem:dense} and \eqref{F1}--\eqref{F4}.
No limit eigenvalues $\lambda_n^{(\infty)}$,
no limit eigenvectors $\Phi_n^{(\infty)}$,
no operator difference
$\widetilde{\mathcal{H}}_k - \Hc$,
and no Paper~VIII or Paper~IX results
appear.
\end{remark}

\begin{lemma}[PSWF operator monotonicity]
\label{lem:monotone}
For $0 < c \le c'$ and all
$f \in L^2(\mathbb{R})$:
\[
  \ip{\Hc f}{f} \le \ip{\mathcal{H}_{c'} f}{f}.
\]
In particular, $c \mapsto \ip{\Hc f}{f}$
is non-decreasing and bounded by
$\|f\|^2$.
\end{lemma}

\begin{proof}
Let $\hat{f}$ denote the Fourier
transform of $f$.
By the Fourier representation of the
PSWF concentration operator on
$L^2([0,1])$ (see \cite{Slepian1961};
Paper~V, Prop.~2.1):
\[
  \ip{\Hc f}{f}
  = \int_{-\infty}^{\infty}
    \left|\int_0^1
    e^{i c\xi x} f(x)\,dx\right|^2
    \mathbf{1}_{|\xi| \le 1}\,d\xi
    \cdot (\text{normalisation}).
\]
Equivalently, by Landau--Pollak
(\cite{Slepian1961}, Eq.~(3)):
\[
  \ip{\Hc f}{f}
  = \|P_c (Pf)\|_{L^2(\mathbb{R})}^2,
\]
where $P_c$ is the Fourier
projection onto $[-c,c]$.
Since $c \le c'$ implies
$[-c,c] \subset [-c',c']$,
the projection $P_c$ is dominated
by $P_{c'}$ in the operator order:
$P_c \le P_{c'}$.
Hence $\ip{\Hc f}{f} \le
\ip{\mathcal{H}_{c'} f}{f}$.
Boundedness by $\|f\|^2$ follows
from $P_c \le I$.
\end{proof}

% ---- FIX F32-2: Compactness via Parseval/Fourier ----

\begin{lemma}[Hilbert--Schmidt property
  and compactness]
\label{lem:hs-compact}
For each fixed $c_0 > 0$,
$\Hc[c_0]$ is a Hilbert--Schmidt operator
on $L^2(\mathbb{R})$,
hence compact.
\end{lemma}

\begin{proof}
We show $\mathcal{H}_{c_0}^{[0,1]}$
is Hilbert--Schmidt on $L^2([0,1])$;
compactness of $\Hc[c_0]$ then follows
from $\Hc[c_0] = P^*\mathcal{H}_{c_0}^{[0,1]}P$
and bounded-compact-bounded composition
(\cite{ReedSimon1975}, Thm.~VI.22).

\textit{HS on $L^2([0,1])$.}
Expand using the Fourier basis
$\{e^{2\pi i n x}\}_{n\in\mathbb{Z}}$
of $L^2([0,1])$.
The matrix element is
\[
  a_{mn}
  := \ip{\mathcal{H}_{c_0}^{[0,1]}
  e^{2\pi i n \cdot}}{e^{2\pi i m \cdot}}
  = \int_0^1\int_0^1
    K_{c_0}(x,y)
    e^{2\pi i n y}
    e^{-2\pi i m x}
    \,dy\,dx,
\]
where $K_{c_0}(x,y)
= \frac{\sin c_0(x-y)}{\pi(x-y)}$
(interpreted as $c_0/\pi$ at $x=y$
by continuous extension).
By Parseval on $[0,1]^2$:
\[
  \sum_{m,n\in\mathbb{Z}} |a_{mn}|^2
  = \|K_{c_0}\|_{L^2([0,1]^2)}^2.
\]
Now:
\[
  \|K_{c_0}\|_{L^2([0,1]^2)}^2
  = \int_0^1\int_0^1
    \left|\frac{\sin c_0(x-y)}{\pi(x-y)}
    \right|^2 dx\,dy.
\]
Substitute $u = x-y \in [-1,1]$;
the integrand satisfies
$|K_{c_0}(x,y)| \le c_0/\pi$
(since $|\sin(t)/t| \le 1$
and $\sin(c_0 u)/(\pi u) = c_0/\pi$
at $u=0$).
Hence:
\[
  \|K_{c_0}\|_{L^2([0,1]^2)}^2
  \le (c_0/\pi)^2 \cdot |[0,1]^2|
  = (c_0/\pi)^2 < \infty.
\]
Thus $\sum_{m,n} |a_{mn}|^2 < \infty$,
which is the Hilbert--Schmidt condition
(\cite{ReedSimon1975}, Thm.~VI.22).
\end{proof}

% ---- FIX F32-3: Density as explicit lemma ----

\begin{lemma}[Density of $\mathcal{D}$]
\label{lem:dense}
Let
$\mathcal{D} := \{f \in L^2(\mathbb{R})
: \lim_{c\to\infty}\Hc f
\text{ exists in } L^2\}$.
Then $\mathcal{D}$ is dense in
$L^2(\mathbb{R})$.
\end{lemma}

\begin{proof}
It suffices to show that every
compactly supported $f \in L^2$ lies
in $\mathcal{D}$.

Let $f \in L^2(\mathbb{R})$ have
support in $[-R, R]$ for some $R > 0$.
By the Fourier representation
(Lemma~\ref{lem:monotone}):
\[
  \Hc f = P^* P_c(\hat{f}\,\hat{\cdot}\,),
\]
where $P_c$ is Fourier projection
onto $[-c, c]$.
As $c \to \infty$, $P_c\hat{g} \to \hat{g}$
in $L^2(\mathbb{R})$ for every
$g \in L^2(\mathbb{R})$
(Plancherel + dominated convergence
on the Fourier side).
Hence $\Hc f \to P^* Pf = \Hc[\infty] f$
in $L^2$, and
$f \in \mathcal{D}$.

Since compactly supported $L^2$-functions
are dense in $L^2(\mathbb{R})$,
$\mathcal{D}$ is dense.
\end{proof}

\begin{remark}[Summary of setup]
\label{rem:setup-summary}
Lemma~\ref{lem:monotone} provides:
for all $f$, the map
$c \mapsto \ip{\Hc f}{f}$ is
non-decreasing and bounded;
hence the limit form
\[
  \qlim(f,f)
  := \lim_{c\to\infty}\ip{\Hc f}{f}
  = \sup_{c>0}\ip{\Hc f}{f}
\]
exists.
Lemma~\ref{lem:hs-compact} provides:
each $\Hc[c_0]$ is compact.
Lemma~\ref{lem:dense} provides:
$\mathcal{D}$ is dense.
These three lemmas are the complete
basis for the form thread.
\end{remark}

\section{Strong-Limit Operator
  \texorpdfstring{$\Hstrong$}{H-str}}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Density-extension principle]
\label{lem:sc}
Let $(A_c)_{c>0} \subset \mathcal{B}(\mathcal{H})$
with $M := \sup_c\|A_c\| < \infty$,
and let $\mathcal{D} \subset \mathcal{H}$
be a dense subspace on which
$A_c f \to Bf$ for every $f \in \mathcal{D}$.
Then $A_c f$ converges for every
$f \in \mathcal{H}$, the limit defines
$A \in \mathcal{B}(\mathcal{H})$
with $\|A\| \le M$, and $A_c \to A$
strongly.
\end{lemma}

\begin{proof}
For any $f \in \mathcal{H}$ and
$\varepsilon > 0$, choose
$g \in \mathcal{D}$ with
$\|f-g\| < \varepsilon/(3M)$.
For large $c, c'$:
$\|A_c f - A_{c'} f\|
\le 2M\|f-g\|
+ \|A_c g - A_{c'} g\|
\le \varepsilon$.
Completeness gives $Af$;
$A_c \to A$ strongly.
\end{proof}

\begin{lemma}[Existence of $\Hstrong$]
\label{lem:hlim-bounded}
There exists a unique
$\Hstrong \in \mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hstrong\| \le 1$ and
$\Hc \to \Hstrong$ strongly.
\end{lemma}

\begin{proof}
Apply Lemma~\ref{lem:sc} with
$A_c := \Hc$, dense domain
$\mathcal{D}$ from Lemma~\ref{lem:dense},
and $M := 1$ from \eqref{F4}.
\end{proof}

\begin{corollary}[Form equals operator]
\label{cor:qlim-op}
$\qlim(f,f) = \ip{\Hstrong f}{f}$
for all $f \in L^2(\mathbb{R})$.
\end{corollary}
\begin{proof}
$\qlim(f,f)
= \lim_c\ip{\Hc f}{f}
= \ip{\Hstrong f}{f}$
by strong convergence and inner product
continuity.
\end{proof}

\begin{corollary}[Self-adjointness]
\label{cor:sa-strong}
$\Hstrong$ is self-adjoint, positive,
$\sigma(\Hstrong) \subset [0,1]$.
\end{corollary}
\begin{proof}
Symmetry from strong limit of symmetric
family; BSA principle for bounded
symmetric $\Hstrong$
(\cite{ReedSimon1975}, Thm.~III.6);
positivity from $\qlim \ge 0$.
\end{proof}

\section{Limit Form and Mosco
  Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
$\qlim(f,f) := \lim_{c\to\infty}
\ip{\Hc f}{f}$
(exists by Lemma~\ref{lem:monotone}
and \eqref{F1});
extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc) \xrightarrow{\mathrm{M}} \qlim$ if:
\begin{enumerate}[\rm(M1)]
\item For every $c_n \to \infty$
      and $f_n \rightharpoonup f$:
      $\qlim(f,f) \le
      \liminf_n q_{c_n}(f_n,f_n)$.
\item For every $f$, there exist
      $c_n \to \infty$ and
      $f_n \to f$ strongly with
      $q_{c_n}(f_n,f_n) \to \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Complete continuity
  of fixed form]
\label{lem:wc}
For each fixed $c_0 > 0$,
the map $f \mapsto q_{c_0}(f,f)$
is sequentially weakly continuous:
$f_n \rightharpoonup f
\Rightarrow
q_{c_0}(f_n,f_n) \to q_{c_0}(f,f)$.
\end{lemma}
\begin{proof}
Since $\Hc[c_0]$ is compact
(Lemma~\ref{lem:hs-compact}),
it is completely continuous:
$f_n \rightharpoonup f$ implies
$\Hc[c_0] f_n \to \Hc[c_0] f$
strongly
(\cite{ReedSimon1975}, Thm.~VI.1;
equivalence of compactness and
complete continuity on Hilbert spaces).
Hence:
\begin{align*}
  q_{c_0}(f_n,f_n)
  &= \ip{\Hc[c_0] f_n}{f_n} \\
  &\to \ip{\Hc[c_0] f}{f}
  = q_{c_0}(f,f),
\end{align*}
where convergence uses strong
$\Hc[c_0] f_n \to \Hc[c_0] f$
and weak $f_n \rightharpoonup f$
(strong-weak product $\to$ inner product;
\cite{ReedSimon1975}, Lem.~II.3).
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc \xrightarrow{\mathrm{M}} \qlim$.
\end{theorem}
\begin{proof}
\textit{(M2).}
Take $f_n := f$, $c_n \to \infty$:
$q_{c_n}(f,f)
\to \qlim(f,f)$
by definition and
Lemma~\ref{lem:monotone}.

\textit{(M1).}
Let $c_n \to \infty$,
$f_n \rightharpoonup f$.
For every fixed $c_0 > 0$,
once $c_n \ge c_0$,
Lemma~\ref{lem:monotone} gives
$q_{c_n}(f_n,f_n) \ge q_{c_0}(f_n,f_n)$.
Hence:
\[
  \liminf_n q_{c_n}(f_n,f_n)
  \ge \liminf_n q_{c_0}(f_n,f_n)
  = q_{c_0}(f,f)
  \quad
  (\text{Lemma~\ref{lem:wc}}).
\]
This holds for every fixed $c_0$.
The left side is independent of $c_0$,
so:
\[
  \liminf_n q_{c_n}(f_n,f_n)
  \ge \sup_{c_0>0}q_{c_0}(f,f)
  = \qlim(f,f).
  \qedhere
\]
\end{proof}

\section{Closedness and
  Riesz--Kato Identification}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of $\qlim$]
\label{thm:form-closed}
$\qlim$ is a closed form.
\end{theorem}
\begin{proof}
$\qlim(f,f) = \ip{\Hstrong f}{f}$
(Corollary~\ref{cor:qlim-op});
$\Hstrong \in \mathcal{B}(L^2)$.
The form norm $\|f\|_{\qlim}^2
:= \|f\|^2 + \qlim(f,f)$
is equivalent to $\|f\|^2$
(by boundedness of $\Hstrong$);
hence $\qlim$ is closed.
\end{proof}

\begin{theorem}[Riesz--Kato identification]
\label{thm:friedrichs}
$T_{\qlim} = \Hstrong$,
where $T_{\qlim}$ is the unique
bounded self-adjoint operator with
$\qlim(f,g) = \ip{T_{\qlim}f}{g}$
for all $f,g$.
\end{theorem}
\begin{proof}
Corollary~\ref{cor:qlim-op} gives
$\ip{\Hstrong f}{g} = \qlim(f,g)$
for all $f,g$; uniqueness of the
Riesz representative gives
$T_{\qlim} = \Hstrong$.
\end{proof}

\begin{remark}[Status of this section]
\label{rem:rate-status}
No result in
Sections~\ref{sec:setup}--%
\ref{sec:bridge}
depends on this section.
The rate is stated for completeness.
\end{remark}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
For all $f \in L^2(\mathbb{R})$
and $c \ge c_0(\kappa)$:
\[
  0 \le \qlim(f,f) - \qc(f,f)
  \le C_\kappa\, c^{-1/4}\, \|f\|^2.
\]
\end{theorem}

\begin{proof}[Proof (external citation)]
Lower bound: Lemma~\ref{lem:monotone}.
Upper bound: Paper~VIII, Cor.~4.3
establishes
$\|\Hstrong - \Hc\| \le C_\kappa c^{-1/4}$.
Hence:
$\qlim(f,f) - \qc(f,f)
= \ip{(\Hstrong-\Hc)f}{f}
\le \|\Hstrong - \Hc\|\cdot\|f\|^2
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{proof}

\begin{remark}[Obstruction to
  independent proof]
\label{rem:no-indep-proof}
Any attempt to derive the upper bound
independently runs into the following:
passing from scalar eigenvalue asymptotics
$\lambda_n^{(c)} \nearrow \lambda_n^{(\infty)}$
to a uniform operator-norm bound requires
control of the eigenvectors
$\phi_n^{(c)}$ as $c$ varies.
Representations of the form
$\qlim(f,f) = \sum_n \lambda_n^{(\infty)}
|\langle f,\phi_n^{(c)}\rangle|^2$
mix two non-commuting limits
(fixed-$c$ basis, $c\to\infty$ weights)
and are not valid.
The correct proof requires uniform
eigenvector asymptotics,
which is the content of Paper~VIII.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the \emph{only} section
that imports results from Paper~IX.

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      Paper~IX, Definition~2.3 constructs
      \[
        \Hspectral f
        := \sum_{n \ge 0}
           \lambda_n^{(\infty)}
           \ip{f}{\Phi_n^{(\infty)}}
           \Phi_n^{(\infty)},
      \]
      with domain
      $\Dom(\Hspectral)
      = \{f : \sum_n
      (\lambda_n^{(\infty)})^2
      |\ip{f}{\Phi_n^{(\infty)}}|^2
      < \infty\}$.
      Here $\{\Phi_n^{(\infty)},
      \lambda_n^{(\infty)}\}$ are the
      limit PSWF eigensystem.
      $\Hspectral$ is symmetric,
      densely defined, and closable
      (Paper~IX, Thm.~2.5, 3.2).

% ---- FIX F32-4: Boundedness of closure made explicit ----
\item\label{IXb}
      \textit{The closure
      $\overline{\Hspectral}$ is bounded.}
      On $\Dom(\Hspectral)$:
      \[
        \|\Hspectral f\|^2
        = \sum_n
          (\lambda_n^{(\infty)})^2
          |\ip{f}{\Phi_n^{(\infty)}}|^2
        \le \sum_n
          |\ip{f}{\Phi_n^{(\infty)}}|^2
        \le \|f\|^2,
      \]
      using $\lambda_n^{(\infty)} \in [0,1]$
      (Paper~IX, Def.~2.3)
      and $\{\Phi_n^{(\infty)}\}$ being
      an orthonormal set.
      Hence $\Hspectral$ extends by
      continuity to a bounded operator
      $\overline{\Hspectral}
      \in \mathcal{B}(L^2)$
      with $\|\overline{\Hspectral}\|
      \le 1$.
      (This is the graph-closure
      principle for bounded symmetric
      operators on a dense domain:
      \cite{ReedSimon1975},
      Thm.~I.7.)

\item\label{IXc}
      \textit{Agreement on $\mathcal{D}$.}
      For every $f \in \mathcal{D}$:
      $\Hc f \to \Hstrong f$
      (definition, Lemma~\ref{lem:hlim-bounded}),
      and also
      $\Hc f \to \overline{\Hspectral} f$
      (Paper~IX, Remark~2.6).
      Uniqueness of $L^2$-limits:
      $\Hstrong f
      = \overline{\Hspectral} f$
      on all of $\mathcal{D}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong = \overline{\Hspectral}$
as bounded operators on
$L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Both operators are bounded on
$L^2(\mathbb{R})$
(\eqref{IXb} and
Lemma~\ref{lem:hlim-bounded}).
They agree on the dense set
$\mathcal{D}$
(\eqref{IXc}).
Two bounded operators that agree on
a dense subspace of a Hilbert space
are equal everywhere
(\cite{ReedSimon1975}, Lem.~II.3;
applied to their difference).
\end{proof}

\begin{remark}[After the bridge]
\label{rem:bridge}
We write $\Hlim :=
\Hstrong = \overline{\Hspectral}$.
\end{remark}

\begin{remark}[$\Hlim$ is not a
  multiplication operator]
\label{rem:hlim-not-mult}
$\sigma(\Hlim) \subset
\overline{\{\lambda_n^{(\infty)}
: n \ge 0\}}$;
Landau--Pollak accumulation in
$(0,1)$ (\cite{Slepian1961})
shows $\sigma(\Hlim)
\not\subset \{0,1\}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every $z \in \mathbb{C}
\setminus [0,1]$:
$(\Hc - z)^{-1}
\xrightarrow{\mathrm{s}}
(\Hlim - z)^{-1}$.
\end{corollary}
\begin{proof}
By Theorem~\ref{thm:mosco},
$\qc \xrightarrow{\mathrm{M}} \qlim$.
By Corollary~\ref{cor:sa-strong},
$\sigma(\Hlim) \subset [0,1]$.
Apply \cite{KuwaeShioya2003},
Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
\label{rem:unconditional}
Without any hypothesis:
$\Hlim = \Hlim^* \ge 0$,
$\|\Hlim\| \le 1$,
$T_{\qlim} = \Hlim$,
$\qc \xrightarrow{\mathrm{M}} \qlim$,
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{s}} (\Hlim-z)^{-1}$,
$\Hstrong = \overline{\Hspectral}$.
With external Paper~VIII input:
rate $O(c^{-1/4})$.
\end{remark}

\begin{corollary}[Conditional RH]
\label{cor:rh}
Assume Paper~VIII, Cor.~4.3 (as in
Theorem~\ref{thm:rate-ext}),
CCM (\cite{CCM2025}, Thm.~1.3),
and Paper~IX, Hypothesis~4.1.
Then every non-trivial zero $\rho$
of $\zeta(s)$ satisfies
$\mathrm{Re}(\rho) = \tfrac{1}{2}$.
\textit{No unconditional claim about
zeta zeros is made in this paper.}
\end{corollary}
\begin{proof}
Under Paper~IX, Hyp.~4.1,
$\Hlim$ is essentially self-adjoint
(Paper~IX, Cor.~5.2).
Under CCM, Thm.~1.3:
$\rho \in \sigma_{\mathrm{app}}(\Hlim)
= \sigma(\Hlim) \subset \mathbb{R}$;
the CCM spectral embedding forces
$\mathrm{Re}(\rho) = 1/2$.
\end{proof}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Completeness]
\label{prob:completeness}
Prove Paper~IX, Hypothesis~4.1:
$\{\Phi_n^{(\infty)}\}$ is complete
in $L^2(\mathbb{R})$.
\end{problem}

\begin{problem}[Independent rate proof]
\label{prob:rate}
Give an independent proof of
$\qlim(f,f) - \qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$
without using Paper~VIII;
see Remark~\ref{rem:no-indep-proof}
for the key obstruction.
\end{problem}

\begin{problem}[Quantitative resolvent]
\label{prob:resolvent}
Find $F(c,z) \to 0$ with
$\|(\Hc-z)^{-1}-(\Hlim-z)^{-1}\|
\le F(c,z)$.
\end{problem}

\begin{problem}[Spectrum of $\Hlim$]
\label{prob:spectrum}
Is $\sigma(\Hlim) = [0,1]$?
\end{problem}
