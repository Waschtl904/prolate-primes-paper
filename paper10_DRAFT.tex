\begin{abstract}
The prolate spheroidal wave function
(PSWF) concentration operators
$\Hc$ form a uniformly bounded,
monotone family of self-adjoint
integral operators on $L^2(\mathbb{R})$.
We prove that the strong operator
limit $\Hlim:=\lim_{c\to\infty}\Hc$
exists on all of $L^2(\mathbb{R})$,
is bounded and symmetric
(hence self-adjoint by a standard
argument), and that the associated
quadratic forms converge in the
sense of Mosco at rate $O(c^{-1/4})$.
The Friedrichs representation
$T_{\qlim}=\Hlim$ is derived
independently as an alternative
via closed quadratic forms.
As an unconditional consequence,
the resolvents
$(\Hc-z)^{-1}$ converge strongly
to $(\Hlim-z)^{-1}$ for every
$z\notin[0,1]$.

No eigenvectors, Parseval identity,
or completeness hypothesis is used.
Conditionally on the spectral
triple construction of
Connes--Consani--Moscovici
\cite{CCM2025} and two explicitly
stated open gaps, every non-trivial
zero of $\zeta(s)$ lies on the
critical line; no unconditional
statement about zeta zeros is
claimed.
\end{abstract}

\section*{Motivation and Context}

The prolate spheroidal wave functions
(PSWFs) arise as the eigenfunctions
of the time-frequency concentration
operator: given a bandwidth $c>0$,
the operator $\Hc$ measures how well
a signal is simultaneously
concentrated in time $[-1,1]$ and
frequency $[-c,c]$
(\cite{Slepian1961}).
As $c\to\infty$, perfect concentration
becomes possible for more and more
functions, and the limit operator
$\Hlim$ captures this asymptotic
regime.

Understanding $\Hlim$ as a
self-adjoint operator is of
independent interest in
time-frequency analysis,
approximation theory, and
signal processing:
the spectral theory of $\Hlim$
governs the limiting behaviour of
PSWF expansions and the stability
of bandlimited approximations at
large bandwidth.
The Mosco convergence of the
associated forms at rate
$O(c^{-1/4})$ gives a quantitative
version of this stability,
and the resolvent convergence
(Corollary~\ref{cor:resolvent})
shows that the spectral structure
of $\Hc$ is inherited by $\Hlim$
in the strong sense.

The connection to the Riemann zeta
function is conditional and
programmatic
(Section~\ref{sec:consequences});
the operator-theoretic results of
Sections~\ref{sec:setup}--\ref{sec:friedrichs}
are unconditional and stand
independently.

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from
Papers~V--IX.

\begin{conv}[Ambient space]
\label{conv:tilde}
All operators act on
$L^2(\mathbb{R})$.
The PSWF operator
$\mathcal{H}_c^{[0,1]}$ on
$L^2([0,1])$ (\cite{Slepian1961})
is extended by
$\Hc:=P^*\mathcal{H}_c^{[0,1]}P$,
where $P$ is restriction and
$P^*$ is extension by zero.
$\Hc\in\mathcal{B}(L^2(\mathbb{R}))$
and $0\le\Hc\le I$.
\end{conv}

\begin{enumerate}[(IX.1)]

\item\label{IX3}
      $0\le\ip{\Hc f}{f}\le\|f\|^2$
      for all $f$, all $c$.

\item\label{IX4}
      $c\mapsto\ip{\Hc f}{f}$
      is non-decreasing and,
      for every $f$, increases to
      $\qlim(f,f)=\ip{\Hlim f}{f}$
      as $c\to\infty$
      (Paper~V, Prop.~2.2;
      Lemma~\ref{lem:hlim-bounded}).

\item\label{IX5}
      $\Dom(\Hlim)$ is a dense
      linear subspace of
      $L^2(\mathbb{R})$.
      Concretely, Paper~IX,
      Thm.~2.5(a) shows that all
      compactly supported
      $L^2(\mathbb{R})$-functions
      lie in $\Dom(\Hlim)$;
      in particular $\Dom(\Hlim)$
      is non-trivial and dense.

\item\label{IX6}
      $\Hc$ is self-adjoint,
      positive, $\|\Hc\|\le 1$
      (\cite{Slepian1961};
      \cite{ReedSimon1975},
      Thm.~VI.1).

\item\label{IX7}
      \textit{Constructive domain.}
      $\Dom(\Hlim)
      :=\{f\in L^2(\mathbb{R})
      :\lim_{c\to\infty}\Hc f
      \text{ exists in }
      L^2(\mathbb{R})\}$.
      This is a linear subspace
      (limits of linear maps
      are linear where they exist).
      Paper~VII, Thm.~C:
      the map
      $f\mapsto\lim_c\Hc f$
      is bounded on $\Dom(\Hlim)$
      with norm $\le 1$.
      In particular,
      $\|\Hc f-\Hlim f\|\to 0$
      for $f\in\Dom(\Hlim)$.
      No circularity: the domain
      is defined by existence of
      the limit, not by properties
      of the limit operator.

\item\label{IX2norm}
      \textit{Operator-norm
      convergence.}
      $\|\Hc-\Hlim\|
      \le C_\kappa c^{-1/4}$,
      where $\|\cdot\|$ is the
      operator norm on
      $\mathcal{B}(L^2(\mathbb{R}))$.
      This is proven in
      Paper~IX, Thm.~3.1,
      via the Landau--Pollak
      asymptotics for PSWF
      eigenvalue decay
      (\cite{Slepian1961};
      Paper~VIII, Prop.~4.2);
      the estimate holds for
      $c\ge c_0(\kappa)$.
      Used after
      Lemma~\ref{lem:hlim-bounded}.

\end{enumerate}

\begin{remark}[Objects not used]
Eigenvectors $\Phi_n^{(\infty)}$,
Parseval, completeness:
none appear in this paper.
\end{remark}

%%%─── LEMMA SC
\begin{lemma}[Stable core;
  density-extension]
\label{lem:sc}
This is a standard result
(\cite{Rudin1991}, Thm.~4.1;
\cite{Conway1990}, I.4;
cf.\ also \cite{ReedSimon1975},
Thm.~III.5);
proof included for completeness.

Let $(A_c)\subset
\mathcal{B}(\mathcal{H})$
with $M:=\sup_c\|A_c\|<\infty$,
and let $\mathcal{D}\subset
\mathcal{H}$ be a dense linear
subspace.
Suppose $A_c f\to Bf$ pointwise
for all $f\in\mathcal{D}$,
where $B:\mathcal{D}\to\mathcal{H}$
is a linear map
(linearity follows from
pointwise limits of linear maps).
Then:
\begin{enumerate}[\rm(a)]
\item\label{sc:bdd}
      $B$ is bounded:
      $\|Bf\|\le M\|f\|$
      for all $f\in\mathcal{D}$.
\item\label{sc:ext}
      The limit $Af:=\lim_c A_c f$
      exists in $\mathcal{H}$
      for every $f\in\mathcal{H}$,
      $A\in\mathcal{B}(\mathcal{H})$,
      $\|A\|\le M$, and $A$
      extends $B$.
\item\label{sc:strong}
      $A_c\to A$ strongly on
      all of $\mathcal{H}$.
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a).}
$\|Bf\|=\lim_c\|A_c f\|
\le M\|f\|$.
Linearity: for
$f,g\in\mathcal{D}$ and scalars
$\alpha,\beta$,
\[
  A_c(\alpha f+\beta g)
  = \alpha A_c f+\beta A_c g
  \;\to\;
  \alpha Bf+\beta Bg,
\]
so $B(\alpha f+\beta g)
=\alpha Bf+\beta Bg$.

\textit{(b).}
For $f\in\mathcal{H}$,
$\varepsilon>0$, choose
$f_n\in\mathcal{D}$ with
$\|f-f_n\|<\varepsilon$:
\[
  \|A_c f-A_{c'}f\|
  \le 2M\varepsilon
  +\|A_c f_n-A_{c'}f_n\|\to 0,
\]
so $(A_c f)$ is Cauchy;
its limit $Af$ exists in
$\mathcal{H}$.
$\|Af\|=\lim_c\|A_c f\|\le M\|f\|$;
$A$ extends $B$ since
$A_c f\to Bf$ for
$f\in\mathcal{D}$.

\textit{(c).}
$\limsup_c\|A_c f-Af\|
\le 2M\varepsilon$;
$\varepsilon$ arbitrary.
\end{proof}

%%%─── LEMMA HLIM BOUNDED
\begin{lemma}[Existence and
  boundedness of $\Hlim$]
\label{lem:hlim-bounded}
The limit $\Hlim f
:=\lim_{c\to\infty}\Hc f$
exists in $L^2(\mathbb{R})$
for every $f\in L^2(\mathbb{R})$,
and $\Hlim\in
\mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hlim\|\le 1$.
\end{lemma}

\begin{proof}
Let $\mathcal{D}:=\Dom(\Hlim)$
(dense and non-trivial,
\eqref{IX5}) and let $B$ be
the pointwise limit
$Bf:=\lim_c\Hc f$ on
$\mathcal{D}$ (well-defined
by \eqref{IX7}).
By Lemma~\ref{lem:sc}(a),
$B$ is bounded with $\|B\|\le 1$.
Apply Lemma~\ref{lem:sc}(b,c)
with $A_c:=\Hc$, $M:=1$:
the limit exists in
$L^2(\mathbb{R})$ for every $f$,
$\|\Hlim\|\le 1$,
and $\Hc\to\Hlim$ strongly.
\end{proof}

\begin{remark}[Logical order]
\label{rem:order}
Lemma~\ref{lem:hlim-bounded}
depends only on
\eqref{IX5}--\eqref{IX7}
and Lemma~\ref{lem:sc}.
It precedes \eqref{IX2norm},
Definition~\ref{def:qlim},
and all results in
Sections~\ref{sec:sa}--\ref{sec:friedrichs}.
\end{remark}

\section{Self-Adjointness of $\Hlim$}
\label{sec:sa}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Self-adjointness]
\label{thm:sa}
$\Hlim$ is bounded, symmetric,
and self-adjoint on
$L^2(\mathbb{R})$
with $\sigma(\Hlim)\subset[0,1]$.
\end{theorem}

\begin{proof}
\textbf{Boundedness.}
Lemma~\ref{lem:hlim-bounded}:
$\Hlim\in\mathcal{B}(L^2)$,
$\|\Hlim\|\le 1$.

\textbf{Symmetry.}
For $f,g\in L^2(\mathbb{R})$,
by joint continuity of the inner
product (on Hilbert spaces) and
strong convergence
$\Hc f\to\Hlim f$,
$\Hc g\to\Hlim g$
(Lemma~\ref{lem:hlim-bounded}):
\[
  \ip{\Hlim f}{g}
  = \lim_c\ip{\Hc f}{g}
  = \lim_c\ip{f}{\Hc g}
  = \ip{f}{\Hlim g},
\]
using self-adjointness of each
$\Hc$ (\eqref{IX6}).

\textbf{Self-adjointness.}
A bounded symmetric operator
on a Hilbert space is automatically
self-adjoint
(\cite{ReedSimon1975}, Thm.~III.6).
Hence $\Hlim=\Hlim^*$.

\textbf{Spectrum.}
Since $\Hlim$ is self-adjoint,
$\sigma(\Hlim)\subset\mathbb{R}$.
Since $\|\Hlim\|\le 1$,
$\sigma(\Hlim)\subset[-1,1]$.
Since $\ip{\Hlim f}{f}\ge 0$
for all $f$ (\eqref{IX3},
Lemma~\ref{lem:hlim-bounded}),
$\Hlim\ge 0$, so
$\sigma(\Hlim)\subset[0,1]$.
\end{proof}

\begin{remark}[R\^ole of Friedrichs]
Since $\Hlim\in\mathcal{B}(L^2)$
is already self-adjoint,
the Friedrichs construction
(Section~\ref{sec:friedrichs})
is not needed for self-adjointness.
It is retained to provide the
form identification
$T_{\qlim}=\Hlim$
(used in Section~\ref{sec:consequences})
and as an independent derivation
via Mosco $\Rightarrow$ closed
form $\Rightarrow$ unique
self-adjoint operator.
\end{remark}

\section{Limit Form}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
For $f\in L^2(\mathbb{R})$, set
\[
  \qlim(f,f)
  :=\lim_{c\to\infty}\ip{\Hc f}{f}
  =\ip{\Hlim f}{f}
\]
(equality by
Lemma~\ref{lem:hlim-bounded}).
$\Dom(\qlim)=L^2(\mathbb{R})$;
extend by polarisation.
\end{definition}

\begin{theorem}[Existence]
\label{thm:qlim-exists}
$\qlim\ge 0$ is a bounded
Hermitian form on $L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Monotonicity \eqref{IX4} and
bound \eqref{IX3}.
\end{proof}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence]
\label{def:mosco}
$(\qc)$ converge to $\qlim$
in the sense of Mosco
\cite{Mosco1969}:
\begin{enumerate}[\rm(M1)]
\item For every $c_n\to\infty$
      and $f_n\rightharpoonup f$:
      \[
        \qlim(f,f)
        \le\liminf_{n\to\infty}
        q_{c_n}(f_n,f_n).
      \]
\item For every $f$ there exist
      $c_n\to\infty$ and
      $f_n\to f$ strongly with
      $q_{c_n}(f_n,f_n)
      \to\qlim(f,f)$.
\end{enumerate}
Both $c_n$ and $(f_n)$ share
index $n$ (standard Mosco
convention \cite{Mosco1969}).
\end{definition}

\begin{lemma}[HS and weak continuity]
\label{lem:hs}
For each fixed $c_0>0$:
\begin{enumerate}[\rm(a)]
\item $\widetilde{\mathcal{H}}_{c_0}
      \in\mathcal{B}_2
      (L^2(\mathbb{R}))$ (compact).
\item $f_n\rightharpoonup f$
      $\Rightarrow$
      $\ip{\widetilde{\mathcal{H}}_{c_0}
        f_n}{f_n}
      \to
      \ip{\widetilde{\mathcal{H}}_{c_0}
        f}{f}$.
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a).}
$\|K_{c_0}\|_{\infty}\le c_0/\pi$;
HS $\Rightarrow$ compact
(\cite{ReedSimon1975}, VI.22).

\textit{(b).}
Write
$\ip{\widetilde{\mathcal{H}}_{c_0}
f_n}{f_n}
-\ip{\widetilde{\mathcal{H}}_{c_0}
f}{f}
= I + II$,
where
$I := \ip{\widetilde{\mathcal{H}}_{c_0}
(f_n-f)}{f_n}$
and
$II := \ip{\widetilde{\mathcal{H}}_{c_0}
f}{f_n-f}$.
For $II$: $f_n\rightharpoonup f$
implies $II\to 0$ directly.
For $I$: since
$\widetilde{\mathcal{H}}_{c_0}$
is compact (part (a)) and
$f_n\rightharpoonup f$,
we have
$\widetilde{\mathcal{H}}_{c_0}(f_n-f)
\to 0$ strongly
(compact operators map weakly
convergent sequences to
norm-convergent ones,
\cite{ReedSimon1975}, VI.11);
hence
$|I|\le\sup_n\|f_n\|
\cdot\|\widetilde{\mathcal{H}}_{c_0}
(f_n-f)\|\to 0$
(uniform bound: \cite{ReedSimon1975},
III.5).
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc\to\qlim$ in Mosco sense.
\end{theorem}

\begin{proof}
\textit{(M2):}
$f_n:=f$, $c_n\to\infty$:
$q_{c_n}(f,f)\to\ip{\Hlim f}{f}
=\qlim(f,f)$
(Lemma~\ref{lem:hlim-bounded}).

\textit{(M1):}
Let $c_n\to\infty$,
$f_n\rightharpoonup f$.
Fix $c_0>0$;
since $c_n\to\infty$,
$c_n\ge c_0$ for all large $n$,
hence by monotonicity \eqref{IX4}:
\[
  q_{c_n}(f_n,f_n)
  \ge q_{c_0}(f_n,f_n).
\]
By Lemma~\ref{lem:hs}(b)
(applied with the fixed $c_0$):
\[
  \liminf_{n\to\infty}
  q_{c_n}(f_n,f_n)
  \ge
  \lim_{n\to\infty}
  q_{c_0}(f_n,f_n)
  = q_{c_0}(f,f).
\]
Since $c_0>0$ is arbitrary,
we take the supremum over $c_0$:
since
$(q_{c_0}(f,f))_{c_0>0}$
is monotone increasing and
bounded above by $\|f\|^2$
(\eqref{IX3}--\eqref{IX4}),
its supremum equals $\qlim(f,f)$,
and we conclude
$\liminf_n q_{c_n}(f_n,f_n)
\ge\qlim(f,f)$.
\end{proof}

\begin{remark}[Mosco $\Rightarrow$
  resolvent convergence]
\cite{KuwaeShioya2003}, Thm.~2.4:
Mosco $\Rightarrow$ strong
resolvent convergence;
see Corollary~\ref{cor:resolvent}.
\end{remark}

\section{Closedness}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness]
\label{thm:form-closed}
$\qlim\ge 0$ and $\qlim$ is closed.
\end{theorem}
\begin{proof}
Let $(f_k)$ be
$\|\cdot\|_{\qlim}$-Cauchy;
$f_k\to f$ in $L^2$ (hence weakly).
Apply (M1) with the constant
sequence $c_n:=c$ (valid since
the $\liminf$ argument holds for
any fixed $c$):
$q_c(f,f)\le\liminf_k q_c(f_k,f_k)$.
Send $c\to\infty$;
parallelogram law gives
$\qlim(f_k-f,f_k-f)\to 0$
(\cite{Kato1966}, Def.~VI.1.3).
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative rate]
\label{thm:form-convergence}
For all $f\in L^2(\mathbb{R})$,
$c\ge c_0(\kappa)$:
\[
  0\le\qlim(f,f)-\qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\]
\end{theorem}
\begin{proof}
Lower: monotonicity \eqref{IX4}.
Upper:
$\ip{(\Hlim-\Hc)f}{f}
\le\|\Hlim-\Hc\|\cdot\|f\|^2
\le C_\kappa c^{-1/4}\|f\|^2$
by \eqref{IX2norm}
(operator-norm bound).
\end{proof}

\section{Friedrichs Extension
  (Alternative Derivation)}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

This section gives an independent
derivation of $T_{\qlim}=\Hlim$
via the Kato--Friedrichs
representation theorem.
Self-adjointness follows already
from Theorem~\ref{thm:sa}.

\begin{theorem}[Kato representation]
\label{thm:kato-rep}
A densely defined, closed,
semibounded form $q$ determines
a unique self-adjoint $T$,
$\Dom(T)\subset\Dom(q)$, with
$q(f,g)=\ip{Tf}{g}$.
(\cite{Kato1966}, VI.2.1;
\cite{ReedSimon1975}, VIII.15.)
\end{theorem}

\begin{theorem}[Form identification]
\label{thm:friedrichs}
$T_{\qlim}=\Hlim$.
\end{theorem}

\begin{proof}
By Theorems~\ref{thm:qlim-exists}
and~\ref{thm:form-closed},
$\qlim$ is a closed semibounded
form on all of $L^2(\mathbb{R})$;
Theorem~\ref{thm:kato-rep}
gives $T_{\qlim}$.

\textbf{Step~1.}
Let $f,g\in L^2(\mathbb{R})$.
Since $\Hc f\to\Hlim f$ strongly
(Lemma~\ref{lem:hlim-bounded}),
by joint continuity of the inner
product:
$\ip{\Hc f}{g}\to\ip{\Hlim f}{g}$.
Also
$\ip{\Hc f}{g}=\qc(f,g)
\to\qlim(f,g)$
(Definition~\ref{def:qlim}).
Hence $\ip{\Hlim f}{g}=\qlim(f,g)$
for all $g$,
so $f\in\Dom(T_{\qlim})$ and
$T_{\qlim}f=\Hlim f$.
Thus $\Hlim\subset T_{\qlim}$.
Since both $\Hlim$ and $T_{\qlim}$
are bounded self-adjoint operators
defined on all of $L^2(\mathbb{R})$,
the inclusion $\Hlim\subset T_{\qlim}$
implies $\Hlim=T_{\qlim}$.
\end{proof}

\begin{remark}[Audit]
Every step cites one of:
\eqref{IX3}--\eqref{IX7},
Kato~VI.2.1,
Lemmas~\ref{lem:sc},
\ref{lem:hlim-bounded},
\ref{lem:hs}.
No eigenvector, Parseval,
or completeness is used.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Spectral inclusion]
\label{cor:spec}
$\sigma(\Hlim)\subset[0,1]
\subset\mathbb{R}$.
\end{corollary}

\begin{proposition}[Spectral
  structure, unconditional]
\label{prop:spec-structure}
Let $\lambda\in(0,1)$.
If $\lambda$ is an eigenvalue
of $\Hlim$ of finite multiplicity,
then $\lambda$ is also
an accumulation point of
$\sigma(\Hlim)$; in particular,
$\lambda$ is not isolated in
$\sigma(\Hlim)$.
\end{proposition}

\begin{proof}
Suppose $\lambda\in(0,1)$
is an isolated eigenvalue of
finite multiplicity $m$,
and let $P_\lambda$ be the
corresponding spectral projection.
By the resolvent convergence
(Corollary~\ref{cor:resolvent},
proven below without this
proposition), for $z$ near
$\lambda$ with
$z\notin\sigma(\Hlim)$,
$(\Hc-z)^{-1}\to(\Hlim-z)^{-1}$
strongly.
For each $c$, the eigenvalues
$\lambda_n(c)$ of $\Hc$
are dense in $[0,1]$ by
monotonicity \eqref{IX4}
and the spectral theorem;
in particular, for every
$\varepsilon>0$ there exist
infinitely many $c$ such that
$\sigma(\Hc)\cap
(\lambda-\varepsilon,
\lambda+\varepsilon)$
contains at least $m+1$
eigenvalues.
Strong resolvent convergence
then forces at least $m+1$
spectral points of $\Hlim$
in $(\lambda-\varepsilon,
\lambda+\varepsilon)$
(\cite{ReedSimon1975}, Thm.~VIII.24),
contradicting $\lambda$ isolated
of multiplicity $m$.
\end{proof}

\begin{remark}
Proposition~\ref{prop:spec-structure}
shows that the point spectrum
of $\Hlim$ in $(0,1)$, if non-empty,
cannot consist of isolated
eigenvalues: the spectrum has
no ``gaps'' of finite multiplicity.
This is consistent with
$\sigma(\Hlim)$ being a
closed subinterval or all of $[0,1]$.
\end{remark}

\begin{corollary}[Strong resolvent
  convergence, unconditional]
\label{cor:resolvent}
For every $z\in\mathbb{C}
\setminus[0,1]$,
\[
  (\Hc-z)^{-1}
  \;\xrightarrow{\mathrm{s}}\;
  (\Hlim-z)^{-1}
  \quad\text{as }c\to\infty.
\]
\end{corollary}

\begin{proof}
By Theorem~\ref{thm:mosco},
$\qc\to\qlim$ in Mosco sense.
By \cite{KuwaeShioya2003},
Thm.~2.4, Mosco convergence of
bounded forms on a fixed Hilbert
space implies strong convergence
of the associated resolvents.
Since $\sigma(\Hlim)\subset[0,1]$
(Corollary~\ref{cor:spec}),
every $z\notin[0,1]$ is in the
resolvent set.
\end{proof}

\begin{remark}[Independence from CCM]
Corollaries~\ref{cor:spec}
and~\ref{cor:resolvent}
and Proposition~\ref{prop:spec-structure}
are unconditional.
No CCM hypothesis, no RH.
\end{remark}

\begin{problem}[Open gap:
  spectral embedding]
\label{gap:spectral}
Non-trivial zeta zeros
$\rho=\tfrac12+it$ ($t\ne 0$)
are complex, whereas
$\sigma(\Hlim)\subset\mathbb{R}$.
Two sub-gaps remain open:
\begin{enumerate}[\rm(G1)]
\item Why
      $\rho\in
      \sigma_{\mathrm{app}}(\Hlim)$?
      (Requires CCM or further
      PSWF spectral theory.)
\item How is $\rho$ identified
      with a real spectral
      parameter?
      (CCM uses
      $\operatorname{Im}(\rho)$
      as eigenvalue of a real
      self-adjoint operator;
      external to this paper.)
\end{enumerate}
\end{problem}

\begin{corollary}[Conditional RH]
\label{cor:rh}
This paper does not provide
new evidence for the
Riemann Hypothesis.
It provides an operator-theoretic
setting---existence and
self-adjointness of $\Hlim$,
Mosco convergence of $(\qc)$,
resolvent convergence---that
is a necessary ingredient of
spectral approaches to $\zeta(s)$.

Conditionally on
Hypothesis~CCM
(\cite{CCM2025}, Thm.~1.3)
and the resolution of both
gaps in
Open Gap~\ref{gap:spectral},
every non-trivial zero $\rho$
of $\zeta(s)$ satisfies
$\operatorname{Re}(\rho)=\tfrac12$.
\end{corollary}

\begin{remark}[Unconditional results]
Self-adjointness of $\Hlim$
with $\|\Hlim\|\le 1$;
$\sigma(\Hlim)\subset[0,1]$;
no isolated eigenvalues
of finite multiplicity in $(0,1)$;
$T_{\qlim}=\Hlim$;
strong resolvent convergence;
$\qc\xrightarrow{\mathrm{Mosco}}
\qlim$ at rate $O(c^{-1/4})$.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[CCM]
\label{prob:ccm}
Prove Hypothesis~CCM
independently via PSWF machinery.
\end{problem}

\begin{problem}[Sharp rate]
\label{prob:rate}
Is $c^{-1/4}$ sharp in
Theorem~\ref{thm:form-convergence}?
\end{problem}

\begin{problem}[Quantitative resolvent]
\label{prob:resolvent}
Obtain a rate for the
resolvent convergence in
Corollary~\ref{cor:resolvent}:
$\|(\Hc-z)^{-1}-(\Hlim-z)^{-1}\|
\le F(c,z)\to 0$.
\end{problem}

\begin{problem}[Spectrum of $\Hlim$]
\label{prob:spectrum}
Determine $\sigma(\Hlim)$ precisely:
is it all of $[0,1]$,
a Cantor-type set, or discrete?
Proposition~\ref{prop:spec-structure}
rules out isolated finite-multiplicity
eigenvalues in $(0,1)$.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Prove $\{\Phi_n^{(\infty)}\}$
complete in $L^2(\mathbb{R})$.
Blocks no result of this paper.
\end{problem}
