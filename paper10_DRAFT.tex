\begin{abstract}
The prolate spheroidal wave function
(PSWF) concentration operators
$\Hc$ form a uniformly bounded,
monotone family of self-adjoint
integral operators on $L^2(\mathbb{R})$.
We prove that the strong operator
limit $\Hlim:=\lim_{c\to\infty}\Hc$
exists on all of $L^2(\mathbb{R})$,
is bounded and symmetric
(hence self-adjoint by a standard
argument), and that the associated
quadratic forms converge in the
sense of Mosco at rate $O(c^{-1/4})$.
The Friedrichs representation
$T_{\qlim}=\Hlim$ is derived
independently as an alternative
via closed quadratic forms.
As an unconditional consequence,
the resolvents
$(\Hc-z)^{-1}$ converge strongly
to $(\Hlim-z)^{-1}$ for every
$z\notin[0,1]$.

No eigenvectors, Parseval identity,
or completeness hypothesis is used.
Conditionally on the spectral
triple construction of
Connes--Consani--Moscovici
\cite{CCM2025} and two explicitly
stated open gaps, every non-trivial
zero of $\zeta(s)$ lies on the
critical line; no unconditional
statement about zeta zeros is
claimed.
\end{abstract}

\section*{Motivation and Context}

The prolate spheroidal wave functions
(PSWFs) arise as the eigenfunctions
of the time-frequency concentration
operator: given a bandwidth $c>0$,
the operator $\Hc$ measures how well
a signal is simultaneously
concentrated in time $[-1,1]$ and
frequency $[-c,c]$
(\cite{Slepian1961}).
As $c\to\infty$, perfect concentration
becomes possible for more and more
functions, and the limit operator
$\Hlim$ captures this asymptotic
regime.

Understanding $\Hlim$ as a
self-adjoint operator is of
independent interest in
time-frequency analysis,
approximation theory, and
signal processing:
the spectral theory of $\Hlim$
governs the limiting behaviour of
PSWF expansions and the stability
of bandlimited approximations at
large bandwidth.
The Mosco convergence of the
associated forms at rate
$O(c^{-1/4})$ gives a quantitative
version of this stability,
and the resolvent convergence
(Corollary~\ref{cor:resolvent})
shows that the spectral structure
of $\Hc$ is inherited by $\Hlim$
in the strong sense.

The connection to the Riemann zeta
function is conditional and
programmatic
(Section~\ref{sec:consequences});
the operator-theoretic results of
Sections~\ref{sec:setup}--\ref{sec:friedrichs}
are unconditional and stand
independently.

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from
Papers~V--IX.

\begin{conv}[Ambient space]
\label{conv:tilde}
All operators act on
$L^2(\mathbb{R})$.
The PSWF operator
$\mathcal{H}_c^{[0,1]}$ on
$L^2([0,1])$ (\cite{Slepian1961})
is extended by
$\Hc:=P^*\mathcal{H}_c^{[0,1]}P$,
where $P$ is restriction and
$P^*$ is extension by zero.
$\Hc\in\mathcal{B}(L^2(\mathbb{R}))$
and $0\le\Hc\le I$.
\end{conv}

\begin{enumerate}[(IX.1)]

\item\label{IX3}
      $0\le\ip{\Hc f}{f}\le\|f\|^2$
      for all $f$, all $c$.

\item\label{IX4}
      \textit{Monotone limit.}
      For every $f\in L^2(\mathbb{R})$,
      the map $c\mapsto\ip{\Hc f}{f}$
      is non-decreasing and bounded
      above by $\|f\|^2$
      (\eqref{IX3}).
      Hence the limit
      \[
        \qlim(f,f)
        :=\lim_{c\to\infty}
        \ip{\Hc f}{f}
        =\sup_{c>0}\ip{\Hc f}{f}
      \]
      exists in $\mathbb{R}$
      for every $f$
      (monotone convergence
      in $\mathbb{R}$;
      Paper~V, Prop.~2.2).
      No operator is presupposed
      here; $\qlim(f,f)$ is defined
      purely from $\Hc$.

\item\label{IX5}
      \textit{Dense convergence
      domain.}
      Let
      \[
        \mathcal{D}
        :=\{f\in L^2(\mathbb{R})
        :\lim_{c\to\infty}\Hc f
        \text{ exists in }
        L^2(\mathbb{R})\}.
      \]
      Paper~IX, Thm.~2.5(a):
      all compactly supported
      $L^2(\mathbb{R})$-functions
      lie in $\mathcal{D}$;
      in particular $\mathcal{D}$
      is non-trivial and dense.
      $\mathcal{D}$ is a linear
      subspace (limits of linear
      maps are linear where
      they exist).
      Define the pre-operator
      $B:\mathcal{D}
      \to L^2(\mathbb{R})$ by
      $Bf:=\lim_{c\to\infty}\Hc f$.
      Paper~VII, Thm.~C:
      $\|Bf\|\le\|f\|$
      for all $f\in\mathcal{D}$.
      \textbf{Note:}
      $\Hlim$ is not yet defined
      here; the symbol $B$ is used
      until Lemma~\ref{lem:hlim-bounded}
      constructs the global extension.

\item\label{IX6}
      $\Hc$ is self-adjoint,
      positive, $\|\Hc\|\le 1$
      (\cite{Slepian1961};
      \cite{ReedSimon1975},
      Thm.~VI.1).

%%% U1: circularity in IX2norm eliminated
\item\label{IX2norm}
      \textit{Operator-norm
      convergence (external input;
      used only in
      Section~\ref{sec:form-convergence}).}
      $\|\Hc-\Hlim\|
      \le C_\kappa c^{-1/4}$
      (operator norm on
      $\mathcal{B}(L^2(\mathbb{R}))$,
      globally on all of
      $L^2(\mathbb{R})$).
      Proven in Paper~IX, Thm.~3.1,
      via Landau--Pollak asymptotics
      for PSWF eigenvalue decay
      (\cite{Slepian1961};
      Paper~VIII, Prop.~4.2);
      holds for $c\ge c_0(\kappa)$.
      The estimate is proved in
      Paper~IX with respect to the
      strong-limit operator
      constructed via a uniform-
      boundedness/density argument
      (Lemma~\ref{lem:sc}),
      and does not presuppose the
      existence of $\Hlim$ as an
      a priori object.
      In particular, the proof in
      Paper~IX establishes the
      estimate for the same operator
      that is defined in
      Lemma~\ref{lem:hlim-bounded},
      so no circularity arises.
      This estimate is imported
      as an external input;
      no part of its proof is
      reproduced here.

\end{enumerate}

\begin{lemma}[Stable core;
  density-extension]
\label{lem:sc}
This is a standard result
(\cite{Rudin1991}, Thm.~5.4;
\cite{Conway1990}, I.4;
cf.\ also \cite{ReedSimon1975},
Thm.~III.5);
proof included for completeness.

Let $(A_c)\subset
\mathcal{B}(\mathcal{H})$
with $M:=\sup_c\|A_c\|<\infty$,
and let $\mathcal{D}\subset
\mathcal{H}$ be a dense linear
subspace.
Suppose $A_c f\to Bf$ for all
$f\in\mathcal{D}$, where
$B:\mathcal{D}\to\mathcal{H}$
is linear.
Then:
\begin{enumerate}[\rm(a)]
\item\label{sc:bdd}
      $\|Bf\|\le M\|f\|$
      for all $f\in\mathcal{D}$.
\item\label{sc:ext}
      For every $f\in\mathcal{H}$,
      the net $(A_c f)_c$ is
      Cauchy in $\mathcal{H}$
      and hence convergent.
      Denoting $Af:=\lim_c A_c f$,
      we obtain
      $A\in\mathcal{B}(\mathcal{H})$,
      $\|A\|\le M$,
      and $A|_{\mathcal{D}}=B$.
\item\label{sc:strong}
      $A_c\to A$ strongly.
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a).}
$\|Bf\|=\lim_c\|A_c f\|\le M\|f\|$.

\textit{(b).}
For $f\in\mathcal{H}$ and
$\varepsilon>0$,
choose $g\in\mathcal{D}$ with
$\|f-g\|<\varepsilon$, then
\[
  \|A_c f-A_{c'}f\|
  \le \|A_c(f-g)\|
  +\|A_c g-A_{c'}g\|
  +\|A_{c'}(f-g)\|
  \le 2M\varepsilon
  +\|A_c g-A_{c'}g\|.
\]
Since $A_c g\to Bg$,
the right-hand side $\to 2M\varepsilon$
as $c,c'\to\infty$.
As $\varepsilon>0$ is arbitrary,
$(A_c f)_c$ is a Cauchy net
in $\mathcal{H}$ for every fixed
$f\in\mathcal{H}$;
since $\mathcal{H}$ is complete,
$Af:=\lim_c A_c f$ exists.
$\|Af\|\le M\|f\|$ and
$A|_{\mathcal{D}}=B$ follow immediately,
so $A\in\mathcal{B}(\mathcal{H})$
with $\|A\|\le M$.

\textit{(c).}
$\limsup_c\|A_c f-Af\|
\le 2M\varepsilon$;
$\varepsilon$ arbitrary.
\end{proof}

\begin{remark}[No extra completeness
  argument]
\label{rem:sc-complete}
The only completeness property
used above is that of the Hilbert
space $\mathcal{H}$ itself
(Cauchy nets converge).
No completeness of an
eigenvector system, no Parseval
identity, and no basis argument
is involved.
\end{remark}

\begin{lemma}[Existence and
  boundedness of $\Hlim$]
\label{lem:hlim-bounded}
The pre-operator $B$ of
\eqref{IX5} extends to a unique
bounded operator
$\Hlim\in\mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hlim\|\le 1$
and $\Hc\to\Hlim$ strongly.
\end{lemma}

\begin{proof}
Apply Lemma~\ref{lem:sc} with
$A_c:=\Hc$,
$\mathcal{D}$ and $B$ as in
\eqref{IX5}, $M:=1$.
$\sup_c\|\Hc\|\le 1$ by
\eqref{IX6};
$\mathcal{D}$ is dense by
\eqref{IX5};
$\Hc f\to Bf$ for all
$f\in\mathcal{D}$ by definition.
Part~(a) gives $\|Bf\|\le\|f\|$
(consistent with Paper~VII, Thm.~C).
Parts~(b,c) yield the limit
$Af:=\lim_c\Hc f$ in $L^2$
for all $f$,
$A\in\mathcal{B}(L^2)$,
$\|A\|\le 1$,
$\Hc\to A$ strongly.
We define $\Hlim:=A$;
from this point on, $\Hlim$
denotes this globally defined
bounded operator.
%%% U5: uniqueness sentence
By uniqueness of limits in the
strong operator topology,
$\Hlim$ is uniquely determined
by this construction.
\end{proof}

\begin{corollary}[Form equals operator]
\label{cor:qlim-op}
For every $f\in L^2(\mathbb{R})$,
\[
  \qlim(f,f)
  = \ip{\Hlim f}{f}.
\]
\end{corollary}
\begin{proof}
By \eqref{IX4},
$\qlim(f,f)=\lim_c\ip{\Hc f}{f}$.
By Lemma~\ref{lem:hlim-bounded},
$\Hc f\to\Hlim f$ strongly;
by continuity of the inner product,
$\ip{\Hc f}{f}\to\ip{\Hlim f}{f}$.
\end{proof}

\begin{remark}[Logical order]
\label{rem:order}
The symbol $\Hlim$ is not used
before Lemma~\ref{lem:hlim-bounded}.
In particular, \eqref{IX4} and
\eqref{IX5} do not reference
$\Hlim$; the equality
$\qlim(f,f)=\ip{\Hlim f}{f}$
follows as
Corollary~\ref{cor:qlim-op}.
Lemma~\ref{lem:hlim-bounded}
depends only on
\eqref{IX3}, \eqref{IX5},
\eqref{IX6}, and
Lemma~\ref{lem:sc}.
\eqref{IX2norm} is used only
in Section~\ref{sec:form-convergence}.
\end{remark}

\section{Self-Adjointness of $\Hlim$}
\label{sec:sa}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[BSA principle]
\label{cor:bsa}
Let $T\in\mathcal{B}(\mathcal{H})$
be symmetric
($\ip{Tf}{g}=\ip{f}{Tg}$
for all $f,g\in\mathcal{H}$).
Then $T$ is self-adjoint:
$T=T^*$.
\end{corollary}
\begin{proof}
Since $T$ is everywhere defined
and bounded, $\Dom(T^*)=\mathcal{H}$
and $T^*=T$
(\cite{ReedSimon1975}, Thm.~III.6;
\cite{Rudin1991}, Thm.~12.11).
\end{proof}

\begin{theorem}[Self-adjointness
  of $\Hlim$]
\label{thm:sa}
$\Hlim\in\mathcal{B}(L^2(\mathbb{R}))$
is self-adjoint and positive,
with $\sigma(\Hlim)\subset[0,1]$.
\end{theorem}

\begin{proof}
\textbf{Step~1: Boundedness.}
$\Hlim\in\mathcal{B}(L^2)$,
$\|\Hlim\|\le 1$
(Lemma~\ref{lem:hlim-bounded}).

\textbf{Step~2: Symmetry.}
For all $f,g\in L^2(\mathbb{R})$,
using $\Hc f\to\Hlim f$ strongly
(Lemma~\ref{lem:hlim-bounded})
and joint continuity of
$\ip{\cdot}{\cdot}$:
\[
  \ip{\Hlim f}{g}
  = \lim_c\ip{\Hc f}{g}
  = \lim_c\ip{f}{\Hc g}
  = \ip{f}{\Hlim g}.
\]
The middle equality uses
symmetry of each $\Hc$
(\eqref{IX6}).

\textbf{Step~3: Self-adjointness.}
$\Hlim$ is bounded, symmetric,
and defined on all of $L^2(\mathbb{R})$;
Corollary~\ref{cor:bsa} gives
$\Hlim=\Hlim^*$.

\textbf{Step~4: Spectrum.}
(i) $\Hlim=\Hlim^*\Rightarrow
\sigma(\Hlim)\subset\mathbb{R}$.
(ii) $\|\Hlim\|\le 1\Rightarrow
\sigma(\Hlim)\subset[-1,1]$
(spectral radius $\le$ norm).
(iii) Positivity is preserved
under strong limits of uniformly
bounded positive operators:
for all $f\in L^2(\mathbb{R})$,
$\ip{\Hc f}{f}\ge 0$ for every $c$
(\eqref{IX6}), and
$\ip{\Hc f}{f}\to\ip{\Hlim f}{f}$
(Lemma~\ref{lem:hlim-bounded}),
hence $\ip{\Hlim f}{f}\ge 0$.
By Corollary~\ref{cor:qlim-op}
this equals $\qlim(f,f)\ge 0$
(\eqref{IX3}), so
$\Hlim\ge 0$ and
$\sigma(\Hlim)\subset[0,1]$.
\end{proof}

\begin{remark}[R\^ole of Friedrichs]
Since $\Hlim\in\mathcal{B}(L^2)$
is already self-adjoint,
the Friedrichs construction
(Section~\ref{sec:friedrichs})
is not needed for self-adjointness.
It is retained to provide the
form identification
$T_{\qlim}=\Hlim$
and as an independent derivation
via Mosco $\Rightarrow$ closed
form $\Rightarrow$ unique
self-adjoint operator.
\end{remark}

\section{Limit Form}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
For $f\in L^2(\mathbb{R})$,
the limit
\[
  \qlim(f,f)
  :=\lim_{c\to\infty}\ip{\Hc f}{f}
\]
exists by \eqref{IX4}
(monotone non-decreasing,
bounded above).
Extend by polarisation to obtain
a Hermitian form
$\qlim:\Dom(\qlim)\times\Dom(\qlim)
\to\mathbb{C}$
with $\Dom(\qlim)=L^2(\mathbb{R})$.
\end{definition}

\begin{remark}
Definition~\ref{def:qlim} is
phrased entirely in terms of $\Hc$;
no reference to $\Hlim$ is made.
The equality $\qlim(f,f)=
\ip{\Hlim f}{f}$ holds
by Corollary~\ref{cor:qlim-op}.
\end{remark}

\begin{theorem}[Existence]
\label{thm:qlim-exists}
$\qlim\ge 0$ is a bounded
Hermitian form on $L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Non-negativity: \eqref{IX3}.
Boundedness:
$\qlim(f,f)\le\|f\|^2$
by \eqref{IX3}.
\end{proof}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence]
\label{def:mosco}
$(\qc)$ converge to $\qlim$
in the sense of Mosco
\cite{Mosco1969}:
\begin{enumerate}[\rm(M1)]
\item For every $c_n\to\infty$
      and $f_n\rightharpoonup f$:
      $\qlim(f,f)
      \le\liminf_{n}q_{c_n}(f_n,f_n)$.
\item For every $f$ there exist
      $c_n\to\infty$, $f_n\to f$
      strongly with
      $q_{c_n}(f_n,f_n)
      \to\qlim(f,f)$.
\end{enumerate}
(Standard convention: $c_n$ and
$f_n$ share index $n$
\cite{Mosco1969}.)
\end{definition}

%%%─── LEMMA HS  %%% U2: compactness chain written out step by step
\begin{lemma}[HS and weak continuity]
\label{lem:hs}
For each fixed $c_0>0$:
\begin{enumerate}[\rm(a)]
\item $\widetilde{\mathcal{H}}_{c_0}
      \in\mathcal{B}_2
      (L^2(\mathbb{R}))$ (compact).
\item $f_n\rightharpoonup f$
      $\Rightarrow$
      $\ip{\widetilde{\mathcal{H}}_{c_0}
        f_n}{f_n}
      \to
      \ip{\widetilde{\mathcal{H}}_{c_0}
        f}{f}$.
\end{enumerate}
\end{lemma}
\begin{proof}
\textit{(a).}
$\|K_{c_0}\|_{\infty}\le c_0/\pi$
$\Rightarrow$ HS
$\Rightarrow$ compact
(\cite{ReedSimon1975}, VI.22).

\textit{(b).}
Write the difference as $I+II$:
$II:=\ip{\widetilde{\mathcal{H}}_{c_0}f}{f_n-f}\to 0$
by weak convergence.
$I:=\ip{\widetilde{\mathcal{H}}_{c_0}(f_n-f)}{f_n}$:
since $f_n\rightharpoonup f$,
we have $f_n - f \rightharpoonup 0$.
Since $\widetilde{\mathcal{H}}_{c_0}$
is compact and
$f_n - f \rightharpoonup 0$,
the sequence
$\widetilde{\mathcal{H}}_{c_0}(f_n-f)$
is relatively compact in
$L^2(\mathbb{R})$.
Moreover, every weakly convergent
subsequence of
$\widetilde{\mathcal{H}}_{c_0}(f_n-f)$
has weak limit $0$
(since $\widetilde{\mathcal{H}}_{c_0}$
is bounded and
$f_n - f \rightharpoonup 0$),
hence every norm-convergent
subsequence converges strongly
to $0$.
Since the sequence is relatively
compact and every accumulation
point equals $0$, the whole
sequence converges strongly to $0$
(\cite{ReedSimon1975}, VI.11).
Moreover, $\sup_n\|f_n\|<\infty$
by weak convergence
(Banach--Steinhaus;
\cite{ReedSimon1975}, III.5).
Hence
$|I|\le\sup_n\|f_n\|
\cdot\|\widetilde{\mathcal{H}}_{c_0}(f_n-f)\|
\to 0$.
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc\to\qlim$ in Mosco sense.
\end{theorem}
\begin{proof}
\textit{(M2).}
Take $f_n:=f$, $c_n\to\infty$:
$q_{c_n}(f,f)
=\ip{\widetilde{\mathcal{H}}_{c_n}f}{f}
\to\ip{\Hlim f}{f}
=\qlim(f,f)$
by Corollary~\ref{cor:qlim-op}.

\textit{(M1).}
Let $c_n\to\infty$,
$f_n\rightharpoonup f$.
Fix $c_0>0$.
For each fixed $n$ with $c_n\ge c_0$,
the form monotonicity
(Inherited Result~\eqref{IX4})
states that $q_c(h,h)$ is
non-decreasing in $c$ for every
fixed $h\in L^2(\mathbb{R})$.
Applying this pointwise with
$h:=f_n$ and $c:=c_n\ge c_0$
yields
$q_{c_n}(f_n,f_n)\ge q_{c_0}(f_n,f_n)$.
By Lemma~\ref{lem:hs}(b):
\[
  \liminf_n q_{c_n}(f_n,f_n)
  \ge\lim_n q_{c_0}(f_n,f_n)
  = q_{c_0}(f,f).
\]
This lower bound $q_{c_0}(f,f)$
is independent of $n$,
so taking the supremum over
$c_0>0$ preserves the inequality:
\[
  \liminf_n q_{c_n}(f_n,f_n)
  \ge\sup_{c_0>0}q_{c_0}(f,f).
\]
The function $c_0\mapsto q_{c_0}(f,f)$
is non-decreasing by \eqref{IX4},
so $\sup_{c_0>0}q_{c_0}(f,f)
=\lim_{c_0\to\infty}q_{c_0}(f,f)
=\qlim(f,f)$
by Definition~\ref{def:qlim}.
Hence
$\liminf_n q_{c_n}(f_n,f_n)
\ge\qlim(f,f)$.
\end{proof}

%%% U3: Mosco remark on tools used
\begin{remark}[Tools in (M1)]
\label{rem:mosco-tools}
We emphasise that (M1) uses
only monotonicity of
$c\mapsto q_c(f,f)$ and
compactness of
$\widetilde{\mathcal{H}}_{c_0}$;
no spectral decomposition
of $\Hc$ or $\Hlim$ is involved.
\end{remark}

\begin{remark}[Mosco $\Rightarrow$
  resolvent convergence]
\cite{KuwaeShioya2003}, Thm.~2.4.
See Corollary~\ref{cor:resolvent}.
\end{remark}

%%═══════════════════════════════════════════════════════════════════
\section{Closedness}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
Let $(f_k)$ be
$\|\cdot\|_{\qlim}$-Cauchy;
then $f_k\to f$ in $L^2$
(norm-Cauchy $\Rightarrow$
$L^2$-Cauchy by $\|\qlim\|\le 1$),
hence weakly.
Apply (M1) with constant
$c_n:=c$ and $f_k\rightharpoonup f$:
$\qc(f,f)
\le\liminf_k\qc(f_k,f_k)$.
Send $c\to\infty$ (using
Definition~\ref{def:qlim});
the parallelogram law yields
$\qlim(f_k-f,f_k-f)\to 0$
(\cite{Kato1966}, VI.1.3).
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative rate]
\label{thm:form-convergence}
For all $f\in L^2(\mathbb{R})$,
$c\ge c_0(\kappa)$:
\[
  0\le\qlim(f,f)-\qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\]
\end{theorem}
\begin{proof}
Lower bound: \eqref{IX4}.
Upper bound:
$\qlim(f,f)-\qc(f,f)
=\ip{(\Hlim-\Hc)f}{f}
\le\|\Hlim-\Hc\|\cdot\|f\|^2
\le C_\kappa c^{-1/4}\|f\|^2$
(Corollary~\ref{cor:qlim-op};
\eqref{IX2norm}).
\end{proof}

\section{Friedrichs Extension
  (Alternative Derivation)}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

This section gives an independent
derivation of $T_{\qlim}=\Hlim$
via the Kato--Friedrichs
representation theorem.
Self-adjointness of $\Hlim$
follows already from
Theorem~\ref{thm:sa}; this
section is logically independent.

\begin{theorem}[Kato representation]
\label{thm:kato-rep}
A densely defined, closed,
semibounded form $q$ determines
a unique self-adjoint $T$,
$\Dom(T)\subset\Dom(q)$, with
$q(f,g)=\ip{Tf}{g}$.
(\cite{Kato1966}, VI.2.1;
\cite{ReedSimon1975}, VIII.15.)
\end{theorem}

\begin{theorem}[Form identification]
\label{thm:friedrichs}
$T_{\qlim}=\Hlim$.
\end{theorem}
\begin{proof}
By Theorems~\ref{thm:qlim-exists}
and~\ref{thm:form-closed},
$\qlim$ is a closed semibounded
form on all of $L^2(\mathbb{R})$;
Theorem~\ref{thm:kato-rep}
gives a unique self-adjoint
$T_{\qlim}$.

For every $f,g\in L^2(\mathbb{R})$:
by Lemma~\ref{lem:hlim-bounded}
($\Hc f\to\Hlim f$ strongly)
and joint continuity of
$\ip{\cdot}{\cdot}$:
$\ip{\Hc f}{g}\to\ip{\Hlim f}{g}$.
By Definition~\ref{def:qlim}:
$\ip{\Hc f}{g}=\qc(f,g)
\to\qlim(f,g)$.
Hence $\ip{\Hlim f}{g}=\qlim(f,g)$
for all $g$, so
$f\in\Dom(T_{\qlim})$ and
$T_{\qlim}f=\Hlim f$:
$\Hlim\subset T_{\qlim}$.
Since both operators are bounded
and defined on all of
$L^2(\mathbb{R})$,
$\Hlim\subset T_{\qlim}$
implies $\Hlim=T_{\qlim}$.
%%% U4: coincidence sentence
Thus the operator constructed
via strong limits
(Lemma~\ref{lem:hlim-bounded})
coincides with the one obtained
abstractly from the form
$\qlim$ via the Kato representation
(Theorem~\ref{thm:kato-rep}).
\end{proof}

\begin{remark}[Full audit]
Every step uses exactly one of:
\eqref{IX3}, \eqref{IX4},
\eqref{IX5}, \eqref{IX6},
\eqref{IX2norm},
Lemmas~\ref{lem:sc},
\ref{lem:hlim-bounded},
\ref{lem:hs},
Corollary~\ref{cor:qlim-op},
Kato VI.2.1.
No eigenvector, Parseval,
or completeness is invoked.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Spectral inclusion]
\label{cor:spec}
$\sigma(\Hlim)\subset[0,1]
\subset\mathbb{R}$.
\end{corollary}

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every
$z\in\mathbb{C}\setminus[0,1]$:
\[
  (\Hc-z)^{-1}
  \xrightarrow{\mathrm{s}}
  (\Hlim-z)^{-1}.
\]
\end{corollary}
\begin{proof}
Mosco convergence
(Theorem~\ref{thm:mosco})
$+$ \cite{KuwaeShioya2003},
Thm.~2.4;
$\sigma(\Hlim)\subset[0,1]$
(Corollary~\ref{cor:spec}).
\end{proof}

\begin{remark}[All unconditional]
Corollaries~\ref{cor:spec},
\ref{cor:resolvent}:
no CCM, no RH.
\end{remark}

\begin{problem}[Open gap:
  spectral embedding]
\label{gap:spectral}
Two sub-gaps bridging to RH
remain open:
\begin{enumerate}[\rm(G1)]
\item Why
      $\rho\in
      \sigma_{\mathrm{app}}(\Hlim)$?
\item How is
      $\operatorname{Im}(\rho)$
      identified as a real
      spectral parameter?
      (External to this paper;
      CCM uses a real
      self-adjoint operator.)
\end{enumerate}
\end{problem}

\begin{corollary}[Conditional RH]
\label{cor:rh}
Conditionally on
Hypothesis~CCM
(\cite{CCM2025}, Thm.~1.3)
and the resolution of
Open Gap~\ref{gap:spectral},
every non-trivial zero $\rho$
of $\zeta(s)$ satisfies
$\operatorname{Re}(\rho)=\tfrac12$.

\textit{This paper makes no
unconditional claim about
zeta zeros.}
\end{corollary}

\begin{remark}[Unconditional
  results of this paper]
\begin{enumerate}[--]
\item $\Hlim\in\mathcal{B}(L^2)$,
      $\|\Hlim\|\le 1$,
      $\Hlim=\Hlim^*\ge 0$.
\item $\sigma(\Hlim)\subset[0,1]$.
\item $T_{\qlim}=\Hlim$.
\item $(\Hc-z)^{-1}
      \xrightarrow{s}(\Hlim-z)^{-1}$
      for $z\notin[0,1]$.
\item $\qc\xrightarrow{\mathrm{Mosco}}
      \qlim$ at rate $O(c^{-1/4})$.
\end{enumerate}
\end{remark}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[CCM]
\label{prob:ccm}
Prove Hypothesis~CCM
independently via PSWF machinery.
\end{problem}

\begin{problem}[Sharp rate]
\label{prob:rate}
Is $c^{-1/4}$ sharp in
Theorem~\ref{thm:form-convergence}?
\end{problem}

\begin{problem}[Quantitative resolvent]
\label{prob:resolvent}
$\|(\Hc-z)^{-1}-(\Hlim-z)^{-1}\|
\le F(c,z)\to 0$: find $F$.
\end{problem}

\begin{problem}[Spectrum of $\Hlim$]
\label{prob:spectrum}
Is $\sigma(\Hlim)=[0,1]$,
a Cantor set, or discrete?
\end{problem}

\begin{problem}[Spectral structure
  of $\Hlim$]
\label{prob:spec-structure}
Does $\sigma(\Hlim)$ have isolated
points in $(0,1)$?
The Landau--Pollak eigenvalue
distribution (\cite{Slepian1961})
suggests a limiting density on
$[0,1]$, but deriving structural
properties of $\sigma(\Hlim)$ from
this via strong resolvent convergence
(Corollary~\ref{cor:resolvent})
remains open: strong (as opposed to
norm) resolvent convergence does
not directly control multiplicity
or isolation of spectral points
in the limit operator.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Prove $\{\Phi_n^{(\infty)}\}$
complete in $L^2(\mathbb{R})$.
This is not used anywhere above.
\end{problem}
