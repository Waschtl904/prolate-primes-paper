\begin{abstract}
We study the concentration operators
$\Hc = P^*\mathcal{H}_c^{[0,1]}P$,
where $\mathcal{H}_c^{[0,1]}$ is the
classical Slepian--Pollak
time-band-limiting operator on
$L^2([0,1])$.
The family $\{\Hc\}_{c>0}$ consists
of uniformly bounded,
monotonically increasing,
compact self-adjoint operators on
$L^2(\mathbb{R})$.

The foundational identity
(Lemma~\ref{lem:ftb})
\[
  \ip{\Hc f}{f}
  = \|\Bc(P^*Pf)\|^2
\]
reduces the entire operator family to
a Fourier band projection conjugated
by time-restriction.
All results in this paper are
consequences of this single identity
and the inclusion
$[-c,c]\subset[-c',c']$ for $c\le c'$;
we call it the \emph{coordinatisation
lemma} of the theory.

\textbf{Dual convergence structure.}
The identity gives rise to two
independent, parallel convergence threads:
\begin{enumerate}[(A)]
\item \textbf{Operator thread
      (SOT convergence):}
      Uniform boundedness and a dense
      subset of $L^2(\mathbb{R})$
      on which $\Hc f$ converges
      (Lemma~\ref{lem:dense})
      yield, via the SOT extension
      principle (Lemma~\ref{lem:sc}),
      a unique strong limit
      $\Hc\xrightarrow{\mathrm{SOT}}
      \Hstrong\in\mathcal{B}(L^2)$.
\item \textbf{Form thread
      (Mosco convergence):}
      Operator monotonicity and
      Hilbert--Schmidt compactness
      of each $\Hc[c_0]$
      (Lemma~\ref{lem:hs-compact})
      yield, via weak-to-strong
      compactness (Lemma~\ref{lem:wc}),
      Mosco convergence
      $\qc\xrightarrow{\mathrm{M}}
      \qlim$.
\end{enumerate}
The two threads are logically
orthogonal: thread~(A) uses
uniform boundedness and Cauchy nets;
thread~(B) uses monotonicity and
compactness.
They are unified by the identification
$\qlim(f,f)=\ip{\Hstrong f}{f}$,
giving
$T_{\qlim}=\Hstrong$
(Theorem~\ref{thm:friedrichs})
and strong resolvent convergence
(Corollary~\ref{cor:resolvent}).

\textbf{Bridge theorem.}
For a spectrally constructed operator
$\Hspectral$ (Paper~IX),
$\Hstrong=\overline{\Hspectral}$
(Theorem~\ref{thm:bridge}).
The proof is a
\emph{categorical identity argument}:
both sides are SOT limits of the same
family $(\Hc)$;
SOT limits of bounded operators
are unique by pointwise-convergence
on all vectors.
No eigenfunction analysis or
dense-set comparison is needed.

\textbf{Scope.}
This paper is pure operator-limit
theory.
It proves no properties of the
limiting eigenfunctions
$\{\Phi_n^{(\infty)}\}$,
no spectral decomposition of $\Hlim$,
and makes no unconditional
connection to zeros of $\zeta(s)$.
The speculative connection to the
Riemann hypothesis
(Remark~\ref{rem:rh})
is clearly labelled as programmatic.
\end{abstract}

\section{Setup and Foundational Lemmas}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

Let $\Fc$ denote the Fourier transform
$(\Fc f)(\xi)=
\int_{-\infty}^\infty f(x)
e^{-i\xi x}\,dx$,
with Plancherel isometry
$\|\Fc f\|_{L^2}=\sqrt{2\pi}\|f\|_{L^2}$
(\cite{ReedSimon1975}, Ch.~IX).

\begin{conv}[Operators]
\label{conv:ops}
\begin{alignat}{2}
  \Bc&:=\Fc^{-1}\chi_{[-c,c]}\Fc
  &&\;(\Bc^*=\Bc=\Bc^2),\\
  P&:L^2(\mathbb{R})\to L^2([0,1]),
  &&\;\text{restriction; }\|P\|=1,\\
  P^*&:L^2([0,1])\to L^2(\mathbb{R}),
  &&\;\text{extension by zero;
  }\|P^*\|=1,\\
  \mathcal{H}_c^{[0,1]}
  &:L^2([0,1])\to L^2([0,1]),
  &&\;K_c(x,y)
  =\tfrac{\sin c(x-y)}{\pi(x-y)},\\
  \Hc&:=P^*\mathcal{H}_c^{[0,1]}P
  &&\;\in\mathcal{B}(L^2(\mathbb{R})).
\end{alignat}
$K_c(x,x):=c/\pi$.
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
$\Hc$ is self-adjoint,
$0\le\Hc\le I$,
$\|\Hc\|\le 1$
(\cite{Slepian1961};
\cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[Scope of foundational
  thread]
\label{rem:scope}
Sections~\ref{sec:setup}--%
\ref{sec:friedrichs} use only
\eqref{F1} and
Lemmas~\ref{lem:ftb}--\ref{lem:dense}.
No results from Papers~VIII or~IX
appear here.
\end{remark}

%── lem:ftb
\begin{lemma}[Fourier--time--band
  identity: coordinatisation lemma]
\label{lem:ftb}
For all $c>0$, $f\in L^2(\mathbb{R})$:
\begin{align}
  \label{eq:ftb}
  \ip{\Hc f}{f}
  &=\|\Bc(P^*Pf)\|^2,\\
  \label{eq:ftb-op}
  \mathcal{H}_c^{[0,1]}
  &=P\,\Bc\,P^*.
\end{align}
This is the structural origin of all
subsequent results; every property
of $\Hc$ derived below follows
from \eqref{eq:ftb} and the inclusion
$[-c,c]\subset[-c',c']$.
\end{lemma}

\begin{proof}
\textit{Step~1: Convolution identity.}
$K_c(u)=\frac{\sin cu}{\pi u}
\leftrightarrow
(\Fc K_c)(\xi)=\chi_{[-c,c]}(\xi)$
(\cite{Slepian1961}, Eq.~(3);
\cite{ReedSimon1975}, Ex.~IX.7).
Convolution theorem
(\cite{ReedSimon1975}, Thm.~IX.2):
\begin{equation}\label{eq:conv-bc}
  K_c*h=\Bc h
  \;\forall h\in L^2(\mathbb{R}).
\end{equation}

\textit{Step~2: Restriction.}
For $g\in L^2([0,1])$,
$\tilde g:=P^*g$.
Since $\tilde g=0$ a.e.\ outside
$[0,1]$:
$(K_c*\tilde g)(x)
=\int_0^1 K_c(x-y)g(y)\,dy
=(\mathcal{H}_c^{[0,1]}g)(x)$.
With \eqref{eq:conv-bc}:
$\mathcal{H}_c^{[0,1]}g=P\Bc P^*g$,
giving \eqref{eq:ftb-op}.

\textit{Step~3.}
For $g=Pf$:
$\ip{\Hc f}{f}
=\ip{P\Bc P^*Pf}{Pf}
=\|\Bc(P^*Pf)\|^2$.
\end{proof}

%── lem:monotone
\begin{lemma}[Operator monotonicity
  --- Thread~B input]
\label{lem:monotone}
$0<c\le c'\Rightarrow
\ip{\Hc f}{f}\le
\ip{\mathcal{H}_{c'}f}{f}$.
\end{lemma}
\begin{proof}
$[-c,c]\subset[-c',c']\Rightarrow
\Bc\le B_{c'}$
(\cite{ReedSimon1975}, Ch.~VIII.1);
apply \eqref{eq:ftb}.
\end{proof}

%── lem:hs-compact
\begin{lemma}[Hilbert--Schmidt ---
  Thread~B input]
\label{lem:hs-compact}
For fixed $c_0>0$,
$\Hc[c_0]$ is Hilbert--Schmidt,
hence compact.
\end{lemma}
\begin{proof}
$\|\mathcal{H}_{c_0}^{[0,1]}
\|_{\mathrm{HS}}^2
\le\int_{-\infty}^\infty
|K_{c_0}(u)|^2\,du
=c_0/\pi$
(Fubini $+$ Parseval);
compactness:
\cite{ReedSimon1975}, Thm.~VI.22.
\end{proof}

%── lem:dense
\begin{lemma}[Dense domain ---
  Thread~A input]
\label{lem:dense}
Define
$\mathcal{D}:=\{f\in L^2(\mathbb{R}):
\Hc f\to\text{ in }L^2
\text{ as }c\to\infty\}$.
Then $C_c^\infty\subset\mathcal{D}$,
so $\mathcal{D}$ is dense.
\end{lemma}
\begin{proof}
For $f\in C_c^\infty$,
$\tilde g:=P^*Pf$:
$\|\Hc f - P^*P\tilde g\|
\le\|\Bc\tilde g-\tilde g\|\to 0$
(\cite{ReedSimon1975}, Thm.~IX.14;
stability under $P^*P$).
\end{proof}

\begin{remark}[Logical status]
\label{rem:dense-trivial}
Lemma~\ref{lem:dense} is
pre-input to Thread~A only.
Once $\Hc\xrightarrow{\mathrm{SOT}}
\Hstrong$ is established,
$\mathcal{D}=L^2$ automatically;
the density claim is then
superseded.
\end{remark}

\section{Thread~A: Strong Operator
  Limit}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Thread~A uses]
\label{rem:threadA-uses}
Inputs: \eqref{F1};
Lem.~\ref{lem:dense}.
Tool: Cauchy nets in Banach space.
Output:
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$.
\end{remark}

\begin{lemma}[SOT extension
  principle]
\label{lem:sc}
Let $(A_c)_{c>0}$ be a directed net
(ordered by $c\to\infty$) in
$\mathcal{B}(\mathcal{H})$
with $M:=\sup_c\|A_c\|<\infty$,
and $\mathcal{D}\subset\mathcal{H}$
dense with $\lim_c A_c g$ existing
for every $g\in\mathcal{D}$.
Then $\exists!\,A\in\mathcal{B}(\mathcal{H})$
with $\|A\|\le M$ and
$A_c\xrightarrow{\mathrm{SOT}}A$.
\end{lemma}

\begin{proof}
Fix $f$, $\varepsilon>0$;
choose $g\in\mathcal{D}$ with
$\|f-g\|<\varepsilon/(3M)$.
For $c,c'$ above $c_0(g,\varepsilon)$
in the directed set $(0,\infty)$:
\[
  \|A_c f-A_{c'}f\|
  \le 2M\|f-g\|+\|A_c g-A_{c'}g\|
  <\varepsilon.
\]
$(A_c f)$ is Cauchy in the complete
space $\mathcal{H}$;
$Af:=\lim_c A_c f$ exists.
Boundedness, linearity, uniqueness:
\cite{ReedSimon1975}, Thm.~I.26.
\end{proof}

\begin{lemma}[Thread~A conclusion]
\label{lem:hlim-bounded}
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$
for a unique
$\Hstrong\in\mathcal{B}(L^2)$,
$\|\Hstrong\|\le 1$.
\end{lemma}
\begin{proof}
Lem.~\ref{lem:sc};
$M=1$ from \eqref{F1};
$\mathcal{D}$ from Lem.~\ref{lem:dense}.
\end{proof}

\begin{corollary}
\label{cor:qlim-op}
$\qlim(f,f)=\ip{\Hstrong f}{f}$.
\end{corollary}

\begin{corollary}
\label{cor:sa-strong}
$\Hstrong=\Hstrong^*\ge 0$;
$\sigma(\Hstrong)\subset[0,1]$.
\end{corollary}

\section{Thread~B: Mosco Convergence
  of Forms}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Thread~B uses]
\label{rem:threadB-uses}
Inputs: Lem.~\ref{lem:monotone};
Lem.~\ref{lem:hs-compact}.
Tool: compactness $+$ weak convergence.
Output:
$\qc\xrightarrow{\mathrm{M}}\qlim$.
Thread~B is \emph{logically orthogonal}
to Thread~A: it uses neither
uniform boundedness in the
Cauchy-net sense nor
$\mathcal{D}$.
\end{remark}

\begin{definition}[Limit form]
\label{def:qlim}
$\qlim(f,f):=
\sup_{c>0}\ip{\Hc f}{f}$;
extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc)\xrightarrow{\mathrm{M}}\qlim$
iff:
\begin{enumerate}[\rm(M1)]
\item $c_n\to\infty$,
      $f_n\rightharpoonup f\Rightarrow
      \qlim(f,f)\le
      \liminf_n q_{c_n}(f_n,f_n)$.
\item $\forall f$,
      $\exists c_n$, $f_n\to f$:
      $q_{c_n}(f_n,f_n)\to\qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Compact$+$weak
  $\Rightarrow$ form continuity]
\label{lem:wc}
Let $T\in\mathcal{B}(\mathcal{H})$
be compact,
$f_n\rightharpoonup f$.
Then:
(a) $Tf_n\to Tf$ strongly
(\cite{ReedSimon1975}, Thm.~VI.1);
(b) $\ip{Tf_n}{f_n}\to\ip{Tf}{f}$
(\cite{ReedSimon1975}, Lem.~II.3).
\end{lemma}

\begin{theorem}[Thread~B conclusion:
  Mosco convergence]
\label{thm:mosco}
$\qc\xrightarrow{\mathrm{M}}\qlim$.
\end{theorem}

\begin{proof}
\textit{(M2).} Trivial from
Def.~\ref{def:qlim}.

\textit{(M1).}
Let $c_n\to\infty$,
$f_n\rightharpoonup f$.
Fix $c_0>0$; once $c_n\ge c_0$:
$q_{c_n}(f_n,f_n)
\ge q_{c_0}(f_n,f_n)$
(Lem.~\ref{lem:monotone}).
$\Hc[c_0]$ compact
(Lem.~\ref{lem:hs-compact}),
so by Lem.~\ref{lem:wc}(b):
$\liminf_n q_{c_n}(f_n,f_n)
\ge q_{c_0}(f,f)$.
Take $\sup_{c_0}$:
$\liminf_n q_{c_n}(f_n,f_n)
\ge\qlim(f,f)$.
\end{proof}

\section{Identification and Closedness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of
  $\qlim$]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
Bounded symmetric forms are
automatically closed
(\cite{ReedSimon1975}, Thm.~VIII.15);
alternatively, form and $L^2$
norms are equivalent
($\|f\|^2\le\|f\|_{\qlim}^2
\le 2\|f\|^2$).
\end{proof}

\begin{theorem}[Unification of
  threads: Riesz--Kato]
\label{thm:friedrichs}
$T_{\qlim}=\Hstrong$.
\end{theorem}
\begin{proof}
$\ip{\Hstrong f}{g}=\qlim(f,g)$
(Cor.~\ref{cor:qlim-op});
uniqueness of Riesz representative.
\end{proof}

\begin{remark}[Threads united]
Thm.~\ref{thm:friedrichs} is the
unification step:
Thread~A produced $\Hstrong$;
Thread~B produced $\qlim$;
the Riesz identification
$T_{\qlim}=\Hstrong$ shows
they are the same object.
\end{remark}

\section{Quantitative Rate
  (External, Paper~VIII)}
\label{sec:rate}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}
No other section depends on
Thm.~\ref{thm:rate-ext};
this section is standalone.
\end{remark}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
For all $f$ and $c\ge c_0(\kappa)$:
$0\le\qlim(f,f)-\qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{theorem}
\begin{proof}[Proof (external)]
Paper~VIII, Cor.~4.3:
$\|\Hstrong-\Hc\|
\le C_\kappa c^{-1/4}$
(operator-norm level;
strictly stronger than Mosco).
Apply to $\ip{(\Hstrong-\Hc)f}{f}$.
\end{proof}

\begin{remark}[Obstruction]
\label{rem:no-indep-proof}
An internal proof would require
uniform eigenvector asymptotics;
mixing fixed-$c$ eigenbasis with
$c\to\infty$ weights is invalid.
See Problem~\ref{prob:rate}.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the sole import
point for Paper~IX.

\begin{enumerate}[(IX.a)]
\item\label{IXa}
      $\Hspectral f
      :=\sum_n
      \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}}
      \Phi_n^{(\infty)}}$;
      densely defined, symmetric,
      closable
      (Paper~IX, Thms.~2.5,~3.2).
\item\label{IXb}
      $\|\Hspectral f\|^2\le\|f\|^2$
      on $\Dom(\Hspectral)$
      ($\lambda_n^{(\infty)}\in[0,1]$
      + Bessel);
      hence $\overline{\Hspectral}
      \in\mathcal{B}(L^2)$
      (\cite{ReedSimon1975},
      Thm.~I.7).
\item\label{IXc}
      Paper~IX, Rem.~2.6:
      $\Hc\xrightarrow{\mathrm{SOT}}
      \overline{\Hspectral}$.
\end{enumerate}

\begin{theorem}[Bridge theorem:
  categorical SOT identity]
\label{thm:bridge}
$\Hstrong=\overline{\Hspectral}$
in $\mathcal{B}(L^2(\mathbb{R}))$.
\end{theorem}

\begin{proof}
Both $\Hstrong$ (Lem.~\ref{lem:hlim-bounded})
and $\overline{\Hspectral}$ (\eqref{IXc})
are SOT limits of the same family
$(\Hc)_{c>0}$.
SOT limits are unique:
if $A_c f\to Af$ and $A_c f\to Bf$
for all $f\in L^2$,
then $(A-B)f=0$ for all $f$,
so $A=B$
(pointwise convergence; $L^2$
Hausdorff).
Hence $\Hstrong=\overline{\Hspectral}$.
\end{proof}

\begin{remark}[Structure of the proof]
No eigenfunction analysis,
no dense-set comparison,
no spectral identification:
the proof is a pure
\emph{categorical identity argument}
in the SOT topology on
$\mathcal{B}(L^2)$.
The sole import from Paper~IX is
\eqref{IXc}.
\end{remark}

\begin{remark}
$\Hlim:=\Hstrong=\overline{\Hspectral}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For $z\notin[0,1]$:
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$.
\end{corollary}
\begin{proof}
Thm.~\ref{thm:mosco};
$\sigma(\Hlim)\subset[0,1]$;
\cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional
  results summary]
Without any hypothesis:
$\Hlim=\Hlim^*\ge 0$;
$\|\Hlim\|\le 1$;
$T_{\qlim}=\Hlim$;
$\qc\xrightarrow{\mathrm{M}}\qlim$;
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$;
$\Hstrong=\overline{\Hspectral}$.
External (operator-norm, Paper~VIII):
rate $O(c^{-1/4})$.
\end{remark}

\begin{remark}[Speculative framework:
  Riemann zeros]
\label{rem:rh}
This remark is
\emph{speculative and programmatic};
no proved result and no implication
from preceding theorems to
zeta zeros is claimed.

Under two unverified hypotheses:
\begin{enumerate}[(H1)]
\item $\{\Phi_n^{(\infty)}\}$ complete;
      $\Hlim$ essentially
      self-adjoint
      (Paper~IX, Hyp.~4.1).
\item Non-trivial zeros $\rho$ lie in
      $\sigma_{\mathrm{app}}(\Hlim)$
      (\cite{CCM2025}, Thm.~1.3).
\end{enumerate}
(H1)--(H2) would give
$\rho\in\sigma(\Hlim)
\subset\mathbb{R}$,
hence $\mathrm{Re}(\rho)=1/2$.
(H2) has no rigorous operator
identification between $\Hlim$ and
any zeta-spectral object;
treat as motivational context only.
\end{remark}
