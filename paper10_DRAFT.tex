\begin{abstract}
We prove the essential self-adjointness
of the PSWF concentration limit
operator $\Hlim$ via Mosco form
convergence and the Kato--Friedrichs
representation theorem.
All operators are realised on the
single ambient space $L^2(\mathbb{R})$
(Convention~\ref{conv:tilde}).
The limit form
$\qlim(f,f):=\lim_{c\to\infty}
\ip{\Hc f}{f}$ is defined purely from
the operator sequence; no eigenvectors,
Parseval identity, or completeness
hypothesis appear anywhere.

The structural backbone is
Lemma~\ref{lem:sc}
(\emph{Stable Core under Strong
Convergence}), a short standard
argument showing that strong
convergence on a dense set plus a
uniform norm bound implies strong
convergence on all of $L^2(\mathbb{R})$.
This single lemma eliminates all
implicit domain-switching in the
Mosco and core-density arguments.

\textbf{Main results.\ }
(1)~$\qlim$ is a closed semibounded
    form on $L^2(\mathbb{R})$
    (Theorems~\ref{thm:qlim-exists},
    \ref{thm:form-closed}).
(2)~$\qc\to\qlim$ in Mosco sense
    with rate $O(c^{-1/4})$
    (Theorems~\ref{thm:mosco},
    \ref{thm:form-convergence}).
(3)~$T_{\qlim}=\overline{\Hlim}$;
    $\Hlim$ is essentially self-adjoint
    (Theorem~\ref{thm:friedrichs}).
(4)~Conditionally on Hypothesis~CCM
    \cite{CCM2025}, all non-trivial
    zeros $\rho$ of $\zeta(s)$ satisfy
    $\operatorname{Re}(\rho)=\tfrac12$
    (Corollary~\ref{cor:rh}).
\end{abstract}

\section{Setup and Inherited Results}
\label{sec:setup}

We retain all notation from
Papers~V--IX.

\begin{conv}[Ambient space and
  operator extension]
\label{conv:tilde}
All operators in this paper act on the
single Hilbert space $L^2(\mathbb{R})$.
The PSWF concentration operator
$\mathcal{H}_c^{[0,1]}$ is originally
defined on $L^2([0,1])$
(\cite{Slepian1961}).
We extend it to $L^2(\mathbb{R})$ by
\[
  \Hc
  := P^*\,\mathcal{H}_c^{[0,1]}\,P,
\]
where
$P:L^2(\mathbb{R})\to L^2([0,1])$
is the restriction operator and
$P^*:L^2([0,1])\to L^2(\mathbb{R})$
extension by zero.
This preserves all spectral,
symmetry, and norm properties;
in particular
$\Hc\in\mathcal{B}(L^2(\mathbb{R}))$
and $0\le\Hc\le I$.
All forms and all weak/strong
convergence are understood in
$L^2(\mathbb{R})$.
\end{conv}

\begin{enumerate}[(IX.1)]
\item\label{IX1}
      $\Hlim$ is symmetric and densely
      defined on $\Dom(\Hlim)$
      (Paper~IX, Def.~2.3, Thm.~2.5).

\item\label{IX2}
      $\Hlim\in\mathcal{B}(L^2(\mathbb{R}))$
      with $\|\Hlim\|\le 1$, and
      $\|\Hc-\Hlim\|_{L^2\to L^2}
      \le C_\kappa c^{-1/4}$
      (Paper~IX, Thm.~3.1).

      \textit{Proof that
      $\|\Hlim\|\le 1$
      is a theorem.}
      $\|\Hc\|\le 1$ for every $c$
      (\eqref{IX6}).
      Strong convergence \eqref{IX7}
      on the dense set $\Dom(\Hlim)$,
      together with the uniform bound
      $\sup_c\|\Hc\|\le 1$,
      implies by
      Lemma~\ref{lem:sc}(a)
      that $\Hc\to\Hlim$ strongly
      on all of $L^2(\mathbb{R})$.
      Hence
      $\|\Hlim f\|
      =\lim_c\|\Hc f\|
      \le\|f\|$
      for all $f\in L^2(\mathbb{R})$.

\item\label{IX3}
      $0\le\ip{\Hc f}{f}\le\|f\|^2$
      for all $f$, all $c$.

\item\label{IX4}
      $c\mapsto\ip{\Hc f}{f}$
      is monotone non-decreasing
      (Paper~V, Prop.~2.2).

\item\label{IX5}
      $\Dom(\Hlim)$ is dense in
      $L^2(\mathbb{R})$
      (Paper~IX, Thm.~2.5(a)).

\item\label{IX6}
      $\mathcal{H}_c^{[0,1]}$ has real
      symmetric kernel
      $K_c(x,y)=
      \tfrac{\sin c(x-y)}{\pi(x-y)}$
      (\cite{Slepian1961}),
      so it is self-adjoint and
      positive on $L^2([0,1])$.
      For a positive self-adjoint
      operator,
      $\|\mathcal{H}_c^{[0,1]}\|
      =\sup_{\|f\|=1}
      \ip{\mathcal{H}_c^{[0,1]}f}{f}$
      (\cite{ReedSimon1975},
      Thm.~VI.1);
      hence $\|\Hc\|\le 1$
      by \eqref{IX3}.

\item\label{IX7}
      $\|\Hc f-\Hlim f\|\to 0$
      for every $f\in\Dom(\Hlim)$
      (Paper~VII, Thm.~C).
\end{enumerate}

%%─── STABLE CORE LEMMA
\begin{lemma}[Stable core under
  strong convergence]
\label{lem:sc}
Let $(A_c)_{c>0}\subset
\mathcal{B}(\mathcal{H})$ be
self-adjoint with
$\sup_c\|A_c\|\le M<\infty$.
Suppose $A_c f\to Af$ for all
$f\in\mathcal{D}$, where
$\mathcal{D}\subset\mathcal{H}$
is dense, and
$A\in\mathcal{B}(\mathcal{H})$.
Then:
\begin{enumerate}[\rm(a)]
\item\label{sc:strong}
      $A_c\to A$ strongly on all
      of $\mathcal{H}$.
\item\label{sc:approx}
      For every $f\in\mathcal{H}$
      and $\varepsilon>0$ there
      exists $f_\varepsilon
      \in\mathcal{D}$ such that
      $\|f_\varepsilon-f\|<\varepsilon$
      and
      $\sup_c
      \|A_c f_\varepsilon
      -Af\|\to 0$
      as $\varepsilon\to 0$.
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a).}
Let $f\in\mathcal{H}$,
$\varepsilon>0$.
Choose $f_n\in\mathcal{D}$ with
$\|f-f_n\|<\varepsilon$.
For any $c$:
\[
  \|A_c f-Af\|
  \le
  \underbrace{\|A_c(f-f_n)\|}
    _{\le\, M\varepsilon}
  +\underbrace{\|A_c f_n-Af_n\|}
    {\xrightarrow{\,c\to\infty\,}0}
  +\underbrace{\|A(f_n-f)\|}
    {\le\,\|A\|\varepsilon}.
\]
Hence
$\limsup_{c}\|A_c f-Af\|
\le(M+\|A\|)\varepsilon$.
Since $\varepsilon>0$ is arbitrary,
$A_c f\to Af$.

\textit{(b).}
Choose $f_\varepsilon\in\mathcal{D}$
with $\|f_\varepsilon-f\|<\varepsilon$.
Then
$\sup_c\|A_c f_\varepsilon-Af\|
\le M\varepsilon
+\sup_c\|A_c f_\varepsilon
  -Af_\varepsilon\|
+\|A\|\varepsilon$.
The middle term vanishes as
$c\to\infty$ for fixed $f_\varepsilon$;
all terms vanish as $\varepsilon\to 0$.
\end{proof}

\begin{remark}[Application to PSWF]
\label{rem:sc-app}
Apply Lemma~\ref{lem:sc} with
$A_c:=\Hc$, $A:=\Hlim$,
$\mathcal{D}:=\Dom(\Hlim)$,
$M:=1$.
Conditions hold by
\eqref{IX6}, \eqref{IX7},
\eqref{IX2}.
Conclusion~\eqref{sc:strong}:
$\Hc\to\Hlim$ strongly on all of
$L^2(\mathbb{R})$.
This is used in
\begin{itemize}
\item the proof of \eqref{IX2}
      ($\|\Hlim\|\le 1$),
\item Definition~\ref{def:qlim}
      ($\qlim(f,f)=\ip{\Hlim f}{f}$
      for all $f$),
\item Mosco~(M2)
      (Theorem~\ref{thm:mosco}),
\item Step~1 of
      Theorem~\ref{thm:friedrichs}.
\end{itemize}
\end{remark}

\section{Limit Form}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
For $f\in L^2(\mathbb{R})$:
\[
  \qlim(f,f)
  :=\lim_{c\to\infty}\ip{\Hc f}{f}.
\]
By Remark~\ref{rem:sc-app}
and $\Hlim\in\mathcal{B}(L^2)$,
the limit equals
$\ip{\Hlim f}{f}\in[0,\|f\|^2]$
for every $f$, so
$\Dom(\qlim)=L^2(\mathbb{R})$.
Extend by polarisation to
a sesquilinear form.
\end{definition}

\begin{theorem}[Existence]
\label{thm:qlim-exists}
$\qlim$ exists for all
$f\in L^2(\mathbb{R})$,
equals $\ip{\Hlim f}{f}$,
and defines a Hermitian
semibounded form on $L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Monotonicity \eqref{IX4} and
boundedness \eqref{IX3} give
existence; the identity
$\qlim(f,f)=\ip{\Hlim f}{f}$
follows from
Lemma~\ref{lem:sc}(a)
(Remark~\ref{rem:sc-app}).
\end{proof}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence]
\label{def:mosco}
Forms $(\qc)$ converge to $\qlim$
in the sense of Mosco
\cite{Mosco1969} on $L^2(\mathbb{R})$:
\begin{enumerate}[\rm(M1)]
\item\label{M1}
      $f_n\rightharpoonup f$
      $\Rightarrow$
      $\qlim(f,f)
      \le\liminf_c\qc(f_n,f_n)$.
\item\label{M2}
      $\forall f\in L^2(\mathbb{R})$,
      $\exists f_c\to f$ strongly with
      $\qc(f_c,f_c)\to\qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[HS and weak continuity]
\label{lem:hs}
All operators act on $L^2(\mathbb{R})$
via Convention~\ref{conv:tilde}.
For each fixed $c_0>0$:
\begin{enumerate}[\rm(a)]
\item $\widetilde{\mathcal{H}}_{c_0}
      \in\mathcal{B}_2(L^2(\mathbb{R}))$
      (Hilbert--Schmidt, hence compact).
\item For every weakly convergent
      sequence $f_n\rightharpoonup f$
      in $L^2(\mathbb{R})$:
      \[
        \ip{\widetilde{\mathcal{H}}_{c_0}
          f_n}{f_n}
        \to
        \ip{\widetilde{\mathcal{H}}_{c_0}
          f}{f}.
      \]
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a).}
For fixed $c_0$:
$|K_{c_0}(x,y)|\le c_0/\pi$
on $[0,1]^2$
(using $|\sin\theta|\le|\theta|$),
so
$\|\mathcal{H}_{c_0}^{[0,1]}
\|_{\mathrm{HS}}^2
\le(c_0/\pi)^2<\infty$
for this fixed $c_0$.
The tilde-extension preserves the
HS-norm.
HS $\Rightarrow$ compact
(\cite{ReedSimon1975}, Thm.~VI.22).

\textit{(b).}
Decompose
$\ip{\widetilde{\mathcal{H}}_{c_0}
f_n}{f_n}
-\ip{\widetilde{\mathcal{H}}_{c_0}f}{f}
=I+II$
as in \eqref{eq:3term}.
\textit{Term~II:}
$\widetilde{\mathcal{H}}_{c_0}f$
is a fixed $L^2$ vector;
$f_n-f\rightharpoonup 0$ gives
$II\to 0$.
\textit{Term~I:}
Weak convergence implies
$\sup_n\|f_n\|\le M$
(\cite{ReedSimon1975}, Thm.~III.5);
compactness gives
$\|\widetilde{\mathcal{H}}_{c_0}
(f_n-f)\|\to 0$
(\cite{ReedSimon1975}, Thm.~VI.11),
so $|I|\le M\cdot
\|\widetilde{\mathcal{H}}_{c_0}
(f_n-f)\|\to 0$.
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc\to\qlim$ in Mosco sense.
\end{theorem}

\begin{proof}
\textit{(M2):}
Take $f_c:=f$.
By Lemma~\ref{lem:sc}(a)
(Remark~\ref{rem:sc-app}),
$\qc(f,f)=\ip{\Hc f}{f}
\to\ip{\Hlim f}{f}=\qlim(f,f)$.

\textit{(M1):}
Let $f_n\rightharpoonup f$.
Fix $c_0$; for $c\ge c_0$:
$\qc(f_n,f_n)
\ge q_{c_0}(f_n,f_n)$
(monotonicity \eqref{IX4});
by Lemma~\ref{lem:hs}(b),
$q_{c_0}(f_n,f_n)
\to q_{c_0}(f,f)$,
so
$\liminf_c\qc(f_n,f_n)
\ge q_{c_0}(f,f)$.
Send $c_0\to\infty$:
$\liminf_c\qc(f_n,f_n)
\ge\qlim(f,f)$.
\end{proof}

\begin{remark}[Mosco $\Rightarrow$
  strong resolvent convergence]
By Kuwae--Shioya
\cite{KuwaeShioya2003}, Thm.~2.4,
Mosco convergence implies strong
resolvent convergence; see
Open Problem~\ref{prob:resolvent}.
\end{remark}

\section{Closedness}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness]
\label{thm:form-closed}
$\qlim\ge 0$ and $\qlim$ is closed.
\end{theorem}
\begin{proof}
$\qlim\ge 0$: each $\ip{\Hc f}{f}\ge 0$.
Closedness: if $f_k\to f$ in $L^2$
(hence $f_k\rightharpoonup f$)
and $\qlim(f_k-f_j,f_k-f_j)\to 0$,
then (M1) gives
$\qlim(f,f)\le\liminf_k\qlim(f_k,f_k)
<\infty$,
and the parallelogram law yields
$\qlim(f_k-f,f_k-f)\to 0$,
so $(\Dom(\qlim),\|\cdot\|_{\qlim})$
is complete
(\cite{Kato1966}, Def.~VI.1.3;
\cite{Mosco1969}, Thm.~1).
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative convergence]
\label{thm:form-convergence}
For all $f\in L^2(\mathbb{R})$,
$c\ge c_0(\kappa)$:
\[
  0\le\qlim(f,f)-\qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\]
\end{theorem}
\begin{proof}
Lower: monotonicity.
Upper:
$\qlim(f,f)-\qc(f,f)
=\ip{(\Hlim-\Hc)f}{f}
\le\|\Hlim-\Hc\|\|f\|^2
\le C_\kappa c^{-1/4}\|f\|^2$
by \eqref{IX2}.
\end{proof}

\section{Friedrichs Extension and
  Self-Adjointness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Kato representation]
\label{thm:kato-rep}
Densely defined, closed, semibounded
$q$ determines a unique self-adjoint
$T$, $\Dom(T)\subset\Dom(q)$, with
$q(f,g)=\ip{Tf}{g}$.
\emph{(\cite{Kato1966}, VI.2.1;
\cite{ReedSimon1975}, VIII.15.)}
\end{theorem}

\begin{lemma}[Core density]
\label{lem:core}
Let $T_{\qlim}$ be from
Theorem~\ref{thm:kato-rep}.
Then $\Dom(\Hlim)$ is a core for
$T_{\qlim}$, i.e.\ dense in
$(\Dom(T_{\qlim}),\|\cdot\|_{T_{\qlim}})$.
\end{lemma}

\begin{proof}
\textbf{Domain inclusion.}
By Step~1 of
Theorem~\ref{thm:friedrichs}
($\Hlim\subset T_{\qlim}$),
$\Dom(\Hlim)\subset\Dom(T_{\qlim})$.

\textbf{Norm equivalence.}
Since $T_{\qlim}\in\mathcal{B}(L^2)$
with $\|T_{\qlim}\|\le 1$,
all three norms
$\|\cdot\|$,
$\|\cdot\|_{T_{\qlim}}$,
$\|\cdot\|_{\qlim}$
are equivalent on $\Dom(T_{\qlim})$
(each is sandwiched between
$\|\cdot\|$ and $\sqrt{2}\|\cdot\|$).
It suffices to show density in
$\|\cdot\|_{\qlim}$.

\textbf{Step~A.}
For $g\in\Dom(T_{\qlim})$, $c>1$,
set $g_c:=(I+c^{-1}\Hc)^{-1}g$.
Since $0\le c^{-1}\Hc\le c^{-1}I$,
we have $I+c^{-1}\Hc\ge I>0$,
so spectral positivity gives
$\|(I+c^{-1}\Hc)^{-1}\|\le 1$.
Algebraically $\Hc g_c=c(g-g_c)$,
hence $\|g-g_c\|\le c^{-1}\|g\|\to 0$
and $\qlim(g-g_c,g-g_c)
=\ip{\Hlim(g-g_c)}{g-g_c}
\le\|g-g_c\|^2\to 0$.

\textbf{Step~B.}
Fix $c,\varepsilon$;
choose $h\in\Dom(\Hlim)$
with $\|h-g_c\|<\varepsilon$.
Since $\Dom(\Hlim)\subset\Dom(T_{\qlim})$,
$h$ is in the right space.
$\qlim(h-g_c,h-g_c)
=\ip{\Hlim(h-g_c)}{h-g_c}
\le\|\Hlim\|\|h-g_c\|^2
\le\varepsilon^2$.

\textbf{Diagonal.}
For each $c$ choose $h_c\in\Dom(\Hlim)$
with $\|h_c-g_c\|_{\qlim}\le c^{-1}$;
then $\|h_c-g\|_{\qlim}\to 0$.
\end{proof}

\begin{theorem}[Essential
  self-adjointness]
\label{thm:friedrichs}
Let $T_{\qlim}$ be from
Theorem~\ref{thm:kato-rep}.
Then $T_{\qlim}=\overline{\Hlim}$
and $\Hlim$ is essentially
self-adjoint.
\end{theorem}

\begin{proof}
\textbf{Step~1} ($\Hlim\subset T_{\qlim}$).
For $f\in\Dom(\Hlim)$,
$g\in L^2$:
$\ip{\Hlim f}{g}
=\lim_c\ip{\Hc f}{g}
=\qlim(f,g)$
(Lemma~\ref{lem:sc}(a),
Remark~\ref{rem:sc-app});
hence $T_{\qlim}f=\Hlim f$.

\begin{remark*}
$\Hlim\subset T_{\qlim}$ and
$T_{\qlim}$ closed $\Rightarrow$
$\Hlim$ closable,
$\overline{\Hlim}\subset T_{\qlim}$.
\end{remark*}

\textbf{Step~2}
($T_{\qlim}\subset\overline{\Hlim}$).
Let $f\in\Dom(T_{\qlim})$.
By Lemma~\ref{lem:core},
choose $h_c\in\Dom(\Hlim)$ with
$h_c\to f$ in $L^2$.
By Step~1 and
$T_{\qlim}\in\mathcal{B}(L^2)$:
$\|\Hlim h_c-T_{\qlim}f\|
=\|T_{\qlim}(h_c-f)\|
\le\|h_c-f\|\to 0$.
Graph sequence
$(h_c,\Hlim h_c)\to(f,T_{\qlim}f)$;
$\overline{\Hlim}$ closed gives
$f\in\Dom(\overline{\Hlim})$.

\textbf{Steps~3--4.}
$T_{\qlim}=\overline{\Hlim}$
self-adjoint $\Rightarrow$
$n_\pm=0$ $\Rightarrow$
$\Hlim$ essentially self-adjoint.
\end{proof}

\begin{remark}[Input audit]
\label{rem:audit}
\begin{itemize}
\item \eqref{IX6}: symmetric kernel
$\xrightarrow{\text{Sp.Thm.}}$
$\|\Hc\|\le 1$.
\item \eqref{IX7}: strong conv.\ on
$\Dom(\Hlim)$
$\xrightarrow{\text{Lem.}\ref{lem:sc}}$
strong conv.\ on all $L^2$.
\item \eqref{IX4}: monotonicity
$\to$ M1 lower bound.
\item \eqref{IX2}: norm rate
$\to$ form rate, Step~B.
\item Kato VI.2.1 $\to$ $T_{\qlim}$.
\item Conv.~\ref{conv:tilde}:
single ambient space.
\end{itemize}
No eigenvector, Parseval, or
completeness is used anywhere.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Essential
  self-adjointness]
\label{cor:esa}
$\Hlim$ is essentially self-adjoint:
$\overline{\Hlim}=\Hlim^*$,
$\sigma_{\rm app}(\Hlim)
=\sigma(\overline{\Hlim})$.
\end{corollary}

\begin{corollary}[Spectral chain]
\label{cor:chain}
\[
  \underbrace{\Delta_\varepsilon
    \asymp c^{-1/2}}_{\text{VI--VIII}}
  \Rightarrow
  \underbrace{\rho\in
    \sigma_{\rm app}(\Hlim)
    }_{\text{VIII [CCM]}}
  \Rightarrow
  \underbrace{\rho\in
    \sigma(\overline{\Hlim})
    }_{\text{X, uncond.}}
  \Rightarrow
  \underbrace{\operatorname{Re}(\rho)
    =\tfrac12}_{\text{Cor.
    \ref{cor:rh}}}.
\]
\end{corollary}

\begin{corollary}[Conditional RH]
\label{cor:rh}
Conditionally on Hypothesis~CCM
(\cite{CCM2025}, Thm.~1.3),
every non-trivial zero $\rho$ of
$\zeta(s)$ satisfies
$\operatorname{Re}(\rho)=\tfrac{1}{2}$.
\end{corollary}

\begin{remark}[Status]
\[
  \underbrace{\text{Papers I--X}}_{
    \text{unconditional}}
  +\underbrace{\text{CCM}}_{
    \text{external}}
  \Rightarrow
  \operatorname{Re}(\rho)=\tfrac12.
\]
Paper~IX, Problem~10 is independent
and blocks no result.
\end{remark}

independently via PSWF machinery.

Is $c^{-1/4}$ sharp in
Theorem~\ref{thm:form-convergence}?

\label{prob:resolvent}
Verify the Kuwae--Shioya framework
\cite{KuwaeShioya2003} for the
PSWF family to obtain
$(\Hc-z)^{-1}\to
(\overline{\Hlim}-z)^{-1}$ strongly.

  reclassified]
\label{prob:completeness}
Prove $\{\Phi_n^{(\infty)}\}$

This does not block any result.
\end{problem}
