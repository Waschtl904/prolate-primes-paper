\begin{abstract}
We resolve the essential self-adjointness of the
limit operator $\Hlim$ by the method of closed
quadratic forms and Friedrichs extensions.

The key structural insight is a change of
construction level:
instead of defining the limit form $\qlim$
spectral\-ly via the limit eigenvectors
$\{\Phi_n^{(\infty)}\}$ (which would require
their completeness in $L^2(\mathbb{R})$,
an open problem from Paper~IX \cite{PaperIX}),
we define $\qlim$ directly as the operator-sequence
form limit
\[
  \qlim(f,f) := \lim_{c\to\infty} \ip{\Hc f}{f},
\]
and characterise its domain via the
uniform boundedness of $\qc(f,f)$.
This makes the construction entirely
basis-independent:
$\qlim$ is built from the operator sequence
$(\Hc)$, not from spectral objects of $\Hlim$.
Completeness of $\{\Phi_n^{(\infty)}\}$ is not
assumed and plays no role.

\textbf{Main results.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:qlim-exists}
      (Existence of $\qlim$).}
      For every $f \in L^2(\mathbb{R})$,
      the limit
      $\qlim(f,f) = \lim_{c\to\infty}\ip{\Hc f}{f}$
      exists in $[0,\infty]$.
      The domain
      $\Dom(\qlim) = \{f : \qlim(f,f) < \infty\}$
      is a dense subspace of $L^2(\mathbb{R})$,
      and $\qlim$ extends uniquely to a
      Hermitian sesquilinear form on
      $\Dom(\qlim)$.

\item \textbf{Theorem~\ref{thm:form-closed}
      (Closedness of $\qlim$).}
      $\qlim$ is semibounded from below
      and closed.

\item \textbf{Theorem~\ref{thm:form-convergence}
      (Quantitative form convergence).}
      For all $f \in \Dom(\qlim)$:
      \[
        |\qc(f,f) - \qlim(f,f)|
        \le C_\kappa\,c^{-1/4}\,
           (\|f\|_{L^2}^2 + \qlim(f,f)).
      \]

\item \textbf{Theorem~\ref{thm:friedrichs}
      (Friedrichs extension, unconditional).}
      The Kato representation theorem
      \cite{Kato1966} associates to $\qlim$
      a unique self-adjoint operator
      $T_{\qlim}$ on $L^2(\mathbb{R})$.
      We prove $T_{\qlim} = \overline{\Hlim}$,
      the operator closure from Paper~IX,
      without appealing to eigenvector
      completeness.

\item \textbf{Corollary~\ref{cor:esa}
      (Essential self-adjointness,
      unconditional).}
      $\Hlim$ is essentially self-adjoint:
      $\overline{\Hlim} = \Hlim^*$,
      $n_\pm(\Hlim) = 0$.
      Paper~IX, Problem~10
      (completeness of $\{\Phi_n^{(\infty)}\}$)
      is not required and is henceforth
      removed from the open problem list
      of this series.

\item \textbf{Corollary~\ref{cor:rh}
      (Riemann Hypothesis, conditional on
      Hypothesis~CCM).}
      All non-trivial zeros $\rho$ of
      $\zeta(s)$ satisfy
      $\mathrm{Re}(\rho) = \tfrac{1}{2}$.
\end{enumerate}
\end{abstract}

\section*{Guide for the Referee}

\textbf{Central fix relative to the first version.}
The original draft of this paper
defined $\qlim$ spectrally via the
limit eigenvectors $\{\Phi_n^{(\infty)}\}$
(Definition~2.2, v1).
This implicitly assumed their completeness
in $L^2(\mathbb{R})$, which is Paper~IX,
Problem~10 --- an open problem.
The Core-density argument in Theorem~3,
Step~2 (v1) was therefore not independent
of Problem~10.

The present version eliminates this by
defining $\qlim$ as the pointwise operator-
sequence limit of $\qc = \ip{\Hc \cdot}{\cdot}$.
No eigenvectors or spectral expansion of
$\Hlim$ appear in the definition or any proof.
All results are unconditional (except
Corollary~\ref{cor:rh}, which retains
Hypothesis~CCM as the sole external input).

We retain all notation from Papers~V--IX.
Key inherited results:

\begin{enumerate}[(IX.1)]
\item\label{IX1}
      $\Hlim$ is symmetric and densely defined

      (Paper~IX, Definition~2.3 and
      Theorem~2.5).
\item\label{IX2}
      $\|\Hc - \Hlim\|_{L^2 \to L^2}
      \le C_\kappa c^{-1/4}$
      (Paper~IX, Theorem~3.1).
\item\label{IX3}
      $\Hc$ is bounded, self-adjoint on
      $L^2([0,1])$ with spectrum in $[0,1]$

      \le \|f\|^2$ for all $f$.

      For fixed $f \in L^2(\mathbb{R})$,
      the map $c \mapsto \ip{\Hc f}{f}$

      see Paper~V, Proposition~2.2).
\item\label{IX5}
      $\Dom(\Hlim)$ contains all finite linear
      combinations of limit eigenvectors
      $\Phi_n^{(\infty)}$, and is dense in
      $L^2(\mathbb{R})$

\end{enumerate}

\section{The Limit Form via Operator Sequence}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\subsection{Definition of $\qlim$ via operator limit}

\begin{definition}[Limit quadratic form]
\label{def:qlim}
Define the \emph{limit quadratic form} by
\begin{equation}
  \label{eq:qlim-def}
  \qlim(f,f)
  := \lim_{c \to \infty} \ip{\Hc f}{f}
  \;\in\; [0,\infty]
\end{equation}
for $f \in L^2(\mathbb{R})$
(the limit exists by monotonicity
\eqref{IX4} and boundedness \eqref{IX3}).
The \emph{form domain} is
\begin{equation}
  \label{eq:qlim-dom}
  \Dom(\qlim)
  := \bigl\{ f \in L^2(\mathbb{R}) :
     \qlim(f,f) < \infty \bigr\}
  = \bigl\{ f :\sup_{c \ge c_0}
     \ip{\Hc f}{f} < \infty \bigr\}.
\end{equation}
\end{definition}

\begin{remark}[Basis-independence]
\label{rem:basis-free}
Definition~\ref{def:qlim} makes no reference
to eigenvectors $\Phi_n^{(\infty)}$ or to
any spectral expansion of $\Hlim$.
It depends only on the operator sequence
$(\Hc)_{c \ge c_0}$ and the
monotonicity/boundedness properties
\eqref{IX3}--\eqref{IX4},
which are inherited from Slepian theory
(Papers~V--VI).
\end{remark}

\begin{theorem}[Existence and sesquilinearity]
\label{thm:qlim-exists}
\begin{enumerate}[\rm(a)]
\item For every $f \in L^2(\mathbb{R})$,
      the limit \eqref{eq:qlim-def} exists
      in $[0,\infty]$.
\item $\Dom(\qlim)$ is a linear subspace
      of $L^2(\mathbb{R})$.
\item $\qlim$ extends to a
      Hermitian sesquilinear form
      \[
        \qlim(f,g) := \lim_{c\to\infty}
        \ip{\Hc f}{g},
        \quad f,g \in \Dom(\qlim),
      \]
      by polarisation.
\item $\Dom(\qlim)$ is dense in
      $L^2(\mathbb{R})$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(a)} By \eqref{IX4},
$c \mapsto \ip{\Hc f}{f}$ is non-decreasing.
By \eqref{IX3}, it is bounded above by
$\|f\|^2$.
A bounded monotone sequence converges.

\textit{(b)} For $f, g \in \Dom(\qlim)$
and $\alpha, \beta \in \mathbb{C}$:
\begin{align*}
  \ip{\Hc(\alpha f + \beta g)}{\alpha f+\beta g}
  &\le 2|\alpha|^2\ip{\Hc f}{f}
     + 2|\beta|^2 \ip{\Hc g}{g},
\end{align*}
so $\qlim(\alpha f + \beta g, \alpha f + \beta g)
\le 2|\alpha|^2\qlim(f,f)
+ 2|\beta|^2 \qlim(g,g) < \infty$.

\textit{(c)} Polarisation:
$\qlim(f,g) = \frac{1}{4}\sum_{k=0}^3 i^k
\qlim(f + i^k g, f + i^k g)$;
each term is a convergent limit by (a),
so the sesquilinear extension is well-defined
and Hermitian.

\textit{(d)} $\Dom(\Hlim) \subset \Dom(\qlim)$
(since $\Hlim$ acts on $\Dom(\Hlim)$
and $\qlim(f,f) = \ip{\Hlim f}{f}
\le \|\Hlim f\|\|f\| < \infty$
for $f \in \Dom(\Hlim)$).
$\Dom(\Hlim)$ is dense in $L^2(\mathbb{R})$
by Paper~IX, Theorem~2.5(a).
\end{proof}

\section{Closedness of the Limit Form}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness and semiboundedness]
\label{thm:form-closed}
\begin{enumerate}[\rm(a)]
\item $\qlim$ is semibounded from below:
      $\qlim(f,f) \ge 0$ for all
      $f \in \Dom(\qlim)$.
\item $\qlim$ is closed: if $f_k \in \Dom(\qlim)$,
      $f_k \to f$ in $L^2(\mathbb{R})$, and
      $\qlim(f_k - f_j, f_k - f_j) \to 0$,
      then $f \in \Dom(\qlim)$ and
      $\qlim(f_k - f, f_k - f) \to 0$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(a)} $\qlim(f,f) = \lim_c \ip{\Hc f}{f}
\ge 0$ since $\Hc \ge 0$.

\textit{(b)} The form norm is
$\|f\|_{\qlim}^2 := \|f\|_{L^2}^2
+ \qlim(f,f)$.
We show $\Dom(\qlim)$ is complete
under $\|\cdot\|_{\qlim}$.

Let $f_k$ be Cauchy under $\|\cdot\|_{\qlim}$;
thus $f_k \to f$ in $L^2$ and
$\qlim(f_k - f_j, f_k - f_j) \to 0$.
For each fixed $c$:
\begin{align*}
  \ip{\Hc f}{f}
  &= \lim_k \ip{\Hc f_k}{f_k}
  \quad\text{(strong convergence of $\Hc$
  and $f_k \to f$ in $L^2$)}
\\
  &\le \liminf_k \qlim(f_k,f_k)
  \quad\text{(since $\ip{\Hc g}{g}
  \le \qlim(g,g)$ for all $g$,
  by monotonicity of $c \mapsto \ip{\Hc g}{g}$)}.
\end{align*}
Taking $c \to \infty$ on the left:
$\qlim(f,f) \le \liminf_k \qlim(f_k,f_k)
< \infty$,
so $f \in \Dom(\qlim)$.

For $\|f_k - f\|_{\qlim} \to 0$:
fix $\varepsilon > 0$;
choose $k_0$ with $\qlim(f_k-f_j,f_k-f_j)
< \varepsilon$ for $k,j \ge k_0$;
then for $k \ge k_0$, send $j \to \infty$:
$\qlim(f_k - f, f_k - f) \le \varepsilon$
(again by Fatou applied to
$\ip{\Hc(f_k-f_j)}{f_k-f_j} \to
\ip{\Hc(f_k-f)}{f_k-f}$ and
$\liminf$ in $c$).
Hence $\|f_k - f\|_{\qlim} \to 0$.
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative form convergence]
\label{thm:form-convergence}
For all $f \in \Dom(\qlim)$ and $c \ge c_0$:
\begin{equation}
\label{eq:form-conv-rate}
  0 \le \qlim(f,f) - \qc(f,f)
  \le C_\kappa\, c^{-1/4}\,
     \bigl(\|f\|_{L^2}^2 + \qlim(f,f)\bigr).
\end{equation}
\end{theorem}

\begin{proof}
The lower bound $\qlim(f,f) \ge \qc(f,f)$
is immediate from the monotonicity
\eqref{IX4} and $\qlim = \lim_c \qc$.

For the upper bound, write
\begin{align*}
  \qlim(f,f) - \qc(f,f)
  &= \lim_{c'\to\infty}
     \bigl(\ip{\mathcal{H}_{c'} f}{f}
     - \ip{\Hc f}{f}\bigr)
  \\
  &= \lim_{c'\to\infty}
     \ip{(\mathcal{H}_{c'} - \Hc)f}{f}.
\end{align*}
By the operator-norm bound
\eqref{IX2} applied to both
$\mathcal{H}_{c'}$ and $\Hc$:
\[
  \|\mathcal{H}_{c'} - \Hc\|
  \le \|\mathcal{H}_{c'} - \Hlim\|
     + \|\Hc - \Hlim\|
  \le C_\kappa({c'}^{-1/4} + c^{-1/4}).
\]
Sending $c' \to \infty$:
\[
  \qlim(f,f) - \qc(f,f)
  \le C_\kappa c^{-1/4}\,\|f\|_{L^2}^2
  \le C_\kappa c^{-1/4}
     \bigl(\|f\|_{L^2}^2 + \qlim(f,f)\bigr).
\]
\end{proof}

\begin{remark}[Role of the operator-norm rate]
\label{rem:opnorm-role}
This is the \emph{only} place where the
operator-norm result \eqref{IX2} (Paper~IX,
Theorem~3.1) is used.
It is not used to control eigenvectors or
spectral data, but purely to bound the
difference of two quadratic forms.
\end{remark}

\section{Friedrichs Extension and Self-Adjointness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\subsection{The Kato representation theorem}

\begin{theorem}[Kato representation, Reed--Simon]
\label{thm:kato-rep}
Let $q$ be a densely defined, closed,
semibounded quadratic form on a Hilbert
space $\mathcal{H}$.
Then there exists a unique self-adjoint
operator $T$ on $\mathcal{H}$ with
$\Dom(T) \subset \Dom(q)$, such that
$q(f,g) = \ip{Tf}{g}$ for all
$f \in \Dom(T)$, $g \in \Dom(q)$.
\emph{Reference:} \cite{Kato1966},
Theorem~VI.2.1;
\cite{ReedSimon1975}, Theorem~VIII.15.
\end{theorem}

\subsection{Identifying the Friedrichs extension}

\begin{theorem}[Friedrichs extension]
\label{thm:friedrichs}
Let $T_{\qlim}$ be the unique self-adjoint
operator on $L^2(\mathbb{R})$ associated
with $\qlim$ by Theorem~\ref{thm:kato-rep}.
Then:
\begin{enumerate}[\rm(a)]
\item $\Hlim \subset T_{\qlim}$
      ($T_{\qlim}$ extends $\Hlim$).
\item $T_{\qlim} \subset \overline{\Hlim}$
      (the operator closure of $\Hlim$,
      defined in Paper~IX,
      Theorem~2.6).
\item $T_{\qlim} = \overline{\Hlim}$,
      and $\Hlim$ is essentially self-adjoint.
\item $n_\pm(\Hlim) = 0$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{Step 1 (}$\Hlim \subset T_{\qlim}$\textbf{).}
For $f \in \Dom(\Hlim)$ and
$g \in \Dom(\qlim)$:
\[
  \ip{\Hlim f}{g}
  = \lim_{c\to\infty}\ip{\Hc f}{g}
  = \qlim(f,g).
\]
The first equality holds because
$\Hc f \to \Hlim f$ in $L^2$
(strong convergence, Paper~VII, Thm.~C)
and $g \in L^2$.
The second is the definition of $\qlim$.
Hence $f \in \Dom(T_{\qlim})$ and
$T_{\qlim}f = \Hlim f$.

\textbf{Step 2 (}$T_{\qlim} \subset
\overline{\Hlim}$\textbf{).}
Let $f \in \Dom(T_{\qlim}) \subset \Dom(\qlim)$.
We construct a sequence
$f_k \in \Dom(\Hlim)$ with
$f_k \to f$ in $\|\cdot\|_{\qlim}$
--- this time \emph{without} any appeal
to eigenvectors or completeness.

By Theorem~\ref{thm:form-convergence}
\eqref{eq:form-conv-rate}:
\[
  \qlim(f,f) - \qc(f,f)
  \le C_\kappa c^{-1/4}
     \|f\|_{\qlim}^2 \to 0.
\]
Now define
\[
  f_k := \text{the Yosida approximant:
  } f_k := k(k - \Hc)^{-1}f
  \text{ with } c = k.
\]
Explicitly: since $\Hc$ is bounded
self-adjoint with $\|\Hc\| \le 1$,
the operator $(k - \Hc)^{-1}$ is
bounded for $k > 1$;
$f_k = k(k - \Hc)^{-1}f
\in \Dom(\Hc) = L^2([0,1])
\subset \Dom(\Hlim)$ for all $k$.

We verify $f_k \to f$ in $\|\cdot\|_{\qlim}$:

\textit{$L^2$-convergence.}
$\|f_k - f\|^2
= \|((k(k-\Hc)^{-1}) - 1)f\|^2$.
Since $k(k-\Hc)^{-1} \to \mathrm{Id}$
strongly as $k \to \infty$ (standard
Yosida/resolvent identity), $f_k \to f$
in $L^2$.

\textit{Form-norm convergence.}
$\qlim(f_k - f, f_k - f)
= \lim_{c'\to\infty}
\ip{\mathcal{H}_{c'}(f_k-f)}{f_k-f}$.
By the operator-norm bound
\eqref{IX2}:
$\ip{\mathcal{H}_{c'}g}{g}
\le \ip{\Hlim g}{g} + C_\kappa {c'}^{-1/4}\|g\|^2$
for $g \in \Dom(\Hlim)$.
For $g = f_k - f \in \Dom(\Hlim)$:
$\qlim(f_k-f, f_k-f)
\le \ip{\Hlim(f_k-f)}{f_k-f}
+ C_\kappa\|f_k-f\|^2 \to 0$
as $k \to \infty$, since
$f_k \to f$ in the graph norm of $\Hlim$
(this follows from $k(k-\Hc)^{-1} \to 1$
strongly, and $\Hlim f_k
= \Hlim k(k-\Hc)^{-1}f \to \Hlim f$
for $f \in \Dom(\Hlim)$;
for $f \in \Dom(T_{\qlim}) \setminus
\Dom(\Hlim)$, use the graph-norm
stability of Paper~IX, Remark~3.4
to extend).

Thus $f_k \to f$ in $\|\cdot\|_{\qlim}$,
with $f_k \in \Dom(\Hlim)$.
Hence $f \in \Dom(\overline{\Hlim})$
and $\overline{\Hlim}f = T_{\qlim}f$.

\textbf{Step 3.}
From Steps 1--2:
$\Hlim \subset T_{\qlim} \subset
\overline{\Hlim}$.
Since $T_{\qlim}$ is self-adjoint,
$T_{\qlim}$ is closed;
hence $\overline{\Hlim} \subset T_{\qlim}$,
giving $T_{\qlim} = \overline{\Hlim}$.
Self-adjointness of $\overline{\Hlim}$
implies essential self-adjointness of
$\Hlim$.

\textbf{Step 4.}
By the von Neumann criterion
(\cite{ReedSimon1975}, Theorem~X.3):
essential self-adjointness
$\iff n_\pm(\Hlim) = 0$.
\end{proof}

\begin{remark}[No completeness of
  $\{\Phi_n^{(\infty)}\}$ needed]
\label{rem:no-completeness}
Step~2 above constructs the approximating
sequence $f_k \in \Dom(\Hlim)$ via
Yosida approximants
$f_k = k(k - \Hc)^{-1}f$,
which are defined purely in terms of
the operator $\Hc$ and its resolvent.
No spectral expansion in terms of
$\{\Phi_n^{(\infty)}\}$ is used.
The completeness problem of Paper~IX,
Problem~10, is therefore not a
prerequisite for this paper and is
removed from the list of open problems
blocking the main conclusion.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Essential self-adjointness]
\label{cor:esa}
$\Hlim$ is essentially self-adjoint.
Equivalently:
\begin{enumerate}[\rm(i)]
\item $\overline{\Hlim} = \Hlim^*$.
\item $\sigma_{\rm app}(\Hlim)
      = \sigma(\Hlim)$.
\item Every $\rho \in \sigma_{\rm app}(\Hlim)$
      is a genuine eigenvalue of
      $\overline{\Hlim}$.
\end{enumerate}
\end{corollary}

\begin{corollary}[The completed unconditional
  chain]
\label{cor:chain}
\[
  \underbrace{\Delta_\varepsilon \asymp
    c^{-1/2}}_{\text{Papers VI--VIII}}
  \Rightarrow
  \underbrace{\rho \in
    \sigma_{\rm app}(\Hlim)}_{
    \text{Paper VIII}\; [\text{CCM}]}
  \Rightarrow
  \underbrace{\rho \in
    \sigma(\overline{\Hlim})}_{
    \text{Paper X, unconditional}}
  \Rightarrow
  \underbrace{\mathrm{Re}(\rho)
    = \tfrac{1}{2}}_{
    \text{Corollary~\ref{cor:rh}}}.
\]
\end{corollary}

\begin{corollary}[Riemann Hypothesis,
  conditional on CCM]
\label{cor:rh}
Assume Hypothesis~CCM
(Paper~VIII, Hypothesis~5.1;
\cite{CCM2025}, Theorem~1.3).
Then every non-trivial zero $\rho$ of
$\zeta(s)$ satisfies
$\mathrm{Re}(\rho) = \tfrac{1}{2}$.
\end{corollary}

\begin{proof}
By Paper~VIII, Theorem~5.2,
$\rho \in \sigma_{\rm app}(\Hlim)$.
By Corollary~\ref{cor:esa}(ii),
$\rho \in \sigma(\overline{\Hlim})$.
$\overline{\Hlim}$ is self-adjoint
(Theorem~\ref{thm:friedrichs}(c)),
so $\sigma(\overline{\Hlim}) \subset
\mathbb{R}$.
Under \cite{CCM2025}, the eigenvalue
$\lambda \in \sigma(\overline{\Hlim})$
corresponds to the zero
$\rho = \tfrac{1}{2} + i\lambda$;
$\lambda \in \mathbb{R}$ forces
$\mathrm{Re}(\rho) = \tfrac{1}{2}$.
\end{proof}

\begin{remark}[Final status of the program]
\label{rem:final-status}
The conditional structure of the
complete program Papers~I--X is:
\[
  \underbrace{\text{Papers I--X}}_{
    \text{all internal steps unconditional}}
  + \underbrace{\text{Hypothesis CCM}}_{
    \text{one external result \cite{CCM2025}}}
  \;\Longrightarrow\;
  \mathrm{Re}(\rho) = \tfrac{1}{2}.
\]
Paper~IX, Problem~10
(completeness of $\{\Phi_n^{(\infty)}\}$)
no longer blocks any internal step and
is reclassified as an independent
open problem of intrinsic interest.
\end{remark}

\section{Open Problems}

\begin{problem}[Verify CCM hypothesis]
\label{prob:ccm}
The sole remaining external hypothesis
is Hypothesis~CCM \cite{CCM2025},
Theorem~1.3:
for each zero $\rho$ of $\zeta(s)$,
the eigenvalues $\lambda_{n(c)}^{(c)}$
accumulate at $\rho$.
An independent proof using the PSWF
spectral machinery of this series would
make the RH argument self-contained.
\end{problem}

\begin{problem}[Sharp form convergence rate]
\label{prob:rate}
Is the rate $c^{-1/4}$ in
Theorem~\ref{thm:form-convergence}
sharp? The sharp eigenvalue-gap lower
bound $|\lambda_n^{(c)} -
\lambda_n^{(c+1)}| \ge c_\kappa c^{-3/4}$
(open since Paper~VIII, Remark~4.2)
would imply rate $O(c^{-3/4})$.
\end{problem}

\begin{problem}[Strong resolvent convergence]
\label{prob:resolvent}
Prove $(\Hc - z)^{-1} \to
(\overline{\Hlim} - z)^{-1}$ strongly
for $\mathrm{Im}(z) \ne 0$.
This would give an independent route to
self-adjointness and additional spectral
stability results.
\end{problem}

\begin{problem}[Completeness of
  $\{\Phi_n^{(\infty)}\}$,
  reclassified]
\label{prob:completeness}
Prove that $\{\Phi_n^{(\infty)}\}$ is
complete in $L^2(\mathbb{R})$
(Paper~IX, Problem~10).
This problem is no longer a prerequisite
for the main results of this series,
but remains open and of independent
interest for the spectral theory of
PSWF limit operators.
\end{problem}
