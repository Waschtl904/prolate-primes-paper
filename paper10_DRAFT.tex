\author{[Author]}
\address{[Address]}

\maketitle

\begin{abstract}
We resolve the essential self-adjointness of the
limit operator $\Hlim$ by the method of
monotone/Mosco form convergence and
Friedrichs extensions.

The key structural point is a change of
construction level. The limit form $\qlim$
is defined directly as the operator-sequence
form limit
\[
  \qlim(f,f) := \lim_{c\to\infty} \ip{\Hc f}{f},
\]
using only the monotonicity and boundedness
of $c\mapsto \ip{\Hc f}{f}$ (inherited from
Slepian theory, Papers~V--VI).
This definition is basis-independent:
no eigenvectors $\Phi_n^{(\infty)}$,
no Parseval identity, and no completeness
hypothesis appear anywhere.
The form $\qlim$ is precisely the
\emph{monotone Mosco limit} of the
forms $\qc = \ip{\Hc \cdot}{\cdot}$
in the sense of Kato \cite{Kato1966},
Chapter~VI, and Mosco \cite{Mosco1969}.

\textbf{Main results.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:qlim-exists}
      (Existence and density).}
      $\qlim$ exists on a dense domain
      $\Dom(\qlim) \supset \Dom(\Hlim)$.

\item \textbf{Theorem~\ref{thm:form-closed}
      (Closedness).}
      $\qlim$ is semibounded from below
      and closed.

\item \textbf{Theorem~\ref{thm:mosco}
      (Mosco convergence).}
      $\qc \to \qlim$ in the sense of
      Mosco convergence on $L^2(\mathbb{R})$,
      with quantitative rate $O(c^{-1/4})$
      (Theorem~\ref{thm:form-convergence}).

\item \textbf{Theorem~\ref{thm:friedrichs}
      (Friedrichs extension, unconditional).}
      The Kato representation theorem
      \cite{Kato1966} associates to $\qlim$
      a unique self-adjoint operator
      $T_{\qlim}$ on $L^2(\mathbb{R})$,
      and $T_{\qlim} = \overline{\Hlim}$.
      The approximating sequence in the
      core-density step is constructed via
      Yosida approximants, using only
      $\Dom(\qlim)$ (not $\Dom(\Hlim)$).

\item \textbf{Corollary~\ref{cor:esa}
      (Essential self-adjointness,
      unconditional).}
      $\Hlim$ is essentially self-adjoint.
      Paper~IX, Problem~10 is reclassified
      as an independent open problem.

\item \textbf{Corollary~\ref{cor:rh}.}
      Under Hypothesis~CCM \cite{CCM2025},
      all non-trivial zeros $\rho$ of
      $\zeta(s)$ satisfy
      $\mathrm{Re}(\rho) = \tfrac{1}{2}$.
\end{enumerate}
\end{abstract}

\section*{Guide for the Referee}

\textbf{Structural fix relative to v1.}
The original draft defined $\qlim$
spectral\-ly via $\{\Phi_n^{(\infty)}\}$
(implying completeness = Paper~IX,
Problem~10).
This version replaces that with the
operator-sequence form limit
(Definition~\ref{def:qlim}),
classified as Mosco/monotone convergence
(Section~\ref{sec:mosco}).
Step~2 of the Friedrichs proof
(Theorem~\ref{thm:friedrichs})
builds the core sequence via Yosida
approximants in $\Dom(\qlim)$,
not $\Dom(\Hlim)$,
eliminating all spectral dependence.

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.
Key inherited results:

\begin{enumerate}[(IX.1)]
\item\label{IX1}
      $\Hlim$ is symmetric and densely defined
      on $\Dom(\Hlim)$
      (Paper~IX, Definition~2.3,
      Theorem~2.5).
\item\label{IX2}
      $\|\Hc - \Hlim\|_{L^2 \to L^2}
      \le C_\kappa c^{-1/4}$
      (Paper~IX, Theorem~3.1).
\item\label{IX3}
      $0 \le \ip{\Hc f}{f} \le \|f\|^2$
      for all $f$, all $c$
      (spectrum of $\Hc$ in $[0,1]$,
      Papers~V--VI).
\item\label{IX4}
      $c \mapsto \ip{\Hc f}{f}$ is
      monotone non-decreasing
      for each fixed $f$
      (Slepian monotonicity,
      Paper~V, Proposition~2.2).
\item\label{IX5}
      $\Dom(\Hlim)$ is dense in
      $L^2(\mathbb{R})$
      (Paper~IX, Theorem~2.5(a)).
\end{enumerate}

\begin{remark}[Objects not used in this paper]
\label{rem:not-used}
The following do not appear anywhere
in this paper:
\begin{itemize}
\item Eigenvectors $\Phi_n^{(\infty)}$
      or their completeness.
\item The Parseval identity for
      $\{\Phi_n^{(\infty)}\}$.
\item Any spectral expansion of $\Hlim$.
\item Paper~IX, Problem~10.
\end{itemize}
The construction proceeds from
the operator sequence $(\Hc)$ alone.
\end{remark}

\section{The Limit Form via Operator Sequence}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit quadratic form]
\label{def:qlim}
Define
\begin{equation}
  \label{eq:qlim-def}
  \qlim(f,f)
  := \lim_{c \to \infty} \ip{\Hc f}{f}
  \;\in\; [0,\infty]
\end{equation}
for $f \in L^2(\mathbb{R})$
(limit exists by \eqref{IX4}--\eqref{IX3}),
and
\begin{equation}
  \label{eq:qlim-dom}
  \Dom(\qlim)
  := \bigl\{
     f \in L^2(\mathbb{R}) :
     \qlim(f,f) < \infty
  \bigr\}.
\end{equation}
The sesquilinear extension to
$\Dom(\qlim) \times \Dom(\qlim)$
is defined by polarisation:
$\qlim(f,g) :=
\frac{1}{4}\sum_{k=0}^3 i^k
\qlim(f+i^k g, f+i^k g)$.
\end{definition}

\begin{theorem}[Existence, sesquilinearity, density]
\label{thm:qlim-exists}
\begin{enumerate}[\rm(a)]
\item The limit \eqref{eq:qlim-def}
      exists in $[0,\infty]$ for every
      $f \in L^2(\mathbb{R})$.
\item $\Dom(\qlim)$ is a dense linear
      subspace of $L^2(\mathbb{R})$;
      it contains $\Dom(\Hlim)$.
\item $\qlim$ is a well-defined
      Hermitian sesquilinear form on
      $\Dom(\qlim)$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(a)} Monotone \eqref{IX4} and
bounded above \eqref{IX3}:
a monotone bounded sequence converges.

\textit{(b)} Linearity: for
$f,g \in \Dom(\qlim)$,
$\qlim(\alpha f+\beta g,\alpha f+\beta g)
\le 2|\alpha|^2\qlim(f,f)
+2|\beta|^2\qlim(g,g)<\infty$.
Density: $\Dom(\Hlim) \subset \Dom(\qlim)$
because for $f \in \Dom(\Hlim)$:
$\qlim(f,f) = \lim_c \ip{\Hc f}{f}
\le \|\Hlim f\|\|f\| < \infty$;
and $\Dom(\Hlim)$ is dense \eqref{IX5}.

\textit{(c)} Follows from polarisation
and (a).
\end{proof}

\section{Mosco Convergence Classification}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence of forms]
\label{def:mosco}
A sequence of closed forms $(q_c)$ on a
Hilbert space $\mathcal{H}$ converges to
a closed form $q$ in the sense of
\emph{Mosco} \cite{Mosco1969} if:
\begin{enumerate}[\rm(M1)]
\item\label{M1}
      \textbf{(liminf condition)}
      For every $f_c \rightharpoonup f$
      weakly in $\mathcal{H}$:
      $q(f,f) \le \liminf_c q_c(f_c,f_c)$.
\item\label{M2}
      \textbf{(limsup condition)}
      For every $f \in \Dom(q)$,
      there exists $f_c \to f$ strongly
      such that $q_c(f_c,f_c) \to q(f,f)$.
\end{enumerate}
\end{definition}

\begin{theorem}[Mosco convergence
  of $\qc$ to $\qlim$]
\label{thm:mosco}
$\qc \to \qlim$ in the sense of Mosco.
\end{theorem}

\begin{proof}
\textit{(M1).}
Let $f_c \rightharpoonup f$ weakly.
For each fixed $c_0$:
\[
  \liminf_{c\to\infty} \qc(f_c,f_c)
  \ge \liminf_{c\to\infty}
     \ip{\mathcal{H}_{c_0}f_c}{f_c}
  = \ip{\mathcal{H}_{c_0}f}{f}
  = q_{c_0}(f,f),
\]
where we used $c \ge c_0$ (by monotonicity
\eqref{IX4}, $\qc \ge q_{c_0}$) and
weak convergence $f_c \rightharpoonup f$
with the compact action of $\mathcal{H}_{c_0}$
(finite-rank perturbation structure of
Slepian operators; see Paper~V, Sec.~3).
Sending $c_0 \to \infty$:
$\liminf \qc(f_c,f_c) \ge \qlim(f,f)$.

\textit{(M2).}
For $f \in \Dom(\qlim)$,
take $f_c = f$ for all $c$.
Then $f_c \to f$ strongly and
$\qc(f_c,f_c) = \qc(f,f) \to \qlim(f,f)$
by Definition~\ref{def:qlim}.
\end{proof}

\begin{remark}[Significance of Mosco convergence]
\label{rem:mosco-significance}
Mosco convergence of forms is the
standard condition under which
the associated self-adjoint operators
converge in the strong resolvent sense
(Kuwae--Shioya \cite{KuwaeShioya2003},
Theorem~2.4; see also
\cite{Kato1966}, Theorem~VIII.3.11).
This provides an independent route to
resolvent convergence (Open
Problem~\ref{prob:resolvent}) and
puts the operator family $(\Hc)$ firmly
within the Mosco--Kato framework for
operator convergence.
\end{remark}

\section{Closedness of the Limit Form}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness and semiboundedness]
\label{thm:form-closed}
\begin{enumerate}[\rm(a)]
\item $\qlim(f,f) \ge 0$ for all
      $f \in \Dom(\qlim)$.
\item $\qlim$ is closed.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(a)} Each $\ip{\Hc f}{f} \ge 0$,
hence $\qlim(f,f) = \lim_c \ip{\Hc f}{f}
\ge 0$.

\textit{(b)} Associate the form norm
$\|f\|_{\qlim}^2 := \|f\|_{L^2}^2
+ \qlim(f,f)$.
Let $f_k \in \Dom(\qlim)$ be Cauchy
under $\|\cdot\|_{\qlim}$;
write $f_k \to f$ in $L^2$.
For each fixed $c$:
\[
  \ip{\Hc f}{f}
  = \lim_k \ip{\Hc f_k}{f_k}
  \le \liminf_k \qlim(f_k,f_k).
\]
The first equality uses $\Hc$ bounded
and $f_k \to f$ in $L^2$;
the inequality uses $\ip{\Hc g}{g}
\le \qlim(g,g)$ (from \eqref{IX4}).
Sending $c\to\infty$:
$\qlim(f,f) \le \liminf_k \qlim(f_k,f_k)
< \infty$,
so $f \in \Dom(\qlim)$.

For $\|f_k-f\|_{\qlim}\to 0$:
given $\varepsilon>0$, pick $k_0$ with
$\qlim(f_k-f_j,f_k-f_j)<\varepsilon$
for $k,j\ge k_0$;
for fixed $c$ and $k\ge k_0$,
send $j\to\infty$:
$\ip{\Hc(f_k-f)}{f_k-f}
\le \varepsilon$ (same argument);
then $\qlim(f_k-f,f_k-f)\le\varepsilon$.
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative form convergence]
\label{thm:form-convergence}
For all $f \in \Dom(\qlim)$, $c \ge c_0$:
\begin{equation}
\label{eq:form-conv-rate}
  0 \le \qlim(f,f) - \qc(f,f)
  \le C_\kappa c^{-1/4}\,\|f\|_{L^2}^2.
\end{equation}
\end{theorem}

\begin{proof}
Lower bound: monotonicity \eqref{IX4}.
Upper bound: for any $c' > c$,
\[
  \qc[c'](f,f)-\qc(f,f)
  = \ip{(\mathcal{H}_{c'}-\Hc)f}{f}
  \le \|\mathcal{H}_{c'}-\Hc\|\,\|f\|^2.
\]
By \eqref{IX2}:
$\|\mathcal{H}_{c'}-\Hc\|
\le \|\mathcal{H}_{c'}-\Hlim\|
+\|\Hc-\Hlim\|
\le C_\kappa({c'}^{-1/4}+c^{-1/4})$.
Send $c'\to\infty$:
$\qlim(f,f)-\qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{proof}

\begin{remark}[Sole use of the
  operator-norm rate]
\label{rem:opnorm-role}
The operator-norm result \eqref{IX2}
(Paper~IX, Theorem~3.1) is used
exactly once, here, to bound the
difference of two quadratic forms.
It controls no spectral or eigenvector data.
\end{remark}

\section{Friedrichs Extension and Self-Adjointness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Kato representation]
\label{thm:kato-rep}
Let $q$ be densely defined, closed,
and semibounded on a Hilbert space
$\mathcal{H}$.
There is a unique self-adjoint $T$
with $\Dom(T)\subset\Dom(q)$ and
$q(f,g)=\ip{Tf}{g}$ for
$f\in\Dom(T)$, $g\in\Dom(q)$.
\emph{(\cite{Kato1966}, Thm.~VI.2.1;
\cite{ReedSimon1975}, Thm.~VIII.15.)}
\end{theorem}

\begin{theorem}[Friedrichs extension
  and essential self-adjointness]
\label{thm:friedrichs}
Let $T_{\qlim}$ be the self-adjoint
operator from Theorem~\ref{thm:kato-rep}
applied to $\qlim$.
Then:
\begin{enumerate}[\rm(a)]
\item $\Hlim \subset T_{\qlim}$.
\item $T_{\qlim} \subset \overline{\Hlim}$.
\item $T_{\qlim} = \overline{\Hlim}$;
      $\Hlim$ is essentially self-adjoint.
\item $n_\pm(\Hlim) = 0$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{Step 1} ($\Hlim \subset T_{\qlim}$).
For $f\in\Dom(\Hlim)$ and
$g\in\Dom(\qlim)$:
\[
  \ip{\Hlim f}{g}
  =\lim_{c\to\infty}\ip{\Hc f}{g}
  =\qlim(f,g).
\]
First equality: $\Hc f\to\Hlim f$
in $L^2$ (Paper~VII, Thm.~C) and
$g\in L^2$.
Second: definition of $\qlim$.
Hence $f\in\Dom(T_{\qlim})$ and
$T_{\qlim}f=\Hlim f$.

\textbf{Step 2} ($T_{\qlim}\subset
\overline{\Hlim}$).
Let $f\in\Dom(T_{\qlim})\subset\Dom(\qlim)$.
We construct
$f_k\in\Dom(\qlim)$, $f_k\to f$
in $\|\cdot\|_{\qlim}$,
with each $f_k$ in the operator domain
of $T_{\qlim}$ and approximable by
elements of $\Dom(\Hlim)$ via the
graph-norm topology
--- without any appeal to
eigenvectors or spectral expansions.

\emph{Construction.}
Set $c_k := k$ and define
the Yosida approximant
\begin{equation}
  \label{eq:yosida}
  f_k := R_k f,
  \quad
  R_k := k(k + T_{\qlim})^{-1}
  \in \mathcal{B}(L^2(\mathbb{R})).
\end{equation}
Note: $R_k$ uses the resolvent of
$T_{\qlim}$ (the \emph{already-constructed}
self-adjoint operator), not of $\Hlim$.
This avoids circularity.
Since $T_{\qlim}$ is self-adjoint:
\begin{itemize}
\item $f_k \in \Dom(T_{\qlim})
      \subset \Dom(\qlim)$.
\item $f_k \to f$ in $L^2$
      (standard resolvent approximation,
      \cite{ReedSimon1975}, Theorem~VIII.5).
\item $T_{\qlim}f_k
      = k T_{\qlim}(k+T_{\qlim})^{-1}f
      \to T_{\qlim}f$ in $L^2$.
\end{itemize}
Hence $f_k\to f$ in the graph norm
of $T_{\qlim}$, which coincides with
$\|\cdot\|_{\qlim}$ on $\Dom(T_{\qlim})$.

\emph{Connecting $f_k$ to $\Dom(\Hlim)$.}
Each $f_k\in\Dom(T_{\qlim})\subset\Dom(\qlim)$.
By Step~1, $T_{\qlim}$ extends $\Hlim$,
so every $g\in\Dom(\Hlim)$ satisfies
$T_{\qlim}g=\Hlim g$.
Apply the graph-norm stability of
Paper~IX, Remark~3.4 (which uses only
the operator-norm bound \eqref{IX2}):
there exist $g_j^{(k)}\in\Dom(\Hlim)$
with $g_j^{(k)}\to f_k$ in
$\|\cdot\|_{\qlim}$.
Hence $f_k\in\overline{\Hlim}^{\|\cdot\|_{\qlim}}$.
Since $f_k\to f$ in $\|\cdot\|_{\qlim}$,
also $f\in\Dom(\overline{\Hlim})$ and
$\overline{\Hlim}f = T_{\qlim}f$.

\textbf{Step 3.}
$\Hlim\subset T_{\qlim}\subset
\overline{\Hlim}$,
$T_{\qlim}$ closed and self-adjoint
$\Rightarrow T_{\qlim}=\overline{\Hlim}$.

\textbf{Step 4.}
Von Neumann: $n_\pm(\Hlim)=0
\iff \Hlim$ essentially self-adjoint.
\end{proof}

\begin{remark}[Domain hierarchy and
  why Step~2 is not circular]
\label{rem:not-circular}
The approximating sequence in Step~2
is constructed in three layers:
\begin{enumerate}[(i)]
\item
      $f_k = R_k f \in \Dom(T_{\qlim})
      \subset \Dom(\qlim)$
      (Yosida approximants of $T_{\qlim}$).
\item
      $f_k \approx g_j^{(k)}
      \in \Dom(\Hlim)$ in form norm
      (graph-norm stability, Paper~IX
      Remark~3.4, using only \eqref{IX2}).
\item
      The only spectral object used is
      $T_{\qlim}$ itself, which is
      constructed from $\qlim$ alone
      (Definition~\ref{def:qlim},
      no eigenvectors).
\end{enumerate}
No completeness of $\{\Phi_n^{(\infty)}\}$
is used at any point.
The domain chain is:
$\Dom(\Hlim) \subset \Dom(T_{\qlim})
= \Dom(\overline{\Hlim}) \subset \Dom(\qlim)$,
with inclusions established by
form-theoretic arguments only.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Essential self-adjointness]
\label{cor:esa}
$\Hlim$ is essentially self-adjoint.
Equivalently:
\begin{enumerate}[\rm(i)]
\item $\overline{\Hlim} = \Hlim^*$.
\item $\sigma_{\rm app}(\Hlim)
      = \sigma(\Hlim)$.
\item Every $\rho\in\sigma_{\rm app}(\Hlim)$
      is a genuine eigenvalue of
      $\overline{\Hlim}$.
\end{enumerate}
\end{corollary}

\begin{corollary}[Completed chain]
\label{cor:chain}
\[
  \underbrace{\Delta_\varepsilon
    \asymp c^{-1/2}}_{
    \text{Papers VI--VIII}}
  \Rightarrow
  \underbrace{\rho \in
    \sigma_{\rm app}(\Hlim)}_{
    \text{Paper VIII [CCM]}}
  \Rightarrow
  \underbrace{\rho \in
    \sigma(\overline{\Hlim})}_{
    \text{Paper X, unconditional}}
  \Rightarrow
  \underbrace{\mathrm{Re}(\rho)
    = \tfrac{1}{2}}_{
    \text{Cor.~\ref{cor:rh}}}.
\]
\end{corollary}

\begin{corollary}[Riemann Hypothesis,
  conditional on CCM]
\label{cor:rh}
Assume Hypothesis~CCM
(Paper~VIII, Hyp.~5.1;
\cite{CCM2025}, Thm.~1.3).
Then every non-trivial zero $\rho$
of $\zeta(s)$ satisfies
$\mathrm{Re}(\rho)=\tfrac{1}{2}$.
\end{corollary}

\begin{remark}[Final status]
\label{rem:final-status}
\[
  \underbrace{\text{Papers I--X}}_{
    \text{unconditional}}
  +\underbrace{\text{Hyp.~CCM}}_{
    \text{external}}
  \;\Longrightarrow\;
  \mathrm{Re}(\rho)=\tfrac{1}{2}.
\]
Paper~IX, Problem~10 is reclassified
as an independent open problem
of intrinsic interest, no longer
blocking any internal step.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[CCM hypothesis]
\label{prob:ccm}
Prove Hypothesis~CCM \cite{CCM2025}
independently via PSWF machinery,
making the RH argument self-contained.
\end{problem}

\begin{problem}[Sharp form rate]
\label{prob:rate}
Is $c^{-1/4}$ in
Theorem~\ref{thm:form-convergence}
sharp? The eigenvalue-gap lower bound
(open since Paper~VIII) would give
$O(c^{-3/4})$.
\end{problem}

\begin{problem}[Strong resolvent
  convergence]
\label{prob:resolvent}
Prove $(\Hc-z)^{-1}\to
(\overline{\Hlim}-z)^{-1}$ strongly.
By Remark~\ref{rem:mosco-significance},
this is a consequence of the
Mosco convergence established in
Theorem~\ref{thm:mosco},
modulo verifying condition \eqref{M1}
for the Slepian family in full generality.
\end{problem}

\begin{problem}[Completeness of
  $\{\Phi_n^{(\infty)}\}$,
  reclassified]
\label{prob:completeness}
Prove that $\{\Phi_n^{(\infty)}\}$
is complete in $L^2(\mathbb{R})$
(Paper~IX, Problem~10).
This is no longer a prerequisite
for any result in this series,
but remains open and of independent
spectral-theoretic interest.
\end{problem}
