  of $\Hlim$]{%
  Mosco--Kato Form Convergence,
  Friedrichs Extension, and
  Self-Adjointness of the PSWF
  Concentration Limit Operator}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\maketitle

\begin{abstract}
The prolate spheroidal wave function
(PSWF) concentration operators
$\Hc$ form a uniformly bounded,
monotone family of self-adjoint
integral operators on $L^2(\mathbb{R})$
that model signal concentration
in the time-frequency plane.
We prove that the strong operator
limit $\Hlim:=\lim_{c\to\infty}\Hc$
exists on all of $L^2(\mathbb{R})$,
is bounded and symmetric
(hence self-adjoint), and that the
associated quadratic forms converge
in the sense of Mosco.
The self-adjointness of $\Hlim$
follows directly from boundedness
and symmetry; the Friedrichs
representation is also given as an
alternative derivation via
closed quadratic forms.

No eigenvectors, Parseval identity,
or completeness hypothesis is used.
Conditionally on the spectral
triple construction of
Connes--Consani--Moscovici
\cite{CCM2025}, the framework
provides a necessary operator-theoretic
ingredient for spectral approaches
to the Riemann zeta function;
no unconditional statement about
zeta zeros is claimed.
\end{abstract}

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.

\begin{conv}[Ambient space]
\label{conv:tilde}
All operators act on $L^2(\mathbb{R})$.
The PSWF operator
$\mathcal{H}_c^{[0,1]}$ on $L^2([0,1])$
(\cite{Slepian1961}) is extended by
$\Hc:=P^*\mathcal{H}_c^{[0,1]}P$,
where $P$ is restriction and $P^*$
extension by zero.
$\Hc\in\mathcal{B}(L^2(\mathbb{R}))$
and $0\le\Hc\le I$.
\end{conv}

\begin{enumerate}[(IX.1)]

\item\label{IX3}
      $0\le\ip{\Hc f}{f}\le\|f\|^2$
      for all $f$, all $c$.

\item\label{IX4}
      $c\mapsto\ip{\Hc f}{f}$
      non-decreasing (Paper~V,
      Prop.~2.2).

\item\label{IX5}
      $\Dom(\Hlim)$ is a dense linear
      subspace of $L^2(\mathbb{R})$
      (Paper~IX, Thm.~2.5(a)).

\item\label{IX6}
      $\Hc$ is self-adjoint,
      positive, $\|\Hc\|\le 1$
      (\cite{Slepian1961};
      \cite{ReedSimon1975},
      Thm.~VI.1).

\item\label{IX7}
      \textit{Constructive domain.}
      $\Dom(\Hlim)
      :=\{f\in L^2(\mathbb{R})
      :\lim_{c\to\infty}\Hc f
      \text{ exists}\}$.
      This is a linear subspace
      (limits of linear maps
      are linear where they exist).
      Paper~VII, Thm.~C:
      the map $f\mapsto\lim_c\Hc f$
      is bounded on $\Dom(\Hlim)$
      with norm $\le 1$;
      $\Dom(\Hlim)$ is dense.
      In particular,
      $\|\Hc f-\Hlim f\|\to 0$
      for $f\in\Dom(\Hlim)$.
      No circularity: the domain
      is defined by existence of
      the limit, not by properties
      of the limit operator.

\item\label{IX2norm}
      \textit{Operator-norm convergence.}
      $\|\Hc-\Hlim\|
      \le C_\kappa c^{-1/4}$,
      where $\|\cdot\|$ is the
      \emph{operator norm} on
      $\mathcal{B}(L^2(\mathbb{R}))$
      (Paper~IX, Thm.~3.1;
      the stated rate uses
      Landau--Pollak asymptotics
      and holds for
      $c\ge c_0(\kappa)$).
      Used after
      Lemma~\ref{lem:hlim-bounded}.

\end{enumerate}

\begin{remark}[Objects not used]
Eigenvectors $\Phi_n^{(\infty)}$,
Parseval, completeness:
none appear in this paper.
\end{remark}

%%─── LEMMA SC
\begin{lemma}[Stable core;
  standard density-extension]
\label{lem:sc}
This is a standard density-extension
argument
(cf.\ \cite{ReedSimon1975},
Thm.~III.5); proof included
for completeness.

Let $(A_c)\subset\mathcal{B}(\mathcal{H})$
with $M:=\sup_c\|A_c\|<\infty$.
Suppose $A_c f\to Bf$ for all
$f\in\mathcal{D}$, where $\mathcal{D}$
is a dense linear subspace and
$B:\mathcal{D}\to\mathcal{H}$
is linear with $\|B\|\le M$.
Then $Af:=\lim_c A_c f$ exists for
all $f$, $A\in\mathcal{B}(\mathcal{H})$,
$\|A\|\le M$, and $A_c\to A$
strongly.
\end{lemma}

\begin{proof}
For $f\in\mathcal{H}$, $\varepsilon>0$,
choose $f_n\in\mathcal{D}$ with
$\|f-f_n\|<\varepsilon$:
$\|A_c f-A_{c'}f\|
\le 2M\varepsilon
+\|A_c f_n-A_{c'}f_n\|\to 0$,
so $(A_c f)$ is Cauchy.
$\|Af\|=\lim_c\|A_c f\|\le M\|f\|$;
linearity clear.
$\limsup_c\|A_c f-Af\|
\le 2M\varepsilon$; $\varepsilon$
arbitrary.
\end{proof}

%%─── LEMMA HLIM BOUNDED
\begin{lemma}[Existence and
  boundedness of $\Hlim$]
\label{lem:hlim-bounded}
$\Hlim f:=\lim_{c\to\infty}\Hc f$
exists for every
$f\in L^2(\mathbb{R})$,
and $\Hlim\in\mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hlim\|\le 1$.
\end{lemma}

\begin{proof}
Let $B:=\lim_c\Hc$ on
$\mathcal{D}:=\Dom(\Hlim)$.
Linearity of $B$: pointwise
limits of linear maps are linear.
Boundedness:
$\|Bf\|=\lim_c\|\Hc f\|\le\|f\|$
(\eqref{IX6}).
Apply Lemma~\ref{lem:sc}
with $A_c:=\Hc$, $M:=1$,
$\mathcal{D}$ dense (\eqref{IX5}),
$B$ as above.
\end{proof}

\begin{remark}[Logical order]
\label{rem:order}
Lemma~\ref{lem:hlim-bounded}
precedes \eqref{IX2norm},
Definition~\ref{def:qlim},
and Theorem~\ref{thm:sa}.
It depends only on
\eqref{IX5}--\eqref{IX7}
and Lemma~\ref{lem:sc}.
\end{remark}

\section{Self-Adjointness of $\Hlim$}
\label{sec:sa}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Self-adjointness]
\label{thm:sa}
$\Hlim$ is bounded, symmetric,
and self-adjoint on $L^2(\mathbb{R})$
with $\sigma(\Hlim)\subset[0,1]$.
\end{theorem}

\begin{proof}
\textbf{Boundedness.}
Lemma~\ref{lem:hlim-bounded}:
$\Hlim\in\mathcal{B}(L^2(\mathbb{R}))$,
$\|\Hlim\|\le 1$.

\textbf{Symmetry.}
For $f,g\in L^2(\mathbb{R})$:
\[
  \ip{\Hlim f}{g}
  = \lim_c\ip{\Hc f}{g}
  = \lim_c\ip{f}{\Hc g}
  = \ip{f}{\Hlim g},
\]
using Cauchy--Schwarz to exchange
limit and inner product,
and self-adjointness of each
$\Hc$ (\eqref{IX6}).

\textbf{Self-adjointness.}
A bounded symmetric operator on
a Hilbert space is automatically
self-adjoint
(\cite{ReedSimon1975}, Thm.~III.6;
cf.\ \cite{Kato1966}, Thm.~III.2.1).
Hence $\Hlim = \Hlim^*$.

\textbf{Spectrum.}
$0\le\ip{\Hlim f}{f}\le\|f\|^2$
for all $f$ (\eqref{IX3},
Lemma~\ref{lem:hlim-bounded}),
so $\sigma(\Hlim)\subset[0,1]$.
\end{proof}

\begin{remark}[Rôle of Friedrichs]
Since $\Hlim\in\mathcal{B}(L^2)$
is already self-adjoint,
the Friedrichs construction
(Section~\ref{sec:friedrichs})
is not needed for self-adjointness
per se.
It is retained because it
(a)~provides the quadratic-form
identification
$T_{\qlim}=\Hlim$,
which is used in the spectral
chain (Section~\ref{sec:consequences}),
and (b)~gives an independent
derivation via
Mosco $\Rightarrow$ closed form
$\Rightarrow$ unique self-adjoint
operator.
\end{remark}

\section{Limit Form}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
For $f\in L^2(\mathbb{R})$, set
$\qlim(f,f)
:=\lim_{c\to\infty}\ip{\Hc f}{f}
=\ip{\Hlim f}{f}$
(the equality from
Lemma~\ref{lem:hlim-bounded}).
$\Dom(\qlim)=L^2(\mathbb{R})$;
extend by polarisation.
\end{definition}

\begin{theorem}[Existence]
\label{thm:qlim-exists}
$\qlim\ge 0$ is a bounded Hermitian
form on $L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Monotonicity \eqref{IX4} and
bound \eqref{IX3}.
\end{proof}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence]
\label{def:mosco}
$(\qc)$ converge to $\qlim$
in the sense of Mosco
\cite{Mosco1969}:
\begin{enumerate}[\rm(M1)]
\item For every $c_n\to\infty$
      and $f_n\rightharpoonup f$:
      $\qlim(f,f)
      \le\liminf_n q_{c_n}(f_n,f_n)$.
\item For every $f$ there exist
      $c_n\to\infty$ and
      $f_n\to f$ strongly with
      $q_{c_n}(f_n,f_n)
      \to\qlim(f,f)$.
\end{enumerate}
Both $c_n$ and $(f_n)$ share
index $n$ (standard Mosco;
cf.\ \cite{Mosco1969}).
\end{definition}

\begin{lemma}[HS and weak continuity]
\label{lem:hs}
For each fixed $c_0>0$:
\begin{enumerate}[\rm(a)]
\item $\widetilde{\mathcal{H}}_{c_0}
      \in\mathcal{B}_2(L^2(\mathbb{R}))$
      (compact).
\item $f_n\rightharpoonup f
      \Rightarrow
      \ip{\widetilde{\mathcal{H}}_{c_0}
        f_n}{f_n}
      \to
      \ip{\widetilde{\mathcal{H}}_{c_0}
        f}{f}$.
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a).}
$\|K_{c_0}\|_{\infty}\le c_0/\pi$
on $[0,1]^2$;
HS $\Rightarrow$ compact
(\cite{ReedSimon1975}, VI.22).

\textit{(b).}
Write $I+II$: $II\to 0$ (weak);
$|I|\le\sup_n\|f_n\|
\cdot\|\widetilde{\mathcal{H}}_{c_0}
(f_n-f)\|\to 0$
(uniform bound: III.5;
compactness: VI.11;
both \cite{ReedSimon1975}).
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc\to\qlim$ in Mosco sense.
\end{theorem}

\begin{proof}
\textit{(M2):}
$f_n:=f$, $c_n\to\infty$:
$q_{c_n}(f,f)\to\ip{\Hlim f}{f}
=\qlim(f,f)$
(Lemma~\ref{lem:hlim-bounded}).

\textit{(M1):}
Let $c_n\to\infty$,
$f_n\rightharpoonup f$.
Fix $c_0>0$;
since $c_n\to\infty$,
$c_n\ge c_0$ for large $n$,
so monotonicity \eqref{IX4}
gives
$q_{c_n}(f_n,f_n)
\ge q_{c_0}(f_n,f_n)$.
Lemma~\ref{lem:hs}(b):
$\liminf_n q_{c_n}(f_n,f_n)
\ge q_{c_0}(f,f)$.
Send $c_0\to\infty$.
\end{proof}

\begin{remark}[Mosco $\Rightarrow$
  resolvent convergence]
\cite{KuwaeShioya2003}, Thm.~2.4:
Mosco $\Rightarrow$ strong resolvent
convergence;
see Problem~\ref{prob:resolvent}.
\end{remark}

\section{Closedness}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness]
\label{thm:form-closed}
$\qlim\ge 0$ and $\qlim$ is closed.
\end{theorem}
\begin{proof}
Let $(f_k)$ be
$\|\cdot\|_{\qlim}$-Cauchy;
$f_k\to f$ in $L^2$
(hence weakly).
Apply (M1) with the constant
sequence $c_n:=c$
(valid: the $\liminf$ argument
holds for any fixed $c$):
$q_c(f,f)
\le\liminf_k q_c(f_k,f_k)$.
Send $c\to\infty$;
parallelogram law gives
$\qlim(f_k-f,f_k-f)\to 0$
(\cite{Kato1966}, Def.~VI.1.3).
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative rate]
\label{thm:form-convergence}
For all $f\in L^2(\mathbb{R})$,
$c\ge c_0(\kappa)$:
\[
  0\le\qlim(f,f)-\qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\]
\end{theorem}
\begin{proof}
Lower: monotonicity \eqref{IX4}.
Upper: $\ip{(\Hlim-\Hc)f}{f}
\le\|\Hlim-\Hc\|\cdot\|f\|^2
\le C_\kappa c^{-1/4}\|f\|^2$
by \eqref{IX2norm}
(operator-norm bound).
\end{proof}

\section{Friedrichs Extension
  (Alternative Derivation)}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

This section gives an independent
derivation of $T_{\qlim}=\Hlim$
via the Kato--Friedrichs
representation theorem,
using Mosco convergence and
closedness from
Sections~\ref{sec:mosco}--\ref{sec:form-closed}.
Self-adjointness of $\Hlim$
follows independently from
Theorem~\ref{thm:sa}.

\begin{theorem}[Kato representation]
\label{thm:kato-rep}
A densely defined, closed,
semibounded form $q$ determines a
unique self-adjoint $T$,
$\Dom(T)\subset\Dom(q)$,
with $q(f,g)=\ip{Tf}{g}$.
(\cite{Kato1966}, VI.2.1;
\cite{ReedSimon1975}, VIII.15.)
\end{theorem}

\begin{theorem}[Form
  identification]
\label{thm:friedrichs}
$T_{\qlim}=\Hlim$.
\end{theorem}

\begin{proof}
By Theorems~\ref{thm:qlim-exists}
and \ref{thm:form-closed},
$\qlim$ is a closed semibounded
form on all of $L^2(\mathbb{R})$.
Apply Theorem~\ref{thm:kato-rep}
to get $T_{\qlim}$.

\textbf{Step~1}
($\Hlim\subset T_{\qlim}$).
Let $f\in L^2(\mathbb{R})$ and
$g\in L^2(\mathbb{R})$.
Since $\Hc f\to\Hlim f$ strongly,
Cauchy--Schwarz gives
$\ip{\Hc f}{g}\to\ip{\Hlim f}{g}$.
Also $\ip{\Hc f}{g}
=\qc(f,g)\to\qlim(f,g)$
(Definition~\ref{def:qlim}).
Hence $\ip{\Hlim f}{g}
=\qlim(f,g)$ for all $g$,
so $T_{\qlim}=\Hlim$.

\textbf{Consequence.}
$T_{\qlim}=\Hlim$ is bounded
and self-adjoint
(Theorem~\ref{thm:sa}),
consistent with uniqueness in
Theorem~\ref{thm:kato-rep}.
\end{proof}

\begin{remark}[Audit]
Every step cites one of:
\eqref{IX3}--\eqref{IX7},
Kato~VI.2.1,
Lemmas~\ref{lem:sc},
\ref{lem:hlim-bounded},
\ref{lem:hs}.
No eigenvector, Parseval,
or completeness is used.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Spectral inclusion]
\label{cor:spec}
$\sigma(\Hlim)\subset[0,1]\subset\mathbb{R}$.
\end{corollary}

\begin{problem}[Open gap:
  spectral embedding]
\label{gap:spectral}
$\sigma(\Hlim)\subset\mathbb{R}$.
Non-trivial zeta zeros
$\rho=\tfrac12+it$ ($t\ne0$)
are complex.
Two sub-gaps remain:
\begin{enumerate}[\rm(G1)]
\item Why $\rho\in
      \sigma_{\mathrm{app}}(\Hlim)$?
      (Requires CCM or further
      PSWF spectral theory.)
\item How is $\rho$ identified with
      a real spectral parameter?
      (CCM uses $\mathrm{Im}(\rho)$
      as eigenvalue of a real
      self-adjoint operator;
      external to this paper.)
\end{enumerate}
\end{problem}

\begin{corollary}[Conditional RH]
\label{cor:rh}
This paper does not provide new
evidence for the Riemann Hypothesis.
It provides an operator-theoretic
setting (existence and
self-adjointness of $\Hlim$,
Mosco convergence of $(\qc)$)
that is a necessary ingredient
of spectral approaches to the
Riemann zeta function.

Conditionally on Hypothesis~CCM
(\cite{CCM2025}, Thm.~1.3)
and the resolution of both gaps
in Open Gap~\ref{gap:spectral},
every non-trivial zero $\rho$
of $\zeta(s)$ satisfies
$\operatorname{Re}(\rho)=\tfrac12$.
\end{corollary}

\begin{remark}[Unconditional results]
Self-adjointness of $\Hlim$,
boundedness $\|\Hlim\|\le 1$,
$T_{\qlim}=\Hlim$,
$\qc\xrightarrow{\mathrm{Mosco}}\qlim$
at rate $O(c^{-1/4})$.
\end{remark}
