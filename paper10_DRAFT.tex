\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of compact
self-adjoint operators on $L^2(\mathbb{R})$,
generated by spectral truncations.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent
stability (unconditional).}
The starting point is
Lemma~\ref{lem:ftb} (Slepian--Pollak
\cite{Slepian1961}), which identifies
$\Hc = P^* \Bc P$ where $\Bc = \Fc^{-1}\chi_{[-c,c]}\Fc$
is the Fourier band projection.
From this single identity, operator
monotonicity, Hilbert--Schmidt compactness,
and density of the strong-convergence domain
$\mathcal{D}$ are derived
(Lemmas~\ref{lem:monotone}--\ref{lem:dense}).
These three lemmas are the complete
foundation for proving Mosco convergence
$\qc \xrightarrow{\mathrm{M}} \qlim$,
closedness of $\qlim$, and strong
resolvent convergence
$(\Hc-z)^{-1} \xrightarrow{\mathrm{s}}
(\Hlim-z)^{-1}$.
A quantitative rate $O(c^{-1/4})$ is
stated as an external theorem
(Theorem~\ref{thm:rate-ext}, citing Paper~VIII)
without independent proof.

\textbf{(B) Bridge theorem (unconditional).}
The strong-limit operator $\Hstrong$
and the spectral-closure operator
$\overline{\Hspectral}$ are identical
as bounded operators on $L^2(\mathbb{R})$.

The conditional connection to zeros of
$\zeta(s)$ (Remark~\ref{rem:rh})
holds under two unverified hypotheses;
no unconditional claim about zeta zeros
is made.
\end{abstract}

\section{Setup and Foundational Lemmas}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

Let $\Fc$ denote the Fourier transform,
$(\Fc f)(\xi) =
\int_{-\infty}^\infty f(x) e^{-i\xi x}\,dx$,
with inverse $\Fc^{-1}$.
All operators act on $L^2(\mathbb{R})$
unless stated otherwise.

\begin{conv}[Operators and notation]
\label{conv:ops}
Define:
\begin{alignat}{2}
  \Bc &:= \Fc^{-1}\chi_{[-c,c]}\Fc
  &&\in \mathcal{B}(L^2(\mathbb{R})),
  \text{ the Fourier band projection,}
  \\
  P &: L^2(\mathbb{R}) \to L^2([0,1]),
  &&\text{ restriction (bounded, $\|P\|=1$),}
  \\
  P^* &: L^2([0,1]) \to L^2(\mathbb{R}),
  &&\text{ extension by zero (bounded, $\|P^*\|=1$),}
  \\
  \mathcal{H}_c^{[0,1]}
  &: L^2([0,1]) \to L^2([0,1]),
  &&\text{ the PSWF concentration operator,}
\end{alignat}
with kernel
$K_c(x,y) :=
\frac{\sin c(x-y)}{\pi(x-y)}$
(value $c/\pi$ at $x=y$).
The extension to $L^2(\mathbb{R})$:
$\Hc := P^*\mathcal{H}_c^{[0,1]}P$.
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
$\Hc$ is self-adjoint on $L^2(\mathbb{R})$,
$0 \le \Hc \le I$,
$\|\Hc\| \le 1$
(\cite{Slepian1961};
\cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[Scope]
\label{rem:scope}
Sections~\ref{sec:setup}--\ref{sec:friedrichs}
use only \eqref{F1} and
Lemmas~\ref{lem:ftb}--\ref{lem:dense}.
No limit eigenvalues, no limit
eighenvectors, no operator differences,
no results from Paper~VIII or IX.
\end{remark}

% THE KEY STRUCTURAL LEMMA: Fourier-Time-Band Identity
% ======================================================
%
\begin{lemma}[Fourier--time--band
  representation]
\label{lem:ftb}
For all $c > 0$ and all $f \in L^2(\mathbb{R})$:
\begin{equation}\label{eq:ftb}
  \ip{\Hc f}{f}
  = \|\Bc(Pf)\|_{L^2(\mathbb{R})}^2,
\end{equation}
where $Pf \in L^2([0,1])$ is identified
with its zero extension in $L^2(\mathbb{R})$.
Equivalently, as operators on $L^2([0,1])$:
\begin{equation}\label{eq:ftb-op}
  \mathcal{H}_c^{[0,1]}
  = P\, \Bc\, P^*.
\end{equation}
\end{lemma}

\begin{proof}
Let $g = Pf \in L^2([0,1])$,
identified with its zero extension
$\tilde{g} \in L^2(\mathbb{R})$.
By the convolution theorem
(Fourier inversion + Plancherel):
\begin{align*}
  (\mathcal{H}_c^{[0,1]} g)(x)
  &= \int_0^1 K_c(x,y) g(y)\,dy
  = \int_{-\infty}^\infty
    K_c(x,y) \tilde{g}(y)\,dy
    \quad (\tilde{g}\ \text{zero outside}\ [0,1])\\
  &= (K_c * \tilde{g})(x)
  \quad \text{(convolution on }\mathbb{R}),
\end{align*}
where the convolution is restricted
to $x \in [0,1]$.
Now $K_c(u) = \Fc^{-1}[
\tfrac{1}{2\pi}\chi_{[-c,c]}](u)$
(standard Fourier pair;
\cite{Slepian1961}, Eq.~(3)).
By the convolution theorem:
\[
  \Fc(K_c * \tilde{g})(\xi)
  = 2\pi \cdot \tfrac{1}{2\pi}
    \chi_{[-c,c]}(\xi)\cdot(\Fc\tilde{g})(\xi)
  = \chi_{[-c,c]}(\xi)(\Fc\tilde{g})(\xi).
\]
Hence $K_c * \tilde{g} =
\Fc^{-1}[\chi_{[-c,c]}\Fc\tilde{g}]
= \Bc\tilde{g}$.
Restricting to $x \in [0,1]$:
\[
  \mathcal{H}_c^{[0,1]} g = P(\Bc P^* g)
  \quad \text{for all } g \in L^2([0,1]),
\]
which is \eqref{eq:ftb-op}.
Applying $P^*$ and taking inner product:
\[
  \ip{\Hc f}{f}
  = \ip{P^*\mathcal{H}_c^{[0,1]}Pf}{f}
  = \ip{\mathcal{H}_c^{[0,1]}Pf}{Pf}
  = \ip{P\Bc P^*Pf}{Pf}
  = \ip{\Bc \tilde{g}}{\tilde{g}}
  = \|\Bc\tilde{g}\|^2,
\]
where the last step uses
$\Bc = \Bc^* = \Bc^2$
(orthogonal projection;
\cite{ReedSimon1975}, Ch.~VIII.1).
This is \eqref{eq:ftb}.
\end{proof}

\begin{remark}[What Lemma~\ref{lem:ftb}
  provides]
\label{rem:ftb}
Lemma~\ref{lem:ftb} is the
\emph{only} point where
Slepian--Pollak PSWF theory is used
in the form thread.
All subsequent lemmas are
pure functional analysis.
The identity \eqref{eq:ftb}
reduces $\Hc$ to the composition
\[
  \Hc = P^* P \Bc P^* P,
\]
i.e.\ a band projection $\Bc$
conjugated by time-restriction.
This is the classical
``double commutant'' structure
of Slepian--Pollak.
\end{remark}

% ---- Monotonicity: now one line from lem:ftb ----

\begin{lemma}[Operator monotonicity]
\label{lem:monotone}
For $0 < c \le c'$ and all $f$:
$\ip{\Hc f}{f} \le
\ip{\mathcal{H}_{c'} f}{f}$.
In particular, $c \mapsto \ip{\Hc f}{f}$
is non-decreasing, bounded by $\|f\|^2$.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:ftb}:
$\ip{\Hc f}{f} = \|\Bc(Pf)\|^2$
and
$\ip{\mathcal{H}_{c'} f}{f}
= \|B_{c'}(Pf)\|^2$.
Since $[-c,c] \subset [-c',c']$,
the orthogonal projections satisfy
$\Bc \le B_{c'}$
(projection order;
\cite{ReedSimon1975}, Ch.~VIII.1);
hence $\|\Bc h\|^2 \le \|B_{c'} h\|^2$
for all $h$.
Apply with $h = Pf$.
\end{proof}

% ---- HS: rebuilt on Parseval of Bc ----

\begin{lemma}[Hilbert--Schmidt and
  compactness]
\label{lem:hs-compact}
For each fixed $c_0 > 0$,
$\Hc[c_0]$ is compact on $L^2(\mathbb{R})$.
\end{lemma}

\begin{proof}
We show $\mathcal{H}_{c_0}^{[0,1]}$
is Hilbert--Schmidt on $L^2([0,1])$;
compactness of $\Hc[c_0]
= P^*\mathcal{H}_{c_0}^{[0,1]}P$
then follows since $P,P^*$ are bounded
(\cite{ReedSimon1975}, Thm.~VI.22).

\textit{HS norm.}
By \eqref{eq:ftb-op},
$\mathcal{H}_{c_0}^{[0,1]}
= P \Bc[c_0] P^*$.
The HS norm satisfies:
\[
  \|\mathcal{H}_{c_0}^{[0,1]}\|_{\mathrm{HS}}^2
  = \|K_{c_0}\|_{L^2([0,1]^2)}^2
  = \int_0^1\!\int_0^1
    |K_{c_0}(x-y)|^2\,dx\,dy.
\]
Substitute $u = x-y$.
By Fubini (integrand non-negative)
and the Jacobian-one change of variables:
\[
  \int_0^1\!\int_0^1
  |K_{c_0}(x-y)|^2\,dx\,dy
  \le \int_{-1}^{1}(1-|u|)|K_{c_0}(u)|^2
  \,du
  \le \int_{-1}^{1}|K_{c_0}(u)|^2\,du.
\]
By Parseval
($K_{c_0}(u)
= \Fc^{-1}[\tfrac{1}{2\pi}\chi_{[-c_0,c_0]}](u)$):
\[
  \int_{-\infty}^{\infty}|K_{c_0}(u)|^2\,du
  = \frac{1}{2\pi}\int_{-c_0}^{c_0}1\,d\xi
  = \frac{c_0}{\pi}.
\]
Hence
$\|\mathcal{H}_{c_0}^{[0,1]}\|_{\mathrm{HS}}^2
\le c_0/\pi < \infty$,
so $\mathcal{H}_{c_0}^{[0,1]}$ is HS,
hence compact
(\cite{ReedSimon1975}, Thm.~VI.22).
\end{proof}

% ---- Density: via operator bound, no Riemann-Lebesgue ----

\begin{lemma}[Density of $\mathcal{D}$]
\label{lem:dense}
Let $\mathcal{D} := \{f \in L^2(\mathbb{R})
: \lim_{c\to\infty}\Hc f
\text{ exists in } L^2\}$.
Then $\mathcal{D}$ is dense.
\end{lemma}

\begin{proof}
It suffices to show
$C_c^\infty(\mathbb{R}) \subset \mathcal{D}$.

Let $f \in C_c^\infty$,
$\tilde{g} := P^*(Pf)
= \mathbf{1}_{[0,1]} f$
(zero extension of $Pf$).
By Lemma~\ref{lem:ftb}:
\[
  \Hc f = P^*(\mathcal{H}_c^{[0,1]}Pf)
  = P^*(P \Bc P^* Pf)
  = P^* P (\Bc \tilde{g}).
\]
We show $\Hc f \to P^*P\tilde{g}
= \mathbf{1}_{[0,1]}\tilde{g}$
in $L^2$.

Since $\tilde{g} \in L^2(\mathbb{R})$
and $\Bc$ is the Fourier projection
onto $\{\hat{h}\ \mathrm{supp} \subset
[-c,c]\}$:
\[
  \Bc\tilde{g}
  \xrightarrow{\mathrm{s}} \tilde{g}
  \quad \text{as } c \to \infty
\]
strongly in $L^2(\mathbb{R})$
(since $\Bc$ increases to the identity:
$\Bc = \Fc^{-1}\chi_{[-c,c]}\Fc
\nearrow I$ strongly by
Plancherel + dominated convergence
on the Fourier side;
\cite{ReedSimon1975}, Thm.~IX.14).

Hence, using $\|P^*P\| \le 1$:
\begin{align*}
  \|\Hc f - P^*P\tilde{g}\|_{L^2}
  &= \|P^*P(\Bc\tilde{g} - \tilde{g})\|_{L^2}
  \le \|\Bc\tilde{g} - \tilde{g}\|_{L^2}
  \to 0,
\end{align*}
so $f \in \mathcal{D}$.
\emph{Uniform domination in $c$:
the bound $\|\Hc f\| \le \|f\|$
from \eqref{F1} shows the family
$(\Hc f)_{c>0}$ is uniformly bounded;
no pointwise argument is needed.}
\end{proof}

\begin{remark}[Three lemmas from one]
\label{rem:setup-summary}
Lemmas~\ref{lem:monotone}--\ref{lem:dense}
all follow from the single
identity \eqref{eq:ftb}.
Monotonicity: projection order.
Compactness: Parseval norm bound.
Density: strong convergence of $\Bc \to I$.
This is a template argument for any
family of compact self-adjoint operators
generated by spectral truncations.
\end{remark}

\section{Strong-Limit Operator}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Density-extension principle]
\label{lem:sc}
Let $(A_c) \subset \mathcal{B}(\mathcal{H})$
with $\sup_c\|A_c\| \le M$;
$\mathcal{D} \subset \mathcal{H}$ dense;
$A_c f \to Bf$ for all $f \in \mathcal{D}$.
Then $A_c \to A$ strongly,
$A \in \mathcal{B}(\mathcal{H})$,
$\|A\| \le M$.
\end{lemma}
\begin{proof}
$3\varepsilon$ argument;
\cite{ReedSimon1975}, Thm.~I.26.
\end{proof}

\begin{lemma}[Existence of $\Hstrong$]
\label{lem:hlim-bounded}
$\exists!\,\Hstrong \in
\mathcal{B}(L^2)$ with $\|\Hstrong\| \le 1$
and $\Hc \to \Hstrong$ strongly.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:sc} with
$\mathcal{D}$ from Lemma~\ref{lem:dense},
$M = 1$.
\end{proof}

\begin{corollary}[Form equals operator]
\label{cor:qlim-op}
$\qlim(f,f) = \ip{\Hstrong f}{f}$.
\end{corollary}
\begin{proof}
Strong convergence + inner product
continuity.
\end{proof}

\begin{corollary}[Self-adjointness]
\label{cor:sa-strong}
$\Hstrong = \Hstrong^* \ge 0$;
$\sigma(\Hstrong) \subset [0,1]$.
\end{corollary}
\begin{proof}
Symmetry by strong limit;
BSA (\cite{ReedSimon1975}, Thm.~III.6);
positivity from $\qlim \ge 0$.
\end{proof}

\section{Limit Form and Mosco
  Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
$\qlim(f,f) :=
\sup_{c>0}\ip{\Hc f}{f}
= \lim_{c\to\infty}\ip{\Hc f}{f}$
(exists by Lemma~\ref{lem:monotone}
and \eqref{F1});
extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc) \xrightarrow{\mathrm{M}} \qlim$:
\begin{enumerate}[\rm(M1)]
\item $c_n\to\infty$,
      $f_n\rightharpoonup f$
      $\Rightarrow$
      $\qlim(f,f) \le
      \liminf_n q_{c_n}(f_n,f_n)$.
\item $\forall f\,\exists c_n\to\infty,
      f_n\to f$:
      $q_{c_n}(f_n,f_n) \to \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Complete continuity]
\label{lem:wc}
For fixed $c_0$:
$f_n \rightharpoonup f
\Rightarrow
q_{c_0}(f_n,f_n) \to q_{c_0}(f,f)$.
\end{lemma}
\begin{proof}
$\Hc[c_0]$ compact
(Lemma~\ref{lem:hs-compact}),
hence completely continuous:
$\Hc[c_0] f_n \xrightarrow{s}
\Hc[c_0] f$
(\cite{ReedSimon1975}, Thm.~VI.1).
Then inner product continuity
(\cite{ReedSimon1975}, Lem.~II.3).
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc \xrightarrow{\mathrm{M}} \qlim$.
\end{theorem}
\begin{proof}
\textit{(M2).}
$f_n := f$, $c_n \to \infty$:
definition + Lemma~\ref{lem:monotone}.

\textit{(M1).}
$c_n\to\infty$, $f_n\rightharpoonup f$.
For fixed $c_0$, once $c_n \ge c_0$:
$q_{c_n}(f_n,f_n) \ge q_{c_0}(f_n,f_n)$
(Lemma~\ref{lem:monotone}).
Hence
$\liminf_n q_{c_n}(f_n,f_n)
\ge q_{c_0}(f,f)$
(Lemma~\ref{lem:wc}).
Taking $\sup_{c_0}$:
$\ge \qlim(f,f)$.
\end{proof}

\begin{theorem}[Closedness]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
$\Hstrong$ bounded
$\Rightarrow$ form norm $\sim$ $L^2$-norm.

\end{proof}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
For all $f$ and $c \ge c_0(\kappa)$:
$0 \le \qlim(f,f) - \qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{theorem}
\begin{proof}[Proof (external)]
Lower bound: Lemma~\ref{lem:monotone}.
Upper bound: Paper~VIII, Cor.~4.3:
$\|\Hstrong - \Hc\| \le C_\kappa c^{-1/4}$;
hence $\ip{(\Hstrong-\Hc)f}{f}
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{proof}

\begin{remark}[Obstruction to
  independent proof]
\label{rem:no-indep-proof}
Any independent proof requires
uniform eigenvector asymptotics
as $c$ varies;
representations mixing fixed-$c$
basis with $c\to\infty$ weights
are not valid.
See Problem~\ref{prob:rate}.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the \emph{only}
section that imports Paper~IX results.

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      Paper~IX, Def.~2.3:
      $\Hspectral f :=
      \sum_n \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}}
      \Phi_n^{(\infty)}$,
      densely defined, symmetric,
      closable.

\item\label{IXb}
      \textit{Bounded, everywhere-defined
      closure.}
      On $\Dom(\Hspectral)$:
      $\|\Hspectral f\|^2
      = \sum_n (\lambda_n^{(\infty)})^2
      |\ip{f}{\Phi_n^{(\infty)}}|^2
      \le \|f\|^2$
      (Bessel + $\lambda_n^{(\infty)}
      \in[0,1]$).
      Hence
      $\overline{\Hspectral}
      \in \mathcal{B}(L^2)$,
      $\|\overline{\Hspectral}\| \le 1$,
      everywhere defined
      (\cite{ReedSimon1975}, Thm.~I.7).

\item\label{IXc}
      \textit{Agreement on $\mathcal{D}$.}
      Both $\Hstrong$ and
      $\overline{\Hspectral}$
      are bounded, everywhere defined.
      For $f \in \mathcal{D}$:
      $\Hc f \to \Hstrong f$
      (Lemma~\ref{lem:hlim-bounded});
      $\Hc f \to \overline{\Hspectral}f$
      (Paper~IX, Rem.~2.6).
      Uniqueness:
      $\Hstrong = \overline{\Hspectral}$
      on $\mathcal{D}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong = \overline{\Hspectral}$
in $\mathcal{B}(L^2(\mathbb{R}))$.
\end{theorem}
\begin{proof}
Both bounded, everywhere defined
(\eqref{IXb}, Lemma~\ref{lem:hlim-bounded}).
Agree on dense $\mathcal{D}$ (\eqref{IXc}).
Apply \cite{ReedSimon1975}, Lem.~II.3.
\end{proof}

\begin{remark}
$\Hlim :=
\Hstrong = \overline{\Hspectral}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every $z \notin [0,1]$:
$(\Hc - z)^{-1}
\xrightarrow{\mathrm{s}}
(\Hlim - z)^{-1}$.
\end{corollary}
\begin{proof}
Mosco (Theorem~\ref{thm:mosco}),
$\sigma(\Hlim) \subset [0,1]$
(Corollary~\ref{cor:sa-strong}),
\cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
Without hypothesis:
$\Hlim = \Hlim^* \ge 0$;
$\|\Hlim\| \le 1$;
$T_{\qlim} = \Hlim$;
$\qc \xrightarrow{\mathrm{M}} \qlim$;
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{s}} (\Hlim-z)^{-1}$;
$\Hstrong = \overline{\Hspectral}$.
External (Paper~VIII):
rate $O(c^{-1/4})$.
\end{remark}

\begin{remark}[Conditional connection
  to the Riemann hypothesis]
\label{rem:rh}
This remark records a formal implication
under two \emph{unverified}
hypotheses:
(H1) Paper~IX, Hypothesis~4.1:
$\{\Phi_n^{(\infty)}\}$ complete;
$\Hlim$ essentially self-adjoint.
(H2) \cite{CCM2025}, Thm.~1.3:
non-trivial zeros $\rho$ of $\zeta(s)$
lie in $\sigma_{\mathrm{app}}(\Hlim)$.
Under (H1)--(H2):
$\rho \in \sigma(\Hlim) \subset \mathbb{R}$,
forcing $\mathrm{Re}(\rho) = 1/2$.
\emph{No unconditional claim.
Neither (H1) nor (H2) is proved here.}
\end{remark}

\begin{problem}[Completeness]
\label{prob:completeness}
Prove Paper~IX, Hypothesis~4.1.

\begin{problem}[Independent rate proof]
\label{prob:rate}
Prove $\qlim(f,f) - \qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$
without Paper~VIII;
see Remark~\ref{rem:no-indep-proof}.

\label{prob:resolvent}
Find $F(c,z) \to 0$ with
$\|(\Hc-z)^{-1}-(\Hlim-z)^{-1}\|
\le F(c,z)$.
\end{problem}

\label{prob:spectrum}
Is $\sigma(\Hlim) = [0,1]$?
\end{problem}
