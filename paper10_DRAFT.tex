  Mosco--Kato Form Convergence,
  Friedrichs Extension, and Essential
  Self-Adjointness of the PSWF
  Concentration Limit Operator}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\maketitle

\begin{abstract}
We prove the essential self-adjointness
of the PSWF concentration limit operator
$\Hlim$ via Mosco form convergence and
Friedrichs extension.
The limit form
$\qlim(f,f):=\lim_{c\to\infty}\ip{\Hc f}{f}$
is defined purely from the operator sequence;
no eigenvectors, Parseval identity,
or completeness hypothesis appear anywhere.

Two structural sub-lemmas are established:
\begin{itemize}
\item \textbf{Lemma~\ref{lem:hs}:}
      $\mathcal{H}_c$ is Hilbert--Schmidt.
      Mosco condition (M1) is closed via
      an explicit three-term estimate,
      with Term~II vanishing by weak
      convergence and Term~I vanishing
      by compactness and Banach--Steinhaus.
      No assumption of uniform norm-bound
      is pre-imposed.
\item \textbf{Lemma~\ref{lem:core}:}
      $\Dom(\Hlim)$ is a core for
      $T_{\qlim}$.
      Since $T_{\qlim}\in\mathcal{B}(L^2)$,
      form-norm and graph-norm on
      $\Dom(T_{\qlim})$ are equivalent;
      core density therefore reduces to
      form-norm density, which is proved
      via Yosida regularisation and
      an explicit $\varepsilon^2$-estimate.
\end{itemize}

\textbf{Main results.}
(1)~$\qlim$ exists on a dense domain
    and is closed
    (Theorems~\ref{thm:qlim-exists},
    \ref{thm:form-closed}).
(2)~$\qc\to\qlim$ in the sense of
    Mosco with rate $O(c^{-1/4})$
    (Theorems~\ref{thm:mosco},
    \ref{thm:form-convergence}).
(3)~$T_{\qlim}=\overline{\Hlim}$;
    $\Hlim$ is essentially self-adjoint
    (Theorem~\ref{thm:friedrichs}).
(4)~Under Hypothesis~CCM \cite{CCM2025},
    all non-trivial zeros $\rho$ of
    $\zeta(s)$ satisfy
    $\mathrm{Re}(\rho)=\tfrac{1}{2}$
    (Corollary~\ref{cor:rh}).
\end{abstract}

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.

\begin{enumerate}[(IX.1)]
\item\label{IX1}
      $\Hlim$ is symmetric and densely
      defined on $\Dom(\Hlim)$
      (Paper~IX, Def.~2.3, Thm.~2.5).

\item\label{IX2}
      $\Hlim\in\mathcal{B}(L^2(\mathbb{R}))$
      with $\|\Hlim\|\le 1$, and
      $\|\Hc-\Hlim\|_{L^2\to L^2}
      \le C_\kappa c^{-1/4}$
      (Paper~IX, Thm.~3.1).

      \textit{Boundedness is a theorem,
      not an assumption.}
      Each $\Hc$ satisfies $\|\Hc\|\le 1$
      by \eqref{IX6}.
      The sequence $(\Hc)_{c>0}$ converges
      strongly on $\Dom(\Hlim)$
      (\eqref{IX7}), which is dense
      (\eqref{IX5}).
      By the Uniform Boundedness Principle
      (\cite{ReedSimon1975}, Thm.~III.5),
      the pointwise limit
      $\Hlim=\lim_c\Hc$ satisfies
      $\|\Hlim\|\le\sup_c\|\Hc\|\le 1$
      and is bounded on all of
      $L^2(\mathbb{R})$.
      In particular, $\Hlim f$ is defined
      for every $f\in L^2(\mathbb{R})$.

\item\label{IX3}
      $0\le\ip{\Hc f}{f}\le\|f\|^2$
      for all $f$, all $c$.

\item\label{IX4}
      $c\mapsto\ip{\Hc f}{f}$
      is monotone non-decreasing
      (Paper~V, Prop.~2.2).

\item\label{IX5}
      $\Dom(\Hlim)$ is dense in
      $L^2(\mathbb{R})$
      (Paper~IX, Thm.~2.5(a)).

\item\label{IX6}
      $\Hc$ is an integral operator on
      $L^2([0,1])$ with symmetric real
      kernel
      $K_c(x,y)=\tfrac{\sin c(x-y)}{\pi(x-y)}$
      (Slepian--Pollak \cite{Slepian1961}).
      Hence $\Hc$ is self-adjoint on
      $L^2([0,1])$.
      For a positive self-adjoint
      operator,
      $\|\Hc\|
      =\sup_{\|f\|=1}\ip{\Hc f}{f}$
      (\cite{ReedSimon1975}, Thm.~VI.1).
      Together with \eqref{IX3} this gives
      $\|\Hc\|\le 1$.

\item\label{IX7}
      $\|\Hc f-\Hlim f\|\to 0$
      for every $f\in\Dom(\Hlim)$
      (Paper~VII, Thm.~C).
\end{enumerate}

\begin{remark}[Objects not used]
Eigenvectors $\Phi_n^{(\infty)}$,
Parseval identity, completeness of
$\{\Phi_n^{(\infty)}\}$, and
Paper~IX, Problem~10 do not appear
anywhere in this paper.
\end{remark}

\section{Limit Form}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit quadratic form]
\label{def:qlim}
For $f\in L^2(\mathbb{R})$:
\[
  \qlim(f,f)
  :=\lim_{c\to\infty}\ip{\Hc f}{f}
  \in[0,\infty];
\]
$\Dom(\qlim):=\{f:\qlim(f,f)<\infty\}$,
extended by polarisation.
\end{definition}

\begin{theorem}[Existence, density]
\label{thm:qlim-exists}
(a) The limit exists for all
    $f\in L^2(\mathbb{R})$.
(b) $\Dom(\Hlim)\subset\Dom(\qlim)$;
    $\Dom(\qlim)$ is dense.
(c) $\qlim$ is Hermitian sesquilinear.
\end{theorem}
\begin{proof}
(a) Monotone \eqref{IX4}, bounded
\eqref{IX3}.
(b) $f\in\Dom(\Hlim)$:
$\qlim(f,f)=\ip{\Hlim f}{f}
\le\|\Hlim f\|\|f\|<\infty$
(using $\Hlim\in\mathcal{B}(L^2)$
from \eqref{IX2});
dense by \eqref{IX5}.
(c) Polarisation.
\end{proof}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence]
\label{def:mosco}
Forms $(\qc)$ converge to $\qlim$
in the sense of Mosco \cite{Mosco1969}:
\begin{enumerate}[\rm(M1)]
\item\label{M1}
      $f_c\rightharpoonup f$ weakly
      $\Rightarrow$
      $\qlim(f,f)\le\liminf_c\qc(f_c,f_c)$.
\item\label{M2}
      $\forall f\in\Dom(\qlim)$,
      $\exists f_c\to f$ strongly with
      $\qc(f_c,f_c)\to\qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[HS compactness and quadratic
  form weak continuity]
\label{lem:hs}
For each fixed $c_0$:
\begin{enumerate}[\rm(a)]
\item $\mathcal{H}_{c_0}\in
      \mathcal{B}_2(L^2([0,1]))$
      (Hilbert--Schmidt).
\item For every sequence with
      $f_c\rightharpoonup f$ weakly:
      \begin{equation}
        \label{eq:qwc}
        \ip{\mathcal{H}_{c_0}f_c}{f_c}
        \longrightarrow
        \ip{\mathcal{H}_{c_0}f}{f}.
      \end{equation}
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a).}
By \eqref{IX6},
$|K_{c_0}(x,y)|\le c_0/\pi$ uniformly
on $[0,1]^2$, hence
$\|\mathcal{H}_{c_0}\|_{\mathrm{HS}}^2
=\int_0^1\int_0^1|K_{c_0}(x,y)|^2\,dx\,dy
\le(c_0/\pi)^2<\infty$.
Hilbert--Schmidt $\Rightarrow$ compact
(\cite{ReedSimon1975}, Thm.~VI.22).

\textit{(b).}
Decompose:
\begin{equation}
  \label{eq:3term}
  \ip{\mathcal{H}_{c_0}f_c}{f_c}
  -\ip{\mathcal{H}_{c_0}f}{f}
  = \underbrace{%
    \ip{\mathcal{H}_{c_0}(f_c-f)}{f_c}
    }_{=:I}
  + \underbrace{%
    \ip{\mathcal{H}_{c_0}f}{f_c-f}
    }_{=:II}.
\end{equation}
\textit{Term~II.}
$\mathcal{H}_{c_0}f$ is a fixed $L^2$
vector; $f_c-f\rightharpoonup 0$
gives $II\to 0$.
\textit{Term~I.}
Every weakly convergent sequence is
norm-bounded by the Uniform Boundedness
Principle (\cite{ReedSimon1975},
Thm.~III.5): $\sup_c\|f_c\|\le M$.
$\mathcal{H}_{c_0}$ compact maps
$f_c-f\rightharpoonup 0$ to
$\|\mathcal{H}_{c_0}(f_c-f)\|\to 0$
(\cite{ReedSimon1975}, Thm.~VI.11),
so $|I|\le M\|\mathcal{H}_{c_0}(f_c-f)\|
\to 0$.
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc\to\qlim$ in the sense of Mosco.
\end{theorem}

\begin{proof}
\textit{(M2).}
Take $f_c:=f$;
$\qc(f,f)\to\qlim(f,f)$
by Definition~\ref{def:qlim}.

\textit{(M1).}
Let $f_c\rightharpoonup f$.
Fix $c_0$; for $c\ge c_0$
(monotonicity \eqref{IX4}):
$\qc(f_c,f_c)\ge q_{c_0}(f_c,f_c)
=\ip{\mathcal{H}_{c_0}f_c}{f_c}$.
By Lemma~\ref{lem:hs}(b),
$\ip{\mathcal{H}_{c_0}f_c}{f_c}
\to q_{c_0}(f,f)$,
so $\liminf_c\qc(f_c,f_c)\ge q_{c_0}(f,f)$.
Send $c_0\to\infty$:
$\liminf_c\qc(f_c,f_c)\ge\qlim(f,f)$.
\end{proof}

\begin{remark}[Mosco $\Rightarrow$
  strong resolvent convergence]
\label{rem:mosco-significance}
By Kuwae--Shioya
\cite{KuwaeShioya2003}, Thm.~2.4,
Mosco convergence of forms implies
strong resolvent convergence.
This provides a route to
Open Problem~\ref{prob:resolvent}.
\end{remark}

\section{Closedness}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness and
  semiboundedness of $\qlim$]
\label{thm:form-closed}
$\qlim\ge 0$ and $\qlim$ is closed.
\end{theorem}
\begin{proof}
$\qlim\ge 0$: each $\ip{\Hc f}{f}\ge 0$.
Closedness: if $f_k\to f$ in $L^2$
(hence weakly) and
$\qlim(f_k-f_j,f_k-f_j)\to 0$,
then (M1) gives
$\qlim(f,f)\le\liminf_k\qlim(f_k,f_k)
<\infty$ (so $f\in\Dom(\qlim)$),
and the parallelogram law in the
pre-Hilbert space
$(\Dom(\qlim),\qlim(\cdot,\cdot))$
yields $\qlim(f_k-f,f_k-f)\to 0$.
Hence $(\Dom(\qlim),\|\cdot\|_{\qlim})$
is complete, i.e.\ $\qlim$ is closed
(\cite{Kato1966}, Def.~VI.1.3;
\cite{Mosco1969}, Thm.~1).
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative convergence]
\label{thm:form-convergence}
For all $f\in L^2(\mathbb{R})$, $c\ge c_0$:
\begin{equation}
  \label{eq:rate}
  0\le\qlim(f,f)-\qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\end{equation}
\end{theorem}
\begin{proof}
Lower bound: monotonicity \eqref{IX4}.
Upper bound: since
$\Hlim,\Hc\in\mathcal{B}(L^2)$,
\begin{align*}
  \qlim(f,f)-\qc(f,f)
  &=\ip{(\Hlim-\Hc)f}{f}\\
  &\le\|\Hlim-\Hc\|\,\|f\|^2
  \le C_\kappa c^{-1/4}\|f\|^2
\end{align*}
by \eqref{IX2}.
\end{proof}

\begin{remark}
The monotone convergence of
$\qlim(f,f)=\langle\Hlim f,f\rangle$
(using $\Hlim\in\mathcal{B}(L^2)$)
shows that $\Dom(\qlim)=L^2(\mathbb{R})$;
the rate \eqref{eq:rate} holds
for all $f\in L^2$.
\end{remark}

\section{Friedrichs Extension and
  Self-Adjointness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Kato representation]
\label{thm:kato-rep}
Densely defined, closed, semibounded
$q$ on $\mathcal{H}$ determines a
unique self-adjoint operator $T$,
$\Dom(T)\subset\Dom(q)$,
with $q(f,g)=\ip{Tf}{g}$ for
$f\in\Dom(T)$, $g\in\Dom(q)$.
\emph{(\cite{Kato1966}, VI.2.1;
\cite{ReedSimon1975}, VIII.15.)}
\end{theorem}

\begin{lemma}[Core density]
\label{lem:core}
Let $T_{\qlim}$ be from
Theorem~\ref{thm:kato-rep}.
Then $\Dom(\Hlim)$ is a core for
$T_{\qlim}$, i.e.\ dense in
$\Dom(T_{\qlim})$ with respect to
the graph norm
$\|f\|_{T_{\qlim}}^2
:=\|f\|^2+\|T_{\qlim}f\|^2$.
\end{lemma}

\begin{proof}
\textbf{Norm equivalence.}
Since $T_{\qlim}\in\mathcal{B}(L^2)$
with $\|T_{\qlim}\|\le 1$
(inherited from $\|\Hlim\|\le 1$
via Step~1 of
Theorem~\ref{thm:friedrichs}),
we have for $f\in\Dom(T_{\qlim})$:
\begin{equation}
  \label{eq:norm-equiv}
  \|f\|^2
  \le\|f\|_{T_{\qlim}}^2
  =\|f\|^2+\|T_{\qlim}f\|^2
  \le 2\|f\|^2.
\end{equation}
Moreover, $\qlim(f,f)
=\ip{T_{\qlim}f}{f}
\le\|T_{\qlim}\|\|f\|^2
\le\|f\|^2$,
so the form norm satisfies
$\|f\|_{\qlim}^2
=\|f\|^2+\qlim(f,f)
\le 2\|f\|^2$.
From \eqref{eq:norm-equiv},
all three norms
$\|\cdot\|$, $\|\cdot\|_{T_{\qlim}}$,
$\|\cdot\|_{\qlim}$
are mutually equivalent on
$\Dom(T_{\qlim})$.

\textbf{It suffices to prove density
in $\|\cdot\|_{\qlim}$.}
By the equivalence above, density
of $\Dom(\Hlim)$ in
$(\Dom(T_{\qlim}),\|\cdot\|_{\qlim})$
implies density in
$(\Dom(T_{\qlim}),\|\cdot\|_{T_{\qlim}})$,
i.e.\ $\Dom(\Hlim)$ is a core
for $T_{\qlim}$.

\textbf{Step~A: Regularise
  $g\mapsto g_c$.}
For $g\in\Dom(T_{\qlim})$ and $c>0$,
set
\begin{equation}
  \label{eq:gc}
  g_c
  :=(\mathrm{Id}+c^{-1}\Hc)^{-1}g
  \in L^2(\mathbb{R}).
\end{equation}
Since $\|\Hc\|\le 1$ (\eqref{IX6}),
the inverse exists with norm $\le 1$.

\textit{Resolvent identity}
(algebraic, no functional calculus):
\begin{equation}
  \label{eq:resolvent-id}
  g = g_c + c^{-1}\Hc g_c
  \implies
  \Hc g_c = c(g-g_c).
\end{equation}

\textit{$L^2$-convergence:}
$\|g-g_c\|=c^{-1}\|\Hc g_c\|
\le c^{-1}\|g\|\to 0$.

\textit{Form-norm convergence:}
for any $c'>0$, since
$\|\mathcal{H}_{c'}\|\le 1$:
$q_{c'}(g-g_c,g-g_c)
\le\|g-g_c\|^2$;
sending $c'\to\infty$:
$\qlim(g-g_c,g-g_c)
\le\|g-g_c\|^2\to 0$.
Hence $g_c\to g$ in
$\|\cdot\|_{\qlim}$.

\textbf{Step~B: Approximate $g_c$
  from $\Dom(\Hlim)$.}
Fix $c$ and $\varepsilon>0$.
By \eqref{IX5} choose
$h\in\Dom(\Hlim)$ with
$\|h-g_c\|<\varepsilon$.
Since $\Hlim\in\mathcal{B}(L^2)$
with $\|\Hlim\|\le 1$ (\eqref{IX2}),
for any $c'>0$:
\begin{align}
  q_{c'}(h-g_c,h-g_c)
  &=\ip{(\mathcal{H}_{c'}-\Hlim)
    (h-g_c)}{h-g_c}
  +\ip{\Hlim(h-g_c)}{h-g_c}.
  \label{eq:split-step-B}
\end{align}
First term $\to 0$ as $c'\to\infty$
by \eqref{IX2}.
Second term:
$|\ip{\Hlim(h-g_c)}{h-g_c}|
\le\|\Hlim\|\|h-g_c\|^2\le\varepsilon^2$.
Sending $c'\to\infty$:
\begin{equation}
  \label{eq:form-step-B}
  \qlim(h-g_c,h-g_c)\le\varepsilon^2.
\end{equation}
Choosing $\varepsilon=\varepsilon_j\to 0$
gives $h_j^{(c)}\in\Dom(\Hlim)$
with $h_j^{(c)}\to g_c$ in
$\|\cdot\|_{\qlim}$.

\textbf{Diagonal completion.}
For each $c$ choose $j(c)$ so that
$\|h_{j(c)}^{(c)}-g_c\|_{\qlim}
\le c^{-1}$; set $h_c:=h_{j(c)}^{(c)}$.
Then
$\|h_c-g\|_{\qlim}
\le\|h_c-g_c\|_{\qlim}
+\|g_c-g\|_{\qlim}\to 0$.
\end{proof}

\begin{theorem}[Friedrichs extension
  and essential self-adjointness]
\label{thm:friedrichs}
Let $T_{\qlim}$ be from
Theorem~\ref{thm:kato-rep}.
Then $T_{\qlim}=\overline{\Hlim}$,
$\Hlim$ is essentially self-adjoint,
and $n_\pm(\Hlim)=0$.
\end{theorem}

\begin{proof}
\textbf{Step~1} ($\Hlim\subset T_{\qlim}$).
For $f\in\Dom(\Hlim)$ and
$g\in\Dom(\qlim)=L^2$:
$\ip{\Hlim f}{g}
=\lim_c\ip{\Hc f}{g}
=\qlim(f,g)$
by strong convergence \eqref{IX7};
hence $T_{\qlim}f=\Hlim f$.

\begin{remark*}
$\Hlim$ is closable: $\Hlim\subset T_{\qlim}$
and $T_{\qlim}$ is closed (self-adjoint),
so every restriction of $T_{\qlim}$ is
closable; in particular
$\overline{\Hlim}\subset T_{\qlim}$.
\end{remark*}

\textbf{Step~2}
($T_{\qlim}\subset\overline{\Hlim}$).
Let $f\in\Dom(T_{\qlim})$.
By Lemma~\ref{lem:core},
choose $h_c\in\Dom(\Hlim)$
with $h_c\to f$ in $\|\cdot\|_{\qlim}$,
hence $h_c\to f$ in $L^2$.
By Step~1, $\Hlim h_c=T_{\qlim}h_c$.
Since $T_{\qlim}\in\mathcal{B}(L^2)$:
\[
  \|\Hlim h_c-T_{\qlim}f\|
  =\|T_{\qlim}(h_c-f)\|
  \le\|T_{\qlim}\|\,\|h_c-f\|
  \to 0.
\]
The graph sequence
$(h_c,\Hlim h_c)\to(f,T_{\qlim}f)$
in $L^2\times L^2$; since
$\overline{\Hlim}$ is closed,
$f\in\Dom(\overline{\Hlim})$ and
$\overline{\Hlim}f=T_{\qlim}f$.

\textbf{Step~3.}
$\Hlim\subset T_{\qlim}
\subset\overline{\Hlim}$
and $T_{\qlim}$ is closed,
giving $T_{\qlim}=\overline{\Hlim}$.

\textbf{Step~4.}
By von Neumann's theorem,
$\Hlim$ is essentially self-adjoint
$\iff$ $n_+(\Hlim)=n_-(\Hlim)=0$.
Since $\overline{\Hlim}=T_{\qlim}$
is self-adjoint, this holds.
\end{proof}

\begin{remark}[Complete input audit]
\label{rem:audit}
Every step uses exactly one of:
\begin{itemize}
\item \eqref{IX6} (symmetric kernel)
      $\xrightarrow{\text{Spectral Thm}}$
      $\|\Hc\|\le 1$
      $\xrightarrow{\text{UBP}}$
      $\|\Hlim\|\le 1$
      $\xrightarrow{\text{HS}}$
      compact
      $\to$ M1 (Lemma~\ref{lem:hs});
\item \eqref{IX4} (monotonicity)
      $\to$ existence of $\qlim$,
      M1 lower bound;
\item \eqref{IX2} (norm rate on all $L^2$)
      $\to$ quantitative rate
      (Thm.~\ref{thm:form-convergence}),
      Step~B of Lemma~\ref{lem:core};
\item \eqref{IX7} (strong convergence)
      $\to$ Step~1;
\item Kato Thm.~VI.2.1
      $\to$ $T_{\qlim}$ exists.
\end{itemize}
No eigenvector, Parseval, or
completeness is used anywhere.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Essential
  self-adjointness]
\label{cor:esa}
$\Hlim$ is essentially self-adjoint:
$\overline{\Hlim}=\Hlim^*$,
$\sigma_{\rm app}(\Hlim)
=\sigma(\Hlim)$.
\end{corollary}

\begin{corollary}[Spectral chain]
\label{cor:chain}
\[
  \underbrace{\Delta_\varepsilon
    \asymp c^{-1/2}}_{\text{VI--VIII}}
  \Rightarrow
  \underbrace{\rho\in
    \sigma_{\rm app}(\Hlim)}_{\text{VIII
    [CCM]}}
  \Rightarrow
  \underbrace{\rho\in
    \sigma(\overline{\Hlim})}_{\text{X,
    uncond.}}
  \Rightarrow
  \underbrace{\mathrm{Re}(\rho)
    =\tfrac{1}{2}}_{\text{Cor.
    \ref{cor:rh}}}.
\]
\end{corollary}

\begin{corollary}[Riemann Hypothesis]
\label{cor:rh}
Under Hypothesis~CCM
(\cite{CCM2025}, Thm.~1.3),
every non-trivial zero $\rho$ of
$\zeta(s)$ satisfies
$\mathrm{Re}(\rho)=\tfrac{1}{2}$.
\end{corollary}

\end{remark}
