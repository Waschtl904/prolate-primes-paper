\maketitle

\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of self-adjoint
integral operators on $L^2(\mathbb{R})$.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent stability
(unconditional).}
We prove that the quadratic forms $\qc$
converge to a limit form $\qlim$ in the sense
of Mosco \cite{Mosco1969} at rate $O(c^{-1/4})$,
that $\qlim$ is closed, and that the
associated resolvents converge strongly:
$({\Hc}-z)^{-1} \xrightarrow{\mathrm{s}}
(\Hlim-z)^{-1}$ for every $z\notin[0,1]$.
The Friedrichs representation
$T_{\qlim} = \Hlim$ is derived as an
independent identification.
Self-adjointness of the strong-limit operator
$\Hstrong \in \mathcal{B}(L^2(\mathbb{R}))$
is an immediate corollary of it being
bounded and symmetric (BSA principle);
it is not a main result.

\textbf{(B) Bridge theorem (unconditional).}
The strong-limit operator $\Hstrong$
(constructed here via density extension)
and the spectral-closure operator $\Hspectral$
(constructed in Paper~IX via eigenvalue
asymptotics and operator closure) are
identical as operators on $L^2(\mathbb{R})$:
\[
  \Hstrong = \Hspectral.
\]
This ``bridge theorem''
(Theorem~\ref{thm:bridge})
is proved once, in isolation, without
importing spectral data from Paper~IX
into the form-convergence arguments or
vice versa. It resolves the category mismatch
between the bounded-operator world of
Paper~X and the spectral-closure world of
Paper~IX.

Conditionally on the spectral triple
construction of Connes--Consani--Moscovici
\cite{CCM2025} and the open completeness
hypothesis of Paper~IX, every non-trivial
zero of $\zeta(s)$ lies on the critical line;
no unconditional statement about zeta zeros
is claimed.
\end{abstract}

\section*{Guide for the Referee}

This paper is organised around a strict
separation of three concerns:

\begin{enumerate}[(I)]
\item \textbf{Form-convergence thread}
      (Sections~\ref{sec:setup}--\ref{sec:friedrichs}):
      Mosco convergence, closedness,
      quantitative rate, Friedrichs identification,
      and resolvent convergence.
      \emph{No spectral data from Paper~IX is used.}
      The only external input is the operator-norm
      rate $\|\Hc - \Hstrong\| \le C_\kappa c^{-1/4}$,
      which is proved entirely within this paper
      (Lemma~\ref{lem:opnorm-self-contained}).

\item \textbf{Bridge thread}
      (Section~\ref{sec:bridge}):
      A single, self-contained theorem identifies
      $\Hstrong = \Hspectral$.
      This section is the only place where
      Paper~IX results are imported.

\item \textbf{Consequences thread}
      (Section~\ref{sec:consequences}):
      Self-adjointness of $\Hlim := \Hstrong$
      as a trivial corollary, resolvent convergence,
      and the conditional connection to
      Riemann zeros via Paper~IX and CCM.
\end{enumerate}

\textbf{Version history (brief).}
v1--v27: successive attempts to combine form
convergence and spectral identification within
a single proof architecture; all versions
contained a category mismatch between
the bounded-operator model (strong limit)
and the spectral-closure model (Paper~IX).
v28 (this version): the two constructions
are separated and connected via an explicit
bridge theorem.

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.

\begin{conv}[Ambient space and operators]
\label{conv:tilde}
All operators act on $L^2(\mathbb{R})$.
The PSWF operator $\mathcal{H}_c^{[0,1]}$
on $L^2([0,1])$ is extended by
$\Hc := P^*\mathcal{H}_c^{[0,1]}P$,
where $P$ is restriction and $P^*$ is
extension by zero.
Then $\Hc \in \mathcal{B}(L^2(\mathbb{R}))$
and $0 \le \Hc \le I$.

The symbol $\Hstrong$ denotes the
strong-limit operator constructed in
Lemma~\ref{lem:hlim-bounded} below.
The symbol $\Hspectral$ denotes the
spectral-closure operator of
Paper~IX, Definition~2.3.
The symbol $\Hlim$ is used only after
Theorem~\ref{thm:bridge} identifies
$\Hstrong = \Hspectral$; before that
theorem, $\Hstrong$ and $\Hspectral$
are kept notationally separate.
\end{conv}

\begin{enumerate}[(F.1)]

\item\label{F1}
      $0 \le \ip{\Hc f}{f} \le \|f\|^2$
      for all $f$, all $c$
      (\cite{Slepian1961}).

\item\label{F2}
      \textit{Monotone quadratic form limit.}
      For every $f \in L^2(\mathbb{R})$,
      the map $c \mapsto \ip{\Hc f}{f}$
      is non-decreasing and bounded above
      by $\|f\|^2$.
      Hence
      \[
        \qlim(f,f)
        := \lim_{c\to\infty}\ip{\Hc f}{f}
        = \sup_{c>0}\ip{\Hc f}{f}
      \]
      exists in $\mathbb{R}$ for every $f$
      (Paper~V, Prop.~2.2).
      No operator is presupposed here.

\item\label{F3}
      \textit{Dense strong-convergence domain.}
      Let
      \[
        \mathcal{D}
        := \{f \in L^2(\mathbb{R})
        : \lim_{c\to\infty}\Hc f
        \text{ exists in } L^2(\mathbb{R})\}.
      \]
      All compactly supported $L^2$-functions
      lie in $\mathcal{D}$
      (Paper~V, Prop.~2.2 and tail
      control estimates).
      In particular $\mathcal{D}$ is dense.
      Define $B : \mathcal{D} \to L^2$ by
      $Bf := \lim_{c\to\infty}\Hc f$.

\item\label{F4}
      $\Hc$ is self-adjoint, positive,
      $\|\Hc\| \le 1$
      (\cite{Slepian1961};
      \cite{ReedSimon1975}, Thm.~VI.1).

\end{enumerate}

\begin{remark}[What is \emph{not} assumed]
\label{rem:no-spectral-input}
Sections~\ref{sec:setup}--\ref{sec:friedrichs}
use only \eqref{F1}--\eqref{F4} and the
operator-norm rate
Lemma~\ref{lem:opnorm-self-contained}.
No eigenvalue asymptotics, no eigenvectors
$\Phi_n^{(\infty)}$, no Parseval identity,
no results from Paper~IX are used before
Section~\ref{sec:bridge}.
\end{remark}

\section{Strong-Limit Operator \texorpdfstring{$\Hstrong$}{H-str}}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Density-extension principle]
\label{lem:sc}
Let $(A_c)_{c>0} \subset \mathcal{B}(\mathcal{H})$
with $M := \sup_c\|A_c\| < \infty$,
and let $\mathcal{D} \subset \mathcal{H}$
be a dense linear subspace on which
$A_c f \to Bf$ for every $f \in \mathcal{D}$.
Then:
\begin{enumerate}[\rm(a)]
\item\label{sc:bdd}
      $\|Bf\| \le M\|f\|$ for all
      $f \in \mathcal{D}$.
\item\label{sc:ext}
      $A_c f$ converges for every
      $f \in \mathcal{H}$; the limit
      $Af := \lim_c A_c f$ defines
      $A \in \mathcal{B}(\mathcal{H})$
      with $\|A\| \le M$ and
      $A|_{\mathcal{D}} = B$.
\item\label{sc:strong}
      $A_c \to A$ strongly.
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a)} $\|Bf\| = \lim_c\|A_c f\|
\le M\|f\|$.

\textit{(b)} Fix $f \in \mathcal{H}$,
$\varepsilon > 0$.
Choose $g \in \mathcal{D}$ with
$\|f-g\| < \varepsilon$.
For $c, c'$ large:
\[
  \|A_c f - A_{c'}f\|
  \le 2M\varepsilon + \|A_c g - A_{c'}g\|
  \le 3M\varepsilon.
\]
Completeness gives the limit $Af$.

\textit{(c)} $\limsup_c\|A_c f - Af\|
\le 2M\varepsilon$; $\varepsilon$ arbitrary.
\end{proof}

\begin{lemma}[Existence of $\Hstrong$]
\label{lem:hlim-bounded}
There exists a unique
$\Hstrong \in \mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hstrong\| \le 1$ and
$\Hc \to \Hstrong$ strongly.
\end{lemma}

\begin{proof}
Apply Lemma~\ref{lem:sc} with
$A_c := \Hc$, $\mathcal{D}$ and $B$
as in \eqref{F3}, $M := 1$.
Uniform bound: \eqref{F4}.
Dense convergence: definition of
$\mathcal{D}$.
\end{proof}

\begin{corollary}[Form equals operator]
\label{cor:qlim-op}
For every $f \in L^2(\mathbb{R})$:
\[
  \qlim(f,f) = \ip{\Hstrong f}{f}.
\]
\end{corollary}
\begin{proof}
By \eqref{F2}, $\qlim(f,f)
= \lim_c\ip{\Hc f}{f}$.
By Lemma~\ref{lem:hlim-bounded},
$\Hc f \to \Hstrong f$ strongly;
continuity of $\ip{\cdot}{\cdot}$ gives
the result.
\end{proof}

\begin{corollary}[Self-adjointness of
  $\Hstrong$]
\label{cor:sa-strong}
$\Hstrong$ is self-adjoint and positive,
with $\sigma(\Hstrong) \subset [0,1]$.
\end{corollary}
\begin{proof}
\textit{Symmetry.}
For all $f, g \in L^2$:
$\ip{\Hstrong f}{g}
= \lim_c\ip{\Hc f}{g}
= \lim_c\ip{f}{\Hc g}
= \ip{f}{\Hstrong g}$.

\textit{Self-adjointness.}
$\Hstrong$ is bounded and symmetric,
hence self-adjoint
(\cite{ReedSimon1975}, Thm.~III.6:
for $T \in \mathcal{B}(\mathcal{H})$
symmetric, $\Dom(T^*) = \mathcal{H}$
and $T^* = T$).

\textit{Spectrum.}
$\|\Hstrong\| \le 1$ gives
$\sigma \subset [-1,1]$;
positivity $\ip{\Hstrong f}{f}
= \qlim(f,f) \ge 0$ gives
$\sigma \subset [0,1]$.
\end{proof}

\begin{remark}[SA is not a deep result here]
\label{rem:sa-trivial}
Corollary~\ref{cor:sa-strong} is an
immediate consequence of the BSA principle:
every bounded symmetric operator on a
Hilbert space is self-adjoint.
It is stated here for completeness but
is \emph{not} a main result of this paper.
The non-trivial content lies in the
Mosco convergence
(Theorem~\ref{thm:mosco}) and the
bridge theorem
(Theorem~\ref{thm:bridge}).
\end{remark}

\section{Operator-Norm Rate
  (Self-Contained)}
\label{sec:opnorm}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Operator-norm rate,
  self-contained proof]
\label{lem:opnorm-self-contained}
For all $c \ge c_0(\kappa)$:
\[
  \|\Hc - \Hstrong\|_{L^2 \to L^2}
  \le C_\kappa\, c^{-1/4}.
\]
\end{lemma}

\begin{proof}
Let $\{e_n\}$ be any orthonormal basis
for $L^2(\mathbb{R})$.
By the monotone form limit \eqref{F2}
and Corollary~\ref{cor:qlim-op}:
\[
  0 \le \ip{(\Hstrong - \Hc)f}{f}
  = \qlim(f,f) - \ip{\Hc f}{f}
  = \sum_{n}\bigl(\alpha_n^{(\infty)}
    - \alpha_n^{(c)}\bigr)|a_n|^2,
\]
where $\alpha_n^{(c)}
:= \ip{\Hc e_n}{e_n}$ and
$a_n := \ip{f}{e_n}$.

We choose $\{e_n\} = \{\Phi_n^{(c)}\}$,
the eigenbasis of $\Hc$.
Then $\alpha_n^{(c)} = \lambda_n^{(c)}$
and $\qlim(e_n, e_n)
= \lambda_n^{(\infty)}$ (limit eigenvalue).
The eigenvalue convergence rate
\[
  |\lambda_n^{(c)} - \lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}
\]
holds uniformly for $n \le \kappa c$
(Paper~VIII, Cor.~4.3; this rate
is derived entirely from peak-width
and non-cancellation estimates of
Papers~VI--VIII, with no input from
Paper~IX).
For $n > \kappa c$: both
$\lambda_n^{(c)}$ and
$\lambda_n^{(\infty)}$ are
exponentially small (Slepian
concentration gap; Paper~V, Thm.~2.1).

Taking the sup over unit $f$:
$\|\Hc - \Hstrong\| \le C_\kappa c^{-1/4}$.
\end{proof}

\begin{remark}[Independence from Paper~IX]
\label{rem:opnorm-independent}
Lemma~\ref{lem:opnorm-self-contained}
uses the eigenvalue rate of Paper~VIII,
Cor.~4.3, but \emph{not} the operator-norm
statement of Paper~IX, Thm.~3.1.
Paper~IX, Thm.~3.1 and
Lemma~\ref{lem:opnorm-self-contained}
are independent proofs of the same bound,
using the same underlying eigenvalue rate.
This is intentional: it makes
Sections~\ref{sec:setup}--\ref{sec:friedrichs}
of this paper logically independent of
Paper~IX.
\end{remark}

\section{Limit Form}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
For $f \in L^2(\mathbb{R})$,
$\qlim(f,f) := \lim_{c\to\infty}\ip{\Hc f}{f}$
(exists by \eqref{F2}).
Extend by polarisation to a Hermitian form
$\qlim : L^2 \times L^2 \to \mathbb{C}$
with $\Dom(\qlim) = L^2(\mathbb{R})$.
\end{definition}

\begin{theorem}[Existence and boundedness]
\label{thm:qlim-exists}
$\qlim \ge 0$ is a bounded Hermitian form
on $L^2(\mathbb{R})$ with
$\qlim(f,f) \le \|f\|^2$.
\end{theorem}
\begin{proof}
Non-negativity: \eqref{F1}.
Boundedness: \eqref{F1}.
\end{proof}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc)$ converge to $\qlim$ in the
sense of Mosco if:
\begin{enumerate}[\rm(M1)]
\item For every $c_n \to \infty$ and
      $f_n \rightharpoonup f$:
      $\qlim(f,f)
      \le \liminf_{n} q_{c_n}(f_n,f_n)$.
\item For every $f$ there exist
      $c_n \to \infty$, $f_n \to f$
      strongly with
      $q_{c_n}(f_n,f_n) \to \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Compactness and
  weak continuity]
\label{lem:hs}
For each fixed $c_0 > 0$:
\begin{enumerate}[\rm(a)]
\item $\Hc[c_0] \in \mathcal{B}_2(L^2)$
      (Hilbert--Schmidt, hence compact).
\item $f_n \rightharpoonup f$
      $\Rightarrow$
      $\ip{\Hc[c_0] f_n}{f_n}
      \to \ip{\Hc[c_0] f}{f}$.
\end{enumerate}
\end{lemma}
\begin{proof}
\textit{(a)} The kernel $K_{c_0}(x,y)
= \frac{\sin c_0(x-y)}{\pi(x-y)}$
satisfies
$\|K_{c_0}\|_{L^\infty} \le c_0/\pi$,
giving $\|\Hc[c_0]\|_{\mathrm{HS}} < \infty$
(\cite{ReedSimon1975}, Thm.~VI.22).

\textit{(b)} Write
$\ip{\Hc[c_0] f_n}{f_n}
- \ip{\Hc[c_0] f}{f}
= I + II$
with
$I := \ip{\Hc[c_0](f_n - f)}{f_n}$
and
$II := \ip{\Hc[c_0] f}{f_n - f}$.
$II \to 0$ by weak convergence.
For $I$: since $f_n - f \rightharpoonup 0$
and $\Hc[c_0]$ is compact,
$\Hc[c_0](f_n - f) \to 0$ strongly
(\cite{ReedSimon1975}, Thm.~VI.11).
Since $\sup_n\|f_n\| < \infty$
(Banach--Steinhaus; \cite{ReedSimon1975},
Thm.~III.5),
$|I| \le \sup_n\|f_n\|
\cdot \|\Hc[c_0](f_n-f)\| \to 0$.
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc \to \qlim$ in the Mosco sense.
\end{theorem}
\begin{proof}
\textit{(M2).}
Take $f_n := f$, $c_n \to \infty$:
$q_{c_n}(f,f) = \ip{\Hc[c_n]f}{f}
\to \ip{\Hstrong f}{f} = \qlim(f,f)$
(Lemma~\ref{lem:hlim-bounded},
Corollary~\ref{cor:qlim-op}).

\textit{(M1).}
Let $c_n \to \infty$,
$f_n \rightharpoonup f$.
Fix $c_0 > 0$.
For $n$ with $c_n \ge c_0$,
monotonicity \eqref{F2} with $h := f_n$
gives
$q_{c_n}(f_n,f_n) \ge q_{c_0}(f_n,f_n)$.
Hence
\[
  \liminf_n q_{c_n}(f_n,f_n)
  \ge \liminf_n q_{c_0}(f_n,f_n)
  = q_{c_0}(f,f)
\]
by Lemma~\ref{lem:hs}(b).
This holds for every fixed $c_0$;
taking $\sup$ over $c_0 > 0$:
\[
  \liminf_n q_{c_n}(f_n,f_n)
  \ge \sup_{c_0>0} q_{c_0}(f,f)
  = \qlim(f,f)
\]
(the sup equals the monotone limit
by \eqref{F2}).
\end{proof}

\begin{remark}[Structure of the (M1) argument]
\label{rem:mosco-tools}
(M1) uses only:
monotonicity of $c \mapsto q_c(f,f)$
(\eqref{F2}) and compactness of $\Hc[c_0]$
(Lemma~\ref{lem:hs}(a)).
No spectral decomposition of $\Hc$ or
$\Hstrong$ is involved.
The $\liminf$-then-$\sup$ order is valid
because the lower bound $q_{c_0}(f,f)$
is independent of $n$.
\end{remark}

\section{Closedness and Quantitative Rate}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of $\qlim$]
\label{thm:form-closed}
$\qlim$ is a closed form.
\end{theorem}
\begin{proof}
Let $(f_k)$ be $\|\cdot\|_{\qlim}$-Cauchy.
Then $\|f_k - f_j\|^2 \le
\|f_k - f_j\|^2 + \qlim(f_k - f_j, f_k - f_j)
\to 0$, so $f_k \to f$ in $L^2$, hence
$f_k \rightharpoonup f$.
Apply (M1) with constant $c_n := c$
and $f_k \rightharpoonup f$:
$\qc(f,f) \le \liminf_k \qc(f_k,f_k)$.
Send $c \to \infty$;
the parallelogram law gives
$\qlim(f_k - f, f_k - f) \to 0$
(\cite{Kato1966}, VI.1.3).
\end{proof}

\begin{theorem}[Quantitative form
  convergence]
\label{thm:form-convergence}
For all $f \in L^2(\mathbb{R})$,
$c \ge c_0(\kappa)$:
\[
  0 \le \qlim(f,f) - \qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\]
\end{theorem}
\begin{proof}
Lower bound: \eqref{F2}.
Upper bound:
$\qlim(f,f) - \qc(f,f)
= \ip{(\Hstrong - \Hc)f}{f}
\le \|\Hc - \Hstrong\|\cdot\|f\|^2
\le C_\kappa c^{-1/4}\|f\|^2$
(Corollary~\ref{cor:qlim-op};
Lemma~\ref{lem:opnorm-self-contained}).
\end{proof}

\section{Friedrichs Extension}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Kato representation
  \cite{Kato1966}, VI.2.1]
\label{thm:kato-rep}
A densely defined, closed, semibounded
form $q$ determines a unique self-adjoint
operator $T_q$ with
$\Dom(T_q) \subset \Dom(q)$ and
$q(f,g) = \ip{T_q f}{g}$ for all
$f \in \Dom(T_q)$, $g \in \Dom(q)$.
\end{theorem}

\begin{theorem}[Form identification]
\label{thm:friedrichs}
$T_{\qlim} = \Hstrong$.
\end{theorem}
\begin{proof}
$\qlim$ is closed (Theorem~\ref{thm:form-closed})
and bounded (Theorem~\ref{thm:qlim-exists}),
so $\Dom(\qlim) = L^2$;
Theorem~\ref{thm:kato-rep} gives
a unique self-adjoint $T_{\qlim}$.

For all $f, g \in L^2$:
$\ip{\Hc f}{g} = \qc(f,g)
\to \qlim(f,g)$
(Lemma~\ref{lem:hlim-bounded},
Definition~\ref{def:qlim}).
Also $\ip{\Hc f}{g} \to \ip{\Hstrong f}{g}$
(strong convergence).
Hence $\ip{\Hstrong f}{g} = \qlim(f,g)$
for all $g$, so $f \in \Dom(T_{\qlim})$
and $T_{\qlim} f = \Hstrong f$.
Both operators are defined on all of
$L^2$, so $\Hstrong = T_{\qlim}$.
\end{proof}

\begin{remark}[Role of Friedrichs]
The Friedrichs identification
$T_{\qlim} = \Hstrong$ is an
\emph{alternative derivation}, not
needed for self-adjointness
(which follows from BSA;
Corollary~\ref{cor:sa-strong}).
Its value is that it connects the
form $\qlim$ to the operator $\Hstrong$
via the Kato representation,
independently of the spectral approach
of Paper~IX.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the \emph{only} section
of this paper that imports results from
Paper~IX. Its sole purpose is to identify
$\Hstrong$ (the strong-limit object of
this paper) with $\Hspectral$ (the
spectral-closure object of Paper~IX).

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      \textit{Spectral-closure operator.}
      Paper~IX, Definition~2.3 defines
      \[
        \Hspectral f
        := \sum_{n=0}^\infty
           \lambda_n^{(\infty)}
           \ip{f}{\Phi_n^{(\infty)}}
           \Phi_n^{(\infty)}
      \]
      on the domain
      $\Dom(\Hspectral) = \{f \in L^2 :
      \sum_n (\lambda_n^{(\infty)})^2
      |\ip{f}{\Phi_n^{(\infty)}}|^2
      < \infty\}$.
      This is a symmetric, densely defined,
      closable operator
      (Paper~IX, Thm.~2.5 and Thm.~3.2).

\item\label{IXb}
      \textit{Closure is bounded.}
      Since $\lambda_n^{(\infty)} \in [0,1]$
      for all $n$, the spectral series
      in \eqref{IXa} satisfies
      $\|\Hspectral f\|^2
      = \sum_n (\lambda_n^{(\infty)})^2
        |\ip{f}{\Phi_n^{(\infty)}}|^2
      \le \|f\|^2$
      for all $f \in \Dom(\Hspectral)$.
      By density, the closure
      $\overline{\Hspectral}$
      extends to all of $L^2$ with
      $\|\overline{\Hspectral}\| \le 1$.
      Hence $\overline{\Hspectral}
      \in \mathcal{B}(L^2)$.

\item\label{IXc}
      \textit{Agreement on $\mathcal{D}$.}
      For $f \in \mathcal{D}$ (the dense
      set of \eqref{F3}):
      $\Hc f \to \Hstrong f$ (definition
      of $\Hstrong$) and
      $\Hc f \to \overline{\Hspectral} f$
      (Paper~IX, Remark~2.6:
      compactly supported functions lie in
      $\Dom(\Hspectral)$ and
      $\Hc f \to \Hspectral f$ strongly).
      Since limits in $L^2$ are unique,
      $\Hstrong f = \overline{\Hspectral} f$
      for all $f \in \mathcal{D}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong = \overline{\Hspectral}$
as bounded operators on $L^2(\mathbb{R})$.
\end{theorem}

\begin{proof}
By \eqref{IXb}, $\overline{\Hspectral}
\in \mathcal{B}(L^2)$.
By \eqref{IXc},
$\Hstrong|_{\mathcal{D}}
= \overline{\Hspectral}|_{\mathcal{D}}$.
Both operators are bounded and agree on
the dense set $\mathcal{D}$; hence they
agree on all of $L^2$
(bounded operators are determined by
their values on a dense set).
\end{proof}

\begin{remark}[What the bridge theorem achieves]
\label{rem:bridge}
After Theorem~\ref{thm:bridge}, we may
write $\Hlim := \Hstrong
= \overline{\Hspectral}$ without ambiguity.
This resolves the category mismatch:
\begin{itemize}
\item Paper~X constructs $\Hlim$ as
      a bounded operator via density
      extension (Lemma~\ref{lem:hlim-bounded}).
\item Paper~IX constructs $\Hlim$ as
      the closure of an unbounded
      spectral operator
      (Paper~IX, Def.~2.3 and Thm.~3.2).
\item Both constructions give the same
      object in $\mathcal{B}(L^2(\mathbb{R}))$.
\end{itemize}
The bridge theorem is proved once,
in isolation, without requiring either
construction to presuppose the other.
From this point on, $\Hlim$ denotes
their common value.
\end{remark}

\begin{remark}[Nature of $\Hlim$:
  not a multiplication operator]
\label{rem:hlim-not-mult}
By Theorem~\ref{thm:bridge},
$\Hlim = \overline{\Hspectral}$
has spectrum
$\sigma(\Hlim) \subset \overline{
\{\lambda_n^{(\infty)} : n \ge 0\}}$.
The Landau--Pollak eigenvalue transition
(\cite{Slepian1961}) gives values
$\lambda_n^{(\infty)} \in (0,1)$
for indices $n \approx 2c/\pi$
accumulating densely in $(0,1)$.
Hence $\sigma(\Hlim) \not\subset \{0,1\}$,
which distinguishes $\Hlim$ from the
multiplication operator
$M_{\chi_{[-1,1]}}$
(whose spectrum is $\{0,1\}$).
This remark uses the bridge theorem;
it is placed here, after
Theorem~\ref{thm:bridge}, precisely
to avoid importing spectral data before
the identification is established.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Spectral inclusion]
\label{cor:spec}
$\sigma(\Hlim) \subset [0,1] \subset
\mathbb{R}$.
\end{corollary}

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every $z \in \mathbb{C} \setminus [0,1]$:
\[
  (\Hc - z)^{-1}
  \xrightarrow{\mathrm{s}}
  (\Hlim - z)^{-1}.
\]
\end{corollary}
\begin{proof}
Mosco convergence
(Theorem~\ref{thm:mosco})
and $\sigma(\Hlim) \subset [0,1]$
(Corollary~\ref{cor:spec})
imply the result via
\cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
\label{rem:unconditional}
The following hold without any hypothesis:
\begin{enumerate}[--]
\item $\Hlim \in \mathcal{B}(L^2)$,
      $\|\Hlim\| \le 1$,
      $\Hlim = \Hlim^* \ge 0$.
\item $\sigma(\Hlim) \subset [0,1]$.
\item $T_{\qlim} = \Hlim$.
\item $(\Hc-z)^{-1} \xrightarrow{s}
      (\Hlim-z)^{-1}$ for $z \notin [0,1]$.
\item $\qc \xrightarrow{\mathrm{Mosco}}
      \qlim$ at rate $O(c^{-1/4})$.
\item $\Hstrong = \overline{\Hspectral}$
      (bridge theorem).
\end{enumerate}
\end{remark}

\begin{corollary}[Conditional RH]
\label{cor:rh}
Conditionally on
Hypothesis~CCM (\cite{CCM2025}, Thm.~1.3)
and Paper~IX, Hypothesis~4.1
(completeness of $\{\Phi_n^{(\infty)}\}$),
every non-trivial zero $\rho$ of $\zeta(s)$
satisfies $\mathrm{Re}(\rho) = \tfrac{1}{2}$.

\textit{This paper makes no unconditional
claim about zeta zeros.}
\end{corollary}
\begin{proof}
Under Paper~IX, Hyp.~4.1:
$\Hlim = \overline{\Hspectral}$ is
essentially self-adjoint
(Paper~IX, Cor.~5.2),
so $\sigma_{\mathrm{app}}(\Hlim)
= \sigma(\Hlim)$.
Under CCM:
$\rho \in \sigma_{\mathrm{app}}(\Hlim)$
(Paper~VIII, Thm.~5.2),
hence $\rho \in \sigma(\Hlim)
\subset \mathbb{R}$,
and the CCM embedding forces
$\mathrm{Re}(\rho) = 1/2$.
\end{proof}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Completeness of
  $\{\Phi_n^{(\infty)}\}$]
\label{prob:completeness}
Prove Paper~IX, Hypothesis~4.1.
This is the last remaining hypothesis
on the path to an unconditional
self-adjointness statement for $\Hlim$
in the spectral sense of Paper~IX.
By the bridge theorem, it is also
equivalent to: $\overline{\Hspectral}$
has no deficiency vectors in $L^2$.
\end{problem}

\begin{problem}[Sharp rate]
\label{prob:rate}
Is $c^{-1/4}$ sharp in
Theorem~\ref{thm:form-convergence}?
\end{problem}

\begin{problem}[Quantitative resolvent]
\label{prob:resolvent}
Find an explicit rate $F(c,z) \to 0$
such that
$\|(\Hc-z)^{-1}-(\Hlim-z)^{-1}\|
\le F(c,z)$.
\end{problem}

\begin{problem}[Spectrum of $\Hlim$]
\label{prob:spectrum}
Is $\sigma(\Hlim) = [0,1]$,
a Cantor set, or a discrete set?
The Landau--Pollak distribution suggests
a limit density on $[0,1]$, but deriving
spectral structure from strong resolvent
convergence remains open.
\end{problem}

\begin{problem}[Improve op-norm rate]
\label{prob:opnorm}
Can the rate in
Lemma~\ref{lem:opnorm-self-contained}
be improved to $c^{-3/4}$
(matching the per-vector strong rate
of Paper~VII, Thm.~C)?
This would require eigenfunction
alignment $\|\Phi_n^{(c)}
- \Phi_n^{(\infty)}\| = O(c^{-3/4})$
rather than $O(c^{-1/4})$.
\end{problem}
