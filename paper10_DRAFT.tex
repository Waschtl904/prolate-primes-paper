\maketitle

\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of compact
self-adjoint operators on $L^2(\mathbb{R})$.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent
stability (unconditional).}
The foundational identity
(Lemma~\ref{lem:ftb}, Slepian--Pollak
\cite{Slepian1961})
$\ip{\Hc f}{f} = \|\Bc(P^*Pf)\|^2$
reduces the PSWF operator to a Fourier
band projection conjugated by
time-restriction.
From this, operator monotonicity,
Hilbert--Schmidt compactness, and
density of $\mathcal{D}$ are derived;
Mosco convergence, closedness, and
strong resolvent convergence follow.
A quantitative rate $O(c^{-1/4})$
is stated as an external theorem.

\textbf{(B) Bridge theorem (unconditional).}
$\Hstrong = \overline{\Hspectral}$
as bounded operators on $L^2(\mathbb{R})$.

The conditional connection to zeros of
$\zeta(s)$ (Remark~\ref{rem:rh})
is purely programmatic and holds only
under two unverified spectral hypotheses;
no unconditional claim is made.
\end{abstract}

\section{Setup and Foundational Lemmas}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

Let $\Fc$ denote the Fourier transform
$(\Fc f)(\xi) =
\int_{-\infty}^\infty f(x)
e^{-i\xi x}\,dx$,
with Plancherel isometry
$\|\Fc f\|_{L^2} =
\sqrt{2\pi}\|f\|_{L^2}$
(convention: \cite{ReedSimon1975},
Ch.~IX).

\begin{conv}[Operators]
\label{conv:ops}
\begin{alignat}{2}
  \Bc &:= \Fc^{-1}\chi_{[-c,c]}\Fc
  &&\quad (\text{Fourier band projection;
  }\Bc^*=\Bc=\Bc^2),\\
  P &: L^2(\mathbb{R}) \to L^2([0,1]),
  &&\quad \text{restriction;
  }\|P\|=1,\\
  P^* &: L^2([0,1]) \to
  L^2(\mathbb{R}),
  &&\quad \text{extension by zero;
  }\|P^*\|=1,\\
  \mathcal{H}_c^{[0,1]}
  &: L^2([0,1]) \to L^2([0,1]),
  &&\quad \text{kernel }
  K_c(x,y) = \tfrac{\sin c(x-y)}
  {\pi(x-y)},\\
  \Hc &:= P^*\mathcal{H}_c^{[0,1]}P
  &&\quad \in \mathcal{B}(L^2(\mathbb{R})).
\end{alignat}
$K_c(x,x) := c/\pi$ by continuity.
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
$\Hc$ is self-adjoint,
$0 \le \Hc \le I$,
$\|\Hc\| \le 1$
(\cite{Slepian1961};
\cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[Scope]
\label{rem:scope}
Sections~\ref{sec:setup}--%
\ref{sec:friedrichs} use only
\eqref{F1} and
Lemmas~\ref{lem:ftb}--\ref{lem:dense}.
No limit eigenvalues, eigenvectors,
or results from Papers~VIII,~IX appear.
\end{remark}

%─ Foundational lemma

\begin{lemma}[Fourier--time--band
  representation]
\label{lem:ftb}
For all $c > 0$ and
$f \in L^2(\mathbb{R})$:
\begin{align}
  \label{eq:ftb}
  \ip{\Hc f}{f}
  &= \|\Bc(P^*Pf)\|_{L^2(\mathbb{R})}^2,\\
  \label{eq:ftb-op}
  \mathcal{H}_c^{[0,1]}
  &= P\,\Bc\,P^*
  \quad \text{(operators on
  }L^2([0,1])).
\end{align}
\end{lemma}

\begin{proof}
\textit{Step 1: Convolution identity
on $L^2(\mathbb{R})$.}
The Fourier pair
$K_c(u) = \tfrac{\sin cu}{\pi u}
\leftrightarrow
(\Fc K_c)(\xi) = \chi_{[-c,c]}(\xi)$
(\cite{Slepian1961}, Eq.~(3);
\cite{ReedSimon1975}, Ex.~IX.7)
shows $K_c \in L^2(\mathbb{R})$.
For any $h \in L^2(\mathbb{R})$,
the convolution theorem
(\cite{ReedSimon1975}, Thm.~IX.2)
gives
\begin{equation}\label{eq:conv-bc}
  K_c * h
  = \Fc^{-1}[\chi_{[-c,c]}\Fc h]
  = \Bc h
  \quad \text{in }L^2(\mathbb{R}).
\end{equation}

\textit{Step 2: Restriction to $[0,1]$.}
Let $g \in L^2([0,1])$ and
$\tilde{g} = P^*g \in L^2(\mathbb{R})$.
Since $\tilde{g}(y) = 0$ a.e.\
outside $[0,1]$,
the integration domain reduces:
for every $x \in \mathbb{R}$,
\[
  (K_c * \tilde{g})(x)
  = \int_{-\infty}^{\infty}
    K_c(x-y)\tilde{g}(y)\,dy
  = \int_0^1 K_c(x-y)g(y)\,dy.
\]
For $x \in [0,1]$, the right side
is $(\mathcal{H}_c^{[0,1]}g)(x)$
by definition of the kernel
operator.
Applying $P$ on both sides and
using \eqref{eq:conv-bc}:
\[
  \mathcal{H}_c^{[0,1]}g
  = P(K_c * \tilde{g})
  = P(\Bc\tilde{g})
  = P\Bc P^* g,
\]
which is \eqref{eq:ftb-op}.

\textit{Step 3: Form identity.}
For $f \in L^2(\mathbb{R})$,
set $g = Pf$, $\tilde{g} = P^*Pf$:
\begin{align*}
  \ip{\Hc f}{f}
  &= \ip{P^*\mathcal{H}_c^{[0,1]}Pf}{f}
  = \ip{\mathcal{H}_c^{[0,1]}Pf}{Pf}
  = \ip{P\Bc P^* Pf}{Pf}
  = \ip{\Bc\tilde{g}}{\tilde{g}}
  = \|\Bc\tilde{g}\|^2,
\end{align*}
where the last step uses $\Bc^2 = \Bc$.
\end{proof}

\begin{remark}[Content of
  Lemma~\ref{lem:ftb}]
\label{rem:ftb}
The identity \eqref{eq:ftb-op}
is the Slepian--Pollak
time-band limiting structure:
restrict to $[0,1]$ via $P^*$,
apply frequency cutoff $[-c,c]$
via $\Bc$, restrict back via $P$.
The key feature used in Step~2
is that $\tilde g = P^*g$
vanishes outside $[0,1]$,
so the global convolution
$K_c * \tilde g$
restricts to the kernel integral
on $[0,1]$ without modification.
All subsequent lemmas are pure
functional analysis on
this structure.
\end{remark}

%--- Monotonicity

\begin{lemma}[Operator monotonicity]
\label{lem:monotone}
For $0 < c \le c'$ and all $f$:
$\ip{\Hc f}{f}
\le \ip{\mathcal{H}_{c'} f}{f}$.
\end{lemma}

\begin{proof}
By \eqref{eq:ftb}:
$\ip{\Hc f}{f} =
\|\Bc(P^*Pf)\|^2$.
Since $[-c,c] \subset [-c',c']$,
orthogonal projections satisfy
$\Bc \le B_{c'}$
(\cite{ReedSimon1975}, Ch.~VIII.1);
hence
$\|\Bc h\|^2 \le \|B_{c'}h\|^2$
for all $h$.
\end{proof}

%--- Hilbert-Schmidt

\begin{lemma}[Hilbert--Schmidt
  and compactness]
\label{lem:hs-compact}
For each fixed $c_0 > 0$,
$\Hc[c_0]$ is Hilbert--Schmidt,
hence compact, on $L^2(\mathbb{R})$.
\end{lemma}

\begin{proof}
We prove $\mathcal{H}_{c_0}^{[0,1]}$
is HS on $L^2([0,1])$;
compactness of
$\Hc[c_0] = P^*\mathcal{H}_{c_0}^{[0,1]}P$
follows since $P,P^*$ are bounded
(\cite{ReedSimon1975}, Thm.~VI.22).

The HS norm is computed directly
from the kernel:
\[
  \|\mathcal{H}_{c_0}^{[0,1]}
  \|_{\mathrm{HS}}^2
  = \int_0^1\int_0^1
    |K_{c_0}(x-y)|^2\,dx\,dy.
\]
Substituting $u = x - y$
(Fubini; integrand non-negative;
Jacobian $=1$):
\[
  = \int_{-1}^{1}(1-|u|)
    |K_{c_0}(u)|^2\,du
  \le \int_{-\infty}^{\infty}
    |K_{c_0}(u)|^2\,du.
\]
By Parseval
($\Fc K_{c_0} = \chi_{[-c_0,c_0]}$):
\[
  \int_{-\infty}^{\infty}
  |K_{c_0}(u)|^2\,du
  = \frac{1}{2\pi}
  \|\chi_{[-c_0,c_0]}\|_{L^2}^2
  = \frac{c_0}{\pi}.
\]
Hence
$\|\mathcal{H}_{c_0}^{[0,1]}
\|_{\mathrm{HS}}^2
\le c_0/\pi < \infty$.
\end{proof}

%--- Density

\begin{lemma}[Density of $\mathcal{D}$]
\label{lem:dense}
Define
$\mathcal{D} :=
\{f \in L^2(\mathbb{R})
: \Hc f \text{ converges in }
L^2 \text{ as } c \to \infty\}$.
Then $\mathcal{D}$ is dense in
$L^2(\mathbb{R})$.
\end{lemma}

\begin{proof}
It suffices to show
$C_c^\infty(\mathbb{R})
\subset \mathcal{D}$.

Let $f \in C_c^\infty$,
$\tilde{g} = P^*Pf
= \mathbf{1}_{[0,1]}f
\in L^2(\mathbb{R})$.
By \eqref{eq:ftb}:
$\Hc f =
P^*P(\Bc\tilde{g})$.

\textit{Strong convergence of $\Bc$.}
The Fourier projection $\Bc$
converges to the identity strongly:
for all $h \in L^2(\mathbb{R})$,
$\|\Bc h - h\|^2 =
\int_{|\xi|>c}|\hat h(\xi)|^2
\,\tfrac{d\xi}{2\pi} \to 0$
by dominated convergence
(\cite{ReedSimon1975}, Thm.~IX.14).
In particular,
$\Bc\tilde{g} \to \tilde{g}$
strongly.

\textit{Bounded operators are
strongly continuous.}
For any $A \in \mathcal{B}(L^2)$
and strongly convergent sequence
$h_n \to h$:
$\|Ah_n - Ah\|
\le \|A\|\cdot\|h_n - h\| \to 0$.
Applying with
$A = P^*P$, $\|P^*P\| \le 1$,
$h_c = \Bc\tilde{g}$, $h = \tilde{g}$:
\[
  \|\Hc f - P^*P\tilde{g}\|
  = \|P^*P(\Bc\tilde{g} - \tilde{g})\|
  \le \|\Bc\tilde{g} - \tilde{g}\|
  \to 0.
\]
Hence $f \in \mathcal{D}$.

\textit{Uniform bound.}
$\|\Hc f\| \le \|f\|$
for all $c$ by \eqref{F1};
no $c$-dependent estimate is used.
\end{proof}

\begin{remark}
Lemmas~\ref{lem:monotone}--%
\ref{lem:dense} all follow from
\eqref{eq:ftb}:
monotonicity via projection order;
HS compactness via Parseval;
density via $\Bc \nearrow I$
$+$ strong continuity of $P^*P$.
\end{remark}

\section{Strong-Limit Operator}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Extension of pointwise
  convergence]
\label{lem:sc}
Let $(A_c) \subset
\mathcal{B}(\mathcal{H})$ with
$\sup_c\|A_c\| \le M < \infty$,
and let $\mathcal{D}
\subset \mathcal{H}$ be dense.
If $A_c f \to Bf$ for all
$f \in \mathcal{D}$, then $A_c
\xrightarrow{\mathrm{SOT}} A$,
where $A \in \mathcal{B}(\mathcal{H})$
extends $B$ and satisfies
$\|A\| \le M$.

More precisely:
pointwise convergence on a dense set,
together with a uniform bound,
implies strong-operator-topology
convergence on all of $\mathcal{H}$.
\end{lemma}
\begin{proof}
For $f \in \mathcal{H}$ and
$\varepsilon > 0$,
choose $g \in \mathcal{D}$ with
$\|f - g\| < \varepsilon/(3M)$.
Then:
$\|A_c f - A_{c'} f\|
\le \|A_c(f-g)\|
+ \|A_c g - A_{c'} g\|
+ \|A_{c'}(g-f)\|$.
The outer terms are $\le 2M
\cdot \varepsilon/(3M)
= 2\varepsilon/3$;
the middle term $\to 0$ since
$g \in \mathcal{D}$.
Hence $(A_c f)$ is Cauchy;
$Af := \lim_c A_c f$ defines
$A \in \mathcal{B}(\mathcal{H})$,
$\|A\| \le M$
(\cite{ReedSimon1975}, Thm.~I.26).
\end{proof}

\begin{lemma}[Existence of
  $\Hstrong$]
\label{lem:hlim-bounded}
$\exists!\,\Hstrong \in
\mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hstrong\| \le 1$ and
$\Hc \xrightarrow{\mathrm{SOT}}
\Hstrong$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:sc} with
$\mathcal{D}$ from
Lemma~\ref{lem:dense},
$M = 1$ from \eqref{F1}.
\end{proof}

\begin{corollary}
\label{cor:qlim-op}
$\qlim(f,f)
= \ip{\Hstrong f}{f}$.
\end{corollary}
\begin{proof}
SOT convergence $+$ inner product
continuity.
\end{proof}

\begin{corollary}
\label{cor:sa-strong}
$\Hstrong = \Hstrong^* \ge 0$;
$\sigma(\Hstrong) \subset [0,1]$.
\end{corollary}
\begin{proof}
Symmetry by SOT limit;
\cite{ReedSimon1975}, Thm.~III.6;
positivity from $\qlim \ge 0$.
\end{proof}

\section{Limit Form and Mosco
  Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
$\qlim(f,f)
:= \sup_{c>0}\ip{\Hc f}{f}
= \lim_{c\to\infty}\ip{\Hc f}{f}$
(Lemma~\ref{lem:monotone}
$+$ \eqref{F1});
extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc)
\xrightarrow{\mathrm{M}} \qlim$ iff:
\begin{enumerate}[\rm(M1)]
\item For every $c_n \to \infty$
      and every weakly convergent
      $f_n \rightharpoonup f$:
      $\qlim(f,f)
      \le \liminf_n
      q_{c_n}(f_n,f_n)$.
\item For every $f$ there exist
      $c_n \to \infty$ and $f_n \to f$
      strongly such that
      $q_{c_n}(f_n,f_n) \to \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Weak-to-strong
  via compactness]
\label{lem:wc}
For fixed $c_0 > 0$ and any
weakly convergent sequence
$f_n \rightharpoonup f$:
\[
  q_{c_0}(f_n,f_n)
  \to q_{c_0}(f,f).
\]
\end{lemma}
\begin{proof}
Since $\Hc[c_0]$ is compact
(Lemma~\ref{lem:hs-compact}),
it maps weakly convergent sequences
to strongly convergent ones
(\cite{ReedSimon1975}, Thm.~VI.1):
$\Hc[c_0]f_n
\xrightarrow{\mathrm{SOT}}
\Hc[c_0]f$.
Hence
$\ip{\Hc[c_0]f_n}{f_n}
\to \ip{\Hc[c_0]f}{f}$
(strong-weak continuity of inner
product;
\cite{ReedSimon1975}, Lem.~II.3).
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc \xrightarrow{\mathrm{M}} \qlim$.
\end{theorem}

\begin{proof}
\textit{(M2).}
Set $f_n := f$, $c_n \to \infty$;
then $q_{c_n}(f,f)
\to \qlim(f,f)$
by Definition~\ref{def:qlim}.

\textit{(M1): two-step argument.}
Let $c_n \to \infty$ and
$f_n \rightharpoonup f$.

\textit{Step 1 (fix $c_0$).}
For any fixed $c_0 > 0$,
once $c_n \ge c_0$:
$q_{c_n}(f_n,f_n)
\ge q_{c_0}(f_n,f_n)$
by Lemma~\ref{lem:monotone}.

\textit{Step 2 (compact form
continuity).}
$\Hc[c_0]$ is compact, and
$f_n \rightharpoonup f$, so
$q_{c_0}(f_n,f_n)
\to q_{c_0}(f,f)$
by Lemma~\ref{lem:wc}.

\textit{Combining.}
From Steps 1--2:
$\liminf_n q_{c_n}(f_n,f_n)
\ge \liminf_n q_{c_0}(f_n,f_n)
= \lim_n q_{c_0}(f_n,f_n)
= q_{c_0}(f,f)$.
Taking $\sup_{c_0}$ and using
Definition~\ref{def:qlim}:
$\liminf_n q_{c_n}(f_n,f_n)
\ge \qlim(f,f)$.
\end{proof}

\section{Closedness and
  Riesz--Kato Identification}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of $\qlim$]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
$\qlim(f,f) = \ip{\Hstrong f}{f}$
(Corollary~\ref{cor:qlim-op}).
Since $\Hstrong$ is bounded:
$\|f\|_{\qlim}^2
:= \|f\|^2 + \qlim(f,f)
= \|f\|^2
+ \ip{\Hstrong f}{f}
\le (1 + \|\Hstrong\|)\|f\|^2
\le 2\|f\|^2$.
Hence the form norm and the
$L^2$-norm are equivalent,
so $\qlim$ is closed.
\end{proof}

\begin{theorem}[Riesz--Kato
  identification]
\label{thm:friedrichs}
$T_{\qlim} = \Hstrong$.
\end{theorem}
\begin{proof}
$\ip{\Hstrong f}{g}
= \qlim(f,g)$ for all $f,g$
(Corollary~\ref{cor:qlim-op});
uniqueness of the Riesz
representative.
\end{proof}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
For all $f$ and $c \ge c_0(\kappa)$:
$0 \le \qlim(f,f) - \qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{theorem}
\begin{proof}[Proof (external)]
Lower bound:
Lemma~\ref{lem:monotone}.
Upper bound:
Paper~VIII, Cor.~4.3:
$\|\Hstrong - \Hc\|
\le C_\kappa c^{-1/4}$;
apply to $\ip{(\Hstrong-\Hc)f}{f}$.
\end{proof}

\begin{remark}[Obstruction to
  independent proof]
\label{rem:no-indep-proof}
Any independent proof requires
uniform eigenvector asymptotics;
expressions mixing fixed-$c$ basis
with $c \to \infty$ weights are
not valid.
See Problem~\ref{prob:rate}.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the sole import
point for Paper~IX.

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      Paper~IX, Def.~2.3:
      $\Hspectral f
      := \sum_n
      \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}}
      \Phi_n^{(\infty)}$;
      densely defined, symmetric,
      closable (Paper~IX,
      Thms.~2.5, 3.2).

\item\label{IXb}
      \textit{Bounded, everywhere-defined
      closure.}
      On $\Dom(\Hspectral)$:
      $\|\Hspectral f\|^2
      = \sum_n
      (\lambda_n^{(\infty)})^2
      |\ip{f}{\Phi_n^{(\infty)}}|^2
      \le \|f\|^2$.
      Hence $\overline{\Hspectral}
      \in \mathcal{B}(L^2)$,
      $\|\overline{\Hspectral}\|
      \le 1$, everywhere defined
      (\cite{ReedSimon1975},
      Thm.~I.7).
      This is the only point where
      Paper~IX is invoked;
      the argument depends on the
      Bessel inequality and on
      $\lambda_n^{(\infty)}
      \in [0,1]$, which are
      established in Paper~IX.

\item\label{IXc}
      \textit{Agreement on
      $\mathcal{D}$.}
      Both $\Hstrong$ and
      $\overline{\Hspectral}$
      are bounded and everywhere
      defined.
      For $f \in \mathcal{D}$:
      $\Hc f \to \Hstrong f$
      (Lem.~\ref{lem:hlim-bounded});
      $\Hc f \to
      \overline{\Hspectral}f$
      (Paper~IX, Rem.~2.6).
      Uniqueness of limits gives
      $\Hstrong =
      \overline{\Hspectral}$
      on $\mathcal{D}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong =
\overline{\Hspectral}$
in $\mathcal{B}(L^2(\mathbb{R}))$.
\end{theorem}
\begin{proof}
Both are bounded, everywhere defined
(\eqref{IXb};
Lem.~\ref{lem:hlim-bounded}).
They agree on the dense set
$\mathcal{D}$ (\eqref{IXc}).
Two bounded operators agreeing on
a dense subset are equal
(\cite{ReedSimon1975}, Lem.~II.3).
\end{proof}

\begin{remark}
$\Hlim
:= \Hstrong
= \overline{\Hspectral}$.
\end{remark}

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every $z \notin [0,1]$:
$(\Hc - z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim - z)^{-1}$.
\end{corollary}
\begin{proof}
Mosco (Theorem~\ref{thm:mosco});
$\sigma(\Hlim) \subset [0,1]$
(Corollary~\ref{cor:sa-strong});
\cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
\label{rem:uncond}
The following hold without
hypothesis:
$\Hlim = \Hlim^* \ge 0$;
$\|\Hlim\|\le 1$;
$T_{\qlim} = \Hlim$;
$\qc \xrightarrow{\mathrm{M}} \qlim$;
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$;
$\Hstrong = \overline{\Hspectral}$.
External (Paper~VIII):
rate $O(c^{-1/4})$.
\end{remark}

\begin{remark}[Speculative and
  programmatic framework:
  connection to Riemann zeros]
\label{rem:rh}
This remark describes a
\emph{speculative, programmatic}
framework.
It contains no proved result;
no implication from the preceding
theorems to zeta zeros is claimed.

Two hypotheses, both unverified,
would together give a conditional
consequence:
\begin{enumerate}[(H1)]
\item Paper~IX, Hyp.~4.1:
      $\{\Phi_n^{(\infty)}\}$ complete;
      $\Hlim$ essentially
      self-adjoint.
\item \cite{CCM2025}, Thm.~1.3:
      non-trivial zeros $\rho$ of
      $\zeta(s)$ lie in
      $\sigma_{\mathrm{app}}(\Hlim)$.
\end{enumerate}
Under (H1)--(H2), the spectral
inclusion would give
$\rho \in \sigma(\Hlim)
\subset \mathbb{R}$,
forcing
$\mathrm{Re}(\rho) = \tfrac{1}{2}$.

However:
(H2) currently has no operator
identification that connects the
zeta zeros directly to $\Hlim$;
it is an assumption borrowed from
the CCM spectral program, for which
no rigorous bridge to the present
operator is established.
A reviewer should read this remark
as motivational context only,
not as a mathematical contribution
of this paper.
\end{remark}
