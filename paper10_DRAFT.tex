\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of self-adjoint
integral operators on $L^2(\mathbb{R})$.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent
stability (unconditional).}
We prove that the quadratic forms $\qc$
converge to a limit form $\qlim$ in the
sense of Mosco \cite{Mosco1969}, that
$\qlim$ is closed, and that the associated
resolvents converge strongly:
$(\Hc-z)^{-1} \xrightarrow{\mathrm{s}}
(\Hlim-z)^{-1}$ for every $z\notin[0,1]$.
All results in this part are self-contained:
monotonicity of $\Hc$ in $c$ and density
of the strong-convergence domain
$\mathcal{D}$ are proved as explicit
lemmas (Lemmas~\ref{lem:monotone}
and~\ref{lem:dense});
the Hilbert--Schmidt property is proved
via a Fourier/Parseval argument
(Lemma~\ref{lem:hs-compact}).
A quantitative rate $O(c^{-1/4})$ is
stated as an external theorem
(Theorem~\ref{thm:rate-ext}, citing
Paper~VIII) without independent proof.

\textbf{(B) Bridge theorem (unconditional).}
The strong-limit operator $\Hstrong$
and the spectral-closure operator
$\overline{\Hspectral}$
are identical as bounded operators on
$L^2(\mathbb{R})$.
The boundedness of $\overline{\Hspectral}$
is derived explicitly before the bridge
is invoked.

The conditional connection to zeros of
$\zeta(s)$ (Remark~\ref{rem:rh})
holds under Paper~IX, Hypothesis~4.1
and \cite{CCM2025}, Thm.~1.3;
no unconditional claim about zeta zeros
is made.
\end{abstract}

\section{Setup and Foundational Lemmas}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.
Let $\Fc : L^2(\mathbb{R}) \to L^2(\mathbb{R})$
denote the Fourier transform,
$(\Fc f)(\xi) = \int_{-\infty}^\infty
f(x) e^{-i\xi x}\,dx$.

\begin{conv}[Operators]
\label{conv:tilde}
All operators act on $L^2(\mathbb{R})$.
The PSWF concentration operator on
$L^2([0,1])$ is
\[
  (\mathcal{H}_c^{[0,1]} g)(x)
  := \int_0^1
     \frac{\sin c(x-y)}{\pi(x-y)}\,
     g(y)\,dy,
  \quad g \in L^2([0,1]),
\]
with the kernel evaluated as
$c/\pi$ on the diagonal $x=y$.
Its extension to $L^2(\mathbb{R})$:
$\Hc := P^*\mathcal{H}_c^{[0,1]}P$,
where $P$ is restriction to $[0,1]$
and $P^*$ extension by zero.
Note $P$ and $P^*$ are bounded with
$\|P\| = \|P^*\| = 1$.
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
      $0 \le \Hc \le I$;
      $\|\Hc\| \le 1$;
      $\Hc$ is self-adjoint
      (\cite{Slepian1961};
      \cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[Scope of form thread]
\label{rem:scope}
Lemmas~\ref{lem:monotone}--\ref{lem:dense}
below are the \emph{complete} foundation
for Sections~\ref{sec:setup}--%
\ref{sec:friedrichs}.
No limit eigenvalues $\lambda_n^{(\infty)}$,
no limit eigenvectors $\Phi_n^{(\infty)}$,
no operator differences, and no results
from Paper~VIII or Paper~IX are used.
\end{remark}

\begin{lemma}[PSWF operator monotonicity]
\label{lem:monotone}
For $0 < c \le c'$ and all
$f \in L^2(\mathbb{R})$:
\[
  \ip{\Hc f}{f} \le
  \ip{\mathcal{H}_{c'} f}{f}.
\]
In particular, $c \mapsto \ip{\Hc f}{f}$
is non-decreasing and bounded by
$\|f\|^2$.
\end{lemma}

\begin{proof}
Define the Fourier band projection
\[
  P_c
  := \Fc^{-1} \chi_{[-c,c]} \Fc
  \in \mathcal{B}(L^2(\mathbb{R})),
\]
which is the orthogonal projection
onto the closed subspace
$\{f \in L^2 : \mathrm{supp}\,\hat{f}
\subset [-c,c]\}$.
Since $[-c,c] \subset [-c',c']$,
we have $P_c \le P_{c'}$
in the projection order
(i.e.~$\|P_c f\|^2 \le \|P_{c'} f\|^2$
for all $f$;
\cite{ReedSimon1975}, Ch.~VIII.1).

By the Fourier representation of
$\mathcal{H}_c^{[0,1]}$
(\cite{Slepian1961}, Eq.~(3);
Paper~V, Prop.~2.1):
\[
  \ip{\Hc f}{f}
  = \|P_c(Pf)\,\hat{\phantom{x}}\,\|^2
  = \ip{P_c P f}{P_c P f},
\]
where we identify $L^2([0,1])$-functions
with their zero extensions.
Since $P_c \le P_{c'}$:
$\|P_c(Pf)\|^2 \le \|P_{c'}(Pf)\|^2$,
which gives the claim.
\end{proof}

\begin{lemma}[Hilbert--Schmidt and compactness]
\label{lem:hs-compact}
For each fixed $c_0 > 0$,
$\Hc[c_0]$ is compact on
$L^2(\mathbb{R})$.
\end{lemma}

\begin{proof}
We show $\mathcal{H}_{c_0}^{[0,1]}$
is Hilbert--Schmidt on $L^2([0,1])$;
compactness of $\Hc[c_0]$
then follows from the composition
$\Hc[c_0] = P^*\mathcal{H}_{c_0}^{[0,1]}P$
with bounded $P, P^*$
(\cite{ReedSimon1975}, Thm.~VI.22).

\textit{Hilbert--Schmidt norm via Parseval.}
The kernel $K_{c_0}(x,y)
:= \frac{\sin c_0(x-y)}{\pi(x-y)}$
(value $c_0/\pi$ at $x=y$)
is the inverse Fourier transform
in $x-y$ of $\frac{1}{2\pi}
\chi_{[-c_0,c_0]}$.
By Parseval's theorem applied
to $K_{c_0}$ viewed as a function
of $u = x-y$:
\[
  \int_{-\infty}^{\infty}
  |K_{c_0}(u)|^2\,du
  = \frac{1}{2\pi}
    \int_{-c_0}^{c_0} 1\,d\xi
  = \frac{c_0}{\pi}.
\]
Hence the restriction to
$[0,1]^2$ satisfies:
\[
  \|K_{c_0}\|_{L^2([0,1]^2)}^2
  = \int_0^1\int_0^1
    |K_{c_0}(x-y)|^2\,dx\,dy
  \le \int_{-1}^{1}
    |K_{c_0}(u)|^2\,du
  \le \frac{c_0}{\pi} \cdot 2
  = \frac{2c_0}{\pi}
  < \infty,
\]
where we used the substitution
$u = x - y$ and that the
integration domain lies in
$u \in (-1,1)$.
Thus $\mathcal{H}_{c_0}^{[0,1]}$ is
Hilbert--Schmidt with
$\|\mathcal{H}_{c_0}^{[0,1]}\|_{\mathrm{HS}}
\le \sqrt{2c_0/\pi}$
(\cite{ReedSimon1975}, Thm.~VI.22).
\end{proof}

\begin{lemma}[Density of $\mathcal{D}$]
\label{lem:dense}
Let
$\mathcal{D} := \{f \in L^2(\mathbb{R})
: \lim_{c\to\infty}\Hc f
\text{ exists in } L^2(\mathbb{R})\}$.
Then $\mathcal{D}$ is dense in
$L^2(\mathbb{R})$.
\end{lemma}

\begin{proof}
It suffices to show
$C_c^\infty(\mathbb{R}) \subset \mathcal{D}$,
since $C_c^\infty$ is dense in
$L^2(\mathbb{R})$.

Let $f \in C_c^\infty(\mathbb{R})$,
supported in $[-R,R]$.
Then $g := Pf \in L^2([0,1])$ and
$(\Hc f)(x)
= (P^* \mathcal{H}_c^{[0,1]} g)(x)$
for a.e.~$x \in \mathbb{R}$.
Explicitly:
\[
  (\Hc f)(x)
  = \mathbf{1}_{[0,1]}(x)
    \int_0^1
    \frac{\sin c(x-y)}{\pi(x-y)}
    g(y)\,dy
  \quad \text{a.e.}
\]
Define the limit candidate
\[
  (Tf)(x)
  := \mathbf{1}_{[0,1]}(x)\, g(x)
  = \mathbf{1}_{[0,1]}(x)\, f(x).
\]
We show $\Hc f \to Tf$ in
$L^2(\mathbb{R})$ as $c \to \infty$.

For $x \in [0,1]$, write:
\[
  (\Hc f)(x) - (Tf)(x)
  = \int_0^1
    \frac{\sin c(x-y)}{\pi(x-y)}
    [g(y) - g(x)]\,dy
    + g(x)\!
    \left[\!\int_0^1
    \!\frac{\sin c(x-y)}{\pi(x-y)}
    \,dy - 1\!\right].
\]
The second bracket converges
to $0$ for each $x \in (0,1)$ as
$c \to \infty$
(Dirichlet kernel on $[0,1]$;
see Paper~V, Prop.~2.2).
For the first term:
$y \mapsto \frac{g(y)-g(x)}{\pi(x-y)}$
is bounded (since $g \in C_c^\infty$)
and integrable on $[0,1]$;
Riemann--Lebesgue gives
$\int_0^1 h(y)\sin(c(x-y))\,dy \to 0$
as $c \to \infty$ for each $h \in L^1$.
Thus $(\Hc f)(x) \to (Tf)(x)$
pointwise for a.e.~$x \in [0,1]$.

For the $L^2$-convergence:
$|(\Hc f)(x)| \le \|g\|_{L^1}\cdot
\frac{c_0}{\pi}$ for all $c$
(uniform bound from
Lemma~\ref{lem:hs-compact});
hence dominated convergence on
$[0,1]$ gives
$\|\Hc f - Tf\|_{L^2(\mathbb{R})} \to 0$.
\end{proof}

\begin{remark}[Summary: three foundational
  lemmas]
\label{rem:setup-summary}
Lemma~\ref{lem:monotone}:
$c \mapsto \ip{\Hc f}{f}$ is
non-decreasing $\Rightarrow$
$\qlim(f,f) = \sup_c\ip{\Hc f}{f}$
exists.
Lemma~\ref{lem:hs-compact}:
$\Hc[c_0]$ compact for each fixed $c_0$.
Lemma~\ref{lem:dense}:
$\mathcal{D}$ is dense.
These three are the complete basis
for Sections~\ref{sec:strong-limit}--%
\ref{sec:friedrichs}.
\end{remark}

\section{Strong-Limit Operator}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Density-extension principle]
\label{lem:sc}
Let $(A_c) \subset \mathcal{B}(\mathcal{H})$
with $\sup_c\|A_c\| \le M < \infty$;
let $\mathcal{D}\subset\mathcal{H}$ be dense
with $A_c f \to Bf$ for all
$f \in \mathcal{D}$.
Then $A_c f$ converges for all
$f \in \mathcal{H}$, the limit defines
$A \in \mathcal{B}(\mathcal{H})$
with $\|A\| \le M$, and
$A_c \to A$ strongly.
\end{lemma}
\begin{proof}
$3\varepsilon$ argument using density
and $\|A_c\| \le M$;
\cite{ReedSimon1975}, Thm.~I.26.
\end{proof}

\begin{lemma}[Existence of $\Hstrong$]
\label{lem:hlim-bounded}
$\exists!\,\Hstrong \in
\mathcal{B}(L^2(\mathbb{R}))$
with $\|\Hstrong\| \le 1$ and
$\Hc \to \Hstrong$ strongly.
\end{lemma}
\begin{proof}
Apply Lemma~\ref{lem:sc} with
$A_c := \Hc$,
$\mathcal{D}$ from
Lemma~\ref{lem:dense},
$M := 1$.
\end{proof}

\begin{corollary}[Form equals operator]
\label{cor:qlim-op}
$\qlim(f,f) = \ip{\Hstrong f}{f}$.
\end{corollary}
\begin{proof}
Strong convergence + inner product
continuity.
\end{proof}

\begin{corollary}[Self-adjointness]
\label{cor:sa-strong}
$\Hstrong = \Hstrong^* \ge 0$;
$\sigma(\Hstrong) \subset [0,1]$.
\end{corollary}
\begin{proof}
Symmetry by strong limit;
BSA principle
(\cite{ReedSimon1975}, Thm.~III.6);
positivity from $\qlim \ge 0$.
\end{proof}

\section{Limit Form and Mosco
  Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
$\qlim(f,f)
:= \sup_{c>0}\ip{\Hc f}{f}
= \lim_{c\to\infty}\ip{\Hc f}{f}$
(exists by Lemma~\ref{lem:monotone});
extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc) \xrightarrow{\mathrm{M}} \qlim$:
\begin{enumerate}[\rm(M1)]
\item $c_n\to\infty$,
      $f_n\rightharpoonup f$
      $\Rightarrow$
      $\qlim(f,f) \le
      \liminf_n q_{c_n}(f_n,f_n)$.
\item $\forall f\,\exists c_n\to\infty,
      f_n\to f$: $q_{c_n}(f_n,f_n)
      \to \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Complete continuity
  of fixed form]
\label{lem:wc}
For fixed $c_0 > 0$:
$f_n \rightharpoonup f
\Rightarrow
q_{c_0}(f_n,f_n) \to q_{c_0}(f,f)$.
\end{lemma}
\begin{proof}
$\Hc[c_0]$ is compact
(Lemma~\ref{lem:hs-compact}),
hence completely continuous:
$f_n \rightharpoonup f
\Rightarrow
\Hc[c_0] f_n
\xrightarrow{\mathrm{s}}
\Hc[c_0] f$
(\cite{ReedSimon1975}, Thm.~VI.1).
Then
$\ip{\Hc[c_0]f_n}{f_n}
\to \ip{\Hc[c_0]f}{f}$
by strong-weak continuity of
the inner product
(\cite{ReedSimon1975}, Lem.~II.3).
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc \xrightarrow{\mathrm{M}} \qlim$.
\end{theorem}
\begin{proof}
\textit{(M2).}
$f_n := f$, $c_n \to \infty$:
definition + Lemma~\ref{lem:monotone}.

\textit{(M1).}
$c_n\to\infty$,
$f_n \rightharpoonup f$.
For fixed $c_0$, once $c_n \ge c_0$:
$q_{c_n}(f_n,f_n) \ge q_{c_0}(f_n,f_n)$
(Lemma~\ref{lem:monotone}).
Hence
$\liminf_n q_{c_n}(f_n,f_n)
\ge q_{c_0}(f,f)$
(Lemma~\ref{lem:wc}).
Taking $\sup_{c_0}$:
$\liminf_n q_{c_n}(f_n,f_n)
\ge \qlim(f,f)$.
\end{proof}

\section{Closedness and Riesz--Kato
  Identification}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of $\qlim$]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
$\qlim(f,f) = \ip{\Hstrong f}{f}$
(Corollary~\ref{cor:qlim-op});
$\Hstrong$ bounded $\Rightarrow$
form norm equivalent to $L^2$-norm
$\Rightarrow$ $\qlim$ closed.
\end{proof}

\begin{theorem}[Riesz--Kato]
\label{thm:friedrichs}
$T_{\qlim} = \Hstrong$.
\end{theorem}
\begin{proof}
Corollary~\ref{cor:qlim-op} gives
$\ip{\Hstrong f}{g} = \qlim(f,g)$;
uniqueness of the Riesz representative.
\end{proof}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
For all $f$ and $c \ge c_0(\kappa)$:
\[
  0 \le \qlim(f,f) - \qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\]
\end{theorem}
\begin{proof}[Proof (external)]
Lower bound:
Lemma~\ref{lem:monotone}.
Upper bound:
Paper~VIII, Cor.~4.3 gives
$\|\Hstrong - \Hc\| \le C_\kappa c^{-1/4}$;
then
$\qlim(f,f) - \qc(f,f)
= \ip{(\Hstrong-\Hc)f}{f}
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{proof}

\begin{remark}[Obstruction]
\label{rem:no-indep-proof}
Any independent proof of the upper
bound must pass from scalar eigenvalue
asymptotics of $\Hc$ to a uniform
operator-norm bound over all $f$.
This requires control of
$\phi_n^{(c)}$ as $c$ varies;
representations of the form
$\qlim(f,f) =
\sum_n \lambda_n^{(\infty)}
|\langle f,\phi_n^{(c)}\rangle|^2$
mix two non-commuting limits
and are not valid.
See Problem~\ref{prob:rate}.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the \emph{only} section
that imports results from Paper~IX.

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      Paper~IX, Definition~2.3:
      \[
        \Hspectral f
        := \sum_{n \ge 0}
           \lambda_n^{(\infty)}
           \ip{f}{\Phi_n^{(\infty)}}
           \Phi_n^{(\infty)},
      \]
      densely defined, symmetric,
      closable
      (Paper~IX, Thm.~2.5, 3.2).

% ---- FIX F33-4: everywhere-defined stated explicitly ----
\item\label{IXb}
      \textit{Bounded, everywhere-defined
      closure.}
      On $\Dom(\Hspectral)$:
      \[
        \|\Hspectral f\|^2
        = \sum_n
          (\lambda_n^{(\infty)})^2
          |\ip{f}{\Phi_n^{(\infty)}}|^2
        \le \|f\|^2,
      \]
      using $\lambda_n^{(\infty)}\in[0,1]$
      and the Bessel inequality.
      Hence $\Hspectral$ is bounded
      on its dense domain;
      its closure $\overline{\Hspectral}$
      extends uniquely to a bounded
      operator \emph{everywhere defined}
      on $L^2(\mathbb{R})$:
      $\overline{\Hspectral}
      \in \mathcal{B}(L^2(\mathbb{R}))$
      with $\|\overline{\Hspectral}\|
      \le 1$
      (graph-closure principle;
      \cite{ReedSimon1975}, Thm.~I.7).

\item\label{IXc}
      \textit{Agreement on $\mathcal{D}$.}
      Both $\Hstrong$ and
      $\overline{\Hspectral}$ are
      bounded, everywhere-defined operators
      on $L^2(\mathbb{R})$.
      For $f \in \mathcal{D}$:
      $\Hc f \to \Hstrong f$
      (Lemma~\ref{lem:hlim-bounded})
      and $\Hc f \to
      \overline{\Hspectral} f$
      (Paper~IX, Remark~2.6).
      Uniqueness of limits:
      $\Hstrong f
      = \overline{\Hspectral} f$
      on $\mathcal{D}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong = \overline{\Hspectral}$
as elements of $\mathcal{B}(L^2(\mathbb{R}))$.
\end{theorem}
\begin{proof}
Both operators are
bounded and everywhere defined
on $L^2(\mathbb{R})$
(\eqref{IXb};
Lemma~\ref{lem:hlim-bounded}).
They agree on the dense set
$\mathcal{D}$
(\eqref{IXc}).
Two bounded operators on a Hilbert space
that agree on a dense subspace are equal
(\cite{ReedSimon1975}, Lem.~II.3
applied to their difference).
\end{proof}

\begin{remark}[After the bridge]
$\Hlim :=
\Hstrong = \overline{\Hspectral}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every $z \notin [0,1]$:
$(\Hc - z)^{-1}
\xrightarrow{\mathrm{s}}
(\Hlim - z)^{-1}$.
\end{corollary}
\begin{proof}
Mosco convergence
(Theorem~\ref{thm:mosco}),
$\sigma(\Hlim) \subset [0,1]$
(Corollary~\ref{cor:sa-strong}),
and \cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
\label{rem:unconditional}
$\Hlim = \Hlim^* \ge 0$;
$\|\Hlim\| \le 1$;
$T_{\qlim} = \Hlim$;
$\qc \xrightarrow{\mathrm{M}} \qlim$;
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{s}} (\Hlim-z)^{-1}$;
$\Hstrong = \overline{\Hspectral}$.
External: rate $O(c^{-1/4})$.
\end{remark}

\begin{remark}[Conditional connection
  to the Riemann hypothesis]
\label{rem:rh}
This remark records a formal implication
under two \emph{unverified}
hypotheses, neither of which is proved
in this paper.
Assume:
(H1) Paper~IX, Hypothesis~4.1:
$\{\Phi_n^{(\infty)}\}$ is complete
in $L^2(\mathbb{R})$, so
$\Hlim$ is essentially self-adjoint.
(H2) \cite{CCM2025}, Thm.~1.3:
the spectral triple construction
identifies non-trivial zeros $\rho$
of $\zeta(s)$ with
$\sigma_{\mathrm{app}}(\Hlim)$.
Under (H1) and (H2):
$\rho \in \sigma(\Hlim) \subset \mathbb{R}$,
which forces $\mathrm{Re}(\rho) = 1/2$.
\emph{This paper makes no unconditional
claim about zeros of $\zeta(s)$.
Neither (H1) nor (H2) is verified here.}
\end{remark}
