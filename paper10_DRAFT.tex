\begin{abstract}
We study the concentration operators
$\Hc = P^*\mathcal{H}_c^{[0,1]}P$,
where $\mathcal{H}_c^{[0,1]}$ is the
classical Slepian--Pollak
time-band-limiting operator on
$L^2([0,1])$.
The family $\{\Hc\}_{c>0}$ consists
of uniformly bounded,
monotonically increasing,
compact self-adjoint operators on
$L^2(\mathbb{R})$.

The foundational identity
(Lemma~\ref{lem:ftb})
\[
  \ip{\Hc f}{f} = \|\Bc(P^*Pf)\|^2
\]
reduces the entire operator family to
a Fourier band projection conjugated
by time-restriction.
All results in this paper are
consequences of this single identity
and the inclusion
$[-c,c]\subset[-c',c']$ for $c\le c'$;
we call it the \emph{coordinatisation
lemma} of the theory.

\textbf{Dual convergence structure.}
The identity gives rise to two
concurrent convergence threads:
\begin{enumerate}[(A)]
\item \textbf{Operator thread
      (SOT):}
      Uniform boundedness and density
      (Lemma~\ref{lem:dense})
      yield, via the SOT extension
      principle (Lemma~\ref{lem:sc}),
      a unique
      $\Hc\xrightarrow{\mathrm{SOT}}
      \Hstrong$.
\item \textbf{Form thread
      (Mosco):}
      Operator monotonicity
      and Hilbert--Schmidt compactness
      yield Mosco convergence
      $\qc\xrightarrow{\mathrm{M}}\qlim$.
\end{enumerate}
The two threads share a common origin
(the coordinatisation lemma) but are
\emph{methodologically independent}:
Thread~(A) uses uniform boundedness
and Cauchy nets;
Thread~(B) uses monotonicity and
compactness.
They converge to the same object via
$T_{\qlim}=\Hstrong$
(Theorem~\ref{thm:friedrichs}),
yielding strong resolvent convergence
(Corollary~\ref{cor:resolvent}).

\textbf{Bridge theorem.}
For a spectrally constructed
$\Hspectral$ (Paper~IX),
$\Hstrong=\overline{\Hspectral}$.
Both sides are SOT limits of the same
directed family $(\Hc)_{c>0}$ in the
same SOT topology;
SOT limits are unique (Hausdorff),
so equality follows without
eigenfunction analysis or
dense-set comparison.

\textbf{Scope.}
This paper is pure operator-limit
theory: no properties of limiting
eigenfunctions are proved, no
unconditional connection to
$\zeta(s)$-zeros is made.
\end{abstract}

\section{Setup and Foundational Lemmas}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

Let $\Fc$ denote the Fourier transform
$(\Fc f)(\xi)=
\int_{-\infty}^\infty f(x)
e^{-i\xi x}\,dx$,
with Plancherel isometry
$\|\Fc f\|_{L^2}=\sqrt{2\pi}\|f\|_{L^2}$
(\cite{ReedSimon1975}, Ch.~IX).

\begin{conv}[Operators]
\label{conv:ops}
\begin{alignat}{2}
  \Bc&:=\Fc^{-1}\chi_{[-c,c]}\Fc
  &&\;(\Bc^*=\Bc=\Bc^2),\\
  P&:L^2(\mathbb{R})\to L^2([0,1]),
  &&\;\text{restriction; }\|P\|=1,\\
  P^*&:L^2([0,1])\to L^2(\mathbb{R}),
  &&\;\text{extension by zero;
  }\|P^*\|=1,\\
  \mathcal{H}_c^{[0,1]}
  &:L^2([0,1])\to L^2([0,1]),
  &&\;K_c(x,y)
  =\tfrac{\sin c(x-y)}{\pi(x-y)},\\
  \Hc&:=P^*\mathcal{H}_c^{[0,1]}P
  &&\;\in\mathcal{B}(L^2(\mathbb{R})).
\end{alignat}
$K_c(x,x):=c/\pi$.
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
$\Hc$ is self-adjoint,
$0\le\Hc\le I$,
$\|\Hc\|\le 1$
(\cite{Slepian1961};
\cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[Scope]
\label{rem:scope}
Sections~\ref{sec:setup}--%
\ref{sec:friedrichs} use only
\eqref{F1} and
Lemmas~\ref{lem:ftb}--\ref{lem:dense}.
\end{remark}

%── lem:ftb
\begin{lemma}[Coordinatisation lemma]
\label{lem:ftb}
For all $c>0$, $f\in L^2(\mathbb{R})$:
\begin{align}
  \label{eq:ftb}
  \ip{\Hc f}{f}
  &=\|\Bc(P^*Pf)\|^2,\\
  \label{eq:ftb-op}
  \mathcal{H}_c^{[0,1]}
  &=P\,\Bc\,P^*.
\end{align}
Every property of $\Hc$ derived below
follows from \eqref{eq:ftb} and
the inclusion
$[-c,c]\subset[-c',c']$ for $c\le c'$.
\end{lemma}

\begin{proof}
\textit{Step~1: Convolution identity.}
$K_c(u)=\frac{\sin cu}{\pi u}
\leftrightarrow
(\Fc K_c)(\xi)=\chi_{[-c,c]}(\xi)$
(\cite{Slepian1961}, Eq.~(3);
\cite{ReedSimon1975}, Ex.~IX.7).
Convolution theorem:
\begin{equation}\label{eq:conv-bc}
  K_c*h=\Bc h
  \;\forall h\in L^2(\mathbb{R}).
\end{equation}

\textit{Step~2: Restriction.}
For $g\in L^2([0,1])$,
$\tilde g:=P^*g$.
Since $\tilde g=0$ a.e.\ outside
$[0,1]$:
$(K_c*\tilde g)(x)
=\int_0^1 K_c(x-y)g(y)\,dy$,
giving $\mathcal{H}_c^{[0,1]}g
=P\Bc P^*g$, i.e., \eqref{eq:ftb-op}.

\textit{Step~3.}
$\ip{\Hc f}{f}
=\ip{P\Bc P^*Pf}{Pf}
=\|\Bc(P^*Pf)\|^2$.
\end{proof}

%── lem:monotone
\begin{lemma}[Monotonicity ---
  Thread~B input]
\label{lem:monotone}
$0<c\le c'\Rightarrow
\ip{\Hc f}{f}\le
\ip{\mathcal{H}_{c'}f}{f}$.
\end{lemma}
\begin{proof}
$[-c,c]\subset[-c',c']\Rightarrow
\Bc\le B_{c'}$;
apply \eqref{eq:ftb}.
\end{proof}

%── lem:hs-compact
\begin{lemma}[Hilbert--Schmidt ---
  Thread~B input]
\label{lem:hs-compact}
For fixed $c_0>0$,
$\Hc[c_0]$ is Hilbert--Schmidt,
hence compact.
\end{lemma}
\begin{proof}
$\|\mathcal{H}_{c_0}^{[0,1]}
\|_{\mathrm{HS}}^2
\le c_0/\pi$
(Fubini $+$ Parseval);
compactness:
\cite{ReedSimon1975}, Thm.~VI.22.
\end{proof}

%── lem:dense
\begin{lemma}[Dense domain ---
  Thread~A input]
\label{lem:dense}
$\mathcal{D}:=\{f\in L^2:
\Hc f\to\text{ in }L^2
\text{ as }c\to\infty\}
\supset C_c^\infty$,
hence $\mathcal{D}$ is dense.
\end{lemma}
\begin{proof}
For $f\in C_c^\infty$,
$\tilde g:=P^*Pf$:
$\|\Hc f-P^*P\tilde g\|
\le\|\Bc\tilde g-\tilde g\|\to 0$
(\cite{ReedSimon1975}, Thm.~IX.14;
stability under bounded $P^*P$).
\end{proof}

\begin{remark}[Logical status]
\label{rem:dense-trivial}
Once
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$
is established,
$\mathcal{D}=L^2$ automatically.
Lemma~\ref{lem:dense} is input to
Thread~A only.
\end{remark}

\section{Thread~A: Strong Operator
  Limit}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Thread~A profile]
\label{rem:threadA}
Inputs: \eqref{F1}, Lem.~\ref{lem:dense}.
Tool: Cauchy nets in complete Banach
space (uniform bound $+$ density).
Output:
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$.
\end{remark}

\begin{lemma}[SOT extension principle]
\label{lem:sc}
Let $(A_c)_{c>0}$ be a
\emph{directed net}
(directed by $c\to\infty$) in
$\mathcal{B}(\mathcal{H})$
with $M:=\sup_c\|A_c\|<\infty$.
Let $\mathcal{D}\subset\mathcal{H}$
be dense with
$\lim_c A_c g$ existing for every
$g\in\mathcal{D}$.
Then there exists a unique
$A\in\mathcal{B}(\mathcal{H})$
with $\|A\|\le M$ and
$A_c\xrightarrow{\mathrm{SOT}}A$.
\end{lemma}

\begin{proof}
Fix $f$, $\varepsilon>0$;
choose $g\in\mathcal{D}$ with
$\|f-g\|<\varepsilon/(3M)$.
For $c,c'>c_0(g,\varepsilon)$ in
the directed set $(0,\infty)$:
\[
  \|A_c f-A_{c'}f\|
  \le 2M\|f-g\|+\|A_c g-A_{c'}g\|
  <\varepsilon.
\]
$(A_c f)$ Cauchy in complete $\mathcal{H}$;
$Af:=\lim_c A_c f$ exists.
Linearity, $\|A\|\le M$, uniqueness:
\cite{ReedSimon1975}, Thm.~I.26.
\end{proof}

\begin{lemma}[Thread~A conclusion]
\label{lem:hlim-bounded}
$\exists!\,\Hstrong\in\mathcal{B}(L^2)$
with $\|\Hstrong\|\le 1$ and
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$.
\end{lemma}
\begin{proof}
Lem.~\ref{lem:sc};
$M=1$; $\mathcal{D}$ from
Lem.~\ref{lem:dense}.
\end{proof}

\begin{corollary}
\label{cor:qlim-op}
$\qlim(f,f)=\ip{\Hstrong f}{f}$.
\end{corollary}

\begin{corollary}
\label{cor:sa-strong}
$\Hstrong=\Hstrong^*\ge 0$;
$\sigma(\Hstrong)\subset[0,1]$.
\end{corollary}

\section{Thread~B: Mosco Convergence
  of Forms}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Thread~B profile]
\label{rem:threadB}
Inputs: Lem.~\ref{lem:monotone},
Lem.~\ref{lem:hs-compact}.
Tools: monotone convergence;
compactness $+$ weak convergence.
Output: $\qc\xrightarrow{\mathrm{M}}
\qlim$.
Thread~B uses neither the Cauchy-net
argument nor $\mathcal{D}$ from
Thread~A: it is methodologically
independent (though both originate
from the same FTB coordinatisation).
\end{remark}

\begin{definition}[Limit form]
\label{def:qlim}
Since $c\mapsto\ip{\Hc f}{f}$ is
monotone increasing
(Lem.~\ref{lem:monotone})
and bounded above by $\|f\|^2$
(\eqref{F1}),
the limit exists for every
$f\in L^2$ by the
monotone convergence principle;
define
\[
  \qlim(f,f):=
  \sup_{c>0}\ip{\Hc f}{f}
  =\lim_{c\to\infty}\ip{\Hc f}{f}.
\]
Extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc)\xrightarrow{\mathrm{M}}\qlim$
iff:
\begin{enumerate}[\rm(M1)]
\item $c_n\to\infty$,
      $f_n\rightharpoonup f\Rightarrow
      \qlim(f,f)\le
      \liminf_n q_{c_n}(f_n,f_n)$.
\item $\forall f$,
      $\exists c_n$, $f_n\to f$
      strongly:
      $q_{c_n}(f_n,f_n)\to
      \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Compact$+$weak
  $\Rightarrow$ form continuity]
\label{lem:wc}
Let $T\in\mathcal{B}(\mathcal{H})$
be compact and
$f_n\rightharpoonup f$ weakly.
Then $\ip{Tf_n}{f_n}\to\ip{Tf}{f}$.
\end{lemma}
\begin{proof}
By \cite{ReedSimon1975}, Thm.~VI.1:
$Tf_n\to Tf$ strongly.
Decompose:
\begin{align*}
  |\ip{Tf_n}{f_n}-\ip{Tf}{f}|
  &\le|\ip{T(f_n-f)}{f_n}|
     +|\ip{Tf}{f_n-f}|\\
  &\le\|T(f_n-f)\|\cdot\sup_n\|f_n\|
     +|\ip{Tf}{f_n-f}|.
\end{align*}
First term $\to 0$:
$\|T(f_n-f)\|\to 0$ (strong);
$\sup_n\|f_n\|<\infty$
(weakly convergent nets are bounded;
\cite{ReedSimon1975}, Thm.~II.12).
Second term $\to 0$:
$f_n\rightharpoonup f$ by hypothesis
(weak convergence tested against
$Tf\in\mathcal{H}$).
\end{proof}

\begin{theorem}[Thread~B conclusion:
  Mosco convergence]
\label{thm:mosco}
$\qc\xrightarrow{\mathrm{M}}\qlim$.
\end{theorem}

\begin{proof}
\textit{(M2).} Trivial from
Def.~\ref{def:qlim}.

\textit{(M1).}
Let $c_n\to\infty$,
$f_n\rightharpoonup f$.
Fix $c_0>0$; once $c_n\ge c_0$:
$q_{c_n}(f_n,f_n)
\ge q_{c_0}(f_n,f_n)$
(Lem.~\ref{lem:monotone}).
$\Hc[c_0]$ compact
(Lem.~\ref{lem:hs-compact}),
so by Lem.~\ref{lem:wc}:
$q_{c_0}(f_n,f_n)\to q_{c_0}(f,f)$.
Hence
$\liminf_n q_{c_n}(f_n,f_n)
\ge q_{c_0}(f,f)$.
Take $\sup_{c_0}$
(Def.~\ref{def:qlim}):
$\liminf_n q_{c_n}(f_n,f_n)
\ge\qlim(f,f)$.
\end{proof}

\section{Identification and Closedness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of
  $\qlim$]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
Bounded symmetric forms are
automatically closed
(\cite{ReedSimon1975}, Thm.~VIII.15);
alternatively, norm equivalence:
$\|f\|^2\le\|f\|_{\qlim}^2
\le 2\|f\|^2$.
\end{proof}

\begin{theorem}[Unification of
  threads: Riesz--Kato]
\label{thm:friedrichs}
$T_{\qlim}=\Hstrong$.
\end{theorem}
\begin{proof}
$\ip{\Hstrong f}{g}=\qlim(f,g)$
(Cor.~\ref{cor:qlim-op});
uniqueness of Riesz representative.
\end{proof}

\begin{remark}[Threads united here]
Thread~A produced $\Hstrong$;
Thread~B produced $\qlim$;
Thm.~\ref{thm:friedrichs} shows
they are the same object.
\end{remark}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
$0\le\qlim(f,f)-\qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$
for $c\ge c_0(\kappa)$.
\end{theorem}
\begin{proof}[Proof (external)]
Paper~VIII, Cor.~4.3:
$\|\Hstrong-\Hc\|
\le C_\kappa c^{-1/4}$
(operator-norm, strictly stronger
than Mosco);
apply to the inner product.
\end{proof}

\begin{remark}[Obstruction to
  internal proof]
\label{rem:no-indep-proof}
Uniform eigenvector asymptotics
would be required;
mixing fixed-$c$ eigenbasis with
$c\to\infty$ weights is invalid.
See Problem~\ref{prob:rate}.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

Sole import point for Paper~IX.

\begin{enumerate}[(IX.a)]
\item\label{IXa}
      $\Hspectral f
      :=\sum_n
      \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}}
      \Phi_n^{(\infty)}$;
      densely defined, symmetric,
      closable
      (Paper~IX, Thms.~2.5,~3.2).
\item\label{IXb}
      $\|\Hspectral f\|^2\le\|f\|^2$
      on $\Dom(\Hspectral)$;
      hence $\overline{\Hspectral}
      \in\mathcal{B}(L^2)$,
      everywhere defined
      (\cite{ReedSimon1975},
      Thm.~I.7).
\item\label{IXc}
      Paper~IX, Rem.~2.6:
      $\Hc\xrightarrow{\mathrm{SOT}}
      \overline{\Hspectral}$
      (same directed family
      $(\Hc)_{c>0}$,
      same SOT topology on
      $\mathcal{B}(L^2)$).
\end{enumerate}

\begin{theorem}[Bridge theorem:
  categorical SOT identity]
\label{thm:bridge}
$\Hstrong=\overline{\Hspectral}$
in $\mathcal{B}(L^2(\mathbb{R}))$.
\end{theorem}

\begin{proof}
Lem.~\ref{lem:hlim-bounded}:
$\Hc\xrightarrow{\mathrm{SOT}}\Hstrong$.
\eqref{IXc}:
$\Hc\xrightarrow{\mathrm{SOT}}
\overline{\Hspectral}$.
Both limits are defined with respect
to the same directed family
$(\Hc)_{c>0}$ and the same SOT
topology on $\mathcal{B}(L^2)$.
SOT limits are unique: if
$A_c f\to Af$ and $A_c f\to Bf$
for all $f\in L^2$,
then $(A-B)f=0$ for all $f$
(pointwise; $L^2$ Hausdorff),
so $A=B$.
Hence $\Hstrong=\overline{\Hspectral}$.
\end{proof}

\begin{remark}[Proof structure]
No eigenfunction analysis,
no dense-set comparison:
the argument is a
\emph{categorical identity} in
the SOT topology on $\mathcal{B}(L^2)$.
Paper~IX contributes precisely
\eqref{IXc}.
\end{remark}

\begin{remark}
$\Hlim:=\Hstrong
=\overline{\Hspectral}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For $z\notin[0,1]$:
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$.
\end{corollary}
\begin{proof}
Thm.~\ref{thm:mosco};
$\sigma(\Hlim)\subset[0,1]$;
\cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
Without hypothesis:
$\Hlim^*=\Hlim\ge 0$;
$\|\Hlim\|\le 1$;
$T_{\qlim}=\Hlim$;
$\qc\xrightarrow{\mathrm{M}}\qlim$;
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{SOT}}
(\Hlim-z)^{-1}$;
$\Hstrong=\overline{\Hspectral}$.
External: rate $O(c^{-1/4})$.
\end{remark}

\begin{remark}[Speculative framework:
  Riemann zeros]
\label{rem:rh}
\emph{Speculative and programmatic;
no unconditional claim.}

\begin{enumerate}[(H1)]
\item $\{\Phi_n^{(\infty)}\}$
      complete;
      $\Hlim$ ess.\ self-adjoint
      (Paper~IX, Hyp.~4.1).
\item Zeros $\rho$ in
      $\sigma_{\mathrm{app}}(\Hlim)$
      (\cite{CCM2025}, Thm.~1.3).
\end{enumerate}
Under (H1)--(H2):
$\rho\in\sigma(\Hlim)
\subset\mathbb{R}$,
so $\mathrm{Re}(\rho)=1/2$.
(H2) lacks a proved operator
identification.
\end{remark}

Prove $\{\Phi_n^{(\infty)}\}$
complete in $L^2(\mathbb{R})$.
\end{problem}

\begin{problem}[Independent rate]
\label{prob:rate}
Prove rate $O(c^{-1/4})$ without
Paper~VIII.
\end{problem}

\begin{problem}[Quantitative
  resolvent]
\label{prob:resolvent}
Find $F(c,z)\to 0$ with
$\|(\Hc-z)^{-1}-(\Hlim-z)^{-1}\|
\le F(c,z)$.
\end{problem}

\label{prob:spectrum}

\end{problem}
