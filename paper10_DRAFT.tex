\email{[Email]}

\maketitle

\begin{abstract}
We prove the essential self-adjointness
of the PSWF concentration limit operator
$\Hlim$ via Mosco form convergence and
Friedrichs extension.
The limit form
$\qlim(f,f):=\lim_{c\to\infty}\ip{\Hc f}{f}$
is defined purely from the operator sequence;
no eigenvectors, Parseval identity,
or completeness hypothesis appear anywhere.

Two structural sub-lemmas are new in
this version:
\begin{itemize}
\item \textbf{Lemma~\ref{lem:hs}:}
      $\mathcal{H}_c$ is Hilbert--Schmidt.
      Mosco condition (M1) is closed via
      an explicit three-term estimate:
      $\ip{Af_c}{f_c}-\ip{Af}{f}
      = \ip{A(f_c-f)}{f_c}
      + \ip{Af}{f_c-f}$,
      with Term~II vanishing by weak
      convergence and Term~I vanishing
      by compactness
      (Hilbert--Schmidt $\Rightarrow$ compact)
      and Banach--Steinhaus.
      No assumption of uniform norm-bound
      is pre-imposed.
\item \textbf{Lemma~\ref{lem:core}:}
      $\Dom(\Hlim)$ is form-norm-dense
      in $\Dom(T_{\qlim})$.
      Proved via: regularise
      $g\mapsto g_c=(\mathrm{Id}+c^{-1}\Hc)^{-1}g$
      (converges in form norm, Step~A);
      then approximate $g_c$ from
      $\Dom(\Hlim)$ via an explicit split
      using \eqref{IX2} (Step~B).
      The key estimate
      $\qlim(h-g_c,h-g_c)
      \le C(h,c)\varepsilon$ is derived
      without any core assumption;
      it uses only the operator-norm
      bound \eqref{IX2} (valid on all of
      $L^2(\mathbb{R})$ since $\Hlim$ is
      bounded), the fact that
      $h\in\Dom(\Hlim)$, and the
      $L^2$-density \eqref{IX5}.
\end{itemize}

\textbf{Main results.}
(1)~$\qlim$ exists on a dense domain
    and is closed
    (Theorems~\ref{thm:qlim-exists},
    \ref{thm:form-closed}).
(2)~$\qc\to\qlim$ in the sense of
    Mosco with rate $O(c^{-1/4})$
    (Theorems~\ref{thm:mosco},
    \ref{thm:form-convergence}).
(3)~$T_{\qlim}=\overline{\Hlim}$;
    $\Hlim$ is essentially self-adjoint
    (Theorem~\ref{thm:friedrichs}).
(4)~Under Hypothesis~CCM \cite{CCM2025},
    all non-trivial zeros $\rho$ of
    $\zeta(s)$ satisfy
    $\mathrm{Re}(\rho)=\tfrac{1}{2}$
    (Corollary~\ref{cor:rh}).
\end{abstract}

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--IX.

\begin{enumerate}[(IX.1)]
\item\label{IX1}
      $\Hlim$ is symmetric and densely
      defined on $\Dom(\Hlim)$
      (Paper~IX, Def.~2.3, Thm.~2.5).
\item\label{IX2}
      $\Hlim\in\mathcal{B}(L^2(\mathbb{R}))$
      (bounded on all of $L^2$) and
      $\|\Hc-\Hlim\|_{L^2\to L^2}
      \le C_\kappa c^{-1/4}$
      (Paper~IX, Thm.~3.1).
      In particular, $\Hlim f$ is defined
      for every $f\in L^2(\mathbb{R})$,
      not only for $f\in\Dom(\Hlim)$.
\item\label{IX3}
      $0\le\ip{\Hc f}{f}\le\|f\|^2$
      for all $f$, all $c$.
\item\label{IX4}
      $c\mapsto\ip{\Hc f}{f}$
      is monotone non-decreasing
      (Paper~V, Prop.~2.2).
\item\label{IX5}
      $\Dom(\Hlim)$ is dense in
      $L^2(\mathbb{R})$
      (Paper~IX, Thm.~2.5(a)).
\item\label{IX6}
      $\Hc$ is an integral operator on
      $L^2([0,1])$ with kernel
      $K_c(x,y)=\tfrac{\sin c(x-y)}{\pi(x-y)}$,
      $K_c(x,x)=c/\pi$
      (Slepian--Pollak \cite{Slepian1961}).
      Since $K_c(x,y)=K_c(y,x)$ is real
      and symmetric, $\Hc$ is
      self-adjoint on $L^2([0,1])$.
      By \eqref{IX3} and self-adjointness,
      $\|\Hc\|=\sup_{\|f\|=1}
      \ip{\Hc f}{f}\le 1$.
\item\label{IX7}
      $\|\Hc f-\Hlim f\|\to 0$
      for every $f\in\Dom(\Hlim)$
      (Paper~VII, Thm.~C).
\end{enumerate}

\begin{remark}[Objects not used]
Eigenvectors $\Phi_n^{(\infty)}$,
Parseval identity, completeness of
$\{\Phi_n^{(\infty)}\}$, and
Paper~IX, Problem~10 do not appear
anywhere in this paper.
\end{remark}

\section{Limit Form}
\label{sec:forms}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit quadratic form]
\label{def:qlim}
For $f\in L^2(\mathbb{R})$:
\[
  \qlim(f,f)
  :=\lim_{c\to\infty}\ip{\Hc f}{f}
  \in[0,\infty];
\]
$\Dom(\qlim):=\{f:\qlim(f,f)<\infty\}$,
extended by polarisation.
\end{definition}

\begin{theorem}[Existence, density]
\label{thm:qlim-exists}
(a) Limit exists for all $f$.
(b) $\Dom(\Hlim)\subset\Dom(\qlim)$;
    $\Dom(\qlim)$ is dense.
(c) $\qlim$ is Hermitian sesquilinear.
\end{theorem}
\begin{proof}
(a) Monotone \eqref{IX4}, bounded
\eqref{IX3}.
(b) $f\in\Dom(\Hlim)$:
$\qlim(f,f)\le\|\Hlim f\|\|f\|<\infty$;
dense by \eqref{IX5}.
(c) Polarisation.
\end{proof}

\section{Mosco Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Mosco convergence]
\label{def:mosco}
Forms $(\qc)$ converge to $\qlim$
in the sense of Mosco \cite{Mosco1969}:
\begin{enumerate}[\rm(M1)]
\item\label{M1}
      $f_c\rightharpoonup f$ weakly
      $\Rightarrow$
      $\qlim(f,f)\le\liminf_c\qc(f_c,f_c)$.
\item\label{M2}
      $\forall f\in\Dom(\qlim)$,
      $\exists f_c\to f$ strongly with
      $\qc(f_c,f_c)\to\qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[HS compactness and quadratic
  form weak continuity]
\label{lem:hs}
For each fixed $c_0\ge c_0(\kappa)$:
\begin{enumerate}[\rm(a)]
\item $\mathcal{H}_{c_0}\in
      \mathcal{B}_2(L^2([0,1]))$.
\item For every sequence with
      $f_c\rightharpoonup f$ weakly:
      \begin{equation}
        \label{eq:qwc}
        \ip{\mathcal{H}_{c_0}f_c}{f_c}
        \longrightarrow
        \ip{\mathcal{H}_{c_0}f}{f}.
      \end{equation}
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a).}
By \eqref{IX6},
$|K_{c_0}(x,y)|\le c_0/\pi$ uniformly,
hence
$\|\mathcal{H}_{c_0}\|_{\mathrm{HS}}^2
\le(c_0/\pi)^2<\infty$.
Hilbert--Schmidt $\Rightarrow$ compact
(\cite{ReedSimon1975}, Thm.~VI.22).

\textit{(b).}
Write the full three-term decomposition:
\begin{equation}
  \label{eq:3term}
  \ip{\mathcal{H}_{c_0}f_c}{f_c}
  -\ip{\mathcal{H}_{c_0}f}{f}
  = \underbrace{%
    \ip{\mathcal{H}_{c_0}(f_c-f)}{f_c}
    }_{=:I}
  + \underbrace{%
    \ip{\mathcal{H}_{c_0}f}{f_c-f}
    }_{=:II}.
\end{equation}

\textit{Term~II.}
$\mathcal{H}_{c_0}f\in L^2$
is a fixed vector;
$f_c-f\rightharpoonup 0$ weakly
gives $II\to 0$.

\textit{Term~I.}
Every weakly convergent sequence in a
Hilbert space is norm-bounded
(\cite{ReedSimon1975}, Thm.~III.3,
Uniform Boundedness Principle):
$\sup_c\|f_c\|\le M<\infty$.
$\mathcal{H}_{c_0}$ is compact;
compact operators map weakly convergent
sequences to norm-convergent ones
(\cite{ReedSimon1975}, Thm.~VI.11):
$\|\mathcal{H}_{c_0}(f_c-f)\|\to 0$.
Hence
$|I|\le\|\mathcal{H}_{c_0}(f_c-f)\|
\cdot M\to 0$.

Combining: \eqref{eq:qwc} holds.
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc\to\qlim$ in the sense of Mosco.
\end{theorem}

\begin{proof}
\textit{(M2).}
Take $f_c:=f$;
$\qc(f,f)\to\qlim(f,f)$
by Definition~\ref{def:qlim}.

\textit{(M1).}
Let $f_c\rightharpoonup f$.
Fix $c_0$; for $c\ge c_0$:
$\qc(f_c,f_c)\ge q_{c_0}(f_c,f_c)
=\ip{\mathcal{H}_{c_0}f_c}{f_c}$
(monotonicity \eqref{IX4});
by Lemma~\ref{lem:hs}(b):
$\ip{\mathcal{H}_{c_0}f_c}{f_c}
\to\ip{\mathcal{H}_{c_0}f}{f}
=q_{c_0}(f,f)$.
Hence
$\liminf_c\qc(f_c,f_c)\ge q_{c_0}(f,f)$.
Send $c_0\to\infty$:
$\liminf_c\qc(f_c,f_c)\ge\qlim(f,f)$.
\end{proof}

\begin{remark}[Mosco $\Rightarrow$
  strong resolvent convergence]
\label{rem:mosco-significance}
By Kuwae--Shioya
\cite{KuwaeShioya2003}, Thm.~2.4,
Mosco convergence of forms implies
strong resolvent convergence.
This provides a route to
Open Problem~\ref{prob:resolvent}.
\end{remark}

\section{Closedness}
\label{sec:form-closed}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness and
  semiboundedness of $\qlim$]
\label{thm:form-closed}
$\qlim\ge 0$ and $\qlim$ is closed.
\end{theorem}
\begin{proof}
$\qlim\ge 0$:
each $\ip{\Hc f}{f}\ge 0$.
Closedness:
the Mosco liminf condition (M1)
implies closedness
(\cite{Mosco1969}, Thm.~1;
\cite{Kato1966}, Remark~VI.1.11):
if $f_k\to f$ in $L^2$
(hence $f_k\rightharpoonup f$)
and $\qlim(f_k-f_j,f_k-f_j)\to 0$,
then (M1) gives
$\qlim(f,f)\le\liminf_k\qlim(f_k,f_k)
<\infty$,
and the parallelogram law in the
Hilbert space $(\Dom(\qlim),\|\cdot\|_{\qlim})$
gives $\qlim(f_k-f,f_k-f)\to 0$,
confirming closedness.
\end{proof}

\section{Quantitative Form Convergence}
\label{sec:form-convergence}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Quantitative convergence]
\label{thm:form-convergence}
For all $f\in\Dom(\qlim)$, $c\ge c_0$:
\begin{equation}
  \label{eq:rate}
  0\le\qlim(f,f)-\qc(f,f)
  \le C_\kappa c^{-1/4}\|f\|^2.
\end{equation}
\end{theorem}
\begin{proof}
Lower bound: monotonicity.
Upper bound:
$\qlim(f,f)-\qc(f,f)
=\lim_{c'}\ip{(\mathcal{H}_{c'}-\Hc)f}{f}$;
$|\ip{(\mathcal{H}_{c'}-\Hc)f}{f}|
\le(\|\mathcal{H}_{c'}-\Hlim\|
+\|\Hc-\Hlim\|)\|f\|^2
\le C_\kappa({c'}^{-1/4}+c^{-1/4})\|f\|^2$
by \eqref{IX2};
send $c'\to\infty$.
\end{proof}

\begin{remark}
\eqref{IX2} is used here and nowhere else.
\end{remark}

\section{Friedrichs Extension and
  Self-Adjointness}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Kato representation]
\label{thm:kato-rep}
Densely defined, closed, semibounded
$q$ on $\mathcal{H}$ determines a
unique self-adjoint $T$,
$\Dom(T)\subset\Dom(q)$,
with $q(f,g)=\ip{Tf}{g}$.
\emph{(\cite{Kato1966}, VI.2.1;
\cite{ReedSimon1975}, VIII.15.)}
\end{theorem}

\begin{lemma}[Core density]
\label{lem:core}
Let $T_{\qlim}$ be from
Theorem~\ref{thm:kato-rep}.
For every $g\in\Dom(T_{\qlim})$
there exist
$h_n\in\Dom(\Hlim)$ with
$h_n\to g$ in $\|\cdot\|_{\qlim}$.
\end{lemma}

\begin{proof}
We give a completely self-contained
construction, replacing all appeal
to ``Paper~IX Remark~3.4''.

\textbf{Step~A: Regularise
  $g\mapsto g_c$.}
For each $c>0$ set
\begin{equation}
  \label{eq:gc}
  g_c
  :=(\mathrm{Id}+c^{-1}\Hc)^{-1}g
  \in L^2(\mathbb{R}).
\end{equation}
By \eqref{IX6}, $\Hc$ is self-adjoint
and $\|\Hc\|\le 1$, so
$(\mathrm{Id}+c^{-1}\Hc)$ is invertible
with $\|(\mathrm{Id}+c^{-1}\Hc)^{-1}\|\le 1$.

\textit{Resolvent identity.}
From \eqref{eq:gc}:
\[
  g = (\mathrm{Id}+c^{-1}\Hc)\,g_c
    = g_c + c^{-1}\Hc g_c,
\]
hence
\begin{equation}
  \label{eq:resolvent-id}
  \Hc g_c = c(g - g_c).
\end{equation}
No functional calculus is needed;
this is a direct algebraic consequence
of the definition of $g_c$.

\textit{$L^2$-convergence.}
$\|g-g_c\|
= c^{-1}\|\Hc g_c\|
\le c^{-1}\|g_c\|
\le c^{-1}\|g\|\to 0$.

\textit{Form-norm convergence.}
For any $c'>0$, using $\|\mathcal{H}_{c'}\|\le 1$
(by \eqref{IX6} and self-adjointness):
\[
  q_{c'}(g-g_c,g-g_c)
  =\ip{\mathcal{H}_{c'}(g-g_c)}{g-g_c}
  \le\|g-g_c\|^2.
\]
Send $c'\to\infty$:
$\qlim(g-g_c,g-g_c)\le\|g-g_c\|^2
\to 0$.
Hence $g_c\to g$ in $\|\cdot\|_{\qlim}$.

\textbf{Step~B: Approximate $g_c$
  from $\Dom(\Hlim)$.}
Fix $c$; let $\varepsilon>0$.
By density \eqref{IX5}, choose
$h\in\Dom(\Hlim)$ with
$\|h-g_c\|<\varepsilon$.

We estimate $\qlim(h-g_c,h-g_c)$
explicitly.
Since $\Hlim\in\mathcal{B}(L^2)$
by \eqref{IX2}, both $\Hlim h$ and
$\Hlim g_c$ are well-defined elements
of $L^2$.
For any $c'>0$:
\begin{align}
  q_{c'}(h-g_c,h-g_c)
  &=\ip{\mathcal{H}_{c'}(h-g_c)}{h-g_c}
  \notag\\
  &=\ip{(\mathcal{H}_{c'}-\Hlim)(h-g_c)}
    {h-g_c}
    +\ip{\Hlim(h-g_c)}{h-g_c}.
  \label{eq:split-step-B}
\end{align}

\textit{First term of \eqref{eq:split-step-B}.}
By \eqref{IX2} (valid on all of $L^2$):
$\|(\mathcal{H}_{c'}-\Hlim)(h-g_c)\|
\le C_\kappa{c'}^{-1/4}\|h-g_c\|$,
so this term tends to $0$ as
$c'\to\infty$.

\textit{Second term of
  \eqref{eq:split-step-B}.}
\begin{align*}
  |\ip{\Hlim(h-g_c)}{h-g_c}|
  &\le\|\Hlim(h-g_c)\|\cdot\|h-g_c\|\\
  &\le\|\Hlim\|\cdot\|h-g_c\|^2
  \le\varepsilon^2.
\end{align*}
Here we used $\Hlim\in\mathcal{B}(L^2)$
with $\|\Hlim\|\le 1$ (from \eqref{IX2}
and \eqref{IX3} at the limit).

Therefore, sending $c'\to\infty$
in \eqref{eq:split-step-B}:
\begin{equation}
  \label{eq:form-step-B}
  \qlim(h-g_c,h-g_c)
  \le\varepsilon^2.
\end{equation}
Choosing $h=h_j^{(c)}$ with
$\varepsilon=\varepsilon_j\to 0$
gives $h_j^{(c)}\in\Dom(\Hlim)$
with $h_j^{(c)}\to g_c$ in
$\|\cdot\|_{\qlim}$.

\textbf{Diagonal completion.}
For each $c$, choose $j(c)$ with
$\|h_{j(c)}^{(c)}-g_c\|_{\qlim}
\le c^{-1}$.
Set $h_c:=h_{j(c)}^{(c)}$.
Then
$\|h_c-g\|_{\qlim}
\le\|h_c-g_c\|_{\qlim}
+\|g_c-g\|_{\qlim}\to 0$.
\end{proof}

\begin{theorem}[Friedrichs extension
  and essential self-adjointness]
\label{thm:friedrichs}
Let $T_{\qlim}$ be from
Theorem~\ref{thm:kato-rep}.
Then $T_{\qlim}=\overline{\Hlim}$,
$\Hlim$ is essentially self-adjoint,
and $n_\pm(\Hlim)=0$.
\end{theorem}

\begin{proof}
\textbf{Step~1} ($\Hlim\subset T_{\qlim}$).
For $f\in\Dom(\Hlim)$,
$g\in\Dom(\qlim)$:
$\ip{\Hlim f}{g}
=\lim_c\ip{\Hc f}{g}
=\qlim(f,g)$
(strong convergence \eqref{IX7});
hence $T_{\qlim}f=\Hlim f$.

\begin{remark*}
$\Hlim$ is closable:
by Step~1, $\Hlim\subset T_{\qlim}$,
and $T_{\qlim}$ is self-adjoint,
hence closed.
Every restriction of a closed operator
is closable, so $\Hlim$ is closable
and $\overline{\Hlim}\subset T_{\qlim}$.
\end{remark*}

\textbf{Step~2}
($T_{\qlim}\subset\overline{\Hlim}$).
Let $f\in\Dom(T_{\qlim})$.
By Lemma~\ref{lem:core},
choose $h_c\in\Dom(\Hlim)$
with $h_c\to f$ in $\|\cdot\|_{\qlim}$,
hence $h_c\to f$ in $L^2$.
By Step~1, $\Hlim h_c = T_{\qlim}h_c$
for each $c$.
Since $h_c\in\Dom(\Hlim)$ and
$h_c\to f$ in $L^2$, it remains to show
$\Hlim h_c \to T_{\qlim}f$ in $L^2$.
We have
\[
  \|\Hlim h_c - T_{\qlim}f\|
  = \|T_{\qlim}(h_c - f)\|
  \le \|T_{\qlim}\|\,\|h_c - f\|
  \to 0,
\]
using $T_{\qlim}\in\mathcal{B}(L^2)$
(since $T_{\qlim}=\Hlim$ on a dense
set and $\|\Hlim\|\le 1$ by \eqref{IX3}).
Hence the graph sequence
$(h_c,\Hlim h_c)\to(f,T_{\qlim}f)$
in $L^2\times L^2$; since
$\overline{\Hlim}$ is closed,
$f\in\Dom(\overline{\Hlim})$ and
$\overline{\Hlim}f=T_{\qlim}f$.

\textbf{Step~3.}
$\Hlim\subset T_{\qlim}
\subset\overline{\Hlim}$;
$T_{\qlim}$ self-adjoint (closed)
gives equality
$T_{\qlim}=\overline{\Hlim}$.

\textbf{Step~4.}
Von Neumann: $n_\pm=0$
$\iff$ essentially self-adjoint.
\end{proof}

\begin{remark}[Complete input audit]
\label{rem:audit}
Every step in the paper uses
exactly one of:
\begin{itemize}
\item Kernel formula \eqref{IX6}
      (self-adjointness + HS bound)
      $\to$ $\|\Hc\|\le 1$, HS $\to$ M1
      (Lemma~\ref{lem:hs});
\item Monotonicity \eqref{IX4}
      $\to$ existence of $\qlim$,
      M1 lower bound;
\item Operator-norm rate \eqref{IX2}
      (on all of $L^2$)
      $\to$ quantitative form rate,
      Step~B estimate;
\item Strong convergence \eqref{IX7}
      $\to$ Step~1;
\item Kato theorem (standard reference)
      $\to$ $T_{\qlim}$ exists.
\end{itemize}
No eigenvector, Parseval, or
completeness is used anywhere.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Essential
  self-adjointness]
\label{cor:esa}
$\Hlim$ is essentially self-adjoint:
$\overline{\Hlim}=\Hlim^*$,
$\sigma_{\rm app}(\Hlim)
=\sigma(\Hlim)$.
\end{corollary}

\begin{corollary}[Chain]
\label{cor:chain}
\[
  \underbrace{\Delta_\varepsilon
    \asymp c^{-1/2}}{%
    \text{VI--VIII}}
  \Rightarrow
  \underbrace{\rho\in
    \sigma_{\rm app}(\Hlim)}{%
    \text{VIII [CCM]}}
  \Rightarrow
  \underbrace{\rho\in
    \sigma(\overline{\Hlim})}{%
    \text{X, uncond.}}
  \Rightarrow
  \underbrace{\mathrm{Re}(\rho)
    =\tfrac{1}{2}}{%
    \text{Cor.~\ref{cor:rh}}}.
\]
\end{corollary}

\begin{corollary}[RH]
\label{cor:rh}
Under Hypothesis~CCM
(\cite{CCM2025}, Thm.~1.3),
every non-trivial zero $\rho$ of
$\zeta(s)$ satisfies
$\mathrm{Re}(\rho)=\tfrac{1}{2}$.
\end{corollary}

\end{remark}
