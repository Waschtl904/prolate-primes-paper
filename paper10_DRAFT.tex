  Mosco Convergence, Resolvent Stability,
  and the Bridge Theorem for
  PSWF Concentration Operators}

\author{[Author]}
\address{[Address]}

\maketitle

\begin{abstract}
The prolate spheroidal wave function (PSWF)
concentration operators $\Hc$ form a uniformly
bounded, monotone family of compact
self-adjoint operators on $L^2(\mathbb{R})$.
This paper has two independent objectives.

\textbf{(A) Form convergence and resolvent
stability (unconditional).}
The foundational identity
(Lemma~\ref{lem:ftb}, Slepian--Pollak
\cite{Slepian1961})
\[
  \ip{\Hc f}{f} = \|\Bc(Pf)\|^2,
\]
reduces the PSWF operator to a Fourier
band projection conjugated by
time-restriction.
The no-boundary-artifact property of
this identification is made explicit
in the proof.
From this single identity, operator
monotonicity, Hilbert--Schmidt compactness,
and density of the strong-convergence
domain $\mathcal{D}$ are derived
(Lemmas~\ref{lem:monotone}--\ref{lem:dense}).
Mosco convergence
$\qc \xrightarrow{\mathrm{M}} \qlim$,
closedness, and strong resolvent
convergence follow.
A quantitative rate $O(c^{-1/4})$
is stated as an external theorem
(Theorem~\ref{thm:rate-ext}, Paper~VIII).

\textbf{(B) Bridge theorem (unconditional).}
$\Hstrong = \overline{\Hspectral}$
as bounded operators on $L^2(\mathbb{R})$.

The conditional connection to zeros of
$\zeta(s)$ (Remark~\ref{rem:rh})
holds under two unverified hypotheses;
no unconditional claim is made.
\end{abstract}

\section{Setup and Foundational Lemmas}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

Let $\Fc$ denote the Fourier transform,
$(\Fc f)(\xi) = \int_{-\infty}^\infty
f(x) e^{-i\xi x}\,dx$,
with Plancherel isometry
$\|\Fc f\|_{L^2} = \sqrt{2\pi}\|f\|_{L^2}$
(convention following \cite{ReedSimon1975},
Ch.~IX).
All operators act on $L^2(\mathbb{R})$
unless stated otherwise.

\begin{conv}[Operators]
\label{conv:ops}
\begin{alignat}{2}
  \Bc &:= \Fc^{-1}\chi_{[-c,c]}\Fc
  && \text{(Fourier band projection;
     $\Bc^* = \Bc = \Bc^2$)},
  \\
  P &: L^2(\mathbb{R})
     \to L^2([0,1]),
  && \text{restriction; } \|P\|=1,
  \\
  P^* &: L^2([0,1])
       \to L^2(\mathbb{R}),
  && \text{extension by zero; }
     \|P^*\|=1,
  \\
  \mathcal{H}_c^{[0,1]}
  &: L^2([0,1]) \to L^2([0,1]),
  && \text{kernel }
     K_c(x,y) =
     \tfrac{\sin c(x-y)}{\pi(x-y)},
  \\
  \Hc &:= P^*\mathcal{H}_c^{[0,1]}P
  && \in \mathcal{B}(L^2(\mathbb{R})).
\end{alignat}
The kernel $K_c$ is evaluated as
$c/\pi$ at $x = y$.
\end{conv}

\begin{enumerate}[(F.1)]
\item\label{F1}
$\Hc$ is self-adjoint,
$0 \le \Hc \le I$, $\|\Hc\| \le 1$
(\cite{Slepian1961};
\cite{ReedSimon1975}, Thm.~VI.1).
\end{enumerate}

\begin{remark}[Scope of form thread]
\label{rem:scope}
Sections~\ref{sec:setup}--\ref{sec:friedrichs}
use only \eqref{F1} and
Lemmas~\ref{lem:ftb}--\ref{lem:dense}.
No limit eigenvalues, eigenvectors,
operator differences, or results
from Paper~VIII or~IX appear.
\end{remark}

%───────────────────────────────────────────────────
% FOUNDATIONAL LEMMA: Fourier-Time-Band
%───────────────────────────────────────────────────
%

\begin{lemma}[Fourier--time--band
  representation]
\label{lem:ftb}
For all $c > 0$ and all
$f \in L^2(\mathbb{R})$:
\begin{equation}\label{eq:ftb}
  \ip{\Hc f}{f}
  = \|\Bc(P^*Pf)\|_{L^2(\mathbb{R})}^2.
\end{equation}
Equivalently, as operators on
$L^2([0,1])$:
\begin{equation}\label{eq:ftb-op}
  \mathcal{H}_c^{[0,1]} = P\,\Bc\,P^*.
\end{equation}
\end{lemma}

\begin{proof}
\textit{Step 1: Convolution identity
on $L^2(\mathbb{R})$.}
For any $h \in L^2(\mathbb{R})$,
the Fourier pair
\[
  K_c(u) = \frac{\sin cu}{\pi u}
  \quad \longleftrightarrow \quad
  (\Fc K_c)(\xi)
  = \chi_{[-c,c]}(\xi)
\]
(\cite{Slepian1961}, Eq.~(3);
\cite{ReedSimon1975}, Ex.~IX.7)
shows $K_c \in L^2(\mathbb{R})$
(since $\Fc K_c = \chi_{[-c,c]}
\in L^2$).
By the convolution theorem
(\cite{ReedSimon1975}, Thm.~IX.2):
\[
  \Fc(K_c * h)(\xi)
  = (\Fc K_c)(\xi) \cdot (\Fc h)(\xi)
  = \chi_{[-c,c]}(\xi)(\Fc h)(\xi),
\]
so
\begin{equation}\label{eq:conv-bc}
  K_c * h = \Fc^{-1}[\chi_{[-c,c]}
  \Fc h] = \Bc h
  \quad \text{in } L^2(\mathbb{R}).
\end{equation}

\textit{Step 2: No-boundary-artifact
under restriction.}
Let $g \in L^2([0,1])$ and set
$\tilde{g} = P^* g \in L^2(\mathbb{R})$,
so $\tilde{g}$ has support in
$[0,1]$.
By Step~1:
$K_c * \tilde{g} = \Bc\tilde{g}
\in L^2(\mathbb{R})$.
For $x \in [0,1]$:
\begin{align*}
  (K_c * \tilde{g})(x)
  &= \int_{-\infty}^{\infty}
     K_c(x-y)\,\tilde{g}(y)\,dy
  = \int_0^1 K_c(x-y)\,g(y)\,dy
  = (\mathcal{H}_c^{[0,1]} g)(x).
\end{align*}
The equality of the first and second
expression uses $\mathrm{supp}\,
\tilde{g} \subset [0,1]$:
\emph{no boundary term arises}
because $P^*g$ has compact support
in $[0,1]$, so the convolution
$K_c * \tilde{g}$ evaluated at
$x \in [0,1]$ integrates only over
$y \in [0,1]$, coinciding exactly
with the kernel integral definition
of $\mathcal{H}_c^{[0,1]} g$.

Applying $P$ on both sides:
\[
  \mathcal{H}_c^{[0,1]} g
  = P(\Bc\tilde{g})
  = P(\Bc P^* g),
\]
which is \eqref{eq:ftb-op}.

\textit{Step 3: Form identity.}
For $f \in L^2(\mathbb{R})$,
set $g = Pf$, $\tilde{g} = P^*Pf$:
\begin{align*}
  \ip{\Hc f}{f}
  &= \ip{P^*\mathcal{H}_c^{[0,1]}Pf}{f}
  = \ip{\mathcal{H}_c^{[0,1]}Pf}{Pf}
  = \ip{P\Bc P^*Pf}{Pf}\\
  &= \ip{\Bc\tilde{g}}{\tilde{g}}
  = \|\Bc\tilde{g}\|^2,
\end{align*}
where the last equality uses
$\Bc = \Bc^*$ and $\Bc^2 = \Bc$.
This is \eqref{eq:ftb}.
\end{proof}

\begin{remark}[Structural content
  of Lemma~\ref{lem:ftb}]
\label{rem:ftb}
The identity \eqref{eq:ftb-op}
is the precise content of
Slepian--Pollak's observation that
$\mathcal{H}_c^{[0,1]}$ is the
\emph{time-band limiting operator}:
first restrict to time interval
$[0,1]$ (via $P^*$),
then apply frequency cutoff $[-c,c]$
(via $\Bc$), then restrict back (via $P$).
The no-boundary-artifact property
(Step~2) is the key: it holds because
$P^*g$ has compact support, so the
convolution against the non-compactly
supported kernel $K_c$ nevertheless
produces no aliasing when restricted
to $[0,1]$.
All subsequent lemmas are pure
functional analysis on this structure.
\end{remark}

% ---- Monotonicity: one line ----

\begin{lemma}[Operator monotonicity]
\label{lem:monotone}
For $0 < c \le c'$:
$\ip{\Hc f}{f} \le
\ip{\mathcal{H}_{c'} f}{f}$
for all $f$.
Hence $c \mapsto \ip{\Hc f}{f}$
is non-decreasing, bounded by
$\|f\|^2$.
\end{lemma}

\begin{proof}
By \eqref{eq:ftb}:
$\ip{\Hc f}{f} = \|\Bc(P^*Pf)\|^2$.
Since $[-c,c] \subset [-c',c']$:
$\Bc \le B_{c'}$ in the
orthogonal projection order
(\cite{ReedSimon1975}, Ch.~VIII.1),
so $\|\Bc h\|^2 \le \|B_{c'}h\|^2$
for all $h$.
Apply with $h = P^*Pf$.
\end{proof}

% ---- HS: Parseval, with remark on cleaner route ----

\begin{lemma}[Hilbert--Schmidt
  and compactness]
\label{lem:hs-compact}
For each fixed $c_0 > 0$,
$\Hc[c_0]$ is compact on
$L^2(\mathbb{R})$.
\end{lemma}

\begin{proof}
We show
$\mathcal{H}_{c_0}^{[0,1]}$
is Hilbert--Schmidt on $L^2([0,1])$.
Compactness of
$\Hc[c_0] = P^*\mathcal{H}_{c_0}^{[0,1]}P$
then follows since
$P, P^*$ are bounded
(\cite{ReedSimon1975}, Thm.~VI.22).

By \eqref{eq:ftb-op},
$\mathcal{H}_{c_0}^{[0,1]}
= P\Bc[c_0]P^*$,
so the Hilbert--Schmidt norm equals
\[
  \|\mathcal{H}_{c_0}^{[0,1]}
  \|_{\mathrm{HS}}^2
  = \|K_{c_0}\|_{L^2([0,1]^2)}^2.
\]
Substitute $u = x - y$
(Fubini; integrand non-negative;
Jacobian $= 1$):
\begin{align*}
  \int_0^1\!\int_0^1
  |K_{c_0}(x-y)|^2\,dx\,dy
  &= \int_{-1}^{1}(1-|u|)
     |K_{c_0}(u)|^2\,du
  \le \int_{-\infty}^{\infty}
     |K_{c_0}(u)|^2\,du.
\end{align*}
By Parseval
($\Fc K_{c_0} = \chi_{[-c_0,c_0]}$;
norm convention
$\|f\|_2^2 =
\frac{1}{2\pi}\|\Fc f\|_2^2$):
\[
  \int_{-\infty}^{\infty}
  |K_{c_0}(u)|^2\,du
  = \frac{1}{2\pi}
    \|\chi_{[-c_0,c_0]}\|_{L^2}^2
  = \frac{1}{2\pi}\cdot 2c_0
  = \frac{c_0}{\pi}.
\]
Hence
$\|\mathcal{H}_{c_0}^{[0,1]}
\|_{\mathrm{HS}}^2 \le c_0/\pi
< \infty$.
\end{proof}

\begin{remark}[Operator-theoretic
  alternative]
\label{rem:hs-alt}
Since $\Bc[c_0]$ is an orthogonal
projection with
$\mathrm{rank}(\Bc[c_0]|_{L^2([0,1])})
= \infty$ (infinitely many PSWF
basis elements) but
$\Bc[c_0]$ is a bounded operator,
compactness of
$P\Bc[c_0]P^* = \mathcal{H}_{c_0}^{[0,1]}$
can also be seen from the
Hilbert--Schmidt norm via Parseval
as above, or alternatively from
the fact that $\Bc[c_0]$ maps
$L^2([0,1])$ into a separable
subspace and the restriction
$P\Bc[c_0]P^*$ is Hilbert--Schmidt;
both routes are equivalent.
\end{remark}

% ---- Density: B_c -> I strong, P*P bounded commutes ----

\begin{lemma}[Density of $\mathcal{D}$]
\label{lem:dense}
Let $\mathcal{D} :=
\{f \in L^2(\mathbb{R})
: \lim_{c\to\infty}\Hc f
\text{ exists in } L^2\}$.
Then $\mathcal{D}$ is dense.
\end{lemma}

\begin{proof}
It suffices to show
$C_c^\infty(\mathbb{R})
\subset \mathcal{D}$,
since $C_c^\infty$ is dense
in $L^2(\mathbb{R})$.

Let $f \in C_c^\infty$,
$\tilde{g} = P^*Pf
= \mathbf{1}_{[0,1]}f$.
By \eqref{eq:ftb}:
\[
  \Hc f
  = P^*(P\Bc P^*Pf)
  = P^*P(\Bc\tilde{g}).
\]
We claim $\Hc f \to
P^*P\tilde{g}$
in $L^2$ as $c \to \infty$.

The Fourier projection $\Bc$
increases to the identity strongly:
\[
  \Bc h \xrightarrow{\mathrm{s}} h
  \quad \forall h \in L^2(\mathbb{R}),
  \quad c \to \infty,
\]
by Plancherel and dominated convergence
on the Fourier side
($\chi_{[-c,c]}(\xi)\hat{h}(\xi)
\to \hat{h}(\xi)$ a.e.,
$|\chi_{[-c,c]}\hat{h}|
\le |\hat{h}| \in L^2$;
\cite{ReedSimon1975}, Thm.~IX.14).
Applying to $h = \tilde{g}$:
$\Bc\tilde{g} \to \tilde{g}$
strongly in $L^2$.

Since $P^*P \in \mathcal{B}(L^2)$
with $\|P^*P\| \le 1$,
bounded operators are continuous
with respect to strong convergence:
\[
  \|P^*P(\Bc\tilde{g})
  - P^*P\tilde{g}\|_{L^2}
  \le \|P^*P\|
  \cdot\|\Bc\tilde{g} - \tilde{g}\|_{L^2}
  \to 0.
\]
Hence $\Hc f \to P^*P\tilde{g}$
in $L^2$, so $f \in \mathcal{D}$.
\emph{Uniform domination:
$\|\Hc f\| \le \|f\|$ for all $c$
by \eqref{F1}; no $c$-dependent
bound is needed.}
\end{proof}

\begin{remark}[Three lemmas
  from one identity]
\label{rem:setup-summary}
Lemmas~\ref{lem:monotone}--\ref{lem:dense}
all follow from \eqref{eq:ftb}.
Monotonicity: projection order
$\Bc \le B_{c'}$.
Compactness: Parseval norm bound
$\le c_0/\pi$.
Density: strong convergence
$\Bc \nearrow I$
+ boundedness of $P^*P$.
This is a template for any
increasing family of compact
self-adjoint operators generated
by spectral truncations.
\end{remark}

\section{Strong-Limit Operator}
\label{sec:strong-limit}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Density-extension
  principle]
\label{lem:sc}
Let $(A_c) \subset
\mathcal{B}(\mathcal{H})$
with $\sup_c\|A_c\| \le M$;
$\mathcal{D} \subset \mathcal{H}$
dense;
$A_c f \to Bf$ for all
$f \in \mathcal{D}$.
Then $A_c \to A$ strongly,
$A \in \mathcal{B}(\mathcal{H})$,
$\|A\| \le M$.
\end{lemma}
\begin{proof}
Uniformly bounded operators are
strongly sequentially closed:
for any $f$, choose $g \in \mathcal{D}$
close to $f$ and use the
$3\varepsilon$ argument
(\cite{ReedSimon1975}, Thm.~I.26);
uniformly bounded net converges
pointwise on all of $\mathcal{H}$.
\end{proof}

\begin{lemma}[Existence of $\Hstrong$]
\label{lem:hlim-bounded}
$\exists!\,\Hstrong \in
\mathcal{B}(L^2)$ with
$\|\Hstrong\| \le 1$ and
$\Hc \to \Hstrong$ strongly.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:sc} with
$\mathcal{D}$ from
Lemma~\ref{lem:dense},
$M = 1$ from \eqref{F1}.
\end{proof}

\begin{corollary}[Form equals
  operator]
\label{cor:qlim-op}
$\qlim(f,f) = \ip{\Hstrong f}{f}$.
\end{corollary}
\begin{proof}
Strong convergence
$+$ inner product continuity.
\end{proof}

\begin{corollary}[Self-adjointness]
\label{cor:sa-strong}
$\Hstrong = \Hstrong^* \ge 0$;
$\sigma(\Hstrong) \subset [0,1]$.
\end{corollary}
\begin{proof}
Symmetry by strong limit;
BSA principle
(\cite{ReedSimon1975}, Thm.~III.6);
positivity from $\qlim \ge 0$.
\end{proof}

\section{Limit Form and Mosco
  Convergence}
\label{sec:mosco}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Limit form]
\label{def:qlim}
$\qlim(f,f) :=
\sup_{c>0}\ip{\Hc f}{f}
= \lim_{c\to\infty}\ip{\Hc f}{f}$
(Lemma~\ref{lem:monotone}
$+$ \eqref{F1});
extend by polarisation.
\end{definition}

\begin{definition}[Mosco convergence
  \cite{Mosco1969}]
\label{def:mosco}
$(\qc) \xrightarrow{\mathrm{M}}
\qlim$:
\begin{enumerate}[\rm(M1)]
\item $c_n\to\infty$,
      $f_n\rightharpoonup f$
      $\Rightarrow$
      $\qlim(f,f)
      \le \liminf_n
      q_{c_n}(f_n,f_n)$.
\item $\forall f\,\exists
      c_n\to\infty, f_n\to f$:
      $q_{c_n}(f_n,f_n)
      \to \qlim(f,f)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Complete continuity]
\label{lem:wc}
For fixed $c_0$:
$f_n \rightharpoonup f
\Rightarrow
q_{c_0}(f_n,f_n) \to q_{c_0}(f,f)$.
\end{lemma}
\begin{proof}
$\Hc[c_0]$ compact
(Lemma~\ref{lem:hs-compact}),
hence completely continuous
(\cite{ReedSimon1975}, Thm.~VI.1):
$\Hc[c_0]f_n
\xrightarrow{\mathrm{s}}\Hc[c_0]f$.
Then $\ip{\Hc[c_0]f_n}{f_n}
\to \ip{\Hc[c_0]f}{f}$
by strong-weak continuity
(\cite{ReedSimon1975}, Lem.~II.3).
\end{proof}

\begin{theorem}[Mosco convergence]
\label{thm:mosco}
$\qc \xrightarrow{\mathrm{M}}
\qlim$.
\end{theorem}
\begin{proof}
\textit{(M2).}
$f_n := f$, $c_n \to \infty$:
definition $+$
Lemma~\ref{lem:monotone}.

\textit{(M1).}
$c_n\to\infty$,
$f_n\rightharpoonup f$.
For fixed $c_0$, once
$c_n \ge c_0$:
$q_{c_n}(f_n,f_n)
\ge q_{c_0}(f_n,f_n)$
(Lemma~\ref{lem:monotone});
hence
$\liminf_n q_{c_n}(f_n,f_n)
\ge q_{c_0}(f,f)$
(Lemma~\ref{lem:wc}).
Taking $\sup_{c_0}$:
$\ge \qlim(f,f)$.
\end{proof}

\section{Closedness and
  Riesz--Kato Identification}
\label{sec:friedrichs}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closedness of $\qlim$]
\label{thm:form-closed}
$\qlim$ is closed.
\end{theorem}
\begin{proof}
$\qlim(f,f) = \ip{\Hstrong f}{f}$
(Corollary~\ref{cor:qlim-op});
$\Hstrong$ bounded
$\Rightarrow$ form norm
$\|f\|_{\qlim}^2
= \|f\|^2 + \qlim(f,f)$
equivalent to $\|f\|^2$
$\Rightarrow$ $\qlim$ closed.
\end{proof}

\begin{theorem}[Riesz--Kato
  identification]
\label{thm:friedrichs}
$T_{\qlim} = \Hstrong$.
\end{theorem}
\begin{proof}
$\ip{\Hstrong f}{g}
= \qlim(f,g)$
(Corollary~\ref{cor:qlim-op});
uniqueness of Riesz representative.
\end{proof}

\begin{theorem}[Quantitative rate,
  external]
\label{thm:rate-ext}
For all $f$ and $c \ge c_0(\kappa)$:
$0 \le \qlim(f,f) - \qc(f,f)
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{theorem}
\begin{proof}[Proof (external)]
Lower bound:
Lemma~\ref{lem:monotone}.
Upper bound: Paper~VIII, Cor.~4.3:
$\|\Hstrong - \Hc\|
\le C_\kappa c^{-1/4}$;
hence
$\qlim(f,f) - \qc(f,f)
= \ip{(\Hstrong-\Hc)f}{f}
\le C_\kappa c^{-1/4}\|f\|^2$.
\end{proof}

\begin{remark}[Obstruction to
  independent proof]
\label{rem:no-indep-proof}
Independent proof requires uniform
eighenvector asymptotics as $c$ varies;
expressions mixing fixed-$c$
basis with $c\to\infty$ weights
are not valid.
See Problem~\ref{prob:rate}.
\end{remark}

\section{The Bridge Theorem}
\label{sec:bridge}
%%═══════════════════════════════════════════════════════════════════

This section is the \emph{only} section
that imports Paper~IX results.

\begin{enumerate}[(IX.a)]

\item\label{IXa}
      Paper~IX, Def.~2.3:
      $\Hspectral f
      := \sum_n \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}}
      \Phi_n^{(\infty)}$;
      densely defined, symmetric,
      closable
      (Paper~IX, Thm.~2.5, 3.2).

\item\label{IXb}
      \textit{Bounded, everywhere-defined
      closure.}
      On $\Dom(\Hspectral)$:
      $\|\Hspectral f\|^2
      = \sum_n
      (\lambda_n^{(\infty)})^2
      |\ip{f}{\Phi_n^{(\infty)}}|^2
      \le \|f\|^2$
      (Bessel $+$
      $\lambda_n^{(\infty)}\in[0,1]$).
      Hence
      $\overline{\Hspectral}
      \in \mathcal{B}(L^2)$,
      $\|\overline{\Hspectral}\|
      \le 1$,
      everywhere defined
      (\cite{ReedSimon1975}, Thm.~I.7).

\item\label{IXc}
      \textit{Agreement on $\mathcal{D}$.}
      Both $\Hstrong$ and
      $\overline{\Hspectral}$
      are bounded and everywhere
      defined.
      For $f \in \mathcal{D}$:
      $\Hc f \to \Hstrong f$
      (Lem.~\ref{lem:hlim-bounded});
      $\Hc f \to
      \overline{\Hspectral}f$
      (Paper~IX, Rem.~2.6).
      Uniqueness of limits:
      $\Hstrong =
      \overline{\Hspectral}$
      on $\mathcal{D}$.

\end{enumerate}

\begin{theorem}[Bridge theorem]
\label{thm:bridge}
$\Hstrong = \overline{\Hspectral}$
in $\mathcal{B}(L^2(\mathbb{R}))$.
\end{theorem}
\begin{proof}
Both are bounded and everywhere
defined (\eqref{IXb};
Lem.~\ref{lem:hlim-bounded}).
They agree on the dense set
$\mathcal{D}$ (\eqref{IXc}).
Two bounded operators that agree
on a dense set are equal
(\cite{ReedSimon1975}, Lem.~II.3).
\end{proof}

\begin{remark}
$\Hlim :=
\Hstrong = \overline{\Hspectral}$.
\end{remark}

\section{Spectral Consequences}
\label{sec:consequences}
%%═══════════════════════════════════════════════════════════════════

\begin{corollary}[Strong resolvent
  convergence]
\label{cor:resolvent}
For every $z \notin [0,1]$:
$(\Hc - z)^{-1}
\xrightarrow{\mathrm{s}}
(\Hlim - z)^{-1}$.
\end{corollary}
\begin{proof}
Mosco (Theorem~\ref{thm:mosco});
$\sigma(\Hlim) \subset [0,1]$
(Corollary~\ref{cor:sa-strong});
\cite{KuwaeShioya2003}, Thm.~2.4.
\end{proof}

\begin{remark}[Unconditional results]
Without hypothesis:
$\Hlim = \Hlim^* \ge 0$;
$\|\Hlim\|\le 1$;
$T_{\qlim} = \Hlim$;
$\qc \xrightarrow{\mathrm{M}} \qlim$;
$(\Hc-z)^{-1}
\xrightarrow{\mathrm{s}}
(\Hlim-z)^{-1}$;
$\Hstrong = \overline{\Hspectral}$.
External (Paper~VIII):
rate $O(c^{-1/4})$.
\end{remark}

\begin{remark}[Conditional connection
  to the Riemann hypothesis]
\label{rem:rh}
Formal implication under two
\emph{unverified} hypotheses:
(H1) Paper~IX, Hyp.~4.1:
$\{\Phi_n^{(\infty)}\}$ complete;
$\Hlim$ essentially self-adjoint.
(H2) \cite{CCM2025}, Thm.~1.3:
non-trivial zeros $\rho$
lie in
$\sigma_{\mathrm{app}}(\Hlim)$.
Under (H1)--(H2):
$\rho \in \sigma(\Hlim)
\subset \mathbb{R}$,
forcing $\mathrm{Re}(\rho)
= \tfrac{1}{2}$.
\emph{No unconditional claim.
Neither (H1) nor (H2) is proved
here.}
\end{remark}
