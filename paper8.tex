\title[Non-Cancellation and $c^{-1/2}$ Lower Bound for PSWFs]{%
Non-Cancellation Control, the $c^{-1/2}$ Lower Bound
for the Peak Width of Prolate Spheroidal Wave Functions,
and Spectral Connection to the Zeros of $\zeta(s)$}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We continue the asymptotic analysis of the
Fourier--Mellin transforms $\Phihat_n^{(c)}$
of the rescaled prolate spheroidal wave functions
(PSWFs) initiated in Papers~V--VII of this series.

Paper~VII \cite{PaperVII} established the sharp
\emph{upper bound} $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)\,c^{-1/2}$.
The matching \emph{lower bound} was identified as
Problem~7.1 of \cite{PaperVII}, requiring pointwise
non-cancellation of $\Phihat_n^{(c)}$ on a set of
measure $\gtrsim c^{-1/2}$ near $\tstar$.

\textbf{What this paper proves.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:nonvanish}
      (Non-Cancellation, proved).}
      There exist $c_1(\kappa)>0$ and, for each
      $c \ge c_0(\kappa)$, $n \le \kappa c$,
      a set $E_c \subset \Zai(c,n)$ with
      $|E_c| \asymp c^{-2/3}$ and
      $|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$
      on $E_c$.
      The Airy core prevents oscillatory cancellation
      near the turning point $\tstar$.

\item \textbf{Problem~\ref{prob:anticonc}
      (Anti-concentration gap, central open problem).}
      Upgrading $|E_c| \asymp c^{-2/3}$ to
      $|E_c| \ge c_1 c^{-1/2}$ requires
      anti-concentration of $\Phihat_n^{(c)}$
      in the crossover zone $\Zcross$:
      Airy oscillations are too sparse to
      cancel the signal on a set larger than $c^{-2/3}$.
      Three strategies are formulated
      (Problem~\ref{prob:anticonc}).

\item \textbf{Theorem~\ref{thm:lower}
      (Lower bound, conditional).}
      \emph{Conditionally on Problem~\ref{prob:anticonc}:}
      $\Delta_\varepsilon(c,n) \ge c_1(\kappa,\varepsilon)\,c^{-1/2}$
      and $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$.

\item \textbf{Theorem~\ref{thm:spectral}
      (Spectral eigenvalue control, unconditional).}
      $c_\kappa c^{-3/4} \le
      |\lambda_n^{(c)} - \lambda_n^{(c+1)}|
      \le C_\kappa c^{-3/4}$.

\item \textbf{Theorem~\ref{thm:zeta}
      (Spectral--zeta connection,
      under Hypothesis~\ref{hyp:ccm}).}
      Each non-trivial zero $\rho$ of $\zeta(s)$
      lies in the approximate point spectrum
      of $\Hlim$ with precision $O(c^{-3/4})$.
\end{enumerate}
\end{abstract}

\section*{Guide for the Referee}

This paper has two main threads.
\begin{enumerate}
\item \textbf{The lower bound thread}
(Sections~\ref{sec:setup}--\ref{sec:lower}):
we prove the Airy-zone amplitude lower bound and
reduce the full $c^{-1/2}$ lower bound to one
precisely formulated anti-concentration problem
(Problem~\ref{prob:anticonc}).
\item \textbf{The spectral thread}
(Sections~\ref{sec:spectral}--\ref{sec:zeta}):
unconditional two-sided eigenvalue bounds and
the CCM spectral connection (under
Hypothesis~\ref{hyp:ccm}).
\end{enumerate}

\noindent
\textbf{The key structural insight:}
The Airy core prevents cancellation pointwise
(\S\ref{sec:nc}), giving amplitude $\ge c_1 c^{-1/2}$
on a set of width $\asymp c^{-2/3}$.
The missing factor $c^{-1/6}$ in the measure
is exactly the \emph{anti-concentration} problem:
Airy oscillations are sparse, but controlling
that sparsity precisely in the crossover zone
requires quantitative zero-spacing estimates for
$\Ai$ (Option~C of Problem~\ref{prob:anticonc}).

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--VII.
In particular, from Paper~VII \cite{PaperVII}:
\begin{itemize}
\item $\Zcross(c,n)
      := \{t : c^{-2/3} \le |t - \tstar| \le c^{-1/3}\}$.
\item $\Zai(c,n) := \{t : |t-\tstar| \le c^{-2/3}\}$,
      the Airy zone, with $|\Zai| = 2c^{-2/3}$.
\item The composite profile
      $\mathcal{A}_n^{(c)}$ from Theorem~A of Paper~VII:
      \begin{equation}
        \label{eq:composite-recall}
        \Phihat_n^{(c)}(t)
        = c^{-1/2}\mathcal{A}_n^{(c)}\!\bigl(c^{1/2}(t-\tstar)\bigr)
        (1 + O(c^{-1/3})),
        \quad t \in \Zcross,
      \end{equation}
      with $|\mathcal{A}_n^{(c)}(x)| \le C_\kappa(1+|x|)^{-1/4}$.
\item The Airy form
      \begin{equation}\label{eq:Airy-form}
        \mathcal{A}_n^{(c)}(x)
        = \kappa_n^{(c)}\Ai\!\bigl(x/(\mu_n^{(c)})^{2/3}\bigr),
      \end{equation}
      with $c_\kappa \le \mu_n^{(c)} \le C_\kappa$
      and $|\kappa_n^{(c)}| \ge c_\kappa > 0$
      (Lemma~\ref{lem:kappa-lower}).
\end{itemize}

Results used without reproof:
\begin{enumerate}[(VII.1)]
\item\label{VII1}
      \textbf{Upper bound} (Paper~VII, Thm.~B):
      $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)c^{-1/2}$.
\item\label{VII2}
      \textbf{SOT convergence with strong rate}
      (Paper~VII, Thm.~C):
      $\|(\Hc - \Hlim)f\| \le C_\kappa c^{-3/4}\|f\|$
      for every fixed unit vector $f$.
      This is a \emph{strong} (pointwise-in-$f$)
      rate, not an operator norm bound.
\item\label{VII3}
      \textbf{Eigenvalue perturbation upper bound}
      (Paper~VII, Lem.~6.2):
      $|\lambda_n^{(c)} - \lambda_n^{(c+1)}|
      \le C_\kappa c^{-3/4}$.
\item\label{VII4}
      \textbf{Mass in $\Zcross$} (Paper~VII, Cor.~3.8):
      $\int_{\Zcross}|\Phihat_n^{(c)}|^2 dt
      \asymp c^{-17/12}$.
\item\label{VII5}
      \textbf{Pointwise decay} (Paper~VI, Thm.~3.1):
      $|\Phihat_n^{(c)}(t)| \le C_\kappa c^{-1/2}$
      for $t \asymp c^{1/2}$.
\item\label{VIIA}
      \textbf{$L^2$-normalisation:}
      $\|\Phihat_n^{(c)}\|_{L^2}^2 = 1/2$.
\end{enumerate}

\section{Non-Cancellation in the Airy Zone}
\label{sec:nc}
%%═══════════════════════════════════════════════════════════════════

\subsection{Strategy}

Since $\Ai(0) = 3^{-2/3}/\Gamma(2/3) > 0$
and $\Ai$ is continuous,
it is bounded below near $y = 0$.
In the Airy zone $\Zai$, the rescaled argument
$y = c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}$
satisfies $|y| \le c^{-1/6}/c_\kappa^{2/3} \to 0$,
so $|\Ai(y)| \ge \Ai(0)/2$ throughout $\Zai$
for large $c$.
This yields a \emph{pointwise} lower bound
$|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$ on all of $\Zai$,
but since $|\Zai| = 2c^{-2/3}$, the resulting
good set has measure $\asymp c^{-2/3}$, not $c^{-1/2}$.
The additional factor $c^{-1/6}$ in the measure
is the anti-concentration problem
(Problem~\ref{prob:anticonc}).

\subsection{Amplitude lower bound for $\kappa_n^{(c)}$}

\begin{lemma}[Amplitude lower bound]
\label{lem:kappa-lower}
For all $c \ge c_0(\kappa)$ and $n \le \kappa c$:
$|\kappa_n^{(c)}| \ge c_\kappa > 0$.
\end{lemma}

\begin{proof}
From \eqref{VII4}, substituting
$u = c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}$:
\[
  c^{-17/12} \lesssim
  c^{-3/2}|\kappa_n^{(c)}|^2(\mu_n^{(c)})^{2/3}
  \int_{\mathbb{R}}|\Ai|^2\,du
  \le C_\kappa' c^{-3/2}|\kappa_n^{(c)}|^2,
\]
so $|\kappa_n^{(c)}|^2 \gtrsim c^{1/12} \ge 1$ for $c \ge 1$.
We use only the lower bound $|\kappa_n^{(c)}| \ge c_\kappa$;
the possible growth $|\kappa_n^{(c)}| \asymp c^{1/24}$
is harmless and consistent with $L^2$-normalisation.
\end{proof}

\subsection{Airy non-vanishing near the origin}

\begin{lemma}[Airy non-vanishing]
\label{lem:airy-nv}
Set $\alpha_0 := \Ai(0)/2 > 0$.
There exists $\delta_0 > 0$ such that
$|\Ai(y)| \ge \alpha_0$ for all $|y| \le \delta_0$.
\end{lemma}

\begin{proof}
Continuity of $\Ai$ and $\Ai(0) > 0$
(\cite{Olver1974}, p.~392).
\end{proof}

\subsection{Main non-cancellation theorem}

\begin{theorem}[Non-Cancellation in the Airy zone]
\label{thm:nonvanish}
There exist $c_1 = c_1(\kappa) > 0$ and
$c_0 = c_0(\kappa)$ such that
for all $c \ge c_0$ and $n \le \kappa c$:
\[
  |\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}
  \quad \text{for all } t \in \Zai(c,n),
\]
and hence the good set $E_c := \Zai$ satisfies
$|E_c| = 2c^{-2/3}$ and
$|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$ on $E_c$.
\end{theorem}

\begin{proof}
In the rescaled variable $x = c^{1/2}(t-\tstar)$,
the Airy zone gives $|x| \le c^{-1/6}$.
For $y = x/(\mu_n^{(c)})^{2/3}$:
$|y| \le c^{-1/6}/c_\kappa^{2/3} \to 0$,
so $|y| \le \delta_0$ for $c \ge c_0$
and $|\Ai(y)| \ge \alpha_0$ by Lemma~\ref{lem:airy-nv}.
From \eqref{eq:composite-recall}--\eqref{eq:Airy-form}
and Lemma~\ref{lem:kappa-lower}:
\[
  |\Phihat_n^{(c)}(t)|
  \ge (1-O(c^{-1/3}))\,|\kappa_n^{(c)}|\,|\Ai(y)|\,c^{-1/2}
  \ge \tfrac{1}{2}\alpha_0 c_\kappa c^{-1/2}
  =: c_1 c^{-1/2}
\]
for $c$ large. Set $E_c = \Zai$;
then $|E_c| = 2c^{-2/3}$ and the amplitude bound holds.
\end{proof}

\begin{remark}[Why this gives $c^{-2/3}$, not $c^{-1/2}$]
\label{rem:measure-gap}
The Airy zone $\Zai$ has width $c^{-2/3}$
in $t$-measure. To obtain $|E_c| \ge c_1 c^{-1/2}$
one needs the lower bound to hold on a set
$c^{-1/6}$ times wider.
A naive Chebyshev argument from
$\int |\Phihat_n^{(c)}|^2 \ge 1/4$ and
$\sup |\Phihat_n^{(c)}|^2 \asymp c^{-1}$
would give $|E_c| \gtrsim c$, not $c^{-1/2}$ ---
this shows the $L^2$ information alone is
too coarse.
What is needed is \emph{anti-concentration}:
the function $\Phihat_n^{(c)}$ cannot be
concentrated on a set of measure $\ll c^{-1/2}$,
because the Airy oscillations in $\Zcross$
are sparse enough to prevent it.
This is the content of Problem~\ref{prob:anticonc}.
\end{remark}

\section{Anti-Concentration Problem and Conditional Lower Bound}
\label{sec:lower}
%%═══════════════════════════════════════════════════════════════════

\subsection{The anti-concentration open problem}

\begin{problem}[Anti-concentration in the crossover zone]
\label{prob:anticonc}
Prove that for all $c \ge c_0(\kappa)$,
$n \le \kappa c$, there exists a set
$E_c \subset \{|t-\tstar| \le c^{-1/2}\}$
with $|E_c| \ge c_1 c^{-1/2}$ and
$|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$ on $E_c$.

Equivalently: $\Phihat_n^{(c)}$ cannot concentrate
its $L^2$-mass in a set of measure $\ll c^{-1/2}$
near $\tstar$.
Three strategies:

\begin{enumerate}[\normalfont(A)]
\item \textbf{Extension into $\Zcross$.}
      Extend the pointwise lower bound of
      Theorem~\ref{thm:nonvanish} beyond $\Zai$
      into $\Zcross$, using the composite asymptotic
      (Paper~VII, Thm.~A) and quantitative
      zero-spacing estimates for $\Ai$:
      adjacent zeros of $\Ai$ are spaced $\asymp 1$
      in the argument variable (\cite{Olver1974}, §10.4),
      so in $t$-measure the zero-free intervals
      have width $\asymp (\mu_n^{(c)})^{2/3}c^{-1/2}
      \asymp c^{-1/2}$.
      Controlling the zero-free fractions in $\Zcross$
      would give $|E_c| \gtrsim c^{-1/2}$.

\item \textbf{Fourier uncertainty / no-concentration.}
      Show that the $L^\infty$ bound
      $|\Phihat_n^{(c)}| \le C_\kappa c^{-1/2}$
      combined with local Fourier uncertainty
      (e.g.\ Nazarov's theorem, \cite{Havin1994})
      prevents concentration in measure $\ll c^{-1/2}$.

\item \textbf{Quantitative oscillation control
      (most promising).}
      Use the Airy structure
      $\Phihat_n^{(c)}(t) = c^{-1/2}\kappa_n^{(c)}
      \Ai\!\bigl(c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}\bigr)
      (1+O(c^{-1/3}))$
      valid for $|c^{1/2}(t-\tstar)| \le c^{1/6}$
      (Paper~VII, Thm.~A).
      The zeros of $\Ai(y)$ for $y \le 0$
      are the sequence $-a_k$
      ($a_1 < a_2 < \cdots$, the positive zeros
      of $\Ai(-\cdot)$),
      spaced $a_{k+1}-a_k \sim \pi k^{-1/3}
      \to \pi$ as $k \to \infty$
      (\cite{Olver1974}, Thm.~10.4.94).
      Between consecutive zeros, $|\Ai| \ge c_0'
      (1+|y|)^{-1/4}$.
      In $t$-coordinates, the zero-free intervals
      have width $\gtrsim (\mu_n^{(c)})^{2/3}c^{-1/2}
      \gtrsim c_\kappa c^{-1/2}$
      (using $\mu_n^{(c)} \ge c_\kappa$).
      Since the relevant range
      $|c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}|
      \le c^{1/6}/c_\kappa^{2/3}$
      contains $O(c^{1/6})$ zero-free intervals,
      there is at least one interval of
      $t$-measure $\gtrsim c^{-1/2}$ within
      $\{|t-\tstar| \le c^{-1/2}\}$
      on which $|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$.
\end{enumerate}

\noindent
Strategy~(C) is the most concrete and directly
utilises the known asymptotic structure.
A complete proof along these lines is the
central goal of Paper~IX.
\end{problem}

\begin{remark}[Why $L^2$ mass alone is too coarse]
\label{rem:L2-insufficient}
The total $L^2$-mass near $\tstar$,
$\int_{|t-\tstar|\le c^{-1/2}}|\Phihat_n^{(c)}|^2 dt
\ge 1/4$, is a strong bound ---
but it is compatible with concentration
on a set of measure $\ll c^{-1/2}$
(e.g.\ a narrow spike of amplitude $\gg c^{-1/2}$
but width $\ll c^{-1/2}$).
The sup-norm bound $|\Phihat_n^{(c)}| \le C_\kappa c^{-1/2}$
rules out such spikes, giving
$|E_c| \ge (1/4)/(C_\kappa^2 c^{-1}) = c/(4C_\kappa^2)$
--- which is trivially too large.
The actual anti-concentration requires the
oscillatory structure of $\Phihat_n^{(c)}$,
not just its size.
\end{remark}

\subsection{$L^2$ mass lower bound near $\tstar$}

\begin{proposition}[$L^2$ mass lower bound]
\label{prop:mass-lower}
For all $c \ge c_0$, $n \le \kappa c$:
\begin{equation}
  \label{eq:mass-lower}
  \int_{\{|t-\tstar| \le c^{-1/2}\}}
  |\Phihat_n^{(c)}|^2\,dt
  \;\ge\; \tfrac{1}{4}.
\end{equation}
\end{proposition}

\begin{proof}
From \eqref{VIIA}: $\int_{\mathbb{R}}|\Phihat_n^{(c)}|^2 = 1/2$.
The tail beyond $|t-\tstar| > c^{-1/2}$
contributes at most
$\int_{\Zcross}|\Phihat_n^{(c)}|^2
+ \int_{|t-\tstar|>c^{-1/3}}|\Phihat_n^{(c)}|^2$.
The first term is $\le Cc^{-17/12}$ by \eqref{VII4}.
For the second, $|\Phihat_n^{(c)}(t)|
\le C_\kappa c^{-1/4}t^{-1/4}$ and $t \ge c^{1/2}-c^{-1/3}$,
so the integral is $\le C_\kappa^2 c^{-3/4}$.
Hence the tail is $O(c^{-3/4}) < 1/4$ for large $c$,
giving \eqref{eq:mass-lower}.
\end{proof}

\subsection{Conditional lower bound on peak width}

\begin{theorem}[Conditional lower bound]
\label{thm:lower}
\emph{Assume Problem~\ref{prob:anticonc} is resolved},
i.e.\ that there exists $E_c \subset
\{|t-\tstar| \le c^{-1/2}\}$ with
$|E_c| \ge c_1 c^{-1/2}$ and
$|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$ on $E_c$.
Then for every $\varepsilon \in (0,1)$:
$\Delta_\varepsilon(c,n) \ge c_1(\kappa,\varepsilon)\,c^{-1/2}$.
\end{theorem}

\begin{proof}
If $\Delta_\varepsilon < c_1 c^{-1/2}/2$,
then the interval $\{|t-\tstar|\le\Delta_\varepsilon\}$
has measure $< c_1 c^{-1/2}$.
Since $|E_c| \ge c_1 c^{-1/2}$,
the set $E_c \setminus \{|t-\tstar|\le\Delta_\varepsilon\}$
is non-empty and has measure
$\ge c_1 c^{-1/2} - 2\Delta_\varepsilon > 0$.
On this set $|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$,
so the tail mass satisfies
\[
  \|\Phihat_n^{(c)}\|_{L^2(|t-\tstar|>\Delta_\varepsilon)}^2
  \ge c_1^2 c^{-1} \cdot (c_1 c^{-1/2} - 2\Delta_\varepsilon)
  \ge c_1^3 c^{-3/2}/2.
\]
For $c$ large this exceeds $\varepsilon^2/2$:
contradiction with $\Delta_\varepsilon$ being the
$(1-\varepsilon)$-energy-capture width.
Hence $\Delta_\varepsilon \ge c_1 c^{-1/2}/2$.
\end{proof}

\begin{corollary}[Sharp two-sided bound, conditional]
\label{cor:sharp}
Conditional on Problem~\ref{prob:anticonc}:
$\Delta_\varepsilon(c,n) \asymp c^{-1/2}$.
\end{corollary}

\begin{proof}
Lower: Theorem~\ref{thm:lower}.
Upper: Paper~VII, Theorem~B \eqref{VII1}.
\end{proof}

\section{Spectral Eigenvalue Control}
\label{sec:spectral}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Two-sided eigenvalue perturbation]
\label{thm:spectral}
For all $c \ge c_0(\kappa)$, $n \le \kappa c$,
$c' = c+1$:
\begin{equation}
  \label{eq:eig-two-sided}
  c_\kappa c^{-3/4}
  \le |\lambda_n^{(c)} - \lambda_n^{(c')}|
  \le C_\kappa c^{-3/4}.
\end{equation}
This result is \textbf{unconditional}.
\end{theorem}

\begin{proof}
Upper: \eqref{VII3}.
Lower: Hellmann--Feynman identity
(\cite{Osipov2013}, \S3.3):
$|d\lambda_n^{(c)}/dc| \ge c_\kappa' c^{-3/4}$
(dominant contribution
$\int_{-\Delta}^{\Delta}\partial_c|\Phihat_n^{(c)}|^2 dt
\asymp c^{-1/4}\cdot c^{-1/2} = c^{-3/4}$,
using $\|\partial_c\Phi_n^{(c)}\|_{L^2} \asymp c^{-1/4}$
from \cite{Osipov2013} Thm.~3.2
and $\Delta \le C_\kappa c^{-1/2}$ from \eqref{VII1});
then integrate over $[c,c+1]$.
\end{proof}

\begin{corollary}[Eigenvalue control for $\Hlim$]
\label{cor:eigenvalue-lim}
(a)~$|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$.
(b)~Eigenvalues cluster near $1/2$ (well-concentrated)
and $0$ (poorly concentrated), transitioning near
$n \approx c/\pi$ (classical PSWF theory
\cite{Slepian1961,Osipov2013}).
\end{corollary}

\begin{proof}
(a): $\sum_{k=c}^\infty k^{-3/4} \le Cc^{-1/4}$
and \eqref{VII3}.
\end{proof}

\section{Spectral Connection to $\zeta(s)$}
\label{sec:zeta}
%%═══════════════════════════════════════════════════════════════════

\begin{hypothesis}[CCM eigenvalue accumulation]
\label{hyp:ccm}
For each non-trivial zero $\rho$ of $\zeta(s)$
and $\kappa \in (0,2/\pi)$, there exist
$n(c) \le \kappa c$ and unit eigenvectors $v_c$ of $\Hc$
with $\Hc v_c = \lambda_{n(c)}^{(c)} v_c$
and $\lambda_{n(c)}^{(c)} \to \rho$.
This is the content of \cite{CCM2025}, Theorem~1.3.
\end{hypothesis}

\begin{lemma}[Strong rate implies approximate spectrum]
\label{lem:approx-pt-spec}
Let $T_c \to T$ with
$\|(T_c-T)f\| \le \omega(c)\|f\|$
for every fixed $f$, with $\omega(c) \to 0$.
If $T_c v_c = \mu_c v_c$, $\|v_c\|=1$,
and $\mu_c \to \mu$,
then $\mu \in \sigma_{\rm app}(T)$,
with
$\|(T-\mu)v_{c_k}\| \le \omega(c_k) + |\mu_{c_k}-\mu|
\to 0$
along any subsequence with
$\omega(c_k)+|\mu_{c_k}-\mu| < 2/k$.
\end{lemma}

\begin{proof}
For fixed $k$, choose $c_k$ with
$\omega(c_k) < 1/k$ and $|\mu_{c_k}-\mu| < 1/k$.
Apply $\|(T-\mu)v_{c_k}\|
\le \|(T-T_{c_k})v_{c_k}\| + |\mu_{c_k}-\mu|
\le \omega(c_k) + 1/k < 2/k$.
The key point: $\omega(c)$ is a
\emph{per-vector} bound (strong rate),
applied here to each specific $v_{c_k}$.
This is \emph{not} an operator norm bound;
it suffices because we need only a
sequence of unit vectors with
$(T-\mu)v_{c_k} \to 0$.
\end{proof}

\begin{theorem}[Spectral--zeta connection]
\label{thm:zeta}
Assume Hypothesis~\ref{hyp:ccm}.
For each non-trivial zero $\rho$ of $\zeta(s)$:
\begin{enumerate}[\rm(i)]
\item The eigenvalue differences
      $|\lambda_n^{(c)}-\lambda_n^{(c+1)}|$
      detect spectral transitions near $\rho$
      with precision $O(c^{-3/4})$
      (Theorem~\ref{thm:spectral}).
\item Under \cite{CCM2025}, Section~4:
      $\zeta_{\Hlim}(s) = \zeta(s) + E(s)$
      with $E$ entire.
\item $\rho \in \sigma_{\rm app}(\Hlim)$
      with $\|(\Hlim-\rho)v_{c_k}\|
      = O(c_k^{-3/4})$,
      by Lemma~\ref{lem:approx-pt-spec}
      applied to $T_c = \Hc$, $T = \Hlim$,
      $\omega(c) = C_\kappa c^{-3/4}$
      (from \eqref{VII2}), and the accumulating
      eigenvectors of Hypothesis~\ref{hyp:ccm}.
\end{enumerate}
\end{theorem}

\begin{remark}[What is new vs.\ CCM2025]
\begin{enumerate}[(a)]
\item Sharp peak-width control $\Delta_\varepsilon \le Cc^{-1/2}$
      (Paper~VII); conditional lower bound here.
\item Quantitative rate $O(c^{-3/4})$ (Paper~VII,
      Thm.~C).
\item Two-sided eigenvalue bounds
      (Theorem~\ref{thm:spectral}, unconditional).
\item Rigorous approximate-spectrum argument with
      explicit separation of CCM input
      (Hypothesis~\ref{hyp:ccm}) from new results.
\end{enumerate}
\end{remark}

\begin{conjecture}[RH via $\Hlim$]
\label{conj:RH}
$\Hlim$ is essentially self-adjoint.
Equivalently, all non-trivial zeros of $\zeta(s)$
lie on $\mathrm{Re}(s) = 1/2$.
\end{conjecture}

\section{Open Problems}
\label{sec:problems}

\begin{problem}[Anti-concentration --- main problem for Paper~IX]
See Problem~\ref{prob:anticonc} above.
Strategy~(C) (quantitative zero-spacing of $\Ai$)
is recommended as the starting point.
The key estimate needed:
the $t$-width of a zero-free interval of
$t \mapsto \Ai(c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3})$
in $\{|t-\tstar| \le c^{-1/2}\}$
is $\asymp c^{-1/2}$
(since Airy zeros are $O(1)$-spaced in argument
and the conversion factor is
$(\mu_n^{(c)})^{2/3} c^{-1/2} \asymp c^{-1/2}$).
\end{problem}

\begin{problem}[Sharp constant]
\label{prob:sharp-const}
Determine $c^{1/2}\Delta_\varepsilon(c,\lfloor\kappa c\rfloor)
as $c\to\infty$.
\end{problem}

\begin{problem}[Operator norm rate]
\label{prob:op-norm-rate}
Determine the decay rate of
$\|\Hc - \Hlim\|_{L^2\to L^2}$.
\end{problem}

\begin{problem}[Essential self-adjointness]
\label{prob:esa}
Prove or disprove Conjecture~\ref{conj:RH}.
\end{problem}

\begin{problem}[Point vs.\ approximate spectrum]
\label{prob:exact-spectrum}
Are the non-trivial zeros of $\zeta(s)$
in the point spectrum (eigenvalues) of $\Hlim$,
or only in $\sigma_{\rm app}(\Hlim)$?
\end{problem}
