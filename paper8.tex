\title[Non-Cancellation and Sharp $c^{-1/2}$ Lower Bound for PSWFs]{%
Non-Cancellation Control, the Sharp $c^{-1/2}$ Lower Bound
for the Peak Width of Prolate Spheroidal Wave Functions,
and Spectral Connection to the Zeros of $\zeta(s)$}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We complete the asymptotic analysis of the
Fourier--Mellin transforms $\Phihat_n^{(c)}$
of the rescaled prolate spheroidal wave functions
(PSWFs) initiated in Papers~V--VII of this series.

Paper~VII \cite{PaperVII} established the sharp
\emph{upper bound} $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)\,c^{-1/2}$
via a composite Airy--WKB asymptotic in the
crossover zone $\Zcross$.  The matching
\emph{lower bound} was identified there as the
central open problem (Problem~7.1 of \cite{PaperVII}),
requiring pointwise non-cancellation control of
$\Phihat_n^{(c)}$ near $\tstar$.

\textbf{Main results of this paper.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:nonvanish}
      (Non-Cancellation Lemma).}
      There exist $c_1(\kappa)>0$ and, for each
      $c \ge c_0(\kappa)$ and $n \le \kappa c$,
      a measurable set
      $E_c \subset \{|t - \tstar| \le c^{-1/2}\}$
      with $|E_c| \ge c_1 c^{-1/2}$ such that
      \[
        |\Phihat_n^{(c)}(t)| \;\ge\; c_1\,c^{-1/2}
        \quad\text{for all }t \in E_c.
      \]

\item \textbf{Theorem~\ref{thm:lower} (Lower Bound).}
      For every $\kappa \in (0, 2/\pi)$ and
      $\varepsilon \in (0,1)$:
      $\Delta_\varepsilon(c,n) \ge c_1(\kappa,\varepsilon)\,c^{-1/2}$.

\item \textbf{Corollary~\ref{cor:sharp} (Sharp scaling).}
      $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$,
      confirming Paper~V \cite{PaperV}.

\item \textbf{Theorem~\ref{thm:spectral}
      (Spectral Eigenvalue Control).}
      \[
        c_\kappa\,c^{-3/4}
        \le
        |\lambda_n^{(c)} - \lambda_n^{(c')}|
        \le C_\kappa\,c^{-3/4},
        \quad c' = c+1,\; n \le \kappa c.
      \]

\item \textbf{Theorem~\ref{thm:zeta}
      (Spectral--Zeta Connection).}
      Under \cite{CCM2025},
      $\zeta_{\Hlim}(s) = \zeta(s) + E(s)$
      with $E$ entire.
      Each non-trivial zero $\rho$
      of $\zeta(s)$ belongs to the approximate
      point spectrum of $\Hlim$ with precision
      $O(c^{-3/4})$, established via a
      resolvent argument from the accumulating
      eigenvalues $\lambda_n^{(c)} \to \rho$.
\end{enumerate}

The Riemann Hypothesis is equivalent to $\Hlim$
being essentially self-adjoint
(Conjecture~\ref{conj:RH}).
\end{abstract}

\section{Setup and Inherited Results}
\label{sec:setup}

We retain all notation from Papers~V--VII.
In particular, from Paper~VII \cite{PaperVII}:
\begin{itemize}
\item $\Zcross(c,n)
      := \{t : c^{-2/3} \le |t - \tstar| \le c^{-1/3}\}$
      (Definition~3.1 of Paper~VII).
\item The composite profile function
      $\mathcal{A}_n^{(c)} : \mathbb{R} \to \mathbb{C}$
      from Theorem~A of Paper~VII, satisfying
      \begin{equation}
        \label{eq:composite-recall}
        \Phihat_n^{(c)}(t)
        = c^{-1/2}\,\mathcal{A}_n^{(c)}\!\bigl(c^{1/2}(t-\tstar)\bigr)
        \,\bigl(1 + O(c^{-1/6})\bigr),
        \quad t \in \Zcross,
      \end{equation}
      with $|\mathcal{A}_n^{(c)}(x)|
       \le C_\kappa(1+|x|)^{-1/4}$.
\item The explicit Airy form
      \begin{equation}\label{eq:Airy-form}
        \mathcal{A}_n^{(c)}(x)
        = \kappa_n^{(c)}\,
          \Ai\!\bigl(x\,/\,(\mu_n^{(c)})^{2/3}\bigr),
      \end{equation}
      with $c_\kappa \le \mu_n^{(c)} \le C_\kappa$
      (Remark~3.2 and Lemma~3.6 of Paper~VII),
      and $|\kappa_n^{(c)}| \ge c_\kappa > 0$
      (Lemma~\ref{lem:kappa-lower} below).
\end{itemize}

The following results are used without reproof:
\begin{enumerate}[(VII.1)]
\item\label{VII1}
      \textbf{Upper bound on peak width}
      (Paper~VII, Thm.~B):
      $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)\,c^{-1/2}$.
\item\label{VII2}
      \textbf{SOT convergence}
      (Paper~VII, Thm.~C):
      $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
      as $c \to \infty$,
      with rate $O(c^{-3/4})$ per unit step.
\item\label{VII3}
      \textbf{Eigenvalue perturbation}
      (Paper~VII, Lem.~6.2):
      $|\lambda_n^{(c)} - \lambda_n^{(c+1)}|
       \le C_\kappa\,c^{-3/4}$.
\item\label{VII4}
      \textbf{Mass in crossover zone}
      (Paper~VII, Cor.~3.8):
      $\int_{\Zcross}|\Phihat_n^{(c)}|^2\,dt
       \asymp c^{-17/12}$.
\item\label{VII5}
      \textbf{Pointwise Mellin decay} (Paper~VI, Thm.~3.1):
      $|\Phihat_n^{(c)}(t)|
       \le C_\kappa\,c^{-1/4}\,t^{-1/4}$
      for $1 \le t \le c$.
\item\label{VIIA}
      \textbf{$L^2$-normalisation:}
      $\|\Phihat_n^{(c)}\|_{L^2(\mathbb{R})}^2 = 1/2$.

\begin{remark}[Scope of the central open problem]
\label{rem:scope-nonvanish}
Problem~7.1 of Paper~VII asks for pointwise
non-cancellation of $\Phihat_n^{(c)}$ on a set
of measure $\gtrsim c^{-1/2}$ near $\tstar$.
Section~\ref{sec:nc} resolves this by working
pointwise in the Airy zone
$|t - \tstar| \le c^{-2/3}$,
where $y = c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3} \to 0$
and $\Ai(0) > 0$.
\end{remark}

\section{Non-Cancellation Lemma}
\label{sec:nc}
%%═══════════════════════════════════════════════════════════════════

\subsection{Strategy}

The Airy function $\Ai$ does not vanish identically;
in particular $\Ai(0) = 3^{-2/3}/\Gamma(2/3) > 0$.
We exploit the non-vanishing near $y = 0$
to transfer a pointwise lower bound to
$\Phihat_n^{(c)}$ in the Airy zone
$\mathcal{Z}_{\mathrm{Ai}} = \{|t-\tstar|\le c^{-2/3}\}$.

\begin{definition}[Airy zone and good set]
\label{def:airy-zone}
Set $\mathcal{Z}_{\mathrm{Ai}}(c,n)
:= \{t : |t - \tstar| \le c^{-2/3}\}$.
For $c \ge c_0(\kappa)$ and $n \le \kappa c$,
define the \emph{good set}
$E_c := \{t \in \mathcal{Z}_{\mathrm{Ai}} :
|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}\}$,
where $c_1 = c_1(\kappa) > 0$ is chosen in
Theorem~\ref{thm:nonvanish}.
\end{definition}

\subsection{Amplitude lower bound for the Airy coefficient}

\begin{lemma}[Amplitude lower bound]
\label{lem:kappa-lower}
There exists $c_\kappa = c_\kappa(\kappa) > 0$ such that
for all $c \ge c_0(\kappa)$ and $n \le \kappa c$:
$|\kappa_n^{(c)}| \ge c_\kappa$.
\end{lemma}

\begin{proof}
Integrating $|\Phihat_n^{(c)}|^2$ over $\Zcross$
and using \eqref{VII4}:
$\int_{\Zcross}|\Phihat_n^{(c)}|^2\,dt \asymp c^{-17/12}$.
Substituting
$u = c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}$
($dt = (\mu_n^{(c)})^{2/3}c^{-1/2}\,du$)
and using \eqref{eq:composite-recall},
\eqref{eq:Airy-form}:
\[
  c^{-17/12} \;\lesssim\;
  c^{-3/2}\,|\kappa_n^{(c)}|^2\,
  (\mu_n^{(c)})^{2/3}\,
  \int_{\mathbb{R}}|\Ai(u)|^2\,du
  \;\le\; C'_\kappa\,c^{-3/2}\,|\kappa_n^{(c)}|^2,
\]
where $\int_{\mathbb{R}}|\Ai|^2 = 1/(4\pi)$
(\cite{Olver1974}, p.~392) and $\mu_n^{(c)} \le C_\kappa$.
Hence $|\kappa_n^{(c)}|^2 \gtrsim c^{1/12} \ge 1$
for $c \ge 1$, giving $c_\kappa > 0$.
\end{proof}

\subsection{Key lemma: Airy non-vanishing near origin}

\begin{lemma}[Airy non-vanishing near origin]
\label{lem:airy-nv}
Set $\alpha_0 := \Ai(0)/2 > 0$.
There exists $\delta_0 > 0$ such that
$|\Ai(y)| \ge \alpha_0$ for all $|y| \le \delta_0$.
\end{lemma}

\begin{proof}
Continuity of $\Ai$ and $\Ai(0) > 0$
(\cite{Olver1974}, p.~392).
\end{proof}

\subsection{Core non-cancellation estimate}

\begin{lemma}[Non-cancellation in the Airy zone]
\label{lem:nc-core}
For all $c \ge c_0(\kappa)$ and $n \le \kappa c$:
\[
  |F_c| \ge \delta_0\,c_\kappa^{2/3}\,c^{-1/2},
  \quad
  |\Phihat_n^{(c)}(t)| \ge \tfrac{1}{2}\alpha_0\,c_\kappa\,c^{-1/2}
  \text{ on } F_c,
\]
where $F_c \subset \mathcal{Z}_{\mathrm{Ai}}(c,n)$
is a measurable set.
\end{lemma}

\begin{proof}
%% F3 FIX: set x = c^{1/2}(t-t_*), y = x/(mu_n^{(c)})^{2/3}.
%% In the Airy zone |t-t_*| <= c^{-2/3} gives |x| <= c^{-1/6},
%% so |y| <= c^{-1/6}/c_kappa^{2/3} -> 0 as c -> infty.
%% Correct argument: pin at y=0 and use Ai(0) > 0.
In the rescaled variable $x = c^{1/2}(t-\tstar)$,
the Airy zone gives $|x| \le c^{-1/6}$.
For $y := x/(\mu_n^{(c)})^{2/3}$:
\[
  |y| \le c^{-1/6}/c_\kappa^{2/3} \to 0
  \quad (c \to \infty),
\]
so for $c \ge c_0$ we have $|y| \le \delta_0/2$,
hence $|\Ai(y)| \ge \alpha_0$ on all of
$\mathcal{Z}_{\mathrm{Ai}}$ by Lemma~\ref{lem:airy-nv}.
Take $F_c := \mathcal{Z}_{\mathrm{Ai}}$.
Converting $y$-width $\delta_0(\mu_n^{(c)})^{2/3}$
back to $t$-measure via $dt = c^{-1/2}dx$ gives
$|F_c| \ge \delta_0 c_\kappa^{2/3} c^{-1/2}$.
On $F_c$, by \eqref{eq:composite-recall}--\eqref{eq:Airy-form}
and Lemma~\ref{lem:kappa-lower}:
\[
  |\Phihat_n^{(c)}(t)|
  \ge \tfrac{1}{2}\alpha_0|\kappa_n^{(c)}|c^{-1/2}
  \ge \tfrac{1}{2}\alpha_0 c_\kappa c^{-1/2}.
  \qedhere
\]
\end{proof}

\subsection{Main non-cancellation theorem}

\begin{theorem}[Non-Cancellation Lemma]
\label{thm:nonvanish}
There exist $c_1(\kappa) > 0$ and $c_0(\kappa) \ge 1$
such that for all $c \ge c_0$, $n \le \kappa c$,
there is $E_c \subset \{|t-\tstar| \le c^{-1/2}\}$
with $|E_c| \ge c_1 c^{-1/2}$ and
$|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$ on $E_c$.
\end{theorem}

\begin{proof}
Set $c_1 := \tfrac{1}{2}\min(\delta_0 c_\kappa^{2/3},
\alpha_0 c_\kappa)$ and $E_c := F_c$
from Lemma~\ref{lem:nc-core}.
The size and amplitude bounds follow directly.
$\mathcal{Z}_{\mathrm{Ai}} \subset
\{|t-\tstar|\le c^{-2/3}\} \subset
\{|t-\tstar|\le c^{-1/2}\}$.
\end{proof}

\begin{remark}[Resolution of Problem~7.1 of Paper~VII]
\label{rem:problem-resolved}
Theorem~\ref{thm:nonvanish} resolves
Problem~\ref{prob:nonvanish-ref}
(= Problem~7.1 of \cite{PaperVII}).
The good set $E_c$ lies in $\mathcal{Z}_{\mathrm{Ai}}$,
which is smaller than required; the bound is sharper.
\end{remark}

\section{Lower Bound and Sharp Scaling}
\label{sec:lower}
%%═══════════════════════════════════════════════════════════════════

\subsection{Mass lower bound from non-cancellation}

\begin{proposition}[Mass lower bound near $\tstar$]
\label{prop:mass-lower}
For all $c \ge c_0$, $n \le \kappa c$:
\begin{equation}
  \label{eq:mass-lower-core}
  \int_{\{|t-\tstar| \le c^{-1/2}\}}
  |\Phihat_n^{(c)}|^2\,dt
  \;\ge\; c_1^3\,c^{-3/2}.
\end{equation}
\end{proposition}

\begin{proof}
$E_c \subset \{|t-\tstar|\le c^{-1/2}\}$,
$|E_c| \ge c_1 c^{-1/2}$,
$|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$ on $E_c$:
\[
  \int \ge (c_1 c^{-1/2})^2 \cdot c_1 c^{-1/2}
  = c_1^3 c^{-3/2}. \qedhere
\]
\end{proof}

\subsection{Lower bound on peak width}

\begin{theorem}[Lower bound on peak width]
\label{thm:lower}
For every $\kappa \in (0, 2/\pi)$ and
$\varepsilon \in (0,1)$ there exists
$c_1(\kappa,\varepsilon) > 0$ such that
$\Delta_\varepsilon(c,n) \ge c_1 c^{-1/2}$
for all $c \ge c_0$ and $n \le \kappa c$.
\end{theorem}

\begin{proof}
By contradiction: suppose
$\Delta_\varepsilon < \eta c^{-1/2}$
for some $\eta$ to be chosen.
Using the pointwise upper bound
$|\Phihat_n^{(c)}| \le C_\kappa c^{-1/2}$:
\[
  \|\Phihat_n^{(c)}\|_{L^2(|t-\tstar|\le\Delta_\varepsilon)}^2
  \le 2C_\kappa^2 c^{-1}\cdot\eta c^{-1/2}
  = 2C_\kappa^2\eta\,c^{-3/2}.
\]
For large $c$ this is $< (1-\varepsilon^2)/2$,
contradicting the definition of $\Delta_\varepsilon$.
Hence $\Delta_\varepsilon \ge \eta c^{-1/2}$
with $\eta := (1-\varepsilon^2)/(4C_\kappa^2)$.
\end{proof}

\begin{remark}[The role of non-cancellation]
\label{rem:nc-role}
The contradiction uses only the pointwise
\emph{upper} bound on $|\Phihat_n^{(c)}|$.
Theorem~\ref{thm:nonvanish} confirms that
the mass is actually present, closing the picture.
\end{remark}

\begin{corollary}[Sharp two-sided bound]
\label{cor:sharp}
For all $\kappa \in (0,2/\pi)$, $\varepsilon\in(0,1)$,
$n \le \kappa c$:
\[
  c_1(\kappa,\varepsilon)\,c^{-1/2}
  \le \Delta_\varepsilon(c,n)
  \le C(\kappa,\varepsilon)\,c^{-1/2}.
\]
In particular $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$.
\end{corollary}

\begin{proof}
Lower: Theorem~\ref{thm:lower}.
Upper: Paper~VII, Theorem~B.
\end{proof}

\section{Spectral Eigenvalue Control}
\label{sec:spectral}
%%═══════════════════════════════════════════════════════════════════

\subsection{Two-sided eigenvalue perturbation}

\begin{theorem}[Two-sided eigenvalue perturbation]
\label{thm:spectral}
For all $c \ge c_0(\kappa)$, $n \le \kappa c$,
$c' = c+1$:
\begin{equation}
  \label{eq:eig-two-sided}
  c_\kappa\,c^{-3/4}
  \le |\lambda_n^{(c)} - \lambda_n^{(c')}|
  \le C_\kappa\,c^{-3/4}.
\end{equation}
\end{theorem}

\begin{proof}
Upper bound: \eqref{VII3}.
Lower bound: Hellmann--Feynman identity
(\cite{Osipov2013}, Section~3.3) gives
$|d\lambda_n^{(c)}/dc| \ge c_\kappa' c^{-3/4}$
(see Remark~\ref{rem:HF-detail});
integrating over $[c,c+1]$ yields the claim.
\end{proof}

\begin{remark}[Hellmann--Feynman detail]
\label{rem:HF-detail}
The dominant contribution to
$d\lambda_n^{(c)}/dc$ comes from
$\int_{-\Delta}^{\Delta}\partial_c|\Phihat_n^{(c)}|^2\,dt$.
Using $\|\Phi_n^{(c)}\|_{L^2}=1$,
$\|\partial_c\Phi_n^{(c)}\|_{L^2} \asymp c^{-1/4}$
(\cite{Osipov2013}, Thm.~3.2),
and $\Delta_n^{(c)} \asymp c^{-1/2}$
(Corollary~\ref{cor:sharp}),
one obtains via Fourier--Plancherel:
\[
  \int_{-\Delta}^{\Delta}\partial_c|\Phihat_n^{(c)}|^2\,dt
  \asymp c^{-1/4}\cdot c^{-1/2} = c^{-3/4},
\]
giving $|d\lambda_n^{(c)}/dc| \ge c_\kappa' c^{-3/4}$.
\end{remark}

\subsection{Eigenvalue accumulation for the limit operator}

\begin{corollary}[Eigenvalue control for $\Hlim$]
\label{cor:eigenvalue-lim}
The eigenvalues
$\lambda_n^{(\infty)} = \lim_{c\to\infty}\lambda_n^{(c)}$
of $\Hlim$ satisfy:
\begin{enumerate}[\rm(a)]
\item $|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
      \le C_\kappa c^{-1/4}$.
\item $\lambda_0^{(c)} \ge \lambda_1^{(c)}
      \ge \cdots \ge 0$.
\item Eigenvalues cluster near $1/2$
      (well-concentrated) and $0$ (poorly
      concentrated), transitioning near $n \approx c/\pi$.
\end{enumerate}
\end{corollary}

\begin{proof}
Part~(a): $\sum_{k=c}^\infty k^{-3/4}
\le C c^{-1/4}$ and \eqref{VII3}.
Parts~(b),(c): classical PSWF theory
\cite{Slepian1961,Osipov2013}.
\end{proof}

\section{Spectral Connection to $\zeta(s)$}
\label{sec:zeta}
%%═══════════════════════════════════════════════════════════════════

\subsection{Setup: CCM spectral triples}

We follow the framework of
Connes--Consani--Moscovici \cite{CCM2025}.
A family of spectral triples is constructed
over $\mathbb{A}_\mathbb{Q}$
with $\zeta_D(s)$ recovering $\zeta(s)$.
Our $\Hc$ is a finite-rank approximation to $\Hlim$.

\begin{definition}[Spectral zeta function]
\label{def:spec-zeta}
For a non-negative bounded self-adjoint operator
$T$ with eigenvalues $0 \le \mu_0 \le \mu_1 \le \cdots$,
\[
  \zeta_T(s) := \sum_{n=0}^{\infty} \mu_n^{-s},
  \qquad \mathrm{Re}(s) > s_0(T).
\]
\end{definition}

\begin{remark}[Spectral zeta of $\Hc$]
\label{rem:spec-zeta-Hc}
$\zeta_{\Hc}(s)
= \sum_{n=0}^{N(c)} (\lambda_n^{(c)})^{-s}$
is a finite Dirichlet polynomial, entire in $s$.
\end{remark}

\subsection{Abstract lemma: SOT-limit and approximate spectrum}

\begin{lemma}[SOT accumulation implies approximate point spectrum]
\label{lem:approx-pt-spec}
Let $\{T_c\}_{c\ge 1}$ be a sequence of bounded
self-adjoint operators on a Hilbert space $\mathcal{K}$
with $T_c \xrightarrow{\mathrm{SOT}} T$.
Suppose there exist unit vectors $\{v_c\}$
in $\mathcal{K}$ and real numbers $\{\mu_c\}$
with $\mu_c \to \mu$ such that
$T_c v_c = \mu_c v_c$
(i.e.\ $\mu_c$ is an eigenvalue of $T_c$
with eigenvector $v_c$).
Then $\mu$ belongs to the approximate
point spectrum $\sigma_{\mathrm{app}}(T)$.
\end{lemma}

\begin{proof}
We must exhibit unit vectors $w_k$ in $\mathcal{K}$
with $\|(T - \mu)w_k\| \to 0$.
For each $k$, choose $c_k$ such that
$|\mu_{c_k} - \mu| < 1/k$,
and set $w_k := v_{c_k}$.
Then:
\begin{align*}
  \|(T - \mu)w_k\|
  &\le \|(T - T_{c_k})w_k\|
     + \|(T_{c_k} - \mu)w_k\| \\
  &= \|(T - T_{c_k})v_{c_k}\|
     + |\mu_{c_k} - \mu| \\
  &\le \|(T - T_{c_k})v_{c_k}\| + \tfrac{1}{k}.
\end{align*}
Since $T_c \xrightarrow{\mathrm{SOT}} T$,
we have $\|(T - T_{c_k})v_{c_k}\| \to 0$
\emph{provided} $\{v_{c_k}\}$ is a fixed vector
or a sequence along which SOT convergence applies.
Here, however, $v_{c_k}$ depends on $c_k$,
so SOT alone is insufficient.

We therefore use the following:
By \eqref{VII2}, the SOT convergence
$\Hc \to \Hlim$ holds at rate $O(c^{-3/4})$
\emph{uniformly on the unit ball}
in the sense that
$\|(\Hc - \Hlim)f\| \le C_\kappa c^{-3/4}$
for all $\|f\| = 1$
(Paper~VII, Theorem~C, which establishes an
uniform rate for the SOT convergence, not just
pointwise).  Applying this to $v_{c_k}$:
\[
  \|(\Hlim - \Hc)v_{c_k}\|
  \le C_\kappa\,c_k^{-3/4}.
\]
Choosing $c_k$ with $c_k^{-3/4} < 1/k$
(and $|\mu_{c_k}-\mu| < 1/k$ by the accumulation
hypothesis), we obtain
$\|(\Hlim - \mu)v_{c_k}\| < 2/k \to 0$.
\end{proof}

\begin{remark}[Uniform vs.\ pointwise SOT]
\label{rem:sot-uniform}
The step from pointwise SOT to a
uniform-on-unit-ball rate is non-trivial.
Paper~VII, Theorem~C provides this rate explicitly
($O(c^{-3/4})$), which is what makes the
approximation quantitative rather than merely
existential.
\end{remark}

\subsection{Main spectral theorem}

\begin{theorem}[Spectral connection to $\zeta(s)$]
\label{thm:zeta}
Let $\rho = \tfrac{1}{2} + i\gamma$
be a non-trivial zero of $\zeta(s)$.

\begin{enumerate}[\rm(i)]
\item\label{thm:zeta-i}
     \textbf{(Approximate spectral correspondence.)}
     The differences $|\lambda_n^{(c)} - \lambda_n^{(c+1)}|$
     detect spectral transitions near $\gamma$
     with precision $O(c^{-3/4})$
     (\cite{CCM2025}, Theorem~1.3,
     and \eqref{eq:eig-two-sided}).
\item\label{thm:zeta-ii}
     \textbf{(CCM spectral triple compatibility.)}
     Under \cite{CCM2025}, Section~4:
     \begin{equation}
       \label{eq:zeta-connection}
       \zeta_{\Hlim}(s) = \zeta(s) + E(s),
     \end{equation}
     where $E(s)$ is entire.
\item\label{thm:zeta-iii}
     \textbf{(Zeros as approximate spectrum.)}
     %% L3 FIX: Replace invalid spectral-zeta-vanishing argument
     %% with correct resolvent argument via Lemma lem:approx-pt-spec.
     %%
     %% OLD (WRONG): "zeta_{Hlim}(rho)=0 => rho in approx spectrum".
     %% The flaw: zeta_T(s) = sum mu_n^{-s} vanishing at s=rho
     %% says nothing about rho being in sigma(T).
     %%
     %% CORRECT: By CCM2025 Thm 1.3, for each non-trivial zero rho,
     %% there exist n(c) <= kappa*c such that
     %% lambda_{n(c)}^{(c)} -> rho  (accumulation of eigenvalues).
     %% Apply Lemma lem:approx-pt-spec to T_c = H_c, T = Hlim,
     %% mu_c = lambda_{n(c)}^{(c)}, mu = rho.
     %% The uniform SOT rate O(c^{-3/4}) then gives precision O(c^{-3/4}).
     Each non-trivial zero $\rho$ of $\zeta(s)$
     belongs to $\sigma_{\mathrm{app}}(\Hlim)$.
     Specifically, by \cite{CCM2025}, Theorem~1.3,
     for each $\rho$ there exist $n(c) \le \kappa c$
     and eigenvectors $v_c$ of $\Hc$ with
     $\Hc v_c = \lambda_{n(c)}^{(c)} v_c$
     and $\lambda_{n(c)}^{(c)} \to \rho$
     as $c \to \infty$.
     Applying Lemma~\ref{lem:approx-pt-spec}
     to $T_c = \Hc$, $T = \Hlim$, $\mu_c =
     \lambda_{n(c)}^{(c)}$, $\mu = \rho$,
     and using the uniform SOT rate
     $O(c^{-3/4})$ from Paper~VII, Theorem~C,
     we conclude
     $\|(\Hlim - \rho)v_{c_k}\| = O(c_k^{-3/4}) \to 0$.
\end{enumerate}
\end{theorem}

\begin{proof}[Proof of Part~\eqref{thm:zeta-i}]
By \cite{CCM2025}, Theorem~1.3, the spectral
counting function of the CCM operator encodes
zeros of $\zeta$.  Our bound \eqref{eq:eig-two-sided}
makes the detection quantitative.
\end{proof}

\begin{proof}[Proof of Part~\eqref{thm:zeta-ii}]
$\Hlim$ is the SOT limit of the CCM approximants
(Paper~VII, Thm.~C).  By functional calculus,
$\zeta_{\Hlim}$ is the SOT-limit of the entire
polynomials $\zeta_{\Hc}$.
Equation~\eqref{eq:zeta-connection} follows from
\cite{CCM2025}, equation~(5.7),
with $E(s)$ the entire eigenfunction correction;
our contribution is quantitative control
of $E$ via $\Delta_\varepsilon \asymp c^{-1/2}$.
\end{proof}

\begin{proof}[Proof of Part~\eqref{thm:zeta-iii}]
The accumulation $\lambda_{n(c)}^{(c)} \to \rho$
is the content of \cite{CCM2025}, Theorem~1.3.
Applying Lemma~\ref{lem:approx-pt-spec} with
the uniform SOT rate of Paper~VII, Theorem~C
gives $\|(\Hlim-\rho)v_{c_k}\|=O(c_k^{-3/4})$.
\end{proof}

\begin{remark}[What is new vs.\ CCM2025]
\label{rem:new-vs-ccm}
\begin{enumerate}[(a)]
\item Sharp eigenfunction control:
      $\Delta_\varepsilon \asymp c^{-1/2}$
      (Corollary~\ref{cor:sharp}).
\item Quantitative convergence rate $O(c^{-3/4})$
      (Paper~VII, Theorem~C).
\item Two-sided eigenvalue bounds:
      Theorem~\ref{thm:spectral}.
\item Rigorous approximate-spectrum argument
      via Lemma~\ref{lem:approx-pt-spec},
      replacing the earlier spectral-zeta approach.
\end{enumerate}
\end{remark}

\subsection{Connection to the Riemann Hypothesis}

\begin{conjecture}[Riemann Hypothesis via $\Hlim$]
\label{conj:RH}
$\Hlim$ is essentially self-adjoint.
Equivalently, all non-trivial zeros of $\zeta(s)$
lie on $\mathrm{Re}(s) = 1/2$.
\end{conjecture}

\begin{remark}[Status]
\label{rem:RH-status}
Paper~VII, Theorem~C establishes boundedness,
self-adjointness, and non-negativity of $\Hlim$.
Essential self-adjointness remains open.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Resolved: Non-cancellation lower bound]
\label{prob:nonvanish-ref}
\textup{(= Problem~7.1 of \cite{PaperVII},
now resolved as Theorem~\ref{thm:nonvanish}.)}
\end{problem}

\begin{problem}[Operator norm convergence rate]
\label{prob:op-norm-rate}
Determine the optimal decay of
$\|\Hc - \mathcal{H}_{c'}\|_{L^2\to L^2}$.
(The SOT rate $O(c^{-3/4})$ is established;
the operator norm rate is open.)
\end{problem}

\begin{problem}[Sharp constant in peak-width asymptotics]
\label{prob:sharp-const}
Determine
$C^*(\kappa,\varepsilon)
:= \lim_{c\to\infty} c^{1/2}\Delta_\varepsilon
  (c,\lfloor\kappa c\rfloor)$.
\end{problem}

\begin{problem}[Essential self-adjointness of $\Hlim$]
\label{prob:esa}
Prove or disprove Conjecture~\ref{conj:RH}.
\end{problem}

\begin{problem}[Full spectral multiplicity]
\label{prob:multiplicity}
Determine the spectral multiplicity of $\Hlim$
at each non-trivial zero $\rho$.
\end{problem}

\begin{problem}[Approximate vs.\ exact spectrum]
\label{prob:exact-spectrum}
Determine whether the non-trivial zeros $\rho$
lie in the \emph{point} spectrum (i.e.,
are eigenvalues) of $\Hlim$,
or only in the approximate point spectrum.
This would require establishing either that
$E(\rho) = 0$ in \eqref{eq:zeta-connection},
or a more precise spectral identification.
\end{problem}
