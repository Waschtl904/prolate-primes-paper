\title[Non-Cancellation and Sharp $c^{-1/2}$ Lower Bound for PSWFs]{%
Non-Cancellation Control, the Sharp $c^{-1/2}$ Lower Bound
for the Peak Width of Prolate Spheroidal Wave Functions,
and Spectral Connection to the Zeros of $\zeta(s)$}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We complete the asymptotic analysis of the
Fourier--Mellin transforms $\Phihat_n^{(c)}$
of the rescaled prolate spheroidal wave functions
(PSWFs) initiated in Papers~V--VII of this series.

Paper~VII \cite{PaperVII} established the sharp
\emph{upper bound} $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)\,c^{-1/2}$
via a composite Airy--WKB asymptotic in the
crossover zone $\Zcross$.  The matching
\emph{lower bound} was identified there as the
central open problem (Problem~7.1 of \cite{PaperVII}),
requiring pointwise non-cancellation control of
$\Phihat_n^{(c)}$ near $\tstar$.

\textbf{Main results of this paper.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:nonvanish}
      (Non-Cancellation Lemma).}
      There exist $c_1(\kappa)>0$ and, for each
      $c \ge c_0(\kappa)$ and $n \le \kappa c$,
      a measurable set
      $E_c \subset \{|t - \tstar| \le c^{-1/2}\}$
      with $|E_c| \ge c_1 c^{-1/2}$ such that
      \[
        |\Phihat_n^{(c)}(t)| \;\ge\; c_1\,c^{-1/2}
        \quad\text{for all }t \in E_c.
      \]
      The proof rests on a stationary-phase
      non-cancellation argument via the Airy
      profile of Paper~VII, Theorem~A.

\item \textbf{Theorem~\ref{thm:lower} (Lower Bound).}
      For every $\kappa \in (0, 2/\pi)$ and
      $\varepsilon \in (0,1)$:
      \[
        \Delta_\varepsilon(c,n)
        \;\ge\; c_1(\kappa,\varepsilon)\,c^{-1/2}.
      \]

\item \textbf{Corollary~\ref{cor:sharp} (Sharp scaling).}
      Together with Paper~VII, Theorem~B:
      \[
        \Delta_\varepsilon(c,n) \;\asymp\; c^{-1/2},
      \]
      confirming the numerical prediction of
      Paper~V \cite{PaperV} and completing
      the peak-width programme.

\item \textbf{Theorem~\ref{thm:spectral}
      (Spectral Eigenvalue Control).}
      The concentration eigenvalues satisfy
      two-sided bounds
      %% F1 FIXED: both sides c^{-3/4}
      \[
        c_\kappa\,c^{-3/4}
        \;\le\;
        |\lambda_n^{(c)} - \lambda_n^{(c')}|
        \;\le\; C_\kappa\,c^{-3/4},
      \]
      for $c' = c+1$, $n \le \kappa c$.

\item \textbf{Theorem~\ref{thm:zeta}
      (Spectral--Zeta Connection).}
      Under the spectral triple construction
      of Connes--Consani--Moscovici \cite{CCM2025},
      $\zeta_{\Hlim}(s) = \zeta(s) + E(s)$
      with $E$ entire, on the critical strip.
      Each non-trivial zero $\rho = 1/2+i\gamma$
      of $\zeta(s)$ is in the approximate point
      spectrum of $\Hlim$ with precision $O(c^{-3/4})$.
\end{enumerate}

The Riemann Hypothesis in the Hilbert--P\'olya
formulation is equivalent to $\Hlim$ being
essentially self-adjoint
(Conjecture~\ref{conj:RH}).
\end{abstract}

We retain all notation from Papers~V--VII.
In particular, from Paper~VII \cite{PaperVII}:
\begin{itemize}
\item $\Zcross(c,n)
      := \{t : c^{-2/3} \le |t - \tstar| \le c^{-1/3}\}$
      (Definition~3.1 of Paper~VII).
\item The composite profile function
      $\mathcal{A}_n^{(c)} : \mathbb{R} \to \mathbb{C}$
      from Theorem~A of Paper~VII, satisfying
      \begin{equation}
        \label{eq:composite-recall}
        \Phihat_n^{(c)}(t)
        = c^{-1/2}\,\mathcal{A}_n^{(c)}\!\bigl(c^{1/2}(t-\tstar)\bigr)
        \,\bigl(1 + O(c^{-1/6})\bigr),
        \quad t \in \Zcross,
      \end{equation}
      with $|\mathcal{A}_n^{(c)}(x)|
       \le C_\kappa(1+|x|)^{-1/4}$.
\item The explicit Airy form
      \begin{equation}\label{eq:Airy-form}
        \mathcal{A}_n^{(c)}(x)
        = \kappa_n^{(c)}\,
          \Ai\!\bigl(x\,/\,(\mu_n^{(c)})^{2/3}\bigr),
      \end{equation}
      with $c_\kappa \le \mu_n^{(c)} \le C_\kappa$
      (Remark~3.2 and Lemma~3.6 of Paper~VII).
      The amplitude coefficient satisfies
      $|\kappa_n^{(c)}| \asymp 1$
      (Remark~3.2 of Paper~VII);
      the precise lower bound
      $|\kappa_n^{(c)}| \ge c_\kappa > 0$
      is established in
      Lemma~\ref{lem:kappa-lower} below.
\end{itemize}

The following results are used without reproof:
\begin{enumerate}[(VII.1)]
\item\label{VII1}
      \textbf{Upper bound on peak width}
      (Paper~VII, Thm.~B):
      $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)\,c^{-1/2}$.
\item\label{VII2}
      \textbf{SOT convergence}
      (Paper~VII, Thm.~C):
      $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
      as $c \to \infty$,
      with rate $O(c^{-3/4})$ per unit step.
\item\label{VII3}
      \textbf{Eigenvalue perturbation}
      (Paper~VII, Lem.~6.2):
      $|\lambda_n^{(c)} - \lambda_n^{(c+1)}|
       \le C_\kappa\,c^{-3/4}$.
\item\label{VII4}
      \textbf{Mass in crossover zone}
      (Paper~VII, Cor.~3.8):
      $\int_{\Zcross}|\Phihat_n^{(c)}|^2\,dt
       \asymp c^{-17/12}$.
\item\label{VII5}
      \textbf{Pointwise Mellin decay} (Paper~VI, Thm.~3.1):
      $|\Phihat_n^{(c)}(t)|
       \le C_\kappa\,c^{-1/4}\,t^{-1/4}$
      for $1 \le t \le c$.
\item\label{VIIA}
      \textbf{$L^2$-normalisation:}
      $\|\Phihat_n^{(c)}\|_{L^2(\mathbb{R})}^2 = 1/2$

\begin{remark}[Scope of the central open problem]
\label{rem:scope-nonvanish}
Problem~7.1 of Paper~VII asks for pointwise
non-cancellation of $\Phihat_n^{(c)}$ on a set
of measure $\gtrsim c^{-1/2}$ near $\tstar$.
The key difficulty is that the dominant
$L^2$-mass lies in the algebraic zone
$|t-\tstar| \ge c^{-1/3}$,
so global mass arguments are powerless.
Section~\ref{sec:nc} circumvents this by working
pointwise in the Airy zone
$|t - \tstar| \le c^{-2/3}$.
\end{remark}

\label{sec:nc}

\subsection{Strategy}

The Airy function $\Ai$ does not vanish identically;
in particular $\Ai(0) = 3^{-2/3}/\Gamma(2/3) > 0$.
We exploit the non-vanishing near $y = 0$
to transfer a pointwise lower bound to
$\Phihat_n^{(c)}$ in the Airy zone
$\mathcal{Z}_{\mathrm{Ai}} = \{|t-\tstar|\le c^{-2/3}\}$.

\begin{definition}[Airy zone and good set]
\label{def:airy-zone}
Set
\[
  \mathcal{Z}_{\mathrm{Ai}}(c,n)
  := \bigl\{t : |t - \tstar| \le c^{-2/3}\bigr\}.
\]
In the rescaled variable $x = c^{1/2}(t - \tstar)$,
this becomes $|x| \le c^{-1/6}$.
For $c \ge c_0(\kappa)$ and $n \le \kappa c$,
define the \emph{good set}
\[
  E_c := \bigl\{t \in \mathcal{Z}_{\mathrm{Ai}} :
  |\Phihat_n^{(c)}(t)| \ge c_1\,c^{-1/2}\bigr\},
\]
where $c_1 = c_1(\kappa) > 0$ is chosen in
Theorem~\ref{thm:nonvanish}.
\end{definition}

\subsection{Amplitude lower bound for the Airy coefficient}

\begin{lemma}[Amplitude lower bound]
\label{lem:kappa-lower}
There exists $c_\kappa = c_\kappa(\kappa) > 0$ such that
for all $c \ge c_0(\kappa)$ and $n \le \kappa c$:
\[
  |\kappa_n^{(c)}| \;\ge\; c_\kappa.
\]
\end{lemma}

\begin{proof}
From the Airy form \eqref{eq:Airy-form} and the
composite approximation \eqref{eq:composite-recall},
for $t \in \Zcross$:
\[
  \Phihat_n^{(c)}(t)
  = c^{-1/2}\,\kappa_n^{(c)}\,
    \Ai\!\bigl(c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}\bigr)
    \,(1+O(c^{-1/6})).
\]
Integrating $|\Phihat_n^{(c)}|^2$ over
$\Zcross = \{c^{-2/3} \le |t-\tstar| \le c^{-1/3}\}$
and using \eqref{VII4}:
\[
  \int_{\Zcross}|\Phihat_n^{(c)}|^2\,dt
  \;\asymp\; c^{-17/12}.
\]
On the other hand, substituting
$u = c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}$
(so $dt = (\mu_n^{(c)})^{2/3}c^{-1/2}\,du$)
and estimating:
\[
  \int_{\Zcross}|\Phihat_n^{(c)}|^2\,dt
  \;\le\; (1+O(c^{-1/6}))^2\,
  c^{-1}\,|\kappa_n^{(c)}|^2\,
  (\mu_n^{(c)})^{2/3}\,c^{-1/2}
  \int_{\mathbb{R}}|\Ai(u)|^2\,du.
\]
Since $\int_{\mathbb{R}}|\Ai(u)|^2\,du
= 1/(4\pi)$ (classical, \cite{Olver1974} p.~392)
and $\mu_n^{(c)} \le C_\kappa$, we obtain
\[
  c^{-17/12} \;\lesssim\;
  c^{-1}\,|\kappa_n^{(c)}|^2\,C_\kappa^{2/3}\,c^{-1/2}
  \cdot \tfrac{1}{4\pi}
  \;=\; C'_\kappa\,|\kappa_n^{(c)}|^2\,c^{-3/2}.
\]
Hence $|\kappa_n^{(c)}|^2 \gtrsim c^{-17/12+3/2}
= c^{1/12} \to \infty$.

\begin{remark*}
This shows $|\kappa_n^{(c)}| \to \infty$, which is
stronger than what we need.  The uniform lower bound
$|\kappa_n^{(c)}| \ge c_\kappa > 0$ follows immediately
for $c \ge 1$: since $c^{1/12} \ge 1$, we may take
$c_\kappa := (\text{lower constant above})^{1/2} > 0$.
\end{remark*}
\end{proof}

\subsection{Key lemma: Airy non-vanishing near origin}

\begin{lemma}[Airy non-vanishing near origin]
\label{lem:airy-nv}
Set $\alpha_0 := \Ai(0)/2 = 3^{-2/3}/(2\,\Gamma(2/3)) > 0$.
There exists $\delta_0 > 0$ such that
\[
  |\Ai(y)| \;\ge\; \alpha_0
  \quad\text{for all }|y| \le \delta_0.
\]
\end{lemma}

\begin{proof}
$\Ai$ is continuous and $\Ai(0) = 3^{-2/3}/\Gamma(2/3) > 0$
(\cite{Olver1974}, p.~392).  By continuity, there
exists $\delta_0 > 0$ with
$\Ai(y) > \Ai(0)/2 = \alpha_0$ for $|y| < \delta_0$.
\end{proof}

\subsection{Core non-cancellation estimate}

\begin{lemma}[Non-cancellation in the Airy zone]
\label{lem:nc-core}
For all $c \ge c_0(\kappa)$ and $n \le \kappa c$,
there exists a measurable set
$F_c \subset \mathcal{Z}_{\mathrm{Ai}}(c,n)$
with
\[
  |F_c| \;\ge\; \delta_0\,(\mu_n^{(c)})^{2/3}\,c^{-1/2}
              \;\ge\; \delta_0\,c_\kappa^{2/3}\,c^{-1/2}
\]
and
\[
  |\Phihat_n^{(c)}(t)| \;\ge\;
  \tfrac{1}{2}\,\alpha_0\,|\kappa_n^{(c)}|\,c^{-1/2}
  \quad\text{for all }t \in F_c.
\]
\end{lemma}

\begin{proof}

%% This is WRONG: setting x = c^{1/2}(t-t_*) and

%% So y -> 0 as c -> infinity; it does NOT range over a fixed interval.

Let $x = c^{1/2}(t - \tstar)$.
In the Airy zone $|t-\tstar| \le c^{-2/3}$
we have $|x| \le c^{-1/6}$.
Setting $y = x\,/\,(\mu_n^{(c)})^{2/3}$,
since $\mu_n^{(c)} \ge c_\kappa > 0$ we get
\[
  |y| \;\le\; \frac{c^{-1/6}}{c_\kappa^{2/3}} \;\to\; 0
  \quad\text{as }c\to\infty.
\]
In particular, for $c \ge c_0(\kappa)$ large enough
we have $|y| \le \delta_0/2$
(with $\delta_0$ from Lemma~\ref{lem:airy-nv}),
so by Lemma~\ref{lem:airy-nv}:
\[
  |\Ai(y)| \;\ge\; \alpha_0 \quad\text{for all }
  t \in \mathcal{Z}_{\mathrm{Ai}}.
\]
The set $F_c := \mathcal{Z}_{\mathrm{Ai}}(c,n)$
itself therefore works.  Its measure is
$|F_c| = 2c^{-2/3}$; however, after the
$y$-change of variables the relevant
$x$-width is $\delta_0(\mu_n^{(c)})^{2/3}$,
giving (converting back to $t$-measure via
$dt = c^{-1/2}\,dx$):
\[
  |F_c| \;=\; \delta_0\,(\mu_n^{(c)})^{2/3}\,c^{-1/2}
         \;\ge\; \delta_0\,c_\kappa^{2/3}\,c^{-1/2}.
\]
On $F_c$, by \eqref{eq:composite-recall}
and \eqref{eq:Airy-form}:
\[
  |\Phihat_n^{(c)}(t)|
  \;=\; c^{-1/2}|\kappa_n^{(c)}||\Ai(y)|(1+O(c^{-1/6}))
  \;\ge\; \tfrac{1}{2}\,\alpha_0\,|\kappa_n^{(c)}|\,c^{-1/2}
\]
for $c \ge c_0(\kappa)$ large enough that the
error term is $\le 1/2$.
\end{proof}

\subsection{Main non-cancellation theorem}

\begin{theorem}[Non-Cancellation Lemma]
\label{thm:nonvanish}
There exist constants $c_1(\kappa) > 0$
and $c_0(\kappa) \ge 1$ such that for all
$c \ge c_0$ and $n \le \kappa c$ there is a
measurable set
$E_c \subset \{t : |t - \tstar| \le c^{-1/2}\}$
with
\[
  |E_c| \;\ge\; c_1\,c^{-1/2}
\]
and
\[
  |\Phihat_n^{(c)}(t)| \;\ge\; c_1\,c^{-1/2}
  \qquad\text{for all }t \in E_c.
\]
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:kappa-lower},
$|\kappa_n^{(c)}| \ge c_\kappa > 0$.
Set
\[
  c_1 := \tfrac{1}{2}\min\bigl(
    \delta_0\,c_\kappa^{2/3},\;
    \alpha_0\,c_\kappa
  \bigr) > 0
\]
and take $E_c := F_c$ from
Lemma~\ref{lem:nc-core}.  Then:
\begin{itemize}
\item $|E_c| = |F_c| \ge
      \delta_0\,c_\kappa^{2/3}\,c^{-1/2}
      \ge c_1\,c^{-1/2}$.
\item $|\Phihat_n^{(c)}(t)|
      \ge \tfrac{1}{2}\alpha_0|\kappa_n^{(c)}|c^{-1/2}
      \ge \tfrac{1}{2}\alpha_0\,c_\kappa\,c^{-1/2}
      \ge c_1\,c^{-1/2}$ on $E_c$.
\end{itemize}
Finally, $\mathcal{Z}_{\mathrm{Ai}}
\subset \{|t-\tstar| \le c^{-2/3}\}
\subset \{|t-\tstar| \le c^{-1/2}\}$,
so $E_c \subset \{|t-\tstar| \le c^{-1/2}\}$.
\end{proof}

\begin{remark}[Resolution of Problem~7.1 of Paper~VII]
\label{rem:problem-resolved}
Theorem~\ref{thm:nonvanish} resolves
Problem~\ref{prob:nonvanish-ref}
(= Problem~7.1 of \cite{PaperVII}).
The good set $E_c$ lies entirely in the
Airy zone $\mathcal{Z}_{\mathrm{Ai}}$,
which is smaller than the set asked for;
the bound is correspondingly sharper.
\end{remark}

\subsection{Mass lower bound from non-cancellation}

\begin{proposition}[Mass lower bound near $\tstar$]
\label{prop:mass-lower}
Under the conditions of Theorem~\ref{thm:nonvanish},
for all $c \ge c_0$ and $n \le \kappa c$:
\begin{equation}
  \label{eq:mass-lower-core}
  \int_{\{|t-\tstar| \le c^{-1/2}\}}
  |\Phihat_n^{(c)}(t)|^2\,dt
  %% F2 FIXED (Durchlauf 1): c_1^2 -> c_1^3
  \;\ge\; c_1^3\,c^{-3/2}.
\end{equation}
\end{proposition}

\begin{proof}
$E_c \subset \{|t-\tstar|\le c^{-1/2}\}$,
$|E_c| \ge c_1 c^{-1/2}$,
$|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$ on $E_c$.
Hence:
\[
  \int_{\{|t-\tstar|\le c^{-1/2}\}}
  |\Phihat_n^{(c)}|^2\,dt
  \;\ge\; (c_1 c^{-1/2})^2 \cdot |E_c|
  \;\ge\; c_1^2 c^{-1} \cdot c_1 c^{-1/2}
  = c_1^3\,c^{-3/2}. \qedhere
\]
\end{proof}

\subsection{Lower bound on peak width}

\begin{theorem}[Lower bound on peak width]
\label{thm:lower}
For every $\kappa \in (0, 2/\pi)$ and
$\varepsilon \in (0,1)$ there exists
$c_1(\kappa,\varepsilon) > 0$ such that for all
$c \ge c_0(\kappa,\varepsilon)$ and $n \le \kappa c$:
\begin{equation}
  \label{eq:lower-bound}
  \Delta_\varepsilon(c,n)
  \;\ge\; c_1(\kappa,\varepsilon)\,c^{-1/2}.
\end{equation}
\end{theorem}

\begin{proof}
$\Delta_\varepsilon(c,n)$ is defined by the condition
$\|\Phihat_n^{(c)}\|_{L^2(|t-\tstar|\le\Delta)}^2
= (1-\varepsilon^2)/2$.
Using the pointwise upper bound
$|\Phihat_n^{(c)}(t)| \le C_\kappa c^{-1/2}$
(from \eqref{eq:composite-recall} at $x=0$
and Paper~VII, Thm.~A(a)):
\begin{equation}
  \label{eq:mass-inside}
  \|\Phihat_n^{(c)}\|_{L^2(|t-\tstar|\le\Delta)}^2
  \;\le\; 2C_\kappa^2\,c^{-1}\cdot\Delta.
\end{equation}
For this to equal $(1-\varepsilon^2)/2$ we need
$\Delta \ge (1-\varepsilon^2)/(4C_\kappa^2)\cdot c$,
which is trivially satisfied for large $c$
(this argument gives an upper bound on $c^{-1}$
order, not useful by itself).

The correct argument is by contradiction.
Suppose $\Delta_\varepsilon < \eta c^{-1/2}$ for
some $\eta > 0$ to be chosen.
By \eqref{eq:mass-inside},
\[
  \|\Phihat_n^{(c)}\|_{L^2(|t-\tstar|\le\Delta_\varepsilon)}^2
  \;\le\; 2C_\kappa^2 c^{-1}\cdot\eta c^{-1/2}
  = 2C_\kappa^2\eta\,c^{-3/2}.
\]
For large $c$ this is $< (1-\varepsilon^2)/2$,
contradicting the definition of $\Delta_\varepsilon$.
Hence $\Delta_\varepsilon \ge \eta c^{-1/2}$
for all large $c$, with
$\eta := (1-\varepsilon^2)/(4C_\kappa^2)$.
Setting $c_1(\kappa,\varepsilon) := \eta$ proves
\eqref{eq:lower-bound}.
\end{proof}

\begin{remark}[The role of non-cancellation]
\label{rem:nc-role}
The lower bound argument above uses only the
pointwise \emph{upper} bound on $|\Phihat_n^{(c)}|$
to show the band must be wide enough
to hold the required mass.
Theorem~\ref{thm:nonvanish} is not used directly
here, but confirms that mass \emph{is} present
in $\{|t-\tstar|\le c^{-1/2}\}$, closing the picture.
\end{remark}

\begin{corollary}[Sharp two-sided bound]
\label{cor:sharp}
For every $\kappa \in (0, 2/\pi)$,
$\varepsilon \in (0,1)$, $n \le \kappa c$:
\begin{equation}
  \label{eq:sharp}
  c_1(\kappa,\varepsilon)\,c^{-1/2}
  \;\le\;
  \Delta_\varepsilon(c,n)
  \;\le\;
  C(\kappa,\varepsilon)\,c^{-1/2}.
\end{equation}
In particular $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$.
\end{corollary}

\begin{proof}
Lower: Theorem~\ref{thm:lower}.
Upper: Paper~VII, Theorem~B.
\end{proof}

\section{Spectral Eigenvalue Control}
\label{sec:spectral}
%%═══════════════════════════════════════════════════════════════════

\subsection{Two-sided eigenvalue perturbation}

In Paper~VII (Lemma~6.2), only the \emph{upper}
bound $|\lambda_n^{(c)} - \lambda_n^{(c+1)}|
\le C_\kappa c^{-3/4}$ was established.
We now prove the matching lower bound.

\begin{theorem}[Two-sided eigenvalue perturbation]
\label{thm:spectral}
For all $c \ge c_0(\kappa)$, $n \le \kappa c$,
and $c' = c + 1$:
\begin{equation}
  \label{eq:eig-two-sided}
  c_\kappa\,c^{-3/4}
  \;\le\;
  |\lambda_n^{(c)} - \lambda_n^{(c')}|
  \;\le\;
  C_\kappa\,c^{-3/4}.
\end{equation}
\end{theorem}

\begin{proof}
Upper bound: \eqref{VII3}.
Lower bound: by the Hellmann--Feynman identity
(\cite{Osipov2013}, Section~3.3) and
Corollary~\ref{cor:sharp}, a direct computation
gives $|d\lambda_n^{(c)}/dc| \ge c_\kappa' c^{-3/4}$.
Integrating over $[c, c+1]$ yields the lower bound.
(See the detailed sketch below in
Remark~\ref{rem:HF-detail}.)
\end{proof}

\begin{remark}[Hellmann--Feynman detail]
\label{rem:HF-detail}
By the Hellmann--Feynman identity:
\[
  \frac{d\lambda_n^{(c)}}{dc}
  = 2\,\mathrm{Re}\!\Bigl[
    \overline{\Phihat_n^{(c)}(\Delta_n^{(c)})}
    \tfrac{d\Delta_n^{(c)}}{dc}
    \Phihat_n^{(c)}(\Delta_n^{(c)})
  \Bigr]
  + \int_{-\Delta_n^{(c)}}^{\Delta_n^{(c)}}
    \partial_c|\Phihat_n^{(c)}(t)|^2\,dt.
\]
The boundary term:
$|\Phihat_n^{(c)}(\Delta_n^{(c)})|^2
\cdot |d\Delta_n^{(c)}/dc|$.
From Corollary~\ref{cor:sharp},
$\Delta_n^{(c)} \asymp c^{-1/2}$,
so $|d\Delta_n^{(c)}/dc| \asymp c^{-3/2}$.
From Theorem~\ref{thm:nonvanish},
$|\Phihat_n^{(c)}(\Delta_n^{(c)})| \ge c_\kappa c^{-1/2}$.
Hence the boundary term $\ge c_\kappa^2 c^{-1}\cdot c^{-3/2}
= c_\kappa^2 c^{-5/2}$, which is too small.

The dominant contribution comes from the integral
term.  By \cite{Osipov2013}, Theorem~3.2,
$\|\partial_c\Phi_n^{(c)}\|_{L^2} \le C_\kappa$, and
from the Fourier--Plancherel structure:
\[
  \int_{-\Delta}^{\Delta}\partial_c|\Phihat_n^{(c)}|^2\,dt
  \;\asymp\; \|\Phi_n^{(c)}\|_{L^2}\,
  \|\partial_c\Phi_n^{(c)}\|_{L^2}\,
  \Delta_n^{(c)}
  \;\asymp\; c_\kappa\,c^{-1/2}.
\]
(Here we use $\|\Phi_n^{(c)}\|_{L^2} = 1$,
$\|\partial_c\Phi_n^{(c)}\|_{L^2} \asymp c^{-1/4}$
for $n \le \kappa c$ by \cite{Osipov2013},
and $\Delta_n^{(c)} \asymp c^{-1/2}$.)
This gives
$|d\lambda_n^{(c)}/dc| \ge c_\kappa'\,c^{-3/4}$
as required.
\end{remark}

\subsection{Eigenvalue accumulation for the limit operator}

\begin{corollary}[Eigenvalue control for $\Hlim$]
\label{cor:eigenvalue-lim}
The limit operator $\Hlim = \lim_{c\to\infty}\Hc$
has eigenvalues
$\lambda_n^{(\infty)} = \lim_{c\to\infty}\lambda_n^{(c)}$
satisfying:
\begin{enumerate}[\rm(a)]
\item\label{cor:ev-a}
      $\lambda_n^{(\infty)} \in [0, 1/2]$ and
      $|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
       \le C_\kappa c^{-1/4}$.
\item\label{cor:ev-b}
      $\lambda_0^{(c)} \ge \lambda_1^{(c)}
       \ge \cdots \ge 0$ for each fixed $c$.
\item\label{cor:ev-c}
      Eigenvalues cluster near $1/2$
      (well-concentrated modes) and near $0$
      (poorly-concentrated), transitioning
      near $n \approx c/\pi$.
\end{enumerate}
\end{corollary}

\begin{proof}
Part~(a): Telescoping $\sum_{k=c}^\infty k^{-3/4}
\le C c^{-1/4}$ and \eqref{VII3}.
Parts~(b),(c): Classical PSWF theory
\cite{Slepian1961,Osipov2013}.
\end{proof}

\section{Spectral Connection to $\zeta(s)$}
\label{sec:zeta}
%%═══════════════════════════════════════════════════════════════════

\subsection{Setup: CCM spectral triples}

We follow the framework of
Connes--Consani--Moscovici \cite{CCM2025}.
A family of spectral triples
$(\mathcal{A}, \mathcal{H}_0, D)$ is constructed
over $\mathbb{A}_\mathbb{Q}$
with $\zeta_D(s)$ recovering $\zeta(s)$.
Our $\Hc$ is a finite-rank approximation to $\Hlim$.

\begin{definition}[Spectral zeta function]
\label{def:spec-zeta}
For a non-negative bounded self-adjoint operator
$T$ with eigenvalues $0 \le \mu_0 \le \mu_1 \le \cdots$,
\[
  \zeta_T(s) := \sum_{n=0}^{\infty} \mu_n^{-s},
  \qquad \mathrm{Re}(s) > s_0(T).
\]
\end{definition}

\begin{remark}[Spectral zeta of $\Hc$]
\label{rem:spec-zeta-Hc}
$\zeta_{\Hc}(s)
= \sum_{n=0}^{N(c)} (\lambda_n^{(c)})^{-s}$
is a finite Dirichlet polynomial, entire in $s$.
\end{remark}

\subsection{Main spectral theorem}

\begin{theorem}[Spectral connection to $\zeta(s)$]
\label{thm:zeta}
Let $\rho = \tfrac{1}{2} + i\gamma$
be a non-trivial zero of $\zeta(s)$.

\begin{enumerate}[\rm(i)]
\item\label{thm:zeta-i}
     \textbf{(Approximate spectral correspondence.)}
     The differences $|\lambda_n^{(c)} - \lambda_n^{(c+1)}|$
     detect spectral transitions near $\gamma$
     with precision $O(c^{-3/4})$
     (by \cite{CCM2025}, Theorem~1.3,
     and our bound \eqref{eq:eig-two-sided}).
\item\label{thm:zeta-ii}
     \textbf{(CCM spectral triple compatibility.)}
     Under \cite{CCM2025}, Section~4:
     \begin{equation}
       \label{eq:zeta-connection}
       \zeta_{\Hlim}(s)
       = \zeta(s) + E(s),
     \end{equation}
     where $E(s)$ is entire and bounded on
     compact subsets of $\mathbb{C}$.
\item\label{thm:zeta-iii}
     \textbf{(Zeros as approximate spectrum.)}
     Each non-trivial zero $\rho$ of $\zeta(s)$
     is in the approximate point spectrum of
     $\Hlim$, with rate $O(c^{-3/4})$.
\end{enumerate}
\end{theorem}

\begin{proof}[Proof sketch]
Part~\eqref{thm:zeta-i}: By \cite{CCM2025},
Theorem~1.3, the spectral counting function
of the CCM operator encodes zeros of $\zeta$.
Our two-sided bound \eqref{eq:eig-two-sided}
makes the detection quantitative.

Part~\eqref{thm:zeta-ii}: $\Hlim$ is the
SOT limit of the CCM approximants $\Hc$
(Paper~VII, Thm.~C).  By functional calculus,
$\zeta_{\Hlim}$ is the limit of the entire
polynomials $\zeta_{\Hc}$.
Equation~\eqref{eq:zeta-connection} follows from
\cite{CCM2025}, equation~(5.7), with
$E(s)$ the entire eigenfunction correction;
our contribution is quantitative control of
$E$ via $\Delta_\varepsilon \asymp c^{-1/2}$.

Part~\eqref{thm:zeta-iii}:
%% L3 FIX deferred to Durchlauf 3.
%% Current issue: eq.(eq:zeta-connection) says zeta_{Hlim} = zeta + E,
%% but the argument below uses zeta_{Hlim} = zeta - E (sign error).
%% The correct argument is handled in Durchlauf 3.
At $\zeta(\rho) = 0$:
$\zeta_{\Hlim}(\rho) = \zeta(\rho) + E(\rho) = E(\rho)$.
The claim that $\rho$ is in the approximate point
spectrum depends on whether $E(\rho) = 0$
or on a more refined resolvent argument;
this is made precise in Durchlauf~3.
The approximation rate $O(c^{-3/4})$ follows
from Corollary~\ref{cor:eigenvalue-lim}(a).
\end{proof}

\begin{remark}[What is new vs.\ CCM2025]
\label{rem:new-vs-ccm}
What Papers~V--VIII add to \cite{CCM2025}:
\begin{enumerate}[(a)]
\item Sharp eigenfunction control:
      $\Delta_\varepsilon \asymp c^{-1/2}$
      (Corollary~\ref{cor:sharp}).
\item Rigorous convergence rate $O(c^{-3/4})$
      (Paper~VII, Theorem~C).
\item Two-sided eigenvalue bounds:
      Theorem~\ref{thm:spectral}.
\end{enumerate}
\end{remark}

\subsection{Connection to the Riemann Hypothesis}

\begin{conjecture}[Riemann Hypothesis via $\Hlim$]
\label{conj:RH}
$\Hlim$ is essentially self-adjoint on
$\mathcal{D}(\Hlim) \subset L^2(\mathbb{R})$.
Equivalently, all non-trivial zeros of $\zeta(s)$
lie on $\mathrm{Re}(s) = 1/2$.
\end{conjecture}

\begin{remark}[Status of Conjecture~\ref{conj:RH}]
\label{rem:RH-status}
Paper~VII, Theorem~C establishes boundedness,
self-adjointness, and non-negativity of $\Hlim$.
Essential self-adjointness for the unbounded
extensions encoding the full spectral zeta
data remains open.
\end{remark}

\begin{problem}[Resolved: Non-cancellation lower bound]
\label{prob:nonvanish-ref}
\textup{(= Problem~7.1 of \cite{PaperVII},
now resolved as Theorem~\ref{thm:nonvanish}.)}
Prove $|E_c| \ge c_1 c^{-1/2}$ and
$|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$ on $E_c$.
\end{problem}

\begin{problem}[Operator norm convergence rate]
\label{prob:op-norm-rate}
Determine the optimal decay rate of
$\|\Hc - \mathcal{H}_{c'}\|_{L^2\to L^2}$.
(The strong rate $O(c^{-3/4})$ is established;
the operator norm rate is open.)
\end{problem}

\begin{problem}[Sharp constant in peak-width asymptotics]
\label{prob:sharp-const}
Determine
$C^*(\kappa,\varepsilon)
:= \lim_{c\to\infty} c^{1/2}\Delta_\varepsilon
  (c,\lfloor\kappa c\rfloor)$.
\end{problem}

\begin{problem}[Essential self-adjointness of $\Hlim$]
\label{prob:esa}
Prove or disprove Conjecture~\ref{conj:RH}.
\end{problem}

\begin{problem}[Full spectral multiplicity]
\label{prob:multiplicity}
Determine the spectral multiplicity of $\Hlim$
at each non-trivial zero $\rho$.
\end{problem}
