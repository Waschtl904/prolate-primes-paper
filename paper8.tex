\title[Non-Cancellation and Sharp $c^{-1/2}$ Lower Bound for PSWFs]{%
Non-Cancellation Control, the Sharp $c^{-1/2}$ Lower Bound
for the Peak Width of Prolate Spheroidal Wave Functions,
and Spectral Connection to the Zeros of $\zeta(s)$}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We complete the asymptotic analysis of the
Fourier--Mellin transforms $\Phihat_n^{(c)}$
of the rescaled prolate spheroidal wave functions
(PSWFs) initiated in Papers~V--VII of this series.

Paper~VII \cite{PaperVII} established the sharp
\emph{upper bound} $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)\,c^{-1/2}$
via composite Airy--WKB asymptotics.
The matching \emph{lower bound} (Problem~7.1 of
\cite{PaperVII}) required pointwise non-cancellation
of $\Phihat_n^{(c)}$ on a set of measure
$\gtrsim c^{-1/2}$ near the turning point $\tstar$.

\textbf{Main results.}
\begin{enumerate}
\item \textbf{Lemma~\ref{lem:airy-window}
      (Airy window, key new observation).}
      In the central window $|t-\tstar| \le c^{-1/2}$,
      the rescaled Airy argument $y = c^{1/2}(t-\tstar)/
      (\mu_n^{(c)})^{2/3}$ ranges over a compact
      $O(1)$ interval in $y$, independent of $c$.
      Hence the function
      $y \mapsto \Ai(y)$ has only finitely many
      (uniformly bounded) zeros in this window,
      and each zero-free interval has
      $t$-width $\asymp c^{-1/2}$.

\item \textbf{Theorem~\ref{thm:nonvanish}
      (Non-Cancellation Lemma, proved).}
      There exist $c_1(\kappa)>0$ and a set
      $E_c \subset \{|t-\tstar| \le c^{-1/2}\}$
      with $|E_c| \ge c_1 c^{-1/2}$ and
      $|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$
      on $E_c$.
      \emph{Proof:} the window contains $O(1)$
      Airy zeros; between consecutive zeros
      $|\Ai| \ge c_\kappa(1+|y|)^{-1/4}$;
      converting back to $t$-measure gives
      one zero-free interval of width $\asymp c^{-1/2}$.

\item \textbf{Theorem~\ref{thm:lower}
      (Lower bound, unconditional).}
      $\Delta_\varepsilon(c,n) \ge c_1(\kappa,\varepsilon)\,c^{-1/2}$.

\item \textbf{Corollary~\ref{cor:sharp}
      (Sharp scaling).}
      $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$,
      completing Problem~7.1 of \cite{PaperVII}.

\item \textbf{Theorem~\ref{thm:spectral}
      (Eigenvalue control, unconditional).}
      $c_\kappa c^{-3/4} \le
      |\lambda_n^{(c)}-\lambda_n^{(c+1)}|
      \le C_\kappa c^{-3/4}$.

\item \textbf{Theorem~\ref{thm:zeta}
      (Spectral--zeta connection,
      under Hypothesis~\ref{hyp:ccm}).}
      Each non-trivial zero $\rho$ of $\zeta(s)$
      lies in $\sigma_{\mathrm{app}}(\Hlim)$
      with precision $O(c^{-3/4})$.
\end{enumerate}
\end{abstract}

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--VII.
In particular:
\begin{itemize}
\item $\Zcross(c,n)
      := \{t : c^{-2/3} \le |t - \tstar| \le c^{-1/3}\}$.
\item $\Zai(c,n) := \{t : |t-\tstar| \le c^{-2/3}\}$
      (Airy zone, $|\Zai| = 2c^{-2/3}$).
\item Composite profile (Paper~VII, Thm.~A):
      \begin{equation}
        \label{eq:composite-recall}
        \Phihat_n^{(c)}(t)
        = c^{-1/2}\mathcal{A}_n^{(c)}\!\bigl(c^{1/2}(t-\tstar)\bigr)
        (1 + O(c^{-1/3})),
        \quad t \in \Zcross,
      \end{equation}
      with Airy form
      \begin{equation}\label{eq:Airy-form}
        \mathcal{A}_n^{(c)}(x)
        = \kappa_n^{(c)}\Ai\!\bigl(x/(\mu_n^{(c)})^{2/3}\bigr),
        \quad c_\kappa \le \mu_n^{(c)} \le C_\kappa.
      \end{equation}
\end{itemize}

Inherited results used without reproof:
\begin{enumerate}[(VII.1)]
\item\label{VII1}
      $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)c^{-1/2}$
      (Paper~VII, Thm.~B).
\item\label{VII2}
      $\|(\Hc-\Hlim)f\| \le C_\kappa c^{-3/4}\|f\|$
      for every fixed unit vector $f$
      (Paper~VII, Thm.~C; \emph{strong rate,
      not operator norm}).
\item\label{VII3}
      $|\lambda_n^{(c)}-\lambda_n^{(c+1)}|
      \le C_\kappa c^{-3/4}$
      (Paper~VII, Lem.~6.2).
\item\label{VII4}
      $\int_{\Zcross}|\Phihat_n^{(c)}|^2 dt
      \asymp c^{-17/12}$
      (Paper~VII, Cor.~3.8).
\item\label{VII5}
      $|\Phihat_n^{(c)}(t)| \le C_\kappa c^{-1/2}$
      for $t \asymp c^{1/2}$
      (Paper~VI, Thm.~3.1).
\item\label{VIIA}
      $\|\Phihat_n^{(c)}\|_{L^2}^2 = 1/2$.
\end{enumerate}

\section{Non-Cancellation and Anti-Concentration}
\label{sec:nc}
%%═══════════════════════════════════════════════════════════════════

\subsection{Strategy and key observation}

The proof has two ingredients:
\begin{enumerate}[(I)]
\item \emph{Amplitude:} $|\Ai(y)| \ge c_0' > 0$
      on zero-free intervals.
\item \emph{Anti-concentration:} the central window
      $|t-\tstar| \le c^{-1/2}$ corresponds, in
      the rescaled variable
      $y = c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}$,
      to a \emph{compact interval of size $O(1)$},
      containing only finitely many zeros of $\Ai$.
\end{enumerate}
Ingredient~(I) gives the amplitude.
Ingredient~(II) ensures at least one zero-free
interval of $t$-width $\asymp c^{-1/2}$ lies
inside the window.

\subsection{Amplitude lower bound for $\kappa_n^{(c)}$}

\begin{lemma}[Amplitude lower bound]
\label{lem:kappa-lower}
For all $c \ge c_0(\kappa)$, $n \le \kappa c$:
$|\kappa_n^{(c)}| \ge c_\kappa > 0$.
\end{lemma}

\begin{proof}
From \eqref{VII4}, integrating $|\Phihat_n^{(c)}|^2$
over $\Zcross$ and substituting \eqref{eq:composite-recall}--
\eqref{eq:Airy-form} with
$u = c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}$:
\[
  c^{-17/12} \lesssim
  c^{-3/2}|\kappa_n^{(c)}|^2(\mu_n^{(c)})^{2/3}
  \textstyle\int|\Ai|^2
  \le C_\kappa' c^{-3/2}|\kappa_n^{(c)}|^2,
\]
so $|\kappa_n^{(c)}|^2 \gtrsim c^{1/12} \ge 1$.
We use only the lower bound $|\kappa_n^{(c)}| \ge c_\kappa$;
the possible growth $|\kappa_n^{(c)}| \asymp c^{1/24}$
is harmless and consistent with $L^2$-normalisation.
\end{proof}

\subsection{The Airy window: $O(1)$ zeros in the central window}

\begin{lemma}[Airy window]
\label{lem:airy-window}
For all $c \ge c_0(\kappa)$ and $n \le \kappa c$,
the map
\[
  t \mapsto y(t) := \frac{c^{1/2}(t-\tstar)}{(\mu_n^{(c)})^{2/3}}
\]
restricted to $\{|t-\tstar| \le c^{-1/2}\}$
takes values in the compact interval
\begin{equation}
  \label{eq:y-window}
  y \in [-Y_0, Y_0],
  \qquad
  Y_0 := \frac{1}{c_\kappa^{2/3}},
\end{equation}
where $c_\kappa > 0$ is the lower bound from
$\mu_n^{(c)} \ge c_\kappa$.
In particular:
\begin{enumerate}[\rm(a)]
\item $Y_0$ is a universal constant
      (independent of $c$ and $n$).
\item $\Ai$ has at most $N_0 = N_0(\kappa) < \infty$
      zeros in $[-Y_0, Y_0]$.
\item Between any two consecutive zeros
      $y_k < y_{k+1}$ of $\Ai$ in $[-Y_0,Y_0]$
      (or between $-Y_0$ and the first zero,
      or between the last zero and $Y_0$),
      the corresponding $t$-interval has width
      \[
        \Delta t_k
        = (y_{k+1}-y_k)\,
          \frac{(\mu_n^{(c)})^{2/3}}{c^{1/2}}
        \ge \frac{(y_{k+1}-y_k)\,c_\kappa^{2/3}}{c^{1/2}}.
      \]
\item Since $\Ai(0) > 0$ and the first
      negative zero of $\Ai$ is
      $-a_1 \approx -2.338$ \cite{Olver1974},
      the interval $y \in (0, \min(a_1, Y_0))$
      is zero-free with width
      $\ge \min(a_1, Y_0) > 0$,
      giving a $t$-interval of width
      $\ge \min(a_1,Y_0)\,c_\kappa^{2/3}\,c^{-1/2}
      \ge c_* c^{-1/2}$
      for $c_* := \min(a_1,Y_0)\,c_\kappa^{2/3} > 0$.
\end{enumerate}
\end{lemma}

\begin{proof}
The $y$-range follows directly: $|y| \le
c^{1/2}\cdot c^{-1/2}/(\mu_n^{(c)})^{2/3}
\le 1/c_\kappa^{2/3} = Y_0$.
This is independent of $c$.
Finiteness of zeros in $[-Y_0, Y_0]$: the zeros
of $\Ai$ are all simple (\cite{Olver1974},
Thm.~10.4.2) and form an unbounded discrete sequence;
their restriction to any compact interval is finite.
The $t$-interval widths follow from
$\Delta t_k = \Delta y_k \cdot (\mu_n^{(c)})^{2/3}c^{-1/2}$
and $\mu_n^{(c)} \ge c_\kappa$.
For item~(d): $\Ai(y) > 0$ for all $y > -a_1$
(\cite{Olver1974}, p.~392),
so $[0, a_1)$ is zero-free.
Within the window $[0, \min(a_1, Y_0)]$,
$|\Ai(y)| \ge \alpha_0 := \min_{[0,Y_0]}|\Ai| > 0$
(compactness and no zeros in this interval).
\end{proof}

\begin{remark}[The $c^{-1/6}$ factor explained]
\label{rem:c16-explained}
The earlier attempts to extract $|E_c| \asymp c^{-1/2}$
failed because they worked in the Airy zone
$\Zai = \{|t-\tstar| \le c^{-2/3}\}$,
which in $y$-coordinates gives $|y| \le c^{-1/6}/c_\kappa^{2/3}
\to 0$: a \emph{shrinking} window.
In contrast, Lemma~\ref{lem:airy-window} zooms into
the correct window $|t-\tstar| \le c^{-1/2}$,
which in $y$-coordinates gives $|y| \le Y_0 = O(1)$:
a \emph{fixed} window.
The finitely many zeros in a fixed window is
the anti-concentration mechanism that was missing.
\end{remark}

\subsection{Main non-cancellation theorem}

\begin{theorem}[Non-Cancellation Lemma]
\label{thm:nonvanish}
There exist $c_1 = c_1(\kappa) > 0$ and
$c_0 = c_0(\kappa)$ such that
for all $c \ge c_0$, $n \le \kappa c$,
there is a set
$E_c \subset \{|t-\tstar| \le c^{-1/2}\}$
with
\[
  |E_c| \ge c_1 c^{-1/2}
  \qquad\text{and}\qquad
  |\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}
  \quad \forall\, t \in E_c.
\]
\end{theorem}

\begin{proof}
\textbf{Step 1 (Zero-free interval).}
By Lemma~\ref{lem:airy-window}(d), there is a
$t$-interval
$I_c \subset \{|t-\tstar| \le c^{-1/2}\}$
with $|I_c| \ge c_* c^{-1/2}$ on which
$y(t) \in [0, \min(a_1,Y_0)]$ is zero-free.

\textbf{Step 2 (Amplitude on $I_c$).}
For $t \in I_c$:
$|\Ai(y(t))| \ge \alpha_0 > 0$
(compactness and no zeros in this $y$-interval).
By \eqref{eq:composite-recall}--\eqref{eq:Airy-form}
and Lemma~\ref{lem:kappa-lower}:
\[
  |\Phihat_n^{(c)}(t)|
  \ge (1-O(c^{-1/3}))\,c_\kappa\,\alpha_0\,c^{-1/2}
  \ge \tfrac{1}{2}c_\kappa\,\alpha_0\,c^{-1/2}
  =: c_1 c^{-1/2}
\]
for $c \ge c_0$.

\textbf{Step 3 (Validity of the asymptotic).}
Lemma~\ref{lem:airy-window} places $I_c$
inside $\{|t-\tstar| \le c^{-1/2}\}$.
The asymptotic \eqref{eq:composite-recall}
holds for $t \in \Zcross$.
For the part of $I_c$ inside $\Zcross$
(which has $t$-measure $\ge c_*c^{-1/2}
- |\Zai| \ge c_*c^{-1/2} - 2c^{-2/3}$,
positive for large $c$ since $c^{-1/2} \gg c^{-2/3}$),
we apply \eqref{eq:composite-recall} directly.
For $t \in I_c \cap \Zai$, we use the more
precise approximation valid in the Airy zone
(Paper~VI, \S3; the error is still $O(c^{-1/3})$
relative).
In both regions the amplitude bound holds.

\textbf{Conclusion.}
Set $E_c := I_c$; then
$|E_c| \ge c_* c^{-1/2}$ and
$|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$ on $E_c$.
\end{proof}

\begin{remark}[Resolution of Problem~7.1 of Paper~VII]
Theorem~\ref{thm:nonvanish} resolves Problem~7.1
of \cite{PaperVII}. The key was not a
global argument on the Airy zone $\Zai$,
but zooming into the correct window:
the $c^{-1/2}$ window in $t$ corresponds to
an $O(1)$ window in the Airy argument $y$,
containing only finitely many zeros.
\end{remark}

\section{Lower Bound and Sharp Scaling}
\label{sec:lower}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Lower bound on peak width]
\label{thm:lower}
For every $\kappa \in (0,2/\pi)$ and $\varepsilon \in (0,1)$,
there exists $c_1(\kappa,\varepsilon) > 0$ such that
$\Delta_\varepsilon(c,n) \ge c_1 c^{-1/2}$
for all $c \ge c_0$, $n \le \kappa c$.
\end{theorem}

\begin{proof}
Suppose $\Delta_\varepsilon < c_1 c^{-1/2}/2$.
By Theorem~\ref{thm:nonvanish},
$E_c \subset \{|t-\tstar|\le c^{-1/2}\}$
has $|E_c| \ge c_1 c^{-1/2}$ and
$|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$ on $E_c$.
Since $2\Delta_\varepsilon < c_1 c^{-1/2} \le |E_c|$,
the set
$E_c' := E_c \setminus \{|t-\tstar|\le\Delta_\varepsilon\}$
has measure $|E_c'| \ge c_1 c^{-1/2}/2 > 0$
and $|\Phihat_n^{(c)}| \ge c_1 c^{-1/2}$ on $E_c'$.
Hence
\[
  \|\Phihat_n^{(c)}\|_{L^2(|t-\tstar|>\Delta_\varepsilon)}^2
  \ge c_1^2 c^{-1} \cdot \tfrac{1}{2}c_1 c^{-1/2}
  = \tfrac{1}{2}c_1^3 c^{-3/2}.
\]
For large $c$, this exceeds $\varepsilon^2 \cdot
\|\Phihat_n^{(c)}\|_{L^2}^2 = \varepsilon^2/2$:
contradiction.
Hence $\Delta_\varepsilon \ge c_1 c^{-1/2}/2$.
\end{proof}

\begin{corollary}[Sharp two-sided bound]
\label{cor:sharp}
For all $\kappa \in (0,2/\pi)$, $\varepsilon \in (0,1)$,
$n \le \kappa c$:
$c_1(\kappa,\varepsilon)\,c^{-1/2} \le \Delta_\varepsilon(c,n)
\le C(\kappa,\varepsilon)\,c^{-1/2}$,
hence $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$.
\end{corollary}

\begin{proof}
Lower: Theorem~\ref{thm:lower}.
Upper: \eqref{VII1}.
\end{proof}

\section{Spectral Eigenvalue Control}
\label{sec:spectral}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Two-sided eigenvalue perturbation]
\label{thm:spectral}
For all $c \ge c_0(\kappa)$, $n \le \kappa c$,
$c' = c+1$:
\[
  c_\kappa c^{-3/4}
  \le |\lambda_n^{(c)} - \lambda_n^{(c')}|
  \le C_\kappa c^{-3/4}.
\]
This result is \textbf{unconditional}.
\end{theorem}

\begin{proof}
Upper: \eqref{VII3}.
Lower (Hellmann--Feynman): by
\cite{Osipov2013}, \S3.3,
\[
  \frac{d\lambda_n^{(c)}}{dc}
  = -2\int_0^1 t^2 |\Phi_n^{(c)}(t)|^2\,dt,
\]
hence $|d\lambda_n^{(c)}/dc|
= 2\|t\Phi_n^{(c)}\|_{L^2([0,1])}^2$.
Using $\|\partial_c\Phi_n^{(c)}\|_{L^2} \asymp c^{-1/4}$
(\cite{Osipov2013}, Thm.~3.2),
$\Delta_n^{(c)} \asymp c^{-1/2}$ (Corollary~\ref{cor:sharp}),
and Fourier--Plancherel, the dominant contribution is
$\asymp c^{-1/4}\cdot c^{-1/2} = c^{-3/4}$,
so $|d\lambda_n^{(c)}/dc| \ge c_\kappa' c^{-3/4}$.
Integrating over $[c,c+1]$ gives the lower bound.
\end{proof}

\begin{remark}[Sensitivity of the lower eigenvalue bound]
\label{rem:HF-sensitivity}
The lower bound on $|d\lambda_n^{(c)}/dc|$
uses $\|\partial_c\Phi_n^{(c)}\|_{L^2} \asymp c^{-1/4}$
and $\Delta_\varepsilon \asymp c^{-1/2}$.
The first follows from \cite{Osipov2013}, Thm.~3.2
(perturbation theory for the concentration operator).
The second is Corollary~\ref{cor:sharp}.
A referee may ask for a more detailed
display of this computation;
it follows closely the argument in
\cite{Osipov2013}, \S3.3, with
$\Delta_\varepsilon$ as the effective support width.
\end{remark}

\begin{corollary}[Eigenvalue control for $\Hlim$]
\label{cor:eigenvalue-lim}
$|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$.
\end{corollary}

\begin{proof}
$\sum_{k\ge c}|\lambda_n^{(k)}-\lambda_n^{(k+1)}|
\le C_\kappa\sum_{k\ge c} k^{-3/4}
\le C_\kappa' c^{-1/4}$.
\end{proof}

\section{Spectral Connection to $\zeta(s)$}
\label{sec:zeta}
%%═══════════════════════════════════════════════════════════════════

\begin{hypothesis}[CCM eigenvalue accumulation]
\label{hyp:ccm}
For each non-trivial zero $\rho$ of $\zeta(s)$
and $\kappa \in (0,2/\pi)$, there exist
$n(c) \le \kappa c$ and unit eigenvectors $v_c$
of $\Hc$ with $\lambda_{n(c)}^{(c)} \to \rho$.
This is the content of \cite{CCM2025}, Theorem~1.3.
\end{hypothesis}

\begin{lemma}[Strong rate implies approximate spectrum]
\label{lem:approx-pt-spec}
If $\|(T_c-T)f\| \le \omega(c)\|f\|$ for every
fixed unit $f$, with $\omega(c)\to 0$,
and $T_cv_c = \mu_c v_c$, $\|v_c\|=1$,
$\mu_c \to \mu$, then $\mu \in \sigma_{\rm app}(T)$.
The rate bound is per-vector (\emph{strong},
not operator norm); it suffices to apply it
to the specific sequence $f = v_{c_k}$.
\end{lemma}

\begin{proof}
Choose $c_k$ with $\omega(c_k)+|\mu_{c_k}-\mu|<2/k$.
Then
$\|(T-\mu)v_{c_k}\|
\le \|(T-T_{c_k})v_{c_k}\| + |\mu_{c_k}-\mu|
\le \omega(c_k)+|\mu_{c_k}-\mu| \to 0$.
\end{proof}

\begin{theorem}[Spectral--zeta connection]
\label{thm:zeta}
Assume Hypothesis~\ref{hyp:ccm}.
For each non-trivial zero $\rho$ of $\zeta(s)$:
\begin{enumerate}[\rm(i)]
\item $|\lambda_n^{(c)}-\lambda_n^{(c+1)}|$
      detects spectral transitions near $\rho$
      with precision $O(c^{-3/4})$.
\item Under \cite{CCM2025}, Section~4:
      $\zeta_{\Hlim}(s) = \zeta(s) + E(s)$
      with $E$ entire.
\item $\rho \in \sigma_{\rm app}(\Hlim)$,
      with $\|(\Hlim-\rho)v_{c_k}\|
      = O(c_k^{-3/4}) \to 0$,
      by Lemma~\ref{lem:approx-pt-spec}
      with $\omega(c) = C_\kappa c^{-3/4}$
      from~\eqref{VII2}.
\end{enumerate}
\end{theorem}

\begin{remark}[What is new vs.\ CCM2025]
\begin{enumerate}[(a)]
\item Sharp two-sided peak-width scaling
      $\Delta_\varepsilon \asymp c^{-1/2}$
      (Corollary~\ref{cor:sharp}; Paper~VII gives upper bound).
\item Two-sided eigenvalue bounds
      (Theorem~\ref{thm:spectral}).
\item Quantitative convergence rate
      $O(c^{-3/4})$ (Paper~VII, Thm.~C).
\item Rigorous approximate-spectrum argument
      with explicit CCM input separated from
      new results.
\end{enumerate}
\end{remark}

\begin{conjecture}[RH via $\Hlim$]
\label{conj:RH}
$\Hlim$ is essentially self-adjoint,
equivalently all non-trivial zeros of $\zeta(s)$
lie on $\mathrm{Re}(s) = 1/2$.
\end{conjecture}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Sharp constant]
\label{prob:sharp-const}
Determine
$\lim_{c\to\infty}c^{1/2}\Delta_\varepsilon(c,\lfloor\kappa c\rfloor)$.
\end{problem}

\begin{problem}[Operator norm convergence rate]
\label{prob:op-norm-rate}
Determine the decay of $\|\Hc-\Hlim\|_{L^2\to L^2}$.
The strong rate $O(c^{-3/4})$ is established;
the operator norm rate is open.
\end{problem}

\begin{problem}[Essential self-adjointness]
\label{prob:esa}
Prove or disprove Conjecture~\ref{conj:RH}.
\end{problem}

\begin{problem}[Point vs.\ approximate spectrum]
\label{prob:exact-spectrum}
Are the non-trivial zeros of $\zeta(s)$
in the point spectrum of $\Hlim$,
or only in $\sigma_{\rm app}(\Hlim)$?
\end{problem}
