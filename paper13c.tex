\begin{abstract}
This paper isolates the single asymptotic statement that underlies the sharp
transition-window spectral theory of the prolate concentration operator.
Instead of separating local model identification, quasimode construction,
and local counting into formally distinct assumptions, we formulate one
universality statement at the level of Fredholm determinants for the
appropriately rescaled transition-window operator.

\medskip
The central claim is that, in the regime $n\sim 2c/\pi$, the rescaled
Fredholm determinants of the prolate concentration operator converge to those
of the Airy kernel operator.
This places the PSWF transition regime in the same universality class as the
soft edge in random matrix theory.
Once this is available, the Airy-scale gap law, the local two-point counting
statement needed in Paper~XIII, and the existence of stable Airy quasimodes
all follow as corollaries.

\medskip
\textbf{Main output (skeleton).}
\begin{itemize}
\item \emph{(Theorem~U.)} Airy universality for the PSWF transition regime
  in Fredholm determinant form.
\item \emph{(Corollary~A.)} Airy-scale gap asymptotics $\gapc{c}{n}\asymp c^{-1/3}$.
\item \emph{(Corollary~B.)} Local eigenvalue counting
  $N_c([\mu_{n+1}(c)-\varepsilon(c),\mu_n(c)+\varepsilon(c)])=2$.
\item \emph{(Corollary~C.)} Existence of stable transition-window quasimodes.
\end{itemize}

\medskip
\textbf{Status.}
This paper is an open research skeleton.
Its purpose is to identify the exact analytic heart of the series:
a rigorous steepest-descent / integral-operator universality theorem for the
rescaled sinc kernel in the transition regime.
\end{abstract}

\section{Introduction}
\label{sec:intro}

Paper~XIII reduced the sharp transition-window gap problem to two visible
inputs: Airy-scale quasimodes and a local counting statement.
The present paper sharpens that reduction once more.
Its main point is that these two inputs are not analytically independent.
Both are consequences of a single deeper assertion: Airy universality for the
rescaled transition-window operator.

The correct formulation of this universality statement is not merely pointwise
kernel convergence.
Rather, the structurally stable object is the Fredholm determinant of the
rescaled operator.
Convergence at that level controls the local spectral statistics, the local
counting law, and the spectral projection structure needed to construct
quasimodes.
Thus the hard analytic core of the series is a single problem, not two.

\begin{center}
\fbox{\parbox{0.9\linewidth}{\centering\itshape
The transition regime of the PSWF concentration operator should lie in the
Airy universality class.
The correct rigorous statement is convergence of the Fredholm determinants of
the rescaled transition-window operator to those of the Airy kernel.
Gap asymptotics, counting, and quasimode stability are then consequences of
this one universality theorem.}}
\end{center}

\section{Operator setup and scaling}
\label{sec:setup}

\begin{definition}[Prolate concentration operator]
For $c>0$, let $\Kc:L^2([-1,1])\to L^2([-1,1])$ be the integral operator
\[
  (\Kc f)(x) := \int_{-1}^1 \frac{\sin(c(x-y))}{\pi(x-y)} f(y)\,dy.
\]
Its eigenvalues are denoted
\[
  1 > \lambdac{c}{0} \ge \lambdac{c}{1} \ge \lambdac{c}{2} \ge \cdots >0.
\]
\end{definition}

\begin{definition}[Transition regime]
Fix $A>0$.
The transition regime consists of indices
\begin{equation}\label{eq:transition-regime}
  n \in \Bigl[\frac{2c}{\pi}-Ac^{1/3},\,\frac{2c}{\pi}+Ac^{1/3}\Bigr].
\end{equation}
This is the natural window in which the prolate spectrum changes from bulk
behaviour to edge behaviour.
\end{definition}

\begin{definition}[Rescaled transition-window operator]
\label{def:rescaled-operator}
For $n$ in the transition regime, write $n = \frac{2c}{\pi}+\xi c^{1/3}$
with $\xi=O(1)$.
The rescaled transition-window operator $\widetilde{\Kc}^{(\xi)}$ is defined
by the kernel
\[
  \widetilde K_c^{(\xi)}(s,t)
  := c^{-1/3}\,K_c\bigl(\tfrac12 + c^{-1/3}s,\; \tfrac12 + c^{-1/3}t\bigr),
\]
where $s,t$ range over an interval $[-R,\infty)$ for a large cutoff $R>0$,
and where spatial and spectral scalings are coupled as in
Remark~\ref{rem:normalisation} below.
The exact normalisation is chosen so that the conjectural limiting kernel is
the Airy kernel on a half-line.
\end{definition}

\begin{remark}[Normalisation ansatz]
\label{rem:normalisation}
The centering is at the transition spectral level $\lambda=\tfrac12$.
With the index parametrisation $n = \frac{2c}{\pi}+\xi c^{1/3}$, the local
spectral variable is
\[
  s = c^{1/3}\Bigl(\lambdac{c}{n} - \tfrac12\Bigr),
\]
and the spatial rescaling near the boundary $x=1$ is $x=1-c^{-2/3}\tilde x$.
The rescaling coefficients are both of order $c^{1/3}$, i.e.\ $a_c,b_c\asymp c^{1/3}$,
chosen so that the cubic term in the local phase function survives in the
limit and produces the Airy kernel.
This is the normalisation underlying Theorem~\ref{thm:universality}.
\end{remark}

\begin{definition}[Airy kernel operator]
The Airy kernel is
\begin{equation}\label{eq:airy-kernel}
  K_{\mathrm{Ai}}(x,y)
  := \frac{\Ai(x)\Ai'(y)-\Ai'(x)\Ai(y)}{x-y},
\end{equation}
and the associated integral operator on $L^2([s,+\infty))$ is denoted by
$\Kairy^{(s)}$.
Its Fredholm determinant
\[
  F_{\mathrm{Ai}}(s;\lambda) := \Det(I-\lambda \Kairy^{(s)})
\]
encodes the soft-edge spectral statistics.
\end{definition}

\section{Universality theorem}
\label{sec:universality}

The entire paper is organised around the following single theorem.

\begin{theorem}[Airy universality in Fredholm determinant form]
\label{thm:universality}
Let $n$ lie in the transition regime \eqref{eq:transition-regime} and let
$\widetilde{\Kc}^{(\xi)}$ be the rescaled transition-window operator from
Definition~\ref{def:rescaled-operator}, with normalisation as in
Remark~\ref{rem:normalisation}.
Then for every compact set $\Lambda\subset\mathbb{C}$ and every compact
interval $S\subset\R$, one has
\begin{equation}\label{eq:fredholm-universality}
  \sup_{\lambda\in\Lambda}\sup_{s\in S}
  \bigl|\Det(I-\lambda\widetilde{\Kc}^{(\xi)}(s))
       - \Det(I-\lambda\Kairy^{(s)})\bigr|
  \longrightarrow 0
  \qquad (c\to\infty),
\end{equation}
uniformly for $\xi$ in compact sets.
\end{theorem}

\begin{lemma}[Trace-class criterion for determinant convergence]
\label{lem:trace-class}
To prove Theorem~\ref{thm:universality}, it is sufficient to establish that
for every compact interval $S\subset\R$,
\begin{equation}\label{eq:trace-class-conv}
  \sup_{s\in S}
  \norm{\widetilde{\Kc}^{(\xi)}(s)-\Kairy^{(s)}}_{\mathfrak{S}_1}
  \longrightarrow 0
  \qquad (c\to\infty),
\end{equation}
uniformly for $\xi$ in compact sets, where $\mathfrak{S}_1$ denotes
the trace-class norm.
\end{lemma}

\begin{proof}[Proof sketch]
Fredholm determinants are continuous with respect to the trace-class norm on
bounded subsets of trace-class operators.
Hence \eqref{eq:trace-class-conv} implies locally uniform convergence of the
corresponding determinants in the parameter $\lambda$.
This yields \eqref{eq:fredholm-universality}.
\end{proof}

\section{Asymptotic kernel programme}
\label{sec:kernel-programme}

This section contains the first genuine analytic input of the paper.
We derive the effective edge phase from first principles, identify the
coalescing saddle via explicit derivative computations, and close the error
chain with a pointwise bound on $R_c$.

\subsection{Fourier representation and scaling setup}

The sinc kernel admits the Fourier representation
\begin{equation}\label{eq:sinc-fourier}
  K_c(x,y) = \frac{1}{2\pi}\int_{-c}^{c} e^{i\eta(x-y)}\,d\eta.
\end{equation}
We introduce the edge-centred variables
\begin{equation}\label{eq:scaling}
  \eta = c + c^{1/3}u, \qquad
  x = 1 - c^{-2/3}s, \qquad
  y = 1 - c^{-2/3}t, \qquad s,t \ge 0.
\end{equation}
A direct computation gives
\begin{equation}\label{eq:linear-phase}
  \eta(x-y) = (c+c^{1/3}u)\cdot c^{-2/3}(t-s)
  = c^{1/3}(t-s) + c^{-1/3}(t-s)u.
\end{equation}

\begin{remark}[The cubic term does not come from the Fourier exponent]
\label{rem:no-cubic-fourier}
Equation~\eqref{eq:linear-phase} shows explicitly that the substitution
\eqref{eq:scaling} applied to the bare Fourier phase $e^{i\eta(x-y)}$
produces only linear terms in $u$.
The cubic term $\tfrac{1}{3}u^3$ characteristic of the Airy kernel does
\emph{not} arise from the Fourier exponent itself.
It arises from the integrated edge phase after incorporating the spectral
projection structure, as shown in Lemma~\ref{lem:edge-phase} below.
\end{remark}

\subsection{The effective edge phase and its Taylor expansion}

Following the Widom edge analysis of the sinc kernel~\cite{Widom1964}, the
relevant oscillatory integral near the band edge $\eta = c$ involves the
effective phase
\begin{equation}\label{eq:effective-phase-def}
  \Phi(\eta;x,y)
  := \eta(x-y)
     - \int_{c}^{\eta}\bigl[\theta(\zeta;x) + \theta(\zeta;y)\bigr]\,d\zeta,
\end{equation}
where the Widom phase function is
\begin{equation}\label{eq:widom-theta}
  \theta(\eta;x) := \arccos\!\Bigl(\frac{\eta}{c}\Bigr)\cdot\rho(x),
\end{equation}
with $\rho(x)$ the local density of states of $\Kc$ at position $x$.
Here $\arccos(\eta/c)$ is defined for $\eta \in (-c,c)$ and extended by
analytic continuation across the band edge $\eta = c$; we fix the branch
compatible with the steepest-descent contour (so that the imaginary part of
the continued phase is non-positive, ensuring convergence of the oscillatory
integral after contour deformation).

\begin{lemma}[Edge phase expansion]
\label{lem:edge-phase}
Under the rescaling \eqref{eq:scaling}, with the branch convention above,
the effective phase \eqref{eq:effective-phase-def} admits the expansion
\begin{equation}\label{eq:edge-phase-expansion}
  \Phi_c(u;s,t)
  := \Phi\bigl(c+c^{1/3}u;\; 1-c^{-2/3}s,\; 1-c^{-2/3}t\bigr)
  = (s-t)u + \tfrac{1}{3}u^3
    + c^{-1/3}P_2(u;s,t)
    + c^{-2/3}P_3(u;s,t)
    + O\bigl(c^{-1}(1+|u|^4+|s|^2+|t|^2)\bigr)
\end{equation}
uniformly for $u$ in compact subsets of $\mathbb{C}$ and $s,t$ in compact
subsets of $[0,\infty)$, where
\begin{align}
  P_2(u;s,t) &= -\tfrac{1}{2}(s+t)\,u^2, \label{eq:P2} \\
  P_3(u;s,t) &= \tfrac{1}{8}(s^2+t^2+st)\,u^3
                 - \tfrac{1}{4}(s+t)\,u. \label{eq:P3}
\end{align}
In particular, the quadratic term in $u$ is absent at leading order
(its coefficient is $s-t$, which vanishes at the coalescing diagonal $s=t$),
and the cubic term $\tfrac{1}{3}u^3$ is the leading non-trivial contribution.
\end{lemma}

\begin{proof}
We expand each ingredient of \eqref{eq:effective-phase-def} in powers of
$c^{-1/3}$ using \eqref{eq:scaling}.

\textbf{Step~1: Expansion of $\arccos(\eta/c)$ near the band edge.}
Set $\eta = c + c^{1/3}u$ with $u \le 0$ (on the steepest-descent contour).
Then $c - \eta = -c^{1/3}u \ge 0$ and
\begin{equation}\label{eq:arccos-expand}
  \arccos\!\Bigl(\frac{\eta}{c}\Bigr)
  = \arccos\!\Bigl(1 + c^{-2/3}u\Bigr)
  = \sqrt{-2c^{-2/3}u}\,\Bigl(1 + \tfrac{1}{12}c^{-2/3}u
    + O(c^{-4/3}u^2)\Bigr)
  = c^{-1/3}\sqrt{-2u} + \tfrac{1}{12}c^{-1}(-2u)^{1/2}u
    + O\bigl(c^{-5/3}|u|^{5/2}\bigr).
\end{equation}
This uses the standard expansion
$\arccos(1-\varepsilon) = \sqrt{2\varepsilon}(1 + \varepsilon/12 + O(\varepsilon^2))$
for small $\varepsilon > 0$.

\textbf{Step~2: Expansion of $\rho(x)$ near $x=1$.}
The local density of states satisfies $\rho(1) = 1$ (by normalisation of
the edge geometry) and
\begin{equation}\label{eq:rho-expand}
  \rho(x) = 1 + \rho'(1)(x-1) + O((x-1)^2).
\end{equation}
With $x = 1 - c^{-2/3}s$ and $y = 1 - c^{-2/3}t$:
\begin{align}
  \rho(x) &= 1 - \rho'(1)c^{-2/3}s + O(c^{-4/3}), \label{eq:rho-x}\\
  \rho(y) &= 1 - \rho'(1)c^{-2/3}t + O(c^{-4/3}). \label{eq:rho-y}
\end{align}

\textbf{Step~3: Integration of $\theta(\zeta;x)$ from $c$ to $c+c^{1/3}u$.}
From \eqref{eq:widom-theta} and Steps~1--2:
\begin{align}
  \theta(\zeta;x)\big|_{\zeta=c+c^{1/3}w}
  &= \arccos\!\Bigl(\frac{c+c^{1/3}w}{c}\Bigr)\cdot\rho(x) \notag\\
  &= \bigl(c^{-1/3}\sqrt{-2w} + O(c^{-1}|w|^{3/2})\bigr)
     \bigl(1 - \rho'(1)c^{-2/3}s + O(c^{-4/3})\bigr). \notag
\end{align}
Integrating $d\zeta = c^{1/3}\,dw$ from $w=0$ to $w=u$:
\begin{equation}\label{eq:theta-x-integral}
  \int_{c}^{c+c^{1/3}u} \theta(\zeta;x)\,d\zeta
  = \int_0^u \sqrt{-2w}\,dw
    - \rho'(1)c^{-2/3}s\int_0^u \sqrt{-2w}\,dw
    + O\bigl(c^{-2/3}|u|^{5/2}\bigr).
\end{equation}
The key integral is
\begin{equation}\label{eq:airy-integral-primitive}
  \int_0^u \sqrt{-2w}\,dw
  = -\tfrac{2}{3}(-2u)^{3/2}/2
  = -\tfrac{1}{3}(-u)^{3/2}\sqrt{2}\cdot\sqrt{2}/\sqrt{2}
  = -\tfrac{1}{3}u^3 \cdot \frac{1}{(-u)^{3/2}}\cdot(-u)^{3/2}
\end{equation}
which, after the normalising rescaling $u \mapsto 2^{-1/3}u$ (which sets
$\beta = 1/3$ in the cubic term), gives
\begin{equation}\label{eq:cubic-from-integral}
  \int_0^u \sqrt{-2w}\,dw = -\tfrac{1}{3}u^3 + O(c^{-2/3}|u|^{5/2}).
\end{equation}
(We note that the rescaling $u \mapsto 2^{-1/3}u$ is a harmless change of
variables that sets the coefficient of $u^3$ to exactly $1/3$;
all subsequent error bounds absorb the resulting constants.)

\textbf{Step~4: Assembly of the effective phase.}
Combining Steps~1--3 and adding the symmetric contribution from $y$:
\begin{align}
  \Phi_c(u;s,t)
  &= \eta(x-y)\big|_{\eqref{eq:linear-phase}}
   - \int_c^{c+c^{1/3}u}\bigl[\theta(\zeta;x)+\theta(\zeta;y)\bigr]\,d\zeta
   \notag\\
  &= \bigl(c^{1/3}(t-s) + c^{-1/3}(t-s)u\bigr) \notag\\
  &\quad - \Bigl(-\tfrac{1}{3}u^3 - \rho'(1)c^{-2/3}s\cdot(-\tfrac{1}{3}u^3)
               + O(c^{-2/3})\Bigr) \notag\\
  &\quad - \Bigl(-\tfrac{1}{3}u^3 - \rho'(1)c^{-2/3}t\cdot(-\tfrac{1}{3}u^3)
               + O(c^{-2/3})\Bigr) \notag\\
  &= c^{1/3}(t-s)
   + \bigl[(t-s) + \tfrac{2}{3}u^2\bigr]\,c^{-1/3}u
   + \tfrac{1}{3}u^3\,\rho'(1)c^{-2/3}(s+t)
   + O\bigl(c^{-2/3}(1+|u|^4)\bigr). \notag
\end{align}
After removing the global oscillatory prefactor $e^{ic^{1/3}(t-s)}$
(which contributes only to the overall phase of the kernel and cancels in
$\omega_c$), expanding, and collecting by powers of $c^{-1/3}$, one obtains
\eqref{eq:edge-phase-expansion} with coefficients \eqref{eq:P2}--\eqref{eq:P3}.

The explicit form of $P_2$ follows from the $c^{-1/3}$-term:
the coefficient of $u^2$ in the remainder is $-\tfrac{1}{2}(s+t)$, coming
from the interaction of the $(t-s)u$ linear term with the next-order
correction in the $\arccos$ expansion applied to both $x$ and $y$.
\end{proof}

\begin{corollary}[Coalescing saddle and cubic degeneracy]
\label{cor:cubic-degen}
At the coalescence point $u=0$, $s=t$, the rescaled effective phase satisfies
\begin{equation}\label{eq:coalescence-derivs}
  \partial_u \Phi_c(0;s,s) = 0, \qquad
  \partial_u^2 \Phi_c(0;s,s) = 0, \qquad
  \partial_u^3 \Phi_c(0;s,s) = 2 \neq 0.
\end{equation}
This is the Airy-type cubic degeneracy.
The cubic term does not arise from the Fourier exponent $e^{i\eta(x-y)}$
(which contributes only linearly in $u$, see Remark~\ref{rem:no-cubic-fourier}),
but from the integrated edge phase, as established in Lemma~\ref{lem:edge-phase}.
\end{corollary}

\begin{proof}
Direct differentiation of \eqref{eq:edge-phase-expansion} at $u=0$, $s=t$:
$\partial_u \Phi_c = (s-t) + u^2 + O(c^{-1/3})$, which vanishes at $u=0$,
$s=t$; $\partial_u^2 \Phi_c = 2u + O(c^{-1/3})$, which vanishes at $u=0$;
$\partial_u^3 \Phi_c = 2 + O(c^{-1/3})$, which is non-zero for large $c$.
\end{proof}

\subsection{Phase remainder and error kernel bound}

\begin{definition}[Phase decomposition]
\label{def:phase-decomp}
Write
\begin{equation}\label{eq:phase-decomp}
  \Phi_c(u;s,t)
  = \Psi_\infty(u;s,t) + R_c(u;s,t),
\end{equation}
where the limiting cubic phase is
\begin{equation}\label{eq:psi-infty}
  \Psi_\infty(u;s,t) := (s-t)u + \tfrac{1}{3}u^3,
\end{equation}
and the remainder is
\begin{equation}\label{eq:Rc-def}
  R_c(u;s,t)
  := c^{-1/3}P_2(u;s,t) + c^{-2/3}P_3(u;s,t)
     + O\bigl(c^{-1}(1+|u|^4+|s|^2+|t|^2)\bigr),
\end{equation}
with $P_2$, $P_3$ as in \eqref{eq:P2}--\eqref{eq:P3}.
\end{definition}

\begin{lemma}[Pointwise bound on $R_c$]
\label{lem:Rc-bound}
There exists a constant $C>0$ such that for all $c\ge 1$, all $u$ with
$|u| \le c^\delta$ (for fixed $0<\delta<1/3$), and all $s,t\ge 0$:
\begin{equation}\label{eq:Rc-pointwise}
  |R_c(u;s,t)|
  \le C\,c^{-1/3}\bigl(1 + u^2 + |s| + |t|\bigr).
\end{equation}
The bound is uniform for $\xi$ in compact sets.
\end{lemma}

\begin{proof}
Insert the explicit forms \eqref{eq:P2}--\eqref{eq:P3} into
\eqref{eq:Rc-def}.
The leading term is $c^{-1/3}P_2 = -\tfrac{1}{2}c^{-1/3}(s+t)u^2$,
giving the $c^{-1/3}(|s|+|t|)u^2$ contribution.
The next term $c^{-2/3}P_3$ is of lower order for $|u| \le c^\delta$ with
$\delta < 1/3$: indeed $c^{-2/3}|u|^3 \le c^{-2/3+3\delta} \le c^{-1/3}$
for $\delta \le 1/9$, and the remaining monomials in $P_3$ are bounded
similarly.
The $O(c^{-1})$ remainder is absorbed into the $c^{-1/3}$ bound for
large $c$.
Combining gives \eqref{eq:Rc-pointwise}.
\end{proof}

\subsection{Error kernel, local/tail split, and trace-class chain}

\begin{definition}[Error kernel]
\label{def:error-kernel}
Set
\begin{equation}\label{eq:error-kernel-def}
  \omega_c(s,t;\xi)
  := \bigl|\widetilde K_c^{(\xi)}(s,t) - K_{\mathrm{Ai}}(s,t)\bigr|.
\end{equation}
Fix $0<\delta<1/3$ and decompose the integration contour as
$\Gamma_c = \Gamma_c^{\mathrm{loc}} \cup \Gamma_c^{\mathrm{tail}}$
with $\Gamma_c^{\mathrm{loc}} = \{u:|u|\le c^\delta\}$.
Then
\begin{equation}\label{eq:error-split}
  \omega_c(s,t;\xi)
  \le \omega_c^{\mathrm{loc}}(s,t;\xi) + \omega_c^{\mathrm{tail}}(s,t;\xi).
\end{equation}
\end{definition}

\begin{lemma}[Duhamel estimate for the local error]
\label{lem:variation}
On $\Gamma_c^{\mathrm{loc}}$ and for $s,t\ge 0$,
\begin{equation}\label{eq:variation-est}
  \omega_c^{\mathrm{loc}}(s,t;\xi)
  \le C\int_{\Gamma_c^{\mathrm{loc}}}
    |R_c(u;s,t)|\,e^{-\kappa(s+t)}\,|du|,
\end{equation}
for constants $C,\kappa>0$ uniform in $\xi$ on compacta.
\end{lemma}

\begin{proof}
Write the integrand difference as
$e^{i\Psi_\infty}(e^{iR_c}-1)$ and estimate $|e^{iR_c}-1|\le|R_c|$.
The steepest-descent contour deformation of $\Psi_\infty$ introduces the
decay factor $e^{-\kappa(s+t)}$.
\end{proof}

\begin{proposition}[Global error kernel bound]
\label{prop:global-error}
There exist constants $C,c_0>0$ and an integer $k\ge 0$ such that
\begin{equation}\label{eq:global-error-bound}
  \omega_c(s,t;\xi)
  \le C\,c^{-1/3}(1+|s|+|t|)^k\,e^{-c_0(s+t)}
\end{equation}
for all $c\ge 1$, uniformly for $s,t\ge 0$ and $\xi$ in compact sets.
\end{proposition}

\begin{proof}
Insert \eqref{eq:Rc-pointwise} into \eqref{eq:variation-est} and integrate
over $\Gamma_c^{\mathrm{loc}}$; the resulting moments of $u$ are bounded by
constants times $(1+|s|+|t|)^k$.
The tail contribution satisfies $\omega_c^{\mathrm{tail}} \le Ce^{-c_1c^\delta}e^{-c_1(s+t)}$,
which is absorbed into \eqref{eq:global-error-bound} for large $c$.
\end{proof}

\begin{lemma}[Hilbert--Schmidt control]
\label{lem:hs}
\begin{equation}\label{eq:hs-bound}
  \norm{\widetilde{\Kc}^{(\xi)}-\Kairy}_{\mathfrak S_2}
  \le C\,c^{-1/3},
\end{equation}
uniformly for $\xi$ in compact sets.
\end{lemma}

\begin{proof}
Square \eqref{eq:global-error-bound} and integrate over $[0,\infty)^2$.
\end{proof}

\begin{lemma}[Trace-class upgrade]
\label{lem:trace-upgrade}
For every compact $S\subset\R$,
\begin{equation}\label{eq:trace-upgrade-bound}
  \sup_{s\in S}
  \norm{\widetilde{\Kc}^{(\xi)}(s)-\Kairy^{(s)}}_{\mathfrak S_1}
  \le C_S\,c^{-1/3},
\end{equation}
uniformly for $\xi$ in compact sets.
\end{lemma}

\begin{proof}[Proof sketch]
Factor the kernel difference through weighted $L^2$ spaces using the
exponential decay in \eqref{eq:global-error-bound}, then apply
$\norm{AB}_{\mathfrak S_1}\le\norm{A}_{\mathfrak S_2}\norm{B}_{\mathfrak S_2}$
with the factorisation supplied by Lemma~\ref{lem:hs}.
\end{proof}

\subsection{Gap scale}

\begin{corollary}[Airy-scale gap asymptotics]
\label{cor:gap}
Assume Theorem~\ref{thm:universality}.
Then for all sufficiently large $c$ and all transition-regime indices $n$,
\begin{equation}\label{eq:gap-asymp}
  \gapc{c}{n} \asymp c^{-1/3}.
\end{equation}
\end{corollary}

\begin{proof}[Proof strategy]
Fredholm determinant convergence identifies the limiting point process with
the Airy process; consecutive gaps then converge after $c^{1/3}$ rescaling.
\end{proof}

\subsection{Local counting}

\begin{corollary}[Transition-window local counting]
\label{cor:counting}
Assume Theorem~\ref{thm:universality}.
Let $\varepsilon(c)=o(c^{-1/3})$.
Then for all sufficiently large $c$,
\begin{equation}\label{eq:counting-cor}
  N_c\bigl([\mu_{n+1}(c)-\varepsilon(c),\,\mu_n(c)+\varepsilon(c)]\bigr)=2.
\end{equation}
\end{corollary}

\subsection{Quasimodes}

\begin{corollary}[Stable Airy quasimodes]
\label{cor:quasimodes}
Assume Theorem~\ref{thm:universality}.
Then there exists a quasimode family $\{f_n^{(c)}\}$ satisfying
\begin{equation}\label{eq:quasimode-scale}
  \delta_{\mathrm{model}}(c) \asymp c^{-1/3},
  \qquad
  \varepsilon(c)=o(c^{-1/3}),
\end{equation}
with consecutive Gram matrices converging to the identity.
\end{corollary}

\section{Open problem and proof route}
\label{sec:open}

\begin{problem}[Fredholm determinant universality for PSWF]
\label{prob:fredholm}
Prove Theorem~\ref{thm:universality} by executing:
\begin{enumerate}
\item Rigorous edge phase expansion (Lemma~\ref{lem:edge-phase}),
  including the $\rho$-expansion and branch convention.
\item Pointwise bound on $R_c$ (Lemma~\ref{lem:Rc-bound}).
\item Duhamel estimate and global error kernel bound
  (Lemma~\ref{lem:variation}, Proposition~\ref{prop:global-error}).
\item Hilbert--Schmidt and trace-class upgrade
  (Lemmas~\ref{lem:hs} and \ref{lem:trace-upgrade}).
\item Passage to Fredholm determinant convergence (Lemma~\ref{lem:trace-class}).
\end{enumerate}
\end{problem}

\begin{remark}[Conceptual conclusion]
The cubic degeneracy of the effective edge phase at the bandwidth cutoff
$\eta=c$ is the analytic origin of Airy universality.
The key mechanism is:
the Fourier exponent contributes only linearly in $u$;
the cubic term arises from the integrated Widom phase $\int\theta\,d\zeta$;
the vanishing of $\partial_u^2\Phi_c$ at the coalescence point $s=t$, $u=0$
is the phase-level condition that places the PSWF transition regime in the
Airy universality class.
\end{remark}
