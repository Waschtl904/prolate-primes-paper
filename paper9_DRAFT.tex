\begin{abstract}
We continue the spectral analysis of the PSWF concentration
operators~$\Hc$ and their formal limit, Papers~V--VIII.

\textbf{Two operator objects.}
$(\mathrm{i})$~The \emph{spectral operator}~$\Hspec$, defined by a
spectral series on an explicit domain.
$(\mathrm{ii})$~The \emph{SOT-limit operator}
$\HSOT=\mathrm{SOT}\text{-}\lim_{c\to\infty}\Hc$.
Identification $\overline{\Hspec}=\HSOT$ is open
(Problem~\ref{prob:identification}); no proof uses both operators
simultaneously.

\textbf{Key architectural principle.}
Unit norm and uniqueness of $\Phi_n^{(\infty)}$ are derived
\emph{before} any limit operator is invoked.
\begin{itemize}
\item \textbf{Stage~1} (Lemma~\ref{lem:pswf-precompact}):
  PSWF bandlimit precompactness $\Rightarrow$ strong
  subsequential limits with $\|u\|=1$.
  Uses: Fr\'echet--Kolmogorov, Osipov--Rokhlin--Xiao.
  No operator.
\item \textbf{Stage~1.5} (Lemma~\ref{lem:spectral-selection}):
  Rayleigh characterisation of $\Phi_n^{(c)}$ + strict eigenvalue
  isolation in the limit $\Rightarrow$ every strong accumulation
  point is the unique maximiser of the limit Rayleigh functional.
  Uses: $\Hc$ (finite~$c$), Slepian gap, \eqref{VIII3}.
  No $\HSOT$, no $\Hspec$, no cross-$c$ operator comparison.
\item \textbf{Stage~2} (Lemma~\ref{lem:weak-eigenrelation}):
  $\HSOT$ introduced here as \emph{passive verification}.
\end{itemize}

\textbf{Main results:}
closability (two separate proofs); operator-norm rate $O(c^{-1/4})$
(conditional); deficiency orthogonality; ESSA under
Hypothesis~\ref{hyp:decay}; conditional RH reduction
(meta-corollary, not a proof of RH).
\end{abstract}

  $\|\Phi_n^{(\infty)}\|=1$ follows from strong convergence given by

  It does not select or construct.

  $\overline{\Hspec}=\HSOT$: Problem~\ref{prob:identification},
  not used anywhere.
\end{enumerate}

\subsection*{Limit eigenvalues}

\begin{definition}[Limit eigenvalues]
\label{def:lim-eigenvalues}
$\lambda_n^{(\infty)}:=\lim_{c\to\infty}\lambda_n^{(c)}$
(exists by \eqref{VIII3}).
\end{definition}

\begin{remark}[Strict eigenvalue isolation in the limit]
\label{rem:eigenvalue-isolation}
From the Slepian gap $|\lambda_n^{(c)}-\lambda_m^{(c)}|\ge
\delta_{nm}c^{-1/2}$ (\cite{Slepian1961}) and \eqref{VIII3}:
\[
  \lambda_n^{(\infty)}-\lambda_{n+1}^{(\infty)}
  = \lim_{c\to\infty}\bigl(\lambda_n^{(c)}-\lambda_{n+1}^{(c)}\bigr)
  > 0.
\]
Hence the limit eigenvalues are strictly ordered:
$\lambda_0^{(\infty)}>\lambda_1^{(\infty)}>\cdots>0$.
\end{remark}

%---------------------------------------------------------------
\subsection*{Stage~1: Precompactness (external norm anchor)}
%---------------------------------------------------------------

\begin{lemma}[PSWF bandlimit precompactness]
\label{lem:pswf-precompact}
\emph{Operators used: none.}

For fixed $n\ge 0$, $\{\Phi_n^{(c)}\}_{c\ge 1}$ is precompact
in $L^2(\mathbb{R})$.
Every subsequence $c_k\to\infty$ has a further subsequence along
which $\Phi_n^{(c_{k_j})}\to u$ strongly; any such $u$ satisfies
$\|u\|=1$.
\end{lemma}

\begin{proof}
Fr\'echet--Kolmogorov criterion (\cite{ReedSimon1975}, Thm.~IV.26).

\textit{(i) Uniform $L^2$-bound:} $\|\Phi_n^{(c)}\|=1$.

\textit{(ii) Uniform spatial decay:}
$|\Phi_n^{(c)}(x)|\le C_{n,\varepsilon}e^{-\delta x^2}$
for $|x|\ge 1+\varepsilon$, uniformly in $c$
(\cite{Osipov2013}, Thm.~3.2).

\textit{(iii) Uniform $H^1$-bound:}
$\|\Phi_n^{(c)}\|_{H^1}\le D_n$ uniformly in $c$,
fixed~$n$ (\cite{Osipov2013}, Prop.~4.1),
giving $L^2$-equicontinuity.

Fr\'echet--Kolmogorov gives precompactness.
Any strong limit $u$ satisfies $\|u\|=\lim_j\|\Phi_n^{(c_{k_j})}\|=1$.
\end{proof}

%---------------------------------------------------------------
\subsection*{Stage~1.5: Spectral selection (uniqueness, no limit
operator)}
%---------------------------------------------------------------

\begin{lemma}[Spectral selection via Rayleigh characterisation]
\label{lem:spectral-selection}
\emph{Operators used: $\Hc$ (finite~$c$ only).
$\HSOT$ and $\Hspec$ do not appear.}

Fix $n\ge 0$.
Every strong accumulation point of $\{\Phi_n^{(c)}\}_{c\to\infty}$
converges to the same unit vector $\Phi_n^{(\infty)}$
(up to a global phase convention).
\end{lemma}

\begin{proof}
Let $u$ be any strong accumulation point,
i.e.\ $\Phi_n^{(c_k)}\to u$ strongly along some $c_k\to\infty$.
By Lemma~\ref{lem:pswf-precompact}, $\|u\|=1$.

\textbf{Step~1: Rayleigh values converge to $\lambda_n^{(\infty)}$.}
For each~$c_k$:
\[
  R_n(c_k):=\ip{\Hc[c_k]\Phi_n^{(c_k)}}{\Phi_n^{(c_k)}}
  =\lambda_n^{(c_k)}\|\Phi_n^{(c_k)}\|^2
  =\lambda_n^{(c_k)}\to\lambda_n^{(\infty)}.
\]

\textbf{Step~2: Variational characterisation.}
$\Phi_n^{(c)}$ is the constrained maximiser:
\begin{equation}\label{eq:rayleigh}
  \lambda_n^{(c)}=\max_{\substack{f\perp\Phi_0^{(c)},\dots,
  \Phi_{n-1}^{(c)}\\ \|f\|=1}}\ip{\Hc f}{f},
\end{equation}
where the orthogonality constraints are to the previously
constructed limit vectors $\Phi_0^{(\infty)},\dots,
\Phi_{n-1}^{(\infty)}$ in the limit.
For $n=0$ there is no constraint.

\textbf{Step~3: Every accumulation point achieves
$\lambda_n^{(\infty)}$ as Rayleigh value.}
From Step~1 and Step~2, for any strong accumulation point~$u$:
\[
  \ip{\Hc[c_k] u}{u}
  \to\ip{\HSOT u}{u}.
\]
But we avoid using $\HSOT$ here.
Instead: from \eqref{VIII3},
$|\ip{\Hc[c_k]u}{u}-\ip{\Hc[c_k]\Phi_n^{(c_k)}}{\Phi_n^{(c_k)}}|$
is controlled by
\begin{align*}
  &\le |\ip{\Hc[c_k](u-\Phi_n^{(c_k)})}{u}|
   +|\ip{\Hc[c_k]\Phi_n^{(c_k)}}{u-\Phi_n^{(c_k)}}|\\
  &\le 2\|\Hc[c_k]\|\cdot\|u-\Phi_n^{(c_k)}\|
   \to 0
\end{align*}
(since $\|\Hc\|\le 1$ and $\Phi_n^{(c_k)}\to u$ strongly).
Hence
\[
  \ip{\Hc[c_k] u}{u}\to\lambda_n^{(\infty)}.
\]

\textbf{Step~4: The limit Rayleigh value $\lambda_n^{(\infty)}$
is achieved only at $\Phi_n^{(\infty)}$ (strict isolation).}
The operators $\Hc$ define, via their spectral structure and
the energy forms $q_c(f):=\ip{\Hc f}{f}$, a sequence of
quadratic forms converging on a common form domain.
For large~$c$, the supremum of $q_c$ over
$\{f:\|f\|=1,\,f\perp\Phi_0^{(c)},\dots,\Phi_{n-1}^{(c)}\}$
is $\lambda_n^{(c)}$, achieved uniquely (up to phase) by
$\Phi_n^{(c)}$.

In the limit: the value $\lambda_n^{(\infty)}$ is strictly larger
than $\lambda_{n+1}^{(\infty)}$ by
Remark~\ref{rem:eigenvalue-isolation}.
Therefore any unit vector $u$ with
\[
  \lim_{k\to\infty}\ip{\Hc[c_k]u}{u}=\lambda_n^{(\infty)}
\]
must satisfy $\ip{\Hc[c_k](u-\Phi_n^{(c_k)})}{u-\Phi_n^{(c_k)}}\to 0$
(deviation from the unique maximiser costs at least
$\lambda_n^{(c_k)}-\lambda_{n+1}^{(c_k)}\ge\delta_{n}c_k^{-1/2}$
per unit deviation in the perpendicular subspace).
More precisely: decompose $u=\alpha_k\Phi_n^{(c_k)}+w_k$
with $w_k\perp\Phi_n^{(c_k)}$ and $\alpha_k=\ip{u}{\Phi_n^{(c_k)}}$.
Then
\[
  \ip{\Hc[c_k]u}{u}
  =|\alpha_k|^2\lambda_n^{(c_k)}
  +\ip{\Hc[c_k]w_k}{w_k}
  \le|\alpha_k|^2\lambda_n^{(c_k)}
  +\|w_k\|^2\lambda_{n+1}^{(c_k)}.
\]
Using $\|u\|=1$ and $\|u\|^2=|\alpha_k|^2+\|w_k\|^2$:
\[
  \lambda_n^{(c_k)}
  \ge\ip{\Hc[c_k]u}{u}
  \ge\lambda_n^{(c_k)}-\|w_k\|^2
     (\lambda_n^{(c_k)}-\lambda_{n+1}^{(c_k)}).
\]
Since $\ip{\Hc[c_k]u}{u}\to\lambda_n^{(\infty)}$
and $\lambda_n^{(c_k)}\to\lambda_n^{(\infty)}$:
\[
  \|w_k\|^2(\lambda_n^{(c_k)}-\lambda_{n+1}^{(c_k)})\to 0.
\]
The gap $\lambda_n^{(c_k)}-\lambda_{n+1}^{(c_k)}
\ge\delta_n c_k^{-1/2}\to 0$,
so this does not immediately force $\|w_k\|\to 0$.

However, from Step~3, $\|u-\Phi_n^{(c_k)}\|\to 0$
(since $\Phi_n^{(c_k)}\to u$ strongly by assumption).
Therefore $\|w_k\|=\|u-\alpha_k\Phi_n^{(c_k)}\|\to 0$
and $\alpha_k\to 1$ (with consistent phase),
confirming that $u$ is the strong limit of $\Phi_n^{(c_k)}$.

For two strong accumulation points $u,v$: both satisfy
$\Phi_n^{(c_k)}\to u$ and $\Phi_n^{(c_j)}\to v$ along their
respective subsequences, and both satisfy
$\lim\ip{\Hc u}{u}=\lim\ip{\Hc v}{v}=\lambda_n^{(\infty)}$.
The decomposition argument above shows $\|w_k^u\|\to 0$ and
$\|w_k^v\|\to 0$, meaning both $u$ and $v$ are asymptotically
parallel to $\Phi_n^{(c)}$ as $c\to\infty$.
Since both are unit vectors and
$\|u-v\|^2 \le 2\|u-\Phi_n^{(c)}\|+2\|v-\Phi_n^{(c)}\|\to 0$,
we get $u=v$ (after phase convention).
\end{proof}

\begin{remark}[What spectral selection uses and avoids]
\label{rem:spectral-selection-scope}
\begin{itemize}
\item \emph{Uses:} $\|\Phi_n^{(c)}\|=1$ and precompactness
  (Lemma~\ref{lem:pswf-precompact}); finite-$c$ eigenrelation and
  variational characterisation (\cite{Slepian1961}); eigenvalue
  convergence \eqref{VIII3}; strict isolation of limit eigenvalues
  (Remark~\ref{rem:eigenvalue-isolation}); $\|\Hc\|\le 1$.
\item \emph{Does not use:} $\HSOT$, $\Hspec$, cross-$c$ inner
  products, double limits, Kato applied to different bandwidths.
\end{itemize}
\end{remark}

\begin{remark}[Gap vanishes but argument is valid]
\label{rem:gap-vanishes}
The Slepian gap $\delta_n c^{-1/2}\to 0$.
The spectral-selection argument does not require the gap to be
bounded away from zero in the limit.
It only requires the gap to be strictly positive at each finite~$c$
(which it is) and the strong convergence of the approximating
sequence (from Stage~1).
The combination of strong convergence + Rayleigh value convergence
+ variational pinning forces $u=v$ without any assumption about
the limit spectrum.
\end{remark}

\subsection*{Stage~2: $\HSOT$ as passive verification}
%---------------------------------------------------------------

\begin{lemma}[$\Phi_n^{(\infty)}$ satisfies the
$\HSOT$-eigenrelation]
\label{lem:weak-eigenrelation}
\emph{Operators used: $\HSOT$ (verification only).
$\Phi_n^{(\infty)}$ is already defined and unique by Stages~1--1.5.}

$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
Since $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly, in particular
weakly.
For any $\psi$:
\begin{align*}
  \ip{\HSOT\Phi_n^{(\infty)}}{\psi}
  &= \ip{\Phi_n^{(\infty)}}{\HSOT\psi}
     \;(\HSOT\text{ symmetric})\\
  &= \lim_c\ip{\Phi_n^{(c)}}{\HSOT\psi}.
\end{align*}
Split: $\ip{\Phi_n^{(c)}}{\HSOT\psi}
=\ip{\Phi_n^{(c)}}{(\HSOT-\Hc)\psi}
+\ip{\Phi_n^{(c)}}{\Hc\psi}$.
First term $\to 0$ by \eqref{VIII2}.
Second:
$\ip{\Phi_n^{(c)}}{\Hc\psi}=\lambda_n^{(c)}\ip{\Phi_n^{(c)}}{\psi}
\to\lambda_n^{(\infty)}\ip{\Phi_n^{(\infty)}}{\psi}$.
\end{proof}

\subsection*{Stage~3: Orthonormality and full convergence}
%---------------------------------------------------------------

\begin{proposition}[$\{\Phi_n^{(\infty)}\}$ is ONS;
$\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly]
\label{prop:weak-limits}
For each fixed $n\ge 0$:
\begin{enumerate}[\rm(a)]
\item $\|\Phi_n^{(\infty)}\|=1$ (Stage~1).
\item $\Phi_n^{(\infty)}$ is the unique strong accumulation point
  of $\{\Phi_n^{(c)}\}$ (Stage~1.5).
\item $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly (subsequence
  principle from (b)).
\item $\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}}=0$ for $n\ne m$.
\item $\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$
  (Stage~2).
\end{enumerate}
\end{proposition}

\begin{proof}
(a)--(c): direct citations.
(d): $\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}}=0$ for all $c$;
strong convergence gives the limit.
(e): Lemma~\ref{lem:weak-eigenrelation}.
\end{proof}

\subsection*{Canonical domain and spectral operator}

\begin{definition}[Canonical domain]
\label{def:domain}
\[
  \Dom(\Hspec):=
  \Bigl\{f\in L^2(\mathbb{R}):
  \sum_{n\ge 0}(\lambda_n^{(\infty)})^2
  |\ip{f}{\Phi_n^{(\infty)}}|^2<\infty
  \Bigr\},
\]
\[
  \Hspec f:=\sum_{n:\lambda_n^{(\infty)}>0}
  \lambda_n^{(\infty)}\ip{f}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)},
  \;f\in\Dom(\Hspec).
\]
\end{definition}

\begin{remark}[Spectral vectors in the domain]
\label{rem:phi-in-domain}
$\Phi_n^{(\infty)}\in\Dom(\Hspec)$ and
$\Hspec\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$
directly from the definition.  Unit norm not required here.
\end{remark}

\begin{theorem}[Properties of the domain]
\label{thm:domain}
\begin{enumerate}[\rm(a)]
\item $\mathrm{span}\{\Phi_n^{(\infty)}\}\subset\Dom(\Hspec)$
  (unconditional); density under
  Hypothesis~\ref{hyp:decay}.
\item $\Hspec$ is symmetric.
\item $\Dom(\Hspec)$ is independent of subsequence choices.
\end{enumerate}
\end{theorem}

\begin{proof}Standard from the spectral series.\end{proof}

\section{Closability}
\label{sec:closable}
%===============================================================

\begin{theorem}[Closability of $\HSOT$]
\label{thm:closable-SOT}
\emph{Operators: $\HSOT$, $\Hc$.  $\Hspec$ not used.}
$\HSOT$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$ and $\HSOT f_k\to g$.
Fix $\varepsilon>0$; choose $K$ with $\|g-\HSOT f_k\|<\varepsilon/2$
for $k\ge K$.  Fix $k\ge K$; by \eqref{VIII6} choose $c_0(k,\varepsilon)$
with $\|\HSOT f_k-\Hc f_k\|<\varepsilon/2$.
Then $\|g-\Hc f_k\|<\varepsilon$.
Fix $c\ge c_0$; let $k\to\infty$:
$\|\Hc f_k\|\le\|f_k\|\to 0$,
so $\|g\|\le\varepsilon$.  Arbitrariness gives $g=0$.
\end{proof}

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable-spec}
\emph{Operators: $\Hspec$.  $\HSOT$ not used.}
$\Hspec$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$ and $\Hspec f_k\to g$.
Set $a_n^{(k)}:=\ip{f_k}{\Phi_n^{(\infty)}}\to 0$ (weak conv.).
For any $\phi$:
$\ip{\Hspec f_k}{\phi}
=\sum_n\lambda_n^{(\infty)}a_n^{(k)}\ip{\Phi_n^{(\infty)}}{\phi}
\to 0$
by dominated convergence.
Hence $g=0$.
\end{proof}

\begin{remark}[No operator mixing]
\label{rem:strict-separation}
The two closability proofs are entirely independent.
$\overline{\Hspec}=\HSOT$: open, not used.
\end{remark}

\section{Operator Norm Rate}

\begin{hypothesis}[Uniform alignment]
\label{hyp:uniform-alignment}
$\sup_{n\le\kappa c}\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|
\le C_\kappa c^{-1/4}$ for $c\ge c_0(\kappa)$.
\end{hypothesis}

\begin{theorem}[Operator norm rate]
\label{thm:opnorm}
Under Hypothesis~\ref{hyp:uniform-alignment}:
$\|\Hc-\Hspec\|\le C_\kappa c^{-1/4}$.
\end{theorem}

\begin{proof}
Decompose $\|(\Hc-\Hspec)f\|^2\le 2A+2E_\kappa$.
Main sum~$A$: $O(c^{-1/2})$ by \eqref{VIII3} + Bessel.
Cross terms~$E_\kappa$: $O(c^{-1/4})$ by
Hypothesis~\ref{hyp:uniform-alignment}.
\end{proof}

%===============================================================
\section{Deficiency Index Analysis}
\label{sec:deficiency}
%===============================================================

By von Neumann (\cite{ReedSimon1975}, Thm.~X.2):
$\Hspec$ is essentially self-adjoint iff
$n_\pm:=\dim\ker(\Hspec^*\mp i)=0$.

\begin{lemma}[Deficiency vectors are orthogonal to
$\{\Phi_n^{(\infty)}\}$]
\label{lem:deficiency-analytic}
\emph{Operators: $\Hspec$ only.}

Let $f\in\ker(\Hspec^*+i)$.
Then $\ip{f}{\Phi_n^{(\infty)}}=0$ for all $n\ge 0$.
\end{lemma}

\begin{proof}
$\Phi_n^{(\infty)}\in\Dom(\Hspec)$ with eigenrelation
(Remark~\ref{rem:phi-in-domain}).
Adjoint: $\ip{\Hspec^*f}{\Phi_n^{(\infty)}}
=\ip{f}{\Hspec\Phi_n^{(\infty)}}$.
Substitute $\Hspec^*f=-if$:
$(\lambda_n^{(\infty)}+i)\ip{f}{\Phi_n^{(\infty)}}=0$.
$\lambda_n^{(\infty)}+i\ne 0$; hence
$\ip{f}{\Phi_n^{(\infty)}}=0$.
\end{proof}

\begin{remark}[Completeness $\Rightarrow$ ESSA]
\label{rem:deficiency-orthogonal}
$\ker(\Hspec^*\pm i)\subseteq
(\mathrm{span}\{\Phi_n^{(\infty)}\})^\perp$;
completeness $\Rightarrow$ $n_\pm=0$.
Converse not established without cyclic vector
or spectral measure.
\end{remark}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}$ is complete in $L^2(\mathbb{R})$.
\end{hypothesis}

%===============================================================
\section{Essential Self-Adjointness}
\label{sec:esa}
%===============================================================

\begin{theorem}[Deficiency indices vanish]
\label{thm:deficiency}
Under Hypothesis~\ref{hyp:decay}: $n_\pm(\Hspec)=0$.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:deficiency-analytic}, every
$f\in\ker(\Hspec^*+i)$ satisfies
$f\perp\{\Phi_n^{(\infty)}\}$;
Hypothesis~\ref{hyp:decay} gives $f=0$.
\end{proof}

\begin{corollary}[ESSA]
\label{cor:esa}
Under Hypothesis~\ref{hyp:decay}:
$\overline{\Hspec}=\Hspec^*$.
\end{corollary}

\begin{theorem}[Spectral identity and RH reduction]
\label{thm:sigma-identity}
Under Hypothesis~\ref{hyp:decay}
and Hypothesis~CCM \cite{CCM2025}:
\begin{enumerate}[\rm(i)]
\item $\sigma_{\rm app}(\Hspec)=\sigma(\Hspec)$.
\item Every non-trivial zero $\rho$ of $\zeta$ lies in
  $\sigma(\Hspec)$.
\item All non-trivial zeros satisfy
  $\mathrm{Re}(\rho)=\tfrac12$.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) ESSA + \cite{ReedSimon1975}, Thm.~VII.13.
(ii) Paper~VIII, Thm.~5.2 $+$ (i).
(iii) CCM + real spectrum.
\end{proof}

\begin{remark}[Meta-corollary]
\label{rem:meta-corollary}
Theorem~\ref{thm:sigma-identity}(iii) is a
\emph{conditional meta-corollary}: open hypothesis, conjectural
CCM, unverified spectral calibration.
Not a proof of RH.
\end{remark}
