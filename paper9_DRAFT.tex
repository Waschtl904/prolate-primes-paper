\begin{abstract}
We continue the spectral analysis of the PSWF concentration
operators $\Hc$ and their formal limit (Papers~V--VIII).

\textbf{Two operator objects.}
(i)~The \emph{spectral operator} $\Hspec$, defined by a spectral
series on an explicit domain.
(ii)~The \emph{SOT-limit operator}
$\HSOT=\mathrm{SOT}\text{-}\lim_{c\to\infty}\Hc$.
Identification $\overline{\Hspec}=\HSOT$ is open
(Problem~\ref{prob:identification}); no proof uses both simultaneously.

\textbf{Key architectural principle.}
Uniqueness of $\Phi_n^{(\infty)}$ is derived from
\emph{norm-resolvent convergence} and Kato perturbation theory,
not from subsequence arguments:
\begin{itemize}
\item \textbf{Stage~1} (Lem.~\ref{lem:pswf-precompact}):
  Precompactness gives consistent norm bounds.
  No uniqueness claim here.
\item \textbf{Stage~1.5} (Lem.~\ref{lem:spectral-selection}):
  The Slepian gap ensures $\lambda_n^{(c)}$ is an isolated simple
  eigenvalue of $\Hc$ at each finite~$c$.
  Under Hypothesis~\ref{hyp:norm-resolvent}
  (norm-resolvent convergence $\Hc\to\HSOT$),
  Kato's analytic perturbation theorem
  (\cite{Kato1966}, Thm.~VIII.1.8)
  gives $\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0$
  \emph{without passing to subsequences}.
  Uniqueness is then trivial: any strong accumulation
  point of $\{\Phi_n^{(c)}\}$ must equal $\Phi_n^{(\infty)}$.
\item \textbf{Stage~2} (Lem.~\ref{lem:weak-eigenrelation}):
  $\HSOT$ introduced as passive verification only.
\end{itemize}

\textbf{Main results:} closability (two independent proofs);
conditional operator-norm rate $O(c^{-1/4})$; deficiency
orthogonality; ESSA under Hypothesis~\ref{hyp:decay};
conditional RH reduction (meta-corollary, not a proof of RH).
\end{abstract}

\section*{Guide for the Referee}
\begin{enumerate}
\item \textbf{Central uniqueness mechanism.}
  Uniqueness of $\Phi_n^{(\infty)}$ rests on
  Hypothesis~\ref{hyp:norm-resolvent} (norm-resolvent
  convergence) together with the Slepian gap
  (\cite{Slepian1961}) and Kato's theorem
  (\cite{Kato1966}, Thm.~VIII.1.8).
  No interleaving, no subsequence selection, no inner-product
  comparisons across different values of~$c$.
\item \textbf{What Stage~1 does and does not prove.}
  Lem.~\ref{lem:pswf-precompact} (Fr\'echet--Kolmogorov)
  shows precompactness and $\|u\|=1$ for any subsequential
  limit. It does \emph{not} prove uniqueness.
  Uniqueness is the content of Stage~1.5 via Kato.
\item \textbf{Operator separation.}
  $\Hspec$ and $\HSOT$ never appear in the same computation.
\item \textbf{$\HSOT$ as verification (Stage~2).}
  Lem.~\ref{lem:weak-eigenrelation} is purely verificationary.
\item \textbf{Open hypotheses.}
  Hyp.~\ref{hyp:norm-resolvent} (norm-resolvent convergence)
  and Hyp.~\ref{hyp:decay} (completeness) are stated
  explicitly; all results depending on them are labelled.
\end{enumerate}

\subsection*{Central hypothesis: norm-resolvent convergence}
%-------------------------------------------------------------

\begin{hypothesis}[Norm-resolvent convergence]
\label{hyp:norm-resolvent}
For every $z\in\mathbb{C}\setminus\mathbb{R}$,
\[
  \|(\Hc-z)^{-1}-(\HSOT-z)^{-1}\|_{\mathcal{B}(L^2)}
  \to 0
  \qquad\text{as }c\to\infty.
\]
\end{hypothesis}

\begin{remark}[Status of Hyp.~\ref{hyp:norm-resolvent}]
\label{rem:nrc-status}
SOT convergence $\Hc\xrightarrow{\mathrm{SOT}}\HSOT$
(\eqref{VIII6}) does not in general imply
norm-resolvent convergence.
Hyp.~\ref{hyp:norm-resolvent} is the key additional
assumption of this paper.
For the bounded self-adjoint family $\Hc$,
norm-resolvent convergence is implied by
\[
  \|\Hc - \HSOT\|_{\mathcal{B}(L^2)} \to 0
  \quad\text{(operator-norm convergence)};
\]
see \cite{Kato1966}, Thm.~VIII.1.5.
Proving either form is open
(Problem~\ref{prob:norm-resolvent}).
All results in this section are conditional
on Hyp.~\ref{hyp:norm-resolvent}.
\end{remark}

%-------------------------------------------------------------
\subsection*{Stage~1 --- Precompactness (auxiliary)}
%-------------------------------------------------------------

\begin{lemma}[PSWF bandlimit precompactness]
\label{lem:pswf-precompact}
\emph{Operators used: none.}

For fixed $n\ge 0$, $\{\Phi_n^{(c)}\}_{c\ge 1}$ is
precompact in $L^2(\mathbb{R})$.
Every sequence $c_k\to\infty$ has a subsequence with
$\Phi_n^{(c_{k_j})}\to u$ strongly; any such $u$ satisfies
$\|u\|=1$.
\end{lemma}

\begin{proof}
FrĂ©chet--Kolmogorov (\cite{ReedSimon1975}, Thm.~IV.26).
\begin{enumerate}[\rm(i)]
\item $\|\Phi_n^{(c)}\|=1$.
\item $|\Phi_n^{(c)}(x)|\le C_{n,\varepsilon}e^{-\delta x^2}$
  for $|x|\ge 1+\varepsilon$, uniform in~$c$
  (\cite{Osipov2013}, Thm.~3.2).
\item $\|\Phi_n^{(c)}\|_{H^1}\le D_n$ uniform in~$c$,
  fixed~$n$ (\cite{Osipov2013}, Prop.~4.1).
\end{enumerate}
FrĂ©chet--Kolmogorov gives precompactness;
$\|u\|=\lim_j\|\Phi_n^{(c_{k_j})}\|=1$.
\end{proof}

\begin{remark}[What Stage~1 does not prove]
\label{rem:stage1-limit}
Lem.~\ref{lem:pswf-precompact} gives existence of
subsequential limits with unit norm.
It does \emph{not} prove that all such limits coincide.
Uniqueness of the limit requires additional structure
in~$c$, provided in Stage~1.5 below.
\end{remark}

%-------------------------------------------------------------
\subsection*{Stage~1.5 --- Spectral selection via Kato theory}
%-------------------------------------------------------------

\begin{lemma}[Inner-product identification]
\label{lem:inner-product-identification}
Let $u, v\in L^2(\mathbb{R})$ with $\|u\|=\|v\|=1$.
If $\mathrm{Re}\,\ip{u}{v}=1$, then $u=v$.
\end{lemma}

\begin{proof}
$\|u-v\|^2=2-2\,\mathrm{Re}\,\ip{u}{v}=0$.
\end{proof}

\begin{lemma}[Spectral selection]
\label{lem:spectral-selection}
\emph{Operators: $\Hc$ (Kato perturbation at finite~$c$),
$\HSOT$ (as limit operator in Hyp.~\ref{hyp:norm-resolvent}).
Conditional on Hyp.~\ref{hyp:norm-resolvent}.}

For fixed $n\ge 0$:
\[
  \|\Phi_n^{(c)}-\Phi_n^{(\infty)}\| \to 0
  \quad\text{as }c\to\infty
\]
(full convergence, no subsequence extraction needed).
In particular, $\Phi_n^{(c)}$ has a \emph{unique} strong
limit $\Phi_n^{(\infty)}\in L^2(\mathbb{R})$ with
$\|\Phi_n^{(\infty)}\|=1$.
\end{lemma}

\begin{proof}
\textbf{Step~1 (isolated simple eigenvalue at each finite $c$).}
The Slepian gap (\cite{Slepian1961})
\[
  \lambda_n^{(c)}-\lambda_{n+1}^{(c)} > 0
\]
ensures that $\lambda_n^{(c)}$ is a \emph{simple isolated}
eigenvalue of the self-adjoint operator $\Hc$ for each
finite~$c$.
The associated eigenprojection is the rank-1 operator
\begin{equation}\label{eq:rank1}
  P_n^{(c)}f = \ip{f}{\Phi_n^{(c)}}\Phi_n^{(c)},
  \quad f\in L^2(\mathbb{R}).
\end{equation}

\textbf{Step~2 (norm-resolvent convergence implies
eigenvalue stability).}
Under Hyp.~\ref{hyp:norm-resolvent},
$\Hc\to\HSOT$ in norm-resolvent sense.
By Kato (\cite{Kato1966}, Thm.~VIII.1.8) applied
to the family $\{\Hc\}$ of bounded self-adjoint operators
converging in norm-resolvent to $\HSOT$:
if $\mu$ is an isolated eigenvalue of $\HSOT$ of
multiplicity~$m$, then for large~$c$ there are exactly
$m$ eigenvalues of $\Hc$ (counted with multiplicity)
converging to~$\mu$, and the associated spectral
projections converge in operator norm:
\begin{equation}\label{eq:proj-conv}
  \|P_n^{(c)}-P_n^{(\infty)}\|_{\mathcal{B}(L^2)} \to 0.
\end{equation}

\textbf{Step~3 (from projection convergence to
eigenfunction convergence).}
Since $P_n^{(\infty)}$ has rank~1 with unit eigenvector
$\Phi_n^{(\infty)}$ (adopting a phase convention),
and $P_n^{(c)}$ has rank~1, from \eqref{eq:proj-conv}:
\[
  \|P_n^{(c)}\Phi_n^{(\infty)}-\Phi_n^{(\infty)}\|
  = \|(P_n^{(c)}-P_n^{(\infty)})\Phi_n^{(\infty)}\|
  \le \|P_n^{(c)}-P_n^{(\infty)}\|\,\|\Phi_n^{(\infty)}\|
  \to 0.
\]
But $P_n^{(c)}\Phi_n^{(\infty)}
=\ip{\Phi_n^{(\infty)}}{\Phi_n^{(c)}}\Phi_n^{(c)}$.
Since $|\ip{\Phi_n^{(\infty)}}{\Phi_n^{(c)}}|\to 1$
(from $\|P_n^{(c)}-P_n^{(\infty)}\|\to 0$; see
\cite{Kato1966}, p.~34, or direct computation),
we get
\begin{equation}\label{eq:phi-conv}
  \|\Phi_n^{(c)}-\Phi_n^{(\infty)}\| \to 0
\end{equation}
after adjusting phases globally so that
$\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}$ is
eventually real and positive.

\textbf{Step~4 (uniqueness is now trivial).}
Equation~\eqref{eq:phi-conv} holds for the full
family (not a subsequence).
If $u$ is any strong accumulation point of
$\{\Phi_n^{(c)}\}$, i.e.\
$\Phi_n^{(c_k)}\to u$ for some $c_k\to\infty$,
then \eqref{eq:phi-conv} gives
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$ along the same
sequence, hence $u=\Phi_n^{(\infty)}$ by uniqueness
of strong limits in $L^2$.
\end{proof}

\begin{remark}[Why norm-resolvent, not SOT]
\label{rem:kato-route}
SOT convergence (\eqref{VIII6}) does not imply
spectral projection convergence in general;
counterexamples exist even for bounded operators.
Norm-resolvent convergence is the minimal assumption
under which Kato's theorem applies and gives
\eqref{eq:proj-conv}; see \cite{ReedSimon1975},
Thm.~VIII.23~(a).
This is the correct route identified by the structural
analysis of the proof: without a continuity assumption
in~$c$, uniqueness of the limit cannot be established
from precompactness alone.
\end{remark}

\begin{remark}[What the proof uses and avoids]
\label{rem:selection-scope}
\emph{Uses:} Slepian gap (\cite{Slepian1961});
Hyp.~\ref{hyp:norm-resolvent} (norm-resolvent conv.);
Kato Thm.~VIII.1.8 (\cite{Kato1966});
uniqueness of strong limits in~$L^2$.

\emph{Does not use:}
interleaving of subsequences;
cross-$c$ inner-product comparisons;
subnet arguments;
spectrum of $\HSOT$ beyond eigenvalue isolation;
$\Hspec$.
\end{remark}

\begin{remark}[Vanishing gap causes no difficulty]
\label{rem:gap-not-needed}
The Slepian gap $\lambda_n^{(c)}-\lambda_{n+1}^{(c)}
\asymp c^{-1/2}\to 0$.
Kato's theorem requires only that $\lambda_n^{(\infty)}$
is an isolated eigenvalue of $\HSOT$
(Rem.~\ref{rem:eigenvalue-isolation});
the shrinking gap at finite~$c$ causes no difficulty
because the argument operates at the level of the
limit operator and works backwards to finite~$c$.
\end{remark}

\subsection*{Stage~2 --- $\HSOT$ eigenrelation (verification)}
%-------------------------------------------------------------

\begin{lemma}[$\Phi_n^{(\infty)}$ satisfies
the $\HSOT$-eigenrelation]
\label{lem:weak-eigenrelation}
\emph{Operators: $\HSOT$ (verification only).}

$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
$\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly
(Lem.~\ref{lem:spectral-selection}), hence weakly.
For any $\psi$:
\[
  \ip{\HSOT\Phi_n^{(\infty)}}{\psi}
  =\ip{\Phi_n^{(\infty)}}{\HSOT\psi}
  =\lim_c\ip{\Phi_n^{(c)}}{\HSOT\psi}.
\]
Split $\ip{\Phi_n^{(c)}}{\HSOT\psi}
=\ip{\Phi_n^{(c)}}{(\HSOT-\Hc)\psi}
+\ip{\Phi_n^{(c)}}{\Hc\psi}$.
First term $\to 0$ by \eqref{VIII2}.
Second: $\lambda_n^{(c)}\ip{\Phi_n^{(c)}}{\psi}
\to\lambda_n^{(\infty)}\ip{\Phi_n^{(\infty)}}{\psi}$.
\end{proof}

\subsection*{Stage~3 --- Orthonormality}
%-------------------------------------------------------------

\begin{proposition}[$\{\Phi_n^{(\infty)}\}$ is ONS]
\label{prop:weak-limits}
Conditional on Hyp.~\ref{hyp:norm-resolvent}.
\begin{enumerate}[\rm(a)]
\item $\|\Phi_n^{(\infty)}\|=1$.
\item $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly.
\item $\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}}=0$,
  $n\ne m$.
\item $\HSOT\Phi_n^{(\infty)}
  =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(a) $\|\Phi_n^{(c)}\|=1$, strong limit preserves norm.
(b) Lem.~\ref{lem:spectral-selection}.
(c) $\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}}=0$ + strong conv.
(d) Lem.~\ref{lem:weak-eigenrelation}.
\end{proof}

\subsection*{Canonical domain and spectral operator}

\begin{definition}[Canonical domain]
\label{def:domain}
\[
  \Dom(\Hspec):=\Bigl\{f\in L^2:\,
  \textstyle\sum_n(\lambda_n^{(\infty)})^2
  |\ip{f}{\Phi_n^{(\infty)}}|^2<\infty\Bigr\},
\]
\[
  \Hspec f:=\sum_n\lambda_n^{(\infty)}
  \ip{f}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)},
  \quad f\in\Dom(\Hspec).
\]
\end{definition}

\begin{theorem}[Domain properties]
\label{thm:domain}
(a)~$\mathrm{span}\{\Phi_n^{(\infty)}\}\subset\Dom(\Hspec)$;
dense under Hyp.~\ref{hyp:decay}.
(b)~$\Hspec$ symmetric.
(c)~$\Dom(\Hspec)$ independent of subsequence choices.
\end{theorem}

\section{Closability}
\label{sec:closable}
%=============================================================

\begin{theorem}[Closability of $\HSOT$]
\label{thm:closable-SOT}
$\HSOT$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$, $\HSOT f_k\to g$.
For each fixed~$c$, $\Hc$ is bounded:
$\|\Hc f_k\|\le\|f_k\|\to 0$.
By \eqref{VIII6} (SOT): $\HSOT f_k-\Hc f_k\to 0$
pointwise, so $\Hc f_k\to g$ as well.
Hence $\|g\|=\lim_k\|\Hc f_k\|=0$, i.e.\ $g=0$.
\end{proof}

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable-spec}
$\Hspec$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$, $\Hspec f_k\to g$.
For any $\phi$: $\ip{\Hspec f_k}{\phi}
=\sum_n\lambda_n^{(\infty)}\ip{f_k}{\Phi_n^{(\infty)}}
\ip{\Phi_n^{(\infty)}}{\phi}\to 0$.
Hence $g=0$.
\end{proof}

\section{Deficiency Index Analysis}
\label{sec:deficiency}

\begin{lemma}[Deficiency vectors $\perp\{\Phi_n^{(\infty)}\}$]
\label{lem:deficiency-analytic}
If $f\in\ker(\Hspec^*+i)$ then
$\ip{f}{\Phi_n^{(\infty)}}=0$ for all~$n$.
\end{lemma}

\begin{proof}
$(\lambda_n^{(\infty)}+i)\ip{f}{\Phi_n^{(\infty)}}=0$;
$\lambda_n^{(\infty)}+i\ne 0$.
\end{proof}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}$ is complete in $L^2(\mathbb{R})$.
\end{hypothesis}

\section{Essential Self-Adjointness}
\label{sec:esa}
%=============================================================

\begin{theorem}[$n_\pm=0$]
\label{thm:deficiency}
Under Hyp.~\ref{hyp:decay}: $n_\pm(\Hspec)=0$.
\end{theorem}

\begin{proof}
$f\in\ker(\Hspec^*+i)\Rightarrow f\perp\{\Phi_n^{(\infty)}\}
\Rightarrow f=0$.
\end{proof}

\begin{corollary}[ESSA]
\label{cor:esa}
Under Hyp.~\ref{hyp:decay}: $\overline{\Hspec}=\Hspec^*$.
\end{corollary}

\begin{theorem}[Spectral identity and RH reduction]
\label{thm:sigma-identity}
Under Hyp.~\ref{hyp:decay}, Hyp.~\ref{hyp:norm-resolvent},
and CCM~\cite{CCM2025}:
(i)~$\sigma_{\rm app}(\Hspec)=\sigma(\Hspec)$;
(ii)~every non-trivial zero $\rho$ of~$\zeta$ lies in
$\sigma(\Hspec)$;
(iii)~$\mathrm{Re}(\rho)=\tfrac12$.
\end{theorem}

\begin{remark}[Meta-corollary]
\label{rem:meta-corollary}
Thm.~\ref{thm:sigma-identity}(iii) depends on
Hyp.~\ref{hyp:decay} (open), Hyp.~\ref{hyp:norm-resolvent}
(open), CCM (conjectural),
and an unverified spectral calibration.
Not a proof of RH.
\end{remark}
