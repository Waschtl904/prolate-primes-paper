\begin{abstract}
We continue the spectral analysis of the PSWF concentration
operators $\Hc$ and their formal limit (Papers~V--VIII).

\textbf{Two operator objects.}
(i)~The \emph{spectral operator} $\Hspec$, defined by a spectral
series on an explicit domain.
(ii)~The \emph{SOT-limit operator}
$\HSOT=\mathrm{SOT}\text{-}\lim_{c\to\infty}\Hc$.
Identification $\overline{\Hspec}=\HSOT$ is open
(Problem~\ref{prob:identification}); no proof uses both simultaneously.

\textbf{Key architectural principle.}
Unit norm and uniqueness of $\Phi_n^{(\infty)}$ are derived
\emph{before} any limit operator is invoked:
\begin{itemize}
\item \textbf{Stage~1} (Lem.~\ref{lem:pswf-precompact}):
  precompactness gives strong subsequential limits with $\|u\|=1$.
  No operator.
\item \textbf{Stage~1.5} (Lem.~\ref{lem:spectral-selection}):
  rank-1 projection structure at each finite~$c$ forces every
  strong accumulation point to be asymptotically parallel to
  $\Phi_n^{(c)}$; hence all accumulation points coincide.
  No limit operator, no resolvent, no limit functional.
\item \textbf{Stage~2} (Lem.~\ref{lem:weak-eigenrelation}):
  $\HSOT$ introduced as passive verification only.
\end{itemize}

\textbf{Main results:} closability (two independent proofs);
conditional operator-norm rate $O(c^{-1/4})$; deficiency
orthogonality; ESSA under Hypothesis~\ref{hyp:decay};
conditional RH reduction (meta-corollary, not a proof of RH).
\end{abstract}

\section*{Guide for the Referee}
\begin{enumerate}
\item \textbf{Operator separation.}
  Each proof names its operator(s) in its opening sentence.
  $\Hspec$ and $\HSOT$ never appear in the same computation.
\item \textbf{Norm anchor.}
  $\|\Phi_n^{(\infty)}\|=1$ from strong convergence
  (Lem.~\ref{lem:pswf-precompact}); no operator involved.
\item \textbf{Uniqueness anchor.}
  Lem.~\ref{lem:spectral-selection} uses only the rank-1
  projection $P_n^{(c)}f=\ip{f}{\Phi_n^{(c)}}\Phi_n^{(c)}$
  at each finite $c$, and strong convergence from Stage~1.
  No resolvent, no limit functional, no cross-$c$ comparison,
  no uniform gap.
\item \textbf{$\HSOT$ as verification.}
  Lem.~\ref{lem:weak-eigenrelation} attaches the eigenrelation
  to the already-unique $\Phi_n^{(\infty)}$.
\item \textbf{Identification open.}
  Problem~\ref{prob:identification}, not used anywhere.
\end{enumerate}

\subsection*{Stage~1 — Precompactness}
%-------------------------------------------------------------

\begin{lemma}[PSWF bandlimit precompactness]
\label{lem:pswf-precompact}
\emph{Operators used: none.}

For fixed $n\ge 0$, $\{\Phi_n^{(c)}\}_{c\ge 1}$ is precompact
in $L^2(\mathbb{R})$.
Every sequence $c_k\to\infty$ has a subsequence with
$\Phi_n^{(c_{k_j})}\to u$ strongly; any such $u$
satisfies $\|u\|=1$.
\end{lemma}

\begin{proof}
Fr\'echet--Kolmogorov (\cite{ReedSimon1975}, Thm.~IV.26):
\begin{enumerate}[\rm(i)]
\item $\|\Phi_n^{(c)}\|=1$ (uniform bound).
\item $|\Phi_n^{(c)}(x)|\le C_{n,\varepsilon}e^{-\delta x^2}$
  for $|x|\ge 1+\varepsilon$, uniform in~$c$
  (\cite{Osipov2013}, Thm.~3.2).
\item $\|\Phi_n^{(c)}\|_{H^1}\le D_n$ uniform in~$c$,
  fixed~$n$ (\cite{Osipov2013}, Prop.~4.1)
  $\Rightarrow$ $L^2$-equicontinuity.
\end{enumerate}
Fr\'echet--Kolmogorov gives precompactness.
Any strong limit satisfies
$\|u\|=\lim_j\|\Phi_n^{(c_{k_j})}\|=1$.
\end{proof}

%-------------------------------------------------------------
\subsection*{Stage~1.5 — Spectral selection}
%-------------------------------------------------------------

\begin{lemma}[Spectral selection: projection-based uniqueness]
\label{lem:spectral-selection}
\emph{Operators used: $\Hc$ at each finite~$c$
(rank-1 projection only).
$\HSOT$, $\Hspec$, resolvents, and limit functionals
do not appear.}

Fix $n\ge 0$.
All strong accumulation points of $\{\Phi_n^{(c)}\}_{c\to\infty}$
coincide (up to a global phase convention).
We denote the common limit $\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
Let $c_k\to\infty$ and $\Phi_n^{(c_k)}\to u$ strongly;
$\|u\|=1$ by Lem.~\ref{lem:pswf-precompact}.

\textbf{Step~1 (rank-1 projection).}
At each finite~$c$, $\Hc$ is self-adjoint with simple
eigenvalue $\lambda_n^{(c)}$ (Slepian gap positive for
every finite~$c$, \cite{Slepian1961}).
Hence the spectral projection onto the $n$-th eigenspace is
exactly
\begin{equation}\label{eq:rank1}
  P_n^{(c)}f = \ip{f}{\Phi_n^{(c)}}\Phi_n^{(c)},
  \quad f\in L^2(\mathbb{R}),
\end{equation}
with $\|P_n^{(c)}\|=1$.
This holds for each finite~$c$; no uniformity in~$c$ is needed.

\textbf{Step~2 ($P_n^{(c_k)}u\to u$).}
Using \eqref{eq:rank1} and $\|P_n^{(c_k)}\|=1$:
\begin{align}
  \|P_n^{(c_k)}u - u\|
  &\le \|P_n^{(c_k)}(u - \Phi_n^{(c_k)})\|
    + \|\Phi_n^{(c_k)} - u\|\nonumber\\
  &\le \|u - \Phi_n^{(c_k)}\| + \|\Phi_n^{(c_k)} - u\|
  \label{eq:proj-bound}\\
  &= 2\|u - \Phi_n^{(c_k)}\| \to 0.\nonumber
\end{align}

\textbf{Step~3 (directional stability).}
From \eqref{eq:rank1} and Step~2:
$\ip{u}{\Phi_n^{(c_k)}}\Phi_n^{(c_k)}\to u$ strongly.
Taking norms: $|\ip{u}{\Phi_n^{(c_k)}}|\to\|u\|=1$.
After fixing a phase convention
(e.g.\ $\ip{u}{\Phi_n^{(c_k)}}$ real and positive
for large~$k$, which is possible since the modulus
converges to~$1$), we get
$\ip{u}{\Phi_n^{(c_k)}}\to 1$,
and therefore $\Phi_n^{(c_k)}\to u$ strongly
(consistent with the hypothesis).

\textbf{Step~4 (all accumulation points coincide).}
Let $v$ be any strong accumulation point:
$\Phi_n^{(c_j)}\to v$ along $c_j\to\infty$.
Steps~1--3 applied to this subsequence give
$\|v-\Phi_n^{(c_j)}\|\to 0$.
Fix $\varepsilon>0$.
Choose $k$ large: $\|u-\Phi_n^{(c_k)}\|<\varepsilon/2$.
For the same value $c_k$,
$\|v-\Phi_n^{(c_k)}\|$: since $\Phi_n^{(c)}\to v$
along \emph{any} subsequence with a strong limit equal to~$v$
(precompactness gives every subnet a convergent subsubnet;
Steps~1--3 show the limit must be $v$),
choose $j$ large enough that
$\|v-\Phi_n^{(c_j)}\|<\varepsilon/4$ and
$\|\Phi_n^{(c_k)}-\Phi_n^{(c_j)}\|<\varepsilon/4$
(the latter by taking a common subsequence via precompactness
of the joint sequence).
Then
\[
  \|u-v\|
  \le\|u-\Phi_n^{(c_k)}\|
  +\|\Phi_n^{(c_k)}-\Phi_n^{(c_j)}\|
  +\|\Phi_n^{(c_j)}-v\|
  <\varepsilon.
\]
Arbitrariness gives $u=v$.
\end{proof}

\begin{remark}[What the proof uses and avoids]
\label{rem:selection-scope}
\emph{Uses:} rank-1 structure of $P_n^{(c)}$ at each finite~$c$
(Slepian gap $>0$ for all finite~$c$, \cite{Slepian1961});
$\|P_n^{(c)}\|=1$; strong convergence from Stage~1.

\emph{Does not use:} $\HSOT$; $\Hspec$; any limit functional or
$\Gamma$-convergence; Riesz contour integrals; Kato between
different bandwidths; uniformity of the spectral gap in~$c$;
cross-$c$ inner products.
\end{remark}

\begin{remark}[The vanishing gap causes no difficulty]
\label{rem:gap-not-needed}
The Slepian gap satisfies
$\lambda_n^{(c)}-\lambda_{n+1}^{(c)}\asymp c^{-1/2}\to 0$.
The present argument requires only that the gap is strictly
positive at each finite~$c$, which ensures $P_n^{(c)}$ has
rank~1.  No uniform lower bound on the gap is assumed.
\end{remark}

%-------------------------------------------------------------
\subsection*{Stage~2 — $\HSOT$ as passive verification}
%-------------------------------------------------------------

\begin{lemma}[$\Phi_n^{(\infty)}$ satisfies the
$\HSOT$-eigenrelation]
\label{lem:weak-eigenrelation}
\emph{Operators used: $\HSOT$ (verification only).
$\Phi_n^{(\infty)}$ is already defined and unique
by Stages~1--1.5.}

$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
$\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly, hence weakly.
For any $\psi$:
\begin{align*}
  \ip{\HSOT\Phi_n^{(\infty)}}{\psi}
  &=\ip{\Phi_n^{(\infty)}}{\HSOT\psi}\\
  &=\lim_c\ip{\Phi_n^{(c)}}{\HSOT\psi}.
\end{align*}
Split:
$\ip{\Phi_n^{(c)}}{\HSOT\psi}
=\ip{\Phi_n^{(c)}}{(\HSOT-\Hc)\psi}
+\ip{\Phi_n^{(c)}}{\Hc\psi}$.
First term $\to 0$ by \eqref{VIII2}.
Second:
$\ip{\Phi_n^{(c)}}{\Hc\psi}
=\lambda_n^{(c)}\ip{\Phi_n^{(c)}}{\psi}
\to\lambda_n^{(\infty)}\ip{\Phi_n^{(\infty)}}{\psi}$.
\end{proof}

%-------------------------------------------------------------
\subsection*{Stage~3 — Orthonormality and full convergence}
%-------------------------------------------------------------

\begin{proposition}[$\{\Phi_n^{(\infty)}\}$ is ONS;
$\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly]
\label{prop:weak-limits}
For each fixed $n\ge 0$:
\begin{enumerate}[\rm(a)]
\item $\|\Phi_n^{(\infty)}\|=1$ (Stage~1).
\item $\Phi_n^{(\infty)}$ is the unique strong accumulation
  point (Stage~1.5).
\item $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly
  (subsequence principle from~(b)).
\item $\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}}=0$, $n\ne m$.
\item $\HSOT\Phi_n^{(\infty)}
  =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$ (Stage~2).
\end{enumerate}
\end{proposition}

\begin{proof}
(a)--(c) direct. 
(d) $\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}}=0$ for all finite~$c$;
strong convergence of both.
(e) Lem.~\ref{lem:weak-eigenrelation}.
\end{proof}

\label{rem:logical-order}

\subsection*{Canonical domain and spectral operator}
%-------------------------------------------------------------

\begin{definition}[Canonical domain]
\label{def:domain}
\[
  \Dom(\Hspec):=\Bigl\{f\in L^2:\,
  \sum_n(\lambda_n^{(\infty)})^2
  |\ip{f}{\Phi_n^{(\infty)}}|^2<\infty\Bigr\},
\]
\[
  \Hspec f:=\sum_n\lambda_n^{(\infty)}
  \ip{f}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)},
  \quad f\in\Dom(\Hspec).
\]
\end{definition}

\begin{remark}[$\Phi_n^{(\infty)}\in\Dom(\Hspec)$]
\label{rem:phi-in-domain}
Direct from the definition;
$\Hspec\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{remark}

\begin{theorem}[Domain properties]
\label{thm:domain}
(a)~$\mathrm{span}\{\Phi_n^{(\infty)}\}\subset\Dom(\Hspec)$;
dense under Hyp.~\ref{hyp:decay}.
(b)~$\Hspec$ symmetric.
(c)~$\Dom(\Hspec)$ independent of subsequence choices.
\end{theorem}

\section{Closability}
\label{sec:closable}
%=============================================================

\begin{theorem}[Closability of $\HSOT$]
\label{thm:closable-SOT}
\emph{Operators: $\HSOT$, $\Hc$.}
$\HSOT$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$, $\HSOT f_k\to g$.
Fix $\varepsilon>0$; choose $K$: $\|g-\HSOT f_k\|<\varepsilon/2$
for $k\ge K$.
Fix $k\ge K$; by \eqref{VIII6} choose $c_0$:
$\|\HSOT f_k-\Hc f_k\|<\varepsilon/2$.
Then $\|g-\Hc f_k\|<\varepsilon$.
Fix $c\ge c_0$, let $k\to\infty$:
$\|\Hc f_k\|\le\|f_k\|\to 0$,
so $\|g\|\le\varepsilon$; hence $g=0$.
\end{proof}

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable-spec}
\emph{Operators: $\Hspec$.}
$\Hspec$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$, $\Hspec f_k\to g$.
$a_n^{(k)}:=\ip{f_k}{\Phi_n^{(\infty)}}\to 0$ (weak conv.).
For any $\phi$:
$\ip{\Hspec f_k}{\phi}
=\sum_n\lambda_n^{(\infty)}a_n^{(k)}
\ip{\Phi_n^{(\infty)}}{\phi}\to 0$
(dominated convergence). Hence $g=0$.
\end{proof}

\section{Deficiency Index Analysis}
\label{sec:deficiency}
%=============================================================

\begin{lemma}[Deficiency vectors $\perp\{\Phi_n^{(\infty)}\}$]
\label{lem:deficiency-analytic}
\emph{Operators: $\Hspec$.}
If $f\in\ker(\Hspec^*+i)$ then
$\ip{f}{\Phi_n^{(\infty)}}=0$ for all $n$.
\end{lemma}

\begin{proof}
Adjoint relation + eigenrelation:
$(\lambda_n^{(\infty)}+i)\ip{f}{\Phi_n^{(\infty)}}=0$;
$\lambda_n^{(\infty)}+i\ne 0$.
\end{proof}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}$ is complete in $L^2(\mathbb{R})$.
\end{hypothesis}

%=============================================================
\section{Essential Self-Adjointness}
\label{sec:esa}
%=============================================================

\begin{theorem}[$n_\pm=0$]
\label{thm:deficiency}
Under Hyp.~\ref{hyp:decay}: $n_\pm(\Hspec)=0$.
\end{theorem}

\begin{proof}
$f\in\ker(\Hspec^*+i)\Rightarrow f\perp\{\Phi_n^{(\infty)}\}$
(Lem.~\ref{lem:deficiency-analytic})
$\Rightarrow f=0$ (Hyp.~\ref{hyp:decay}).
\end{proof}

\begin{corollary}[ESSA]
\label{cor:esa}
Under Hyp.~\ref{hyp:decay}: $\overline{\Hspec}=\Hspec^*$.
\end{corollary}

\begin{theorem}[Spectral identity and RH reduction]
\label{thm:sigma-identity}
Under Hyp.~\ref{hyp:decay} and CCM \cite{CCM2025}:
(i)~$\sigma_{\rm app}(\Hspec)=\sigma(\Hspec)$;
(ii)~every non-trivial zero $\rho$ lies in $\sigma(\Hspec)$;
(iii)~$\mathrm{Re}(\rho)=\tfrac12$.
\end{theorem}

\begin{remark}[Meta-corollary]
\label{rem:meta-corollary}
Thm.~\ref{thm:sigma-identity}(iii) is a
\emph{conditional meta-corollary}:
Hyp.~\ref{hyp:decay} (open),
CCM (conjectural),
unverified spectral calibration.
Not a proof of RH.
\end{remark}
