\section*{Honest Summary for the Referee}
%============================================================

This paper establishes the following \emph{unconditional} results:
\begin{enumerate}[(U1)]
\item\label{U1}
  The PSWF family $\{\Phi_n^{(c)}\}$ is precompact in $L^2$;
  every subsequential limit has unit norm
  (Fr\'echet--Kolmogorov; no operator).
\item\label{U2}
  For each fixed finite $c$, the eigenprojection $P_n^{(c)}$
  has rank~1 (Slepian gap).
\item\label{U3}
  $\Hspec$ (defined by spectral series) is closable and symmetric.
\item\label{U4}
  $\HSOT:=\mathrm{SOT}\text{-}\lim_c\Hc$ is a well-defined
  bounded self-adjoint operator on all of $L^2$
  (SOT limit of bounded self-adjoint contractions;
  see Rem.~\ref{rem:HSOT-bounded}).
\end{enumerate}

The following results are \emph{conditional}:
\begin{enumerate}[(C1)]
\item\label{C1}
  \textbf{Under Hyp.~\ref{hyp:norm-resolvent}
  and Hyp.~\ref{hyp:HSOT-discrete}:}
  $\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0$
  (full convergence, no subsequence selection).
\item\label{C2}
  \textbf{Under (C1) and Hyp.~\ref{hyp:decay}:}
  completeness of $\{\Phi_n^{(\infty)}\}$ implies ESSA of
  $\Hspec$.
  The converse direction
  (ESSA $\Rightarrow$ completeness) is \emph{open}
  (see Rem.~\ref{rem:essa-one-direction}).
\item\label{C3}
  \textbf{Under Hyp.~\ref{hyp:uniform-alignment}:}
  $\|\Hc-\Hspec\|\le C_\kappa c^{-1/4}$;
  norm-resolvent convergence is then conjectured
  under an additional Riesz-basis stability condition
  (see Rem.~\ref{rem:opnorm-bridge}).
\end{enumerate}

\section{Setup and Inherited Results}
\label{sec:setup}
%=============================================================

We work in $L^2(\mathbb{R})$.
The PSWF concentration operators $\Hc$ are bounded,
self-adjoint, $\|\Hc\|\le 1$, with simple positive
eigenvalues $\lambda_0^{(c)}>\lambda_1^{(c)}>\cdots>0$
and orthonormal eigenfunctions $\Phi_n^{(c)}$.

Inherited results:
\begin{enumerate}[(VIII.1)]
\item\label{VIII2}
  $\|(\Hc-\HSOT)f\|\le C_\kappa c^{-3/4}\|f\|$
  for fixed unit $f$ (Paper~VII, Thm.~C).
\item\label{VIII3}
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|\le C_\kappa c^{-1/4}$
  (Paper~VIII, Cor.~4.3).
\item\label{VIII6}
  $\Hc\xrightarrow{\mathrm{SOT}}\HSOT$
  (Paper~X, Lem.~2.2).
\end{enumerate}

\begin{remark}[$\HSOT$ is bounded self-adjoint; no closability needed]
\label{rem:HSOT-bounded}
The SOT limit of a net of bounded self-adjoint operators
$T_\alpha$ with $\sup_\alpha\|T_\alpha\|\le M$ is a bounded
self-adjoint operator with $\|T\|\le M$
(\cite{ReedSimon1975}, Thm.~VI.1 and preceding discussion).
Applied here: each $\Hc$ is self-adjoint and
$\|\Hc\|\le 1$, so $\HSOT$ is bounded self-adjoint with
$\|\HSOT\|\le 1$ and $\Dom(\HSOT)=L^2$.
Every bounded everywhere-defined operator is closable;
no separate closability argument for $\HSOT$ is needed.
\end{remark}

\section{Open Hypotheses}
\label{sec:hypotheses}
%=============================================================

\begin{hypothesis}[Discrete spectrum of $\HSOT$]
\label{hyp:HSOT-discrete}
$\HSOT$ has simple eigenvalues
$\lambda_0^{(\infty)}>\lambda_1^{(\infty)}>\cdots>0$
(consistent with \eqref{VIII3}),
and the corresponding eigenvectors $\{\Phi_n^{(\infty)}\}$
span an eigenspace decomposition of $\HSOT$.
\end{hypothesis}

\begin{hypothesis}[Norm-resolvent convergence]
\label{hyp:norm-resolvent}
For some (equivalently every) $z\in\mathbb{C}\setminus\mathbb{R}$:
\[
  \|(\Hc-z)^{-1}-(\HSOT-z)^{-1}\|_{\mathcal{B}(L^2)}
  \to 0.
\]
\end{hypothesis}

\begin{remark}[SOT does not imply norm-resolvent]
\label{rem:SOT-not-enough}
SOT convergence \eqref{VIII6} does not in general imply
norm-resolvent convergence;
counterexamples exist among uniformly bounded
self-adjoint families (\cite{ReedSimon1975}, Ex.~VIII.7.14).
Hyp.~\ref{hyp:norm-resolvent} is a strictly stronger
assumption.
\end{remark}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}_{n\ge 0}$ is complete
in $L^2(\mathbb{R})$.
\end{hypothesis}

\begin{hypothesis}[Uniform alignment]
\label{hyp:uniform-alignment}
For each $\kappa>0$:
$\sup_{n\le\kappa c}\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|
\le C_\kappa c^{-1/4}$.
\end{hypothesis}

\begin{remark}[Dependency diagram]
\label{rem:dep}
\[
  \text{Hyp.~\ref{hyp:uniform-alignment}}
  \xRightarrow{(*)}\text{Hyp.~\ref{hyp:norm-resolvent}}
  \xRightarrow{+\text{Hyp.~\ref{hyp:HSOT-discrete}}}
  \text{(C1)}
  \xRightarrow{+\text{Hyp.~\ref{hyp:decay}}}
  \text{(C2) ESSA}.
\]
The step $(*)$ is conjectural;
see Rem.~\ref{rem:opnorm-bridge}.
\end{remark}

\section{Unconditional Results}
\label{sec:unconditional}
%=============================================================

\subsection{Precompactness}

\begin{lemma}[PSWF precompactness]
\label{lem:pswf-precompact}
\emph{No operators.}
For fixed $n\ge 0$, $\{\Phi_n^{(c)}\}_{c\ge 1}$
is precompact in $L^2$.
Every subsequence has a further subsequence converging
strongly to some unit vector.
\end{lemma}

\begin{proof}
Fr\'echet--Kolmogorov (\cite{ReedSimon1975}, Thm.~IV.26)
with: (i)~$\|\Phi_n^{(c)}\|=1$;
(ii)~uniform exponential decay for fixed~$n$
(\cite{Osipov2013}, Thm.~3.2);
(iii)~uniform $H^1$ bound for fixed~$n$
(\cite{Osipov2013}, Prop.~4.1).
\end{proof}

\begin{remark}[Precompactness does not give uniqueness]
\label{rem:precompact-not-unique}
A precompact family may have multiple strong
accumulation points.
Uniqueness of the limit cannot be established
from precompactness alone.
\end{remark}

\subsection{Rank-1 structure at finite $c$}

\begin{lemma}[Rank-1 eigenprojection]
\label{lem:rank1}
\emph{Operator: $\Hc$ at fixed finite $c$.}
For each finite $c$, the Slepian gap
$\lambda_n^{(c)}>\lambda_{n+1}^{(c)}$
(\cite{Slepian1961}) gives:
\[
  P_n^{(c)}f=\ip{f}{\Phi_n^{(c)}}\Phi_n^{(c)},
  \quad \|P_n^{(c)}\|=1.
\]
No uniformity in $c$ is claimed.
\end{lemma}

\subsection{Spectral operator: definition and closability}

\begin{definition}[Canonical domain and $\Hspec$]
\label{def:domain}
\[
  \Dom(\Hspec):=\Bigl\{f\in L^2:\,
  {\textstyle\sum_n}(\lambda_n^{(\infty)})^2
  |\ip{f}{\Phi_n^{(\infty)}}|^2<\infty\Bigr\},
\]
\[
  \Hspec f:=\sum_n\lambda_n^{(\infty)}
  \ip{f}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)}.
\]
Here $\{\Phi_n^{(\infty)}\}$ is any fixed choice of
unit vectors forming an ONS in $L^2$;
well-definedness of $\Hspec$ depends only on the ONS
property, not on how the $\Phi_n^{(\infty)}$ are obtained.
\end{definition}

\begin{theorem}[Closability and symmetry of $\Hspec$]
\label{thm:closable-spec}
$\Hspec$ is closable and symmetric.
\end{theorem}

\begin{proof}
\textit{Symmetry:} Direct from definition.

\textit{Closability:}
Let $f_k\to 0$ in $L^2$, $\Hspec f_k\to g$.
For each $n$: $\ip{f_k}{\Phi_n^{(\infty)}}\to 0$.
For any $\phi\in L^2$:
$\ip{\Hspec f_k}{\phi}
=\sum_n\lambda_n^{(\infty)}\ip{f_k}{\Phi_n^{(\infty)}}
\ip{\Phi_n^{(\infty)}}{\phi}\to 0$.
Hence $g=0$.
\end{proof}

\section{Conditional: Full Eigenfunction Convergence}
\label{sec:conditional-convergence}
%=============================================================

\begin{theorem}[Full norm convergence of $\Phi_n^{(c)}$]
\label{thm:phi-conv}
\emph{Conditional on
Hyp.~\ref{hyp:norm-resolvent} and
Hyp.~\ref{hyp:HSOT-discrete}.}

$\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0$
as $c\to\infty$.
\end{theorem}

\begin{proof}
\textbf{Step~1.}
By Hyp.~\ref{hyp:HSOT-discrete},
$\lambda_n^{(\infty)}$ is a simple isolated eigenvalue
of $\HSOT$ with eigenprojection $P_n^{(\infty)}$ of rank~1.

\textbf{Step~2.}
Under Hyp.~\ref{hyp:norm-resolvent},
Kato (\cite{Kato1966}, Thm.~VIII.1.8) gives:
\[
  \|P_n^{(c)}-P_n^{(\infty)}\|_{\mathcal{B}(L^2)}\to 0.
\]

\textbf{Step~3.}
Since both $P_n^{(c)}$ and $P_n^{(\infty)}$ have rank~1,
norm convergence of the projections implies norm
convergence of the unit eigenvectors up to a global phase
(\cite{Kato1966}, p.~34):
\[
  |\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}|^2
  =\mathrm{tr}(P_n^{(c)}P_n^{(\infty)})\to 1,
\]
hence $\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|^2
=2-2\,\mathrm{Re}\,\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}\to 0$
(positive phase convention).
\end{proof}

\section{Conditional: Eigenrelation for $\HSOT$}
\label{sec:eigenrelation}
%=============================================================

\begin{lemma}[$\HSOT$-eigenrelation]
\label{lem:weak-eigenrelation}
\emph{Conditional on Thm.~\ref{thm:phi-conv}.
Uses: boundedness of $\HSOT$, \eqref{VIII2}, \eqref{VIII3}.}

$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
By Thm.~\ref{thm:phi-conv},
$\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0$.
Since $\HSOT$ is bounded (Rem.~\ref{rem:HSOT-bounded}):
\begin{equation}\label{eq:HSOT-norm-conv}
  \|\HSOT\Phi_n^{(c)}-\HSOT\Phi_n^{(\infty)}\|
  \le\|\HSOT\|\,\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0.
\end{equation}
Using \eqref{VIII2}:
$\HSOT\Phi_n^{(c)}
=\Hc\Phi_n^{(c)}+(\HSOT-\Hc)\Phi_n^{(c)}
=\lambda_n^{(c)}\Phi_n^{(c)}+r_c$,
where $\|r_c\|\le C_\kappa c^{-3/4}\to 0$.
From \eqref{VIII3}: $\lambda_n^{(c)}\Phi_n^{(c)}
\to\lambda_n^{(\infty)}\Phi_n^{(\infty)}$ in norm.
Combining with \eqref{eq:HSOT-norm-conv}:
$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\medskip
\noindent\textit{Remark on logical order.}
The step \eqref{eq:HSOT-norm-conv} uses
\emph{norm convergence} of $\Phi_n^{(c)}$
(Thm.~\ref{thm:phi-conv}), which is conditional.
There is no interchange of limits here:
the limit $c\to\infty$ is taken after
applying the bounded operator $\HSOT$
to the norm-convergent sequence.
\end{proof}

\section{Conditional ONS}
\label{sec:ons}
%=============================================================

\begin{proposition}[Conditional orthonormal system]
\label{prop:ons}
\emph{Conditional on Thm.~\ref{thm:phi-conv}.}
\begin{enumerate}[\rm(a)]
\item $\|\Phi_n^{(\infty)}\|=1$.
\item $\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}}=0$,
  $n\ne m$.
\item $\HSOT\Phi_n^{(\infty)}
  =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{enumerate}
Completeness (making this an ONB) requires
Hyp.~\ref{hyp:decay}.
\end{proposition}

\begin{proof}
(a)~Norm convergence preserves unit norm.
(b)~$\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}}=0$ and norm conv.
(c)~Lem.~\ref{lem:weak-eigenrelation}.
\end{proof}

\section{Conditional: Essential Self-Adjointness}
\label{sec:esa}
%=============================================================

\begin{theorem}[Completeness implies ESSA]
\label{thm:esa}
\emph{Conditional on Thm.~\ref{thm:phi-conv}
and Hyp.~\ref{hyp:decay}.}

If $\{\Phi_n^{(\infty)}\}$ is complete in $L^2$
(Hyp.~\ref{hyp:decay}), then $\Hspec$ is essentially
self-adjoint.
\end{theorem}

\begin{proof}
Let $f\in\ker(\Hspec^*\pm i)$.
For each $n$:
$(\lambda_n^{(\infty)}\pm i)\ip{f}{\Phi_n^{(\infty)}}=0$,
hence $\ip{f}{\Phi_n^{(\infty)}}=0$.
By Hyp.~\ref{hyp:decay}: $f=0$.
So $n_\pm(\Hspec)=0$, i.e.\ $\Hspec$ is essentially
self-adjoint.
\end{proof}

\begin{remark}[Converse direction is open]
\label{rem:essa-one-direction}
Thm.~\ref{thm:esa} proves only the implication
completeness $\Rightarrow$ ESSA.
The converse, ESSA $\Rightarrow$ completeness,
would require showing that every vector orthogonal
to $\{\Phi_n^{(\infty)}\}$ lies in a deficiency space
$\ker(\Hspec^*\pm i)$, which in turn requires
an explicit description of $\Dom(\Hspec^*)$
beyond what is available here.
The converse is therefore left as an open problem
(Prob.~\ref{prob:essa-converse}).
\end{remark}

\begin{corollary}[ESSA under completeness]
\label{cor:esa}
\emph{Conditional on Thm.~\ref{thm:phi-conv}
and Hyp.~\ref{hyp:decay}.}
$\overline{\Hspec}=\Hspec^*$.
\end{corollary}

\section{Conditional: Operator Norm Rate}
\label{sec:opnorm}
%=============================================================

\begin{theorem}[Operator norm rate]
\label{thm:opnorm}
\emph{Conditional on Hyp.~\ref{hyp:uniform-alignment}.}
$\|\Hc-\Hspec\|\le C_\kappa c^{-1/4}$.
\end{theorem}

\begin{remark}[Operator-norm convergence does not
automatically follow from uniform alignment]
\label{rem:opnorm-bridge}
Hyp.~\ref{hyp:uniform-alignment} controls
$\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|$
on the truncated index range $n\le\kappa c$.
Operator-norm convergence $\|\Hc-\Hspec\|\to 0$
would follow if in addition:
\begin{enumerate}[\rm(i)]
\item the contribution from $n>\kappa c$ is negligible
  uniformly in $c$; and
\item $\{\Phi_n^{(\infty)}\}$ forms a Riesz basis
  (or has uniform frame bounds) in $L^2$.
\end{enumerate}
Neither (i) nor (ii) is established here.
Consequently, the implication
Hyp.~\ref{hyp:uniform-alignment}
$\Rightarrow$ Hyp.~\ref{hyp:norm-resolvent}
(Rem.~\ref{rem:dep}) is conjectural,
not a theorem.
\end{remark}

\section{Spectral Connection to Zeta: A Meta-Remark}
\label{sec:rh-remark}
%=============================================================

\begin{remark}[What would be needed for a zeta theorem]
\label{rem:rh-meta}
A genuine spectral theorem connecting $\Hspec$ to
Riemann zeta zeros would require, at minimum:
\begin{enumerate}[(a)]
\item Proof of Hyp.~\ref{hyp:decay} (completeness):
  currently open.
\item Proof of Hyp.~\ref{hyp:norm-resolvent}:
  currently open.
\item A spectral \emph{isomorphism} (not merely
  approximate-spectrum inclusion) between $\Hspec$
  and a trace-class operator whose eigenvalues
  are in bijection with nontrivial zeta zeros.
\item A Weil-type or explicit-formula trace identity
  linking spectral data of $\Hspec$ to $\zeta$;
  this is not provided by CCM~\cite{CCM2025} alone.
\end{enumerate}
The result of Paper~VIII gives
$\rho\in\sigma_{\mathrm{app}}(\Hspec)$
(conditional on CCM); this is an inclusion,
not a bijection, and does not imply
$\mathrm{Re}(\rho)=\tfrac12$ without (c) and (d).
Any claim of the form
``$\Hspec$ proves the Riemann Hypothesis''
is not supported by the present paper.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%=============================================================

\begin{problem}[Norm-resolvent convergence]
\label{prob:norm-resolvent}
Prove Hyp.~\ref{hyp:norm-resolvent}.
\end{problem}

\begin{problem}[Discrete spectrum of $\HSOT$]
\label{prob:HSOT-discrete}
Prove Hyp.~\ref{hyp:HSOT-discrete}.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Prove Hyp.~\ref{hyp:decay}.
\end{problem}

\begin{problem}[Uniform alignment]
\label{prob:uniform-alignment}
Prove Hyp.~\ref{hyp:uniform-alignment}.
\end{problem}

\begin{problem}[Converse of ESSA]
\label{prob:essa-converse}
Does ESSA of $\Hspec$ imply completeness of
$\{\Phi_n^{(\infty)}\}$?
This requires an explicit description of
$\Dom(\Hspec^*)$.
\end{problem}

\begin{problem}[Operator identification]
\label{prob:identification}
Prove $\overline{\Hspec}=\HSOT$.
\end{problem}

\begin{problem}[Riesz-basis property]
\label{prob:riesz}
Do the vectors $\{\Phi_n^{(\infty)}\}$ form a Riesz
basis in $L^2$?
This would close the gap in Rem.~\ref{rem:opnorm-bridge}.
\end{problem}
