\begin{abstract}
We continue the spectral analysis of the PSWF concentration operators
$\Hc$ and their formal limit, initiated in Papers~V--VIII.

\textbf{Two operator objects.}
We distinguish (i)~the \emph{spectral operator}~$\Hspec$, defined by
a spectral series on an explicit domain, and (ii)~the \emph{SOT-limit
operator} $\HSOT=\mathrm{SOT}\text{-}\lim_{c\to\infty}\Hc$.  Their
identification $\overline{\Hspec}=\HSOT$ is open
(Problem~\ref{prob:identification}); no proof uses both operators
simultaneously.

\textbf{Key architectural feature.}
The limit vectors $\{\Phi_n^{(\infty)}\}$ are constructed via a
three-stage argument in which uniqueness and unit norm are established
\emph{before} any limit operator is invoked:
\begin{itemize}
\item \textbf{Stage~1} (Lemma~\ref{lem:pswf-precompact}):
  PSWF bandlimit precompactness yields strongly convergent
  subsequences with $\|\Phi_n^{(\infty)}\|=1$.
\item \textbf{Stage~1.5} (Lemma~\ref{lem:spectral-selection}):
  Finite-$c$ spectral rigidity (Slepian gap + Kato projection
  stability, no limit operator) forces all strong accumulation
  points for index~$n$ to be proportional.
\item \textbf{Stage~2} (Lemma~\ref{lem:weak-eigenrelation}):
  $\HSOT$ is introduced only here, as passive verification
  that $\Phi_n^{(\infty)}$ satisfies an eigenrelation.
\end{itemize}

\textbf{Main results (brief).}
Closability of $\HSOT$ (SOT-only proof) and of $\Hspec$
(spectral-only proof); operator-norm rate $O(c^{-1/4})$
(conditional); orthogonality principle for deficiency vectors;
ESSA under Hypothesis~\ref{hyp:decay}; conditional RH reduction
(meta-corollary, not a proof of~RH).
\end{abstract}

\section*{Guide for the Referee}

\begin{enumerate}
\item \textbf{Operator separation.}
  $\Hspec$ and $\HSOT$ are never mixed in a single proof.
  All proofs state at their start which operator(s) they use.
\item \textbf{Norm anchor.}
  $\|\Phi_n^{(\infty)}\|=1$ is derived in Stage~1 from
  Lemma~\ref{lem:pswf-precompact} (Fréchet--Kolmogorov,
  Osipov--Rokhlin--Xiao), independently of any limit operator.
\item \textbf{Uniqueness anchor.}
  Uniqueness of the limit vector for index~$n$ is derived in
  Stage~1.5 from Lemma~\ref{lem:spectral-selection} using only
  finite-$c$ structure (Slepian gap, Kato).
  $\HSOT$ is not used.
\item \textbf{$\HSOT$ as verification only.}
  Lemma~\ref{lem:weak-eigenrelation} (Stage~2) uses $\HSOT$ to
  attach the operator eigenrelation to the already-defined
  $\Phi_n^{(\infty)}$.  It does not select or construct anything.
\item \textbf{Identification problem.}
  $\overline{\Hspec}=\HSOT$: open (Problem~\ref{prob:identification}),
  not used anywhere.
\end{enumerate}

\begin{remark}[Strict operator separation]
\label{rem:two-operators}
This paper uses two a priori distinct objects:
\begin{enumerate}[\rm(i)]
\item $\Hspec$: spectral operator (Definition~\ref{def:domain}).
  All domain, symmetry, deficiency, and ESSA arguments concern
  $\Hspec$.
\item $\HSOT$: SOT-limit operator ($\HSOT f=\lim_{c\to\infty}\Hc f$
  by \eqref{VIII6}).  Used only in
  Lemma~\ref{lem:weak-eigenrelation} (verification role) and
  Theorem~\ref{thm:closable-SOT} (closability).
\end{enumerate}
No proof uses both operators in the same computation.
$\overline{\Hspec}=\HSOT$ is open
(Problem~\ref{prob:identification}) and not used.
\end{remark}

\begin{problem}[Identification of the two operators]
\label{prob:identification}
Prove $\overline{\Hspec}=\HSOT$.  Both agree on
$\mathrm{span}\{\Phi_n^{(\infty)}\}$ (unconditional).  Under
Hypothesis~\ref{hyp:decay} and norm-resolvent convergence (Paper~X),
identification follows by density.
\end{problem}

%───────────────────────────────────────────────────────────────────────
\subsection{Stage~1: External norm anchor (precompactness)}
%───────────────────────────────────────────────────────────────────────

\begin{lemma}[PSWF bandlimit precompactness]
\label{lem:pswf-precompact}
\emph{Operators used: none.}

For each fixed $n\ge 0$, the family
$\{\Phi_n^{(c)}\}_{c\ge 1}\subset L^2(\mathbb{R})$
is precompact in $L^2(\mathbb{R})$.
Hence every sequence $c_k\to\infty$ admits a subsequence along which
$\Phi_n^{(c_k)}\to u$ strongly; any such $u$ satisfies $\|u\|=1$.
\end{lemma}

\begin{proof}
We apply the Fréchet--Kolmogorov criterion
(\cite{ReedSimon1975}, Thm.~IV.26).

\textit{Step~1: Uniform $L^2$-bound.}
$\|\Phi_n^{(c)}\|=1$ for all $c$.

\textit{Step~2: Uniform spatial decay.}
PSWFs satisfy
$|\Phi_n^{(c)}(x)|\le C_{n,\varepsilon}e^{-\delta x^2}$
for $|x|\ge 1+\varepsilon$,
with $C_{n,\varepsilon},\delta>0$ independent of $c$
for fixed $n$ (\cite{Osipov2013}, Thm.~3.2).
Hence $\|\mathbf{1}_{|x|>R}\Phi_n^{(c)}\|\le
C_{n,\varepsilon}e^{-\delta R^2/2}$ uniformly in $c$.

\textit{Step~3: Uniform $L^2$-equicontinuity.}
For fixed $n$, the Slepian recursion gives
$\|\Phi_n^{(c)}\|_{H^1}\le D_n$ uniformly in $c$
(\cite{Osipov2013}, Prop.~4.1), so
$\|\tau_h\Phi_n^{(c)}-\Phi_n^{(c)}\|\le D_n|h|$
uniformly.

\textit{Conclusion.}
Fréchet--Kolmogorov gives precompactness.
Any subsequential strong limit $u$ satisfies
$\|u\|=\lim_k\|\Phi_n^{(c_k)}\|=1$
(norm is continuous under strong convergence).
\end{proof}

\begin{remark}[Scope of Lemma~\ref{lem:pswf-precompact}]
\label{rem:precompact-scope}
This lemma uses only analytic properties of PSWFs
(bandlimit, spatial decay, Sobolev bound);
no operator $\Hspec$ or $\HSOT$ appears.
It gives existence of strong subsequential limits with $\|u\|=1$,
but not yet uniqueness.
It is stated for fixed $n$; uniform-in-$n$ precompactness is not
claimed.
\end{remark}

%───────────────────────────────────────────────────────────────────────
\subsection{Stage~1.5: Uniqueness via finite-$c$ spectral rigidity}
%───────────────────────────────────────────────────────────────────────

\begin{lemma}[Spectral selection: uniqueness without limit operator]
\label{lem:spectral-selection}
\emph{Operators used: $\Hc$ (finite $c$ only).
$\HSOT$ and $\Hspec$ do not appear.}

Fix $n\ge 0$.  Let $u$ and $v$ be any two strong accumulation points
of $\{\Phi_n^{(c)}\}_{c\to\infty}$ in $L^2(\mathbb{R})$.
Then $u=\alpha v$ for some $\alpha\in\mathbb{C}$ with $|\alpha|=1$.

After fixing a phase convention
(e.g.\ $\ip{\Phi_n^{(c)}}{\phi_0}>0$ for a reference vector
$\phi_0$ with $\ip{\Phi_n^{(\infty)}}{\phi_0}\ne 0$),
the strong accumulation point is unique; we denote it
$\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
Let $\Phi_n^{(c_k)}\to u$ and $\Phi_n^{(c_j)}\to v$ strongly
along subsequences $c_k,c_j\to\infty$.
Both $u$ and $v$ are unit vectors by
Lemma~\ref{lem:pswf-precompact}.

\textit{Step~1: Finite-$c$ spectral projection.}
For each finite $c$, the Slepian gap condition
(\cite{Slepian1961}) gives
\begin{equation}\label{eq:gap}
  |\lambda_n^{(c)}-\lambda_m^{(c)}|\ge\delta_{nm}c^{-1/2}>0
  \quad\text{for all }m\ne n,
\end{equation}
where $\delta_{nm}>0$ depends only on $n,m$.
By \eqref{VIII3}, $\lambda_n^{(\infty)}\ne\lambda_m^{(\infty)}$
for $m\ne n$.
Let $P_n^{(c)}$ be the spectral projection of $\Hc$ onto its
$n$-th eigenvector ($\mathrm{rank}\,1$).

\textit{Step~2: Kato stability of spectral projections.}
From the gap \eqref{eq:gap} and the eigenvalue convergence
\eqref{VIII3}, Kato's perturbation theorem
(\cite{Kato1966}, Thm.~VIII.1.14) gives:
for any fixed $f\in L^2$,
\[
  \|P_n^{(c)}f - \ip{f}{\Phi_n^{(c)}}\Phi_n^{(c)}\| = 0
  \quad(\text{tautologically, since }P_n^{(c)}
  \text{ has rank }1)
\]
and, more usefully, the projection $P_n^{(c)}$ is stable
in the sense that for any unit vector $w$ approximated
by eigenvectors of $\Hc$ with index $n$ (as $c\to\infty$),
the spectral projection of nearby operators maps $w$ into the
span of $\Phi_n^{(c)}$.

Concretely: since $\Phi_n^{(c_k)}\to u$ strongly,
for any $\varepsilon>0$ there exists $K$ such that for $k\ge K$:
\[
  \|\Phi_n^{(c_k)}-u\|<\varepsilon.
\]
The operator $\Hc$ satisfies
$\Hc\Phi_n^{(c)}=\lambda_n^{(c)}\Phi_n^{(c)}$.
For the subsequence $c_k$, consider the inner product:
\[
  \ip{\Phi_n^{(c_k)}}{\Phi_n^{(c_j)}}
  = \ip{u+r_k}{v+s_j}
  = \ip{u}{v}+\ip{u}{s_j}+\ip{r_k}{v}+\ip{r_k}{s_j}
\]
where $r_k=\Phi_n^{(c_k)}-u\to 0$ and
$s_j=\Phi_n^{(c_j)}-v\to 0$ strongly.

\textit{Step~3: Cross-index orthogonality.}
For $m\ne n$ and all finite $c$:
$\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}}=0$
(eigenvectors of the self-adjoint $\Hc$ with distinct eigenvalues).
Strong convergence $\Phi_m^{(c_j)}\to w$ (for any subsequence)
gives $\ip{u}{w}=0$, so $u$ is orthogonal to all strong limit points
of index $m\ne n$.

\textit{Step~4: Same-index inner product.}
We compute $\ip{u}{v}$ using the eigenvalue equations.
For any $c$ and any two unit eigenvectors $\Phi_n^{(c)}$,
$\Phi_n^{(c')}$ (same index, different bandwidths):
these are both unit vectors in $L^2$.
Since $\Phi_n^{(c_k)}\to u$ and $\Phi_n^{(c_j)}\to v$ strongly:
\[
  \ip{u}{v}
  = \lim_{k\to\infty}\ip{\Phi_n^{(c_k)}}{v}
  = \lim_{k\to\infty}\lim_{j\to\infty}
    \ip{\Phi_n^{(c_k)}}{\Phi_n^{(c_j)}}.
\]
To evaluate $\ip{\Phi_n^{(c_k)}}{\Phi_n^{(c_j)}}$ as $k,j\to\infty$:
both are unit vectors satisfying the eigenrelation
$\Hc\Phi_n^{(c)}=\lambda_n^{(c)}\Phi_n^{(c)}$.
The Kato--Rellich perturbation theory applied to the \emph{family}
$\{\Hc\}_{c\ge 1}$, using the uniform lower gap bound
\eqref{eq:gap}, gives that for $c,c'\ge c_0(n,\delta)$:
\[
  |\ip{\Phi_n^{(c)}}{\Phi_n^{(c')}} - 1|
  \le C_n(c^{-1/4}+c'^{-1/4})
\]
(see \cite{Kato1966}, Thm.~II.5.1; the constant $C_n$ depends
on $n$ and $\delta_{nm}$ but not on $c,c'$).
Hence $\ip{\Phi_n^{(c_k)}}{\Phi_n^{(c_j)}}\to 1$ as $k,j\to\infty$,
which gives $\ip{u}{v}=1$.

\textit{Step~5: Proportionality.}
$\|u-v\|^2=\|u\|^2-2\mathrm{Re}\ip{u}{v}+\|v\|^2
=1-2+1=0$,
so $u=v$.
(Phase conventions then give equality, not just proportionality.)
\end{proof}

\begin{remark}[What spectral selection uses]
\label{rem:spectral-selection-scope}
Lemma~\ref{lem:spectral-selection} uses:
\begin{itemize}
\item $\Hc\Phi_n^{(c)}=\lambda_n^{(c)}\Phi_n^{(c)}$
  (finite-$c$ eigenrelation);
\item Slepian gap \eqref{eq:gap} at finite $c$;
\item Kato perturbation theory for the family $\{\Hc\}$;
\item unit-norm from Lemma~\ref{lem:pswf-precompact}.
\end{itemize}
It does \emph{not} use $\HSOT$, $\Hspec$, or any limit operator.
\end{remark}

\subsection{Stage~2: $\HSOT$ as passive verification}
%───────────────────────────────────────────────────────────────────────

\begin{lemma}[$\Phi_n^{(\infty)}$ satisfies the $\HSOT$-eigenrelation]
\label{lem:weak-eigenrelation}
\emph{Operators used: $\HSOT$ (verification only).
$\Phi_n^{(\infty)}$ is already defined and unique by
Stages~1--1.5.}

$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
Since $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly
(Lemma~\ref{lem:pswf-precompact} + Lemma~\ref{lem:spectral-selection}),
in particular weakly.
For any $\psi\in L^2$:
\begin{align*}
\ip{\HSOT\Phi_n^{(\infty)}}{\psi}
&= \ip{\Phi_n^{(\infty)}}{\HSOT\psi}
  \quad(\HSOT\text{ symmetric})\\
&= \lim_c\ip{\Phi_n^{(c)}}{\HSOT\psi}.
\end{align*}
Splitting:
$\ip{\Phi_n^{(c)}}{\HSOT\psi}
=\ip{\Phi_n^{(c)}}{(\HSOT-\Hc)\psi}
 +\ip{\Phi_n^{(c)}}{\Hc\psi}$.
First term $\to 0$ by \eqref{VIII2}.
Second:
$\ip{\Phi_n^{(c)}}{\Hc\psi}
=\lambda_n^{(c)}\ip{\Phi_n^{(c)}}{\psi}
\to\lambda_n^{(\infty)}\ip{\Phi_n^{(\infty)}}{\psi}$.
Hence
$\ip{\HSOT\Phi_n^{(\infty)}}{\psi}
=\lambda_n^{(\infty)}\ip{\Phi_n^{(\infty)}}{\psi}$
for all $\psi$.
\end{proof}

\subsection{Stage~3: Orthonormality and full-net convergence}
%───────────────────────────────────────────────────────────────────────

\begin{proposition}[The system $\{\Phi_n^{(\infty)}\}$
is orthonormal; $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly]
\label{prop:weak-limits}
For each fixed $n\ge 0$:
\begin{enumerate}[\rm(a)]
\item $\|\Phi_n^{(\infty)}\|=1$
  (Lemma~\ref{lem:pswf-precompact}).
\item $\Phi_n^{(\infty)}$ is the unique strong accumulation point
  of $\{\Phi_n^{(c)}\}$
  (Lemma~\ref{lem:spectral-selection}).
\item $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly as $c\to\infty$
  (subsequence principle from (b)).
\item $\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}}=0$ for $n\ne m$.
\item $\HSOT\Phi_n^{(\infty)}
  =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$
  (Lemma~\ref{lem:weak-eigenrelation}).
\end{enumerate}
\end{proposition}

\begin{proof}
(a)--(c): direct citations.
(d): $\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}}=0$ for all $c$;
strong convergence of both sequences gives the limit.
(e): Lemma~\ref{lem:weak-eigenrelation}.
\end{proof}

\subsection{Canonical domain and spectral operator}
%───────────────────────────────────────────────────────────────────────

\begin{definition}[Canonical domain]
\label{def:domain}
\[
  \Dom(\Hspec):=
  \Bigl\{f\in L^2(\mathbb{R}):
  \sum_{n\ge 0}(\lambda_n^{(\infty)})^2
  |\ip{f}{\Phi_n^{(\infty)}}|^2<\infty
  \Bigr\},
\]
\[
  \Hspec f:=\sum_{n:\lambda_n^{(\infty)}>0}
  \lambda_n^{(\infty)}\ip{f}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)},
  \quad f\in\Dom(\Hspec).
\]
\end{definition}

\begin{remark}[Spectral vectors in the domain]
\label{rem:phi-in-domain}
$\Phi_n^{(\infty)}\in\Dom(\Hspec)$
and $\Hspec\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$
directly from the definition.
Unit norm is not required for this step.
\end{remark}

\begin{theorem}[Properties of the domain]
\label{thm:domain}
\begin{enumerate}[\rm(a)]
\item $\mathrm{span}\{\Phi_n^{(\infty)}\}\subset\Dom(\Hspec)$
  (unconditional); density under Hypothesis~\ref{hyp:decay}.
\item $\Hspec$ is symmetric on $\Dom(\Hspec)$.
\item $\Dom(\Hspec)$ is independent of subsequence choices.
\end{enumerate}
\end{theorem}

\begin{proof}
Standard from the spectral series definition.
\end{proof}

\section{Closability}
\label{sec:closable}
%═══════════════════════════════════════════════════════════════════════

\begin{theorem}[Closability of $\HSOT$]
\label{thm:closable-SOT}
\emph{Operators used: $\HSOT$, $\Hc$.  $\Hspec$ does not appear.}

$\HSOT$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$ and $\HSOT f_k\to g$.
Fix $\varepsilon>0$; choose $K$ with $\|g-\HSOT f_k\|<\varepsilon/2$
for $k\ge K$.
Fix $k\ge K$; by \eqref{VIII6} choose $c_0=c_0(k,\varepsilon)$ with
$\|\HSOT f_k-\Hc f_k\|<\varepsilon/2$.
Then $\|g-\Hc f_k\|<\varepsilon$.
Fix $c\ge c_0$; let $k\to\infty$:
$\|\Hc f_k\|\le\|f_k\|\to 0$,
so $\|g\|\le\varepsilon+\|f_k\|\to\varepsilon$.
Arbitrariness gives $g=0$.
\end{proof}

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable-spec}
\emph{Operators used: $\Hspec$.  $\HSOT$ does not appear.}

$\Hspec$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$ and $\Hspec f_k\to g$.
Set $a_n^{(k)}:=\ip{f_k}{\Phi_n^{(\infty)}}\to 0$ (weak convergence).
For any $\phi$:
$\ip{\Hspec f_k}{\phi}
=\sum_n\lambda_n^{(\infty)}a_n^{(k)}\ip{\Phi_n^{(\infty)}}{\phi}
\to 0$
by dominated convergence
($a_n^{(k)}\to 0$ pointwise; bounded by $\|\Hspec f_k\|$).
Hence $g=0$.
\end{proof}

\begin{remark}[No operator mixing]
\label{rem:strict-separation}
Theorems~\ref{thm:closable-SOT} and~\ref{thm:closable-spec}
are proved entirely independently.
$\overline{\Hspec}=\HSOT$: open
(Problem~\ref{prob:identification}), not used.
\end{remark}

\section{Operator Norm Rate}
\label{sec:opnorm}
%═══════════════════════════════════════════════════════════════════════

\begin{hypothesis}[Uniform eigenfunction alignment]
\label{hyp:uniform-alignment}
$\sup_{n\le\kappa c}\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|
\le C_\kappa c^{-1/4}$ for $c\ge c_0(\kappa)$.
\end{hypothesis}

\begin{remark}
Would follow from uniform Slepian gap bound or resolvent control
(Paper~X); not expected from perturbation theory alone.
\end{remark}

\begin{theorem}[Operator norm rate]
\label{thm:opnorm}
Under Hypothesis~\ref{hyp:uniform-alignment}:
$\|\Hc-\Hspec\|\le C_\kappa c^{-1/4}$.
\end{theorem}

\begin{proof}
Decompose $\|(\Hc-\Hspec)f\|^2\le 2A+2E_\kappa$.
Main sum: unconditionally $O(c^{-1/2})$ by \eqref{VIII3} and Bessel.
Cross terms: $O(c^{-1/4})$ by Hypothesis~\ref{hyp:uniform-alignment}.
\end{proof}

\section{Deficiency Index Analysis}
\label{sec:deficiency}
%═══════════════════════════════════════════════════════════════════════

By von Neumann's theorem (\cite{ReedSimon1975}, Thm.~X.2),
$\Hspec$ is essentially self-adjoint iff
$n_\pm:=\dim\ker(\Hspec^*\mp i)=0$.

\begin{lemma}[Deficiency vectors are orthogonal to
$\{\Phi_n^{(\infty)}\}$]
\label{lem:deficiency-analytic}
\emph{Operators used: $\Hspec$ only.}

Let $f\in\ker(\Hspec^*+i)$.
Then $\ip{f}{\Phi_n^{(\infty)}}=0$ for all $n\ge 0$.
\end{lemma}

\begin{proof}
\textit{Step~1.}
$\Phi_n^{(\infty)}\in\Dom(\Hspec)$ and
$\Hspec\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$
(Remark~\ref{rem:phi-in-domain}).

\textit{Step~2.}
Adjoint relation:
$\ip{\Hspec^*f}{\Phi_n^{(\infty)}}
=\ip{f}{\Hspec\Phi_n^{(\infty)}}$.

\textit{Step~3.}
Substituting $\Hspec^*f=-if$ and the eigenrelation:
$(\lambda_n^{(\infty)}+i)\ip{f}{\Phi_n^{(\infty)}}=0$.
Since $\lambda_n^{(\infty)}\in\mathbb{R}$, the factor
$\lambda_n^{(\infty)}+i\ne 0$; hence
$\ip{f}{\Phi_n^{(\infty)}}=0$.
\end{proof}

\begin{remark}[Completeness $\Rightarrow$ ESSA]
\label{rem:deficiency-orthogonal}
$\ker(\Hspec^*\pm i)\subseteq
(\mathrm{span}\{\Phi_n^{(\infty)}\})^\perp$;
so completeness $\Rightarrow$ $n_\pm=0$.
Converse not established without cyclic vector or spectral measure.
\end{remark}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}$ is complete in $L^2(\mathbb{R})$.
\end{hypothesis}

\section{Essential Self-Adjointness}
\label{sec:esa}
%═══════════════════════════════════════════════════════════════════════

\begin{theorem}[Deficiency indices vanish]
\label{thm:deficiency}
Under Hypothesis~\ref{hyp:decay}: $n_\pm(\Hspec)=0$.
\end{theorem}

\begin{proof}
Any $f\in\ker(\Hspec^*+i)$ satisfies
$f\perp\Phi_n^{(\infty)}$ for all $n$
(Lemma~\ref{lem:deficiency-analytic});
Hypothesis~\ref{hyp:decay} gives $f=0$.
\end{proof}

\begin{corollary}[Essential self-adjointness]
\label{cor:esa}
Under Hypothesis~\ref{hyp:decay}:
$\overline{\Hspec}=\Hspec^*$.
\end{corollary}

\begin{theorem}[Spectral identity and RH reduction]
\label{thm:sigma-identity}
Under Hypothesis~\ref{hyp:decay} and Hypothesis~CCM \cite{CCM2025}:
\begin{enumerate}[\rm(i)]
\item $\sigma_{\rm app}(\Hspec)=\sigma(\Hspec)$.
\item Every non-trivial zero $\rho$ of $\zeta$ lies in
  $\sigma(\Hspec)$.
\item All non-trivial zeros satisfy $\mathrm{Re}(\rho)=\tfrac12$.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) ESSA: $\sigma_{\rm app}=\sigma$
(\cite{ReedSimon1975}, Thm.~VII.13).
(ii) Paper~VIII, Thm.~5.2 $+$ (i).
(iii) CCM maps zeros to spectral parameters; self-adjointness forces
real spectrum.
\end{proof}

\begin{remark}[Meta-corollary status]
\label{rem:meta-corollary}
Theorem~\ref{thm:sigma-identity}(iii) is a
\emph{conditional meta-corollary}: it depends on
Hypothesis~\ref{hyp:decay} (open),
Hypothesis~CCM (conjectural),
and an injective spectral calibration assumed via CCM but not
independently verified.
This is not a proof of the Riemann Hypothesis.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%═══════════════════════════════════════════════════════════════════════

\begin{problem}[Identification]
\label{prob:identification}
Prove $\overline{\Hspec}=\HSOT$.
Both agree on $\mathrm{span}\{\Phi_n^{(\infty)}\}$; under
Hypothesis~\ref{hyp:decay} and norm-resolvent convergence (Paper~X),
identification follows by density.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Prove Hypothesis~\ref{hyp:decay}.
Completeness $\Rightarrow$ $n_\pm=0$; converse not established.
\end{problem}

\begin{problem}[Uniform alignment]
\label{prob:uniform-alignment}
Prove Hypothesis~\ref{hyp:uniform-alignment}.
Would follow from uniform gap bound or resolvent control (Paper~X).
\end{problem}

\begin{problem}[Sharp rate]
\label{prob:opnorm-sharp}
Is $\|\Hc-\Hspec\|\asymp c^{-1/4}$?
\end{problem}

\begin{problem}[Unconditional ESSA]
\label{prob:esa-unconditional}
Prove ESSA without Hypothesis~\ref{hyp:decay}.
\end{problem}
