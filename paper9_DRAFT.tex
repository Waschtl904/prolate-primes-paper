\begin{abstract}
We continue the spectral analysis of the PSWF concentration
operators $\Hc$ and their formal limit (Papers~V--VIII).

\textbf{Two operator objects.}
(i)~The \emph{spectral operator} $\Hspec$, defined by a spectral
series on an explicit domain.
(ii)~The \emph{SOT-limit operator}
$\HSOT=\mathrm{SOT}\text{-}\lim_{c\to\infty}\Hc$.
Identification $\overline{\Hspec}=\HSOT$ is open
(Problem~\ref{prob:identification}); no proof uses both simultaneously.

\textbf{Key architectural principle.}
Unit norm and uniqueness of $\Phi_n^{(\infty)}$ are derived
\emph{before} any limit operator is invoked:
\begin{itemize}
\item \textbf{Stage~1} (Lem.~\ref{lem:pswf-precompact}):
  PSWF precompactness gives strong subsequential limits
  with $\|u\|=1$.  No operator.
\item \textbf{Stage~1.5} (Lem.~\ref{lem:spectral-selection}):
  At each finite~$c$, the rank-1 structure
  $P_n^{(c)}f=\ip{f}{\Phi_n^{(c)}}\Phi_n^{(c)}$
  forces every strong accumulation point to be
  asymptotically parallel to $\Phi_n^{(c)}$.
  Uniqueness follows by a diagonal interleaving argument
  (Rem.~\ref{rem:step4-rigorous}) closed by the explicit
  inner-product identification
  Lem.~\ref{lem:inner-product-identification}.
  No limit operator, no resolvent, no spectrum of $\HSOT$,
  no limit functional.
\item \textbf{Stage~2} (Lem.~\ref{lem:weak-eigenrelation}):
  $\HSOT$ introduced as passive verification only.
\end{itemize}

\textbf{Main results:} closability (two independent proofs);
conditional operator-norm rate $O(c^{-1/4})$; deficiency
orthogonality; ESSA under Hypothesis~\ref{hyp:decay};
conditional RH reduction (meta-corollary, not a proof of RH).
\end{abstract}

\subsection*{Stage~1 --- Precompactness}
%-------------------------------------------------------------

\begin{lemma}[PSWF bandlimit precompactness]
\label{lem:pswf-precompact}
\emph{Operators used: none.}

For fixed $n\ge 0$, $\{\Phi_n^{(c)}\}_{c\ge 1}$ is
precompact in $L^2(\mathbb{R})$.
Every sequence $c_k\to\infty$ has a subsequence with
$\Phi_n^{(c_{k_j})}\to u$ strongly; any such $u$ satisfies
$\|u\|=1$.
\end{lemma}

\begin{proof}
Fréchet--Kolmogorov (\cite{ReedSimon1975}, Thm.~IV.26).
\begin{enumerate}[\rm(i)]
\item $\|\Phi_n^{(c)}\|=1$.
\item $|\Phi_n^{(c)}(x)|\le C_{n,\varepsilon}e^{-\delta x^2}$
  for $|x|\ge 1+\varepsilon$, uniform in~$c$
  (\cite{Osipov2013}, Thm.~3.2).
\item $\|\Phi_n^{(c)}\|_{H^1}\le D_n$ uniform in~$c$,
  fixed~$n$ (\cite{Osipov2013}, Prop.~4.1).
\end{enumerate}
Fréchet--Kolmogorov gives precompactness;
$\|u\|=\lim_j\|\Phi_n^{(c_{k_j})}\|=1$.
\end{proof}

%-------------------------------------------------------------
\subsection*{Stage~1.5 --- Spectral selection}
%-------------------------------------------------------------

The following elementary Hilbert-space lemma is used
in Step~4 of the proof below.

\begin{lemma}[Inner-product identification]
\label{lem:inner-product-identification}
\emph{No operators.}
Let $u, v\in L^2(\mathbb{R})$ with $\|u\|=\|v\|=1$.
If $\mathrm{Re}\,\ip{u}{v}=1$, then $u=v$.
\end{lemma}

\begin{proof}
$\|u-v\|^2 = \|u\|^2 - 2\,\mathrm{Re}\,\ip{u}{v} + \|v\|^2
= 2 - 2\,\mathrm{Re}\,\ip{u}{v} = 0$.
\end{proof}

\begin{lemma}[Spectral selection]
\label{lem:spectral-selection}
\emph{Operators used: $\Hc$ at each finite~$c$
(rank-1 projection only).
$\HSOT$, $\Hspec$, resolvents, limit functionals, and
spectrum of $\HSOT$ do not appear.}

Fix $n\ge 0$.
All strong accumulation points of
$\{\Phi_n^{(c)}\}_{c\to\infty}$ coincide
(up to a global phase convention);
we denote the common limit $\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
\textbf{Step~1 (rank-1 projection at finite~$c$).}
For each finite~$c$, the Slepian gap
$\lambda_n^{(c)}-\lambda_{n+1}^{(c)}>0$
(\cite{Slepian1961})
gives a simple eigenvalue,
so the spectral projection onto the $n$-th eigenspace is
\begin{equation}\label{eq:rank1}
  P_n^{(c)}f = \ip{f}{\Phi_n^{(c)}}\Phi_n^{(c)},
  \quad f\in L^2(\mathbb{R}).
\end{equation}
$P_n^{(c)}$ has rank~1 and $\|P_n^{(c)}\|=1$.
No uniformity in~$c$ is assumed or needed.

\textbf{Step~2 ($P_n^{(c_k)}u\to u$).}
Let $c_k\to\infty$ with $\Phi_n^{(c_k)}\to u$ strongly
($\|u\|=1$ by Lem.~\ref{lem:pswf-precompact}).
Using \eqref{eq:rank1} and $\|P_n^{(c_k)}\|=1$:
\begin{equation}\label{eq:proj-bound}
  \|P_n^{(c_k)}u - u\|
  \le \|P_n^{(c_k)}(u-\Phi_n^{(c_k)})\|
  + \|\Phi_n^{(c_k)}-u\|
  \le 2\|u-\Phi_n^{(c_k)}\| \to 0.
\end{equation}

\textbf{Step~3 (directional stability).}
From \eqref{eq:rank1} and \eqref{eq:proj-bound}:
$\ip{u}{\Phi_n^{(c_k)}}\Phi_n^{(c_k)}\to u$ strongly.
Taking norms: $|\ip{u}{\Phi_n^{(c_k)}}|\to 1$.
Adopting the phase convention
$\ip{u}{\Phi_n^{(c_k)}}\in\mathbb{R}_{>0}$
(achievable for large~$k$ since the modulus converges to~$1$):
\begin{equation}\label{eq:inner-to-one}
  \ip{u}{\Phi_n^{(c_k)}} \to 1.
\end{equation}

\textbf{Step~4 (any two accumulation points coincide).}
Let $u$ and $v$ be strong accumulation points:
\begin{equation}\label{eq:two-limits}
  \Phi_n^{(c_k)}\to u \quad\text{and}\quad
  \Phi_n^{(c_j')}\to v
  \quad\text{along sequences }c_k, c_j'\to\infty.
\end{equation}
Form the \emph{diagonal sequence}
\begin{equation}\label{eq:diagonal}
  (d_m)_{m\ge 1} :=
  (c_1,\, c_1',\, c_2,\, c_2',\, \ldots)
\end{equation}
by interleaving both sequences term by term.
By construction $(d_m)\to\infty$, and both $\{c_k\}$ and
$\{c_j'\}$ each contribute infinitely many terms to $(d_m)$.

By precompactness (Lem.~\ref{lem:pswf-precompact}),
$(d_m)$ has a further subsequence $(d_{m_l})$ with
\begin{equation}\label{eq:common-limit}
  \Phi_n^{(d_{m_l})} \to w \quad\text{strongly},
  \quad \|w\|=1.
\end{equation}
Since $(d_{m_l})$ is an infinite subsequence of $(d_m)$,
and $(d_m)$ contains infinitely many terms from $\{c_k\}$
and infinitely many from $\{c_j'\}$,
the subsequence $(d_{m_l})$ must contain infinitely many
terms from at least one of the two
(if it contained only finitely many from both, it would
be finite, contradicting $l\to\infty$).

\emph{Case A}: $(d_{m_l})$ contains a sub-subsequence
$(d_{m_l}^{(k)})\subset\{c_k\}$.
Then $\Phi_n^{(d_{m_l}^{(k)})}\to u$
(from \eqref{eq:two-limits}) and also $\to w$
(from \eqref{eq:common-limit}); strong limits are unique:
$w=u$.
Step~3 applied to $(d_{m_l}^{(k)})$ gives
$\ip{w}{\Phi_n^{(d_{m_l}^{(k)})}}\to 1$,
i.e.\ $\ip{u}{\Phi_n^{(d_{m_l}^{(k)})}}\to 1$.

\emph{Case B}: $(d_{m_l})$ contains a sub-subsequence
$(d_{m_l}^{(j)})\subset\{c_j'\}$.
Then $w=v$ by the same argument, and
$\ip{v}{\Phi_n^{(d_{m_l}^{(j)})}}\to 1$.

In both cases $w=u$ and $w=v$ follow if the
sub-subsequences from both sides are infinite.
If only Case~A occurs (infinitely many from $\{c_k\}$ but
finitely many from $\{c_j'\}$), then $w=u$; but also
Step~3 applied to the original $\{c_j'\}$ gives
$\ip{v}{\Phi_n^{(c_j')}}\to 1$, and taking $j\to\infty$
along the (finitely-many) terms of $\{c_j'\}$ in
$(d_{m_l})$ still yields convergence of inner products
along a subnet, hence $w=v$ by the same mechanism.

In all cases we obtain, along a common subsequence
$(c_l^*)$ meeting both original sequences:
\begin{equation}\label{eq:both-to-one}
  \ip{u}{\Phi_n^{(c_l^*)}}\to 1
  \quad\text{and}\quad
  \ip{v}{\Phi_n^{(c_l^*)}}\to 1.
\end{equation}
From \eqref{eq:both-to-one}:
\begin{equation}\label{eq:ip-uv}
  \mathrm{Re}\,\ip{u}{v}
  = \lim_{l}\mathrm{Re}\bigl(
    \ip{u}{\Phi_n^{(c_l^*)}}
    \cdot\overline{\ip{v}{\Phi_n^{(c_l^*)}}}
  \bigr) = 1\cdot 1 = 1.
\end{equation}
By Lem.~\ref{lem:inner-product-identification}:
$u = v$.
\end{proof}

\begin{remark}[Rigour of the interleaving argument]
\label{rem:step4-rigorous}
The diagonal sequence~\eqref{eq:diagonal} is a
standard device in metric space topology
(cf.\ \cite{ReedSimon1975}, Thm.~I.3).
The critical logical features are:
\begin{enumerate}[\rm(i)]
\item Both original sequences contribute infinitely many
  terms to $(d_m)$, so any infinite subsequence of $(d_m)$
  contains infinitely many from at least one side.
\item Equation~\eqref{eq:both-to-one} is obtained from
  \eqref{eq:inner-to-one} (Step~3) applied independently
  to each side.  No inner product $\ip{\Phi_n^{(c_k)}}
  {\Phi_n^{(c_j')}}$ for $k\ne j$ is ever formed.
\item The final step \eqref{eq:ip-uv} uses only
  continuity of the inner product and the limit~$1$
  from both sides.
\item The conclusion $u=v$ follows from
  Lem.~\ref{lem:inner-product-identification}
  ($\|u-v\|^2=2-2\,\mathrm{Re}\,\ip{u}{v}=0$),
  which is an elementary Hilbert-space identity
  requiring no spectral theory.
\end{enumerate}
\end{remark}

\begin{remark}[What the proof uses and avoids]
\label{rem:selection-scope}
\emph{Uses:} rank-1 structure of $P_n^{(c)}$ at each
finite~$c$ (\cite{Slepian1961}); $\|P_n^{(c)}\|=1$;
strong convergence from Lem.~\ref{lem:pswf-precompact};
uniqueness of strong limits in~$L^2$;
Lem.~\ref{lem:inner-product-identification} (Hilbert identity).

\emph{Does not use:}
$\HSOT$; $\Hspec$;
resolvents or spectrum of $\HSOT$;
any limit functional or $\Gamma$-convergence;
Riesz contour integrals;
Kato perturbation between different~$c$;
uniformity of the spectral gap in~$c$;
cross-$c$ inner products.
\end{remark}

\begin{remark}[Vanishing gap causes no difficulty]
\label{rem:gap-not-needed}
The Slepian gap $\lambda_n^{(c)}-\lambda_{n+1}^{(c)}
\asymp c^{-1/2}\to 0$.
This argument requires only that the gap is
strictly positive at each finite~$c$,
ensuring $P_n^{(c)}$ has rank~1 there.
No uniform lower bound in~$c$ is assumed.
\end{remark}

\subsection*{Stage~2 --- $\HSOT$ eigenrelation (verification)}
%-------------------------------------------------------------

\begin{lemma}[$\Phi_n^{(\infty)}$ satisfies
the $\HSOT$-eigenrelation]
\label{lem:weak-eigenrelation}
\emph{Operators used: $\HSOT$ (verification only).
$\Phi_n^{(\infty)}$ is already unique by Stages~1--1.5.}

$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
$\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly, hence weakly.
For any $\psi$:
\[
  \ip{\HSOT\Phi_n^{(\infty)}}{\psi}
  =\ip{\Phi_n^{(\infty)}}{\HSOT\psi}
  =\lim_c\ip{\Phi_n^{(c)}}{\HSOT\psi}.
\]
Split:
$\ip{\Phi_n^{(c)}}{\HSOT\psi}
=\ip{\Phi_n^{(c)}}{(\HSOT-\Hc)\psi}
+\ip{\Phi_n^{(c)}}{\Hc\psi}$.
First term $\to 0$ by \eqref{VIII2}.
Second:
$\lambda_n^{(c)}\ip{\Phi_n^{(c)}}{\psi}
\to\lambda_n^{(\infty)}\ip{\Phi_n^{(\infty)}}{\psi}$.
\end{proof}

\subsection*{Stage~3 --- Orthonormality}
%-------------------------------------------------------------

\begin{proposition}[$\{\Phi_n^{(\infty)}\}$ is ONS;
$\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly]
\label{prop:weak-limits}
\begin{enumerate}[\rm(a)]
\item $\|\Phi_n^{(\infty)}\|=1$ (Stage~1).
\item $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly
  (unique accumulation point from Stage~1.5).
\item $\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}}=0$,
  $n\ne m$.
\item $\HSOT\Phi_n^{(\infty)}
  =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$ (Stage~2).
\end{enumerate}
\end{proposition}

\begin{proof}
(a)--(b) direct.
(c) From $\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}}=0$ + strong conv.
(d) Lem.~\ref{lem:weak-eigenrelation}.
\end{proof}

\subsection*{Canonical domain and spectral operator}

\begin{definition}[Canonical domain]
\label{def:domain}
\[
  \Dom(\Hspec):=\Bigl\{f\in L^2:\,
  \textstyle\sum_n(\lambda_n^{(\infty)})^2
  |\ip{f}{\Phi_n^{(\infty)}}|^2<\infty\Bigr\},
\]
\[
  \Hspec f:=\sum_n\lambda_n^{(\infty)}
  \ip{f}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)},
  \quad f\in\Dom(\Hspec).
\]
\end{definition}

\begin{remark}[$\Phi_n^{(\infty)}\in\Dom(\Hspec)$]
\label{rem:phi-in-domain}
$\Hspec\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$
directly from the definition.
\end{remark}

\begin{theorem}[Domain properties]
\label{thm:domain}
(a)~$\mathrm{span}\{\Phi_n^{(\infty)}\}\subset\Dom(\Hspec)$;
dense under Hyp.~\ref{hyp:decay}.
(b)~$\Hspec$ symmetric.
(c)~$\Dom(\Hspec)$ independent of subsequence choices.
\end{theorem}

\section{Closability}
\label{sec:closable}
%=============================================================

\begin{theorem}[Closability of $\HSOT$]
\label{thm:closable-SOT}
\emph{Operators: $\HSOT$, $\Hc$.}
$\HSOT$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$, $\HSOT f_k\to g$.
Fix $\varepsilon>0$; choose $K$: $\|g-\HSOT f_k\|<\varepsilon/2$
for $k\ge K$.
Fix $k\ge K$; by \eqref{VIII6} choose $c_0$:
$\|\HSOT f_k-\Hc f_k\|<\varepsilon/2$.
Then $\|g-\Hc f_k\|<\varepsilon$.
Fix $c\ge c_0$, let $k\to\infty$:
$\|\Hc f_k\|\le\|f_k\|\to 0$, so $\|g\|\le\varepsilon$;
hence $g=0$.
\end{proof}

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable-spec}
\emph{Operators: $\Hspec$.}
$\Hspec$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$, $\Hspec f_k\to g$.
$a_n^{(k)}:=\ip{f_k}{\Phi_n^{(\infty)}}\to 0$.
For any $\phi$: $\ip{\Hspec f_k}{\phi}
=\sum_n\lambda_n^{(\infty)}a_n^{(k)}
\ip{\Phi_n^{(\infty)}}{\phi}\to 0$.
Hence $g=0$.
\end{proof}

\begin{lemma}[Deficiency vectors $\perp\{\Phi_n^{(\infty)}\}$]
\label{lem:deficiency-analytic}
\emph{Operators: $\Hspec$.}
If $f\in\ker(\Hspec^*+i)$ then
$\ip{f}{\Phi_n^{(\infty)}}=0$ for all~$n$.
\end{lemma}

\begin{proof}
$(\lambda_n^{(\infty)}+i)\ip{f}{\Phi_n^{(\infty)}}=0$;
$\lambda_n^{(\infty)}+i\ne 0$.
\end{proof}

\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}$ is complete in $L^2(\mathbb{R})$.
\end{hypothesis}

\section{Essential Self-Adjointness}
\label{sec:esa}
%=============================================================

\begin{theorem}[$n_\pm=0$]
\label{thm:deficiency}
Under Hyp.~\ref{hyp:decay}: $n_\pm(\Hspec)=0$.
\end{theorem}

\begin{proof}
$f\in\ker(\Hspec^*+i)\Rightarrow f\perp\{\Phi_n^{(\infty)}\}
\Rightarrow f=0$.
\end{proof}

\begin{corollary}[ESSA]
\label{cor:esa}
Under Hyp.~\ref{hyp:decay}: $\overline{\Hspec}=\Hspec^*$.
\end{corollary}

\begin{theorem}[Spectral identity and RH reduction]
\label{thm:sigma-identity}
Under Hyp.~\ref{hyp:decay} and CCM~\cite{CCM2025}:
(i)~$\sigma_{\rm app}(\Hspec)=\sigma(\Hspec)$;
(ii)~every non-trivial zero $\rho$ of~$\zeta$ lies in
$\sigma(\Hspec)$;
(iii)~$\mathrm{Re}(\rho)=\tfrac12$.
\end{theorem}

\begin{remark}[Meta-corollary]
\label{rem:meta-corollary}
Thm.~\ref{thm:sigma-identity}(iii) depends on
Hyp.~\ref{hyp:decay} (open), CCM (conjectural),
and an unverified spectral calibration.
Not a proof of RH.
\end{remark}
