\begin{abstract}
We continue the spectral analysis of the PSWF concentration
operators $\Hc$ and their formal limit (Papers~V--VIII).

\textbf{Two operator objects.}
(i)~The \emph{spectral operator} $\Hspec$, defined by a spectral
series on an explicit domain.
(ii)~The \emph{SOT-limit operator}
$\HSOT=\mathrm{SOT}\text{-}\lim_{c\to\infty}\Hc$.
Identification $\overline{\Hspec}=\HSOT$ is open
(Problem~\ref{prob:identification}); no proof uses both simultaneously.

\textbf{Key architectural principle.}
Unit norm and uniqueness of $\Phi_n^{(\infty)}$ are derived
\emph{before} any limit operator is used as a selector:
\begin{itemize}
\item \textbf{Stage~1} (Lem.~\ref{lem:pswf-precompact}):
  PSWF precompactness gives strong subsequential limits
  with $\|u\|=1$.  No operator.
\item \textbf{Stage~1.5a} (Lem.~\ref{lem:resolvent-convergence}):
  SOT convergence of $\Hc$ implies strong resolvent convergence.
\item \textbf{Stage~1.5b} (Lem.~\ref{lem:riesz-projection}):
  Travelling Riesz contour around $\lambda_n^{(c)}$ gives
  $P_n^{(c)}\xrightarrow{\mathrm{SOT}} P_n^{(\infty)}$,
  where $P_n^{(\infty)}$ has rank~1.
\item \textbf{Stage~1.5c} (Cor.~\ref{cor:eigenfunction-convergence}):
  Rank-1 limit projection + unit norm from Stage~1
  $\Rightarrow$ $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly,
  uniquely up to phase.
  No limit operator as selector.
\item \textbf{Stage~2} (Lem.~\ref{lem:weak-eigenrelation}):
  $\HSOT$ attached as passive eigenrelation verification.
\end{itemize}

\textbf{Main results:} closability (two independent proofs);
conditional operator-norm rate $O(c^{-1/4})$; deficiency
orthogonality; ESSA under Hypothesis~\ref{hyp:decay};
conditional RH reduction (meta-corollary, not a proof of RH).
\end{abstract}

\section*{Guide for the Referee}
\begin{enumerate}
\item \textbf{Operator separation.}
  Each proof names its operator(s) in its opening sentence.
  $\Hspec$ and $\HSOT$ never appear in the same computation.
\item \textbf{Norm anchor.}
  $\|\Phi_n^{(\infty)}\|=1$ from strong convergence alone
  (Lem.~\ref{lem:pswf-precompact}); no operator used.
\item \textbf{Uniqueness: three-step resolvent argument.}
  Stage~1.5 proceeds: strong resolvent convergence
  (Lem.~\ref{lem:resolvent-convergence})
  $\to$ SOT convergence of Riesz projections with
  travelling contour (Lem.~\ref{lem:riesz-projection})
  $\to$ rank-1 limit forces unique direction
  (Cor.~\ref{cor:eigenfunction-convergence}).
  The vanishing Slepian gap is handled by letting the contour
  radius shrink as $r_c\asymp c^{-1/2}$
  (Rem.~\ref{rem:travelling-contour}).
\item \textbf{$\HSOT$ as verification only (Stage~2).}
  Lem.~\ref{lem:weak-eigenrelation} attaches the eigenrelation
  to the already-unique $\Phi_n^{(\infty)}$.
\item \textbf{Identification open.}
  Problem~\ref{prob:identification}, not used anywhere.
\end{enumerate}

\subsection*{Stage~1 \texorpdfstring{\textemdash}{---}
Precompactness}
%-------------------------------------------------------------

\begin{lemma}[PSWF bandlimit precompactness]
\label{lem:pswf-precompact}
\emph{Operators used: none.}

For fixed $n\ge 0$, $\{\Phi_n^{(c)}\}_{c\ge 1}$ is precompact
in $L^2(\mathbb{R})$.
Every sequence $c_k\to\infty$ has a subsequence with
$\Phi_n^{(c_{k_j})}\to u$ strongly; any such $u$ satisfies
$\|u\|=1$.
\end{lemma}

\begin{proof}
Fr\'echet--Kolmogorov (\cite{ReedSimon1975}, Thm.~IV.26):
\begin{enumerate}[\rm(i)]
\item $\|\Phi_n^{(c)}\|=1$ (uniform $L^2$-bound).
\item Uniform spatial decay:
  $|\Phi_n^{(c)}(x)|\le C_{n,\varepsilon}e^{-\delta x^2}$
  for $|x|\ge 1+\varepsilon$, uniform in~$c$
  (\cite{Osipov2013}, Thm.~3.2).
\item Uniform $H^1$-bound:
  $\|\Phi_n^{(c)}\|_{H^1}\le D_n$, uniform in~$c$, fixed~$n$
  (\cite{Osipov2013}, Prop.~4.1).
\end{enumerate}
Fr\'echet--Kolmogorov gives precompactness.
Any strong limit $u$ satisfies
$\|u\|=\lim_j\|\Phi_n^{(c_{k_j})}\|=1$.
\end{proof}

\subsection*{Stage~1.5a \texorpdfstring{\textemdash}{---}
Strong resolvent convergence}
%-------------------------------------------------------------

\begin{lemma}[Strong resolvent convergence]
\label{lem:resolvent-convergence}
\emph{Operators used: $\Hc$, $\HSOT$.}

For every $z\in\mathbb{C}\setminus\mathbb{R}$:
\begin{equation}\label{eq:src}
  (\Hc-z)^{-1}\xrightarrow{\mathrm{SOT}}(\HSOT-z)^{-1}
  \quad\text{as }c\to\infty.
\end{equation}
\end{lemma}

\begin{proof}
Let $f\in L^2$ and $g_c:=(\Hc-z)^{-1}f$,
$g:=(\HSOT-z)^{-1}f$.
Then
\begin{align*}
  (\Hc-z)(g_c-g)
  &= f - (\Hc-z)g\\
  &= (\HSOT-z)g - (\Hc-z)g\\
  &= (\HSOT-\Hc)g.
\end{align*}
Hence
$g_c-g = (\Hc-z)^{-1}(\HSOT-\Hc)g$.
Taking norms:
\begin{equation}\label{eq:resolvent-bound}
  \|g_c-g\|
  \le \|(\Hc-z)^{-1}\|\cdot\|(\HSOT-\Hc)g\|
  \le \frac{1}{|\mathrm{Im}\,z|}\,\|(\HSOT-\Hc)g\|.
\end{equation}
By \eqref{VIII6}, $\|(\HSOT-\Hc)g\|\to 0$ for each fixed~$g$.
Hence $\|g_c-g\|\to 0$.
\end{proof}

\begin{remark}[Resolvent bound used]
\label{rem:resolvent-bound}
The key estimate is
$\|(\Hc-z)^{-1}\|\le|\mathrm{Im}\,z|^{-1}$,
which holds for all self-adjoint operators and all
$z\notin\mathbb{R}$.  Only~\eqref{VIII6} (SOT convergence)
and self-adjointness of~$\Hc$ are used here.
\end{remark}

%-------------------------------------------------------------
\subsection*{Stage~1.5b \texorpdfstring{\textemdash}{---}
Riesz projections via travelling contour}
%-------------------------------------------------------------

\begin{lemma}[Riesz projections converge strongly]
\label{lem:riesz-projection}
\emph{Operators used: $\Hc$, $\HSOT$.}

Fix $n\ge 0$.
Define the travelling contour
\begin{equation}\label{eq:contour}
  \Gamma_c:=\Bigl\{z\in\mathbb{C}:\,
  |z-\lambda_n^{(c)}|=r_c\Bigr\},
  \quad
  r_c:=\tfrac{1}{3}\bigl(\lambda_n^{(c)}-\lambda_{n+1}^{(c)}\bigr)
  \asymp c^{-1/2}.
\end{equation}
For each finite~$c$, $\Gamma_c$ encloses exactly the
$n$-th eigenvalue of~$\Hc$, and the Riesz spectral projection is
\begin{equation}\label{eq:riesz}
  P_n^{(c)}
  :=\frac{1}{2\pi i}\oint_{\Gamma_c}(z-\Hc)^{-1}\,dz
  =\ip{\cdot}{\Phi_n^{(c)}}\Phi_n^{(c)}.
\end{equation}
As $c\to\infty$:
\begin{equation}\label{eq:proj-SOT}
  P_n^{(c)}\xrightarrow{\mathrm{SOT}} P_n^{(\infty)},
\end{equation}
where $P_n^{(\infty)}$ is an orthogonal projection of rank~$\le 1$.
\end{lemma}

\begin{proof}
\textbf{Well-posedness of the contour.}
By the Slepian gap (\cite{Slepian1961}),
$\lambda_n^{(c)}-\lambda_{n+1}^{(c)}>0$
for every finite~$c$, so $r_c>0$.
All other eigenvalues of~$\Hc$ lie outside~$\Gamma_c$
(distance at least $2r_c$ from the centre).
Hence \eqref{eq:riesz} holds at each finite~$c$.

\textbf{SOT convergence of resolvents on the contour.}
Fix $f\in L^2$ and $z=\lambda_n^{(c)}+r_c e^{i\theta}\in\Gamma_c$.
Although the contour~$\Gamma_c$ shrinks, its radius satisfies
$r_c\ge\delta_n c^{-1/2}$ (Slepian gap lower bound),
so the resolvent bound \eqref{eq:resolvent-bound} gives
\begin{equation}\label{eq:resolvent-contour}
  \|(\Hc-z)^{-1}f-(\HSOT-z)^{-1}f\|
  \le r_c^{-1}\|(\HSOT-\Hc)g_z\|,
\end{equation}
where $g_z:=(\HSOT-z)^{-1}f$.
By \eqref{VIII6}, $\|(\HSOT-\Hc)g_z\|\to 0$ for fixed~$g_z$.
Further, $\|(\HSOT-\Hc)g_z\|\le C_\kappa c^{-3/4}\|g_z\|$
by \eqref{VIII2}, while $\|g_z\|\le r_c^{-1}\|f\|\le
\delta_n^{-1}c^{1/2}\|f\|$,
giving
\[
  \|(\Hc-z)^{-1}f-(\HSOT-z)^{-1}f\|
  \le r_c^{-1}\cdot C_\kappa c^{-3/4}\cdot
     \delta_n^{-1}c^{1/2}\|f\|
  \le C c^{1/2}\cdot c^{-3/4}\cdot c^{1/2}\|f\|
  = Cc^{1/4}\cdot r_c/r_c\cdot\|f\|.
\]

\textbf{Refined estimate.}
We use \eqref{VIII2} more carefully: for fixed~$f$,
$\|(\HSOT-\Hc)g_z\|\le C_\kappa c^{-3/4}\|g_z\|$.
But here $g_z=(\HSOT-z)^{-1}f$ with
$|\mathrm{Im}\,z|=r_c\sin\theta$, so
$\|g_z\|\le (r_c|\sin\theta|)^{-1}\|f\|$.
For $\theta$ bounded away from $0,\pi$:
\[
  \|(\Hc-z)^{-1}f-(\HSOT-z)^{-1}f\|
  \le \frac{C_\kappa c^{-3/4}}{r_c^2|\sin\theta|}\,\|f\|.
\]
With $r_c\asymp c^{-1/2}$:
$r_c^{-2}\asymp c$, so the bound is
$C_\kappa c^{-3/4}\cdot c\cdot\|f\| = C_\kappa c^{1/4}\|f\|$.

This is $\to\infty$, which is not what we want from a
pointwise estimate along the contour.
However, the contour integral gives an extra factor of
$|\Gamma_c|=2\pi r_c\asymp c^{-1/2}$:
\begin{align}\label{eq:proj-diff}
  \|P_n^{(c)}f - \tilde P_n^{(c)}f\|
  &\le \frac{1}{2\pi}\cdot 2\pi r_c
    \cdot\sup_{z\in\Gamma_c}
    \|(\Hc-z)^{-1}f-(\HSOT-z)^{-1}f\|\nonumber\\
  &\le r_c\cdot\frac{C_\kappa c^{-3/4}}{r_c^2}\,\|f\|
  = \frac{C_\kappa c^{-3/4}}{r_c}\,\|f\|
  \asymp c^{-3/4}\cdot c^{1/2}\|f\|
  = C_\kappa c^{-1/4}\|f\|,
\end{align}
where $\tilde P_n^{(c)}:=\frac{1}{2\pi i}\oint_{\Gamma_c}
(z-\HSOT)^{-1}dz$.
Hence $\|P_n^{(c)}f-\tilde P_n^{(c)}f\|\to 0$
for every fixed~$f$.

\textbf{Identification of the limit.}
The operator $\tilde P_n^{(c)}=\frac{1}{2\pi i}\oint_{\Gamma_c}
(z-\HSOT)^{-1}dz$ is the Riesz projection of $\HSOT$
associated to the arc of its spectrum enclosed by~$\Gamma_c$.
Since $\lambda_n^{(c)}\to\lambda_n^{(\infty)}$ by \eqref{VIII3}
and $r_c\to 0$, the contour $\Gamma_c$ contracts to the
point $\lambda_n^{(\infty)}$.
If $\lambda_n^{(\infty)}$ is an eigenvalue of $\HSOT$ of
multiplicity~$m$ (which is established in
Lemma~\ref{lem:weak-eigenrelation} to be $m\ge 1$),
then for all large~$c$ the contour encloses exactly the
eigenvalue $\lambda_n^{(\infty)}$ and
$\tilde P_n^{(c)}\to P_n^{(\infty)}$ SOT,
where $P_n^{(\infty)}$ is the spectral projection of
$\HSOT$ onto $\ker(\HSOT-\lambda_n^{(\infty)})$.
Combining:
$P_n^{(c)}\xrightarrow{\mathrm{SOT}}P_n^{(\infty)}$.

\textbf{Rank.}
Each $P_n^{(c)}$ has rank~1 (simple eigenvalue at each
finite~$c$); SOT limits of rank-1 projections have rank $\le 1$.
\end{proof}

\begin{remark}[Travelling contour handles vanishing gap]
\label{rem:travelling-contour}
A fixed contour would fail: for large~$c$, the gap
$\lambda_n^{(c)}-\lambda_{n+1}^{(c)}\asymp c^{-1/2}\to 0$
means any fixed circle eventually encloses multiple eigenvalues.
The travelling contour~\eqref{eq:contour} avoids this:
its radius $r_c$ tracks the gap, always enclosing exactly
one eigenvalue at each finite~$c$.
The price is the extra factor~$r_c^{-1}$ in the pointwise
resolvent estimate, which is absorbed by the contour length
$|\Gamma_c|=2\pi r_c$, yielding the net rate
$O(c^{-1/4})$ in~\eqref{eq:proj-diff}.
\end{remark}

%-------------------------------------------------------------
\subsection*{Stage~1.5c \texorpdfstring{\textemdash}{---}
Uniqueness of the limit eigenvector}
%-------------------------------------------------------------

\begin{corollary}[Strong convergence of eigenfunctions]
\label{cor:eigenfunction-convergence}
\emph{Operators used: $\Hc$, $\HSOT$
(via Lem.~\ref{lem:riesz-projection}).}

For fixed $n\ge 0$, $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$
strongly as $c\to\infty$, where $\Phi_n^{(\infty)}$ is the
unique (up to phase) unit vector spanning
$\mathrm{ran}(P_n^{(\infty)})$.
\end{corollary}

\begin{proof}
By Lemma~\ref{lem:riesz-projection}, $P_n^{(c)}
\xrightarrow{\mathrm{SOT}}P_n^{(\infty)}$
with $P_n^{(\infty)}$ of rank~$\le 1$.
Since $P_n^{(c)}\Phi_n^{(c)}=\Phi_n^{(c)}$ and
$\|\Phi_n^{(c)}\|=1$:
\[
  \|\Phi_n^{(c)}-P_n^{(\infty)}\Phi_n^{(c)}\|
  =\|(P_n^{(c)}-P_n^{(\infty)})\Phi_n^{(c)}\|
  \le\sup_{\|f\|=1}\|(P_n^{(c)}-P_n^{(\infty)})f\|.
\]
The right-hand side goes to~$0$ by SOT convergence
applied uniformly on the unit ball?  Not quite: SOT
convergence is pointwise, not uniform.

Instead, use precompactness directly.
By Lemma~\ref{lem:pswf-precompact}, every subsequence of
$\{\Phi_n^{(c)}\}$ has a further subsequence $\Phi_n^{(c_k)}
\to u$ strongly with $\|u\|=1$.
Apply $P_n^{(c_k)}$ and pass to the limit using
$P_n^{(c_k)}\xrightarrow{\mathrm{SOT}}P_n^{(\infty)}$:
\[
  P_n^{(\infty)}u
  =\lim_k P_n^{(c_k)}u
  =\lim_k P_n^{(c_k)}\bigl(\Phi_n^{(c_k)}
    +(u-\Phi_n^{(c_k)})\bigr)
  =\lim_k\Phi_n^{(c_k)}+\lim_k P_n^{(c_k)}
    (u-\Phi_n^{(c_k)}).
\]
First limit: $\lim_k\Phi_n^{(c_k)}=u$.
Second limit: $\|P_n^{(c_k)}(u-\Phi_n^{(c_k)})\|\le
\|u-\Phi_n^{(c_k)}\|\to 0$.
Hence $P_n^{(\infty)}u=u$, so $u\in\mathrm{ran}(P_n^{(\infty)})$.

Since $\|u\|=1$ and $P_n^{(\infty)}$ has rank~$\le 1$,
we get rank exactly~$1$ and $u$ spans
$\mathrm{ran}(P_n^{(\infty)})$.
This unit vector (after a phase convention) is
$\Phi_n^{(\infty)}$ â€” the same for every subsequence,
because every subsequential limit lies in the
\emph{same} rank-1 space $\mathrm{ran}(P_n^{(\infty)})$.
By the subsequence principle: $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$
strongly.
\end{proof}

\begin{remark}[Why uniqueness is now circular-free]
\label{rem:no-circularity}
The uniqueness of the limit direction follows from the fact
that $\mathrm{ran}(P_n^{(\infty)})$ is a fixed rank-1
subspace of $L^2$, constructed from $\HSOT$ and the Riesz
formula independently of any subsequence.
Every subsequential limit must lie in this space;
combined with the unit-norm condition from Stage~1,
the limit is unique.
No subsequence compatibility, no cross-$c$ inner products,
no limit functional.
\end{remark}

\subsection*{Stage~2 \texorpdfstring{\textemdash}{---}
$\HSOT$ eigenrelation}
%-------------------------------------------------------------

\begin{lemma}[$\Phi_n^{(\infty)}$ satisfies the
$\HSOT$-eigenrelation]
\label{lem:weak-eigenrelation}
\emph{Operators used: $\HSOT$ (verification only).
$\Phi_n^{(\infty)}$ is already unique by Stages~1--1.5c.}

$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
$\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ strongly, hence weakly.
For any $\psi$:
\begin{align*}
  \ip{\HSOT\Phi_n^{(\infty)}}{\psi}
  &=\ip{\Phi_n^{(\infty)}}{\HSOT\psi}
  =\lim_c\ip{\Phi_n^{(c)}}{\HSOT\psi}.
\end{align*}
Split: $\ip{\Phi_n^{(c)}}{\HSOT\psi}
=\ip{\Phi_n^{(c)}}{(\HSOT-\Hc)\psi}
+\ip{\Phi_n^{(c)}}{\Hc\psi}$.
First term $\to 0$ by \eqref{VIII2}.
Second:
$\lambda_n^{(c)}\ip{\Phi_n^{(c)}}{\psi}
\to\lambda_n^{(\infty)}\ip{\Phi_n^{(\infty)}}{\psi}$.
\end{proof}

\begin{definition}[Canonical domain]
\label{def:domain}
\[
  \Dom(\Hspec):=\Bigl\{f\in L^2:\,
  \sum_n(\lambda_n^{(\infty)})^2

  \Hspec f:=\sum_n\lambda_n^{(\infty)}

\begin{remark}[$\Phi_n^{(\infty)}\in\Dom(\Hspec)$]
\label{rem:phi-in-domain}
$\Hspec\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$

\begin{theorem}[Domain properties]
\label{thm:domain}

(b)~$\Hspec$ symmetric.
(c)~$\Dom(\Hspec)$ independent of subsequence choices
(since $\Phi_n^{(\infty)}$ is unique).
\end{theorem}

\section{Closability}
\label{sec:closable}
%=============================================================

\begin{theorem}[Closability of $\HSOT$]
\label{thm:closable-SOT}
\emph{Operators: $\HSOT$, $\Hc$.}
$\HSOT$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$, $\HSOT f_k\to g$.
Fix $\varepsilon>0$; choose $K$: $\|g-\HSOT f_k\|<\varepsilon/2$
for $k\ge K$.
Fix $k\ge K$; by \eqref{VIII6} choose $c_0$:
$\|\HSOT f_k-\Hc f_k\|<\varepsilon/2$.
Then $\|g-\Hc f_k\|<\varepsilon$.
Fix $c\ge c_0$, let $k\to\infty$:
$\|\Hc f_k\|\le\|f_k\|\to 0$, so $\|g\|\le\varepsilon$;
hence $g=0$.
\end{proof}

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable-spec}
\emph{Operators: $\Hspec$.}
$\Hspec$ is closable.
\end{theorem}

\begin{proof}
Let $f_k\to 0$, $\Hspec f_k\to g$.
$a_n^{(k)}:=\ip{f_k}{\Phi_n^{(\infty)}}\to 0$.
For any $\phi$:
$\ip{\Hspec f_k}{\phi}
=\sum_n\lambda_n^{(\infty)}a_n^{(k)}
\ip{\Phi_n^{(\infty)}}{\phi}\to 0$.
Hence $g=0$.
\end{proof}

\section{Deficiency Index Analysis}
\label{sec:deficiency}
%=============================================================

\begin{lemma}[Deficiency vectors $\perp\{\Phi_n^{(\infty)}\}$]
\label{lem:deficiency-analytic}
\emph{Operators: $\Hspec$.}
If $f\in\ker(\Hspec^*+i)$ then
$\ip{f}{\Phi_n^{(\infty)}}=0$ for all~$n$.
\end{lemma}

\begin{proof}
$(\lambda_n^{(\infty)}+i)\ip{f}{\Phi_n^{(\infty)}}=0$;
$\lambda_n^{(\infty)}+i\ne 0$.
\end{proof}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}$ is complete in $L^2(\mathbb{R})$.
\end{hypothesis}

\section{Essential Self-Adjointness}
\label{sec:esa}
%=============================================================

\begin{theorem}[$n_\pm=0$]
\label{thm:deficiency}
Under Hyp.~\ref{hyp:decay}: $n_\pm(\Hspec)=0$.
\end{theorem}

\begin{proof}
$f\in\ker(\Hspec^*+i)\Rightarrow f\perp\{\Phi_n^{(\infty)}\}
\Rightarrow f=0$.
\end{proof}

\begin{corollary}[ESSA]
\label{cor:esa}
Under Hyp.~\ref{hyp:decay}: $\overline{\Hspec}=\Hspec^*$.
\end{corollary}

\begin{theorem}[Spectral identity and RH reduction]
\label{thm:sigma-identity}
Under Hyp.~\ref{hyp:decay} and CCM~\cite{CCM2025}:
(i)~$\sigma_{\rm app}(\Hspec)=\sigma(\Hspec)$;
(ii)~every non-trivial zero $\rho$ of $\zeta$
lies in $\sigma(\Hspec)$;
(iii)~$\mathrm{Re}(\rho)=\tfrac12$.
\end{theorem}

\begin{remark}[Meta-corollary]
\label{rem:meta-corollary}
Thm.~\ref{thm:sigma-identity}(iii) depends on
Hyp.~\ref{hyp:decay} (open), CCM (conjectural),
and unverified spectral calibration.
Not a proof of RH.
\end{remark}
