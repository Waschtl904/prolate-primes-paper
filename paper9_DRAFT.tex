\section*{Honest Summary for the Referee}
%============================================================

This paper establishes the following \emph{unconditional} results:
\begin{enumerate}[(U1)]
\item\label{U1}
  The PSWF family $\{\Phi_n^{(c)}\}$ is precompact in $L^2$
  (Fr\'echet--Kolmogorov; no operator).
\item\label{U2}
  For fixed $c<\infty$, the eigenprojection $P_n^{(c)}$ has
  rank~1 (Slepian gap; no limit operator).
\item\label{U3}
  Any strong subsequential limit $u$ of $\{\Phi_n^{(c_k)}\}$
  satisfies $\|u\|=1$.
\item\label{U4}
  $\Hspec$ (defined via spectral series) is closable and symmetric.
\item\label{U5}
  $\HSOT := \mathrm{SOT}\text{-}\lim_c \Hc$ is a well-defined
  bounded self-adjoint operator; it is closable trivially
  (it is bounded).
\end{enumerate}

The following results are \emph{conditional}:
\begin{enumerate}[(C1)]
\item\label{C1}
  \textbf{Conditional on Hyp.~\ref{hyp:norm-resolvent}
  (norm-resolvent convergence) and
  Hyp.~\ref{hyp:HSOT-discrete}
  (discrete spectrum of $\HSOT$):}
  $\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0$
  (full convergence, no subsequence extraction).
\item\label{C2}
  \textbf{Conditional on (C1) and
  Hyp.~\ref{hyp:decay} (completeness):}
  $\Hspec$ is essentially self-adjoint.
  Note: ESSA is \emph{equivalent} to completeness of
  $\{\Phi_n^{(\infty)}\}$; it is not an independent theorem.
\item\label{C3}
  \textbf{Conditional on (C2), Hyp.~\ref{hyp:norm-resolvent},
  and CCM~\cite{CCM2025}:}
  a spectral connection to zeta zeros.
  This is a \emph{meta-remark}, not a theorem.
  See Sec.~\ref{sec:rh-remark}.
\end{enumerate}

\section{Setup and Inherited Results}
\label{sec:setup}
%=============================================================

We work in $L^2(\mathbb{R})$.
The PSWF concentration operators $\Hc$ are integral operators
defined by the kernel $\sin(c(x-y))/(\pi(x-y))$;
they are bounded, self-adjoint, and $\|\Hc\|\le 1$.
Eigenvalues $\lambda_0^{(c)}\ge\lambda_1^{(c)}\ge\cdots>0$;
eigenfunctions $\Phi_n^{(c)}$, normalised: $\|\Phi_n^{(c)}\|=1$.

We use the following results from earlier papers:
\begin{enumerate}[(VIII.1)]
\item\label{VIII2}
  $\|(\Hc-\HSOT)f\|\le C_\kappa c^{-3/4}\|f\|$
  for fixed unit~$f\in L^2$ (Paper~VII, Thm.~C).
\item\label{VIII3}
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|\le C_\kappa c^{-1/4}$,
  where $\lambda_n^{(\infty)}:=\lim_c\lambda_n^{(c)}$
  (Paper~VIII, Cor.~4.3).
\item\label{VIII6}
  $\Hc\xrightarrow{\mathrm{SOT}}\HSOT$, where $\HSOT$ is a
  bounded self-adjoint operator defined as the SOT limit
  (Paper~X, Lem.~2.2).
\end{enumerate}

\begin{remark}[$\HSOT$ is bounded self-adjoint]
\label{rem:HSOT-bounded}
$\HSOT$ is the strong operator limit of bounded self-adjoint
contractions, hence bounded self-adjoint with $\|\HSOT\|\le 1$.
In particular, $\HSOT$ is defined on all of $L^2$,
$\Dom(\HSOT)=L^2$, and it is everywhere closable
(every bounded operator is closable).
\emph{No closability argument for $\HSOT$ is needed.}
\end{remark}

\section{Open Hypotheses}
\label{sec:hypotheses}
%=============================================================

We collect the open assumptions on which the
conditional results depend.

\begin{hypothesis}[Norm-resolvent convergence]
\label{hyp:norm-resolvent}
For some (equivalently, every) $z\in\mathbb{C}\setminus\mathbb{R}$:
\[
  \|(\Hc-z)^{-1}-(\HSOT-z)^{-1}\|_{\mathcal{B}(L^2)}
  \to 0 \quad\text{as }c\to\infty.
\]
\end{hypothesis}

\begin{remark}[Why SOT does not suffice]
\label{rem:SOT-not-enough}
SOT convergence $\Hc\xrightarrow{\mathrm{SOT}}\HSOT$
(\eqref{VIII6}) does not in general imply
norm-resolvent convergence.
Counterexamples exist even among uniformly bounded
self-adjoint families; see \cite{ReedSimon1975},
Example VIII.7.14.
Hyp.~\ref{hyp:norm-resolvent} is a strictly stronger
assumption.
It would follow from operator-norm convergence
$\|\Hc-\HSOT\|\to 0$
(Kato \cite{Kato1966}, Thm.~VIII.1.5),
which is in turn implied by
Hyp.~\ref{hyp:uniform-alignment} summed over all $n$.
The logical chain is:
\[
  \text{Hyp.~\ref{hyp:uniform-alignment}}
  \Rightarrow\|\Hc-\HSOT\|\to 0
  \Rightarrow\text{Hyp.~\ref{hyp:norm-resolvent}}.
\]
\end{remark}

\begin{hypothesis}[Discrete spectrum of $\HSOT$]
\label{hyp:HSOT-discrete}
$\HSOT$ has discrete spectrum with simple eigenvalues
$\lambda_0^{(\infty)}>\lambda_1^{(\infty)}>\cdots>0$,
with corresponding eigenvectors $\Phi_n^{(\infty)}$.
\end{hypothesis}

\begin{remark}[Status of Hyp.~\ref{hyp:HSOT-discrete}]
\label{rem:HSOT-discrete-status}
This is the spectral structure of $\HSOT$ required for
Kato's theorem to apply.
For the specific operator $\HSOT$ arising from PSWFs,
the strict ordering $\lambda_n^{(\infty)}>\lambda_{n+1}^{(\infty)}$
follows from the Slepian gap at finite~$c$ and
\eqref{VIII3}, by taking limits.
Completeness of the eigensystem is addressed separately
in Hyp.~\ref{hyp:decay}.
\end{remark}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}_{n\ge 0}$ is a complete
orthonormal system in $L^2(\mathbb{R})$.
\end{hypothesis}

\begin{hypothesis}[Uniform alignment]
\label{hyp:uniform-alignment}
For each $\kappa>0$:
\[
  \sup_{n\le\kappa c}\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|
  \le C_\kappa\, c^{-1/4}.
\]
\end{hypothesis}

\begin{remark}[Dependency diagram]
\label{rem:dep}
\[
  \text{Hyp.~\ref{hyp:uniform-alignment}}
  \Rightarrow
  \text{Hyp.~\ref{hyp:norm-resolvent}}
  \xRightarrow{+\,\text{Hyp.~\ref{hyp:HSOT-discrete}}}
  \text{(C1)}
  \xRightarrow{+\,\text{Hyp.~\ref{hyp:decay}}}
  \text{(C2) ESSA}.
\]
Proving Hyp.~\ref{hyp:uniform-alignment} unconditionally
would close (C1) and (C2) simultaneously
(given Hyp.~\ref{hyp:HSOT-discrete} and
Hyp.~\ref{hyp:decay}).
\end{remark}

\section{Unconditional Results}
\label{sec:unconditional}
%=============================================================

\subsection{Precompactness}

\begin{lemma}[PSWF precompactness]
\label{lem:pswf-precompact}
\emph{No operators used.}
For fixed $n\ge 0$, $\{\Phi_n^{(c)}\}_{c\ge 1}$
is precompact in $L^2(\mathbb{R})$.
Every subsequence has a further subsequence converging
strongly to some unit vector.
\end{lemma}

\begin{proof}
Fr\'echet--Kolmogorov (\cite{ReedSimon1975}, Thm.~IV.26)
applied using:
(i)~$\|\Phi_n^{(c)}\|=1$;
(ii)~uniform exponential decay of $\Phi_n^{(c)}$ for fixed~$n$
(\cite{Osipov2013}, Thm.~3.2);
(iii)~uniform $H^1$ bound for fixed~$n$
(\cite{Osipov2013}, Prop.~4.1).
\end{proof}

\subsection{Rank-1 eigenprojections at finite $c$}

\begin{lemma}[Rank-1 structure]
\label{lem:rank1}
\emph{Operator: $\Hc$ at fixed finite $c$.}
The Slepian gap $\lambda_n^{(c)}>\lambda_{n+1}^{(c)}>0$
(\cite{Slepian1961}) ensures $\lambda_n^{(c)}$ is a
simple isolated eigenvalue, and
\[
  P_n^{(c)}f = \ip{f}{\Phi_n^{(c)}}\Phi_n^{(c)},
  \quad\|P_n^{(c)}\|=1.
\]
This holds at each individual $c$; no uniformity in $c$ is claimed.
\end{lemma}

\subsection{Spectral operator $\Hspec$: definition and closability}

\begin{definition}[Canonical domain and spectral operator]
\label{def:domain}
\[
  \Dom(\Hspec):=\Bigl\{f\in L^2:\,
  \sum_n(\lambda_n^{(\infty)})^2
  |\ip{f}{\Phi_n^{(\infty)}}|^2<\infty\Bigr\},
\]
\[
  \Hspec f:=\sum_n\lambda_n^{(\infty)}
  \ip{f}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)},
  \quad f\in\Dom(\Hspec),
\]
where $\{\Phi_n^{(\infty)}\}$ are the unit vectors from
Lem.~\ref{lem:pswf-precompact} (chosen along any
convergent subsequence; well-definedness of $\Hspec$
is conditional on Hyp.~\ref{hyp:norm-resolvent}
for full convergence, or alternatively on any
choice of subsequential limits that form an ONS).
\end{definition}

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable-spec}
Under any choice of $\{\Phi_n^{(\infty)}\}$ forming
an ONS: $\Hspec$ is closable and symmetric.
\end{theorem}

\begin{proof}
\textit{Symmetry:}
$\ip{\Hspec f}{g}
=\sum_n\lambda_n^{(\infty)}\ip{f}{\Phi_n^{(\infty)}}
\overline{\ip{g}{\Phi_n^{(\infty)}}}
=\ip{f}{\Hspec g}$
for $f,g\in\Dom(\Hspec)$.

\textit{Closability:}
Let $f_k\to 0$ in $L^2$ and $\Hspec f_k\to g$.
For each $n$: $\ip{f_k}{\Phi_n^{(\infty)}}\to 0$.
For any test vector $\phi$:
\[
  \ip{\Hspec f_k}{\phi}
  =\sum_n\lambda_n^{(\infty)}
  \ip{f_k}{\Phi_n^{(\infty)}}\ip{\Phi_n^{(\infty)}}{\phi}\to 0.
\]
Hence $\ip{g}{\phi}=0$ for all $\phi$,
so $g=0$.
\end{proof}

\section{Conditional Result: Full Eigenfunction Convergence}
\label{sec:conditional-convergence}
%=============================================================

\begin{theorem}[Full convergence of $\Phi_n^{(c)}$]
\label{thm:phi-conv}
\emph{Conditional on
Hyp.~\ref{hyp:norm-resolvent} and
Hyp.~\ref{hyp:HSOT-discrete}.}

For fixed $n\ge 0$:
\[
  \|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0
  \quad\text{as }c\to\infty
\]
(full convergence, no subsequence selection).
\end{theorem}

\begin{proof}
\textbf{Step~1.}
Hyp.~\ref{hyp:HSOT-discrete} gives that
$\lambda_n^{(\infty)}$ is a simple isolated eigenvalue
of $\HSOT$, with eigenprojection $P_n^{(\infty)}$
of rank~1.

\textbf{Step~2.}
Under Hyp.~\ref{hyp:norm-resolvent},
Kato (\cite{Kato1966}, Thm.~VIII.1.8) gives:
\[
  \|P_n^{(c)}-P_n^{(\infty)}\|_{\mathcal{B}(L^2)}\to 0.
\]

\textbf{Step~3.}
Since $P_n^{(c)}$ and $P_n^{(\infty)}$ both have rank~1,
norm convergence of projections implies
norm convergence of the corresponding unit eigenvectors
(up to a global phase):
\[
  \|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0.
\]
See \cite{Kato1966}, p.~34, or the direct computation:
$|\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}|^2
=\mathrm{tr}(P_n^{(c)}P_n^{(\infty)})
\to\mathrm{tr}(P_n^{(\infty)})^2=1$,
hence $\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|^2
=2-2\,\mathrm{Re}\,\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}\to 0$
(with the positive phase convention).
\end{proof}

\begin{remark}[What unconditional results give]
\label{rem:what-unconditional}
Without Hyp.~\ref{hyp:norm-resolvent}
and Hyp.~\ref{hyp:HSOT-discrete},
Lem.~\ref{lem:pswf-precompact} gives only:
precompactness and unit norm of subsequential limits.
Uniqueness of the limit cannot be established
from precompactness alone.
See Rem.~\ref{rem:stage1-limit} below.
\end{remark}

\begin{remark}[Precompactness does not give uniqueness]
\label{rem:stage1-limit}
A precompact sequence in a metric space may have
multiple cluster points.
The existence of a unique limit requires either
full convergence (as in Thm.~\ref{thm:phi-conv})
or an independent uniqueness argument
(e.g.\ variational, monotone, or analytic).
No such argument is available here without
additional structure in~$c$.
\end{remark}

\section{Eigenrelation for $\HSOT$ (Verification)}
\label{sec:eigenrelation}
%=============================================================

\begin{lemma}[Eigenrelation]
\label{lem:weak-eigenrelation}
\emph{Conditional on Thm.~\ref{thm:phi-conv}.}

$\HSOT\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
By Thm.~\ref{thm:phi-conv},
$\Phi_n^{(c)}\to\Phi_n^{(\infty)}$ in norm,
hence strongly.
Since $\HSOT$ is bounded
(Rem.~\ref{rem:HSOT-bounded}),
\[
  \HSOT\Phi_n^{(c)} \to \HSOT\Phi_n^{(\infty)}
  \quad\text{in norm.}
\]
But $\Hc\Phi_n^{(c)}=\lambda_n^{(c)}\Phi_n^{(c)}$
and $\|(\Hc-\HSOT)\Phi_n^{(c)}\|
\le C_\kappa c^{-3/4}\to 0$
(\eqref{VIII2}), so
$\HSOT\Phi_n^{(c)}=
\lambda_n^{(c)}\Phi_n^{(c)}
+O(c^{-3/4})$.
Taking the limit:
$\HSOT\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\medskip
\noindent
\emph{Note on the limit interchange.}
The step $\HSOT\Phi_n^{(c)}\to\HSOT\Phi_n^{(\infty)}$
uses boundedness of $\HSOT$ and norm convergence
of $\Phi_n^{(c)}$; no interchange of limits is required.
\end{proof}

\section{Orthonormality and Conditional ONS}
\label{sec:ons}
%=============================================================

\begin{proposition}[$\{\Phi_n^{(\infty)}\}$ is a conditional ONS]
\label{prop:ons}
\emph{Conditional on Thm.~\ref{thm:phi-conv}.}
\begin{enumerate}[\rm(a)]
\item $\|\Phi_n^{(\infty)}\|=1$.
\item $\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}}=0$, $n\ne m$.
\item $\HSOT\Phi_n^{(\infty)}
  =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{enumerate}
Completeness (making it an ONB) requires additionally
Hyp.~\ref{hyp:decay}.
\end{proposition}

\begin{proof}
(a) norm convergence preserves unit norm.
(b) $\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}}=0$ +
norm convergence.
(c) Lem.~\ref{lem:weak-eigenrelation}.
\end{proof}

\section{Essential Self-Adjointness}
\label{sec:esa}
%=============================================================

\begin{theorem}[ESSA $\Leftrightarrow$ completeness]
\label{thm:esa}
\emph{Conditional on Thm.~\ref{thm:phi-conv}.}

The following are equivalent:
\begin{enumerate}[\rm(i)]
\item $\Hspec$ is essentially self-adjoint.
\item $\{\Phi_n^{(\infty)}\}$ is complete in $L^2$
  (Hyp.~\ref{hyp:decay}).
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(ii)$\Rightarrow$(i):}
Let $f\in\ker(\Hspec^*\pm i)$.
For each $n$:
$(\lambda_n^{(\infty)}\pm i)\ip{f}{\Phi_n^{(\infty)}}=0$,
hence $\ip{f}{\Phi_n^{(\infty)}}=0$.
By completeness (ii), $f=0$.
So $\ker(\Hspec^*\pm i)=\{0\}$, i.e.\ $n_\pm=0$,
i.e.\ $\Hspec$ is essentially self-adjoint.

\textit{(i)$\Rightarrow$(ii):}
If $f\perp\Phi_n^{(\infty)}$ for all $n$ but $f\ne 0$,
then $f$ is in the domain of $\Hspec^*$ with
$\Hspec^* f = \mp i\cdot 0 = 0$.
Actually: $f\in\ker(\Hspec^*)$ if $\Hspec f=0$ on all
basis vectors, so $f$ would be a deficiency vector;
hence $n_\pm>0$ and $\Hspec$ is not essentially
self-adjoint.
\end{proof}

\begin{remark}[ESSA is not an independent theorem]
\label{rem:essa-equivalent}
Thm.~\ref{thm:esa} shows that ESSA and completeness
are \emph{equivalent} under our setup.
Neither can be established without the other.
The result should therefore be stated as:
``$\Hspec$ is essentially self-adjoint
\emph{if and only if} $\{\Phi_n^{(\infty)}\}$
is complete in $L^2$.''
This is mathematically honest and does not overstate
what has been proved.
\end{remark}

\section{Spectral Connection to Zeta: A Meta-Remark}
\label{sec:rh-remark}
%=============================================================

\begin{remark}[What would be needed for a zeta connection]
\label{rem:rh-meta}
A genuine spectral theorem connecting $\Hspec$
to the Riemann zeta function would require, at minimum:
\begin{enumerate}[(a)]
\item A proof that $\{\Phi_n^{(\infty)}\}$ is complete
  (Hyp.~\ref{hyp:decay}): currently open.
\item A proof of norm-resolvent convergence
  (Hyp.~\ref{hyp:norm-resolvent}): currently open.
\item A spectral isomorphism between $\Hspec$
  and a trace-class operator whose eigenvalues
  are in bijection with zeta zeros;
  this goes beyond approximation-spectrum inclusions.
\item A Weil-type or trace formula linking spectral data
  of $\Hspec$ to the explicit formula for $\zeta$;
  this is not provided by CCM~\cite{CCM2025} alone.
\end{enumerate}
Item~(iii) in particular is not established in the
present paper or in Papers~V--VIII.
The result of Paper~VIII (Thm.~5.2) gives an
\emph{approximate-spectrum inclusion}:
$\rho\in\sigma_{\mathrm{app}}(\Hspec)$
conditional on CCM;
this does not imply $\mathrm{Re}(\rho)=1/2$
without (c) and (d).

For these reasons, any statement of the form
``$\Hspec$ proves RH'' is not justified.
The connection remains a \emph{conditional research program}.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%=============================================================

\begin{problem}[Norm-resolvent convergence]
\label{prob:norm-resolvent}
Prove Hyp.~\ref{hyp:norm-resolvent}.
Sufficient condition: $\|\Hc-\HSOT\|\to 0$,
which follows from Hyp.~\ref{hyp:uniform-alignment}
via the operator-norm bound
$\|\Hc-\Hspec\|\le C_\kappa c^{-1/4}$
(Thm.~\ref{thm:opnorm}) plus
$\|\Hspec-\HSOT\|$ (open).
\end{problem}

\begin{problem}[Completeness of $\{\Phi_n^{(\infty)}\}$]
\label{prob:completeness}
Prove Hyp.~\ref{hyp:decay}.
This is equivalent to ESSA of $\Hspec$
(Thm.~\ref{thm:esa}).
\end{problem}

\begin{problem}[Discrete spectrum of $\HSOT$]
\label{prob:HSOT-discrete}
Prove Hyp.~\ref{hyp:HSOT-discrete}.
Strict eigenvalue ordering follows from the
Slepian gap (\cite{Slepian1961}) and \eqref{VIII3};
completeness of the eigensystem is the content of
Hyp.~\ref{hyp:decay}.
\end{problem}

\begin{problem}[Uniform alignment]
\label{prob:uniform-alignment}
Prove Hyp.~\ref{hyp:uniform-alignment}.
This is the strongest hypothesis;
it implies Hyp.~\ref{hyp:norm-resolvent}
(see Rem.~\ref{rem:dep}).
\end{problem}

\begin{problem}[Identification of operators]
\label{prob:identification}
Prove $\overline{\Hspec}=\HSOT$.
\end{problem}
