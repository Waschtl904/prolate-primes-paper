\maketitle

\begin{abstract}
We continue the spectral analysis of the PSWF
concentration operators $\Hc$ and their formal limit
$\Hlim$, initiated in Papers~V--VIII of this series.

The central obstacle identified in Paper~VIII
\cite{PaperVIII} is the gap between
$\sigma_{\rm app}(\Hlim)$ and $\sigma(\Hlim)$:
the approximate spectrum is too large unless
$\Hlim$ is (essentially) self-adjoint.
Paper~VIII states this as Conjecture~8.1.

\textbf{Main results of this paper.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:domain}
      (Canonical domain).}
      We define a symmetric operator $\Hlim$
      via a spectral series on an explicitly described
      domain containing all finite linear combinations
      of $\{\Phi_n^{(\infty)}\}$.
      Its identification with the SOT-limit operator
      is addressed in Theorem~\ref{thm:closable}.

\item \textbf{Theorem~\ref{thm:closable}
      (Closability).}
      $\Hlim$ is closable,
      proved directly from SOT-convergence
      $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
      without invoking completeness of
      $\{\Phi_n^{(\infty)}\}$.

\item \textbf{Theorem~\ref{thm:opnorm}
      (Operator norm rate).}
      $\|\Hc - \Hlim\|_{L^2 \to L^2}
      \le C_\kappa\, c^{-1/4}$,
      conditional on Hypothesis~\ref{hyp:uniform-alignment}
      (uniform eigenfunction alignment).
      The eigenvalue-sum term is unconditionally
      $O(c^{-1/4})$.

\item \textbf{Lemma~\ref{lem:deficiency-analytic}
      (Orthogonality principle, unconditional).}
      Every $f \in \ker(\Hlim^* \pm i)$ is
      orthogonal to all $\Phi_n^{(\infty)}$.
      In particular, completeness of
      $\{\Phi_n^{(\infty)}\}$ implies
      $\ker(\Hlim^* \pm i) = \{0\}$,
      hence essential self-adjointness.
      The converse direction
      ($n_\pm = 0$ implying completeness)
      is not established here
      and requires additional spectral structure
      (see Remark~\ref{rem:deficiency-orthogonal}).

\item \textbf{Corollary~\ref{cor:esa}
      (Conditional conclusion).}
      Under Hypothesis~\ref{hyp:decay}
      (completeness of $\{\Phi_n^{(\infty)}\}$,
      currently open)
      and Hypothesis~CCM
      (\cite{CCM2025}, currently conjectural),
      all non-trivial zeros $\rho$ of $\zeta(s)$
      satisfy $\mathrm{Re}(\rho) = 1/2$.
      This is a conditional reduction,
      not a proof of the Riemann hypothesis.
\end{enumerate}
\end{abstract}

\section*{Guide for the Referee}

This paper has three main threads:
\begin{enumerate}
\item \textbf{Domain thread}
      (Sections~\ref{sec:domain}--\ref{sec:closable}):
      $\Hlim$ is well-defined and symmetric on
      the span of $\{\Phi_n^{(\infty)}\}$.
      The spectral definition defines a symmetric operator;
      its identification with the SOT-limit is
      established unconditionally via closability.
      Density of the domain is conditional
      on Hypothesis~\ref{hyp:decay}.
\item \textbf{Operator-norm thread}
      (Section~\ref{sec:opnorm}):
      Unconditional: eigenvalue-sum term $O(c^{-1/4})$.
      Conditional on Hypothesis~\ref{hyp:uniform-alignment}:
      cross-term estimate and full $O(c^{-1/4})$ rate.
      Hypothesis~\ref{hyp:uniform-alignment} is a
      non-trivial global spectral assumption
      discussed in Remark~\ref{rem:uniform-alignment-status}.
\item \textbf{Deficiency thread}
      (Sections~\ref{sec:deficiency}--\ref{sec:esa}):
      The central unconditional result is the
      orthogonality principle
      (Lemma~\ref{lem:deficiency-analytic}):
      deficiency vectors are orthogonal to all
      $\Phi_n^{(\infty)}$.
      This yields the implication
      Completeness $\Rightarrow$ ESSA,
      but not in general the converse
      (see Remark~\ref{rem:deficiency-orthogonal}).
      The conclusion $n_\pm = 0$ and the RH reduction
      are conditional on Hypothesis~\ref{hyp:decay}
      and Hypothesis~CCM
      (Remark~\ref{rem:ccm-status}).
\end{enumerate}

\section{Setup and Inherited Notation}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--VIII.
Key inherited results:

\begin{enumerate}[(VIII.1)]
\item\label{VIII1}
      $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$
      (Paper~VIII, Cor.~2.3).
\item\label{VIII2}
      $\|(\Hc - \Hlim)f\| \le C_\kappa c^{-3/4}\|f\|$
      for every fixed unit $f$ (strong rate;
      Paper~VII, Thm.~C).
\item\label{VIII3}
      $|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
      \le C_\kappa c^{-1/4}$
      (Paper~VIII, Cor.~4.3, telescoping sum).
\item\label{VIII4}
      Non-cancellation: there exist $E_c$ with
      $|E_c| \ge c_* c^{-1/2}$ and
      $|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$ on $E_c$
      (Paper~VIII, Thm.~3.3).
\item\label{VIII5}
      $\rho \in \sigma_{\rm app}(\Hlim)$ for every
      non-trivial zero $\rho$ of $\zeta(s)$,
      conditional on CCM2025 (Paper~VIII, Thm.~5.2).
\item\label{VIII6}
      SOT-convergence: $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
      in $\mathcal{B}(L^2(\mathbb{R}))$
      (Paper~X, Lem.~2.2).
\end{enumerate}

The operators $\Hc$ are self-adjoint contractions
on $L^2(\mathbb{R})$ ($\|\Hc\|_{L^2 \to L^2} \le 1$;
see Papers~V--VI).
The operator $\Hlim$ is defined in
Section~\ref{sec:domain}.

\section{Canonical Domain of $\Hlim$}
\label{sec:domain}
%%═══════════════════════════════════════════════════════════════════

\subsection{The limit eigenvalues and eigenvectors}

Let $\{\Phi_n^{(c)}\}_{n \ge 0}$ be the orthonormal
eigenbasis of $\Hc$ with eigenvalues $\lambda_n^{(c)}$.
By \eqref{VIII3}, for each fixed $n$,
$\lambda_n^{(c)} \to \lambda_n^{(\infty)}$ as $c \to \infty$.

\begin{definition}[Limit eigenvalues]
\label{def:lim-eigenvalues}
For $n \ge 0$, define
$\lambda_n^{(\infty)} := \lim_{c \to \infty} \lambda_n^{(c)}$.
Existence follows from \eqref{VIII3}.
\end{definition}

\begin{remark}[Values of $\lambda_n^{(\infty)}$]
\label{rem:lambda-values}
For $n \le (1-\delta)2c/\pi$:
$\lambda_n^{(\infty)} \in (0,1]$.
For $n$ above the threshold:
$\lambda_n^{(\infty)} = 0$.
The case $\lambda_n^{(\infty)} = 0$
is handled explicitly where needed
(Remarks~\ref{rem:lambda-zero}
and~\ref{rem:deficiency-zero-eigenvalue}).
\end{remark}

\begin{proposition}[Weak convergence of eigenvectors]
\label{prop:weak-limits}
After passing to a subsequence,
$\Phi_n^{(c)} \rightharpoonup \Phi_n^{(\infty)}$
weakly in $L^2(\mathbb{R})$,
where $\{\Phi_n^{(\infty)}\}_{n \ge 0}$ is
an orthonormal system
($\|\Phi_n^{(\infty)}\| = 1$ for all $n$).
\end{proposition}

\begin{proof}
Banach--Alaoglu gives weakly convergent
subsequences from bounded (unit-vector) sequences.
Uniform eigenvalue separation
(\cite{Slepian1961}) ensures mutual orthogonality
of subsequential limits.
\end{proof}

\subsection{The canonical domain}

\begin{definition}[Canonical domain]
\label{def:domain}
Define
\[
  \Dom(\Hlim) :=
  \Bigl\{
    f \in L^2(\mathbb{R}) :
    \sum_{n=0}^\infty
    \bigl(\lambda_n^{(\infty)}\bigr)^2
    \bigl|\ip{f}{\Phi_n^{(\infty)}}\bigr|^2
    < \infty
  \Bigr\},
\]
and for $f \in \Dom(\Hlim)$:
$\Hlim f :=
\sum_{n : \lambda_n^{(\infty)} > 0}
\lambda_n^{(\infty)}
\ip{f}{\Phi_n^{(\infty)}}
\Phi_n^{(\infty)}$.
\end{definition}

\begin{remark}[Definition vs.\ SOT-limit]
\label{rem:domain-vs-SOT}
Definition~\ref{def:domain} introduces a
\emph{symmetric operator} acting on functions
with sufficient spectral decay.
A priori it is not automatic that this
coincides with the operator obtained as
SOT-limit of $\Hc$: the spectral definition
could in principle miss components of $f$
orthogonal to all $\Phi_n^{(\infty)}$
(i.e.\ components in the potential
null-space of $\Hlim$).
This subtlety is resolved by
Theorem~\ref{thm:closable}:
closability of the spectrally-defined
$\Hlim$ --- proved independently of
completeness --- establishes that its
closure is compatible with
the SOT-limit.
\end{remark}

\begin{theorem}[Properties of the canonical domain]
\label{thm:domain}
\begin{enumerate}[\rm(a)]
\item $\Dom(\Hlim)$ contains all finite
      linear combinations of $\{\Phi_n^{(\infty)}\}$.
      Under Hypothesis~\ref{hyp:decay},
      $\Dom(\Hlim)$ is dense in $L^2(\mathbb{R})$.
\item $\Hlim$ is symmetric on $\Dom(\Hlim)$.
\item $\Dom(\Hlim)$ is independent of
      the subsequence in Proposition~\ref{prop:weak-limits}.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(a)} Finite combinations have finitely
many nonzero terms in the domain sum,
hence lie in $\Dom(\Hlim)$.
Under Hypothesis~\ref{hyp:decay}, the finite
span is dense, so $\Dom(\Hlim)$ is dense.

\textit{(b)} Symmetry follows from
absolute convergence of the series
(Cauchy--Schwarz + domain condition).

\textit{(c)} $\lambda_n^{(\infty)}$ does not
depend on the subsequence choice.
\end{proof}

\begin{remark}[What density requires]
\label{rem:density-requires}
The two statements of Thm.~\ref{thm:domain}(a)
are logically separate:
\begin{itemize}
\item $\Dom(\Hlim) \supset
      \mathrm{span}\{\Phi_n^{(\infty)}\}$
      (unconditional).
\item $\Dom(\Hlim)$ dense in $L^2(\mathbb{R})$
      (conditional on Hypothesis~\ref{hyp:decay}).
\end{itemize}
Closability (Section~\ref{sec:closable})
and the Deficiency analysis
(Section~\ref{sec:deficiency})
use only the unconditional part.
\end{remark}

\section{Closability}
\label{sec:closable}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closability of $\Hlim$]
\label{thm:closable}
$\Hlim$ is closable.
The proof requires neither completeness of
$\{\Phi_n^{(\infty)}\}$ nor the spectral
definition of $\Hlim$.
\end{theorem}

\begin{proof}
Let $f_k \to 0$ in $L^2$ with $\Hlim f_k \to g$.
We show $g = 0$.

For each $k$ and $c$:
$\|g - \Hc f_k\|
\le \|g - \Hlim f_k\|
  + \|\Hlim f_k - \Hc f_k\|$.

Fix $\varepsilon > 0$.

\textit{Step 1.}
$\Hlim f_k \to g$:
choose $K$ with
$\|g - \Hlim f_k\| < \varepsilon/2$
for $k \ge K$.

\textit{Step 2.}
SOT-convergence \eqref{VIII6}:
for fixed $k \ge K$, choose $c_0$ with
$\|\Hlim f_k - \Hc f_k\| < \varepsilon/2$
for $c \ge c_0$.

So $\|g - \Hc f_k\| < \varepsilon$
for $k \ge K$, $c \ge c_0$.

\textit{Step 3.}
$\|\Hc\| \le 1$ and $f_k \to 0$:
$\|\Hc f_k\| \to 0$ for fixed $c$.

\textit{Conclusion.}
$\|g\| \le \|g - \Hc f_k\| + \|\Hc f_k\|
< \varepsilon + \|\Hc f_k\|$;
letting $k \to \infty$: $\|g\| \le \varepsilon$.
Since $\varepsilon$ was arbitrary, $g = 0$.
\end{proof}

\begin{remark}[Unconditional status]
\label{rem:closable-unconditional}
Closability uses only \eqref{VIII6}
(SOT-convergence, Paper~X)
and $\|\Hc\| \le 1$.
No completeness, no spectral definition.
\end{remark}

\begin{remark}[Uniqueness under completeness]
\label{rem:closure-unique}
Under Hypothesis~\ref{hyp:decay},
$\overline{\Hlim}$ is the unique self-adjoint
operator with eigensystem
$\{(\lambda_n^{(\infty)}, \Phi_n^{(\infty)})\}$
(spectral theorem;
\cite{ReedSimon1975}, Theorem~VIII.5).
\end{remark}

\section{Operator Norm Convergence Rate}
\label{sec:opnorm}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Strong convergence of eigenvectors]
\label{lem:eigenvector-strong}
For each fixed $n \ge 0$:
$\Phi_n^{(c)} \to \Phi_n^{(\infty)}$
strongly in $L^2(\mathbb{R})$.
\end{lemma}

\begin{proof}
Weak convergence (Proposition~\ref{prop:weak-limits})
plus norm-preservation
($\|\Phi_n^{(c)}\| = 1 = \|\Phi_n^{(\infty)}\|$)
implies strong convergence in a Hilbert space
(\cite{ReedSimon1975}, Prop.~II.11).
\end{proof}

\begin{remark}[The $\lambda_n^{(\infty)} = 0$ case]
\label{rem:lambda-zero}
For $\lambda_n^{(\infty)} = 0$:
the contribution $\lambda_n^{(c)} \Phi_n^{(c)} \to 0$
exponentially; no separate tracking is needed.
\end{remark}

\begin{lemma}[Eigenfunction alignment]
\label{lem:eigenvector-alignment}
For each \emph{fixed} $n \ge 0$:
$\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|_{L^2} \to 0$
as $c \to \infty$.

More precisely:
$\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|^2
= 2(1 - \mathrm{Re}\,\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}})
\to 0$.

The rate $O(c^{-1/4})$ is \emph{heuristically
consistent} with the eigenvalue rate \eqref{VIII3}
(see Remark~\ref{rem:rate-justification}),
but has not been proved uniformly over
$n \le \kappa c$.
For the operator-norm bound in
Theorem~\ref{thm:opnorm},
uniform alignment is listed as a separate
hypothesis (Hypothesis~\ref{hyp:uniform-alignment}).
\end{lemma}

\begin{proof}
Since $\Phi_n^{(c)} \rightharpoonup \Phi_n^{(\infty)}$
(Proposition~\ref{prop:weak-limits}),
$\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}} \to
\|\Phi_n^{(\infty)}\|^2 = 1$,
giving $\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\| \to 0$.
\end{proof}

\begin{remark}[Rate heuristic]
\label{rem:rate-justification}
For fixed $n$, the rate
$\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|
= O(c^{-1/4})$ follows heuristically from:
eigenvalue rate $c^{-1/4}$ (\eqref{VIII3})
divided by the eigenvalue gap
$\lambda_n^{(c)} - \lambda_{n+1}^{(c)} \asymp c^{-1/2}$
(Paper~VIII, Thm.~4.1),
yielding $c^{-3/4}/c^{-1/2} = c^{-1/4}$
from the perturbation-theoretic inequality
(see \cite{Kato1966}, Thm.~V.4.10).
This argument uses the \emph{strong}
rate \eqref{VIII2} (per vector), not
the operator norm, so it is not circular.
However, extending this pointwise bound
uniformly to all $n \le \kappa c$ requires
additional control
(Hypothesis~\ref{hyp:uniform-alignment}).
\end{remark}

\begin{hypothesis}[Uniform eigenfunction alignment]
\label{hyp:uniform-alignment}
For all $c \ge c_0(\kappa)$ and all $n \le \kappa c$:
\[
  \sup_{n \le \kappa c}
  \|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|_{L^2}
  \le C_\kappa\, c^{-1/4}.
\]
\end{hypothesis}

\begin{remark}[Depth of Hypothesis~\ref{hyp:uniform-alignment}]
\label{rem:uniform-alignment-status}
Hypothesis~\ref{hyp:uniform-alignment} is
stronger than the pointwise statement of
Lemma~\ref{lem:eigenvector-alignment}:
it requires the $c^{-1/4}$ bound uniformly
over all $n \le \kappa c$,
not just for each fixed $n$.
This is a \emph{global spectral problem}:
controlling the alignment of eigenvectors
uniformly across the entire sub-threshold
regime is in depth comparable to
semiclassical completeness problems,
and is not expected to follow from
simple perturbation-theoretic arguments.
The uniform version would follow from a
uniform eigenvalue-gap lower bound
$\lambda_n^{(c)} - \lambda_{n+1}^{(c)} \ge \delta_\kappa c^{-1/2}$
for all $n \le \kappa c$ (open since Paper~VIII,
Rem.~4.2),
or from a uniform resolvent bound from Paper~X.
In the absence of such a bound,
Hypothesis~\ref{hyp:uniform-alignment}
should be regarded as a non-trivial
open assumption
(Problem~\ref{prob:uniform-alignment}).
\end{remark}

\begin{theorem}[Operator norm rate]
\label{thm:opnorm}
Assume Hypothesis~\ref{hyp:uniform-alignment}.
Then for all $c \ge c_0(\kappa)$:
\begin{equation}
\label{eq:opnorm-rate}
  \|\Hc - \Hlim\|_{L^2 \to L^2}
  \;\le\; C_\kappa\, c^{-1/4}.
\end{equation}
\end{theorem}

\begin{proof}
Let $\|f\| = 1$.
Decompose:
\begin{equation}
\label{eq:opnorm-decomp}
  \|(\Hc - \Hlim)f\|^2
  \le 2 A(c,f) + 2 E_\kappa(c,f),
\end{equation}
where $A(c,f)$ is the eigenvalue-sum term
and $E_\kappa(c,f)$ collects cross terms
and the exponential tail.

\textit{Main sum} $A(c,f)$:
$|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$ uniformly (\eqref{VIII3})
and $\sum_n |\ip{f}{\Phi_n^{(c)}}|^2 \le 1$
(Bessel), so $A(c,f) \le C_\kappa^2 c^{-1/2}$.
This bound is \emph{unconditional}.

\textit{Tail} $n > \kappa c$:
eigenvalues $\lambda_n^{(c)} \le e^{-\gamma_\kappa c}$
are exponentially small; contribution negligible.

\textit{Cross terms} $E_\kappa(c,f)$:
These arise from $\Phi_n^{(c)} \ne \Phi_n^{(\infty)}$.
By Hypothesis~\ref{hyp:uniform-alignment},
$\sup_{n \le \kappa c}
\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|
\le C_\kappa c^{-1/4}$,
so $E_\kappa(c,f) = O(c^{-1/4})$.

Combining: $\|\Hc - \Hlim\|_{L^2 \to L^2}
\le C_\kappa c^{-1/4}$.
\end{proof}

\begin{remark}[Unconditional partial bound]
\label{rem:opnorm-unconditional-partial}
Without Hypothesis~\ref{hyp:uniform-alignment},
the eigenvalue-sum term alone gives:
$\|\Hc - \Hlim\|_{L^2 \to L^2}
\le C_\kappa c^{-1/4} + R(c)$,
where $R(c) \to 0$ (from SOT-convergence)
but whose rate is unknown.
The full $O(c^{-1/4})$ rate is conditional.
\end{remark}

\begin{remark}[Domain stability]
\label{rem:domain-stability}
For $f_c$ with $\|f_c\| + \|\Hc f_c\| \le M$,
the limit points in $L^2$ lie in $\Dom(\Hlim)$.
\end{remark}

\section{Deficiency Index Analysis}
\label{sec:deficiency}
%%═══════════════════════════════════════════════════════════════════

\subsection{Setup}

By the von Neumann theory
(\cite{ReedSimon1975}, Theorem~X.2),
a symmetric operator is essentially
self-adjoint iff $n_\pm := \dim\ker(\Hlim^* \mp i)
= 0$.

\subsection{Properties of deficiency vectors}

\begin{lemma}[Orthogonality of deficiency vectors]
\label{lem:deficiency-analytic}
Let $f \in \ker(\Hlim^* + i)$.
Then $\ip{f}{\Phi_n^{(\infty)}} = 0$
for all $n \ge 0$.
\end{lemma}

\begin{proof}
From $\Hlim^* f = -if$ and symmetry of $\Hlim$:
$(\lambda_n^{(\infty)} + i)
\ip{f}{\Phi_n^{(\infty)}} = 0$.
Since $\lambda_n^{(\infty)} \in [0,1] \subset \mathbb{R}$,
$\lambda_n^{(\infty)} + i \ne 0$
($\mathrm{Im}(\lambda_n^{(\infty)} + i) = 1 \ne 0$),
so $\ip{f}{\Phi_n^{(\infty)}} = 0$.
This includes the case $\lambda_n^{(\infty)} = 0$.
\end{proof}

\begin{remark}[The $\lambda_n^{(\infty)} = 0$ case]
\label{rem:deficiency-zero-eigenvalue}
The argument $(\lambda_n^{(\infty)} + i) \ne 0$
holds unconditionally for all
$\lambda_n^{(\infty)} \in \mathbb{R}$,
including $\lambda_n^{(\infty)} = 0$.
No division by $\lambda_n^{(\infty)}$ is needed.
\end{remark}

\begin{remark}[Central reduction: Completeness $\Rightarrow$ ESSA]
\label{rem:deficiency-orthogonal}
Lemma~\ref{lem:deficiency-analytic} establishes:
\[
  \ker(\Hlim^* \pm i) \;\subseteq\;
  \bigl(\mathrm{span}\{\Phi_n^{(\infty)}\}\bigr)^\perp.
\]
This immediately yields the implication:
\[
  \{\Phi_n^{(\infty)}\} \text{ complete in }
  L^2(\mathbb{R})
  \;\Longrightarrow\;
  \ker(\Hlim^* \pm i) = \{0\}.
\]
The converse implication --- that
$n_\pm(\Hlim) = 0$ implies completeness of
$\{\Phi_n^{(\infty)}\}$ --- does \emph{not}
hold in general for a symmetric operator
with prescribed orthonormal system.
Vanishing deficiency indices control only
the deficiency spaces
$\ker(\Hlim^* \pm i)$;
they do not imply cyclicity or completeness
of a given orthonormal system without an
additional spectral representation theorem
(e.g.\ a cyclic vector for $\Hlim$,
or a spectral measure supported on
$\{\lambda_n^{(\infty)}\}$).
Such structure is not established in this paper.
\end{remark}

\subsection{The completeness hypothesis}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}_{n \ge 0}$
is complete in $L^2(\mathbb{R})$:
$\overline{\mathrm{span}\{\Phi_n^{(\infty)}\}}
= L^2(\mathbb{R})$.
\end{hypothesis}

\begin{remark}[Evidence]
\label{rem:hyp-evidence}
\begin{enumerate}
\item Each $\{\Phi_n^{(c)}\}$ is a complete ONB
      for its target space (Slepian).
\item $\Phi_n^{(\infty)} \ne 0$ for all $n$
      (\eqref{VIII4}).
\item $\Phi_n^{(c)} \to \Phi_n^{(\infty)}$
      strongly for each fixed $n$
      (Lemma~\ref{lem:eigenvector-strong}),
      consistent with limit completeness,
      but uniform control over all $n$
      is still needed.
\end{enumerate}
Rigorous proof: open
(Problem~\ref{prob:completeness}).
\end{remark}

\section{Essential Self-Adjointness}
\label{sec:esa}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Deficiency indices]
\label{thm:deficiency}
Assume Hypothesis~\ref{hyp:decay}.
Then $n_\pm(\Hlim) = 0$.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:deficiency-analytic},
every $f \in \ker(\Hlim^* + i)$ satisfies
$f \perp \Phi_n^{(\infty)}$ for all $n$.
By Hypothesis~\ref{hyp:decay},
$\{\Phi_n^{(\infty)}\}$ is complete,
so $f = 0$.
Identical for $\ker(\Hlim^* - i)$.
\end{proof}

\begin{corollary}[Essential self-adjointness]
\label{cor:esa}
Assume Hypothesis~\ref{hyp:decay}.
Then $\overline{\Hlim} = \Hlim^*$
(\cite{ReedSimon1975}, Theorem~X.3).
\end{corollary}

\begin{theorem}[Spectral identity]
\label{thm:sigma-identity}
Assume Hypothesis~\ref{hyp:decay} and
Hypothesis~CCM.
Then:
\begin{enumerate}[\rm(i)]
\item $\sigma_{\rm app}(\Hlim) = \sigma(\Hlim)$.
\item Every non-trivial zero $\rho$ of $\zeta(s)$
      lies in $\sigma(\Hlim)$.
\item All non-trivial zeros satisfy
      $\mathrm{Re}(\rho) = 1/2$.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) Essential self-adjointness gives
$\sigma_{\rm app} = \sigma$
(\cite{ReedSimon1975}, Thm.~VII.13).
(ii) Paper~VIII, Thm.~5.2 gives
$\rho \in \sigma_{\rm app}(\Hlim)$; use (i).
(iii) CCM identifies non-trivial zeros
with $\mathrm{Im}(\rho)$ of eigenvalues;
self-adjointness forces the spectrum real,
hence $\mathrm{Re}(\rho) = 1/2$.
\end{proof}

\begin{remark}[Status of the RH reduction]
\label{rem:conditional}
Theorem~\ref{thm:sigma-identity}(iii)
depends on:
(1) Hypothesis~\ref{hyp:decay} (completeness;
currently open, see Problem~\ref{prob:completeness});
(2) Hypothesis~CCM (\cite{CCM2025}).
The result is therefore a
\emph{conditional reduction},
not a proof of the Riemann hypothesis.
\end{remark}

\begin{remark}[Status of Hypothesis~CCM]
\label{rem:ccm-status}
Hypothesis~CCM (\cite{CCM2025}) posits
a spectral embedding of the non-trivial
zeros of $\zeta(s)$ into the spectrum of
$\Hlim$ via the Connes--Consani--Moscovici
construction.
This hypothesis is currently
\emph{conjectural} and is not known to be
equivalent to the Riemann hypothesis.
In particular:
\begin{itemize}
\item the spectral assignment
      $\rho \mapsto \sigma_{\rm app}(\Hlim)$
      depends on the precise formulation
      of the CCM spectral triple and
      on the identification of zeros
      with spectral parameters;
\item it is not established that the converse
      holds: elements of $\sigma(\Hlim)$
      need not correspond to zeros of $\zeta$
      without further input.
\end{itemize}
The conclusions of
Theorem~\ref{thm:sigma-identity} are
therefore contingent on both
Hypothesis~\ref{hyp:decay} and
Hypothesis~CCM in an essential way.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Completeness of $\{\Phi_n^{(\infty)}\}$]
\label{prob:completeness}
Prove Hypothesis~\ref{hyp:decay}:
$\{\Phi_n^{(\infty)}\}_{n \ge 0}$ is complete
in $L^2(\mathbb{R})$.
By Theorem~\ref{thm:deficiency},
completeness implies $n_\pm(\Hlim) = 0$,
hence essential self-adjointness.
The converse (whether ESSA implies completeness
of the prescribed system $\{\Phi_n^{(\infty)}\}$)
is not established; it would require
a cyclic vector or spectral measure argument
(see Remark~\ref{rem:deficiency-orthogonal}).
\emph{Direct approach}: show that no nonzero
$f \in L^2$ is orthogonal to all
$\Phi_n^{(\infty)}$, using non-cancellation
\eqref{VIII4} uniformly over $n$.
\end{problem}

\begin{problem}[Uniform eigenfunction alignment]
\label{prob:uniform-alignment}
Prove Hypothesis~\ref{hyp:uniform-alignment}:
$\sup_{n \le \kappa c}
\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|
\le C_\kappa c^{-1/4}$.
A proof would follow from a uniform
eigenvalue-gap bound
$\lambda_n^{(c)} - \lambda_{n+1}^{(c)}
\ge \delta_\kappa c^{-1/2}$ for all $n \le \kappa c$
(open since Paper~VIII, Rem.~4.2),
or from a uniform resolvent bound
from Paper~X.
\end{problem}

\begin{problem}[Sharp operator-norm rate]
\label{prob:opnorm-sharp}
Is $\|\Hc - \Hlim\|_{L^2 \to L^2}
\asymp c^{-1/4}$?
Upper bound: Theorem~\ref{thm:opnorm}
(conditional).
Lower bound: requires sharp eigenvalue-gap
estimate (open).
\end{problem}

\begin{problem}[Density of $\{\lambda_n^{(\infty)}\}$]
\label{prob:density}
Is $\{\lambda_n^{(\infty)} : n \ge 0\}$
dense in $[0,1]$?
By Paper~XI, Thm.~4.1, this is equivalent
to $\sigma(\Hlim) = [0,1]$.
\end{problem}

\begin{problem}[Unconditional essential self-adjointness]
\label{prob:esa-unconditional}
Prove essential self-adjointness of $\Hlim$
without Hypothesis~\ref{hyp:decay}.
Note that this is \emph{not} automatically
equivalent to completeness of
$\{\Phi_n^{(\infty)}\}$ (see
Remark~\ref{rem:deficiency-orthogonal})
without further spectral structure.
\end{problem}
