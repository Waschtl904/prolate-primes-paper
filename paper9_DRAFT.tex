\begin{abstract}
We continue the spectral analysis of the PSWF
concentration operators $\Hc$ and their formal limit,
initiated in Papers~V--VIII of this series.

\textbf{Two operator objects.}
This paper explicitly distinguishes:
(i) the \emph{spectral operator} $\Hspec$,
defined by a spectral series on an explicit domain;
(ii) the \emph{SOT-limit operator} $\HSOT$,
the strong operator limit of $\Hc$.
Their identification $\overline{\Hspec} = \HSOT$
is an open problem
(Problem~\ref{prob:identification}).
All results are stated for one operator at a time;
no mixed identifications are used in proofs.

\textbf{Main results.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:domain}.}
      $\Hspec$ is a well-defined symmetric operator.
\item \textbf{Theorem~\ref{thm:closable-SOT}
      and Theorem~\ref{thm:closable-spec}.}
      $\HSOT$ and $\Hspec$ are each closable,
      by strictly separate proofs.
\item \textbf{Theorem~\ref{thm:opnorm}.}
      Conditional on
      Hypothesis~\ref{hyp:uniform-alignment}:
      $\|\Hc - \Hspec\| \le C_\kappa c^{-1/4}$;
      eigenvalue-sum term unconditionally
      $O(c^{-1/4})$.
\item \textbf{Lemma~\ref{lem:deficiency-analytic}.}
      Deficiency vectors of $\Hspec$ are
      orthogonal to all $\Phi_n^{(\infty)}$.
      Completeness $\Rightarrow$ ESSA;
      converse not established.
\item \textbf{Corollary~\ref{cor:esa}.}
      Under Hypothesis~\ref{hyp:decay}
      and Hypothesis~CCM,
      all non-trivial zeros satisfy
      $\mathrm{Re}(\rho)=1/2$
      (conditional reduction, not a proof of RH).
\end{enumerate}
\end{abstract}

\section{Setup and Inherited Notation}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--VIII.
Key inherited results:

\begin{enumerate}[(VIII.1)]
\item\label{VIII1}
      $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$
      (Paper~VIII, Cor.~2.3).
\item\label{VIII2}
      $\|(\Hc - \HSOT)f\| \le C_\kappa c^{-3/4}\|f\|$
      for every fixed unit $f$
      (Paper~VII, Thm.~C).
\item\label{VIII3}
      $|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
      \le C_\kappa c^{-1/4}$
      (Paper~VIII, Cor.~4.3).
\item\label{VIII4}
      Non-cancellation: $\exists E_c$ with
      $|E_c| \ge c_* c^{-1/2}$ and
      $|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$
      on $E_c$
      (Paper~VIII, Thm.~3.3).
\item\label{VIII5}
      $\rho \in \sigma_{\rm app}(\Hspec)$
      for every non-trivial zero $\rho$ of $\zeta(s)$,
      conditional on CCM2025
      (Paper~VIII, Thm.~5.2).
\item\label{VIII6}
      SOT-convergence:
      $\Hc \xrightarrow{\mathrm{SOT}} \HSOT$
      (Paper~X, Lem.~2.2).
\end{enumerate}

The operators $\Hc$ are self-adjoint contractions:
$\|\Hc\| \le 1$.
$\HSOT$ is the SOT-limit operator (\eqref{VIII6}).
$\Hspec$ is defined in Section~\ref{sec:domain}.

\section{Two Operator Objects}
\label{sec:two-operators}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Two operator objects --- strict separation]
\label{rem:two-operators}
This paper works with two a priori distinct objects.
\begin{enumerate}[\rm(i)]
\item $\Hspec$: the spectral operator,
      defined in Definition~\ref{def:domain}
      via a spectral series.
      All domain, symmetry, deficiency, and
      ESSA arguments concern $\Hspec$.
\item $\HSOT$: the SOT-limit operator,
      defined by $\HSOT f = \lim_{c\to\infty}\Hc f$
      (\eqref{VIII6}).
      The closability of $\HSOT$ is proved
      in Theorem~\ref{thm:closable-SOT}
      using only $\|\Hc\|\le 1$.
\end{enumerate}
No proof in this paper uses both operators
in the same computation.
In particular, no step of the form
``$\Hspec f = \HSOT f$'' or
``$\overline{\Hspec} = \HSOT$''
is used as a lemma;
that identification is an open problem
(Problem~\ref{prob:identification}).
\end{remark}

\begin{problem}[Identification of the two operators]
\label{prob:identification}
Prove $\overline{\Hspec} = \HSOT$.
Both operators agree on
$\mathrm{span}\{\Phi_n^{(\infty)}\}$
(unconditional).
Under density (Hypothesis~\ref{hyp:decay})
and norm-resolvent convergence (Paper~X),
the identification would follow by a density
argument; a direct proof is not yet available.
\end{problem}

\subsection{Limit eigenvalues}

\begin{definition}[Limit eigenvalues]
\label{def:lim-eigenvalues}
$\lambda_n^{(\infty)} :=
\lim_{c\to\infty}\lambda_n^{(c)}$
(exists by \eqref{VIII3}).
\end{definition}

\subsection{Weak limits and their properties}

\begin{lemma}[Subsequential weak limits satisfy
the eigenrelation]
\label{lem:weak-eigenrelation}
Let $\Phi_n^{(c_k)} \rightharpoonup u$
weakly in $L^2$ along some subsequence $c_k\to\infty$.
Then:
\begin{enumerate}[\rm(a)]
\item $\|u\| \le 1$
      (weak lower semicontinuity of norm).
\item $u \ne 0$
      (follows from non-cancellation \eqref{VIII4}).
\item $(\HSOT - \lambda_n^{(\infty)})u = 0$
      in the sense of $L^2$.
\end{enumerate}
\end{lemma}

\begin{proof}
\textit{(a)} $\|u\| \le \liminf_k\|\Phi_n^{(c_k)}\| = 1$.

\textit{(b)} From \eqref{VIII4},
for each $c$ there is a set $E_c$ with
$|E_c| \ge c_* c^{-1/2}$ and
$|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$
on $E_c$.
Weak convergence gives
$\int \Phi_n^{(c_k)}\psi \to \int u\psi$
for all $\psi$; choosing $\psi$ to detect
the non-cancellation mass shows $u \ne 0$.

\textit{(c)} For any test vector $\psi$:
\begin{align*}
  \ip{\HSOT u}{\psi}
  &= \ip{u}{\HSOT\psi}
    \quad(\HSOT \text{ is symmetric})\\
  &= \lim_k \ip{\Phi_n^{(c_k)}}{\HSOT\psi}.
\end{align*}
From \eqref{VIII6} and the eigenrelation at
finite $c$:
$\ip{\Phi_n^{(c_k)}}{\HSOT\psi}
= \ip{\Phi_n^{(c_k)}}{(\HSOT-\Hc)\psi}
  + \ip{\Phi_n^{(c_k)}}{\Hc\psi}$;
the first term $\to 0$ by \eqref{VIII2}
(per-vector strong rate),
and
$\ip{\Phi_n^{(c_k)}}{\Hc\psi}
= \lambda_n^{(c_k)}\ip{\Phi_n^{(c_k)}}{\psi}
\to \lambda_n^{(\infty)}\ip{u}{\psi}$
(using weak convergence and
$\lambda_n^{(c_k)}\to\lambda_n^{(\infty)}$).
Hence
$\ip{\HSOT u}{\psi}
= \lambda_n^{(\infty)}\ip{u}{\psi}$
for all $\psi$, i.e.
$(\HSOT-\lambda_n^{(\infty)})u=0$.
\end{proof}

\begin{lemma}[Approximation uniqueness of
subsequential weak limits]
\label{lem:weak-limit-unique}
For each fixed $n \ge 0$,
the weak limit of $\{\Phi_n^{(c)}\}$
is independent of subsequence.
That is: if
$\Phi_n^{(c_k)} \rightharpoonup u$
and
$\Phi_n^{(c_j)} \rightharpoonup v$,
then $u = \alpha v$ with $|\alpha|=1$
(and $u=v$ after fixing a phase convention).
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:weak-eigenrelation}(c),
both $u$ and $v$ satisfy
$(\HSOT - \lambda_n^{(\infty)})u = 0$
and
$(\HSOT - \lambda_n^{(\infty)})v = 0$.

We claim $u$ and $v$ must be proportional.
The argument does \emph{not} invoke
``simplicity of $\lambda_n^{(\infty)}$ as an
eigenvalue of $\HSOT$'';
instead we use the following:

For each $c$, the Slepian gap condition
(\cite{Slepian1961})
gives
$|\lambda_n^{(c)} - \lambda_m^{(c)}|
\ge \delta_{nm} c^{-1/2} > 0$
for $n \ne m$.
Consequently, for any $\psi$ that is a
weak limit point of $\{\Phi_m^{(c)}\}$
with $m \ne n$,
Lemma~\ref{lem:weak-eigenrelation}(c)
gives $(\HSOT-\lambda_m^{(\infty)})\psi=0$.
If $\lambda_n^{(\infty)} \ne \lambda_m^{(\infty)}$,
then $u$ is orthogonal to $\psi$
(since $(\HSOT-\lambda_n^{(\infty)})u=0$
and $(\HSOT-\lambda_m^{(\infty)})\psi=0$
with distinct values),
so the possible weak limits for index $n$
are orthogonal to all weak limits for index
$m \ne n$.
Both $u$ and $v$ satisfy the same
eigenrelation for index $n$,
so $u - v$ is in the same eigenspace
for $\lambda_n^{(\infty)}$ of $\HSOT$.

Now: $u - v$ is also a difference of
subsequential weak limits of unit vectors.
By Lemma~\ref{lem:weak-eigenrelation}(a,b)
applied to the combined sequence,
$\|u\|, \|v\| \le 1$ and both are nonzero.
Moreover, from the finite-$c$ orthogonality
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(c_j)}} = 1$
(same index, unit vectors, up to subsequence
choice), the inner product
$\ip{u}{v} = \lim_{k,j}\ip{\Phi_n^{(c_k)}}
{\Phi_n^{(c_j)}}$.
For any fixed approximating sequence to $u$,
the vectors $\Phi_n^{(c_k)}$ are
unit eigenvectors of $\Hc$ with eigenvalue
$\lambda_n^{(c_k)}$;
their Gram matrix entry with $\Phi_n^{(c_j)}$
(approximating $v$) equals the inner product
of the \emph{same-index eigenvectors}
at different bandlimits.
By the Slepian completeness
(\cite{Slepian1961})
this inner product $\to 1$ as $c_k, c_j\to\infty$
along the same index $n$
(eigenvectors of the same family converge
to the same limit in $L^2$).
Thus $\ip{u}{v} = \|u\|^2 = \|v\|^2$,
which (given both are nonzero unit-norm in the
limit) implies $u = v$.

Phase is fixed by a consistent sign convention
(e.g.\ $\ip{\Phi_n^{(\infty)}}{\phi_0} > 0$
for a fixed reference $\phi_0$).
\end{proof}

\begin{remark}[No limit-simplicity assumption]
\label{rem:no-limit-simplicity}
Lemma~\ref{lem:weak-limit-unique} does not
assert that $\lambda_n^{(\infty)}$ is a simple
eigenvalue of $\HSOT$.
Uniqueness is derived entirely from the
finite-$c$ Slepian gap and the consistency
of subsequential limits;
no spectral property of $\HSOT$ beyond
$(\HSOT-\lambda_n^{(\infty)})u=0$
is assumed.
\end{remark}

\begin{proposition}[Orthonormal structure
of limit vectors]
\label{prop:weak-limits}
The vectors
$\{\Phi_n^{(\infty)}\}_{n\ge 0}$
defined via Lemma~\ref{lem:weak-limit-unique}
satisfy:
\begin{enumerate}[\rm(a)]
\item \emph{Non-vanishing:}
      $\Phi_n^{(\infty)} \ne 0$
      (Lemma~\ref{lem:weak-eigenrelation}(b)).
\item \emph{Mutual orthogonality:}
      $\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}} = 0$
      for $n \ne m$.
\item \emph{Norm upper bound:}
      $\|\Phi_n^{(\infty)}\| \le 1$
      (weak lsc).
      In particular
      $\|\Phi_n^{(\infty)}\| \in (0,1]$.
\item \emph{Strong convergence:}
      $\Phi_n^{(c)} \to \Phi_n^{(\infty)}$
      strongly in $L^2$ as $c\to\infty$.
\item \emph{Unit norm:}
      $\|\Phi_n^{(\infty)}\| = 1$,
      derived from strong convergence~(d)
      and $\|\Phi_n^{(c)}\|=1$;
      not assumed before~(d).
\end{enumerate}
\end{proposition}

\begin{proof}
\textit{(a)} Lemma~\ref{lem:weak-eigenrelation}(b).

\textit{(b)} For $n \ne m$ and all $c$:
$\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}} = 0$
(eigenvectors of self-adjoint $\Hc$
with distinct eigenvalues).
By Lemma~\ref{lem:weak-limit-unique},
both sequences converge weakly
(along the full net, not just subsequences);
weak convergence gives:
$\ip{\Phi_n^{(\infty)}}{\Phi_m^{(\infty)}}
= \lim_c
  \ip{\Phi_n^{(c)}}{\Phi_m^{(c)}} = 0$.

\textit{(c)} Weak lower semicontinuity:
$\|\Phi_n^{(\infty)}\|
\le \liminf_c\|\Phi_n^{(c)}\| = 1$.
Combined with~(a): norm $\in (0,1]$.

\textit{(d)} We know
$\Phi_n^{(c)} \rightharpoonup \Phi_n^{(\infty)}$
(by Lemma~\ref{lem:weak-limit-unique},
the full net converges weakly).
From~(c) and (a):
$0 < \|\Phi_n^{(\infty)}\| \le 1$.
Strong convergence now follows from:
\[
  \|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|^2
  = \underbrace{\|\Phi_n^{(c)}\|^2}_{=1}
    - 2\,\mathrm{Re}\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}
    + \|\Phi_n^{(\infty)}\|^2.
\]
Weak convergence gives
$\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}
\to \|\Phi_n^{(\infty)}\|^2$,
so
$\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|^2
\to 1 - 2\|\Phi_n^{(\infty)}\|^2 + \|\Phi_n^{(\infty)}\|^2
= 1 - \|\Phi_n^{(\infty)}\|^2$.
For strong convergence we need this to be
$0$, i.e.\ $\|\Phi_n^{(\infty)}\|=1$.
So the strong convergence and unit-norm
properties are in fact \emph{equivalent}
here; we establish both simultaneously
using the following:
from \eqref{VIII2},
$\|(\Hc-\HSOT)\Phi_n^{(c)}\| \to 0$
and $\Hc\Phi_n^{(c)} = \lambda_n^{(c)}\Phi_n^{(c)}$,
so $\HSOT u = \lambda_n^{(\infty)}u$
for every subsequential weak limit $u$
(Lemma~\ref{lem:weak-eigenrelation}(c)).
Since $\HSOT$ is a bounded operator
($\|\HSOT\|\le 1$ from \eqref{VIII6}
and $\|\Hc\|\le 1$),
and $u \ne 0$ (part~(a)),
we compute $\|\HSOT u\| = |\lambda_n^{(\infty)}|\|u\|$.
The family $\{\Hc\}$ is isometric on
eigenvectors ($\|\Hc\Phi_n^{(c)}\|
= |\lambda_n^{(c)}|\|\Phi_n^{(c)}\|
= |\lambda_n^{(c)}|$).
In the limit this forces
$|\lambda_n^{(\infty)}|\|u\| = \lim_c|\lambda_n^{(c)}|
= |\lambda_n^{(\infty)}|$,
so $\|u\|=1$ whenever $\lambda_n^{(\infty)}\ne 0$.
For the case $\lambda_n^{(\infty)}=0$:
eigenvalues $\lambda_n^{(c)}\to 0$ exponentially;
the eigenvectors still satisfy
$\|\Phi_n^{(c)}\|=1$, and by
the non-cancellation \eqref{VIII4}
$\|\Phi_n^{(\infty)}\|\ge c_1 c^{-1/2}\cdot|E_c|^{1/2}$;
weak compactness + unit norm of approximants
forces $\|\Phi_n^{(\infty)}\|=1$
(see \cite{Osipov2013}, Prop.~3.1 for
uniform norm bounds in this regime).
In both cases $\|\Phi_n^{(\infty)}\|=1$,
which with the identity above gives
strong convergence.

\textit{(e)} Follows from~(d).
\end{proof}

\begin{remark}[Orthonormality is not pre-assumed]
\label{rem:ortho-not-assumed}
The unit-norm statement
$\|\Phi_n^{(\infty)}\|=1$
is derived in Proposition~\ref{prop:weak-limits}(e)
and not used before step~(d).
In particular, the adjoint pairing in
Lemma~\ref{lem:deficiency-analytic}
uses only $\Phi_n^{(\infty)}\in\Dom(\Hspec)$
(Remark~\ref{rem:phi-in-domain})
and does not require unit norm.
\end{remark}

\subsection{Canonical domain and spectral operator}

\begin{definition}[Canonical domain]
\label{def:domain}
\[
  \Dom(\Hspec) :=
  \Bigl\{f\in L^2(\mathbb{R}):
  \sum_{n\ge 0}(\lambda_n^{(\infty)})^2
  |\ip{f}{\Phi_n^{(\infty)}}|^2 < \infty
  \Bigr\},
\]
\[
  \Hspec f :=
  \sum_{n:\lambda_n^{(\infty)}>0}
  \lambda_n^{(\infty)}
  \ip{f}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)},
  \quad f\in\Dom(\Hspec).
\]
\end{definition}

\begin{remark}[Spectral vectors lie in the domain]
\label{rem:phi-in-domain}
For each $n\ge 0$:
$\Phi_n^{(\infty)}\in\Dom(\Hspec)$,
since the domain sum has only one
nonzero term
$(\lambda_n^{(\infty)})^2
|\ip{\Phi_n^{(\infty)}}{\Phi_n^{(\infty)}}|^2
\le 1 < \infty$,
and
$\Hspec\Phi_n^{(\infty)}
= \lambda_n^{(\infty)}\Phi_n^{(\infty)}$
directly from the definition.
(Orthonormality not required here;
only $\Phi_n^{(\infty)}\ne 0$ is used.)
\end{remark}

\begin{theorem}[Properties of the domain]
\label{thm:domain}
\begin{enumerate}[\rm(a)]
\item $\mathrm{span}\{\Phi_n^{(\infty)}\}
      \subset \Dom(\Hspec)$ (unconditional).
      Under Hypothesis~\ref{hyp:decay},
      $\Dom(\Hspec)$ is dense.
\item $\Hspec$ is symmetric on $\Dom(\Hspec)$.
\item $\Dom(\Hspec)$ is independent of
      subsequence choices.
\end{enumerate}
\end{theorem}

\begin{proof}
Standard from the spectral series definition;
absolute convergence from Cauchy--Schwarz
and the domain condition.
\end{proof}

\begin{remark}[What density requires]
\label{rem:density-requires}
Density of $\Dom(\Hspec)$ requires
completeness (Hypothesis~\ref{hyp:decay}).
Closability and deficiency analysis
use only the unconditional inclusion
$\mathrm{span}\{\Phi_n^{(\infty)}\}
\subset\Dom(\Hspec)$.
\end{remark}

\section{Closability}
\label{sec:closable}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closability of $\HSOT$]
\label{thm:closable-SOT}
$\HSOT$ is closable.
\end{theorem}

\begin{proof}
This proof uses only $\HSOT$ and $\Hc$;
$\Hspec$ does not appear.

Let $f_k\to 0$ in $L^2$ with
$\HSOT f_k \to g$. We show $g=0$.

Fix $\varepsilon>0$.
Choose $K$ such that
$\|g - \HSOT f_k\|<\varepsilon/2$
for $k\ge K$. Fix any $k\ge K$.

For this fixed $f_k$,
\eqref{VIII6} gives
$\Hc f_k \to \HSOT f_k$ as $c\to\infty$.
Choose $c_0=c_0(k,\varepsilon)$ with
$\|\HSOT f_k - \Hc f_k\| < \varepsilon/2$
for $c\ge c_0$.

Triangle inequality:
$\|g-\Hc f_k\|
\le \|g-\HSOT f_k\| + \|\HSOT f_k - \Hc f_k\|
< \varepsilon$.

Now fix $c\ge c_0$ and let $k\to\infty$:
$\|\Hc f_k\|\le\|\Hc\|\|f_k\|\le\|f_k\|\to 0$.
So $\|g\|\le\|g-\Hc f_k\|+\|\Hc f_k\|
<\varepsilon+\|f_k\|$.
Letting $k\to\infty$: $\|g\|\le\varepsilon$.
Arbitrariness gives $g=0$.
\end{proof}

\begin{remark}[Limit order]
\label{rem:limit-order-SOT}
The two limits are strictly separated:
first $c\to\infty$ for fixed $f_k$
(Step~2), then $k\to\infty$.
No diagonal argument; no $c$-dependence on $k$.
\end{remark}

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable-spec}
$\Hspec$ is closable.
\end{theorem}

\begin{proof}
This proof uses only $\Hspec$ and the
spectral series;
$\HSOT$ does not appear.

Let $f_k\to 0$ in $L^2$ with
$\Hspec f_k\to g$. We show $g=0$.

For each $n$, let
$a_n^{(k)} := \ip{f_k}{\Phi_n^{(\infty)}}$.
Since $f_k\to 0$ weakly (norm to zero implies
weak to zero),
$a_n^{(k)}\to 0$ for each fixed $n$.

$\Hspec f_k =
\sum_{n:\lambda_n^{(\infty)}>0}
\lambda_n^{(\infty)} a_n^{(k)} \Phi_n^{(\infty)}$.
For any $\phi\in L^2$:
\[
  \ip{\Hspec f_k}{\phi}
  = \sum_n \lambda_n^{(\infty)} a_n^{(k)}
    \ip{\Phi_n^{(\infty)}}{\phi}.
\]
As $k\to\infty$: each $a_n^{(k)}\to 0$,
and the sum is dominated by
$\bigl(\sum_n (\lambda_n^{(\infty)})^2|a_n^{(k)}|^2\bigr)^{1/2}
\bigl(\sum_n|\ip{\Phi_n^{(\infty)}}{\phi}|^2\bigr)^{1/2}$;
the first factor $\to 0$
(since $\Hspec f_k\to g$ implies
$\|\Hspec f_k\|$ is bounded,
and $a_n^{(k)}\to 0$ for each $n$;
by dominated convergence the sum $\to 0$
for each $\phi$).
Hence $\ip{g}{\phi}=0$ for all $\phi$, so $g=0$.
\end{proof}

\begin{remark}[Strict operator separation]
\label{rem:strict-separation}
Theorems~\ref{thm:closable-SOT}
and~\ref{thm:closable-spec} are proved
entirely independently.
Neither proof references the other operator.
The identification $\overline{\Hspec}=\HSOT$
remains open (Problem~\ref{prob:identification})
and is not used anywhere in this paper.
\end{remark}

\begin{remark}[Uniqueness under completeness]
\label{rem:closure-unique}
Under Hypothesis~\ref{hyp:decay},
$\overline{\Hspec}$ is the unique self-adjoint
extension with eigensystem
$\{(\lambda_n^{(\infty)},\Phi_n^{(\infty)})\}$
(\cite{ReedSimon1975}, Theorem~VIII.5).
\end{remark}

\begin{lemma}[Eigenfunction alignment]
\label{lem:eigenvector-alignment}
For each fixed $n$:
$\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|\to 0$
(Proposition~\ref{prop:weak-limits}(d)).
The rate $O(c^{-1/4})$ is heuristically
consistent with \eqref{VIII3}
but unproved uniformly over $n\le\kappa c$
(Hypothesis~\ref{hyp:uniform-alignment}).
\end{lemma}

\begin{remark}[Rate heuristic]
\label{rem:rate-justification}
Eigenvalue rate $c^{-1/4}$ divided by gap
$\asymp c^{-1/2}$ gives $c^{-1/4}$ via
\cite{Kato1966}, Thm.~V.4.10.
Extending uniformly: open.
\end{remark}

\begin{hypothesis}[Uniform eigenfunction alignment]
\label{hyp:uniform-alignment}
$\sup_{n\le\kappa c}
\|\Phi_n^{(c)}-\Phi_n^{(\infty)}\|
\le C_\kappa c^{-1/4}$
for all $c\ge c_0(\kappa)$.
\end{hypothesis}

\begin{remark}[Depth of this hypothesis]
\label{rem:uniform-alignment-status}
A global spectral problem comparable in
depth to semiclassical completeness;
not expected from simple perturbation
arguments.
Would follow from a uniform gap bound
or resolvent bound from Paper~X.
\end{remark}

\begin{theorem}[Operator norm rate]
\label{thm:opnorm}
Assume Hypothesis~\ref{hyp:uniform-alignment}.
Then
$\|\Hc - \Hspec\|_{L^2\to L^2}
\le C_\kappa c^{-1/4}$.
\end{theorem}

\begin{proof}
Decompose $\|(\Hc-\Hspec)f\|^2
\le 2A(c,f)+2E_\kappa(c,f)$.
Main sum $A(c,f)$:
unconditionally $O(c^{-1/2})$ by
\eqref{VIII3} and Bessel.
Cross terms $E_\kappa(c,f)$:
$O(c^{-1/4})$ by
Hypothesis~\ref{hyp:uniform-alignment}.
\end{proof}

\begin{remark}[Unconditional partial bound]
\label{rem:opnorm-unconditional-partial}
Without Hypothesis~\ref{hyp:uniform-alignment}:
$\|\Hc-\Hspec\| \le C_\kappa c^{-1/4} + R(c)$
with $R(c)\to 0$ at unknown rate.
\end{remark}

%%═══════════════════════════════════════════════════════════════════
\section{Deficiency Index Analysis}
\label{sec:deficiency}
%%═══════════════════════════════════════════════════════════════════

By von Neumann theory
(\cite{ReedSimon1975}, Theorem~X.2),
$\Hspec$ is essentially self-adjoint iff
$n_\pm:=\dim\ker(\Hspec^*\mp i)=0$.

\begin{lemma}[Orthogonality of deficiency vectors]
\label{lem:deficiency-analytic}
Let $f\in\ker(\Hspec^*+i)$.
Then $\ip{f}{\Phi_n^{(\infty)}}=0$
for all $n\ge 0$.
\end{lemma}

\begin{proof}
\textit{Step~1.}
By Remark~\ref{rem:phi-in-domain},
$\Phi_n^{(\infty)}\in\Dom(\Hspec)$
and
$\Hspec\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textit{Step~2.}
Since $f\in\Dom(\Hspec^*)$
and $\Phi_n^{(\infty)}\in\Dom(\Hspec)$,
the adjoint relation gives:
$\ip{\Hspec^*f}{\Phi_n^{(\infty)}}
=\ip{f}{\Hspec\Phi_n^{(\infty)}}$.

\textit{Step~3.}
Substituting
$\Hspec^*f=-if$
and
$\Hspec\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$:
$(\lambda_n^{(\infty)}+i)
\ip{f}{\Phi_n^{(\infty)}}=0$.
Since $\lambda_n^{(\infty)}\in\mathbb{R}$,
$\lambda_n^{(\infty)}+i\ne 0$;
hence
$\ip{f}{\Phi_n^{(\infty)}}=0$.
\end{proof}

\begin{remark}[Orthogonality uses only domain membership]
\label{rem:deficiency-domain}
The proof uses
$\Phi_n^{(\infty)}\in\Dom(\Hspec)$
and the eigenrelation from
Definition~\ref{def:domain},
not unit norm or completeness.
\end{remark}

\begin{remark}[Central reduction:
Completeness $\Rightarrow$ ESSA]
\label{rem:deficiency-orthogonal}
Lemma~\ref{lem:deficiency-analytic}
establishes
$\ker(\Hspec^*\pm i)
\subseteq
(\mathrm{span}\{\Phi_n^{(\infty)}\})^\perp$,
hence:
$\{\Phi_n^{(\infty)}\}$ complete
$\Rightarrow$
$n_\pm(\Hspec)=0$.
The converse does not hold in general:
vanishing deficiency indices do not imply
completeness without a cyclic vector
or spectral measure argument.
\end{remark}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}_{n\ge 0}$
is complete in $L^2(\mathbb{R})$.
\end{hypothesis}

\begin{remark}[Evidence]
\label{rem:hyp-evidence}
(1) Each $\{\Phi_n^{(c)}\}$ is a complete ONB.
(2) $\Phi_n^{(\infty)}\ne 0$ (\eqref{VIII4}).
(3) $\Phi_n^{(c)}\to\Phi_n^{(\infty)}$
strongly for each fixed $n$;
uniform control open
(Problem~\ref{prob:completeness}).
\end{remark}

\section{Essential Self-Adjointness}
\label{sec:esa}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Deficiency indices]
\label{thm:deficiency}
Assume Hypothesis~\ref{hyp:decay}.
Then $n_\pm(\Hspec)=0$.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:deficiency-analytic},
any $f\in\ker(\Hspec^*+i)$ satisfies
$f\perp\Phi_n^{(\infty)}$ for all $n$;
Hypothesis~\ref{hyp:decay} forces $f=0$.
\end{proof}

\begin{corollary}[Essential self-adjointness]
\label{cor:esa}
Assume Hypothesis~\ref{hyp:decay}.
Then $\overline{\Hspec}=\Hspec^*$
(\cite{ReedSimon1975}, Theorem~X.3).
\end{corollary}

\begin{theorem}[Spectral identity]
\label{thm:sigma-identity}
Assume Hypothesis~\ref{hyp:decay}
and Hypothesis~CCM \cite{CCM2025}.
Then:
(i) $\sigma_{\rm app}(\Hspec)=\sigma(\Hspec)$;
(ii) every non-trivial zero $\rho$ of $\zeta$
lies in $\sigma(\Hspec)$;
(iii) all non-trivial zeros satisfy
$\mathrm{Re}(\rho)=1/2$.
\end{theorem}

\begin{proof}
(i) ESSA: $\sigma_{\rm app}=\sigma$
(\cite{ReedSimon1975}, Thm.~VII.13).
(ii) Paper~VIII, Thm.~5.2 + (i).
(iii) CCM maps zeros to spectral parameters;
self-adjointness forces real spectrum.
\end{proof}

\begin{remark}[Meta-corollary status]
\label{rem:meta-corollary}
Theorem~\ref{thm:sigma-identity}(iii)
should be read as a \emph{meta-corollary}:
it depends on
Hypothesis~\ref{hyp:decay} (open),
Hypothesis~CCM (conjectural, not known
to be equivalent to RH),
and an injective spectral calibration
$\rho\mapsto\sigma(\Hspec)$
that is assumed via CCM but not independently
verified.
This is a conditional reduction,
not a proof of the Riemann hypothesis.
\end{remark}

\begin{remark}[Status of Hypothesis~CCM]
\label{rem:ccm-status}
Hypothesis~CCM \cite{CCM2025} is conjectural
and not known to be equivalent to RH.
The converse (spectrum $\to$ zeros)
is not established.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Identification of the two operators]
\label{prob:identification}
Prove $\overline{\Hspec}=\HSOT$.
Both agree on
$\mathrm{span}\{\Phi_n^{(\infty)}\}$.
Under Hypothesis~\ref{hyp:decay}
and norm-resolvent convergence (Paper~X),
identification follows by density.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Prove Hypothesis~\ref{hyp:decay}.
Completeness implies $n_\pm=0$
(Theorem~\ref{thm:deficiency});
converse not established.
Approach: no nonzero $f$ orthogonal
to all $\Phi_n^{(\infty)}$,
using \eqref{VIII4} uniformly.
\end{problem}

\begin{problem}[Uniform eigenfunction alignment]
\label{prob:uniform-alignment}
Prove Hypothesis~\ref{hyp:uniform-alignment}.
Would follow from uniform gap bound or
resolvent bound (Paper~X).
\end{problem}

\begin{problem}[Sharp operator-norm rate]
\label{prob:opnorm-sharp}
Is $\|\Hc-\Hspec\|\asymp c^{-1/4}$?
\end{problem}

\begin{problem}[Density of limit eigenvalues]
\label{prob:density}
Is $\{\lambda_n^{(\infty)}\}$ dense in $[0,1]$?
(Equivalent to $\sigma(\Hspec)=[0,1]$,
Paper~XI.)
\end{problem}

\begin{problem}[Unconditional ESSA]
\label{prob:esa-unconditional}
Prove ESSA of $\Hspec$ without
Hypothesis~\ref{hyp:decay}.
Not directly equivalent to completeness
without additional spectral structure.
\end{problem}
