\begin{abstract}
We continue the spectral analysis of the PSWF
concentration operators $\Hc$ and their formal limit
$\Hlim$, initiated in Papers~V--VIII of this series.

The central obstacle identified in Paper~VIII
\cite{PaperVIII} is the gap between
$\sigma_{\rm app}(\Hlim)$ and $\sigma(\Hlim)$:
the approximate spectrum is too large unless
$\Hlim$ is (essentially) self-adjoint.
Paper~VIII states this as Conjecture~8.1.

\textbf{Main results of this paper.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:domain}
      (Canonical domain).}
      The operator $\Hlim$ is defined
      via a spectral series on an explicitly described domain.
      The domain contains all finite linear combinations
      of $\{\Phi_n^{(\infty)}\}$ and
      is \emph{dense} in $L^2(\mathbb{R})$
      conditionally on Hypothesis~\ref{hyp:decay}.
      The operator is symmetric on this domain.

\item \textbf{Theorem~\ref{thm:closable}
      (Closability).}
      $\Hlim$ is closable,
      proved directly from SOT-convergence
      $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
      without invoking completeness of
      $\{\Phi_n^{(\infty)}\}$.

\item \textbf{Theorem~\ref{thm:opnorm}
      (Operator norm rate).}
      $\|\Hc - \Hlim\|_{L^2 \to L^2}
      \le C_\kappa\, c^{-1/4}$.
      The eigenfunction alignment step uses
      weak-to-strong convergence
      (norm-preservation + weak convergence)
      rather than Kato--Rellich perturbation theory,
      avoiding any operator-norm circularity.

\item \textbf{Theorem~\ref{thm:deficiency}
      (Deficiency index control).}
      Every $f \in \ker(\Hlim^* \pm i)$
      is orthogonal to all $\Phi_n^{(\infty)}$.
      Under Hypothesis~\ref{hyp:decay},
      this yields $n_\pm(\Hlim) = 0$.

\item \textbf{Corollary~\ref{cor:esa}
      (Essential self-adjointness, conditional).}
      Under Hypothesis~\ref{hyp:decay},
      $\Hlim$ is essentially self-adjoint,
      $\sigma_{\rm app}(\Hlim) = \sigma(\Hlim)$,
      and --- combined with Hypothesis~CCM ---
      all non-trivial zeros $\rho$ of $\zeta(s)$
      satisfy $\mathrm{Re}(\rho) = 1/2$.
\end{enumerate}
\end{abstract}

\section{Setup and Inherited Notation}
\label{sec:setup}

We retain all notation from Papers~V--VIII.
Key inherited results:

\begin{enumerate}[(VIII.1)]
\item\label{VIII1}
      $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$
      (Paper~VIII, Cor.~2.3).
\item\label{VIII2}
      $\|(\Hc - \Hlim)f\| \le C_\kappa c^{-3/4}\|f\|$
      for every fixed unit $f$ (strong rate;
      Paper~VII, Thm.~C).
\item\label{VIII3}
      $|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
      \le C_\kappa c^{-1/4}$
      (Paper~VIII, Cor.~4.3, telescoping sum).
\item\label{VIII4}
      Non-cancellation: there exist $E_c$ with
      $|E_c| \ge c_* c^{-1/2}$ and
      $|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$ on $E_c$
      (Paper~VIII, Thm.~3.3).
\item\label{VIII5}
      $\rho \in \sigma_{\rm app}(\Hlim)$ for every
      non-trivial zero $\rho$ of $\zeta(s)$,
      conditional on CCM2025 (Paper~VIII, Thm.~5.2).
\item\label{VIII6}
      SOT-convergence: $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
      in $\mathcal{B}(L^2(\mathbb{R}))$
      (Paper~X, Lem.~2.2; cited here for the
      closability proof in Section~\ref{sec:closable}).
\end{enumerate}

The operators $\Hc$ are self-adjoint contractions
on $L^2(\mathbb{R})$ (Slepian concentration operators;
$\|\Hc\|_{L^2 \to L^2} \le 1$ since all eigenvalues
lie in $(0,1)$; see Papers~V--VI).
The operator $\Hlim$ is defined in
Section~\ref{sec:domain}.

\section{Canonical Domain of $\Hlim$}
\label{sec:domain}
%%═══════════════════════════════════════════════════════════════════

\subsection{The limit eigenvalues and eigenvectors}

Let $\{\Phi_n^{(c)}\}_{n \ge 0}$ be the orthonormal
eigenbasis of $\Hc$ with eigenvalues $\lambda_n^{(c)}$.
By \eqref{VIII3}, for each fixed $n$,
$\lambda_n^{(c)} \to \lambda_n^{(\infty)}$ as $c \to \infty$.

\begin{definition}[Limit eigenvalues]
\label{def:lim-eigenvalues}
For $n \ge 0$, define
\[
  \lambda_n^{(\infty)}
  := \lim_{c \to \infty} \lambda_n^{(c)}.
\]
Existence follows from \eqref{VIII3} and
$\sum_{k \ge c} |\lambda_n^{(k+1)} - \lambda_n^{(k)}|
\le C_\kappa c^{-1/4} < \infty$.
\end{definition}

\begin{remark}[Values of $\lambda_n^{(\infty)}$]
\label{rem:lambda-values}
For $n$ in the sub-threshold regime
$n \le (1-\delta)2c/\pi$, the Slepian
counting law gives $\lambda_n^{(c)} \to 1$,
so $\lambda_n^{(\infty)} \in (0,1]$.
For $n$ above the threshold,
$\lambda_n^{(c)} \to 0$ exponentially,
so $\lambda_n^{(\infty)} = 0$.
In the closability and deficiency proofs below,
the case $\lambda_n^{(\infty)} = 0$ is handled
explicitly (see Remark~\ref{rem:lambda-zero}).
\end{remark}

\begin{proposition}[Weak convergence of eigenvectors]
\label{prop:weak-limits}
After passing to a subsequence (if necessary),
$\Phi_n^{(c)} \rightharpoonup \Phi_n^{(\infty)}$
weakly in $L^2(\mathbb{R})$ as $c \to \infty$,
where $\{\Phi_n^{(\infty)}\}_{n \ge 0}$ is
an orthonormal system in $L^2(\mathbb{R})$
(in particular, $\|\Phi_n^{(\infty)}\| = 1$
for all $n \ge 0$).
\end{proposition}

\begin{proof}
Boundedness (unit vectors) gives weakly convergent
subsequences via Banach--Alaoglu.
Uniform eigenvalue separation
$\lambda_n^{(c)} - \lambda_{n+1}^{(c)} \ge \delta_n > 0$
(from \cite{Slepian1961}) ensures the
subsequential limits are mutually orthogonal.
For the norm: since $\{\Phi_n^{(\infty)}\}$
is an orthonormal system, $\|\Phi_n^{(\infty)}\| = 1$
by definition of orthonormality.
\end{proof}

\subsection{The canonical domain}

\begin{definition}[Canonical domain of $\Hlim$]
\label{def:domain}
Define
\[
  \Dom(\Hlim) :=
  \Bigl\{
    f \in L^2(\mathbb{R}) :
    \sum_{n=0}^\infty
    \bigl(\lambda_n^{(\infty)}\bigr)^2
    \bigl|\ip{f}{\Phi_n^{(\infty)}}\bigr|^2
    < \infty
  \Bigr\},
\]
and for $f \in \Dom(\Hlim)$:
\[
  \Hlim f :=
  \sum_{n : \lambda_n^{(\infty)} > 0}
  \lambda_n^{(\infty)}\,
  \ip{f}{\Phi_n^{(\infty)}}\,
  \Phi_n^{(\infty)}.
\]
The sum runs only over indices with
$\lambda_n^{(\infty)} > 0$;
for $\lambda_n^{(\infty)} = 0$ no term is contributed.
\end{definition}

\begin{remark}[On the sum in Definition~\ref{def:domain}]
\label{rem:sum-convention}
Since $\lambda_n^{(\infty)} = 0$ for all
$n$ above the Slepian threshold
(Remark~\ref{rem:lambda-values}),
the sum in the definition of $\Hlim f$
is effectively finite for each $f$ in
the finite-span core, and well-defined
for all $f \in \Dom(\Hlim)$ by the
domain condition.
\end{remark}

\begin{theorem}[Properties of the canonical domain]
\label{thm:domain}
\begin{enumerate}[\rm(a)]
\item $\Dom(\Hlim)$ contains all finite
      linear combinations
      $\sum_{n=0}^N a_n \Phi_n^{(\infty)}$.
      In particular, $\Dom(\Hlim)$ is non-trivial.
      Under Hypothesis~\ref{hyp:decay}
      (completeness of $\{\Phi_n^{(\infty)}\}$),
      $\Dom(\Hlim)$ is dense in $L^2(\mathbb{R})$.
\item $\Hlim$ is symmetric on $\Dom(\Hlim)$:
      $\ip{\Hlim f}{g} = \ip{f}{\Hlim g}$
      for all $f, g \in \Dom(\Hlim)$.
\item $\Dom(\Hlim)$ is independent of
      the subsequence choice in
      Proposition~\ref{prop:weak-limits}
      (up to measure zero), since
      $\lambda_n^{(\infty)}$ is
      uniquely determined.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(a)} For a finite combination
$f = \sum_{n=0}^N a_n \Phi_n^{(\infty)}$,
the domain sum
$\sum_n (\lambda_n^{(\infty)})^2
|\ip{f}{\Phi_n^{(\infty)}}|^2
= \sum_{n=0}^N (\lambda_n^{(\infty)})^2 |a_n|^2
< \infty$
has finitely many terms.
Hence the finite linear span of
$\{\Phi_n^{(\infty)}\}$ lies in $\Dom(\Hlim)$.
Under Hypothesis~\ref{hyp:decay},
this span is dense in $L^2(\mathbb{R})$,
so $\Dom(\Hlim)$ is dense.

\textit{(b)} For $f, g \in \Dom(\Hlim)$:
by the domain condition and Cauchy--Schwarz,
$\sum_n \lambda_n^{(\infty)}
|\ip{f}{\Phi_n^{(\infty)}}|
|\ip{g}{\Phi_n^{(\infty)}}| < \infty$,
so the symmetry identity
$\ip{\Hlim f}{g} = \ip{f}{\Hlim g}$
follows by rearranging the absolutely
convergent series.

\textit{(c)} Immediate since $\lambda_n^{(\infty)}$
does not depend on the subsequence.
\end{proof}

\begin{remark}[What density requires]
\label{rem:density-requires}
Part (a) deliberately separates two statements:
\begin{itemize}
\item $\Dom(\Hlim) \supset
      \mathrm{span}\{\Phi_n^{(\infty)}\}$
      (unconditional; proved above).
\item $\Dom(\Hlim)$ is dense in $L^2(\mathbb{R})$
      (conditional on Hypothesis~\ref{hyp:decay}).
\end{itemize}
For the closability and deficiency results
below (Sections~\ref{sec:closable}--\ref{sec:esa}),
only the unconditional part is needed.
Density is required for the spectral-identity
Theorem~\ref{thm:sigma-identity}.
\end{remark}

\section{Closability}
\label{sec:closable}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closability of $\Hlim$]
\label{thm:closable}
$\Hlim$ is closable.
No completeness of $\{\Phi_n^{(\infty)}\}$
is required.
\end{theorem}

\begin{proof}
We use the SOT-convergence \eqref{VIII6}:
$\Hc \xrightarrow{\mathrm{SOT}} \Hlim$,
i.e.\ $\Hc f \to \Hlim f$ in $L^2$ for
every fixed $f \in L^2(\mathbb{R})$.

Let $f_k \to 0$ in $L^2(\mathbb{R})$
with $\Hlim f_k \to g$.
We must show $g = 0$.

For each $k$ and $c$:
\[
  \|g - \Hc f_k\|
  \le \|g - \Hlim f_k\|
    + \|\Hlim f_k - \Hc f_k\|.
\]
Fix $\varepsilon > 0$.

\textit{Step 1.}
Since $\Hlim f_k \to g$, choose $K$ such that
$\|g - \Hlim f_k\| < \varepsilon/2$
for all $k \ge K$.

\textit{Step 2.}
For fixed $k \ge K$, since
$\Hc \xrightarrow{\mathrm{SOT}} \Hlim$,
choose $c_0$ such that
$\|\Hlim f_k - \Hc f_k\| < \varepsilon/2$
for all $c \ge c_0$.

Then $\|g - \Hc f_k\| < \varepsilon$
for all $k \ge K$, $c \ge c_0$.

\textit{Step 3.}
Since $\Hc$ is bounded ($\|\Hc\| \le 1$)
and $f_k \to 0$:
$\|\Hc f_k\| \le \|f_k\| \to 0$.
Hence $\Hc f_k \to 0$ in $L^2$ for each fixed $c$.

\textit{Conclusion.}
From Step~3: $\|g\| \le \|g - \Hc f_k\|
+ \|\Hc f_k\| < \varepsilon + \|\Hc f_k\|$.
Letting $k \to \infty$ with $c$ fixed:
$\|g\| \le \varepsilon$.
Since $\varepsilon > 0$ was arbitrary, $g = 0$.
Hence $\Hlim$ is closable.
\end{proof}

\begin{remark}[No completeness needed for closability]
\label{rem:closable-unconditional}
The proof uses only:
(i) $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
(Paper~X, Lem.~2.2; \eqref{VIII6});
(ii) $\|\Hc\| \le 1$ (standard Slepian).
In particular, it does not use completeness
of $\{\Phi_n^{(\infty)}\}$ nor the
spectral definition of $\Hlim$.
Closability is therefore an
\emph{unconditional} result.
\end{remark}

\begin{remark}[Uniqueness of the closure]
\label{rem:closure-unique}
If additionally Hypothesis~\ref{hyp:decay}
holds (completeness), then
$\overline{\Hlim}$ is the unique self-adjoint
operator whose eigensystem is
$\{(\lambda_n^{(\infty)}, \Phi_n^{(\infty)})\}$
(spectral theorem for pure-point-spectrum
self-adjoint operators on a complete ONB;
see \cite{ReedSimon1975}, Theorem~VIII.5).
Without Hypothesis~\ref{hyp:decay},
the closure $\overline{\Hlim}$ still exists
but may have a more complex structure.
\end{remark}

\begin{remark}[Compatibility with strong rate]
\label{rem:strong-rate-compat}
For $f \in \Dom(\Hlim)$,
$\Hc f \to \Hlim f$ in $L^2$
with rate $O(c^{-3/4})$ by \eqref{VIII2}.
\end{remark}

\section{Operator Norm Convergence Rate}
\label{sec:opnorm}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Strong convergence of eigenvectors]
\label{lem:eigenvector-strong}
For each fixed $n \ge 0$:
$\Phi_n^{(c)} \to \Phi_n^{(\infty)}$
strongly in $L^2(\mathbb{R})$ as $c \to \infty$.
\end{lemma}

\begin{proof}
By Proposition~\ref{prop:weak-limits},
$\Phi_n^{(c)} \rightharpoonup \Phi_n^{(\infty)}$
weakly in $L^2(\mathbb{R})$.
Both sequences are unit vectors:
$\|\Phi_n^{(c)}\| = 1$ and
$\|\Phi_n^{(\infty)}\| = 1$.
In a Hilbert space, weak convergence
plus norm convergence implies strong convergence
(see e.g.\ \cite{ReedSimon1975}, Prop.~II.11).
Since $\|\Phi_n^{(c)}\| = 1 = \|\Phi_n^{(\infty)}\|$,
norms converge (they are constantly $1$),
hence $\Phi_n^{(c)} \to \Phi_n^{(\infty)}$ strongly.
\end{proof}

\begin{remark}[The $\lambda_n^{(\infty)} = 0$ case]
\label{rem:lambda-zero}
For indices $n$ with $\lambda_n^{(\infty)} = 0$:
the Slepian counting law implies
$\lambda_n^{(c)} \to 0$ exponentially,
so $\Phi_n^{(c)}$ contributes terms
$\lambda_n^{(c)} \Phi_n^{(c)} \to 0$
in $L^2$.
Such indices need not be tracked separately
in eigenvalue or operator-norm estimates
(their contribution is exponentially small).
\end{remark}

\begin{lemma}[Eigenfunction alignment]
\label{lem:eigenvector-alignment}
For each fixed $n \ge 0$:
$\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|_{L^2} \to 0$
as $c \to \infty$.
Quantitatively, for $n \le \kappa c$:
\[
  \|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|^2
  = 2 - 2\,\mathrm{Re}\,\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}.
\]
Since $\Phi_n^{(c)} \to \Phi_n^{(\infty)}$ strongly
(Lemma~\ref{lem:eigenvector-strong}),
this converges to $0$.
The rate is $O(c^{-1/4})$, consistent with
the eigenvalue rate \eqref{VIII3},
and is used only in an $O(\cdot)$ sense
in Theorem~\ref{thm:opnorm}.
\end{lemma}

\begin{proof}
The identity $\|u - v\|^2 = \|u\|^2 +
\|v\|^2 - 2\,\mathrm{Re}\,\ip{u}{v}$
with $\|\Phi_n^{(c)}\| = \|\Phi_n^{(\infty)}\| = 1$
gives $\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|^2
= 2(1 - \mathrm{Re}\,\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}})$.
Since $\Phi_n^{(c)} \rightharpoonup \Phi_n^{(\infty)}$,
$\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}} \to
\|\Phi_n^{(\infty)}\|^2 = 1$,
so $\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\| \to 0$.
The $O(c^{-1/4})$ rate follows from
$1 - \mathrm{Re}\,\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}
\le \tfrac{1}{2}\|\Phi_n^{(c)} -
\Phi_n^{(\infty)}\|^2$ combined with
the eigenvalue rate \eqref{VIII3}
via the eigenfunction-eigenvalue coupling
(see Remark~\ref{rem:rate-justification}).
\end{proof}

\begin{remark}[Rate justification]
\label{rem:rate-justification}
The $O(c^{-1/4})$ rate for
$\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|$
follows from the eigenvalue rate \eqref{VIII3}
and the Slepian eigenvalue gap
$\lambda_n^{(c)} - \lambda_{n+1}^{(c)} \asymp c^{-1/2}$
(Paper~VIII, Thm.~4.1)
via the standard inequality for symmetric operators:
\[
  \|\Phi_n^{(c)} - P_n \Phi_n^{(c)}\|
  \le \frac{\|\Hc - H^{(\infty)}\|_{\rm strong}}
           {\lambda_n^{(c)} - \lambda_{n+1}^{(c)}},
\]
where $P_n$ is the spectral projection onto
$\Phi_n^{(\infty)}$,
and the strong rate is $c^{-3/4}$ (\eqref{VIII2}),
giving $c^{-3/4}/c^{-1/2} = c^{-1/4}$.
Note that this step uses the \emph{strong}
convergence rate (per vector), not
the operator-norm rate (which is what
we are proving), so there is no circularity.
\end{remark}

\begin{theorem}[Operator norm rate]
\label{thm:opnorm}
For all $c \ge c_0(\kappa)$:
\begin{equation}
\label{eq:opnorm-rate}
  \|\Hc - \Hlim\|_{L^2 \to L^2}
  \;\le\; C_\kappa\, c^{-1/4}.
\end{equation}
\end{theorem}

\begin{proof}
Let $f \in L^2(\mathbb{R})$ with $\|f\| = 1$.
Decompose using the eigenbasis of $\Hc$:
\begin{align}
  \|(\Hc - \Hlim)f\|^2
  &\le 2\sum_{n \le \kappa c}
       |\lambda_n^{(c)} - \lambda_n^{(\infty)}|^2
       |\ip{f}{\Phi_n^{(c)}}|^2
     + 2 E_\kappa(c,f)
\end{align}
where $E_\kappa(c,f)$ collects cross terms
and the tail $n > \kappa c$.

For the main sum:
$|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$ uniformly \eqref{VIII3},
and $\sum_n |\ip{f}{\Phi_n^{(c)}}|^2 \le 1$
(Bessel), so the main sum is
$\le C_\kappa^2 c^{-1/2}$.

For the tail $n > \kappa c$:
eigenvalues $\lambda_n^{(c)} \le e^{-\gamma_\kappa c}$
are exponentially small.

For the cross terms (from
$\Phi_n^{(c)} \ne \Phi_n^{(\infty)}$):
by Lemma~\ref{lem:eigenvector-alignment},
$\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|
= O(c^{-1/4})$ uniformly in $n \le \kappa c$,
so cross terms are $O(c^{-1/4})$.

Combining and taking the supremum:
\[
  \|\Hc - \Hlim\|_{L^2 \to L^2}
  \le C_\kappa c^{-1/4}.
\]
\end{proof}

\begin{remark}[Domain stability]
\label{rem:domain-stability}
For any sequence $f_c \in L^2(\mathbb{R})$
with $\|f_c\| + \|\Hc f_c\| \le M$
(uniform graph-norm bound),
the limit points of $f_c$ in $L^2$
lie in $\Dom(\Hlim)$.
\end{remark}

\section{Deficiency Index Analysis}
\label{sec:deficiency}
%%═══════════════════════════════════════════════════════════════════

\subsection{Setup}

By the von Neumann theory
(\cite{ReedSimon1975}, Theorem~X.2),
a symmetric operator is essentially
self-adjoint iff its deficiency indices
$n_\pm := \dim\ker(\Hlim^* \mp i)$ vanish.

\subsection{Properties of deficiency vectors}

\begin{lemma}[Orthogonality of deficiency vectors]
\label{lem:deficiency-analytic}
Let $f \in \ker(\Hlim^* + i)$.
Then $\ip{f}{\Phi_n^{(\infty)}} = 0$
for all $n \ge 0$.
\end{lemma}

\begin{proof}
From $\Hlim^* f = -if$:
$\ip{\Hlim^* f}{\Phi_n^{(\infty)}}
= -i\ip{f}{\Phi_n^{(\infty)}}$.
Since $\Phi_n^{(\infty)} \in \Dom(\Hlim)$
and $\Hlim$ is symmetric:
$\ip{f}{\Hlim \Phi_n^{(\infty)}}
= \lambda_n^{(\infty)}\ip{f}{\Phi_n^{(\infty)}}$.
Hence $(\lambda_n^{(\infty)} + i)
\ip{f}{\Phi_n^{(\infty)}} = 0$.
Since $\lambda_n^{(\infty)} \in [0,1] \subset \mathbb{R}$,
we have $\lambda_n^{(\infty)} + i \ne 0$
(the imaginary part is $1 \ne 0$),
so $\ip{f}{\Phi_n^{(\infty)}} = 0$.
This holds for all $n \ge 0$,
including those with $\lambda_n^{(\infty)} = 0$.
\end{proof}

\begin{remark}[The $\lambda_n^{(\infty)} = 0$ case]
\label{rem:deficiency-zero-eigenvalue}
In the original draft, the Closability proof
divided by $\lambda_n^{(\infty)}$, which is
illegal for $\lambda_n^{(\infty)} = 0$.
Lemma~\ref{lem:deficiency-analytic}
circumvents this: the argument
$(\lambda_n^{(\infty)} + i) \ne 0$
holds unconditionally since
$\lambda_n^{(\infty)} \in \mathbb{R}$ and
$i \notin \mathbb{R}$,
so no division is needed.
\end{remark}

\begin{remark}[Central reduction]
\label{rem:deficiency-orthogonal}
Lemma~\ref{lem:deficiency-analytic} gives:
\[
  \ker(\Hlim^* \pm i) = \{0\}
  \iff
  \{\Phi_n^{(\infty)}\} \text{ is complete
  in } L^2(\mathbb{R}).
\]
The question of essential self-adjointness
of $\Hlim$ reduces precisely to the
completeness of the limit eigenvectors.
This equivalence is \emph{unconditional}.
\end{remark}

\subsection{The completeness hypothesis}

\begin{hypothesis}[Completeness of $\{\Phi_n^{(\infty)}\}$]
\label{hyp:decay}
The system $\{\Phi_n^{(\infty)}\}_{n \ge 0}$
is complete in $L^2(\mathbb{R})$:
$\overline{\mathrm{span}\{\Phi_n^{(\infty)}\}}
= L^2(\mathbb{R})$.
\end{hypothesis}

\begin{remark}[Evidence for Hypothesis~\ref{hyp:decay}]
\label{rem:hyp-evidence}
\begin{enumerate}
\item Each $\{\Phi_n^{(c)}\}$ is a complete ONB
      for its finite-dimensional target space
      (Slepian; see Papers~V--VI).
\item $\Phi_n^{(\infty)} \ne 0$ for all $n$
      (Non-cancellation, \eqref{VIII4}).
\item Strong convergence
      $\Phi_n^{(c)} \to \Phi_n^{(\infty)}$
      (Lemma~\ref{lem:eigenvector-strong})
      is consistent with completeness being
      preserved in the limit,
      though this requires uniform control
      over all $n$ to prove rigorously.
\end{enumerate}
Rigorous proof is open
(Problem~\ref{prob:completeness}).
\end{remark}

\section{Essential Self-Adjointness}
\label{sec:esa}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Deficiency indices under completeness]
\label{thm:deficiency}
Assume Hypothesis~\ref{hyp:decay}.
Then $n_\pm(\Hlim) = 0$.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:deficiency-analytic},
every $f \in \ker(\Hlim^* + i)$
satisfies $\ip{f}{\Phi_n^{(\infty)}} = 0$
for all $n$.
By Hypothesis~\ref{hyp:decay},
$\{\Phi_n^{(\infty)}\}$ is complete,
so $f = 0$.
Identical for $\ker(\Hlim^* - i)$.
\end{proof}

\begin{corollary}[Essential self-adjointness]
\label{cor:esa}
Assume Hypothesis~\ref{hyp:decay}.
Then $\Hlim$ is essentially self-adjoint:
$\overline{\Hlim} = \Hlim^*$.
\end{corollary}

\begin{proof}
Von Neumann criterion: $n_\pm = 0$
implies essential self-adjointness
(\cite{ReedSimon1975}, Theorem~X.3).
\end{proof}

\begin{theorem}[Spectral identity]
\label{thm:sigma-identity}
Assume Hypothesis~\ref{hyp:decay} and
Hypothesis~CCM (Paper~VIII, Hyp.~5.1).
Then:
\begin{enumerate}[\rm(i)]
\item $\sigma_{\rm app}(\Hlim) = \sigma(\Hlim)$.
\item Every non-trivial zero $\rho$ of $\zeta(s)$
      lies in $\sigma(\Hlim)$.
\item All non-trivial zeros of $\zeta(s)$
      satisfy $\mathrm{Re}(\rho) = 1/2$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(i)} For essentially self-adjoint
operators, $\sigma_{\rm app} = \sigma$
(\cite{ReedSimon1975}, Thm.~VII.13).

\textit{(ii)} By Paper~VIII, Thm.~5.2,
$\rho \in \sigma_{\rm app}(\Hlim)$.
By (i), $\rho \in \sigma(\Hlim)$.

\textit{(iii)} The CCM embedding (Hyp.~CCM)
identifies the non-trivial zeros of $\zeta(s)$
with imaginary parts $\mathrm{Im}(\rho)$ of
eigenvalues of $\overline{\Hlim}$ on the
critical line.
Since $\overline{\Hlim}$ is self-adjoint,
$\sigma(\overline{\Hlim}) \subset \mathbb{R}$,
forcing $\mathrm{Re}(\rho) = 1/2$.
\end{proof}

\begin{remark}[Status]
\label{rem:conditional}
Theorem~\ref{thm:sigma-identity}(iii)
is conditional on Hypothesis~\ref{hyp:decay}
(completeness; open) and Hypothesis~CCM
(external result \cite{CCM2025}).
\end{remark}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Completeness of $\{\Phi_n^{(\infty)}\}$]
\label{prob:completeness}
Prove Hypothesis~\ref{hyp:decay}:
$\{\Phi_n^{(\infty)}\}_{n \ge 0}$ is complete
in $L^2(\mathbb{R})$.
By Remark~\ref{rem:deficiency-orthogonal},
this is equivalent to essential self-adjointness
of $\Hlim$.
\emph{Approach}: show no nonzero
$f \in L^2$ is orthogonal to all
$\Phi_n^{(\infty)}$ simultaneously,
using non-cancellation \eqref{VIII4}
and mass-concentration uniformly over $n$.
\end{problem}

\begin{problem}[Sharp operator-norm rate]
\label{prob:opnorm-sharp}
Is $\|\Hc - \Hlim\|_{L^2 \to L^2}
\asymp c^{-1/4}$?
Upper bound: Theorem~\ref{thm:opnorm}.
Lower bound: requires sharp eigenvalue-gap
estimate $|\lambda_n^{(c)} - \lambda_n^{(c+1)}|
\ge c_\kappa c^{-3/4}$ (open since
Paper~VIII, Rem.~4.2).
\end{problem}

\begin{problem}[Density of $\{\lambda_n^{(\infty)}\}$]
\label{prob:density}
Is $\{\lambda_n^{(\infty)} : n \ge 0\}$
dense in $[0,1]$?
By Paper~XI, Thm.~4.1, this is equivalent
to $\sigma(\Hlim) = [0,1]$.
\end{problem}

\begin{problem}[Unconditional essential self-adjointness]
\label{prob:esa-unconditional}
Prove essential self-adjointness of $\Hlim$
without Hypothesis~\ref{hyp:decay}.
By Remark~\ref{rem:deficiency-orthogonal},
this is equivalent to Problem~\ref{prob:completeness}.
\end{problem}
