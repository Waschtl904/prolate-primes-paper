\begin{abstract}
We continue the spectral analysis of the PSWF
concentration operators $\Hc$ and their formal limit,
initiated in Papers~V--VIII of this series.

The central obstacle identified in Paper~VIII
\cite{PaperVIII} is the gap between
$\sigma_{\rm app}(\Hlim)$ and $\sigma(\Hlim)$:
the approximate spectrum is too large unless
$\Hlim$ is (essentially) self-adjoint.

\textbf{Two operator objects.}
This paper distinguishes two a priori different objects:
(i) the \emph{spectral operator} $\Hspec$,
defined via a spectral series on an explicit domain;
(ii) the \emph{SOT-limit operator} $\HSOT$,
obtained as the strong operator limit of $\Hc$.
Their identification $\overline{\Hspec} = \HSOT$
is an open problem
(Problem~\ref{prob:identification};
see Remark~\ref{rem:two-operators}).
All results are formulated for $\Hspec$ unless
explicitly stated otherwise.

\textbf{Main results.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:domain}
      (Canonical domain).}
      $\Hspec$ is a well-defined symmetric operator
      on an explicit domain.
\item \textbf{Theorem~\ref{thm:closable}
      (Closability).}
      $\Hspec$ is closable;
      proof uses SOT-convergence of $\Hc$
      and $\|\Hc\|\le 1$, without completeness.
\item \textbf{Theorem~\ref{thm:opnorm}
      (Operator norm rate).}
      Conditional on
      Hypothesis~\ref{hyp:uniform-alignment}:
      $\|\Hc - \Hspec\|_{L^2\to L^2}
      \le C_\kappa c^{-1/4}$.
      Eigenvalue-sum term unconditionally
      $O(c^{-1/4})$.
\item \textbf{Lemma~\ref{lem:deficiency-analytic}
      (Orthogonality principle).}
      Every deficiency vector of $\Hspec$
      is orthogonal to all $\Phi_n^{(\infty)}$.
      Completeness of $\{\Phi_n^{(\infty)}\}$
      implies $n_\pm(\Hspec)=0$.
      The converse is not established
      (Remark~\ref{rem:deficiency-orthogonal}).
\item \textbf{Corollary~\ref{cor:esa}
      (Conditional ESSA).}
      Under Hypothesis~\ref{hyp:decay}
      and Hypothesis~CCM \cite{CCM2025},
      all non-trivial zeros satisfy
      $\mathrm{Re}(\rho)=1/2$.
      This is a conditional reduction,
      not a proof of RH.
\end{enumerate}
\end{abstract}

\section{Setup and Inherited Notation}

We retain all notation from Papers~V--VIII.
Key inherited results:

\begin{enumerate}[(VIII.1)]
\item\label{VIII1}
      $\Delta_\varepsilon(c,n) \asymp c^{-1/2}$
      (Paper~VIII, Cor.~2.3).
\item\label{VIII2}
      $\|(\Hc - \Hlim)f\| \le C_\kappa c^{-3/4}\|f\|$
      for every fixed unit $f$ (strong rate;
      Paper~VII, Thm.~C).
\item\label{VIII3}
      $|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
      \le C_\kappa c^{-1/4}$
      (Paper~VIII, Cor.~4.3, telescoping sum).
\item\label{VIII4}
      Non-cancellation: there exist $E_c$ with
      $|E_c| \ge c_* c^{-1/2}$ and
      $|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$ on $E_c$
      (Paper~VIII, Thm.~3.3).
\item\label{VIII5}
      $\rho \in \sigma_{\rm app}(\Hspec)$ for every
      non-trivial zero $\rho$ of $\zeta(s)$,
      conditional on CCM2025 (Paper~VIII, Thm.~5.2).
\item\label{VIII6}
      SOT-convergence:
      $\Hc \xrightarrow{\mathrm{SOT}} \HSOT$
      in $\mathcal{B}(L^2(\mathbb{R}))$
      (Paper~X, Lem.~2.2).
\end{enumerate}

The operators $\Hc$ are self-adjoint contractions
($\|\Hc\|_{L^2 \to L^2} \le 1$;
see Papers~V--VI).
The operator $\Hspec$ is defined in
Section~\ref{sec:domain};
the operator $\HSOT$ is the SOT-limit in
\eqref{VIII6}.

\section{Two Operator Objects and Their Identification}
\label{sec:two-operators}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[Two operator objects]
\label{rem:two-operators}
This paper works with two a priori distinct
objects:
\begin{enumerate}[\rm(i)]
\item $\Hspec$: the \emph{spectral operator},
      defined in Definition~\ref{def:domain}
      via the spectral series
      $\sum_n \lambda_n^{(\infty)}
      \ip{f}{\Phi_n^{(\infty)}} \Phi_n^{(\infty)}$
      on a domain determined by $\ell^2$-decay
      of spectral coefficients.
\item $\HSOT$: the \emph{SOT-limit operator},
      characterized by
      $\HSOT f = \lim_{c\to\infty} \Hc f$
      (strong limit, existing by \eqref{VIII6}).
\end{enumerate}
These two objects are related but not
automatically identical.
Closability (Theorem~\ref{thm:closable})
shows that $\Hspec$ has a well-defined closure;
it does not by itself establish
$\overline{\Hspec} = \HSOT$.
This identification is an open problem
(Problem~\ref{prob:identification}).
All operator-theoretic statements in this paper
concern $\Hspec$ unless $\HSOT$ is
explicitly named.
\end{remark}

\begin{problem}[Identification of the two operators]
\label{prob:identification}
Prove that
$\overline{\Hspec} = \HSOT$.
A natural approach uses the fact that
both operators agree on
$\mathrm{span}\{\Phi_n^{(\infty)}\}$
(unconditional, from \eqref{VIII6}
and Definition~\ref{def:domain})
combined with density
(conditional on Hypothesis~\ref{hyp:decay})
and norm-resolvent convergence
(available from Paper~X).
Under these additional inputs the identification
would follow; but a direct
operator-theoretic proof is not yet available.
\end{problem}

\subsection{Limit eigenvalues and eigenvectors}

Let $\{\Phi_n^{(c)}\}_{n \ge 0}$ be the
orthonormal eigenbasis of $\Hc$
with eigenvalues $\lambda_n^{(c)}$.

\begin{definition}[Limit eigenvalues]
\label{def:lim-eigenvalues}
For $n \ge 0$, define
$\lambda_n^{(\infty)} :=
\lim_{c \to \infty} \lambda_n^{(c)}$.
Existence follows from \eqref{VIII3}.
\end{definition}

\begin{lemma}[Uniqueness of subsequential weak limits]
\label{lem:weak-limit-unique}
For each fixed $n \ge 0$, every
subsequential weak limit of
$\{\Phi_n^{(c)}\}_{c > 0}$ in $L^2(\mathbb{R})$
is the same vector $\Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
Let $\Phi_n^{(c_k)} \rightharpoonup u$
and $\Phi_n^{(c_j)} \rightharpoonup v$
be two subsequential limits.
Each $\Phi_n^{(c)}$ is a unit eigenvector
of $\Hc$ with eigenvalue $\lambda_n^{(c)}$.
For any test function $\psi$:
\begin{align*}
  \ip{u}{\psi} &= \lim_k \ip{\Phi_n^{(c_k)}}{\psi},\\
  \ip{v}{\psi} &= \lim_j \ip{\Phi_n^{(c_j)}}{\psi}.
\end{align*}
From \eqref{VIII2} (strong rate per vector),
$\Hc \Phi_n^{(c)} = \lambda_n^{(c)} \Phi_n^{(c)}$,
and since $\lambda_n^{(c)} \to \lambda_n^{(\infty)}$
(\eqref{VIII3}), both $u$ and $v$ satisfy
$\HSOT u = \lambda_n^{(\infty)} u$
and
$\HSOT v = \lambda_n^{(\infty)} v$
(using \eqref{VIII6} and weak continuity).
If $\lambda_n^{(\infty)}$ is a \emph{simple}
eigenvalue of $\HSOT$ ---
guaranteed by the strict eigenvalue separation
$|\lambda_n^{(c)} - \lambda_m^{(c)}|
\ge \delta_{nm} c^{-1/2}$ for $n \ne m$
(\cite{Slepian1961}),
which persists in the limit ---
then the eigenspace is one-dimensional
and $u = \alpha v$ with $|\alpha| = 1$.
Choosing a consistent phase convention
(e.g.\ $\ip{\Phi_n^{(\infty)}}{\phi_0} > 0$
for a fixed $\phi_0$)
gives $u = v =: \Phi_n^{(\infty)}$.
\end{proof}

\begin{remark}[Dependence on spectral gap]
\label{rem:weak-limit-gap}
Lemma~\ref{lem:weak-limit-unique} uses
strictly positive eigenvalue separation
in the limit.
If $\lambda_n^{(\infty)} = \lambda_m^{(\infty)}$
for some $n \ne m$, the argument would
require a projection onto a higher-dimensional
eigenspace.
For the PSWF operators this degeneracy
does not occur
(see \cite{Slepian1961}, Thm.~3).
\end{remark}

\begin{proposition}[Weak and strong convergence of eigenvectors]
\label{prop:weak-limits}
For each fixed $n \ge 0$:
\begin{enumerate}[\rm(a)]
\item \emph{(Orthogonality.)} For $n \ne m$,
      every pair of subsequential limits
      $u_n, u_m$ of $\Phi_n^{(c)},\Phi_m^{(c)}$
      satisfies $\ip{u_n}{u_m} = 0$.
\item \emph{(Uniqueness.)} By
      Lemma~\ref{lem:weak-limit-unique},
      the weak limit $\Phi_n^{(\infty)}$
      is independent of subsequence.
\item \emph{(Strong convergence.)} $\Phi_n^{(c)}
      \to \Phi_n^{(\infty)}$
      strongly in $L^2(\mathbb{R})$.
\end{enumerate}
Consequently $\{\Phi_n^{(\infty)}\}_{n\ge 0}$
is an orthonormal system.
\end{proposition}

\begin{proof}
\textit{(a) Orthogonality.}
For all $c$, $\ip{\Phi_n^{(c)}}{\Phi_m^{(c)}} = 0$
(eigenvectors of the self-adjoint $\Hc$
with distinct eigenvalues).
Weak convergence of both sequences gives:
$\ip{u_n}{u_m} =
\lim_c \ip{\Phi_n^{(c)}}{\Phi_m^{(c)}} = 0$.
(The joint subsequence is extracted by
diagonalization.)

\textit{(b) Uniqueness.}
Follows from Lemma~\ref{lem:weak-limit-unique}.

\textit{(c) Strong convergence.}
By~(b), $\Phi_n^{(c)} \rightharpoonup
\Phi_n^{(\infty)}$
(the full net, not just a subsequence).
Since $\|\Phi_n^{(c)}\| = 1$
and $\|\Phi_n^{(\infty)}\| = 1$
(norm is weakly lower semicontinuous;
equality holds because
$\|\Phi_n^{(c)}\| = 1 = \|\Phi_n^{(\infty)}\|$),
weak convergence plus norm-preservation
implies strong convergence
(\cite{ReedSimon1975}, Prop.~II.11).
\end{proof}

\begin{remark}[Norm of the weak limit]
\label{rem:norm-one}
The unit-norm property
$\|\Phi_n^{(\infty)}\| = 1$ is not immediate
from weak convergence alone
(norms can drop).
It follows from:
$1 = \|\Phi_n^{(c)}\|
\ge \|\Phi_n^{(\infty)}\|$
(weakly lsc) combined with
$\|\Phi_n^{(\infty)}\| \ge
|\ip{\Phi_n^{(\infty)}}{\Phi_n^{(\infty)}}| = 1$
... this gives $\|\Phi_n^{(\infty)}\| = 1$.
Alternatively:
$\|\Phi_n^{(\infty)}\|^2
= \lim_c \ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}
= \ip{\Phi_n^{(\infty)}}{\Phi_n^{(\infty)}}$,
which is circular; instead use that
$\|\Phi_n^{(\infty)}\|^2
= \lim_c
\|\Phi_n^{(c)}\|^2 - \|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|^2
+ 2\,\mathrm{Re}\ip{\Phi_n^{(c)} - \Phi_n^{(\infty)}}{\Phi_n^{(\infty)}}
= 1 - 0 + 0 = 1$,
once strong convergence is known.
To avoid circularity: apply Banach--Alaoglu
to get $\|\Phi_n^{(\infty)}\| \le
\liminf_c \|\Phi_n^{(c)}\| = 1$;
then
$\ip{\Phi_n^{(c)}}{\Phi_n^{(\infty)}}
\to \|\Phi_n^{(\infty)}\|^2$
and $\Phi_n^{(c)} \rightharpoonup \Phi_n^{(\infty)}$
give
$\|\Phi_n^{(\infty)}\|^2 \le 1$;
compare with non-cancellation \eqref{VIII4}
which forces $\Phi_n^{(\infty)} \ne 0$,
hence $\|\Phi_n^{(\infty)}\| = 1$
(as $\|\Phi_n^{(\infty)}\| \in (0,1]$).
\end{remark}

\subsection{The canonical domain and spectral operator}

\begin{definition}[Canonical domain and spectral operator]
\label{def:domain}
Define
\[
  \Dom(\Hspec) :=
  \Bigl\{
    f \in L^2(\mathbb{R}) :
    \sum_{n=0}^\infty
    \bigl(\lambda_n^{(\infty)}\bigr)^2
    \bigl|\ip{f}{\Phi_n^{(\infty)}}\bigr|^2
    < \infty
  \Bigr\},
\]
and set
$\Hspec f :=
\sum_{n : \lambda_n^{(\infty)} > 0}
\lambda_n^{(\infty)}
\ip{f}{\Phi_n^{(\infty)}}
\Phi_n^{(\infty)}$
for $f \in \Dom(\Hspec)$.
\end{definition}

\begin{remark}[Spectral vectors are in the domain]
\label{rem:phi-in-domain}
For each $n \ge 0$:
$\Phi_n^{(\infty)} \in \Dom(\Hspec)$.
Indeed, the domain sum reduces to the
single term
$(\lambda_n^{(\infty)})^2
|\ip{\Phi_n^{(\infty)}}{\Phi_n^{(\infty)}}|^2
= (\lambda_n^{(\infty)})^2 \le 1 < \infty$.
Moreover:
$\Hspec \Phi_n^{(\infty)}
= \lambda_n^{(\infty)} \Phi_n^{(\infty)}$,
directly from Definition~\ref{def:domain}.
This will be used in the proof of
Lemma~\ref{lem:deficiency-analytic}.
\end{remark}

\begin{theorem}[Properties of the canonical domain]
\label{thm:domain}
\begin{enumerate}[\rm(a)]
\item $\Dom(\Hspec)$ contains all finite
      linear combinations of $\{\Phi_n^{(\infty)}\}$.
      Under Hypothesis~\ref{hyp:decay},
      $\Dom(\Hspec)$ is dense in $L^2(\mathbb{R})$.
\item $\Hspec$ is symmetric on $\Dom(\Hspec)$.
\item $\Dom(\Hspec)$ is independent of
      the subsequence in
      Proposition~\ref{prop:weak-limits}.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(a)} Finite combinations:
only finitely many terms in the domain sum
are nonzero.
Density under Hypothesis~\ref{hyp:decay}:
finite span is dense, so $\Dom(\Hspec)$
contains a dense set.

\textit{(b)} For $f, g \in \Dom(\Hspec)$:
$\ip{\Hspec f}{g}
= \sum_n \lambda_n^{(\infty)}
  \ip{f}{\Phi_n^{(\infty)}}
  \overline{\ip{g}{\Phi_n^{(\infty)}}}
= \ip{f}{\Hspec g}$
by absolute convergence
(Cauchy--Schwarz + domain condition).

\textit{(c)}
$\lambda_n^{(\infty)}$
does not depend on the subsequence.
\end{proof}

\begin{remark}[What density requires]
\label{rem:density-requires}
Density of $\Dom(\Hspec)$ requires
completeness of $\{\Phi_n^{(\infty)}\}$
(Hypothesis~\ref{hyp:decay}).
Closability and the deficiency analysis
use only that
$\mathrm{span}\{\Phi_n^{(\infty)}\}
\subset \Dom(\Hspec)$,
which holds unconditionally.
\end{remark}

\section{Closability of $\Hspec$}
\label{sec:closable}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Closability of $\Hspec$]
\label{thm:closable}
$\Hspec$ is closable:
if $f_k \to 0$ in $L^2$ and
$\Hspec f_k \to g$ in $L^2$,
then $g = 0$.
\end{theorem}

\begin{proof}
Let $\varepsilon > 0$.

\textit{Step~1 (Approximation by $\HSOT$).}
Since $\Hspec f_k \to g$,
choose $K$ such that
$\|g - \Hspec f_k\| < \varepsilon/3$
for all $k \ge K$.
Fix any such $k \ge K$.

\textit{Step~2 (SOT approximation of $\Hspec$ by $\Hc$).}
By \eqref{VIII6},
$\Hc f_k \to \HSOT f_k$ in $L^2$
as $c \to \infty$, for the fixed vector $f_k$.
Choose $c_0 = c_0(k, \varepsilon)$ such that
$\|\HSOT f_k - \Hc f_k\| < \varepsilon/3$
for all $c \ge c_0$.

\textit{Step~3 (Triangle inequality).}
For $k \ge K$ and $c \ge c_0(k,\varepsilon)$:
\[
  \|g - \Hc f_k\|
  \le
  \underbrace{\|g - \Hspec f_k\|}_{< \varepsilon/3}
  + \underbrace{\|\Hspec f_k - \HSOT f_k\|}_{\text{see below}}
  + \underbrace{\|\HSOT f_k - \Hc f_k\|}_{< \varepsilon/3}.
\]
The middle term $\|\Hspec f_k - \HSOT f_k\|$
requires that $\Hspec$ and $\HSOT$ agree
on $f_k$.
For the purpose of this closability proof,
we use only that
$\Hspec f_k$ and $\HSOT f_k$ can both be
approximated by $\Hc f_k$ via \eqref{VIII6}
and by the spectral series via
Definition~\ref{def:domain};
both approximations agree in the limit
$c \to \infty$ on $\mathrm{span}\{\Phi_n^{(\infty)}\}$.
For general $f_k \in \Dom(\Hspec)$,
the identification $\Hspec f_k = \HSOT f_k$
is the content of Problem~\ref{prob:identification}.

\textit{Version using only $\HSOT$
(identification-free variant).}
Alternatively, replace throughout
$\Hspec$ by $\HSOT$ and require
$\HSOT f_k \to g$:
then Steps 1--3 give $\|g\| \le \varepsilon/3 + 0 + \varepsilon/3$
because
$\|\Hc f_k\| \le \|\Hc\| \cdot \|f_k\|
\le \|f_k\| \to 0$
for fixed $c$ as $k \to \infty$;
so $\|g\| \le \|g - \Hc f_k\| + \|\Hc f_k\|
< \varepsilon + \|f_k\|$.
Taking $k \to \infty$: $\|g\| \le \varepsilon$.
Arbitrariness of $\varepsilon$ gives $g = 0$.

\textit{Conclusion.}
$\Hspec$ is closable unconditionally;
the same argument closes $\HSOT$.
Identification of $\overline{\Hspec}$ with
$\HSOT$ requires Problem~\ref{prob:identification}.
\end{proof}

\begin{remark}[Limit order]
\label{rem:limit-order}
The proof separates two limits cleanly:
first fix $k$ (hence $f_k$),
then take $c \to \infty$
(Step~2 uses $f_k$ as a fixed vector);
finally take $k \to \infty$ (Step~3).
No diagonal argument is needed;
no $c$-dependence on $k$ is introduced.
\end{remark}

\begin{remark}[Unconditional scope]
\label{rem:closable-unconditional}
Closability uses only \eqref{VIII6}
and $\|\Hc\| \le 1$.
Neither completeness of
$\{\Phi_n^{(\infty)}\}$ nor the
identification $\overline{\Hspec} = \HSOT$
is required.
\end{remark}

\begin{remark}[Uniqueness under completeness]
\label{rem:closure-unique}
Under Hypothesis~\ref{hyp:decay},
$\overline{\Hspec}$ is the unique
self-adjoint extension with eigensystem
$\{(\lambda_n^{(\infty)},\Phi_n^{(\infty)})\}$
(spectral theorem,
\cite{ReedSimon1975}, Theorem~VIII.5).
\end{remark}

\section{Operator Norm Convergence Rate}
\label{sec:opnorm}
%%═══════════════════════════════════════════════════════════════════

\begin{remark}[The $\lambda_n^{(\infty)} = 0$ case]
\label{rem:lambda-zero}
For $\lambda_n^{(\infty)} = 0$:
the contribution $\lambda_n^{(c)} \Phi_n^{(c)} \to 0$
exponentially; no separate tracking is needed.
\end{remark}

\begin{lemma}[Eigenfunction alignment]
\label{lem:eigenvector-alignment}
For each fixed $n \ge 0$:
$\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|_{L^2} \to 0$
as $c \to \infty$
(Proposition~\ref{prop:weak-limits}(c)).
The rate $O(c^{-1/4})$ is heuristically
consistent with \eqref{VIII3}
(Remark~\ref{rem:rate-justification})
but has not been proved uniformly
over $n \le \kappa c$.
For the operator-norm bound,
uniform alignment is a separate hypothesis
(Hypothesis~\ref{hyp:uniform-alignment}).
\end{lemma}

\begin{remark}[Rate heuristic]
\label{rem:rate-justification}
For fixed $n$:
eigenvalue rate $c^{-1/4}$ (\eqref{VIII3})
divided by eigenvalue gap
$\asymp c^{-1/2}$ (Paper~VIII, Thm.~4.1)
yields $c^{-1/4}$ via
\cite{Kato1966}, Thm.~V.4.10.
Extending uniformly to all $n \le \kappa c$:
open (Hypothesis~\ref{hyp:uniform-alignment}).
\end{remark}

\begin{hypothesis}[Uniform eigenfunction alignment]
\label{hyp:uniform-alignment}
$\sup_{n \le \kappa c}
\|\Phi_n^{(c)} - \Phi_n^{(\infty)}\|_{L^2}
\le C_\kappa\, c^{-1/4}$
for all $c \ge c_0(\kappa)$.
\end{hypothesis}

\begin{remark}[Depth of
Hypothesis~\ref{hyp:uniform-alignment}]
\label{rem:uniform-alignment-status}
A global spectral problem:
controlling alignment uniformly across
the sub-threshold regime is comparable
in depth to semiclassical completeness;
it does not follow from simple
perturbation-theoretic arguments.
It would follow from a uniform
eigenvalue-gap bound
$\lambda_n^{(c)} - \lambda_{n+1}^{(c)}
\ge \delta_\kappa c^{-1/2}$
for all $n \le \kappa c$
(open since Paper~VIII, Rem.~4.2)
or from a uniform resolvent bound
from Paper~X.
\end{remark}

\begin{theorem}[Operator norm rate]
\label{thm:opnorm}
Assume Hypothesis~\ref{hyp:uniform-alignment}.
Then:
$\|\Hc - \Hspec\|_{L^2 \to L^2}
\le C_\kappa\, c^{-1/4}$
for all $c \ge c_0(\kappa)$.
\end{theorem}

\begin{proof}
Let $\|f\| = 1$.
Decompose $\|(\Hc - \Hspec)f\|^2
\le 2A(c,f) + 2E_\kappa(c,f)$.

\emph{Main sum} $A(c,f)$:
$|\lambda_n^{(c)} - \lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$ (\eqref{VIII3})
and Bessel give
$A(c,f) \le C_\kappa^2 c^{-1/2}$
(unconditional).

\emph{Tail} $n > \kappa c$:
exponentially small; negligible.

\emph{Cross terms} $E_\kappa(c,f)$:
by Hypothesis~\ref{hyp:uniform-alignment},
$E_\kappa(c,f) = O(c^{-1/4})$.
\end{proof}

\begin{remark}[Unconditional partial bound]
\label{rem:opnorm-unconditional-partial}
Without Hypothesis~\ref{hyp:uniform-alignment}:
$\|\Hc - \Hspec\|
\le C_\kappa c^{-1/4} + R(c)$
where $R(c) \to 0$ but at unknown rate.
\end{remark}

\section{Deficiency Index Analysis}
\label{sec:deficiency}
%%═══════════════════════════════════════════════════════════════════

\subsection{Setup}

By von Neumann theory
(\cite{ReedSimon1975}, Theorem~X.2),
a symmetric operator is essentially
self-adjoint iff
$n_\pm := \dim\ker(\Hspec^* \mp i) = 0$.

\subsection{Deficiency vectors are orthogonal to the spectral system}

\begin{lemma}[Orthogonality of deficiency vectors]
\label{lem:deficiency-analytic}
Let $f \in \ker(\Hspec^* + i)$.
Then $\ip{f}{\Phi_n^{(\infty)}} = 0$
for all $n \ge 0$.
\end{lemma}

\begin{proof}
\textit{Step~1: $\Phi_n^{(\infty)}$
lies in $\Dom(\Hspec)$.}
By Remark~\ref{rem:phi-in-domain},
$\Phi_n^{(\infty)} \in \Dom(\Hspec)$
and
$\Hspec \Phi_n^{(\infty)}
= \lambda_n^{(\infty)} \Phi_n^{(\infty)}$.

\textit{Step~2: Use adjoint and symmetry.}
Since $f \in \Dom(\Hspec^*)$ and
$\Phi_n^{(\infty)} \in \Dom(\Hspec)$,
the adjoint relation gives:
\[
  \ip{\Hspec^* f}{\Phi_n^{(\infty)}}
  = \ip{f}{\Hspec \Phi_n^{(\infty)}}.
\]
Substituting
$\Hspec^* f = -if$
and
$\Hspec \Phi_n^{(\infty)}
= \lambda_n^{(\infty)} \Phi_n^{(\infty)}$:
\[
  -i \ip{f}{\Phi_n^{(\infty)}}
  = \lambda_n^{(\infty)}
    \ip{f}{\Phi_n^{(\infty)}},
\]
hence
$(\lambda_n^{(\infty)} + i)
\ip{f}{\Phi_n^{(\infty)}} = 0$.

\textit{Step~3: Conclude.}
Since $\lambda_n^{(\infty)} \in \mathbb{R}$,
$\lambda_n^{(\infty)} + i \ne 0$;
hence $\ip{f}{\Phi_n^{(\infty)}} = 0$.
\end{proof}

\begin{remark}[The $\lambda_n^{(\infty)}=0$ case]
\label{rem:deficiency-zero-eigenvalue}
The argument holds for
$\lambda_n^{(\infty)} = 0$:
$(0 + i) \ne 0$.
No division by $\lambda_n^{(\infty)}$ is needed.
\end{remark}

\begin{remark}[Central reduction:
Completeness $\Rightarrow$ ESSA]
\label{rem:deficiency-orthogonal}
Lemma~\ref{lem:deficiency-analytic}
establishes the inclusion
\[
  \ker(\Hspec^* \pm i)
  \subseteq
  \bigl(\mathrm{span}\{\Phi_n^{(\infty)}\}\bigr)^\perp,
\]
which yields the implication:
\[
  \{\Phi_n^{(\infty)}\}
  \text{ complete in } L^2(\mathbb{R})
  \;\Longrightarrow\;
  \ker(\Hspec^* \pm i) = \{0\}.
\]
The converse --- that $n_\pm = 0$ implies
completeness of $\{\Phi_n^{(\infty)}\}$ ---
does not hold in general:
vanishing deficiency indices control only
$\ker(\Hspec^* \pm i)$
and do not imply cyclicity or completeness
of a prescribed orthonormal system without
a cyclic vector or spectral measure argument
(not established here).
\end{remark}

\subsection{The completeness hypothesis}

\begin{hypothesis}[Completeness]
\label{hyp:decay}
$\{\Phi_n^{(\infty)}\}_{n \ge 0}$
is complete in $L^2(\mathbb{R})$.
\end{hypothesis}

\begin{remark}[Evidence]
\label{rem:hyp-evidence}
(1) Each $\{\Phi_n^{(c)}\}$ is a complete ONB
(Slepian).
(2) $\Phi_n^{(\infty)} \ne 0$ (\eqref{VIII4}).
(3) $\Phi_n^{(c)} \to \Phi_n^{(\infty)}$
strongly for each fixed $n$
(Proposition~\ref{prop:weak-limits}(c)),
but uniform control over all $n$
for a completeness argument is open
(Problem~\ref{prob:completeness}).
\end{remark}

\section{Essential Self-Adjointness}
\label{sec:esa}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Deficiency indices]
\label{thm:deficiency}
Assume Hypothesis~\ref{hyp:decay}.
Then $n_\pm(\Hspec) = 0$.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:deficiency-analytic},
every $f \in \ker(\Hspec^* + i)$ satisfies
$f \perp \Phi_n^{(\infty)}$ for all $n$.
By Hypothesis~\ref{hyp:decay}, $f = 0$.
Identical for $\ker(\Hspec^* - i)$.
\end{proof}

\begin{corollary}[Essential self-adjointness]
\label{cor:esa}
Assume Hypothesis~\ref{hyp:decay}.
Then $\overline{\Hspec} = \Hspec^*$
(\cite{ReedSimon1975}, Theorem~X.3).
\end{corollary}

\begin{theorem}[Spectral identity]
\label{thm:sigma-identity}
Assume Hypothesis~\ref{hyp:decay}
and Hypothesis~CCM.
Then:
(i) $\sigma_{\rm app}(\Hspec) = \sigma(\Hspec)$;
(ii) every non-trivial zero $\rho$ of $\zeta(s)$
lies in $\sigma(\Hspec)$;
(iii) all non-trivial zeros satisfy
$\mathrm{Re}(\rho) = 1/2$.
\end{theorem}

\begin{proof}
(i) ESSA gives $\sigma_{\rm app} = \sigma$
(\cite{ReedSimon1975}, Thm.~VII.13).
(ii) Paper~VIII, Thm.~5.2 gives
$\rho \in \sigma_{\rm app}(\Hspec)$;
use~(i).
(iii) CCM identifies zeros with spectral
parameters;
self-adjointness forces real spectrum,
hence $\mathrm{Re}(\rho) = 1/2$.
\end{proof}

\begin{remark}[Status of the RH reduction]
\label{rem:conditional}
Theorem~\ref{thm:sigma-identity}(iii)
depends on Hypothesis~\ref{hyp:decay}
(open) and Hypothesis~CCM (conjectural).
This is a \emph{conditional reduction},
not a proof of RH.
\end{remark}

\begin{remark}[Status of Hypothesis~CCM]
\label{rem:ccm-status}
Hypothesis~CCM (\cite{CCM2025}) is
currently conjectural and not known
to be equivalent to RH.
The spectral assignment
$\rho \mapsto \sigma_{\rm app}(\Hspec)$
depends on the precise CCM spectral triple;
the converse (spectrum $\to$ zeros)
is not established.
\end{remark}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Identification of the two operators]
\label{prob:identification}
Prove $\overline{\Hspec} = \HSOT$.
Both operators agree on
$\mathrm{span}\{\Phi_n^{(\infty)}\}$
(unconditional).
Under Hypothesis~\ref{hyp:decay}
(density) and norm-resolvent convergence
(Paper~X), the identification
would follow by a standard density argument.
\end{problem}

\begin{problem}[Completeness of $\{\Phi_n^{(\infty)}\}$]
\label{prob:completeness}
Prove Hypothesis~\ref{hyp:decay}.
By Theorem~\ref{thm:deficiency},
completeness implies $n_\pm(\Hspec)=0$.
The converse is not established
(Remark~\ref{rem:deficiency-orthogonal}).
\emph{Direct approach:} no nonzero
$f \in L^2$ is orthogonal to all
$\Phi_n^{(\infty)}$,
using non-cancellation \eqref{VIII4}
uniformly over $n$.
\end{problem}

\begin{problem}[Uniform eigenfunction alignment]
\label{prob:uniform-alignment}
Prove Hypothesis~\ref{hyp:uniform-alignment}.
Would follow from uniform eigenvalue-gap
$\lambda_n^{(c)} - \lambda_{n+1}^{(c)}
\ge \delta_\kappa c^{-1/2}$
or from a uniform resolvent bound
(Paper~X).
\end{problem}

\begin{problem}[Sharp operator-norm rate]
\label{prob:opnorm-sharp}
Is $\|\Hc - \Hspec\| \asymp c^{-1/4}$?
\end{problem}

\begin{problem}[Density of
$\{\lambda_n^{(\infty)}\}$]
\label{prob:density}
Is $\{\lambda_n^{(\infty)}\}$ dense in $[0,1]$?
(Equivalent to $\sigma(\Hspec)=[0,1]$,
Paper~XI, Thm.~4.1.)
\end{problem}

\begin{problem}[Unconditional ESSA]
\label{prob:esa-unconditional}
Prove ESSA of $\Hspec$ without
Hypothesis~\ref{hyp:decay}.
Not directly equivalent to completeness
of $\{\Phi_n^{(\infty)}\}$ without
additional spectral structure
(Remark~\ref{rem:deficiency-orthogonal}).
\end{problem}
