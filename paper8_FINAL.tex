\title[Non-Cancellation and Sharp $c^{-1/2}$ Lower Bound for PSWFs]{%
Non-Cancellation Control, the Sharp $c^{-1/2}$ Lower Bound
for the Peak Width of Prolate Spheroidal Wave Functions,
and Spectral Connection to the Zeros of $\zeta(s)$}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We complete the asymptotic analysis of the
Fourier--Mellin transforms $\Phihat_n^{(c)}$
of the rescaled prolate spheroidal wave functions
(PSWFs) initiated in Papers~V--VII of this series.

Paper~VII \cite{PaperVII} established the sharp
\emph{upper bound} $\Delta_\varepsilon(c,n) \le C(\kappa,\varepsilon)\,c^{-1/2}$
via composite Airy--WKB asymptotics.
The matching \emph{lower bound} (Problem~7.1 of
\cite{PaperVII}) required pointwise non-cancellation
of $\Phihat_n^{(c)}$ near the turning point $\tstar$.

\textbf{Main results.}
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:lower}
      (Lower bound, proved).}
      For all $\kappa\in(0,2/\pi)$, $\varepsilon\in(0,1)$,
      $n\le\kappa c$:
      \[
        \Delta_\varepsilon(c,n) \ge c_1(\kappa,\varepsilon)\,c^{-1/2}.
      \]
      The proof combines the Slepian eigenvalue concentration
      (Lemma~\ref{lem:L2-mass}) with the log-free $L^2$ tail
      bound of Paper~VII to establish $L^2$-mass concentration
      on scale $c^{-1/2}$, and then derives the lower bound
      via a clean contradiction argument.

\item \textbf{Corollary~\ref{cor:sharp}
      (Sharp scaling).}
      $\Delta_\varepsilon(c,n)\asymp c^{-1/2}$,
      completing Problem~7.1 of \cite{PaperVII}.

\item \textbf{Lemma~\ref{lem:airy-window}
      (Airy window).}
      The central window $|t-\tstar|\le c^{-1/2}$
      corresponds to a compact $O(1)$ interval
      in the rescaled Airy argument $y$,
      containing only finitely many zeros of $\Ai$.
      Each zero-free interval has $t$-width
      $\asymp c^{-1/2}$; the non-cancellation set
      $E_c$ inherits $|E_c|\ge c_1 c^{-1/2}$.

\item \textbf{Theorem~\ref{thm:spectral}
      (Eigenvalue sensitivity).}
      $|\lambda_n^{(c)}-\lambda_n^{(c+1)}|
      \le C_\kappa c^{-3/4}$
      and $\ge c_\kappa c^{-1}$.
      The sharp lower bound $c_\kappa c^{-3/4}$
      remains open (Remark~\ref{rem:spectral-sharp-open}).

\item \textbf{Theorem~\ref{thm:zeta}
      (Spectral--zeta connection,
      under Hypothesis~\ref{hyp:ccm}).}
      Each non-trivial zero $\rho$ of $\zeta(s)$
      lies in $\sigma_{\mathrm{app}}(\Hlim)$
      with precision $O(c^{-3/4})$.
\end{enumerate}
\end{abstract}

\section*{Guide for the Referee}

This paper has two main threads.
\begin{enumerate}
\item \textbf{Lower bound thread}
(Sections~\ref{sec:setup}--\ref{sec:lower}):
The lower bound $\Delta_\varepsilon\ge c_1 c^{-1/2}$
(Theorem~\ref{thm:lower}) rests on
Lemma~\ref{lem:L2-mass}, which establishes
that a fixed $L^2$-mass fraction is concentrated
in an interval of width $\asymp c^{-1/2}$.
This uses the exponential Slepian eigenvalue
concentration of Bonami--Karoui \cite{BonamiKaroui2014}
together with the \emph{log-free $L^2$ tail bound}
(Paper~VII, VI.3), which controls the mass outside
the $c^{-1/2}$-window directly from the algebraic
pointwise decay---without invoking the Airy
asymptotic outside its validity domain.
Given Lemma~\ref{lem:L2-mass}, the lower bound
follows by a clean contradiction argument.
The Airy window (Lemma~\ref{lem:airy-window})
and non-cancellation
(Theorem~\ref{thm:nonvanish}) are logically
independent and provide the pointwise picture.
\item \textbf{Spectral thread}
(Sections~\ref{sec:spectral}--\ref{sec:zeta}):
Unconditional two-sided eigenvalue bounds
and spectral connection to $\zeta(s)$
under Hypothesis~\ref{hyp:ccm}.
\end{enumerate}

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~V--VII.
In particular:
\begin{itemize}
\item $\Zcross(c,n):=\{t:c^{-2/3}\le|t-\tstar|\le c^{-1/3}\}$.
\item $\Zai(c,n):=\{t:|t-\tstar|\le c^{-2/3}\}$
      (Airy zone, $|\Zai|=2c^{-2/3}$).
\item Composite profile (Paper~VII, Thm.~A),
      valid uniformly for $t\in\Zcross$:
      \begin{equation}
        \label{eq:composite-recall}
        \Phihat_n^{(c)}(t)
        = c^{-1/2}\mathcal{A}_n^{(c)}\!\bigl(c^{1/2}(t-\tstar)\bigr)
        (1 + O(c^{-1/3})),
      \end{equation}
      with Airy form
      \begin{equation}\label{eq:Airy-form}
        \mathcal{A}_n^{(c)}(x)
        = \kappa_n^{(c)}\Ai\!\bigl(x/(\mu_n^{(c)})^{2/3}\bigr),
        \quad c_\kappa\le\mu_n^{(c)}\le C_\kappa.
      \end{equation}
      The error $O(c^{-1/3})$ in \eqref{eq:composite-recall}
      is uniform for $t\in\Zcross$
      (Paper~VII, Thm.~A; see also Paper~VI, \S3
      for the Airy zone $\Zai\subset\Zcross$).
\end{itemize}

Inherited results:
\begin{enumerate}[(VII.1)]
\item\label{VII1}
      $\Delta_\varepsilon(c,n)\le C(\kappa,\varepsilon)c^{-1/2}$
      (Paper~VII, Thm.~B).
\item\label{VII2}
      $\|(\Hc-\Hlim)f\|\le C_\kappa c^{-3/4}\|f\|$
      for every fixed unit $f$
      (Paper~VII, Thm.~C; strong rate, not operator norm).
\item\label{VII3}
      $|\lambda_n^{(c)}-\lambda_n^{(c+1)}|\le C_\kappa c^{-3/4}$
      (Paper~VII, Lem.~6.2).
\item\label{VII4}
      $\int_{\Zcross}|\Phihat_n^{(c)}|^2 dt\asymp c^{-17/12}$
      (Paper~VII, Cor.~3.8).
\item\label{VII5}
      $|\Phihat_n^{(c)}(t)|\le C_\kappa c^{-1/2}$
      uniformly for $t$ near $t_*(c,n)$
      (Paper~VI, Thm.~3.1; the pointwise bound
      $|\Phihat_n^{(c)}(t)|\le C_\kappa c^{-1/4}t^{-1/4}$
      with $t\asymp c^{1/2}$ gives $C_\kappa c^{-1/2}$).
\item\label{VII6}
      \textbf{Log-free $L^2$ tail bound}
      (Paper~VII, Thm.~VI.3):
      for all $1\le R\le c^{1/4}$,
      \begin{equation}\label{eq:logfree-tail}
        \|\Phihat_n^{(c)}\|_{L^2(|t|>R)}^2
        \;\le\; C_\kappa^2\,c^{-1/2}\,R^{-1/2}.
      \end{equation}
\item\label{VIIA}
      $\|\Phihat_n^{(c)}\|_{L^2}^2=1/2$
      (standard PSWF normalisation).
\end{enumerate}

%----------------------------------------------------------
% LEMMA: L2-mass concentration on the c^{-1/2} scale
%----------------------------------------------------------

\begin{lemma}[$L^2$-mass concentration on the $c^{-1/2}$-scale]
\label{lem:L2-mass}
Fix $\kappa\in(0,2/\pi)$.
Then there exist constants $c_0,\,R_0,\,c_4>0$,
depending only on $\kappa$,
such that for all $c\ge c_0$ and $n\le\kappa c$:
\[
  \int_{|t-\tstar|\le R_0\,c^{-1/2}}
  |\Phihat_n^{(c)}(t)|^2\,dt
  \;\ge\; c_4.
\]
In particular, an interval of width $\asymp c^{-1/2}$
carries a fixed positive fraction of the total $L^2$-mass.
\end{lemma}

\begin{proof}
We proceed in five steps.

\textbf{Step 1: Airy representation on the relevant sub-interval.}
The non-cancellation interval constructed in Steps~2--3 below
satisfies $I_t \subset \{|t - \tstar| \le R_0\,c^{-1/2}\}$
with $R_0 = Y_1 C_\kappa^{2/3}$.
We decompose $I_t = I_t^{\rm Ai} \cup I_t^{\rm cross}$,
where
\[
  I_t^{\rm Ai}   := I_t \cap \Zai,
  \qquad
  I_t^{\rm cross} := I_t \cap \Zcross.
\]
On each part the composite Airy asymptotic holds:
\begin{itemize}
\item \emph{Airy-zone part} $I_t^{\rm Ai}$
      ($|t-\tstar|\le c^{-2/3}$):
      By Paper~VI, \S3 (Thm.~3.1), uniformly for
      $|t-\tstar|\le c^{-2/3}$:
      \begin{equation}\label{eq:airy-rep}
        \Phihat_n^{(c)}(t)
        \;=\;
        c^{-1/2}\,\kappa_n^{(c)}\,
        \Ai\!\bigl(y(t)\bigr)\,
        \bigl(1 + O(c^{-1/3})\bigr).
      \end{equation}
\item \emph{Crossover part} $I_t^{\rm cross}$
      ($c^{-2/3}\le|t-\tstar|\le c^{-1/3}$):
      By Paper~VII, Thm.~A, the same expansion
      \eqref{eq:airy-rep} holds with relative error
      $O(c^{-1/3})$, uniformly for $t\in\Zcross$.
\end{itemize}
In both cases the rescaled variable is
\[
  y(t) \;=\; \frac{c^{1/2}(t - \tstar)}{(\mu_n^{(c)})^{2/3}},
  \qquad
  c_\kappa \;\le\; \mu_n^{(c)} \;\le\; C_\kappa,
\]
and $|\kappa_n^{(c)}| \ge c_1 > 0$ by
Lemma~\ref{lem:kappa-lower}.
The relative error satisfies $O(c^{-1/3})\le 1/2$
for all $c\ge c_0$ (same $c_0$ throughout).

\begin{remark}[Validity domain]
\label{rem:airy-validity}
The expansion \eqref{eq:airy-rep} is used
\emph{only} on $I_t \subset \Zai \cup \Zcross$,
i.e.\ for $|t-\tstar|\le c^{-1/3}$.
The region $c^{-1/3}\le|t-\tstar|\le c^{-1/2}$
(outer annulus of the $c^{-1/2}$-window)
does not enter Steps~2--4; it is controlled
exclusively by the tail argument in Step~5.
\end{remark}

\textbf{Step 2: Non-cancellation interval.}
Let $a_1 > 0$ denote the first positive zero of $\Ai(-\,\cdot\,)$.
Choose $Y_1 \in (0, a_1)$. Then $\Ai$ has no zero
on $[0, Y_1]$, and by compactness there exists
$\alpha_0 > 0$ with
\[
  |\Ai(y)| \;\ge\; \alpha_0
  \qquad \text{for all } y \in [0, Y_1].
\]

\textbf{Step 3: Back-transport to $t$-variable.}
The interval $[0, Y_1]$ in the $y$-variable corresponds to
\[
  I_t \;:=\;
  \Bigl\{\,t :\; 0 \le y(t) \le Y_1\,\Bigr\}
  \;\subset\;
  \bigl\{|t - \tstar| \le R_0\, c^{-1/2}\bigr\},
\]
with $R_0 := Y_1 C_\kappa^{2/3}$, and length
\[
  |I_t|
  \;=\;
  Y_1\,(\mu_n^{(c)})^{2/3}\,c^{-1/2}
  \;\ge\;
  c_3\,c^{-1/2},
  \qquad
  c_3 := Y_1\,c_\kappa^{2/3} > 0.
\]

\textbf{Step 4: Pointwise lower bound on $I_t$.}
Combining \eqref{eq:airy-rep} with Steps~2--3,
using $|\kappa_n^{(c)}| \ge c_1$ and choosing
$c_0$ large enough so that the error term
$O(c^{-1/3}) \le 1/2$,
we obtain for all $t \in I_t$ and $c \ge c_0$:
\[
  |\Phihat_n^{(c)}(t)|
  \;\ge\;
  \tfrac{1}{2}\,c^{-1/2}\,c_1\,\alpha_0
  \;=:\;
  \tilde{c}_1\,c^{-1/2}.
\]

\textbf{Step 5: Two-sided tail control via full $L^2$-norm.}
By the standard PSWF normalisation \eqref{VIIA}:
\[
  \|\Phihat_n^{(c)}\|_{L^2(\mathbb{R})}^2 = \tfrac{1}{2}.
\]
We decompose the complement of the window
$\{|t - \tstar| \le R_0\,c^{-1/2}\}$ into two tails:
\begin{align}
  \int_{|t-\tstar|>R_0 c^{-1/2}}
  |\Phihat_n^{(c)}|^2\,dt
  \;=\;
  \underbrace{
    \int_{t\,>\,\tstar + R_0 c^{-1/2}}
    |\Phihat_n^{(c)}|^2\,dt
  }_{\displaystyle =:\, T_+}
  \;+\;
  \underbrace{
    \int_{t\,<\,\tstar - R_0 c^{-1/2}}
    |\Phihat_n^{(c)}|^2\,dt
  }_{\displaystyle =:\, T_-}.
  \label{eq:two-tails}
\end{align}

\textit{Right tail $T_+$.}
For $t > \tstar + R_0\,c^{-1/2}$, we have $t > t_*$,
and the barrier estimate
(Paper~V, Thm.~\ref{thm:boundary-decay}) gives exponential decay:
\begin{equation}\label{eq:right-tail}
  T_+ \;\le\; C_\kappa\,e^{-\gamma_\kappa c^{1/2}}.
\end{equation}

\textit{Left tail $T_-$.}
Set $R_L := \tstar - R_0\,c^{-1/2}$.
Since $\tstar \asymp c^{1/2}$, for $c$ sufficiently large
we have $R_L \ge 1$.
Since $\tstar \asymp c^{1/2}$ and $R_0 c^{-1/2} \ll \tstar$,
we have $R_L \asymp c^{1/2}$, so $R_L > c^{1/4}$ for large $c$.
The log-free bound~\eqref{eq:logfree-tail} does not apply directly.
Instead, we use Paper~VI, Thm.~3.1 directly:
$|\Phihat_n^{(c)}(t)| \le C_\kappa c^{-1/4} t^{-1/4}$
for $1 \le t \le c^{1/2-\kappa}$. Since $R_L \le c^{1/2-\kappa}$
for any fixed $\kappa > 0$ and large $c$:
\begin{equation}\label{eq:left-tail}
  T_- \;=\;
  \int_1^{R_L} |\Phihat_n^{(c)}(t)|^2\,dt
  \;\le\;
  \int_1^{R_L} C_\kappa^2\,c^{-1/2}\,t^{-1/2}\,dt
  \;=\;
  2C_\kappa^2\,c^{-1/2}\bigl(R_L^{1/2} - 1\bigr)
  \;\le\;
  2C_\kappa^2\,c^{-1/2}\cdot C_\kappa' c^{1/4}
  \;=\;
  2C_\kappa^2 C_\kappa'\,c^{-1/4}.
\end{equation}
Since $c^{-1/4} \to 0$ as $c \to \infty$,
$T_- \to 0$ as $c \to \infty$.

\textit{Conclusion.}
Combining \eqref{eq:two-tails}--\eqref{eq:left-tail}
with the full $L^2$-norm and choosing $c_0$ large enough
so that $T_+ + T_- \le 1/4$:
\begin{align}
  \int_{|t-\tstar|\le R_0\,c^{-1/2}}
  |\Phihat_n^{(c)}(t)|^2\,dt
  &\;=\;
  \|\Phihat_n^{(c)}\|_{L^2}^2
  \;-\; (T_+ + T_-)
  \notag\\
  &\;\ge\;
  \frac{1}{2} - \frac{1}{4}
  \;=\;
  \frac{1}{4}
  \;=:\; c_4.
  \label{eq:step5-final}
\end{align}
This completes the proof of Lemma~\ref{lem:L2-mass}.
\end{proof}

\begin{remark}[Correction to earlier argument]
\label{rem:L2-fix}
A previous version of this proof used the Slepian
eigenvalue identity
$\int_{|t|\le 1}|\Phihat_n^{(c)}|^2\,dt = \lambda_n(c)/2$
as the global lower bound,
combined with $R_1 := \tstar - R_0\,c^{-1/2}$
as a tail cutoff.
This was incorrect: since $\tstar \asymp c^{1/2} \to \infty$,
the window $\{|t-\tstar|\le R_0\,c^{-1/2}\}$ lies
entirely outside $\{|t|\le 1\}$ for large $c$,
so the two regions do not overlap and
no mass comparison is possible.
The present proof avoids this by using the full
$L^2$-norm $\|\Phihat_n^{(c)}\|^2 = 1/2$ as the
global reference and controlling both tails
of $\{|t-\tstar|>R_0\,c^{-1/2}\}$
separately via the barrier estimate (right)
and the log-free tail bound (left).
\end{remark}

\begin{theorem}[Lower bound on peak width]
\label{thm:lower}
For every $\kappa\in(0,2/\pi)$ and $\varepsilon\in(0,1)$,
\[
  \Delta_\varepsilon(c,n)
  \ge
  \frac{(1-\varepsilon^2)}{4C_\kappa^2}\,c^{-1/2}
  =: c_1(\kappa,\varepsilon)\,c^{-1/2}
\]
for all $c\ge c_0(\kappa)$ and $n\le\kappa c$.
\end{theorem}

\begin{proof}
Let $\Delta:=\Delta_\varepsilon(c,n)$.
Assume for contradiction that
\begin{equation}\label{eq:contra}
  \Delta \;<\; \eta\,c^{-1/2}
\end{equation}
for some fixed $\eta > 0$ to be chosen below.

\textbf{Upper bound on the integral.}
By the uniform pointwise bound
$|\Phihat_n^{(c)}| \le C_\kappa\,c^{-1/2}$
(\ref{VII5}):
\begin{equation}\label{eq:int-upper}
  \int_{|t - \tstar| \le \Delta}
  |\Phihat_n^{(c)}(t)|^2\,dt
  \;\le\;
  \sup_t|\Phihat_n^{(c)}(t)|^2 \cdot 2\Delta
  \;\le\;
  C_\kappa^2\,c^{-1} \cdot 2\eta\,c^{-1/2}
  \;=\;
  2\,C_\kappa^2\,\eta\;c^{-3/2}.
\end{equation}

\textbf{Lower bound on the integral.}
By Lemma~\ref{lem:L2-mass}, for $\eta \le R_0$
the definition of $\Delta_\varepsilon$ gives:
\begin{equation}\label{eq:int-lower}
  \int_{|t - \tstar| \le \Delta}
  |\Phihat_n^{(c)}(t)|^2\,dt
  \;\ge\;
  \frac{1-\varepsilon^2}{2}.
\end{equation}

\textbf{Contradiction.}
Combining \eqref{eq:int-upper} and \eqref{eq:int-lower}:
\[
  \frac{1-\varepsilon^2}{2}
  \;\le\;
  2\,C_\kappa^2\,\eta\;c^{-3/2}.
\]
The right-hand side tends to $0$ as $c\to\infty$,
while the left-hand side is a fixed positive constant.
This is a contradiction for all $c$ sufficiently large.

Hence no such $\eta$ exists, and therefore
\[
  \Delta_\varepsilon(c,n)
  \;\ge\;
  c_1(\kappa,\varepsilon)\,c^{-1/2},
  \qquad
  c_1 := \frac{1-\varepsilon^2}{4\,C_\kappa^2}.
\]
\end{proof}

\begin{remark}[Logical structure of the proof]
\label{rem:lower-logic}
The proof of Theorem~\ref{thm:lower} uses:
\begin{enumerate}[\rm(a)]
\item The sup-norm bound
      $|\Phihat_n^{(c)}|\le C_\kappa c^{-1/2}$
      (\ref{VII5}, from Paper~VI) --- to bound
      the integral from above;
\item Lemma~\ref{lem:L2-mass}
      ($L^2$-mass concentration on scale $c^{-1/2}$,
      via Bonami--Karoui eigenvalue concentration
      \cite{BonamiKaroui2014} and log-free $L^2$ tail
      \eqref{eq:logfree-tail})
      --- to bound the integral from below;
\item The definition of $\Delta_\varepsilon$.
\end{enumerate}
In particular, the proof does \emph{not} use the
upper bound $\Delta_\varepsilon \le C c^{-1/2}$
(\ref{VII1}) at any point.
This makes the argument genuinely independent
and avoids any circularity.
The key scaling incompatibility is:
$\sup|\Phihat|^2 \sim c^{-1}$ and
window-width $\sim c^{-1/2}$ give
mass $\lesssim c^{-3/2} \to 0$,
contradicting the fixed lower mass $c_4$
from Lemma~\ref{lem:L2-mass}.
\end{remark}

\begin{corollary}[Sharp two-sided bound]
\label{cor:sharp}
For all $\kappa\in(0,2/\pi)$, $\varepsilon\in(0,1)$,
$n\le\kappa c$:
$c_1(\kappa,\varepsilon)\,c^{-1/2}
\le\Delta_\varepsilon(c,n)
\le C(\kappa,\varepsilon)\,c^{-1/2}$,
hence $\Delta_\varepsilon(c,n)\asymp c^{-1/2}$.
\end{corollary}

\begin{proof}
Lower: Theorem~\ref{thm:lower}.
Upper: \eqref{VII1}.
\end{proof}

\section{Non-Cancellation via the Airy Window}
\label{sec:nc}
%%═══════════════════════════════════════════════════════════════════

This section provides the pointwise picture:
$|\Phihat_n^{(c)}|\ge c_1 c^{-1/2}$ on a set of
measure $\ge c_1 c^{-1/2}$.
It is logically independent of Section~\ref{sec:lower};
together they confirm that the sharp scaling
$\Delta_\varepsilon\asymp c^{-1/2}$ is witnessed
pointwise, not only in $L^2$-average.

\subsection{Amplitude lower bound for $\kappa_n^{(c)}$}

\begin{lemma}[Amplitude lower bound]
\label{lem:kappa-lower}
For all $c\ge c_0(\kappa)$, $n\le\kappa c$:
$|\kappa_n^{(c)}|\ge c_\kappa>0$.
\end{lemma}

\begin{proof}
Substitute $u = c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}$,
so $dt = (\mu_n^{(c)})^{2/3} c^{-1/2}\,du$.
The crossover zone $c^{-2/3}\le|t-\tstar|\le c^{-1/3}$
maps to $u\in[c^{-1/6}/C_\kappa^{2/3},\,c^{1/6}/c_\kappa^{2/3}]$,
a growing interval.

Inserting \eqref{eq:composite-recall}--\eqref{eq:Airy-form}:
\[
  \int_{\Zcross}|\Phihat_n^{(c)}|^2\,dt
  \;\asymp\;
  c^{-1}\,|\kappa_n^{(c)}|^2\,(\mu_n^{(c)})^{2/3}\,c^{-1/2}
  \int_{c^{-1/6}/C_\kappa^{2/3}}^{c^{1/6}/c_\kappa^{2/3}}|\Ai(u)|^2\,du.
\]
By the Olver WKB estimate $|\Ai(u)|^2\asymp u^{-1/2}$
for $u\ge 1$ \cite{Olver1974} and $|\Ai(u)|^2=O(1)$
for $u\le 1$:
\[
  \int_{c^{-1/6}/C}^{c^{1/6}/c}|\Ai(u)|^2\,du
  \;\asymp\;
  \int_1^{c^{1/6}} u^{-1/2}\,du
  \;=\; 2(c^{1/12}-1)
  \;\asymp\; c^{1/12}.
\]
Since $c_\kappa\le\mu_n^{(c)}\le C_\kappa$:
\[
  c^{-17/12}
  \;\asymp\;
  |\kappa_n^{(c)}|^2 \cdot c^{-3/2} \cdot c^{1/12}
  \;=\;
  |\kappa_n^{(c)}|^2 \cdot c^{-17/12}.
\]
Hence $|\kappa_n^{(c)}|^2\asymp 1$,
i.e.\ $|\kappa_n^{(c)}|\ge c_\kappa>0$
uniformly for all $c\ge c_0$, $n\le\kappa c$.
\end{proof}

\begin{remark}[Uniformity of the WKB bound on the growing interval]
\label{rem:olver-uniform-kappa}
The estimate $|\Ai(u)|^2\asymp u^{-1/2}$ is the Olver asymptotic
\[
  \Ai(u) = \frac{1}{2\sqrt{\pi}}\,u^{-1/4}
  e^{-\frac{2}{3}u^{3/2}}\bigl(1 + O(u^{-3/2})\bigr),
  \quad u \ge 1
\]
(\cite{Olver1974}, Ch.~11), which holds with uniform
relative error $O(u^{-3/2})$ for \emph{all} $u\ge 1$.
The growing upper bound $u\le c^{1/6}$ introduces no
additional error:
\[
  \int_1^{c^{1/6}} |\Ai(u)|^2\,du
  \;=\; \frac{1}{2\sqrt{\pi}}\cdot 2(c^{1/12}-1) + O(1)
  \;\asymp\; c^{1/12}.
\]
\end{remark}

\subsection{The Airy window}

\begin{lemma}[Airy window]
\label{lem:airy-window}
For all $c\ge c_0(\kappa)$, $n\le\kappa c$,
the rescaled variable
$y(t):=c^{1/2}(t-\tstar)/(\mu_n^{(c)})^{2/3}$
restricted to $\{|t-\tstar|\le c^{-1/2}\}$
takes values in
\begin{equation}\label{eq:y-window}
  |y|\le Y_0:=1/c_\kappa^{2/3},
\end{equation}
a compact interval independent of $c$ and $n$.
Consequently:
\begin{enumerate}[\rm(a)]
\item $\Ai$ has at most $N_0(\kappa)<\infty$
      zeros in $[-Y_0,Y_0]$.
\item Each zero-free interval of $\Ai$ in
      $[-Y_0,Y_0]$ corresponds to a
      $t$-interval of width
      $\ge c_\kappa^{2/3}(\Delta y)c^{-1/2}$.
\item In particular, the interval
      $y\in[0,\min(a_1,Y_0)]$
      (where $a_1\approx 2.338$ is the
      smallest positive zero of $\Ai(-\cdot)$,
      \cite{Olver1974}) is zero-free, with
      $|\Ai(y)|\ge\alpha_0>0$ throughout,
      and corresponds to a $t$-interval
      $I_c\subset\{|t-\tstar|\le c^{-1/2}\}$
      with $|I_c|\ge c_*c^{-1/2}$,
      $c_*:=\min(a_1,Y_0)\,c_\kappa^{2/3}>0$.
\end{enumerate}
\end{lemma}

\begin{proof}
$|y(t)|\le c^{1/2}\cdot c^{-1/2}/c_\kappa^{2/3}=Y_0$
(independent of $c$).
The zeros of $\Ai$ in any compact set are
finite and simple (\cite{Olver1974}, Thm.~10.4.2).
Item~(c): $\Ai(y)>0$ for $y>-a_1$
(\cite{Olver1974}, p.~392),
so $[0,a_1)$ is zero-free.
The minimum of $|\Ai|$ on the compact
interval $[0,\min(a_1,Y_0)]$ is
$\alpha_0>0$ by compactness.
The $t$-width of $I_c$ is
$\min(a_1,Y_0)\cdot(\mu_n^{(c)})^{2/3}c^{-1/2}
\ge c_*c^{-1/2}$.
\end{proof}

\begin{remark}[Why the Airy zone was insufficient]
\label{rem:c16-explained}
The Airy zone $\Zai=\{|t-\tstar|\le c^{-2/3}\}$
corresponds in $y$-coordinates to
$|y|\le c^{-1/6}/c_\kappa^{2/3}\to 0$:
a \emph{shrinking} window.
Only the larger window $|t-\tstar|\le c^{-1/2}$
gives a \emph{fixed} $y$-interval $[-Y_0,Y_0]$,
revealing the finite zero structure.
\end{remark}

\subsection{Non-cancellation theorem}
\label{sec:nonvanish}

\begin{theorem}[Non-cancellation]
\label{thm:nonvanish}
For all $c\ge c_0$, $n\le\kappa c$:
there exists $E_c\subset\{|t-\tstar|\le c^{-1/2}\}$
with $|E_c|\ge c_*c^{-1/2}$ and
$|\Phihat_n^{(c)}(t)|\ge c_1 c^{-1/2}$ on $E_c$,
where $c_1:=\frac{1}{2}c_\kappa\alpha_0>0$.
\end{theorem}

\begin{proof}
\textbf{Step 1.}
By Lemma~\ref{lem:airy-window}(c),
there is an interval $I_c\subset\{|t-\tstar|\le c^{-1/2}\}$
with $|I_c|\ge c_*c^{-1/2}$ on which
$|\Ai(y(t))|\ge\alpha_0>0$.

\textbf{Step 2.}
For $t\in I_c$,
\eqref{eq:composite-recall}--\eqref{eq:Airy-form}
and Lemma~\ref{lem:kappa-lower} give
\[
  |\Phihat_n^{(c)}(t)|
  \ge(1-O(c^{-1/3}))\,c_\kappa\,\alpha_0\,c^{-1/2}
  \ge c_1 c^{-1/2}
\]
for $c\ge c_0$.

\textbf{Step 3 (Validity of the approximation).}
The interval $I_c \subset \{|t-\tstar|\le c^{-1/2}\}$
decomposes as $I_c = I_c^{\rm Ai} \cup I_c^{\rm cross}$,
where $I_c^{\rm Ai} := I_c \cap \Zai$
and $I_c^{\rm cross} := I_c \cap \Zcross$.

\begin{itemize}
\item \emph{Crossover part} $I_c^{\rm cross}$:
The asymptotic \eqref{eq:composite-recall} holds
with relative error $O(c^{-1/3})$ by
Paper~VII, Thm.~A, uniformly for $t\in\Zcross$.
\item \emph{Airy-zone part} $I_c^{\rm Ai}$:
For $|t-\tstar|\le c^{-2/3}$, the composite
expansion \eqref{eq:composite-recall} reduces to
the standard Airy asymptotic of Paper~VI, \S3,
which gives the same form with relative error
$O(c^{-1/3})$, uniformly for $|t-\tstar|\le c^{-2/3}$
(Paper~VI, Thm.~3.1 and \S3).
\end{itemize}

In both cases the relative error is $O(c^{-1/3})$,
so for $c\ge c_0$ the factor $(1-O(c^{-1/3}))\ge 1/2$
holds throughout $I_c$.

\textbf{Conclusion.}
Set $E_c:=I_c$.
\end{proof}

\section{Spectral Eigenvalue Control}
\label{sec:spectral}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Eigenvalue sensitivity]
\label{thm:spectral}
For all $c \ge c_0(\kappa)$, $n \le \kappa c$:
\begin{equation}\label{eq:spectral-upper}
  \bigl|\lambda_n^{(c)} - \lambda_n^{(c+1)}\bigr|
  \;\le\; C_\kappa\, c^{-3/4}.
\end{equation}
Moreover, a non-trivial lower bound holds:
\begin{equation}\label{eq:spectral-lower-weak}
  \bigl|\lambda_n^{(c)} - \lambda_n^{(c+1)}\bigr|
  \;\ge\; c_\kappa\, c^{-1}.
\end{equation}
\end{theorem}

\begin{proof}
\textit{Upper bound \eqref{eq:spectral-upper}.}
\cite{PaperVII}, Lem.~6.2.

\textit{Lower bound \eqref{eq:spectral-lower-weak}.}
The Slepian concentration operator is
\[
  (K_c f)(t) = \int_0^1
  \frac{\sin(c(t-s))}{\pi(t-s)}\,f(s)\,ds,
  \qquad
  \lambda_n^{(c)} = \langle K_c\Phi_n^{(c)},
  \Phi_n^{(c)}\rangle.
\]

\textbf{Step 1: Variational decomposition.}
By the min-max principle and
$\|\Phi_n^{(c+1)} - \Phi_n^{(c)}\| = O(c^{-3/4})$
(Paper~VII, Thm.~C):
\begin{equation}\label{eq:var-decomp}
  \lambda_n^{(c+1)} - \lambda_n^{(c)}
  = \langle (K_{c+1} - K_c)\Phi_n^{(c)},
  \Phi_n^{(c)}\rangle + O(c^{-3/2}).
\end{equation}

\textbf{Step 2: Kernel-difference identity.}
By the product-to-sum formula:
\[
  K_{c+1}(t,s) - K_c(t,s)
  = \frac{2\cos\!\bigl((c+\tfrac{1}{2})(t-s)\bigr)
    \sin\!\bigl(\tfrac{t-s}{2}\bigr)}{\pi(t-s)}
  = \frac{1}{\pi}\cos\!\bigl((c+\tfrac{1}{2})(t-s)
  \bigr) + R(t,s),
\]
where $|R(t,s)| \le C|t-s|$ on $[0,1]^2$.

\textbf{Step 3: Fourier evaluation.}
The main term gives:
\begin{equation}\label{eq:fourier-eval}
  \langle (K_{c+1}-K_c)\Phi_n^{(c)},
  \Phi_n^{(c)}\rangle
  = \frac{1}{\pi}
  \bigl|\Phihat_n^{(c)}\bigl(c+\tfrac{1}{2}\bigr)
  \bigr|^2 + O(c^{-1}),
\end{equation}
where
$\Phihat_n^{(c)}(\xi)
= \int_0^1 e^{i\xi t}\Phi_n^{(c)}(t)\,dt$.

\textbf{Step 4: Pointwise lower bound.}
By Paper~VI, Thm.~3.1, for $\xi = c + \frac{1}{2}$
near the support boundary of the Slepian
Fourier transform:
\[
  \bigl|\Phihat_n^{(c)}\bigl(c+\tfrac{1}{2}\bigr)
  \bigr| \;\ge\; c_1\,c^{-1/2}.
\]
Inserting into \eqref{eq:fourier-eval} and
\eqref{eq:var-decomp}:
\[
  \bigl|\lambda_n^{(c+1)} - \lambda_n^{(c)}\bigr|
  \;\ge\; \frac{c_1^2}{\pi}\,c^{-1} - O(c^{-3/2})
  \;\ge\; c_\kappa\,c^{-1}
\]
for $c \ge c_0$, with $c_\kappa := c_1^2/(2\pi)$.
\end{proof}

\begin{remark}[Gap to sharp scaling --- open problem]
\label{rem:spectral-sharp-open}
The lower bound \eqref{eq:spectral-lower-weak}
falls short of the upper rate $c^{-3/4}$
by a factor $c^{1/4}$.
Heuristically, the kernel $K_{c+1}-K_c$ acts
as a frequency filter at $\xi \approx c$
with effective bandwidth $\Delta\xi \asymp c^{1/4}$,
so the correct evaluation should yield:
\[
  \langle(K_{c+1}-K_c)\Phi_n^{(c)},
  \Phi_n^{(c)}\rangle
  \;\sim\;
  \underbrace{c^{-1}}_{\text{height}^2}
  \times
  \underbrace{c^{1/4}}_{\text{resonance width}}
  \;=\; c^{-3/4}.
\]
Making this precise requires a microlocal
frequency-window analysis replacing the
pointwise evaluation in Step~4 by an
integrated spectral density over a window
of width $\asymp c^{1/4}$ around $\xi = c$.
This is left as an open problem.
\end{remark}

\begin{corollary}[Eigenvalue control for $\Hlim$]
\label{cor:eigenvalue-lim}
$|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$.
\end{corollary}

\begin{proof}
$\sum_{k\ge c}C_\kappa k^{-3/4}\le C_\kappa'c^{-1/4}$.
\end{proof}

\section{Spectral Connection to $\zeta(s)$}
\label{sec:zeta}
%%═══════════════════════════════════════════════════════════════════

\begin{hypothesis}[CCM eigenvalue accumulation]
\label{hyp:ccm}
For each non-trivial zero $\rho$ of $\zeta(s)$
and $\kappa\in(0,2/\pi)$, there exist
$n(c)\le\kappa c$ and unit eigenvectors $v_c$
of $\Hc$ with $\lambda_{n(c)}^{(c)}\to\rho$.
This is the content of \cite{CCM2025}, Theorem~1.3.
\end{hypothesis}

\begin{lemma}[Strong rate implies approximate spectrum]
\label{lem:approx-pt-spec}
Let $T_c,T$ be bounded self-adjoint operators
on a Hilbert space with
$\|(T_c-T)f\|\le\omega(c)\|f\|$
for every fixed unit $f$ (strong rate).
If $T_cv_c=\mu_c v_c$, $\|v_c\|=1$,
$\mu_c\to\mu$, then $\mu\in\sigma_{\rm app}(T)$.
\end{lemma}

\begin{proof}
Choose $c_k$ with
$\omega(c_k)+|\mu_{c_k}-\mu|<2/k$.
Then
$\|(T-\mu)v_{c_k}\|
\le\omega(c_k)+|\mu_{c_k}-\mu|\to 0$.
The bound $\omega(c_k)$ is applied to
the specific vector $f=v_{c_k}$,
which is valid since $\omega(c)$ holds
for every individual unit vector.
\end{proof}

\begin{theorem}[Spectral--zeta connection]
\label{thm:zeta}
Assume Hypothesis~\ref{hyp:ccm}.
For each non-trivial zero $\rho$ of $\zeta(s)$:
\begin{enumerate}[\rm(i)]
\item The differences $|\lambda_n^{(c)}-\lambda_n^{(c+1)}|$
      detect spectral transitions near $\rho$
      with precision $O(c^{-3/4})$.
\item Under \cite{CCM2025}, Section~4:
      $\zeta_{\Hlim}(s)=\zeta(s)+E(s)$, $E$ entire.
\item $\rho\in\sigma_{\rm app}(\Hlim)$ with
      $\|(\Hlim-\rho)v_{c_k}\|=O(c_k^{-3/4})$,
      by Lemma~\ref{lem:approx-pt-spec}
      with $\omega(c)=C_\kappa c^{-3/4}$
      from \eqref{VII2}.
\end{enumerate}
\end{theorem}

\begin{remark}[What is new vs.\ CCM2025]
This paper contributes:
(a) the sharp two-sided
$\Delta_\varepsilon\asymp c^{-1/2}$
(Corollary~\ref{cor:sharp});
(b) the two-sided eigenvalue bound
(Theorem~\ref{thm:spectral});
(c) the quantitative rate $O(c^{-3/4})$
(Paper~VII, Thm.~C);
(d) the rigorous approximate-spectrum
argument with explicit CCM hypothesis.
\end{remark}

\begin{conjecture}[RH via $\Hlim$]
\label{conj:RH}
$\Hlim$ is essentially self-adjoint,
equivalently all non-trivial zeros of $\zeta(s)$
lie on $\mathrm{Re}(s)=1/2$.
\end{conjecture}

\section{Open Problems}
\label{sec:problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Sharp constant]
Determine
$\lim_{c\to\infty}c^{1/2}\Delta_\varepsilon
(c,\lfloor\kappa c\rfloor)$.
\end{problem}

\begin{problem}[Operator norm convergence rate]
Determine the decay of $\|\Hc-\Hlim\|_{L^2\to L^2}$;
the strong rate $O(c^{-3/4})$ is established.
\end{problem}

\begin{problem}[Essential self-adjointness]
Prove or disprove Conjecture~\ref{conj:RH}.
\end{problem}

\begin{problem}[Point vs.\ approximate spectrum]
Are the non-trivial zeros of $\zeta(s)$
eigenvalues of $\Hlim$, or only in
$\sigma_{\rm app}(\Hlim)$?
\end{problem}
