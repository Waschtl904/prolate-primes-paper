\begin{abstract}
We continue the operator-theoretic analysis of the limiting PSWF
concentration operator $\Hlim$, established in Papers~IX--X.
$\Hlim$ is the SOT-limit of $\Hc$ (Paper~X);
its eigenstructure is an orthonormal system
$\{\Phi_n^{(\infty)},\lambda_n^{(\infty)}\}$
constructed via a Cantor diagonal argument
(Lemma~\ref{lem:diagonal}).
Without assuming completeness, we establish:
\begin{enumerate}
\item \textbf{Spectral inclusion}
      (Thm.~\ref{thm:spectral-inclusion}, unconditional):
      $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)\subset[0,1]$.
\item \textbf{Semiclassical spectral transfer}
      (Thm.~\ref{thm:weyl}, under Hyp.~\ref{hyp:SOH}):
      for each $\lambda\in(0,1)$, there exists
      $n(\lambda)\ge 0$ with
      $\lambda_{n(\lambda)}^{(\infty)}
      = \lambda + O(c(\lambda)^{-1/4})$
      and
      $n(\lambda)
      = \lfloor\tfrac{2}{\pi}c(\lambda)\rfloor
        + O(\log c(\lambda))$,
      where $c(\lambda)$ is the finite-$c$
      transition bandwidth (Def.~\ref{def:clambda}).
\item \textbf{Density criterion}
      (Thm.~\ref{thm:density-criterion}):
      density of $\{\lambda_n^{(\infty)}\}$ implies
      $\sigma(\Hlim)=[0,1]$.
\end{enumerate}
Riesz-basis stability deferred to Paper~XII.
No unconditional connection to $\zeta(s)$-zeros.
\end{abstract}

\begin{theorem}[Main results]
\label{thm:main}
Let $\Hlim$ and
$\{\Phi_n^{(\infty)},\lambda_n^{(\infty)}\}$
be as in Lemma~\ref{lem:diagonal}.
\begin{enumerate}[\rm(A)]
\item \textbf{(Eigenvalue equation,
      Lem.~\ref{lem:eigenvalue-eq}.)}
      $\Hlim\,\Phi_n^{(\infty)}
      =\lambda_n^{(\infty)}\,\Phi_n^{(\infty)}$.
      Uses SOT (Paper~X), rate \eqref{XI1}
      (Paper~VIII), and $\|\Phi_n^{(\infty)}\|=1$
      (Paper~IX, used at exactly this step).
\item \textbf{(Spectral inclusion,
      Thm.~\ref{thm:spectral-inclusion}.)}
      $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)\subset[0,1]$.
      Unconditional.
\item \textbf{(Semiclassical transfer,
      Thm.~\ref{thm:weyl}.)}
      Under Hyp.~\ref{hyp:SOH}
      (no spectral collapse):
      for each $\lambda\in(0,1)$,
      $\exists\, n(\lambda)\ge 0$:
      \[
        \lambda_{n(\lambda)}^{(\infty)}
        = \lambda + O\!\left(c(\lambda)^{-1/4}\right),
        \quad
        n(\lambda)
        = \Bigl\lfloor\tfrac{2}{\pi}c(\lambda)\Bigr\rfloor
          + O(\log c(\lambda)).
      \]
\item \textbf{(Density criterion,
      Thm.~\ref{thm:density-criterion}.)}
      Density of $\{\lambda_n^{(\infty)}\}$ in $[0,1]$
      implies $\sigma(\Hlim)=[0,1]$.
\end{enumerate}
(A)--(B) unconditional;
(C) under Hyp.~\ref{hyp:SOH};
(D) under Conj.~\ref{conj:density}.
\end{theorem}

\subsection{Approximation convention}

\begin{definition}[Approximation with rate]
\label{conv:approx}
For $\lambda\in(0,1)$ and $n\in\mathbb{Z}_{\ge 0}$,
we write
$\lambda_n^{(\infty)} = \lambda + O(c^{-1/4})$
to mean: along the subsequence $c_k$ of
Lemma~\ref{lem:diagonal},
\[
  |\lambda_n^{(c_k)} - \lambda|
  \le C_\kappa c_k^{-1/4}
\]
for all $k$ with $n\le\kappa c_k$,
where $C_\kappa$ is the constant of \eqref{XI1}.
Informal ``$\approx$'' in remarks always refers
to this bound; theorem and lemma statements
use only the explicit $O(\cdot)$ form.
\end{definition}

\subsection{Common diagonal subsequence}

\begin{lemma}[Common diagonal subsequence]
\label{lem:diagonal}
There exists $c_k\nearrow\infty$ such that
for every $n\ge 0$ simultaneously:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\rightharpoonup\Phi_n^{(\infty)}$
      weakly;
\item $\lambda_n^{(c_k)}\to\lambda_n^{(\infty)}$;
\item $\mathcal{H}_{c_k}
      \xrightarrow{\mathrm{SOT}}\Hlim$.
\end{enumerate}
$\{\Phi_n^{(\infty)}\}$ is orthonormal.
\end{lemma}

\begin{proof}
\textbf{Step~1.} Banach--Alaoglu on the unit ball.
\textbf{Step~2 (Cantor diagonal).}
Extract $c_k^{(n)}$ inductively;
set $c_k:=c_k^{(k)}$.
(ii) from $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$ (Paper~VIII, Cor.~4.3).
\textbf{Step~3.} (iii) along any sequence
(Paper~X, Lem.~2.2).
\textbf{Step~4.} Orthonormality: off-diagonal inner
products vanish by weak convergence;
$\|\Phi_n^{(\infty)}\|=1$ from Paper~IX, Prop.~2.2.
\end{proof}

\begin{remark}[Scope]
All results along $c_k$.
Uniqueness up to phase: Prob.~\ref{prob:uniqueness}.
\end{remark}

\subsection{Inherited results}

\begin{enumerate}[(XI.1)]
\item\label{XI1}
  \textbf{Eigenvalue rate} (Paper~VIII, Cor.~4.3):
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}$, $n\le\kappa c$.
\item\label{XI2}
  \textbf{Orthonormal system}
  (Lem.~\ref{lem:diagonal} + Paper~IX, Prop.~2.2):
  $\Phi_n^{(c_k)}\rightharpoonup\Phi_n^{(\infty)}$;
  $\|\Phi_n^{(\infty)}\|=1$;
  $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}
  =\delta_{mn}$.
\item\label{XI3}
  \textbf{SOT, pointwise on $L^2$}
  (Paper~X, Lem.~2.2):
  $\mathcal{H}_{c_k}
  \xrightarrow{\mathrm{SOT}}\Hlim$;
  $\Hlim$ bounded, $\|\Hlim\|\le 1$;
  $\Dom(\Hlim)=L^2(\mathbb{R})$.
\item\label{XI4}
  \textbf{Non-cancellation} (Paper~VIII, Thm.~3.3):
  $|\Phihat_n^{(c)}|\ge c_1 c^{-1/2}$ on
  $E_c$, $|E_c|\ge c_*c^{-1/2}$.
\item\label{XI5}
  \textbf{Form rate} (Paper~VIII, Cor.~4.3):
  $0\le q_{\lim}-q_c\le C_\kappa c^{-1/4}\|{\cdot}\|^2$.
\item\label{XI6}
  \textbf{Slepian counting}
  \cite{Slepian1961,Osipov2013}:
  For each $\delta>0$:
  $n<(1{-}\delta)2c/\pi
  \Rightarrow\lambda_n^{(c)}\ge 1-e^{-\gamma_\delta c}$;
  $n>(1{+}\delta)2c/\pi
  \Rightarrow\lambda_n^{(c)}\le e^{-\gamma_\delta c}$.
  For each $\varepsilon>0$:
  $\#\{n:\lambda_n^{(c)}\in[\varepsilon,1{-}\varepsilon]\}
  =O(\log c)$,
  window centred at $n=2c/\pi$.
\end{enumerate}

\begin{lemma}[Non-vanishing]
\label{lem:nonvanish-limit}
$\|\Phi_n^{(\infty)}\|=1$.
\end{lemma}
\begin{proof}Orthonormality \eqref{XI2}.\end{proof}

\begin{lemma}[Strong convergence]
\label{lem:strong-conv}
Let $x_k\rightharpoonup x$ weakly in a Hilbert
space with $\|x_k\|=1=\|x\|$.
Then $x_k\to x$ strongly.
\end{lemma}
\begin{proof}
$\|x_k-x\|^2=2-2\,\mathrm{Re}\ip{x_k}{x}\to 0$.
\end{proof}

\begin{remark}[Paper~IX is essential here]
\label{rem:norm-essential}
$\|\Phi_n^{(c_k)}\|=1$: unit eigenvectors
(no reference needed).
$\|\Phi_n^{(\infty)}\|=1$: Paper~IX, Prop.~2.2.
These two norm-one inputs are independent;
the limit norm does \emph{not} follow from
weak convergence alone.
This is the \emph{only} point in the paper
where Paper~IX is essential:
all other arguments use only SOT (Paper~X)
and the eigenvalue rate (Paper~VIII).
\end{remark}

\begin{lemma}[Eigenvalue equation]
\label{lem:eigenvalue-eq}
$\Hlim\,\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\,\Phi_n^{(\infty)}$.
\end{lemma}
\begin{proof}
Since $\Hlim$ is bounded (Paper~X, \eqref{XI3}),
$\Dom(\Hlim)=L^2(\mathbb{R})$, so
$\Phi_n^{(\infty)}\in\Dom(\Hlim)$.
By Lem.~\ref{lem:strong-conv} and \eqref{XI2},
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$ strongly
(uses Paper~IX; Rem.~\ref{rem:norm-essential}).

We estimate
\begin{equation}\label{eq:ABC}
  \|\Hlim\,\Phi_n^{(\infty)}
  - \lambda_n^{(\infty)}\Phi_n^{(\infty)}\|
  \le A_k + B_k + C_k,
\end{equation}
where
\begin{align*}
  A_k &:= \|\Hlim\,\Phi_n^{(\infty)}
           - \mathcal{H}_{c_k}\Phi_n^{(\infty)}\|,\\
  B_k &:= \|\mathcal{H}_{c_k}
           (\Phi_n^{(\infty)}-\Phi_n^{(c_k)})\|,\\
  C_k &:= \|\lambda_n^{(c_k)}\Phi_n^{(c_k)}
           - \lambda_n^{(\infty)}\Phi_n^{(\infty)}\|.
\end{align*}
\textbf{$A_k\to 0$:}
SOT \eqref{XI3} applied pointwise to
$\Phi_n^{(\infty)}\in L^2$.

\textbf{$B_k\to 0$:}
$\|\mathcal{H}_{c_k}\|\le 1$ \eqref{XI3}, so
$B_k\le\|\Phi_n^{(\infty)}-\Phi_n^{(c_k)}\|\to 0$
by strong convergence.

\textbf{$C_k\to 0$:}
$\|\lambda_n^{(c_k)}\Phi_n^{(c_k)}
 - \lambda_n^{(\infty)}\Phi_n^{(\infty)}\|
\le |\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
 + |\lambda_n^{(\infty)}|
   \|\Phi_n^{(c_k)}-\Phi_n^{(\infty)}\|
\to 0$
by \eqref{XI1} and strong convergence.

Since the left-hand side of \eqref{eq:ABC}
is independent of $k$ and the right-hand side
tends to $0$, the equation holds.
\end{proof}

\begin{remark}[Why SOT suffices]
(i)~$\Dom(\Hlim)=L^2$ (bounded operator, Paper~X).
(ii)~Unit eigenvectors.
(iii)~$\|\Phi_n^{(\infty)}\|=1$ from Paper~IX
     (only use of Paper~IX).
(iv)~Quantitative rate $c^{-1/4}$.
No resolvent or compactness needed.
\end{remark}

\begin{theorem}[Spectral inclusion]
\label{thm:spectral-inclusion}
\begin{enumerate}[\rm(a)]
\item $\lambda_n^{(\infty)}\in\sigma_p(\Hlim)$
      for each $n\ge 0$.
\item $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)$.
\end{enumerate}
\end{theorem}
\begin{proof}
(a)~Lem.~\ref{lem:eigenvalue-eq};
$\Phi_n^{(\infty)}\ne 0$.
(b)~$\sigma(\Hlim)$ closed.
\end{proof}

\begin{remark}[Point, approximate, and full spectrum]
\label{rem:sigma-app}
\[
  \overline{\{\lambda_n^{(\infty)}\}}
  \subset\sigapp(\Hlim)
  \subset\sigma(\Hlim)
  \subset[0,1].
\]
Outer bounds unconditional.
$\sigapp(\Hlim)=\sigma(\Hlim)$:
conditional on completeness (Paper~IX).
Paper~XI's unconditional contribution:
$\sigma_p(\Hlim)$ non-empty and explicitly identified.
\end{remark}

\section{Semiclassical Spectral Transfer}
\label{sec:weyl}
%%в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ

\begin{definition}[Transition bandwidth]
\label{def:clambda}
For $\lambda\in(0,1)$, set
\begin{equation}\label{eq:clambda}
  c(\lambda)
  := \inf\bigl\{c > 0 :
     \lambda_{\lfloor 2c/\pi\rfloor}^{(c)}
     \le \lambda\bigr\}.
\end{equation}
The bandwidth at which the Slepian transition
eigenvalue first crosses level $\lambda$.
Defined entirely from finite-$c$ data,
independently of $\{\lambda_n^{(\infty)}\}$.
\end{definition}

\begin{remark}[Quasi-monotonicity of the transition map]
\label{rem:quasi-mono}
The map $c\mapsto\lambda_{\lfloor 2c/\pi\rfloor}^{(c)}$
is \emph{not} globally monotone:
the floor causes integer-jumps in the index.
However, $c\mapsto\lambda_n^{(c)}$ is smooth
for fixed $n$ (analytic perturbation theory),
and consecutive index eigenvalues satisfy
$|\lambda_n^{(c)}-\lambda_{n\pm 1}^{(c)}|
= O(c^{-1/4})$ in the transition window
(Slepian counting, \eqref{XI6}).
Hence, the jumps in the floor-indexed map
are at most $O(c^{-1/4})$, and the map is
\emph{piecewise continuous with jumps of
size $O(c^{-1/4})$}.
This suffices for the IVT argument of
Lemma~\ref{lem:sweeping}.
\end{remark}

\begin{lemma}[Transition sweeping]
\label{lem:sweeping}
For every compact $K\subset(0,1)$
and every $\varepsilon>0$,
there exists $C_{K,\varepsilon}<\infty$ such that
for every $\lambda\in K$,
there exists $c\in[c(\lambda),c(\lambda)+C_{K,\varepsilon}]$
and an index $n(c)=\lfloor 2c/\pi\rfloor+O(1)$
with
\begin{equation}\label{eq:sweeping}
  \Bigl|\lambda_{n(c)}^{(c)}
  - \lambda\Bigr| < \varepsilon.
\end{equation}
In particular, the Slepian transition eigenvalues
cover any compact subset of $(0,1)$ as $c\to\infty$.
\end{lemma}
\begin{proof}
By \eqref{XI6}, for each $c$ there exist
$O(\log c)$ indices near $2c/\pi$ with
$\lambda_n^{(c)}\in[\varepsilon_0,1-\varepsilon_0]$.
Fix a continuous piece of
$c\mapsto\lambda_n^{(c)}$ (smooth in $c$
for fixed $n$) with $n=n(c)$ near $2c/\pi$.
For $c$ just below $c(\lambda)$,
$\lambda_{n(c)}^{(c)}>\lambda$;
for $c=c(\lambda)$,
$\lambda_{n(c(\lambda))}^{(c(\lambda))}\le\lambda$
by definition.
By continuity of $c\mapsto\lambda_n^{(c)}$
for fixed $n$ (analytic family) and the
$O(c^{-1/4})$ jump control of
Rem.~\ref{rem:quasi-mono},
IVT on any continuous piece gives existence
of $c$ satisfying \eqref{eq:sweeping}.
Uniformity on compact $K$ follows from
compactness and the logistic profile
\cite{Widom1964,Osipov2013}.
\end{proof}

\begin{remark}[Why IVT alone is insufficient]
\label{rem:IVT-insufficient}
For fixed $c$, IVT gives attainment of every
$\mu\in(0,1)$ by \emph{some} $\lambda_n^{(c)}$,
but not by a transition-window index
$n\approx 2c/\pi$ specifically.
Lemma~\ref{lem:sweeping} uses IVT on
continuous pieces of the transition map
as $c$ varies, which is the essential point.
\end{remark}

\begin{lemma}[Existence and monotonicity of $c(\lambda)$]
\label{lem:clambda-exists}
For every $\lambda\in(0,1)$:
$c(\lambda)\in(0,\infty)$ and
$c(\lambda)$ is non-decreasing in $\lambda$.
\end{lemma}
\begin{proof}
Finiteness: Lem.~\ref{lem:sweeping}.
Positivity: for small $c$ the transition
eigenvalue exceeds $\lambda$
(\eqref{XI6}, $c\to 0$).
Monotonicity: inclusion of sets in \eqref{eq:clambda}.
\end{proof}

\begin{definition}[Counting function]
\label{def:counting}
$N(\lambda):=
\#\{n\ge 0:\lambda_n^{(\infty)}\le\lambda\}$
(non-decreasing, right-continuous;
finite for $\lambda\in[0,1)$ under SOH;
finiteness not guaranteed without SOH).
\end{definition}

\begin{hypothesis}[Spectral Ordering (SOH)]
\label{hyp:SOH}
$m\ne n\implies\lambda_m^{(\infty)}
\ne\lambda_n^{(\infty)}$.
\end{hypothesis}

\begin{remark}[SOH implies absence of spectral collapse]
\label{rem:SOH-essential}
SOH (injectivity of the spectral labeling)
\emph{implies} absence of spectral collapse:
no $\lambda\in(0,1)$ can be the limit of
infinitely many distinct $\lambda_n^{(\infty)}$,
for if it were, injectivity would fail at every
accumulation point.
The converse --- absence of spectral collapse
implying SOH --- may require additional
compactness or uniform gap assumptions
(e.g.\ \eqref{eq:gap-lower})
and is not claimed here.

Without SOH, the limiting spectrum could undergo
\emph{spectral collapse}: infinitely many
eigenvalues accumulating at a single point,
causing $N(\lambda)$ to be infinite,
the transfer index $n(\lambda)$ to be ambiguous
beyond $O(1)$, and Theorem~\ref{thm:weyl}
to break entirely.
SOH is therefore a \emph{structural} condition
asserting that the limiting spectral measure
carries no atoms of infinite multiplicity.
It would follow from the uniform gap bound
\begin{equation}\label{eq:gap-lower}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge C_\kappa c^{-3/4},\quad n\le\kappa c,
\end{equation}
combined with \eqref{XI1}
(Open Problem~\ref{prob:SOH}).
\end{remark}

\begin{remark}[Gap control]
\label{rem:SOH-status}
Estimate \eqref{eq:gap-lower} is the lower bound
of Prob.~\ref{prob:rate-sharp};
SOH is downstream of gap sharpness.
\end{remark}

\begin{theorem}[Semiclassical spectral transfer]
\label{thm:weyl}
Under Hypothesis~\ref{hyp:SOH},
with $c(\lambda)$ from Def.~\ref{def:clambda}:
\begin{enumerate}[\rm(a)]
\item \textbf{(Transfer law.)}
      For each $\lambda\in(0,1)$,
      there exists $n(\lambda)\ge 0$ such that
      \begin{equation}\label{eq:transfer-exist}
        \lambda_{n(\lambda)}^{(\infty)}
        = \lambda + O\!\left(c(\lambda)^{-1/4}\right)
      \end{equation}
      and
      \begin{equation}\label{eq:transfer-index}
        n(\lambda)
        = \Bigl\lfloor\tfrac{2}{\pi}c(\lambda)\Bigr\rfloor
          + O(\log c(\lambda)).
      \end{equation}
      This is a form of \emph{inverse spectral
      localization}: the Slepian bandwidth index
      $\lfloor 2c(\lambda)/\pi\rfloor$ recovers
      (up to $O(\log c)$ ambiguity) the correct
      label for the limiting eigenvalue closest
      to $\lambda$.
      It is \emph{not} a classical Weyl counting law.
\item \textbf{(Transition window.)}
      For $\lambda\in[\varepsilon,1{-}\varepsilon]$:
      $N(\lambda)-N(\lambda-\varepsilon)
      =O(\log c(\lambda))$.
\item \textbf{(Spectral floor.)}
      $N(\lambda)=0$ for
      $\lambda<\lambda_0^{(\infty)}
      :=\lim_k\lambda_0^{(c_k)}$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(a).}
Apply Lem.~\ref{lem:sweeping} with
$K=\{\lambda\}$ and $\varepsilon_0:=c(\lambda)^{-1/4}$:
there exist $c_*\in[c(\lambda),c(\lambda)+C]$
and $n_0=\lfloor 2c_*/\pi\rfloor+O(1)$
with $|\lambda_{n_0}^{(c_*)}-\lambda|<\varepsilon_0$.
The transition window for $c_*$ has
$O(\log c_*)=O(\log c(\lambda))$ elements
(\eqref{XI6}).
By \eqref{XI1}:
$|\lambda_{n_0}^{(c_*)} - \lambda_{n_0}^{(\infty)}|
\le C_\kappa c_*^{-1/4}$.
Setting $n(\lambda):=n_0$ gives
\eqref{eq:transfer-exist}--\eqref{eq:transfer-index}.

\textbf{(b).} From \eqref{XI6} and \eqref{XI1}.
\textbf{(c).}
No $\lambda_n^{(\infty)}$ below
$\lambda_0^{(\infty)}=\lim_k\lambda_0^{(c_k)}$.
\end{proof}

\begin{remark}[Why (a) is existential]
\label{rem:weyl-existential}
We exhibit one index near $2c(\lambda)/\pi$
whose limit eigenvalue is close to $\lambda$.
The global counting law
$N(\lambda)\sim\frac{2}{\pi}c(\lambda)$
would additionally require:
uniform continuity of $c(\lambda)$ in $\lambda$
and absence of $O(\log c)$-clustering.
Neither is available; see Prob.~\ref{prob:SOH}.
\end{remark}

\begin{remark}[Non-circularity]
$c(\lambda)$ from finite-$c$ data;
$N(\lambda)$ from the limit.
Non-trivial: two apriori unrelated scales
coincide with explicit $O(\log c)$ error.
\end{remark}

\label{sec:density}
%%в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ

\begin{theorem}[Density criterion]
\label{thm:density-criterion}
If $\{\lambda_n^{(\infty)}\}$ is dense in $[0,1]$,
then $\sigma(\Hlim)=[0,1]$.
\end{theorem}
\begin{proof}
Each $\lambda_n^{(\infty)}\in\sigma(\Hlim)$
(Thm.~\ref{thm:spectral-inclusion}(a));
$\sigma(\Hlim)$ closed; density gives
$[0,1]\subset\sigma(\Hlim)$.
Reverse: Paper~X.
\end{proof}

\begin{corollary}
Under density, $\sigma(\Hlim)$ has no isolated
points in $(0,1)$.
\end{corollary}

\section{The Density Question}
\label{sec:density-question}
%%в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ

\begin{conjecture}[Density]
\label{conj:density}
$\{\lambda_n^{(\infty)}\}$ is dense in $[0,1]$.
\end{conjecture}

\subsection{Evidence and barrier}

By \eqref{XI6}, for each $c$ there are
$O(\log c)$ eigenvalues in $[\varepsilon,1{-}\varepsilon]$.
By \eqref{XI1}, each lies within
$C_\kappa c^{-1/4}$ of $\lambda_n^{(\infty)}$.

\begin{remark}[Two-parameter reformulation]
\label{rem:density-gap}
Density is equivalent to:
\begin{quote}
  For every open $I\subset(0,1)$,
  $\exists k_j\to\infty$, $n_j\ge 0$:
  $\lambda_{n_j}^{(c_{k_j})}\in I$ for all $j$.
\end{quote}
IVT gives attainment for each fixed $c$;
the difficulty is
\emph{uniform distribution} of the
$O(\log c)$ transition window
across $(0,1)$ as $c\to\infty$:
the open barrier for this conjecture.
\end{remark}

\begin{remark}[Numerical evidence and limits]
\label{rem:density-numerical}
Numerical tables
\cite{Slepian1961,Osipov2013,LandauPollak1962}
show transition eigenvalues filling $[0,1]$
densely for finite $c$, and the window
appears to sweep $(0,1)$ uniformly as $c$ grows.
However, no uniform equidistribution result
is currently available from the analytic
literature; numerical evidence suggests
sweeping but does not constitute a proof.
\end{remark}

\begin{proposition}[Non-emptiness near $1$]
$\{\lambda_n^{(\infty)}\}\cap[1{-}\delta,1)
\ne\emptyset$ for all $\delta\in(0,1)$.
\end{proposition}
\begin{proof}\eqref{XI6} + \eqref{XI1}.\end{proof}

\section{Open Problems}
\label{sec:open-problems}
%%в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ

\begin{problem}[Density]
\label{prob:density}
Prove Conj.~\ref{conj:density};
equivalent to two-parameter coverage
(Rem.~\ref{rem:density-gap}).
\end{problem}

\begin{problem}[Spectral Ordering / No collapse]
\label{prob:SOH}
Prove Hyp.~\ref{hyp:SOH}
(SOH, which implies absence of spectral collapse);
follows from \eqref{eq:gap-lower}.
\end{problem}

\begin{problem}[Uniqueness]
\label{prob:uniqueness}
$\Phi_n^{(\infty)}$ independent of
diagonal subsequence, up to phase.
\end{problem}

\begin{problem}[Form rate sharpness]
\label{prob:rate-sharp}
$(q_{\lim}-q_c)/\|{\cdot}\|^2\asymp c^{-1/4}$?
Lower bound requires \eqref{eq:gap-lower}.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\{\Phi_n^{(\infty)}\}$ complete in $L^2$?
$\Leftrightarrow\sigapp(\Hlim)=\sigma(\Hlim)$.
\end{problem}

\begin{problem}[Riesz-basis stability]
\label{prob:riesz}
$\{\Phi_n^{(\infty)}\}$ Riesz basis?
Implies completeness and
$\HSOT=\overline{\Hspectral}$.
Target: Paper~XII.
\end{problem}

\begin{remark}[Hierarchy]
\ref{prob:density} + \ref{prob:completeness}:
both needed for Hilbert--P\'olya.
\ref{prob:riesz}: resolves completeness + Bridge.
\ref{prob:SOH} $\Leftarrow$ \ref{prob:rate-sharp}.
\ref{prob:uniqueness}: removes subsequential
qualifiers.
\end{remark}

\section{Outlook}
\label{sec:outlook}
%%в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ

Papers~I--XI reduce the Hilbert--P\'olya program
to four independent structural conditions;
no RH implication is claimed without all
being established:
\begin{enumerate}
\item \textbf{Density} (\ref{prob:density}).
\item \textbf{Completeness} (\ref{prob:completeness}).
\item \textbf{No spectral collapse}
      (Hyp.~\ref{hyp:SOH} + Rem.~\ref{rem:SOH-essential}).
\item \textbf{Uniqueness} (\ref{prob:uniqueness}).
\end{enumerate}
Under (1)--(4) plus CCM (\cite{CCM2025}),
RH would follow; this chain is not yet closed.
Problem~\ref{prob:riesz} would resolve (2)
and the Bridge theorem simultaneously;
target of Paper~XII.

Precise state of the programme as of Paper~XI.
