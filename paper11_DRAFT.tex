\begin{abstract}
We continue the operator-theoretic analysis of the limiting PSWF
concentration operator $\Hlim$, established in Papers~IX--X.
$\Hlim$ is the SOT-limit of $\Hc$ (Paper~X);
its eigenstructure is an orthonormal system
$\{\Phi_n^{(\infty)},\lambda_n^{(\infty)}\}$
constructed via a Cantor diagonal argument
(Lemma~\ref{lem:diagonal}).
Without assuming completeness, we establish:
\begin{enumerate}
\item \textbf{Spectral inclusion}
      (Thm.~\ref{thm:spectral-inclusion}, unconditional):
      $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)\subset[0,1]$.
\item \textbf{Semiclassical spectral transfer}
      (Thm.~\ref{thm:weyl},
      under Hyp.~\ref{hyp:NAcc}):
      for each $\lambda\in(0,1)$,
      there exists $n(\lambda)\ge 0$ with
      $\lambda_{n(\lambda)}^{(\infty)}
      = \lambda + O(c(\lambda)^{-1/4})$
      and
      $n(\lambda)
      = \lfloor\tfrac{2}{\pi}c(\lambda)\rfloor
        + O(\log c(\lambda))$,
      where $c(\lambda)$ is the
      window-robust transition bandwidth
      (Def.~\ref{def:clambda}).
\item \textbf{Density criterion}
      (Thm.~\ref{thm:density-criterion}):
      density of $\{\lambda_n^{(\infty)}\}$
      implies $\sigma(\Hlim)=[0,1]$.
\end{enumerate}
Riesz-basis stability deferred to Paper~XII.
No unconditional connection to $\zeta(s)$-zeros.
\end{abstract}

\section{Main Results}
\label{sec:main}

\begin{theorem}[Main results]
\label{thm:main}
Let $\Hlim$ and
$\{\Phi_n^{(\infty)},\lambda_n^{(\infty)}\}$
be as in Lemma~\ref{lem:diagonal}.
\begin{enumerate}[\rm(A)]
\item \textbf{(Eigenvalue equation,
      Lem.~\ref{lem:eigenvalue-eq}.)}
      $\Hlim\,\Phi_n^{(\infty)}
      =\lambda_n^{(\infty)}\,\Phi_n^{(\infty)}$.
      Uses SOT (Paper~X), rate \eqref{XI1}
      (Paper~VIII), and $\|\Phi_n^{(\infty)}\|=1$
      (Paper~IX, used at exactly this step).
\item \textbf{(Spectral inclusion,
      Thm.~\ref{thm:spectral-inclusion}.)}
      $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)\subset[0,1]$.
      Unconditional.
\item \textbf{(Semiclassical transfer,
      Thm.~\ref{thm:weyl}.)}
      Under Hyp.~\ref{hyp:NAcc}:
      for each $\lambda\in(0,1)$,
      $\exists\, n(\lambda)\ge 0$:
      \[
        \lambda_{n(\lambda)}^{(\infty)}
        = \lambda + O\!\left(c(\lambda)^{-1/4}\right),
        \quad
        n(\lambda)
        = \Bigl\lfloor\tfrac{2}{\pi}c(\lambda)\Bigr\rfloor
          + O(\log c(\lambda)).
      \]
\item \textbf{(Density criterion,
      Thm.~\ref{thm:density-criterion}.)}
      Density of $\{\lambda_n^{(\infty)}\}$ in $[0,1]$
      implies $\sigma(\Hlim)=[0,1]$.
\end{enumerate}
(A)--(B) unconditional;
(C) under Hyp.~\ref{hyp:NAcc};
(D) under Conj.~\ref{conj:density}.
\end{theorem}

\subsection{Approximation convention}

\begin{definition}[Approximation with rate]
\label{conv:approx}
For $\lambda\in(0,1)$ and $n\in\mathbb{Z}_{\ge 0}$,
we write
$\lambda_n^{(\infty)} = \lambda + O(c^{-1/4})$
to mean: along the subsequence $c_k$ of
Lemma~\ref{lem:diagonal},
\[
  |\lambda_n^{(c_k)} - \lambda|
  \le C_\kappa c_k^{-1/4}
\]
for all $k$ with $n\le\kappa c_k$,
where $C_\kappa$ is the constant of \eqref{XI1}.
Informal ``$\approx$'' in remarks always refers
to this bound;
statements use only the explicit $O(\cdot)$ form.
\end{definition}

\subsection{Transition window}

\begin{definition}[Transition window]
\label{def:window}
For $c>0$ and a fixed constant $C_0 > 0$
(to be chosen from the Slepian counting
estimates \eqref{XI6}), define
\[
  W_c := \bigl\{n\in\mathbb{Z}_{\ge 0} :
  |n - 2c/\pi| \le C_0\log c\bigr\}.
\]
By \eqref{XI6}, $|W_c| = O(\log c)$ and
every $n\in W_c$ satisfies
$\lambda_n^{(c)}\in[\varepsilon_0,1-\varepsilon_0]$
for some $\varepsilon_0>0$ depending on $C_0$.
\end{definition}

\subsection{Common diagonal subsequence}

\begin{lemma}[Common diagonal subsequence]
\label{lem:diagonal}
There exists $c_k\nearrow\infty$ such that
for every $n\ge 0$ simultaneously:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\rightharpoonup\Phi_n^{(\infty)}$
      weakly;
\item $\lambda_n^{(c_k)}\to\lambda_n^{(\infty)}$;
\item $\mathcal{H}_{c_k}
      \xrightarrow{\mathrm{SOT}}\Hlim$.
\end{enumerate}
$\{\Phi_n^{(\infty)}\}$ is orthonormal.
\end{lemma}

\begin{proof}
\textbf{Step~1.} Banach--Alaoglu on the unit ball.
\textbf{Step~2 (Cantor diagonal).}
Extract $c_k^{(n)}$ inductively;
set $c_k:=c_k^{(k)}$.
(ii) from $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$ (Paper~VIII, Cor.~4.3).
\textbf{Step~3.} (iii) along any sequence
(Paper~X, Lem.~2.2).
\textbf{Step~4.} Orthonormality: off-diagonal
inner products vanish by weak convergence;
$\|\Phi_n^{(\infty)}\|=1$ from Paper~IX,
Prop.~2.2.
\end{proof}

\begin{remark}[Scope]
All results along $c_k$.
Uniqueness up to phase: Prob.~\ref{prob:uniqueness}.
\end{remark}

\subsection{Inherited results}

\item\label{XI2}

\item\label{XI3}
  \textbf{SOT, pointwise on $L^2$}
  (Paper~X, Lem.~2.2):
  $\mathcal{H}_{c_k}
  \xrightarrow{\mathrm{SOT}}\Hlim$;
  $\Hlim$ bounded, $\|\Hlim\|\le 1$;
  $\Dom(\Hlim)=L^2(\mathbb{R})$.
\item\label{XI4}
  \textbf{Non-cancellation} (Paper~VIII, Thm.~3.3):

\item\label{XI5}

  \cite{Slepian1961,Osipov2013}:
  For each $\delta>0$:

  $n>(1{+}\delta)2c/\pi
  \Rightarrow\lambda_n^{(c)}\le e^{-\gamma_\delta c}$.
  For each $\varepsilon>0$:
  $\#\{n:\lambda_n^{(c)}
  \in[\varepsilon,1{-}\varepsilon]\}
  =O(\log c)$,
  window centred at $n=2c/\pi$.
\end{enumerate}

\section{Spectral Inclusion}
\label{sec:spectral-inclusion}
%%══════════════════════════════════════════════════════════════════════

\begin{lemma}[Non-vanishing]
\label{lem:nonvanish-limit}
$\|\Phi_n^{(\infty)}\|=1$.
\end{lemma}
\begin{proof}Orthonormality \eqref{XI2}.\end{proof}

\begin{lemma}[Strong convergence]
\label{lem:strong-conv}
Let $x_k\rightharpoonup x$ weakly in a Hilbert
space with $\|x_k\|=1=\|x\|$.
Then $x_k\to x$ strongly.
\end{lemma}
\begin{proof}
$\|x_k-x\|^2=2-2\,\mathrm{Re}\ip{x_k}{x}\to 0$.
\end{proof}

\begin{remark}[Paper~IX is essential here]
\label{rem:norm-essential}
$\|\Phi_n^{(c_k)}\|=1$: unit eigenvectors.
$\|\Phi_n^{(\infty)}\|=1$: Paper~IX, Prop.~2.2.
These are independent inputs;
the limit norm does \emph{not} follow from
weak convergence alone.
This is the \emph{only} point in the paper
where Paper~IX is essential;
all other arguments use only Paper~X and
Paper~VIII.
\end{remark}

\begin{lemma}[Eigenvalue equation]
\label{lem:eigenvalue-eq}
$\Hlim\,\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\,\Phi_n^{(\infty)}$.
\end{lemma}
\begin{proof}
Since $\Hlim$ is bounded (Paper~X, \eqref{XI3}),
$\Dom(\Hlim)=L^2(\mathbb{R})$, so
$\Phi_n^{(\infty)}\in\Dom(\Hlim)$.
By Lem.~\ref{lem:strong-conv} and \eqref{XI2},
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$ strongly
(uses Paper~IX; Rem.~\ref{rem:norm-essential}).

\begin{remark*}[Non-commutation of limits]
We deliberately \emph{avoid} writing
$\lim_k\mathcal{H}_{c_k}\Phi_n^{(c_k)}
= \Hlim\lim_k\Phi_n^{(c_k)}$:
such interchange is \emph{not} justified by
SOT alone, since SOT is pointwise on
\emph{fixed} vectors, not on sequences.
The three-term decomposition below bypasses
this non-commutation entirely.
\end{remark*}

We estimate
\begin{equation}\label{eq:ABC}
  \|\Hlim\,\Phi_n^{(\infty)}
  - \lambda_n^{(\infty)}\Phi_n^{(\infty)}\|
  \le A_k + B_k + C_k,
\end{equation}
where
\begin{align*}
  A_k &:= \|\Hlim\,\Phi_n^{(\infty)}
           - \mathcal{H}_{c_k}\Phi_n^{(\infty)}\|,\\
  B_k &:= \|\mathcal{H}_{c_k}
           (\Phi_n^{(\infty)}-\Phi_n^{(c_k)})\|,\\
  C_k &:= \|\lambda_n^{(c_k)}\Phi_n^{(c_k)}
           - \lambda_n^{(\infty)}\Phi_n^{(\infty)}\|.
\end{align*}
\textbf{$A_k\to 0$:}
SOT \eqref{XI3} applied pointwise to the
fixed vector $\Phi_n^{(\infty)}\in L^2$.

\textbf{$B_k\to 0$:}
$\|\mathcal{H}_{c_k}\|\le 1$ \eqref{XI3}, so
$B_k\le\|\Phi_n^{(\infty)}-\Phi_n^{(c_k)}\|\to 0$.

\textbf{$C_k\to 0$:}
$\|\lambda_n^{(c_k)}\Phi_n^{(c_k)}
 - \lambda_n^{(\infty)}\Phi_n^{(\infty)}\|
\le |\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
 + |\lambda_n^{(\infty)}|
   \|\Phi_n^{(c_k)}-\Phi_n^{(\infty)}\|
\to 0$
by \eqref{XI1} and strong convergence.

Since the left-hand side of \eqref{eq:ABC}
is independent of $k$ and the right-hand
side tends to $0$, the equation holds.
\end{proof}

\begin{theorem}[Spectral inclusion]
\label{thm:spectral-inclusion}
\begin{enumerate}[\rm(a)]
\item $\lambda_n^{(\infty)}\in\sigma_p(\Hlim)$
      for each $n\ge 0$.
\item $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)$.
\end{enumerate}
\end{theorem}
\begin{proof}
(a) Lem.~\ref{lem:eigenvalue-eq};
$\Phi_n^{(\infty)}\ne 0$.
(b) $\sigma(\Hlim)$ closed.
\end{proof}

\begin{remark}[Spectral hierarchy]
\label{rem:sigma-app}
\[
  \overline{\{\lambda_n^{(\infty)}\}}
  \subset\sigapp(\Hlim)
  \subset\sigma(\Hlim)
  \subset[0,1].
\]
Outer bounds unconditional (Paper~X).
$\sigapp=\sigma$: conditional on completeness
(Paper~IX).
Unconditional contribution of Paper~XI:
$\sigma_p(\Hlim)$ non-empty.
\end{remark}

\section{Semiclassical Spectral Transfer}
\label{sec:weyl}

\begin{definition}[Transition bandwidth, window-robust]
\label{def:clambda}
Fix a constant $C_0>0$ from \eqref{XI6}.
For $\lambda\in(0,1)$, define
\begin{equation}\label{eq:clambda}
  c(\lambda)
  := \inf\bigl\{c > 0 :
     \exists\, n\in W_c\text{ with }
     \lambda_n^{(c)}\le\lambda\bigr\},
\end{equation}
where $W_c = \{n : |n - 2c/\pi|\le C_0\log c\}$
(Def.~\ref{def:window}).
\end{definition}

\begin{remark}[Window-robustness]
\label{rem:clambda-robust}
Definition~\ref{def:clambda} replaces
the floor-indexed definition
$\inf\{c : \lambda_{\lfloor 2c/\pi\rfloor}^{(c)}
\le\lambda\}$
used in previous drafts.
The window $W_c$ tracks the full transition zone;
no single index is privileged.
This makes $c(\lambda)$ robust against
index jumps at integer values of $2c/\pi$.
\end{remark}

\begin{lemma}[Index stability near $c(\lambda)$]
\label{lem:index-stability}
Let $\lambda\in(0,1)$ and
$n_0\in W_{c(\lambda)}$.
There exists $\delta=\delta(\lambda)>0$ such that
for all $c\in[c(\lambda)-\delta, c(\lambda)+\delta]$:
\begin{enumerate}[\rm(i)]
\item $n_0\in W_c$
      (i.e.\ $|n_0 - 2c/\pi|\le C_0\log c$);
\item the map $c\mapsto\lambda_{n_0}^{(c)}$
      is smooth (analytic in $c$,
      \cite{Kato1966}).
\end{enumerate}
\end{lemma}
\begin{proof}
(i)~$n_0\in W_{c(\lambda)}$ means
$|n_0 - 2c(\lambda)/\pi|\le C_0\log c(\lambda)$.
For $c$ in a $\delta$-neighbourhood,
$|n_0 - 2c/\pi|
\le |n_0 - 2c(\lambda)/\pi|
 + 2\delta/\pi
\le C_0\log c(\lambda) + 2\delta/\pi$.
Choosing $\delta$ small enough that
$C_0\log c(\lambda)+2\delta/\pi
\le C_0\log(c(\lambda)+\delta)$
gives $n_0\in W_c$.
(ii)~For fixed $n_0$, the PSWF family
$c\mapsto\lambda_{n_0}^{(c)}$ is analytic
(\cite{Kato1966}, analytic perturbation theory).
\end{proof}

\begin{lemma}[Transition sweeping]
\label{lem:sweeping}
For every compact $K\subset(0,1)$
and every $\varepsilon>0$,
there exists $C_{K,\varepsilon}<\infty$ such that
for every $\lambda\in K$:
there exist $c'\in[c(\lambda)-C_{K,\varepsilon},
c(\lambda)]$ and an index
$n'\in W_{c(\lambda)}$ with
\begin{equation}\label{eq:sweeping}
  \bigl|\lambda_{n'}^{(c')} - \lambda\bigr|
  < \varepsilon.
\end{equation}
\end{lemma}
\begin{proof}
Let $n_0\in W_{c(\lambda)}$ be any index
satisfying $\lambda_{n_0}^{(c(\lambda))}\le\lambda$
(exists by definition of $c(\lambda)$).
By Lem.~\ref{lem:index-stability}(i),
$n_0\in W_c$ for all $c$ near $c(\lambda)$.

For fixed $n_0$, the map $c\mapsto\lambda_{n_0}^{(c)}$
is strictly increasing in $c$
(\cite[Ch.~3, Prop.~3.2]{Osipov2013}):
increasing the bandwidth $c$ enlarges the
time-frequency concentration region, hence
raises all Slepian eigenvalues.

\begin{remark*}[Monotonicity]
For fixed $c$, the sequence
$n\mapsto\lambda_n^{(c)}$ is strictly decreasing
(higher index $\Rightarrow$ lower concentration).
For fixed $n$, the map
$c\mapsto\lambda_n^{(c)}$ is strictly increasing
(larger bandwidth $\Rightarrow$ higher concentration).
Both facts are classical; see \cite[Ch.~3]{Osipov2013}.
\end{remark*}

For $c$ slightly less than $c(\lambda)$:
by definition of the infimum in \eqref{eq:clambda},
no $n\in W_c$ satisfies $\lambda_n^{(c)}\le\lambda$,
so in particular $\lambda_{n_0}^{(c)}>\lambda$.
At $c=c(\lambda)$:
$\lambda_{n_0}^{(c(\lambda))}\le\lambda$.

Since $c\mapsto\lambda_{n_0}^{(c)}$ is
strictly increasing (analytic by
Lem.~\ref{lem:index-stability}(ii)),
it crosses $\lambda$ at a \emph{unique}
$c'\in[c(\lambda)-\delta,c(\lambda)]$:
\[
  \lambda_{n_0}^{(c')} = \lambda.
\]
Setting $n':=n_0$ establishes \eqref{eq:sweeping}
with $\varepsilon=0$ (hence all $\varepsilon>0$),
and the crossing is unique by strict monotonicity.
Uniformity on $K$ follows from compactness
and continuous dependence of $c(\lambda)$
on $\lambda$ (\cite{Widom1964,Osipov2013}).
\end{proof}

\begin{remark}[Role of window-robustness]
Lem.~\ref{lem:sweeping} is valid because:
(a) $n_0\in W_{c(\lambda)}$ is chosen from
the \emph{window}, not as a fixed floor index;
(b) Lem.~\ref{lem:index-stability} guarantees
$n_0$ remains in the window for nearby $c$,
justifying the monotone crossing on the fixed-index map;
(c) the infimum definition ensures sign change
without requiring monotonicity of the floor map.
\end{remark}

\begin{lemma}[Existence and monotonicity of $c(\lambda)$]
\label{lem:clambda-exists}
For every $\lambda\in(0,1)$:
$c(\lambda)\in(0,\infty)$
and $c(\lambda)$ is non-decreasing.
\end{lemma}
\begin{proof}
Finiteness: by Lem.~\ref{lem:sweeping}
the transition window eventually covers $\lambda$.
Positivity: for small $c$, all $\lambda_n^{(c)}>\lambda$
for $n\in W_c$ (\eqref{XI6}).
Monotonicity: if $\lambda<\lambda'$ then
$\{c:\exists n\in W_c,\,\lambda_n^{(c)}\le\lambda\}
\subset\{c:\exists n\in W_c,\,\lambda_n^{(c)}\le\lambda'\}$.
\end{proof}

\begin{hypothesis}[No Accumulation (NAcc)]
\label{hyp:NAcc}
For all $\lambda\in(0,1)$:
\[
  N(\lambda)
  := \#\bigl\{n\ge 0:
     \lambda_n^{(\infty)}\le\lambda\bigr\}
  < \infty.
\]
\end{hypothesis}

\begin{remark}[NAcc: meaning and status]
\label{rem:NAcc}
Hyp.~\ref{hyp:NAcc} asserts that the limiting
eigenvalues have no accumulation point in $(0,1)$:
if $N(\lambda)<\infty$ for all $\lambda\in(0,1)$,
then $\{\lambda_n^{(\infty)}\}$ can only
accumulate at $0$ or $1$.
Note that
\begin{itemize}
\item NAcc implies SOH (injectivity)
      since if $\lambda_m^{(\infty)}
      =\lambda_n^{(\infty)}=\mu$ for $m\ne n$,
      then $N(\mu)\ge 2$;
      but NAcc alone does not contradict
      multiplicity $\ge 2$ at isolated points.
      The \emph{operative} condition for
      Thm.~\ref{thm:weyl} is finiteness of
      $N(\lambda)$, which is NAcc.
\item NAcc follows from the uniform gap bound
      \begin{equation}\label{eq:gap-lower}
        |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
        \ge C_\kappa c^{-3/4},
        \quad n\le\kappa c,
      \end{equation}
      combined with \eqref{XI1}:
      distinct limiting eigenvalues are then
      separated by $\Omega(c^{-3/4})$,
      preventing accumulation.
      Establishing \eqref{eq:gap-lower} is
      Open Problem~\ref{prob:NAcc}.
\end{itemize}
\end{remark}

\begin{remark}[NAcc: strength and status]
\label{rem:NAcc-strength}
Hypothesis~\ref{hyp:NAcc} is a \emph{strong
structural assumption}: it is equivalent to
a uniform spectral separation condition in the
transition regime (see \eqref{eq:gap-lower}).
No unconditional proof is currently available.
Establishing the gap lower bound
$|\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge C_\kappa c^{-3/4}$
is an independent hard problem
(Open Problem~\ref{prob:NAcc});
we do not claim it is accessible by current methods.
\end{remark}

\begin{theorem}[Semiclassical spectral transfer]
\label{thm:weyl}
Under Hypothesis~\ref{hyp:NAcc},
with $c(\lambda)$ from Def.~\ref{def:clambda}:
\begin{enumerate}[\rm(a)]
\item \textbf{(Transfer law.)}
      For each $\lambda\in(0,1)$,
      there exists $n(\lambda)\ge 0$ such that
      \begin{equation}\label{eq:transfer-exist}
        \lambda_{n(\lambda)}^{(\infty)}
        = \lambda + O\!\left(c(\lambda)^{-1/4}\right)
      \end{equation}
      and
      \begin{equation}\label{eq:transfer-index}
        n(\lambda)
        = \Bigl\lfloor\tfrac{2}{\pi}c(\lambda)\Bigr\rfloor
          + O(\log c(\lambda)).
      \end{equation}
      This is a form of \emph{inverse spectral
      localization}: the Slepian bandwidth index
      recovers the correct label for the
      closest limiting eigenvalue up to
      $O(\log c)$ ambiguity.
      It is \emph{not} a classical Weyl law,
      and provides localization but not a
      global counting law.
\item \textbf{(Transition window.)}
      For $\lambda\in[\varepsilon,1{-}\varepsilon]$:
      $N(\lambda)-N(\lambda-\varepsilon)
      =O(\log c(\lambda))$.
\item \textbf{(Spectral floor.)}
      $N(\lambda)=0$ for
      $\lambda<\lambda_0^{(\infty)}
      :=\lim_k\lambda_0^{(c_k)}$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(a).}
By Lem.~\ref{lem:sweeping}, there exist
$c'\in[c(\lambda)-C, c(\lambda)]$
and $n'\in W_{c(\lambda)}$ with
$\lambda_{n'}^{(c')}=\lambda$ exactly.
Since $n'\in W_{c(\lambda)}$,
$|n'-2c(\lambda)/\pi|\le C_0\log c(\lambda)$,
so \eqref{eq:transfer-index} holds.
By \eqref{XI1}:
$|\lambda_{n'}^{(c')} - \lambda_{n'}^{(\infty)}|
\le C_\kappa (c')^{-1/4}$,
giving \eqref{eq:transfer-exist} with
$n(\lambda):=n'$.

\textbf{(b).} From \eqref{XI6} and \eqref{XI1}.
\textbf{(c).}
No $\lambda_n^{(\infty)}$ below
$\lambda_0^{(\infty)}=\lim_k\lambda_0^{(c_k)}$.
\end{proof}

\begin{remark}[Why existential]
\label{rem:weyl-existential}
The transfer law exhibits one index in the
transition window whose limit eigenvalue is
close to $\lambda$.
A global counting law
$N(\lambda)\sim\frac{2}{\pi}c(\lambda)$
would additionally require uniform continuity
of $c(\lambda)$ and absence of $O(\log c)$-clustering.
Neither is available.
\end{remark}

\begin{remark}[Non-circularity]
$c(\lambda)$ from finite-$c$ data
(Def.~\ref{def:clambda});
$N(\lambda)$ from the limit.
Non-trivial: two apriori unrelated scales
coincide with explicit $O(\log c)$ error.
\end{remark}

\section{Density Criterion}
\label{sec:density}
%%══════════════════════════════════════════════════════════════════════

\begin{theorem}[Density criterion]
\label{thm:density-criterion}
If $\{\lambda_n^{(\infty)}\}$ is dense in $[0,1]$,
then $\sigma(\Hlim)=[0,1]$.
\end{theorem}
\begin{proof}
Each $\lambda_n^{(\infty)}\in\sigma(\Hlim)$
(Thm.~\ref{thm:spectral-inclusion}(a));
closure and density give
$[0,1]\subset\sigma(\Hlim)$.
Reverse inclusion: Paper~X.
\end{proof}

\begin{corollary}
Under density, $\sigma(\Hlim)$ has no
isolated points in $(0,1)$.
\end{corollary}

%%══════════════════════════════════════════════════════════════════════
\section{The Density Question}
\label{sec:density-question}
%%══════════════════════════════════════════════════════════════════════

\begin{conjecture}[Density]
\label{conj:density}
$\{\lambda_n^{(\infty)}\}$ is dense in $[0,1]$.
\end{conjecture}

\subsection{Evidence and barrier}

By \eqref{XI6}, for each $c$ the window
$W_c$ contains $O(\log c)$ eigenvalues in
$[\varepsilon,1-\varepsilon]$.
By \eqref{XI1}, each lies within
$C_\kappa c^{-1/4}$ of $\lambda_n^{(\infty)}$.

\begin{remark}[Two-parameter reformulation]
\label{rem:density-gap}
Density is equivalent to:
\begin{quote}
  For every open $I\subset(0,1)$,
  $\exists k_j\to\infty$, $n_j\ge 0$:
  $\lambda_{n_j}^{(c_{k_j})}\in I$ for all $j$.
\end{quote}
The difficulty is uniform distribution of
the $O(\log c)$ transition window across
$(0,1)$ as $c\to\infty$: the open barrier.
\end{remark}

\begin{remark}[Numerical evidence]
\label{rem:density-numerical}
Numerical tables \cite{Slepian1961,Osipov2013}
show transition eigenvalues filling $[0,1]$
densely as $c$ grows; no analytic equidistribution
result is currently available.
\end{remark}

\begin{proposition}[Non-emptiness near $1$]
$\{\lambda_n^{(\infty)}\}\cap[1{-}\delta,1)
\ne\emptyset$ for all $\delta\in(0,1)$.
\end{proposition}
\begin{proof}\eqref{XI6} + \eqref{XI1}.\end{proof}

\section{Open Problems}
\label{sec:open-problems}
%%══════════════════════════════════════════════════════════════════════

\begin{problem}[Density]
\label{prob:density}
Prove Conj.~\ref{conj:density};
equivalent to two-parameter window coverage
(Rem.~\ref{rem:density-gap}).
\end{problem}

\begin{problem}[No Accumulation / gap lower bound]
\label{prob:NAcc}
Prove Hyp.~\ref{hyp:NAcc};
follows from gap lower bound \eqref{eq:gap-lower}
(Open Problem~\ref{prob:rate-sharp}).
\end{problem}

\begin{problem}[Uniqueness]
\label{prob:uniqueness}
$\Phi_n^{(\infty)}$ independent of
diagonal subsequence, up to phase.
\end{problem}

\begin{problem}[Form rate sharpness]
\label{prob:rate-sharp}
$(q_{\lim}-q_c)/\|{\cdot}\|^2\asymp c^{-1/4}$?
Lower bound requires \eqref{eq:gap-lower}.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\{\Phi_n^{(\infty)}\}$ complete in $L^2$?
$\Leftrightarrow\sigapp(\Hlim)=\sigma(\Hlim)$.
\end{problem}

\begin{problem}[Riesz-basis stability]
\label{prob:riesz}
$\{\Phi_n^{(\infty)}\}$ Riesz basis?
Implies completeness and
$\HSOT=\overline{\Hspectral}$.
Target: Paper~XII.
\end{problem}

\begin{remark}[Hierarchy]
\ref{prob:density} + \ref{prob:completeness}:
both needed for Hilbert--P\'olya.
\ref{prob:riesz}: resolves completeness + Bridge.
\ref{prob:NAcc} $\Leftarrow$ \ref{prob:rate-sharp}.
\ref{prob:uniqueness}: removes subsequential
qualifiers.
\end{remark}

\section{Outlook}
\label{sec:outlook}
%%══════════════════════════════════════════════════════════════════════

Papers~I--XI reduce the Hilbert--P\'olya
program to four structural conditions;
no RH implication is claimed without all
being established:
\begin{enumerate}
\item \textbf{Density}
      (Prob.~\ref{prob:density}).
\item \textbf{Completeness}
      (Prob.~\ref{prob:completeness}).
\item \textbf{No accumulation}
      (Hyp.~\ref{hyp:NAcc},
      Rem.~\ref{rem:NAcc}).
\item \textbf{Uniqueness}
      (Prob.~\ref{prob:uniqueness}).
\end{enumerate}
Under (1)--(4) plus CCM (\cite{CCM2025}),
RH would follow; this chain is not yet closed.
Problem~\ref{prob:riesz} would resolve (2)
and the Bridge theorem;
target of Paper~XII.

\medskip\noindent
The trace formula established in Paper~II
provides a structural link between the spectral
density of $\Hlim$ and the distribution of prime
numbers via the Slepian eigenvalue counting function.
Once Conditions~(1) and~(2) are established,
this connection becomes fully operative and the
Hilbert--P\'olya programme closes conditionally
on CCM (\cite{CCM2025}).

\medskip\noindent
State of the programme as of Paper~XI.
