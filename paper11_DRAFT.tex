\begin{abstract}
We continue the operator-theoretic analysis of the limiting PSWF
concentration operator $\Hlim$, established in Papers~IX--X.
$\Hlim$ is the SOT-limit of $\Hc$ (Paper~X);
its eigenstructure is an orthonormal system
$\{\Phi_n^{(\infty)},\lambda_n^{(\infty)}\}$
constructed along a common diagonal subsequence
(Lemma~\ref{lem:diagonal}).
Without assuming completeness, we establish:
\begin{enumerate}
\item \textbf{Spectral inclusion}
      (Thm.~\ref{thm:spectral-inclusion}, unconditional):
      $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)\subset[0,1]$.
\item \textbf{Semiclassical spectral transfer}
      (Thm.~\ref{thm:weyl}, under Hyp.~\ref{hyp:SOH}):
      $N(\lambda)
      = \lfloor\frac{2}{\pi}c(\lambda)\rfloor
        + O(\log c(\lambda))$,
      where $c(\lambda)$ is the finite-$c$ transition
      bandwidth (Def.~\ref{def:clambda}).
\item \textbf{Density criterion}
      (Thm.~\ref{thm:density-criterion}):
      density of $\{\lambda_n^{(\infty)}\}$ implies
      $\sigma(\Hlim)=[0,1]$.
\end{enumerate}
Riesz-basis stability deferred to Paper~XII.
No unconditional connection to $\zeta(s)$-zeros.
\end{abstract}

\section{Main Results}
\label{sec:main}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Main results]
\label{thm:main}
Let $\Hlim$ and
$\{\Phi_n^{(\infty)},\lambda_n^{(\infty)}\}$
be as in Lemma~\ref{lem:diagonal}.
\begin{enumerate}[\rm(A)]
\item \textbf{(Eigenvalue equation,
      Lem.~\ref{lem:eigenvalue-eq}.)}
      $\Hlim\,\Phi_n^{(\infty)}
      =\lambda_n^{(\infty)}\,\Phi_n^{(\infty)}$.
      Uses only SOT, rate \eqref{XI1}, orthonormality;
      no completeness, no resolvent.
\item \textbf{(Spectral inclusion,
      Thm.~\ref{thm:spectral-inclusion}.)}
      $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)\subset[0,1]$.
      Unconditional.
\item \textbf{(Semiclassical transfer,
      Thm.~\ref{thm:weyl}.)}
      Under Hyp.~\ref{hyp:SOH}:
      $N(\lambda)
      = \lfloor\tfrac{2}{\pi}c(\lambda)\rfloor
        + O(\log c(\lambda))$,
      with $c(\lambda)$ from Def.~\ref{def:clambda}.
\item \textbf{(Density criterion,
      Thm.~\ref{thm:density-criterion}.)}
      Density of $\{\lambda_n^{(\infty)}\}$ in $[0,1]$
      implies $\sigma(\Hlim)=[0,1]$.
\end{enumerate}
(A)--(B) unconditional;
(C) under Hyp.~\ref{hyp:SOH};
(D) under Conj.~\ref{conj:density}.
\end{theorem}

\subsection{Approximation convention}

\begin{definition}[Approximation with rate]
\label{conv:approx}
For $\lambda\in(0,1)$ and $n\in\mathbb{Z}_{\ge 0}$,
we write
\[
  \lambda_n^{(\infty)} = \lambda + O(c^{-1/4})
\]
to mean: along the subsequence $c_k$ of
Lemma~\ref{lem:diagonal},
\[
  |\lambda_n^{(c_k)} - \lambda|
  \le C_\kappa c_k^{-1/4}
\]
for all $k$ with $n\le\kappa c_k$,
where $C_\kappa$ is the constant of \eqref{XI1}.
Informal ``$\approx$'' in remarks always refers
to this quantitative bound;
theorem and lemma statements use only the
precise $O(\cdot)$ form.
\end{definition}

\subsection{Common diagonal subsequence}

\begin{lemma}[Common diagonal subsequence]
\label{lem:diagonal}
There exists $c_k\nearrow\infty$ such that
for every $n\ge 0$ simultaneously:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\rightharpoonup\Phi_n^{(\infty)}$
      weakly;
\item $\lambda_n^{(c_k)}\to\lambda_n^{(\infty)}$;
\item $\mathcal{H}_{c_k}
      \xrightarrow{\mathrm{SOT}}\Hlim$.
\end{enumerate}
$\{\Phi_n^{(\infty)}\}$ is orthonormal.
\end{lemma}

\begin{proof}
\textbf{Step 1.} Banach--Alaoglu on the unit ball.
\textbf{Step 2 (Cantor diagonal).}
Extract $c_k^{(n)}$ inductively;
set $c_k:=c_k^{(k)}$.
(ii) from $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
\le C_\kappa c^{-1/4}$ (Paper~VIII, Cor.~4.3).
\textbf{Step 3.} (iii) along any sequence
(Paper~X, Lem.~2.2).
\textbf{Step 4.} Orthonormality: off-diagonal inner
products vanish by weak convergence;
$\|\Phi_n^{(\infty)}\|=1$ from Paper~IX, Prop.~2.2.
\end{proof}

\begin{remark}[Scope]
All results along $c_k$.
Uniqueness up to phase: Prob.~\ref{prob:uniqueness}.
\end{remark}

\subsection{Inherited results}

\begin{enumerate}[(XI.1)]
\item\label{XI1}
  \textbf{Eigenvalue rate} (Paper~VIII, Cor.~4.3):
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}$, $n\le\kappa c$.
\item\label{XI2}
  \textbf{Orthonormal system}
  (Lem.~\ref{lem:diagonal} + Paper~IX, Prop.~2.2):
  $\Phi_n^{(c_k)}\rightharpoonup\Phi_n^{(\infty)}$;
  $\|\Phi_n^{(\infty)}\|=1$;
  $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}
  =\delta_{mn}$.
\item\label{XI3}
  \textbf{SOT, pointwise on $L^2$}
  (Paper~X, Lem.~2.2):
  $\mathcal{H}_{c_k}
  \xrightarrow{\mathrm{SOT}}\Hlim$;
  $\|\mathcal{H}_{c_k}\|\le 1$.
\item\label{XI4}
  \textbf{Non-cancellation} (Paper~VIII, Thm.~3.3):
  $|\Phihat_n^{(c)}|\ge c_1 c^{-1/2}$ on
  $E_c$, $|E_c|\ge c_*c^{-1/2}$.
\item\label{XI5}
  \textbf{Form rate} (Paper~VIII, Cor.~4.3):
  $0\le q_{\lim}-q_c\le C_\kappa c^{-1/4}\|{\cdot}\|^2$.
\item\label{XI6}
  \textbf{Slepian counting}
  \cite{Slepian1961,Osipov2013}:
  For each $\delta>0$:
  $n<(1{-}\delta)2c/\pi
  \Rightarrow\lambda_n^{(c)}\ge 1-e^{-\gamma_\delta c}$;
  $n>(1{+}\delta)2c/\pi
  \Rightarrow\lambda_n^{(c)}\le e^{-\gamma_\delta c}$.
  For each $\varepsilon>0$:
  $\#\{n:\lambda_n^{(c)}\in[\varepsilon,1{-}\varepsilon]\}
  =O(\log c)$,
  transition centred at $n=2c/\pi$.
\end{enumerate}

\section{Spectral Inclusion}
\label{sec:spectral-inclusion}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Non-vanishing]
\label{lem:nonvanish-limit}
$\|\Phi_n^{(\infty)}\|=1$.
\end{lemma}
\begin{proof}Orthonormality \eqref{XI2}.\end{proof}

\begin{lemma}[Strong convergence]
\label{lem:strong-conv}
Let $x_k\rightharpoonup x$ weakly in a Hilbert
space with $\|x_k\|=1=\|x\|$.
Then $x_k\to x$ strongly.
\end{lemma}
\begin{proof}
$\|x_k-x\|^2
=2-2\,\mathrm{Re}\ip{x_k}{x}\to 0$.
\end{proof}

\begin{remark}[Independence of norm sources]
$\|\Phi_n^{(c_k)}\|=1$: unit eigenvectors.
$\|\Phi_n^{(\infty)}\|=1$: Paper~IX orthonormality.
Norm equality does not follow from weak convergence.
\end{remark}

\begin{lemma}[Eigenvalue equation]
\label{lem:eigenvalue-eq}
$\Hlim\,\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\,\Phi_n^{(\infty)}$.
\end{lemma}
\begin{proof}
\textbf{1.} SOT \eqref{XI3} pointwise:
$\mathcal{H}_{c_k}\,\Phi_n^{(\infty)}
\to\Hlim\,\Phi_n^{(\infty)}$.
\textbf{2.}
$\mathcal{H}_{c_k}\,\Phi_n^{(\infty)}
=\mathcal{H}_{c_k}(\Phi_n^{(\infty)}-\Phi_n^{(c_k)})
+\lambda_n^{(c_k)}\,\Phi_n^{(c_k)}$.
\textbf{3.} Lem.~\ref{lem:strong-conv} + \eqref{XI2}:
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$ strongly.
\textbf{4.}
$\|\mathcal{H}_{c_k}(\Phi_n^{(\infty)}-\Phi_n^{(c_k)})\|
\le\|\Phi_n^{(\infty)}-\Phi_n^{(c_k)}\|\to 0$.
\textbf{5.}
$\|\lambda_n^{(c_k)}\Phi_n^{(c_k)}
-\lambda_n^{(\infty)}\Phi_n^{(\infty)}\|
\to 0$ by \eqref{XI1} + Step~3.
\end{proof}

\begin{remark}[Why SOT suffices]
(i)~Unit eigenvectors;
(ii)~limit norm $=1$ from Paper~IX;
(iii)~quantitative rate $c^{-1/4}$.
No resolvent or compactness needed.
\end{remark}

\begin{theorem}[Spectral inclusion]
\label{thm:spectral-inclusion}
\begin{enumerate}[\rm(a)]
\item $\lambda_n^{(\infty)}\in\sigma_p(\Hlim)$
      for each $n\ge 0$.
\item $\overline{\{\lambda_n^{(\infty)}\}}
      \subset\sigma(\Hlim)$.
\end{enumerate}
\end{theorem}
\begin{proof}
(a)~Lem.~\ref{lem:eigenvalue-eq};
$\Phi_n^{(\infty)}\ne 0$.
(b)~$\sigma(\Hlim)$ closed.
\end{proof}

\begin{remark}[Point, approximate, and full spectrum]
\label{rem:sigma-app}
\[
  \overline{\{\lambda_n^{(\infty)}\}}
  \subset\sigapp(\Hlim)
  \subset\sigma(\Hlim)
  \subset[0,1].
\]
Outer bounds unconditional.
$\sigapp(\Hlim)=\sigma(\Hlim)$:
conditional on completeness (Paper~IX).
Paper~XI's unconditional contribution:
$\sigma_p(\Hlim)$ non-empty and explicitly identified.
\end{remark}

\begin{remark}[Spectral bounds]
Lower: above.
Upper: $\sigma(\Hlim)\subset[0,1]$ (Paper~X).
Equality: Section~\ref{sec:density}.
\end{remark}

\section{Semiclassical Spectral Transfer}
\label{sec:weyl}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Transition bandwidth]
\label{def:clambda}
For $\lambda\in(0,1)$, set
\begin{equation}\label{eq:clambda}
  c(\lambda)
  := \inf\bigl\{c > 0 :
     \lambda_{\lfloor 2c/\pi\rfloor}^{(c)}
     \le \lambda\bigr\}.
\end{equation}
This is the bandwidth at which the Slepian
transition eigenvalue first crosses level $\lambda$.
Defined entirely from finite-$c$ data,
independently of $\{\lambda_n^{(\infty)}\}$.
\end{definition}

\begin{lemma}[Existence and monotonicity of $c(\lambda)$]
\label{lem:clambda-exists}
For every $\lambda\in(0,1)$:
\begin{enumerate}[\rm(i)]
\item $c(\lambda)\in(0,\infty)$
      (the infimum in \eqref{eq:clambda} is finite
      and positive);
\item $c(\lambda)$ is non-decreasing in $\lambda$;
\item under Hypothesis~\ref{hyp:SOH},
      $c(\lambda)$ is strictly increasing.
\end{enumerate}
\end{lemma}

\begin{proof}
\textbf{(i) (finite upper bound).}
For each $\lambda\in(0,1)$, the Slepian eigenvalues
$\lambda_n^{(c)}$ have a logistic-type
asymptotic profile centred at $n=2c/\pi$
(Widom~\cite{Widom1964}, Osipov--Rokhlin--Xiao
\cite{Osipov2013}, Ch.~3):
for every $\lambda_0\in(0,1)$ there exists
$c_0>0$ such that
$\lambda_{\lfloor 2c_0/\pi\rfloor}^{(c_0)}
= \lambda_0$.
Since $\lambda_{\lfloor 2c/\pi\rfloor}^{(c)}$
is continuous in $c$ and sweeps $(0,1)$
as $c$ ranges over $(0,\infty)$
(monotone in $c$ for fixed relative index;
the transition profile shifts continuously:
Landau--Pollak~\cite{LandauPollak1962},
Slepian~\cite{Slepian1961}),
the set $\{c>0:\lambda_{\lfloor 2c/\pi\rfloor}^{(c)}
\le\lambda\}$ is non-empty for every $\lambda>0$.
\textbf{(i) (positive lower bound).}
For $c$ small, $\lambda_{\lfloor 2c/\pi\rfloor}^{(c)}>\lambda$
(transition not yet reached);
hence $c(\lambda)>0$.

\textbf{(ii).}
$c(\lambda)$ is non-decreasing:
if $\lambda_1<\lambda_2$ then
$\{c:\lambda_{\lfloor 2c/\pi\rfloor}^{(c)}\le\lambda_1\}
\subset
\{c:\lambda_{\lfloor 2c/\pi\rfloor}^{(c)}\le\lambda_2\}$,
so $c(\lambda_1)\ge c(\lambda_2)$
(infimum over larger set is smaller).

\textbf{(iii).}
Under Hyp.~\ref{hyp:SOH}, the limiting eigenvalues
are injective; this propagates via \eqref{XI1}
to strict separation of crossing levels,
yielding strict monotonicity.
\end{proof}

\begin{remark}[Why IVT alone is insufficient]
\label{rem:IVT-insufficient}
For each fixed $c$, IVT gives that every
$\mu\in(0,1)$ is attained by some
$\lambda_n^{(c)}$.
However, the covering of $(0,1)$ by the
transition eigenvalue
$\lambda_{\lfloor 2c/\pi\rfloor}^{(c)}$
as $c$ varies requires continuity of
$c\mapsto\lambda_{\lfloor 2c/\pi\rfloor}^{(c)}$
and the logistic-type sweeping property
(established in \cite{Widom1964,Osipov2013}).
Simple IVT in the index direction for fixed $c$
does not suffice.
\end{remark}

\begin{definition}[Counting function]
\label{def:counting}
$N(\lambda):=
\#\{n\ge 0:\lambda_n^{(\infty)}\le\lambda\}$
(non-decreasing, right-continuous;
finite for $\lambda\in[0,1)$ under
Hyp.~\ref{hyp:SOH};
finiteness not guaranteed without SOH).
\end{definition}

\begin{hypothesis}[Spectral Ordering (SOH)]
\label{hyp:SOH}
$m\ne n\implies\lambda_m^{(\infty)}
\ne\lambda_n^{(\infty)}$.
\end{hypothesis}

\begin{remark}[SOH is structurally essential]
\label{rem:SOH-essential}
SOH is not a technical regularity assumption:
it is required for Theorem~\ref{thm:weyl} to
hold in the stated form.
Without SOH:
\begin{itemize}
\item $N(\lambda)$ may be infinite at accumulation
      points of $\{\lambda_n^{(\infty)}\}$;
\item the transfer $N(\lambda)
      \sim\frac{2}{\pi}c(\lambda)$ breaks down;
\item spectral collisions would require a
      fundamentally different counting argument.
\end{itemize}
SOH would follow from the uniform gap bound
\eqref{eq:gap-lower} (Rem.~\ref{rem:SOH-status};
Open Problem~\ref{prob:SOH}).
\end{remark}

\begin{remark}[SOH and gap control]
\label{rem:SOH-status}
SOH follows from:
\begin{equation}\label{eq:gap-lower}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge C_\kappa c^{-3/4},\quad n\le\kappa c,
\end{equation}
combined with \eqref{XI1}.
Estimate \eqref{eq:gap-lower} is the lower bound
of Problem~\ref{prob:rate-sharp}.
\end{remark}

\begin{theorem}[Semiclassical spectral transfer]
\label{thm:weyl}
Under Hypothesis~\ref{hyp:SOH},
with $c(\lambda)$ from Def.~\ref{def:clambda}:
\begin{enumerate}[\rm(a)]
\item \textbf{(Transfer law.)} For $\lambda\in(0,1)$:
\begin{equation}\label{eq:transfer}
  N(\lambda)
  = \Bigl\lfloor\tfrac{2}{\pi}c(\lambda)\Bigr\rfloor
    + O(\log c(\lambda)).
\end{equation}
This is a genuine asymptotic relation between
the limiting spectral count $N(\lambda)$
and the finite-$c$ transition index:
the $O(\log c)$ error comes from the width
of the Slepian transition window \eqref{XI6}
transported by the rate \eqref{XI1}.
\item \textbf{(Transition window.)} For
      $\lambda\in[\varepsilon,1{-}\varepsilon]$:
      $N(\lambda)-N(\lambda-\varepsilon)
      =O(\log c(\lambda))$.
\item \textbf{(Spectral floor.)}
      $N(\lambda)=0$ for
      $\lambda<\lambda_0^{(\infty)}
      :=\lim_k\lambda_0^{(c_k)}$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(a).}
At $c=c(\lambda)$, the index
$n_0:=\lfloor 2c(\lambda)/\pi\rfloor$
satisfies
$\lambda_{n_0}^{(c(\lambda))}\le\lambda$
(by definition of $c(\lambda)$).
By \eqref{XI6}, the transition window
$\{n:\lambda_n^{(c(\lambda))}\in[\varepsilon,1{-}\varepsilon]\}$
has at most $O(\log c(\lambda))$ elements
centred at $n_0$.
By \eqref{XI1}:
$|\lambda_n^{(c(\lambda))}-\lambda_n^{(\infty)}|
\le C_\kappa c(\lambda)^{-1/4}$.
Hence the set
$\{n:\lambda_n^{(\infty)}\le\lambda\}$
differs from $\{n:\lambda_n^{(c(\lambda))}\le\lambda\}$
by at most $O(\log c(\lambda))$ indices
(indices that could cross $\lambda$ due to the
$c^{-1/4}$ perturbation lie inside the
transition window).
Therefore $N(\lambda)
=\lfloor 2c(\lambda)/\pi\rfloor
+O(\log c(\lambda))$.

\textbf{(b).}
Direct from \eqref{XI6} and \eqref{XI1}.

\textbf{(c).}
No $\lambda_n^{(\infty)}$ lies below
$\lambda_0^{(\infty)}$.
\end{proof}

\begin{remark}[Non-circularity]
$c(\lambda)$ is from finite-$c$ data;
$N(\lambda)$ is from the limit.
Equation~\eqref{eq:transfer} is a
non-trivial transfer between two apriori
unrelated scales, with explicit $O(\log c)$ error.
\end{remark}

\begin{remark}[Not a classical Weyl law]
$\Hlim$ is not compact;
$N(\lambda)$ finite only under SOH.
Result is a semiclassical index transfer.
\end{remark}

\section{Density Criterion}
\label{sec:density}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Density criterion]
\label{thm:density-criterion}
If $\{\lambda_n^{(\infty)}\}$ is dense in $[0,1]$,
then $\sigma(\Hlim)=[0,1]$.
\end{theorem}
\begin{proof}
Each $\lambda_n^{(\infty)}\in\sigma(\Hlim)$
(Thm.~\ref{thm:spectral-inclusion}(a));
$\sigma(\Hlim)$ closed; density gives
$[0,1]\subset\sigma(\Hlim)$.
Reverse: Paper~X.
\end{proof}

\begin{corollary}
Under density, $\sigma(\Hlim)$ has no isolated
points in $(0,1)$.
\end{corollary}

\section{The Density Question}
\label{sec:density-question}
%%═══════════════════════════════════════════════════════════════════

\begin{conjecture}[Density]
\label{conj:density}
$\{\lambda_n^{(\infty)}\}$ is dense in $[0,1]$.
\end{conjecture}

\subsection{Evidence and barrier}

By \eqref{XI6}, for each $c$ there are $O(\log c)$
eigenvalues in $[\varepsilon,1{-}\varepsilon]$,
transition window at $n=2c/\pi$.
By \eqref{XI1}, each lies within
$C_\kappa c^{-1/4}$ of $\lambda_n^{(\infty)}$.

\begin{remark}[Two-parameter reformulation]
\label{rem:density-gap}
Density is equivalent to:
\begin{quote}
  For every open $I\subset(0,1)$,
  $\exists k_j\to\infty$, $n_j\ge 0$:
  $\lambda_{n_j}^{(c_{k_j})}\in I$ for all $j$.
\end{quote}
IVT gives attainment for each fixed $c$;
the difficulty is
\emph{uniform distribution} of the
$O(\log c)$ transition window across
all of $(0,1)$ as $c\to\infty$.
This is a moving-window density problem,
analogous to equidistribution in semiclassical
analysis; it is the \emph{open barrier}
for this conjecture.
No new tool for this is available from
Papers~VI--VIII.
\end{remark}

\begin{proposition}[Non-emptiness near $1$]
$\{\lambda_n^{(\infty)}\}\cap[1{-}\delta,1)
\ne\emptyset$ for all $\delta\in(0,1)$.
\end{proposition}
\begin{proof}\eqref{XI6} + \eqref{XI1}.\end{proof}

Numerical tables \cite{Slepian1961,Osipov2013,
LandauPollak1962} consistently show
transition eigenvalues filling $[0,1]$
densely for finite $c$.

\section{Open Problems}
\label{sec:open-problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Density]
\label{prob:density}
Prove Conj.~\ref{conj:density};
equivalent to the two-parameter coverage
(Rem.~\ref{rem:density-gap}).
\end{problem}

\begin{problem}[Spectral Ordering]
\label{prob:SOH}
Prove Hyp.~\ref{hyp:SOH};
follows from \eqref{eq:gap-lower}.
\end{problem}

\begin{problem}[Uniqueness]
\label{prob:uniqueness}
$\Phi_n^{(\infty)}$ independent of the
diagonal subsequence, up to phase.
\end{problem}

\begin{problem}[Form rate sharpness]
\label{prob:rate-sharp}
$(q_{\lim}-q_c)(f,f)/\|f\|^2
\asymp c^{-1/4}$?
Lower bound requires \eqref{eq:gap-lower}.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\{\Phi_n^{(\infty)}\}$ complete in $L^2$?
$\Leftrightarrow\sigapp(\Hlim)=\sigma(\Hlim)$.
\end{problem}

\begin{problem}[Riesz-basis stability]
\label{prob:riesz}
$\{\Phi_n^{(\infty)}\}$ Riesz basis?
Implies completeness,
$\HSOT=\overline{\Hspectral}$.
Target: Paper~XII.
\end{problem}

\begin{remark}[Hierarchy]
\ref{prob:density} + \ref{prob:completeness}:
both needed for Hilbert--P\'olya.
\ref{prob:riesz}: resolves completeness + Bridge.
\ref{prob:SOH} $\Leftarrow$ \ref{prob:rate-sharp}.
\ref{prob:uniqueness}: removes subsequential
qualifiers throughout.
\end{remark}

\section{Outlook}
\label{sec:outlook}
%%═══════════════════════════════════════════════════════════════════

Papers~I--XI reduce the Hilbert--P\'olya program
to the following independent structural
conditions; no RH implication is claimed
without all being established:
\begin{enumerate}
\item \textbf{Density} (\ref{prob:density}).
\item \textbf{Completeness} (\ref{prob:completeness}).
\item \textbf{Spectral ordering}
      (Hyp.~\ref{hyp:SOH}).
\item \textbf{Uniqueness} (\ref{prob:uniqueness}).
\end{enumerate}
Under all of (1)--(4) plus CCM (\cite{CCM2025}),
RH would follow; this chain is not yet closed.
Problem~\ref{prob:riesz} would resolve (2)
and the Bridge theorem simultaneously;
target of Paper~XII.

\medskip\noindent
Precise state of the programme as of Paper~XI.
