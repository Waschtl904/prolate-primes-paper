\begin{abstract}
We continue the operator-theoretic analysis of the limiting PSWF
concentration operator $\Hlim$, established in Papers~IX--X of
this series.
The operator $\Hlim$ is defined as the SOT-limit of the
finite-bandwidth operators $\Hc$ (Paper~X); its eigenstructure
is inherited from Paper~IX via the subsequential orthonormal
system $\{\Phi_n^{(\infty)}\}$.
Without assuming completeness of $\{\Phi_n^{(\infty)}\}$,
we prove:
\begin{enumerate}
\item \textbf{Theorem~\ref{thm:spectral-inclusion}
      (Spectral Inclusion).}
      $\overline{\{\lambda_n^{(\infty)}\}} \subset \sigma(\Hlim)$,
      with each $\lambda_n^{(\infty)}$ a genuine eigenvalue.
\item \textbf{Theorem~\ref{thm:weyl}
      (Weyl-type counting law).}
      The counting function
      $N(\lambda) := \#\{n : \lambda_n^{(\infty)} \le \lambda\}$
      satisfies $N(\lambda) \sim \frac{2}{\pi} c_\lambda$
      where $c_\lambda$ is the bandwidth parameter at which
      $\lambda_{[2c/\pi]}^{(c)} = \lambda$.
\item \textbf{Theorem~\ref{thm:density-criterion}
      (Density Criterion).}
      If $\{\lambda_n^{(\infty)}\}$ is dense in $[0,1]$,
      then $\sigma(\Hlim) = [0,1]$.
\end{enumerate}
The Riesz-basis stability question for $\{\Phi_n^{(\infty)}\}$
and the operator identification $\HSOT = \overline{\Hspectral}$
are deferred to Paper~XII.
No unconditional connection to $\zeta(s)$-zeros is made.
\end{abstract}

\section*{Guide for the Referee}

This paper builds on two distinct layers inherited from
the series:
\begin{itemize}
\item \textbf{Operator layer} (Paper~X):
      $\Hlim$ exists as SOT-limit; bounded; self-adjoint.
\item \textbf{Eigenstructure layer} (Paper~IX):
      $\{\Phi_n^{(\infty)},\lambda_n^{(\infty)}\}$ is an
      orthonormal system of eigendata,
      constructed subsequentially.
\end{itemize}
Paper~XI's contribution is the
\emph{spectral geometry layer}: using both inherited layers
to prove spectral inclusion, a semi-classical counting law,
and isolation of the density question.
No eigenfunction completeness is assumed.
Frame-stability and Riesz-basis analysis are deferred
to Paper~XII.
No unconditional spectral identification with $\zeta$-zeros.

\section{Setup and Inherited Results}
\label{sec:setup}
%%═══════════════════════════════════════════════════════════════════

We retain all notation from Papers~VIII--X.

\begin{enumerate}[(XI.1)]
\item\label{XI1}
  \textbf{Eigenvalue convergence rate} (Paper~VIII, Cor.~4.3):
  \[
    |\lambda_n^{(c)} - \lambda_n^{(\infty)}|
    \;\le\; C_\kappa\, c^{-1/4}
    \quad \text{for all } n \le \kappa c,\; c \ge c_0(\kappa).
  \]
\item\label{XI2}
  \textbf{Orthonormal subsequence limit} (Paper~IX, Prop.~2.2):
  After passing to a diagonal subsequence $c_k \to \infty$,
  $\Phi_n^{(c_k)} \rightharpoonup \Phi_n^{(\infty)}$ weakly
  in $L^2(\mathbb{R})$,
  and $\{\Phi_n^{(\infty)}\}_{n \ge 0}$ is an
  \emph{orthonormal system}: $\ip{\Phi_m^{(\infty)}}{
  \Phi_n^{(\infty)}} = \delta_{mn}$.
  In particular:
  \begin{enumerate}[(a)]
  \item $\|\Phi_n^{(\infty)}\|_{L^2} = 1$ for all $n \ge 0$.
  \item The convergence is subsequential;
        uniqueness of $\Phi_n^{(\infty)}$ up to phase
        requires additional arguments (see
        Remark~\ref{rem:subsequence} below).
  \end{enumerate}
\item\label{XI3}
  \textbf{SOT-convergence} (Paper~X, Lem.~2.2):
  $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
  in $\mathcal{B}(L^2(\mathbb{R}))$
  along the same subsequence as \eqref{XI2}.
  $\Hlim$ is bounded and self-adjoint,
  defined independently of any spectral series.
\item\label{XI4}
  \textbf{Non-cancellation} (Paper~VIII, Thm.~3.3):
  For all $n \le \kappa c$, there exists a Borel set $E_c$
  with $|E_c| \ge c_* c^{-1/2}$ and
  $|\Phihat_n^{(c)}(t)| \ge c_1 c^{-1/2}$ on $E_c$.
  In particular $\Phi_n^{(\infty)} \not\equiv 0$
  (confirmed independently of \eqref{XI2}(a)).
\item\label{XI5}
  \textbf{Quantitative form rate} (Paper~VIII, Cor.~4.3):
  \[
    0 \le q_{\lim}(f,f) - q_c(f,f)
    \le C_\kappa\, c^{-1/4}\|f\|^2.
  \]
  \emph{Unconditional; sole source is Paper~VIII.
  Paper~X, Thm.~5.1 reformulates this in variational
  language without new content.
  Used only in Sections~\ref{sec:weyl}--\ref{sec:density-question}.}
\item\label{XI6}
  \textbf{Slepian counting law} \cite{Slepian1961,Osipov2013}:
  For fixed $c$, the $\lfloor 2c/\pi\rfloor + O(1)$
  eigenvalues with $n < (1-\delta)2c/\pi$ satisfy
  $\lambda_n^{(c)} \ge 1 - e^{-\gamma_\delta c}$;
  those with $n > (1+\delta)2c/\pi$ satisfy
  $\lambda_n^{(c)} \le e^{-\gamma_\delta c}$.
\end{enumerate}

\begin{remark}[Subsequential construction and scope of results]
\label{rem:subsequence}
All results in this paper hold along the fixed
subsequence $c_k \to \infty$ established in
Paper~IX, Prop.~2.2.
The objects $\Phi_n^{(\infty)}$ and $\Hlim$ are defined
along this subsequence;
statements of the form ``$\lim_{c\to\infty}$'' should
be read as ``$\lim_{k\to\infty}$ along $c_k$.''

Uniqueness of the limit objects up to phase:
if the limiting eigenspaces are one-dimensional
(which is expected from Slepian non-degeneracy but
not yet proved in this series),
the $\Phi_n^{(\infty)}$ are unique up to a phase factor
and all subsequences give the same spectral data.
We assume non-degeneracy without proof here;
this is flagged as an open item for Paper~XII
(see also Problem~\ref{prob:uniqueness}).
\end{remark}

\begin{remark}[Dependency architecture of Paper~XI]
\label{rem:dependency-architecture}
Paper~XI sits at the interface of two distinct inherited layers:

\medskip
\noindent
\textbf{What Paper~IX provides (eigenstructure layer):}
\begin{itemize}
\item Existence of a weakly convergent subsequence
      $\Phi_n^{(c_k)} \rightharpoonup \Phi_n^{(\infty)}$
\item Orthonormality: $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}
      = \delta_{mn}$ (hence $\|\Phi_n^{(\infty)}\|=1$)
\item Norm control: $\|\Phi_n^{(c)}\| = 1 \to
      \|\Phi_n^{(\infty)}\| = 1$ (via orthonormality)
\item Spectral labeling consistency:
      $\Phi_n^{(\infty)}$ is labelled by the same
      index $n$ as the finite-$c$ eigenfunctions
\end{itemize}

\medskip
\noindent
\textbf{What Paper~X provides (operator layer):}
\begin{itemize}
\item SOT-existence of $\Hlim$ as a bounded operator
\item Self-adjointness of $\Hlim$
\item Spectral containment $\sigma(\Hlim) \subset [0,1]$
\end{itemize}

\medskip
\noindent
\textbf{What Paper~XI provides (spectral geometry layer):}
\begin{itemize}
\item Spectral inclusion:
      each $\lambda_n^{(\infty)} \in \sigma_p(\Hlim)$
      unconditionally (no completeness)
\item Semi-classical Weyl-type counting law for
      $\{\lambda_n^{(\infty)}\}$
\item Reduction of $\sigma(\Hlim)$-structure to a
      single scalar question (density conjecture)
\item Isolation of Riesz-basis and operator
      identification as the next open layer (Paper~XII)
\end{itemize}

\medskip
\noindent
The eigenstructure layer (Paper~IX) is structurally
\emph{necessary}: SOT-convergence alone does not control
norm or orthogonality of limit eigenvectors.
Attempts to derive $\|\Phi_n^{(\infty)}\|=1$ from
non-cancellation (Paper~VIII) alone lead to circular
arguments (see Remark~\ref{rem:norm-circulariy}).
\end{remark}

\begin{remark}[Why Paper~IX orthonormality cannot be eliminated]
\label{rem:norm-circulariy}
SOT-convergence $\Hc \to \Hlim$ and non-cancellation
(Paper~VIII, Thm.~3.3) together provide:
\begin{enumerate}[(i)]
\item A weak cluster point $\Phi_n^{(\infty)}$
      with $\Phi_n^{(\infty)} \ne 0$
      (from non-cancellation: prevents collapse to zero)
\item $\Hlim \Phi_n^{(\infty)} = \lambda_n^{(\infty)}
      \Phi_n^{(\infty)}$ provided strong convergence
      $\Phi_n^{(c)} \to \Phi_n^{(\infty)}$ holds
\end{enumerate}
However, strong convergence (step (ii)) requires
$\|\Phi_n^{(\infty)}\| = 1$, which in turn requires
either orthonormality (Paper~IX) or an independent
norm-preservation theorem.
No such independent theorem is available from
Paper~VIII alone: non-cancellation gives
$\|\Phi_n^{(c)}\|_{L^2(E_c)} \ge c_1 c_* c^{-1/2} > 0$
on a set $E_c$ of measure $\sim c^{-1/2}$,
but $E_c \to \emptyset$ as $c \to \infty$ and
no global lower norm bound survives the limit.
Hence $\|\Phi_n^{(\infty)}\| = 1$ is an output of
Paper~IX's orthonormality construction,
not derivable from Paper~VIII alone.
\end{remark}

\begin{remark}[Definition of $\Hlim$ in this paper]
\label{rem:operator-definition}
Throughout Paper~XI, $\Hlim$ denotes the SOT-limit
$\Hlim := \mathrm{SOT}\text{-}\lim_{k\to\infty} \mathcal{H}_{c_k}$
(Paper~X, Lem.~2.2): a bounded, self-adjoint operator
on $L^2(\mathbb{R})$, defined without spectral series
or completeness of $\{\Phi_n^{(\infty)}\}$.
\end{remark}

\begin{remark}[What is not assumed]
\label{rem:no-completeness}
Completeness of $\{\Phi_n^{(\infty)}\}$ in $L^2(\mathbb{R})$
(\texttt{prob:completeness}, Papers~IX--X) is
not assumed anywhere in this paper.
Frame-stability and Riesz-basis properties
are deferred to Paper~XII.
\end{remark}

\section{Spectral Inclusion}
\label{sec:spectral-inclusion}
%%═══════════════════════════════════════════════════════════════════

\begin{lemma}[Non-vanishing of limit eigenvectors]
\label{lem:nonvanish-limit}
For each $n \ge 0$, $\|\Phi_n^{(\infty)}\|_{L^2} = 1$.
In particular, $\Phi_n^{(\infty)} \ne 0$.
\end{lemma}

\begin{proof}
By \eqref{XI2}, $\{\Phi_n^{(\infty)}\}$ is an
orthonormal system (Paper~IX, Prop.~2.2),
so $\|\Phi_n^{(\infty)}\| = 1$ by definition.
\end{proof}

\begin{lemma}[Eigenvalue equation for $\Hlim$]
\label{lem:eigenvalue-eq}
For each $n \ge 0$:
$\Hlim\, \Phi_n^{(\infty)}
= \lambda_n^{(\infty)}\, \Phi_n^{(\infty)}$.
\end{lemma}

\begin{proof}
We use \eqref{XI1} (eigenvalue rate),
\eqref{XI2} (orthonormal subsequence limit),
and \eqref{XI3} (SOT-convergence).

\textbf{Step 1 (SOT applied).}
$\Hc\, \Phi_n^{(\infty)}
\xrightarrow{L^2} \Hlim\, \Phi_n^{(\infty)}$
by \eqref{XI3} applied to the fixed vector
$\Phi_n^{(\infty)}$.

\textbf{Step 2 (Decomposition).}
\[
  \Hc\, \Phi_n^{(\infty)}
  = \Hc(\Phi_n^{(\infty)} - \Phi_n^{(c)})
    + \lambda_n^{(c)}\, \Phi_n^{(c)}.
\]

\textbf{Step 3 (Strong convergence of eigenvectors).}
From \eqref{XI2}: $\Phi_n^{(c)} \rightharpoonup \Phi_n^{(\infty)}$
weakly and $\|\Phi_n^{(c)}\| = 1 = \|\Phi_n^{(\infty)}\|$
(the latter from orthonormality \eqref{XI2}(a)).
Weak convergence plus norm convergence in Hilbert
space implies strong convergence:
$\Phi_n^{(c)} \to \Phi_n^{(\infty)}$ in $L^2$.

\textbf{Step 4.}
$\|\Hc(\Phi_n^{(\infty)} - \Phi_n^{(c)})\|
\le \|\Phi_n^{(\infty)} - \Phi_n^{(c)}\| \to 0$
(since $\|\Hc\| \le 1$).

\textbf{Step 5.}
\[
  \|\lambda_n^{(c)} \Phi_n^{(c)}
  - \lambda_n^{(\infty)} \Phi_n^{(\infty)}\|
  \le |\lambda_n^{(c)} - \lambda_n^{(\infty)}|
  + \lambda_n^{(\infty)}
    \|\Phi_n^{(c)} - \Phi_n^{(\infty)}\| \to 0
\]
using \eqref{XI1} and Step~3.

\textbf{Conclusion.}
$\Hlim\, \Phi_n^{(\infty)}
= \lim \lambda_n^{(c)}\, \Phi_n^{(c)}
= \lambda_n^{(\infty)}\, \Phi_n^{(\infty)}$.
Since $\Phi_n^{(\infty)} \ne 0$, this is genuine.
\end{proof}

\begin{remark}[Dependency map of this section]
\label{rem:no-specdef}
The proof of Lemma~\ref{lem:eigenvalue-eq} uses exactly:
\begin{itemize}
\item \eqref{XI3}: SOT-convergence (Paper~X)
\item \eqref{XI1}: eigenvalue rate (Paper~VIII)
\item \eqref{XI2}: weak convergence \emph{and}
      orthonormality (Paper~IX, Prop.~2.2) ---
      the latter is essential for $\|\Phi_n^{(\infty)}\|=1$
      and hence for strong convergence in Step~3
\item $\|\Hc\| \le 1$: Slepian contractivity (Papers~V--VI)
\end{itemize}
It does \emph{not} use:
\begin{itemize}
\item Paper~IX Def.~2.3 (spectral series)
\item Completeness of $\{\Phi_n^{(\infty)}\}$
\item Paper~IX Thm.~4.1 (operator norm rate)
\end{itemize}
Paper~IX is used for \emph{one structural fact}:
orthonormality of the limit system.
See Remark~\ref{rem:norm-circulariy} for why
this cannot be replaced by Paper~VIII alone.
\end{remark}

\begin{theorem}[Spectral inclusion]
\label{thm:spectral-inclusion}
\begin{enumerate}[\rm(a)]
\item For each $n \ge 0$,
      $\lambda_n^{(\infty)} \in \sigma_p(\Hlim)$.
\item $\overline{\{\lambda_n^{(\infty)} : n \ge 0\}}
      \subset \sigma(\Hlim)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(a) Lemma~\ref{lem:eigenvalue-eq} with
$\Phi_n^{(\infty)} \ne 0$ (Lemma~\ref{lem:nonvanish-limit}).
(b) $\sigma(\Hlim)$ is closed
(\cite{ReedSimon1975}, Thm.~VII.1);
it contains each $\lambda_n^{(\infty)}$ by (a).
\end{proof}

\begin{remark}[Scope of part (b)]
\label{rem:closure-scope}
Part (b) is a consequence of part (a) and the
topological closedness of spectra: it establishes
the spectral \emph{lower bound}
$\overline{\{\lambda_n^{(\infty)}\}} \subset \sigma(\Hlim)$
but makes no claim about the structure of
$\sigma(\Hlim)$ beyond this inclusion.
The upper bound $\sigma(\Hlim) \subset [0,1]$
was established unconditionally in Paper~X.
The question of whether $\sigma(\Hlim) = [0,1]$
is the content of Section~\ref{sec:density}.
\end{remark}

\begin{remark}[Comparison with Paper~IX]
\label{rem:comparison-IX}
Paper~IX proves $\sigapp(\Hlim) = \sigma(\Hlim)$
conditionally on completeness.
Theorem~\ref{thm:spectral-inclusion}(a) gives
each $\lambda_n^{(\infty)} \in \sigma_p(\Hlim)$
unconditionally, bypassing essential self-adjointness
entirely.
\end{remark}

\section{Weyl-Type Counting Law}
\label{sec:weyl}
%%═══════════════════════════════════════════════════════════════════

\begin{definition}[Counting function]
\label{def:counting}
$N(\lambda) := \#\{n \ge 0 : \lambda_n^{(\infty)} \le \lambda\}$.
\end{definition}

\begin{remark}[Well-definedness]
$N(\lambda) < \infty$ for $\lambda \in [0,1)$
(Slepian eigenvalues decay exponentially past the threshold);
$N$ is non-decreasing and right-continuous.
\end{remark}

The Slepian--Pollak counting theorem
(\cite{Slepian1961,Osipov2013,LandauPollak1962})
gives: for fixed $c$ and $\varepsilon > 0$,
\begin{equation}
\label{eq:slepian-count}
  \#\{n : \lambda_n^{(c)} \in [\varepsilon, 1-\varepsilon]\}
  = O(\log c),
\end{equation}
with the transition window centred at $n = 2c/\pi$.

\begin{theorem}[Semi-classical counting law]
\label{thm:weyl}
Assume that the map $\lambda \mapsto c(\lambda)$,
defined implicitly by
$\lfloor 2c(\lambda)/\pi\rfloor = N(\lambda)$,
is well-defined and strictly monotone
\emph{(non-degeneracy assumption; see
Remark~\ref{rem:weyl-assumption})}.
Then:
\begin{enumerate}[\rm(a)]
\item $N(\lambda) \sim \dfrac{2c(\lambda)}{\pi}$
      as $\lambda \to 1^-$.
\item For $\lambda \in [\varepsilon,1-\varepsilon]$:
      $N(\lambda) = O(\log c(\lambda))$.
\item $N(\lambda) = 0$ for $\lambda < \lambda_0^{(\infty)}$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(a)} By \eqref{XI6} and \eqref{XI1},
$\lambda_n^{(\infty)} \ge \lambda$ iff
$n \le 2c(\lambda)/\pi + O(\log c(\lambda))$,
giving $N(\lambda) \sim 2c(\lambda)/\pi$.
\textit{(b)} Direct from \eqref{eq:slepian-count}
and \eqref{XI1}.
\textit{(c)} No $\lambda_n^{(\infty)}$ lies below
$\lambda_0^{(\infty)} = \lim_k \lambda_0^{(c_k)}$.
\end{proof}

\begin{remark}[Non-degeneracy assumption]
\label{rem:weyl-assumption}
The monotonicity of $c(\lambda)$ is used but not
proved here; it would follow from strict monotonicity
of $n \mapsto \lambda_n^{(\infty)}$ (index ordering
preserved under the limit), which is expected from
Slepian non-degeneracy but is an open item.
The theorem is therefore conditional on this
non-degeneracy.
\end{remark}

\begin{remark}[Nature of this counting law]
\label{rem:weyl-interp}
This is not a Weyl law in the classical sense
($\Hlim$ is not compact).
It is a \emph{semi-classical index transfer}:
the counting function of the limiting operator
is expressed via the bandwidth parameter of
the approximating finite-$c$ family.
\end{remark}

\section{Density Criterion and Spectral Structure}
\label{sec:density}
%%═══════════════════════════════════════════════════════════════════

\begin{theorem}[Density criterion]
\label{thm:density-criterion}
If $\{\lambda_n^{(\infty)}\}$ is dense in $[0,1]$,
then $\sigma(\Hlim) = [0,1]$.
\end{theorem}

\begin{proof}
$\sigma(\Hlim)$ contains each $\lambda_n^{(\infty)}$
(Theorem~\ref{thm:spectral-inclusion}(a))
and is closed; if $\{\lambda_n^{(\infty)}\}$ is dense
in $[0,1]$ then $[0,1] \subset \sigma(\Hlim)$.
The reverse inclusion $\sigma(\Hlim) \subset [0,1]$
is unconditional (Paper~X, Cor.~\ref{cor:qlim-op}).
\end{proof}

\begin{corollary}[No isolated bulk eigenvalues]
\label{cor:no-isolated}
Under the density hypothesis,
$\sigma(\Hlim)$ has no isolated points in $(0,1)$.
\end{corollary}

\begin{remark}[Resolution of Paper~X problems]
\label{rem:open-problems-resolved}
Conditional on density:
\texttt{prob:spectrum} ($\sigma(\Hlim)=?$) $\to$ $[0,1]$;
\texttt{prob:spec-structure} (isolated points?) $\to$ none.
Both reduce to: is $\{\lambda_n^{(\infty)}\}$ dense?
\end{remark}

\section{The Density Question}
\label{sec:density-question}
%%═══════════════════════════════════════════════════════════════════

\begin{conjecture}[Density]
\label{conj:density}
$\{\lambda_n^{(\infty)}\}_{n \ge 0}$ is dense in $[0,1]$.
\end{conjecture}

\subsection{Analytical evidence}

For each $c$ and $\varepsilon > 0$, \eqref{eq:slepian-count}
places $O(\log c)$ eigenvalues $\lambda_n^{(c)}$ in
$[\varepsilon, 1-\varepsilon]$, with the transition window
shifting as $c$ varies.
By \eqref{XI1}, each $\lambda_n^{(c)}$ is within
$C_\kappa c^{-1/4}$ of $\lambda_n^{(\infty)}$.

\begin{remark}[Gap in the density argument]
\label{rem:density-gap}
The above suggests density but does not prove it.
A rigorous proof requires showing that the
transition eigenvalues $\lambda_n^{(c)}$
(in the $O(\log c)$-window around $n=2c/\pi$)
cover $[0,1]$ densely \emph{as $c$ varies}.
This is a non-trivial two-parameter
(in $n$ and $c$) limit statement and cannot
be reduced to IVT alone;
it requires uniform coverage control not
currently available from Papers~VI--VIII.
\end{remark}

\begin{proposition}[Non-emptiness in $[1-\delta,1)$]
\label{prop:partial-density}
For any $\delta \in (0,1)$,
$\{\lambda_n^{(\infty)}\} \cap [1-\delta,1) \ne \emptyset$.
\end{proposition}

\begin{proof}
Direct from \eqref{XI6} and \eqref{XI1}.
\end{proof}

\subsection{Numerical evidence}

Numerical tables for finite $c$
(\cite{Slepian1961,Osipov2013,LandauPollak1962})
consistently show transition eigenvalues filling
$[0,1]$ densely; the $c^{-1/4}$ rate in \eqref{XI1}
makes these reliable approximations to
$\lambda_n^{(\infty)}$.

\section{Open Problems}
\label{sec:open-problems}
%%═══════════════════════════════════════════════════════════════════

\begin{problem}[Density]
\label{prob:density}
Prove Conjecture~\ref{conj:density}.
Requires uniform coverage control of transition
eigenvalues $\lambda_n^{(c)}$ as $c \to \infty$;
see Remark~\ref{rem:density-gap}.
\end{problem}

\begin{problem}[Uniqueness of limit objects]
\label{prob:uniqueness}
Show that $\Phi_n^{(\infty)}$ is independent
(up to phase) of the choice of subsequence
$c_k \to \infty$ in \eqref{XI2}.
This would remove the subsequential qualifier
from all results in this paper.
Expected to follow from one-dimensionality of
limiting eigenspaces (Slepian non-degeneracy
in the limit), but not yet proved.
\end{problem}

\begin{problem}[Sharpness of form rate]
\label{prob:rate-sharp}
Is $\sup_f (q_{\lim}(f,f) - q_c(f,f))/\|f\|^2
\asymp c^{-1/4}$?
Upper bound is unconditional (Paper~VIII, Cor.~4.3).
Lower bound requires
$|\lambda_n^{(c)} - \lambda_n^{(c+1)}|
\ge c_\kappa c^{-3/4}$ (open, Paper~VIII Rem.~4.2).
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\{\Phi_n^{(\infty)}\}$ complete in $L^2(\mathbb{R})$?
Equivalent to essential self-adjointness of $\Hlim$
and $\sigapp(\Hlim) = \sigma(\Hlim)$.
\end{problem}

\begin{problem}[Riesz-basis stability]
\label{prob:riesz}
Is $\{\Phi_n^{(\infty)}\}$ a Riesz basis for its
closed linear span?
Would imply completeness, provide frame bounds,
and yield $\HSOT = \overline{\Hspectral}$.
Primary target of Paper~XII.
\end{problem}

\begin{remark}[Problem hierarchy]
\label{rem:problems-relation}
Problems~\ref{prob:density} and~\ref{prob:completeness}
are independent.
Both are needed for an unconditional Hilbert--P\'olya
realization.
Problem~\ref{prob:riesz} (Paper~XII) would
resolve completeness and the Bridge theorem
simultaneously.
Problem~\ref{prob:uniqueness} is a prerequisite
for removing the subsequential qualifier from
the entire series.
\end{remark}

\section{Outlook: Reduction of the Hilbert--P\'olya Problem}
\label{sec:outlook}
%%═══════════════════════════════════════════════════════════════════

The results of Papers~I--XI reduce the
Hilbert--P\'olya question to three independent
spectral-analytic problems:

\medskip
\begin{enumerate}
\item \textbf{Density} (Conjecture~\ref{conj:density}):
      $\sigma(\Hlim) = [0,1]$ follows (this paper).
\item \textbf{Completeness} (\texttt{prob:completeness}):
      $\sigapp(\Hlim) = \sigma(\Hlim)$ follows
      (Paper~IX).
\item \textbf{Uniqueness}
      (Problem~\ref{prob:uniqueness}):
      removes subsequential qualifiers throughout.
\end{enumerate}

\medskip\noindent
Under (1) and (2) plus Hypothesis~CCM
(\cite{CCM2025}): $\mathrm{Re}(\rho)=1/2$
for all non-trivial zeros $\rho$ of $\zeta(s)$.

\medskip
The Riesz-basis question
$\HSOT = \overline{\Hspectral}$
(Paper~IX, OP~7; Paper~X, Hyp.~(IX.b))
would resolve completeness and the Bridge theorem
simultaneously and is the primary target of Paper~XII.

\medskip\noindent
This is the precise state of the
Hilbert--P\'olya programme within the
present framework as of Paper~XI.
