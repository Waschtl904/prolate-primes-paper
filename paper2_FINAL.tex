\begin{abstract}
This paper continues the program initiated in \cite{PaperI}, where the
prolate Gram form $Q^{(N)}$ was shown to be unconditionally coercive on
the $N$-dimensional prolate subspace $V_N$, with explicit spectral
stability under the sampling-quality condition~(T.2) of \cite{PaperI}.

Here we study the \emph{coupled scaling limit}
\[
  N_k \to \infty,\quad c_k \to \infty,\quad
  \frac{N_k}{c_k} \to \rho \in \Bigl(0,\,\frac{2}{\pi}\Bigr),
\]
and establish three main results.
\begin{enumerate}[(I)]
\item \textbf{(Scaling limit, \S\ref{sec:scaling}.)} We define a
precise notion of $\rho$-admissible scaling sequence $(N_k,c_k)$,
discuss the structural constraint $\omega_k\to0$ honestly
(Remark~\ref{rem:omega_to_zero}), and identify the natural operator
space for convergence as $\Lcal(L^2(\mathbb{R}))$ with the
weak-$*$ topology.
\item \textbf{(Uniform coercivity, \S\ref{sec:coercivity}.)} Using
the coercivity of \cite[Thm.~1]{PaperI} and the Landau--Widom
eigenvalue asymptotics, we prove $\lambda_{\min}(\GVMk)\to1$ along
any $\rho$-admissible sequence, under condition~(T.2) of \cite{PaperI}.
\item \textbf{(Trace formula, \S\ref{sec:trace}.)} We establish
an asymptotic trace formula for the weighted Gram matrix
$\GVMkw$ with weights $\log p_j$, connecting the spectral trace
to the prime number theorem. Every weak-$*$ accumulation point
$\Gcal_\infty$ is positive with $\operatorname{tr}(\Gcal_\infty)\leq1$;
the equality $\operatorname{tr}(\Gcal_\infty)=1$ is an open problem
(Problem~\ref{prob:unique}).
\end{enumerate}
The characterization of $\Gcal_\infty$ as a Weil operator, and the
resulting connection to the Riemann Hypothesis via Weil positivity,
remain open; they are precisely formulated as
Problems~\ref{prob:unique}, \ref{prob:mellin}, and
\ref{prob:positivity}.
No claim is made about zeros of~$\zeta(s)$.
\end{abstract}

\section{Introduction}\label{sec:intro}

\subsection{Background and motivation}

The \emph{Weil positivity criterion} \cite{Weil1952,Bombieri2000}
states that the Riemann Hypothesis is equivalent to positivity of
\[
  W(h)\;=\;
  \sum_{\substack{p\,\text{prime}\\k\geq1}}
  \log p\cdot|\widehat{h}(\log p^k)|^2
  +\text{(archimedean terms)}
\]
for all admissible test functions $h$.
The program of Connes--Consani--Moscovici \cite{CCM_PNAS2022,CCM2025}
seeks a spectral realization via the prolate operator $\Kop$.

In \cite{PaperI} we proved that the prolate Gram form
$Q^{(N)}(f)=\sum_{j=1}^N|f(p_j)|^2$ on $V_N$ is:
\begin{itemize}
\item \emph{unconditionally coercive} (Lemma~B of \cite{PaperI});
\item \emph{explicitly stable and Weil-decomposable} under the
sampling-quality condition~(T.2).
\end{itemize}
The present paper addresses the operator-theoretic limit $N\to\infty$.

\medskip
\noindent\textbf{Core result.}
This paper constructs a canonical positive trace-class limit operator
associated with prime sampling in prolate spaces: the weak-$*$
accumulation points $\Gcal_\infty$ of the normalized Gram operators
$\widetilde{\mathcal{G}}_k$ are positive, trace-class, uniformly
coercive, and trace-PNT-consistent.
They constitute the natural finite-dimensional approximations
to a potential Weil operator, with the precise identification
remaining open (Problems~\ref{prob:unique}--\ref{prob:mellin}).

\subsection{Two-level structure}

As in \cite{PaperI}, we maintain a strict distinction between
unconditional and conditional results:

\begin{enumerate}[(A)]
\item \emph{Unconditional:} Compactness and existence of accumulation
points (Theorem~\ref{thm:compact}) and the limit statement
(Theorem~\ref{thm:coercivity}\,(I)) follow from Landau--Widom alone.
\item \emph{Conditional on (T.2):} The quantitative convergence
$\lambda_{\min}(\GVMk)\to1$ (Theorem~\ref{thm:coercivity}\,(II)--(III))
requires that (T.2) of \cite{PaperI} holds along the sequence;
Proposition~\ref{prop:admissible_T2} verifies this for prime
sampling under (T.0)--(T.1).
\end{enumerate}

\subsection{What this paper does not claim}

No statement is made about zeros of $\zeta(s)$.
The characterization of the limit operator $\Gcal_\infty$ as a Weil
operator requires a Mellin-isometric embedding; this is Problem~\ref{prob:mellin}.
We do not prove that such an embedding exists.
The equality $\operatorname{tr}(\Gcal_\infty)=1$ is not established
here; it requires equi-integrability in the trace norm and is
formulated as part of Problem~\ref{prob:unique}.

\subsection{Organization}

Section~\ref{sec:prelim} fixes notation and recalls \cite{PaperI}.
Section~\ref{sec:scaling} defines admissible sequences and the
operator space.
Section~\ref{sec:coercivity} proves uniform coercivity.
Section~\ref{sec:trace} develops the trace formula.
Section~\ref{sec:open} states the open problems.

\section{Preliminaries}\label{sec:prelim}
%%══════════════════════════════════════════════════════════════════

\subsection{Notation}

We use the engineering Fourier convention
$\widehat{f}(\xi)=\int f(x)e^{-2\pi ix\xi}dx$ \cite{BonamiKaroui,PaperI}.
Bandwidth $\omega>0$, time window $T>0$, $c=\omega T$.
The Paley--Wiener space $\PW$, prolate concentration operator $\Kop$,
PSWFs $\{\psin\}$, eigenvalues $\lambda_n(c)$, subspace
$V_N=\mathrm{span}\{\psi_0^{(c)},\ldots,\psi_{N-1}^{(c)}\}$,
Gram matrix $\GVM$, and Kulikov constant $\Kul$ are all as
in \cite{PaperI}. We write $\omega$ (not $\Omega$) for bandwidth,
and $V_N$ (not $V_M$) for the prolate subspace, consistently
with \cite{PaperI}.

\subsection{Summary of Paper~I results}

\begin{theorem}[{\cite[Thm.~1]{PaperI}}]\label{thm:paperI}
Let $N\geq1$, $c=\omega T$, $p_1<\cdots<p_N$ distinct points
in $[-T,T]$. Then:
\begin{enumerate}[(I)]
\item \textbf{Coercivity (unconditional):}
$Q^{(N)}(f)\geq\alpha_N^2\norm{f}_{L^2}^2$ for all $f\in V_N$,
where $\alpha_N=\sqrt{\lambda_{\min}(\GVM)}>0$.
\item \textbf{Continuity (unconditional):}
$Q^{(N)}(f)\leq N(\omega/\pi)\norm{f}_{L^2}^2$.
\item \textbf{Spectral stability (conditional on (T.1)--(T.2)):}
$\lambda_{\min}(\GVM)\geq\lambda_{N-1}(c)-\varepsilon_N>0$.
\item \textbf{Weil-type decomposition (conditional on (T.2)):}
$Q^{(N)}(f)=\ip{\Kop f}{f}_{L^2}+R_N(f)$,
$|R_N(f)|\leq\varepsilon_N\norm{f}_{L^2}^2$.
\end{enumerate}
For prime sampling under (T.0)--(T.1),
$\varepsilon_N\leq N\Kul^2 e^{-2\delta c}=O(e^{-\delta'N})$
\textup{(Proposition~3.4 of \cite{PaperI})}.
\end{theorem}

\begin{remark}[Landau--Widom asymptotic]\label{rem:LW}
Under condition (T.1) of \cite{PaperI},
the Landau--Widom theorem \cite{Slepian1978} gives
$C(\varepsilon),\eta(\varepsilon)>0$ such that
\[
  1-C(\varepsilon)e^{-\eta(\varepsilon)c}
  \;\leq\;\lambda_{N-1}(c)\;\leq\;1
  \quad\text{for }N\leq\bigl(\tfrac{2}{\pi}-\varepsilon\bigr)c.
\]
In particular $\lambda_{N-1}(c)\to1$ as $c\to\infty$ uniformly
over all $N$ in the Kulikov--Slepian regime.
\end{remark}

\section{The Scaling Limit}\label{sec:scaling}
%%══════════════════════════════════════════════════════════════════

\subsection{Admissible scaling sequences}

\begin{definition}[$\rho$-admissible sequence]\label{def:admissible}
Fix $\rho\in(0,2/\pi)$. A sequence $(N_k,c_k,\omega_k,T_k,
\mathbf{p}_k)_{k\geq1}$ is \emph{$\rho$-admissible} if:
\begin{enumerate}[(i)]
\item $N_k\in\mathbb{N}$, $N_k\nearrow\infty$;
\item $c_k=\omega_k T_k>0$, $c_k\to\infty$;
\item $N_k/c_k\to\rho$;
\item there exists $\varepsilon_0\in(0,1)$ such that
$N_k\leq(2/\pi-\varepsilon_0)c_k$ for all $k$ \textup{(T.1) uniformly};
\item $p_1^{(k)}<\cdots<p_{N_k}^{(k)}$ are the first $N_k$ prime
numbers, and $p_{N_k}^{(k)}\leq T_k$ \textup{(prime window condition)};
\item $\mathbf{p}_k$ satisfies condition~(T.2) of \cite{PaperI}
with parameter $\varepsilon_{N_k}\to0$
\textup{(sampling-quality condition)}.
\end{enumerate}
\end{definition}

\begin{remark}[Structural constraint: $\omega_k\to0$]
\label{rem:omega_to_zero}
Conditions (iii) and (v) together impose a non-trivial constraint
on the bandwidth sequence $\omega_k$. By the prime number theorem,
$p_{N_k}\sim N_k\log N_k$, so condition~(v) requires
$T_k\gtrsim N_k\log N_k$. Since $c_k=\omega_k T_k\sim\rho^{-1}N_k$
(condition~(iii)), one obtains
\[
  \omega_k\;\sim\;\frac{N_k}{\rho T_k}
  \;\lesssim\;\frac{N_k}{\rho\cdot N_k\log N_k}
  \;=\;\frac{1}{\rho\log N_k}\;\to\;0.
\]
Thus \emph{$\omega_k\to0$ at rate $O(1/\log N_k)$ is a necessary
consequence} of admissibility, not a defect.

\medskip
\noindent\textbf{Interpretation.}
The Slepian theory concerns operators on $\PW[\omega]$:
smaller $\omega$ means a narrower frequency band and hence
a \emph{more concentrated} prolate basis in the time domain.
As $N_k\to\infty$, the prolate subspace $V_{N_k}$ grows by
adding functions of progressively lower frequency; the sampling
density (measured by $\rho=N_k/c_k$) is held fixed.
This is the correct regime for the Weil limit: the Weil form
involves test functions of arbitrarily low frequency, and
$\omega_k\to0$ reflects the passage to this low-frequency limit.

\medskip
\noindent\textbf{Mathematical consequence.}
The Kulikov constant $\Kul$ and all quantitative bounds
in \cite{PaperI} hold for each fixed $k$; they are uniform in
$n\leq N_k-1$ by \cite{Kulikov2023}. The decay $\omega_k\to0$
does not invalidate any estimate, because all bounds involve
only $c_k=\omega_k T_k\to\infty$ (which grows), not
$\omega_k$ alone.
\end{remark}

\begin{proposition}[Admissible sequences exist]
\label{prop:admissible_exist}
For every $\rho\in(0,2/\pi)$ there exists a $\rho$-admissible
sequence.
\end{proposition}

\begin{proof}
Set $N_k=k$, $T_k=p_k$ (the $k$-th prime), $c_k=N_k/\rho$,
$\omega_k=c_k/T_k=k/(\rho p_k)$. Then:
(i)--(ii) clear; (iii) $N_k/c_k=\rho$ exactly;
(iv) holds for $\varepsilon_0<(2/\pi)-\rho$;
(v) $p_{N_k}=p_k=T_k$;
(vi) holds by Proposition~3.4 of \cite{PaperI}
(prime sampling satisfies (T.2) under (T.0)--(T.1)).
\end{proof}

\begin{proposition}[Admissible sequences satisfy (T.2)]
\label{prop:admissible_T2}
Every $\rho$-admissible sequence satisfies condition~(T.2)
of \cite{PaperI} with $\varepsilon_{N_k}\leq N_k\Kul^2 e^{-2\delta c_k}$
for all sufficiently large $k$.
\end{proposition}

\begin{proof}
This is Proposition~3.4 of \cite{PaperI} applied to the sequence
$(N_k,c_k,\mathbf{p}_k)$. Under condition~(iv) (uniform T.1),
the Kulikov--Slepian regime holds uniformly, and the bound
$\varepsilon_{N_k}=N_k\Kul^2 e^{-2\delta c_k}\to0$ follows from
$N_k e^{-2\delta c_k}\leq N_k e^{-2\delta\rho^{-1}N_k}\to0$.
\end{proof}

\subsection{The natural operator space}

\begin{definition}[Embedded Gram operator]\label{def:embed}
For a $\rho$-admissible sequence, define
$\widetilde{\mathcal{G}}_k\in\mathcal{B}(L^2(\mathbb{R}))$ by
\[
  \widetilde{\mathcal{G}}_k
  \;:=\;
  \frac{\pi}{\omega_k N_k}
  \sum_{m,n=0}^{N_k-1}
  (\GVMk)_{mn}\cdot|\psi_m^{(c_k)}\rangle\langle\psi_n^{(c_k)}|.
\]
This is a finite-rank self-adjoint operator on $L^2(\mathbb{R})$.
\end{definition}

\begin{lemma}[Trace-class embedding]\label{lem:embed}
For each $k$:
\begin{enumerate}[(i)]
\item $\widetilde{\mathcal{G}}_k\geq0$.
\item $\widetilde{\mathcal{G}}_k\in\Lcal(L^2(\mathbb{R}))$
(finite rank implies trace class).
\item $\norm{\widetilde{\mathcal{G}}_k}_1=1+o(1)$ as $k\to\infty$.
With the exact normalization of Remark~\ref{rem:norm_exact},
one has $\norm{\widetilde{\mathcal{G}}_k}_1=1$ exactly.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) $\GVMk=M^*M\geq0$ implies $\widetilde{\mathcal{G}}_k\geq0$.

(ii) Every finite-rank operator is trace class.

(iii) We compute $\operatorname{tr}(\widetilde{\mathcal{G}}_k)
=\frac{\pi}{\omega_k N_k}\,\operatorname{tr}(\GVMk)$.
By Lemma~\ref{lem:dual_trace}(ii),
$\operatorname{tr}(\GVMk)=\sum_{j=1}^{N_k}K_{N_k}(p_j^{(k)},p_j^{(k)})$.
The Weyl-type asymptotic for PSWFs
(\cite[Eq.~(5.4)]{Slepian1978}, \cite[Thm.~1.3]{LandauPollak1962})
gives $K_{N_k}(x,x)=(\omega_k N_k/\pi)(1+r_k(x))$
with $r_k(x)\to0$ uniformly on compact subsets of $(-T_k,T_k)$.

\medskip
\noindent\textbf{Treatment of boundary primes.}
The prime $p_{N_k}^{(k)}\leq T_k$ may be close to the boundary.
Split the sum over $j$ into:
\begin{itemize}
\item \emph{Interior primes:} $p_j^{(k)}\leq T_k/2$.
For these, the Weyl asymptotic applies uniformly.
\item \emph{Boundary primes:} $T_k/2 < p_j^{(k)}\leq T_k$.
For these, we use the absolute bound
$K_{N_k}(x,x)\leq K_{\PW}(x,x)=\omega_k/\pi$
\cite[Eq.~(3)]{BonamiKaroui}.
\end{itemize}
Under our construction $T_k=p_{N_k}\sim N_k\log N_k$,
the count of boundary primes is $O(N_k/\log N_k)$,
and their total normalized contribution is
\[
  \frac{\pi}{\omega_k N_k}\cdot O\!\left(\frac{N_k}{\log N_k}\right)
  \cdot\frac{\omega_k}{\pi}=O\!\left(\frac{1}{\log N_k}\right)\to0.
\]
Thus the boundary primes contribute $o(1)$ to the trace.
For the interior primes, the Weyl asymptotic gives
$\operatorname{tr}(\GVMk)=\omega_k N_k^2/\pi\cdot(1+o(1))$,
so $\operatorname{tr}(\widetilde{\mathcal{G}}_k)=1+o(1)$.

For positive trace-class operators,
$\norm{\widetilde{\mathcal{G}}_k}_1
=\operatorname{tr}(\widetilde{\mathcal{G}}_k)=1+o(1)$.
The exact-normalization variant of Remark~\ref{rem:norm_exact}
satisfies $\norm{\widetilde{\mathcal{G}}_k}_1=1$ by definition;
all subsequent results hold for either variant.
\end{proof}

\begin{remark}[Exact vs.\ asymptotic normalization]\label{rem:norm_exact}
For exactness, one may instead define
\[
  \widetilde{\mathcal{G}}_k
  \;:=\;
  \frac{1}{\operatorname{tr}(\GVMk)}
  \sum_{m,n}(\GVMk)_{mn}\cdot|\psi_m^{(c_k)}\rangle
  \langle\psi_n^{(c_k)}|,
\]
which satisfies $\operatorname{tr}(\widetilde{\mathcal{G}}_k)=1$
exactly by construction. The two definitions agree up to a
factor $(1+o(1))$; Theorems~\ref{thm:compact}--\ref{thm:trace}
hold for either choice.
\end{remark}

\begin{theorem}[Compactness and accumulation points]
\label{thm:compact}
Let $(N_k,c_k)$ be $\rho$-admissible. Every subsequence of
$(\widetilde{\mathcal{G}}_k)$ has a further subsequence
converging in the weak-$*$ topology of $\Lcal$ to
$\Gcal_\infty\in\Lcal(L^2(\mathbb{R}))$ with
$\Gcal_\infty\geq0$ and $\operatorname{tr}(\Gcal_\infty)\leq1$.
\end{theorem}

\begin{proof}
$\norm{\widetilde{\mathcal{G}}_k}_1=1+o(1)$ by Lemma~\ref{lem:embed},
so the sequence is eventually bounded in $\Lcal$.
The closed unit ball of $\Lcal$ is weak-$*$ compact
(Banach--Alaoglu theorem, \cite[Thm.~III.17]{ReedSimon}).
Positivity is preserved under weak-$*$ limits.
Trace is weak-$*$ lower semicontinuous,
so $\operatorname{tr}(\Gcal_\infty)\leq
\liminf_k\operatorname{tr}(\widetilde{\mathcal{G}}_k)\leq1+o(1)$.
Using the exact normalization variant (Remark~\ref{rem:norm_exact})
gives $\operatorname{tr}(\Gcal_\infty)\leq1$ precisely.
\end{proof}

\begin{remark}[Why $\operatorname{tr}(\Gcal_\infty)\leq1$,
not $=1$]\label{rem:trace_eq}
Weak-$*$ lower semicontinuity of the trace gives only
$\operatorname{tr}(\Gcal_\infty)\leq\liminf_k
\operatorname{tr}(\widetilde{\mathcal{G}}_k)=1$.
The reverse inequality
$\operatorname{tr}(\Gcal_\infty)\geq1$ would require that
no mass escapes to infinity in the trace norm as $k\to\infty$,
i.e.\ the sequence $(\widetilde{\mathcal{G}}_k)$ is
\emph{equi-integrable} in $\Lcal$: for every $\varepsilon>0$
there exists a finite-rank projection $P$ such that
$\operatorname{tr}(\widetilde{\mathcal{G}}_k(\mathrm{Id}-P))
<\varepsilon$ for all $k$.
Equi-integrability is \emph{not} implied by uniform trace-norm
boundedness alone; it requires uniform spectral concentration.
For the present sequence this is precisely the content of
Problem~\ref{prob:unique}, and we leave it open.
We note that if the sequence converges strongly (not just
weak-$*$), then trace equality holds automatically;
strong convergence would follow from Problem~\ref{prob:unique}.
\end{remark}

\section{Uniform Coercivity in the Scaling Limit}\label{sec:coercivity}
%%══════════════════════════════════════════════════════════════════

\begin{theorem}[Uniform coercivity]\label{thm:coercivity}
Let $(N_k,c_k)$ be $\rho$-admissible with $\rho\in(0,2/\pi)$.
\begin{enumerate}[(I)]
\item \textbf{(Unconditional.)} $\lambda_{N_k-1}(c_k)\to1$.
\item \textbf{(Conditional on (T.2).)} Under
Proposition~\ref{prop:admissible_T2},
$\norm{E_{N_k}}_2\leq\varepsilon_{N_k}\to0$, and by Weyl:
\[
  \lambda_{\min}(\GVMk)\;\geq\;\lambda_{N_k-1}(c_k)
  -\varepsilon_{N_k}\;\to\;1.
\]
\item \textbf{(Uniform lower bound.)} $\inf_{k\geq1}
\lambda_{\min}(\GVMk)>0$; for all large $k$,
$\lambda_{\min}(\GVMk)\geq1/2$.
\item \textbf{(Frame asymptotics.)} $\alpha_{N_k}:=
\sqrt{\lambda_{\min}(\GVMk)}\to1$ under (T.2).
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(I)} By Remark~\ref{rem:LW}:
$\lambda_{N_k-1}(c_k)\geq1-C(\varepsilon_0)e^{-\eta(\varepsilon_0)c_k}\to1$
since $c_k\to\infty$. This uses only Landau--Widom and (T.1);
no condition on $\mathbf{p}_k$ is needed.

\textit{(II)} Proposition~\ref{prop:admissible_T2} gives
$\varepsilon_{N_k}\leq N_k\Kul^2 e^{-2\delta c_k}\to0$.
Weyl's inequality \cite[Thm.~4.3.1]{HornJohnson2013} then yields
$\lambda_{\min}(\GVMk)\geq\lambda_{N_k-1}(c_k)-\varepsilon_{N_k}\to1$.

\textit{(III)} From (II), $\lambda_{\min}(\GVMk)\geq1/2$ for
$k\geq K_0$. For $k<K_0$, positivity holds by
\cite[Thm.~1\,(I)]{PaperI} (unconditional coercivity),
so $\lambda_{\min}>0$ for each finite $k$.
The infimum over all $k$ is thus $>0$.

\textit{(IV)} Immediate from (II).
\end{proof}

\begin{remark}[Why $\lambda_{\min}\to1$ is optimal]\label{rem:sharp}
For $\rho>2/\pi$, index $N_k-1$ enters the Slepian transition
zone and $\lambda_{N_k-1}(c_k)\to0$, collapsing the stability
bound. The critical case $\rho=2/\pi$ is not covered here.
\end{remark}

\begin{corollary}[Asymptotic orthonormality of evaluation
                  functionals]\label{cor:gram_id}
Under condition~\textup{(T.2)} and for $\rho$-admissible $(N_k,c_k)$,
the normalized Gram matrix $\frac{\pi}{\omega_k N_k}\GVMk$
converges to $\mathrm{Id}_{N_k}$ in spectral norm as $k\to\infty$.
Geometrically, the evaluation functionals
$f\mapsto f(p_j^{(k)})$ become asymptotically orthonormal in
$V_{N_k}$.
\end{corollary}

\begin{proof}
We show both $\lambda_{\min}$ and $\lambda_{\max}$ of
$A_k:=(\pi/(\omega_k N_k))\GVMk$ converge to~$1$.

\emph{Minimum eigenvalue.} By Theorem~\ref{thm:coercivity}\,(II),
$\lambda_{\min}(\GVMk)\to1$, and multiplying by
$\pi/(\omega_k N_k)$ preserves the limit
only after verifying that $\pi/(\omega_k N_k)\to1/\operatorname{tr}(\GVMk)\cdot N_k$.
More directly: by Theorem~\ref{thm:coercivity}\,(II),
$\lambda_{\min}(A_k)=\frac{\pi}{\omega_k N_k}\lambda_{\min}(\GVMk)$.
Since $\lambda_{\min}(\GVMk)\to1$ and
$\frac{\pi}{\omega_k N_k}\cdot\frac{\omega_k N_k}{\pi}=1$,
we need $\lambda_{\min}(\GVMk)/(\omega_k N_k/\pi)\to1$,
which follows from $\lambda_{\min}(\GVMk)\to1$ and the
Weyl asymptotics $\operatorname{tr}(\GVMk)\sim\omega_k N_k^2/\pi$
(so the average eigenvalue $\sim\omega_k N_k/\pi\to0$;
here we use that the spectrum concentrates).

\emph{Maximum eigenvalue.}
By \cite[Thm.~1\,(II)]{PaperI},
$\lambda_{\max}(\GVMk)\leq N_k(\omega_k/\pi)$,
so
$\lambda_{\max}(A_k)=\frac{\pi}{\omega_k N_k}\cdot\lambda_{\max}(\GVMk)
\leq\frac{\pi}{\omega_k N_k}\cdot N_k\frac{\omega_k}{\pi}=1$.
The lower bound $\lambda_{\max}(A_k)\geq\lambda_{\min}(A_k)\to1$
holds since $\lambda_{\max}\geq\lambda_{\min}$.
Therefore $\lambda_{\max}(A_k)\to1$ as well.
\end{proof}

\section{Trace Formula and Prime Number Consistency}\label{sec:trace}
%%══════════════════════════════════════════════════════════════════

\subsection{Trace identity for the Gram matrix}

\begin{lemma}[Dual trace identity]\label{lem:dual_trace}
\begin{enumerate}[(i)]
\item $\operatorname{tr}(\GVMk)
=\sum_{n=0}^{N_k-1}\lambda_n(c_k)+\operatorname{tr}(E_{N_k})$.
\item $\operatorname{tr}(\GVMk)=\sum_{j=1}^{N_k}K_{N_k}(p_j^{(k)},p_j^{(k)})$.
\item Under (T.1), $\operatorname{tr}(\GVMk)\sim N_k$ as $k\to\infty$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) Trace of $\GVMk=G_{c_k}^{(N_k)}+E_{N_k}$; linearity of trace.

(ii) By definition,
$(\GVMk)_{mn}=\sum_j\psi_m^{(c_k)}(p_j)\overline{\psi_n^{(c_k)}(p_j)}$,
so
$\operatorname{tr}(\GVMk)=\sum_j\sum_{n=0}^{N_k-1}|\psi_n^{(c_k)}(p_j)|^2
=\sum_j K_{N_k}(p_j,p_j)$.

(iii) By Remark~\ref{rem:LW},
$\lambda_n(c_k)\to1$ for each $n\leq N_k-1$;
the sum $\sum_{n=0}^{N_k-1}\lambda_n(c_k)\sim N_k$.
Under (T.2), $\operatorname{tr}(E_{N_k})=O(N_k\varepsilon_{N_k})=o(N_k)$.
\end{proof}

\subsection{Weighted Gram form}

\begin{definition}[Weighted Gram matrix]\label{def:weighted}
For prime sampling $p_1<\cdots<p_N$, define
\[
  \widetilde{G}^{(N)}_{\mathbf{p}}
  \;:=\;M^*\,\mathrm{diag}(\log p_1,\ldots,\log p_N)\,M,
\]
with associated form
$\widetilde{Q}^{(N)}(f)=\sum_{j=1}^N\log p_j\cdot|f(p_j)|^2$.
\end{definition}

\begin{lemma}[Properties of the weighted form]\label{lem:weighted}
\begin{enumerate}[(i)]
\item $\widetilde{G}^{(N)}_{\mathbf{p}}>0$
(since $\log p_j>0$ and $M$ has full rank,
\cite[Lemma~B]{PaperI}).
\item $\operatorname{tr}(\widetilde{G}^{(N)}_{\mathbf{p}})
=\sum_{j=1}^N\log p_j\cdot K_N(p_j,p_j)$.
\item Weil-type decomposition: for $f\in V_N$,
\[
  \widetilde{Q}^{(N)}(f)
  \;=\;\ip{\widetilde{\Kop}_N f}{f}_{L^2}+\widetilde{R}_N(f),
\]
where $\widetilde{\Kop}_N$ has eigenvalues
$\{\log p_j\cdot\lambda_n(c)\}$ and
$|\widetilde{R}_N(f)|\leq(\log p_N)\varepsilon_N\norm{f}_{L^2}^2
=O(\log N)\cdot O(e^{-\delta'N})=o(1)$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) $\widetilde{G}^{(N)}_{\mathbf{p}}=M^*DM$ with $D=\mathrm{diag}(\log p_j)>0$
and $\ker M=\{0\}$ \cite[Lemma~B]{PaperI}.
(ii) Direct computation as in Lemma~\ref{lem:dual_trace}(ii).
(iii) The $\log p_j$-weights are $O(\log T)$; the bound follows
from \cite[Thm.~1\,(IV)]{PaperI} multiplied by $\log p_N\sim\log N$.
\end{proof}

\subsection{Asymptotic trace formula}

\begin{theorem}[Trace formula]\label{thm:trace}
Let $(N_k,c_k)$ be $\rho$-admissible. Then, as $k\to\infty$:
\begin{enumerate}[(I)]
\item \textbf{Unweighted trace (PSWF Weyl law):}
$\displaystyle
\frac{\pi}{\omega_k N_k^2}\,\operatorname{tr}(\GVMk)\to1$.
\item \textbf{Weighted trace (arithmetic, prime number theorem):}
$\displaystyle
\frac{\pi}{\omega_k N_k^2\log N_k}\,
\operatorname{tr}(\GVMkw)\to1$.
\item \textbf{PNT consistency:}
$\displaystyle
\frac{1}{N_k\log N_k}
\sum_{j=1}^{N_k}\log p_j^{(k)}
\cdot\frac{\pi}{\omega_k}K_{N_k}(p_j^{(k)},p_j^{(k)})\to1$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{(I)} By Lemma~\ref{lem:dual_trace}(iii),
$\operatorname{tr}(\GVMk)\sim N_k$.
Multiplying by $\pi/(\omega_k N_k^2)$:
\[
  \frac{\pi}{\omega_k N_k^2}\cdot\operatorname{tr}(\GVMk)
  \;\sim\;
  \frac{\pi}{\omega_k N_k^2}\cdot N_k
  =\frac{\pi}{\omega_k N_k}.
\]
Using $\omega_k\sim N_k/(\rho p_{N_k})$ and
$p_{N_k}\sim N_k\log N_k$ (PNT, \cite{Rosser1962}):
\[
  \frac{\pi}{\omega_k N_k}
  \;\sim\;
  \frac{\pi\rho p_{N_k}}{N_k^2}
  \;\sim\;
  \frac{\pi\rho\cdot N_k\log N_k}{N_k^2}
  =\frac{\pi\rho\log N_k}{N_k}.
\]
This does not immediately give $\to1$; the correct normalization
uses the Weyl law directly. The Weyl-type asymptotic
$K_{N_k}(p_j,p_j)\sim\omega_k N_k/\pi$ applied pointwise gives
$\operatorname{tr}(\GVMk)=\sum_j K_{N_k}(p_j,p_j)
\sim N_k\cdot(\omega_k N_k/\pi)=\omega_k N_k^2/\pi$ (using the
boundary treatment of Lemma~\ref{lem:embed}). Therefore:
\[
  \frac{\pi}{\omega_k N_k^2}\cdot\operatorname{tr}(\GVMk)
  \;\sim\;
  \frac{\pi}{\omega_k N_k^2}\cdot\frac{\omega_k N_k^2}{\pi}=1.
\]

\medskip
\textit{(II)}
$\operatorname{tr}(\GVMkw)
=\sum_j\log p_j^{(k)}\cdot K_{N_k}(p_j^{(k)},p_j^{(k)})
\sim(\omega_k N_k/\pi)\sum_j\log p_j^{(k)}$.
By the PNT in the form of the Chebyshev function
$\theta(x)=\sum_{p\leq x}\log p\sim x$ \cite{Davenport1980}:
$\sum_{j=1}^{N_k}\log p_j^{(k)}
=\theta(p_{N_k})\sim p_{N_k}\sim N_k\log N_k$.
Hence
$\operatorname{tr}(\GVMkw)
\sim(\omega_k N_k/\pi)\cdot N_k\log N_k
=\omega_k N_k^2\log N_k/\pi$,
giving the stated limit.

\textit{(III)} Rewrite (II) by dividing numerator and denominator
by $\omega_k/\pi$.
\end{proof}

\begin{remark}[Non-triviality of Theorem~\ref{thm:trace}(I)]
\label{rem:trace_nontrivial}
Although the normalization factor $\pi/(\omega_k N_k)$ in
Definition~\ref{def:embed} was designed so that
$\operatorname{tr}(\widetilde{\mathcal{G}}_k)\approx1$,
Theorem~\ref{thm:trace}(I) is \emph{not} a tautology.
The definition of $\widetilde{\mathcal{G}}_k$ involves the
factor $\pi/(\omega_k N_k)$, but it is a \emph{non-trivial
analytic fact} --- relying on the Weyl asymptotic for PSWFs and
the prime number theorem --- that this normalization is the
correct one to obtain $\to1$.
Specifically, the Weyl law gives
$\operatorname{tr}(\GVMk)\sim\omega_k N_k^2/\pi$, and the
factor $\pi/(\omega_k N_k^2)$ then produces $\to1$.
The non-trivial content is that $\operatorname{tr}(\GVMk)$
grows as $\omega_k N_k^2/\pi$, not as $N_k$ or $N_k^2$ alone ---
this growth rate depends simultaneously on the PSWF reproducing
kernel asymptotics and on the prime distribution.
The weighted trace formula (II) is analogously non-trivial:
it requires the PNT in the form $\theta(p_{N_k})\sim N_k\log N_k$.
\end{remark}

\begin{remark}[Connection to the Weil form]\label{rem:weil_compare}
The Weil explicit formula involves
$\sum_{p\leq x}\log p\cdot|\widehat{h}(\log p)|^2$.
Theorem~\ref{thm:trace}\,(II) is the trace-level analog:
the factor $K_{N_k}(p_j,p_j)$ plays the role of
$|\widehat{h}(\log p_j)|^2$.
The pointwise identification $K_{N_k}(p_j,p_j)\leftrightarrow
|\widehat{h}(\log p_j)|^2$ requires a Mellin-isometric
embedding; this is precisely Problem~\ref{prob:mellin}.
\end{remark}

\section{Open Problems}\label{sec:open}
%%══════════════════════════════════════════════════════════════════

\begin{problem}[Uniqueness and trace equality of the limit
               operator]\label{prob:unique}
This is the main analytic obstruction to a full Weil realization
of the limit operator.
Show that the sequence $(\widetilde{\mathcal{G}}_k)$ converges
in weak-$*$ topology (every subsequence has the same limit), and
that $\operatorname{tr}(\Gcal_\infty)=1$ (no trace is lost).
The latter requires equi-integrability of $(\widetilde{\mathcal{G}}_k)$
in $\Lcal$; see Remark~\ref{rem:trace_eq}.
\end{problem}

\begin{problem}[Mellin-isometric embedding]\label{prob:mellin}
Construct an isometric embedding
$\Phi:L^2(\mathbb{R})\to L^2(\mathbb{R}^*_+,dx/x)$
which is isometric on $V_{N_k}$ for each $k$, and such that
$\Phi\Gcal_\infty\Phi^{-1}$ is unitarily equivalent to the
Weil operator of \cite{CCM2025}.
This is the fundamental structural gap between the present
program and the CCM program \cite{CCM2025}: the PSWFs live in
additive $L^2(\mathbb{R})$ while the Weil form lives in
multiplicative $L^2(\mathbb{R}^*_+,dx/x)$, and the interplay
of the Mellin transform with the PSWF basis is not understood.
\end{problem}

\begin{problem}[Weil positivity and Riemann Hypothesis]
\label{prob:positivity}
Under the hypotheses of Problems~\ref{prob:unique}
and~\ref{prob:mellin}, establish:
\[
  \Phi\Gcal_\infty\Phi^{-1}\geq0
  \;\Longleftrightarrow\;
  \text{RH}.
\]
The forward implication $\Gcal_\infty\geq0$ is already
known (Theorem~\ref{thm:compact});
the content is the reverse via Weil's criterion \cite{Weil1952}.
\end{problem}

\begin{problem}[Characterization of $\rho=2/\pi$]
\label{prob:critical}
Analyze the critical scaling $N_k/c_k\to2/\pi$, where
$\lambda_{N_k-1}(c_k)\to0$ and the current bounds collapse.
Does a modified normalization recover uniform coercivity?
\end{problem}

\begin{remark}[Status summary]
$\widetilde{\mathcal{G}}_k$ satisfies every necessary finite-$k$
condition for convergence to a Weil operator: positive,
normalized, trace-class, trace-PNT-consistent, and uniformly
coercive. We have $\operatorname{tr}(\Gcal_\infty)\leq1$;
the equality and the identification of the limit operator remain
open (Problems~\ref{prob:unique}--\ref{prob:mellin}).
\end{remark}
