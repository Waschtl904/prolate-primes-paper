\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to a single
open problem: the intermediate-regime
Hilbert--Schmidt estimate
$\Sigma_{\mathrm{int}} = o(1)$
for $M(c)\sim\beta c$.
This paper identifies the core
analytic obstruction as a
\emph{critical-density Toeplitz
energy problem} and develops
the analytic toolkit required
for the reduction to this formulation.

The central object is the oscillatory
kernel family
\[
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi,
  \quad 1\le k\le c^{\theta_0},
\]
and the associated sum
$\Sigma_{\mathrm{model}}(c)
:= \sum_{k=1}^{\lfloor c^{\theta_0}\rfloor}
|I_c(k)|^2$.
The Dirichlet-kernel reformulation
(\S\ref{sec:model}) and mesoscopic
rescaling identify $\Sigma_{\mathrm{model}}$
with the critical Toeplitz energy
functional
$\Tcc(\psi) := \ip{\Dc\psi}{\psi}$
at the double-scaling
regime $K(c)\cdot h = 1$.

The WKB phase extraction
(Lemmas~\ref{lem:WKB-phase},
\ref{lem:phase-nondeg})
identifies $\theta(\xi)$ explicitly,
proves $\theta'\ne 0$ and
$\theta''\ne 0$ for $\xi\ne 0$,
and establishes that both naive
van~der~Corput
(Corollary~\ref{cor:curvature-available})
and the Weyl/UDPL/exponent-pair
combination
(Proposition~\ref{prop:W4-decoupling})
are insufficient to close the
$\Sigma_{\mathrm{model}}=o(1)$ claim:
the obstruction is not
\emph{analytic weakness} but
\emph{lack of cancellation after
phase freezing}
(\S\ref{sec:toeplitz},
Remark~\ref{rem:core-thesis}).

Section~\ref{sec:toeplitz} develops
the key structural reinterpretation:
$\Sigma_{\mathrm{model}}$ is a
\emph{superposition of shifted
Fourier samplings} of $|\widehat{\psi}|^2$
along the deformed grid
$\{k\alpha_0(\xi_0)/c^{\theta_0}\}$
(Remark~\ref{rem:sampling-rewrite}).
This perspective yields a concrete
sufficient condition for
(T\textsuperscript{*}):
Riemann--Lebesgue decay in the
$\alpha$-direction, derived from
a second Dirichlet-kernel identity
(Proposition~\ref{prop:alpha-oscillation}).

The main structural output is the
\emph{Reduction Theorem}
(Theorem~\ref{thm:reduction}):
\[
  \mathrm{(T^*)} + \mathrm{(L)}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}}=o(1),
  \qquad
  \mathrm{(C)}
  \;\Longrightarrow\;
  \text{H1 fails},
\]
where~(L) (Lipschitz regularity,
Paper~XVI), (C) (local Weyl law,
Paper~XVII), and~(T\textsuperscript{*})
(PSWF Toeplitz decay)
are the three independent open
targets of the programme.
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Inherited structure from Paper~XIV}

Paper~XIV established the following
reduction chain:
\begin{equation}\label{eq:chain-XIV}
  \underbrace{(C) + (D\text{-strong})}
  _{\text{Paper XIV target}}
  \;\Rightarrow\;
  S_N^{(M(c))}(c)\sim c_0\beta > 0
  \;\Rightarrow\;
  \text{H3 fails}
  \;\Rightarrow\;
  \text{H1 fails}.
\end{equation}
The HS decomposition
\begin{equation}\label{eq:HS-decomp}
  \norm{\AcM - D^{(M)}}_{\mathrm{HS}}^2
  = \Sigma_{\mathrm{near}}
  + \Sigma_{\mathrm{int}}
  + \Sigma_{\mathrm{far}}
\end{equation}
satisfies:
$\Sigma_{\mathrm{near}} = O(1/c) = o(1)$
(proved, Paper~XIV, conditional on~(C));
$\Sigma_{\mathrm{far}} = o(1)$
(accessible via stationary phase);
and the sole remaining obstruction:
\begin{equation}\label{eq:int-target}
  \Sigma_{\mathrm{int}}
  = \sum_{\substack{n\ne m\\
      C<|n-m|\le c^{\theta_0}}}
    \left|(\AcM)_{nm}\right|^2
  \stackrel{?}{=} o(1),
  \qquad M(c)\sim\beta c.
\end{equation}

\subsection{Three levels of decay}

\begin{remark}[Three levels of $o(1)$]
\label{rem:three-levels}
The symbol $o(1)$ appears at three
distinct levels throughout this paper;
these are \emph{not} interchangeable:
\begin{enumerate}[\rm(i)]
\item \emph{Pointwise decay}:
  $|(\AcM)_{nm}| = o(1/c)$
  for each fixed pair $(n,m)$
  with $|n-m|\ge C$.
  This is (D-weak); it does not
  imply HS-summability over
  $O(c^2)$ pairs.
\item \emph{HS-summated decay}:
  $\Sigma_{\mathrm{near}}=o(1)$,
  $\Sigma_{\mathrm{far}}=o(1)$,
  $\Sigma_{\mathrm{int}}=o(1)$.
  This is the operative level
  for (D-strong).
\item \emph{Uniform in $\theta_0$}:
  the crude bound
  $O(c^{\theta_0-1})=o(1)$
  holds for each fixed $\theta_0<1$
  but \emph{not} uniformly as
  $\theta_0\to 1^-$.
\end{enumerate}
\end{remark}

\subsection{The model reduction}

The Gram kernel takes the form
\begin{equation}\label{eq:gram-kernel}
  (\AcM)_{nm}
  = \int_{-c_0}^{c_0}
    \widehat{\Phi_n^{(c)}}(\xi)\,
    \overline{\widehat{\Phi_m^{(c)}}(\xi)}
    \,d\xi.
\end{equation}
Under assumption~(C) and the Lipschitz
bound
\begin{equation}\label{eq:Lipschitz-main}
  \norm{\widehat{\Phi_{n+k}^{(c)}}-\widehat{\Phi_n^{(c)}}}_{L^2([-c_0,c_0])}
  = O(k/c),
  \qquad |k|\le c^{\theta_0},
  \tag{L}
\end{equation}
the amplitude correction
$A_n\overline{A_m} - w$
contributes $O(k/c)$ per entry,
which is HS-summable.
The leading term reduces to
the model kernel:
\begin{equation}\label{eq:model-reduction}
  (\AcM)_{nm}
  \approx I_c(|n-m|)
  + O(|n-m|/c),
  \qquad
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi.
\end{equation}
Thus $\Sigma_{\mathrm{int}}=o(1)$
follows from
$\Sigma_{\mathrm{model}}(c)=o(1)$
once~\eqref{eq:Lipschitz-main} holds.
Condition~(L) is the subject of
Paper~XVI; the local Weyl law~(C)
is the subject of Paper~XVII.

%%═══ 2. THE MODEL PROBLEM
\section{The Intermediate-Regime
  Model Problem}
\label{sec:model}

\begin{problem}[Intermediate-regime
  kernel summation;
  model problem]
\label{prob:model-intermediate}

Let $\theta\in C^\infty([-c_0,c_0])$
satisfy $\theta'\ne 0$,
and let $w\in C^\infty([-c_0,c_0])$,
$w>0$, $\int w\,d\xi=1$.

Define
\begin{equation}\label{eq:Ic-def}
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi,
  \qquad k\in\mathbb{Z}_{\ge 1}.
\end{equation}
Fix $\theta_0\in(0,1)$,
$K(c):=\lfloor c^{\theta_0}\rfloor$.

\medskip\noindent
\textbf{Open Target (Model-$\Sigma$).}
\begin{equation}\label{eq:model-sigma}
  \Sigma_{\mathrm{model}}(c)
  := \sum_{k=1}^{K(c)}|I_c(k)|^2
  \;\stackrel{?}{=}\; o(1).
\end{equation}
This is \emph{not} proved here;
reduced to~(T\textsuperscript{*})
in \S\ref{sec:toeplitz}.

\medskip\noindent
\textbf{Kernel reformulation.}
\begin{equation}\label{eq:kernel-rewrite}
  \Sigma_{\mathrm{model}}(c)
  = \int\!\int
    w(\xi)\,w(\eta)\,
    F_c(\xi,\eta)
    \,d\xi\,d\eta,
\end{equation}
\begin{equation}\label{eq:Fc}
  F_c(\xi,\eta)
  := \sum_{k=1}^{K(c)}
     e^{ik(\theta(\xi)-\theta(\eta))/c}.
\end{equation}
This is a \emph{bilinear form};
no separation into pointwise bounds
is admissible.

\medskip\noindent
\textbf{Two-scale structure.}
\begin{itemize}[leftmargin=*]
\item \textbf{Off-diagonal}
  ($|\xi-\eta|\gg c^{1-\theta_0}$):
  IBP gives $|I_c(k)|=O(c/k)$;
  $\Sigma_{\mathrm{off}}
  =O(c^{3\theta_0-2})=o(1)$
  for $\theta_0<2/3$.
\item \textbf{Diagonal strip}
  ($|\xi-\eta|\lesssim c^{1-\theta_0}$):
  strip area $\sim c^{2(1-\theta_0)}$,
  $|F_c|\le c^{\theta_0}$:
  contributes $O(1)$.
  This is the critical Toeplitz regime.
\end{itemize}

\medskip\noindent
\textbf{Mesoscopic rescaling.}
Fix $\xi_0$, set
$\xi=\xi_0+c^{1-\theta_0}u$,
$\eta=\xi_0+c^{1-\theta_0}v$:
\begin{equation}\label{eq:meso-F}
  F_c(u,v)
  = D_{K(c)}\!\big(
       h\,\theta'(\xi_0)(u-v)
     \big),
  \quad
  h:=c^{-\theta_0},
  \quad K(c)\cdot h = 1.
\end{equation}
This is the \emph{double-scaling
Szeg\H{o} regime}: neither
the classical Szeg\H{o} theorem
nor the semiclassical limit applies.
\end{problem}

%%═══ 3. CRITICAL MESOSCOPIC ORTHOGONALITY
\section{Critical Mesoscopic
  Orthogonality}
\label{sec:S3}

\begin{definition}[Toeplitz--Dirichlet
  operator; Critical Toeplitz Energy]
\label{def:TD-operator}
For fixed $\xi_0$ and $c$:
\begin{equation}\label{eq:Dc-def}
  (\Dc f)(u)
  := \int_{\mathbb{R}}
     D_{K(c)}\!\big(h\,\alpha_0(u-v)\big)
     \psi(v)\,f(v)\,dv,
\end{equation}
where $\alpha_0 := \theta'(\xi_0)$,
$h = c^{-\theta_0}$,
$K(c) = \lfloor c^{\theta_0}\rfloor$.
The \emph{critical Toeplitz energy
functional} is:
\begin{equation}\label{eq:Tc-def}
  \Tcc(\psi)
  := \ip{\Dc\psi}{\psi}
  = \sum_{k=1}^{K(c)}
    \left|\widehat{\psi}\!\left(
      \frac{k\alpha_0}{c^{\theta_0}}
    \right)\right|^2,
\end{equation}
so
$\Sigma_{\mathrm{strip}}(c)
= c^{2(1-\theta_0)}\Tcc(\psi)$.
\end{definition}

\begin{proposition}[Critical
  mesoscopic saturation
  (frozen $\xi_0$)]
\label{prop:S3}
For fixed $\xi_0$ and frozen
$\alpha_0 = \theta'(\xi_0)$:

\noindent\textbf{(i)}
$\Sigma_{\mathrm{strip}}(c)
= c^{2(1-\theta_0)}
\sum_{k=1}^{K(c)}
|\widehat{\psi}(k\alpha_0/c^{\theta_0})|^2$.

\noindent\textbf{(ii)}
$\Tcc(\psi)/c^{\theta_0}
\to C(\psi,\alpha_0)
:= \norm{\psi}_{L^2}^2/|\alpha_0| > 0$.

\noindent\textbf{(iii)}
\begin{equation}\label{eq:strip-final}
  \Sigma_{\mathrm{strip}}(c)
  = c^{2-\theta_0}\cdot C(\psi,\alpha_0)
    + o(c^{2-\theta_0})
  = \Theta(c^{2-\theta_0}).
\end{equation}

\noindent\emph{Important scope.}
This is the \textbf{frozen-$\xi_0$}
behaviour at a single basepoint.
Proposition~\ref{prop:S3}
does \emph{not} describe the
behaviour after integration over
$\xi_0\in[-c_0,c_0]$:
that integration is the subject
of Proposition~\ref{prop:model-limit},
and cancellation between different
values of $\alpha_0(\xi_0)$ can
only arise there.
\end{proposition}

\begin{proof}
Riemann sum with mesh
$|\alpha_0|/c^{\theta_0}\to 0$
applied to $|\widehat{\psi}|^2\in L^1\cap C^0$.
Part~(iii): $c^{2(1-\theta_0)}\cdot c^{\theta_0}
= c^{2-\theta_0}$.
\end{proof}

\begin{corollary}[No-go for frozen phase]
\label{cor:S3-negative}
For any fixed $\alpha_0\ne 0$:
$\Sigma_{\mathrm{strip}}(c)
=\Theta(c^{2-\theta_0})\to+\infty$.
No oscillatory mechanism
\emph{pointwise} in $\alpha_0$
produces $o(1)$.
\end{corollary}

\begin{remark}[Scale-balance
  obstruction]
\label{rem:scale-balance}
Weyl/UDPL gives
$|T(c)|=O(c^{3\theta_0/4})$;
strip-area prefactor is
$c^{2(1-\theta_0)}$.
Product:
$c^{2-5\theta_0/4} = \Omega(1)$
for all $\theta_0\in(0,1)$.
Missing input: variation of
$\alpha_0(\xi_0)$ across $\xi_0$.
\end{remark}

\begin{remark}[Two escapes]
\label{rem:two-escapes}
\textbf{Escape~A (global phase
  variation):}
$\xi_0\mapsto\alpha_0(\xi_0)$
strictly monotone
(Lemma~\ref{lem:phase-nondeg}(iv));
integration over $\xi_0$ may
produce cancellation.

\textbf{Escape~B (hidden PSWF
  orthogonality):}
WKB amplitude $w(\xi)$ encodes
PSWF eigenstructure; hidden
cancellation not visible in
the model reduction.

Both require PSWF-specific
structure.
\end{remark}

%%═══ 4. WKB PHASE EXTRACTION
\section{WKB Phase Extraction}
\label{sec:WKB}

\subsection{PSWF ODE and semiclassical
  scaling}

The PSWFs satisfy
\begin{equation}\label{eq:PSWF-ODE}
  \frac{d}{dx}\!\left[
    (1-x^2)\Phi_n^{(c)\prime}
  \right]
  + \left[\chi_n(c)-c^2x^2\right]
    \Phi_n^{(c)} = 0.
\end{equation}
Classical momentum:
\begin{equation}\label{eq:momentum}
  p(\zeta;\alpha)
  :=
  \sqrt{\frac{\lambda(\alpha)-\zeta^2}
             {1-\zeta^2}},
  \quad
  |\zeta|<\sqrt{\lambda(\alpha)}.
\end{equation}

\subsection{Phase function}

\begin{lemma}[WKB phase extraction]
\label{lem:WKB-phase}
\begin{equation}\label{eq:theta-def}
  \theta(\xi)
  := c\int_0^{\xi/c}
     p(\zeta;\alpha)\,d\zeta,
  \qquad
  w(\xi)
  := \frac{|A(\xi/c;\alpha)|^2}
          {p(\xi/c;\alpha)}.
\end{equation}
\end{lemma}

\subsection{Non-degeneracy}

\begin{lemma}[Phase non-degeneracy]
\label{lem:phase-nondeg}
For $\alpha\in(0,1)$,
$|\xi|\le c_0$:

\noindent\textbf{(i)}
$\theta'(\xi) = p(\xi/c;\alpha) > 0$.

\noindent\textbf{(ii)}
For $\xi\ne 0$:
$\theta''(\xi)
= \frac{-(\xi/c)(1-\lambda(\alpha))}
       {c(1-(\xi/c)^2)^{3/2}
         \sqrt{\lambda(\alpha)-(\xi/c)^2}}
\ne 0$.

\noindent\textbf{(iii)}
$|\theta''(\xi)|\ge c_*|\xi|/c$
for $|\xi|\ge\varepsilon$;
$\theta''(0)=0$, $\theta'''(0)\ne 0$.

\noindent\textbf{(iv) Global variation:}
$\xi_0\mapsto\alpha_0(\xi_0)
:=\theta'(\xi_0)$ strictly monotone
with $|\alpha_0'|=|\theta''|\ne 0$.
Structural basis for Escape~A.
\end{lemma}

\begin{proof}
Direct differentiation of $p(\xi/c;\alpha)$.
\end{proof}

\begin{remark}[$\lambda(\alpha)$;
  $\theta''\ne 0$]
\label{rem:lambda-alpha}
$\lambda(\alpha)
= \sin^2(\pi\alpha/2)+O(1/c)$
\cite{Osipov2013}.
\end{remark}

\begin{remark}[Turning point]
\label{rem:turning-zero}
$\theta''(0)=0$; cubic non-degeneracy
$\theta'''(0)\ne 0$ gives Airy control.
\end{remark}

\subsection{van der Corput is
  insufficient}

\begin{corollary}[vdC insufficient]
\label{cor:curvature-available}
$|I_c(k)|\lesssim\sqrt{c/(k\varepsilon)}$,
hence
$\sum_k|I_c(k)|^2\lesssim
(c/\varepsilon)\log c = \Omega(c)$.
Correct tool: bilinear
kernel~\eqref{eq:kernel-rewrite}.
\end{corollary}

%%═══ 5. WEYL EQUIDISTRIBUTION
\section{Weyl Equidistribution and
  Decoupling}
\label{sec:Weyl}

\begin{remark}[Scope of Weyl/UDPL]
\label{rem:what-proved}
Proposition~\ref{prop:W4-decoupling}
controls the \emph{linear} sum
$T(c)=\sum_k\int
e^{ik\theta'(\xi_0)/c^{\theta_0}}
\psi\,d\xi_0$,
not $\Sigma_{\mathrm{model}}
=\sum_k|I_c(k)|^2$.
Gap exponent:
$2-\theta_0 - 3\theta_0/4
= 2-7\theta_0/4 > 0$
for all $\theta_0\in(0,1)$.
\end{remark}

\subsection{Joint $k$/$\xi_0$
  decoupling}

\begin{lemma}[UDPL]
\label{lem:UDPL}
Let $h\in(0,1]$,
$\Phi_h\in C^3([-c_0,c_0])$, with
uniform bounds:
(P1) $|\Phi_h'|\ge\mu_1>0$
for $|\xi_0|\ge\delta$;
(P2) $|\Phi_h''|\ge\mu_2|\xi_0|$
near $0$;
(P3) $|\Phi_h'''|\ge\mu_3>0$.
Then:
\begin{equation}\label{eq:UDPL-opt}
  \left|
    \int e^{i\Phi_h}\psi\,d\xi_0
  \right|
  \le
  C_\psi'\,\mu_1^{-3/5}\mu_3^{-1/5},
  \quad
  \text{uniformly in }h.
\end{equation}
\end{lemma}

\begin{proof}
Split at $\pm\delta$.
Away from $0$: vdC~(P1) gives
$C/(\mu_1\delta)$.
Near $0$: Airy bound~(P3) gives
$C\mu_3^{-1/3}\delta^{2/3}$.
Balance.
\end{proof}

\begin{proposition}[Joint Decoupling
  (W4)]
\label{prop:W4-decoupling}
$|T(c)|
=O(c^{3\theta_0/4+\varepsilon})
=o(c^{\theta_0})$.
\end{proposition}

\begin{proof}
Weyl differencing gives
$|T|^2
\lesssim K^2/H
+(K/H)\sum_{h\le H}|J(h,c)|$
with $|J(h,c)|\le C c^{(1+\theta_0)/2}/h^{1/2}$
(UDPL).
Choose $H=\lfloor c^{(1+\theta_0)/2}\rfloor$:
$|T|^2\lesssim c^{3\theta_0/2}$.
\end{proof}

\begin{remark}[Exponent-pair]
\label{rem:exp-pair}
$A$-process then $B$-process
yields exponent $3\theta_0/4$.
\end{remark}

%%═══ 6. THE CRITICAL TOEPLITZ OBSTRUCTION
\section{The Critical Toeplitz
  Obstruction}
\label{sec:toeplitz}

\begin{remark}[Core thesis]
\label{rem:core-thesis}
\begin{center}
\emph{The obstruction to
$\Sigma_{\mathrm{model}}=o(1)$
is not analytic weakness,
but the absence of cancellation
after phase freezing.}
\end{center}
Every individual tool (vdC, IBP,
Weyl, UDPL) is correctly applied;
the failure is that all these
tools operate
\emph{pointwise in $\xi_0$},
while the necessary cancellation
is \emph{global in $\xi_0$}.
\end{remark}

\begin{remark}[Sampling reinterpretation]
\label{rem:sampling-rewrite}
After the mesoscopic rescaling
and Fourier representation
(Definition~\ref{def:TD-operator}),
one has:
\begin{equation}\label{eq:sampling}
  \Sigma_{\mathrm{model}}(c)
  \approx
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \sum_{k=1}^{K(c)}
  \left|\widehat{\psi}\!\left(
    \frac{k\,\alpha_0(\xi_0)}{c^{\theta_0}}
  \right)\right|^2
  d\xi_0.
\end{equation}
This is \emph{not} a single Riemann sum
(which would saturate at
$\Theta(c^{2-\theta_0})$ per
Proposition~\ref{prop:S3}),
but a \emph{superposition of
shifted Fourier samplings}
of $|\widehat{\psi}|^2$:
for each $\xi_0$, the grid
$\{k\alpha_0(\xi_0)/c^{\theta_0}
: k=1,\ldots,K(c)\}$
is an arithmetic progression
with step $\alpha_0(\xi_0)/c^{\theta_0}$,
and since $\alpha_0(\xi_0)=\theta'(\xi_0)$
varies strictly monotonely with $\xi_0$
(Lemma~\ref{lem:phase-nondeg}(iv)),
these progressions \emph{shift}
as $\xi_0$ varies.
The key question is whether this
shift produces destructive interference
upon integration.
\end{remark}

\begin{proposition}[Alpha-oscillation
  strategy: sufficient condition
  for (T\textsuperscript{*})]
\label{prop:alpha-oscillation}
Let $f_c(\alpha)
:= \sum_{k=1}^{K(c)}
|\widehat{\psi}(k\alpha/c^{\theta_0})|^2$
for $\alpha\in[\alpha_{\min},\alpha_{\max}]$.
Using $|\widehat{\psi}(\xi)|^2
= \int e^{ix\xi}K(x)\,dx$
(where $K = \psi * \widetilde\psi
\in L^1(\mathbb{R})$ is the
auto-correlation of $\psi$), one has:
\begin{equation}\label{eq:fc-dirichlet}
  f_c(\alpha)
  = \int_{\mathbb{R}} K(x)\,
    \underbrace{\sum_{k=1}^{K(c)}
    e^{ixk\alpha/c^{\theta_0}}}
    _{= D_{K(c)}(x\alpha/c^{\theta_0})}\,dx.
\end{equation}
Thus $f_c(\alpha)$
is the \emph{integral transform
of $K$ against the Dirichlet
kernel $D_{K(c)}$ in the
$\alpha$-direction}.

Now integrate over $\alpha$
under the substitution
$\xi_0\mapsto\alpha
= \alpha_0(\xi_0)$
(which is admissible by
Lemma~\ref{lem:phase-nondeg}(iv),
with Jacobian
$d\xi_0/d\alpha
= 1/|\theta''(\xi_0(\alpha))|$):
\begin{align}
  \int_{-c_0}^{c_0}
  f_c(\alpha_0(\xi_0))\,d\xi_0
  &= \int_{\alpha_{\min}}^{\alpha_{\max}}
     f_c(\alpha)\,
     \frac{d\xi_0}{d\alpha}\,d\alpha
  \notag\\
  &= \int_{\mathbb{R}} K(x)
     \int_{\alpha_{\min}}^{\alpha_{\max}}
     D_{K(c)}\!\left(\frac{x\alpha}{c^{\theta_0}}\right)
     g(\alpha)\,d\alpha\;dx,
  \label{eq:alpha-integral}
\end{align}
where $g(\alpha):=d\xi_0/d\alpha
= 1/|\theta''(\xi_0(\alpha))|\in C^0$.

The inner integral over $\alpha$ is
oscillatory: $D_{K(c)}(x\alpha/c^{\theta_0})
= \sum_{k=1}^{K(c)}
e^{ixk\alpha/c^{\theta_0}}$,
so each term
$\int e^{ixk\alpha/c^{\theta_0}}
g(\alpha)\,d\alpha
= \widehat{g}(xk/c^{\theta_0})$.
By the Riemann--Lebesgue lemma,
$\widehat{g}(\xi)\to 0$
as $|\xi|\to\infty$.

\medskip
\noindent\textbf{Sufficient condition
  for (T\textsuperscript{*}).}
If
\begin{equation}\label{eq:RL-condition}
  \frac{1}{K(c)}
  \sum_{k=1}^{K(c)}
  \int_{\mathbb{R}}
  |K(x)|\cdot|\widehat{g}(xk/c^{\theta_0})|\,dx
  = o(1)
  \qquad(c\to\infty),
\end{equation}
then $\int f_c(\alpha_0(\xi_0))\,d\xi_0
= o(c^{\theta_0})$
and hence (T\textsuperscript{*}) holds.
Condition~\eqref{eq:RL-condition}
holds whenever $K\in L^1$ and
$g\in L^1\cap C^0$ with
$|\widehat{g}(\xi)|=o(1)$
as $|\xi|\to\infty$
(which follows from
$g\in L^1$),
\emph{provided the average
over $k$ converges to $0$}.
This is guaranteed by dominated
convergence if
$x\mapsto|K(x)|\cdot\sup_k|\widehat{g}(xk/c^{\theta_0})|
\in L^1$
uniformly in $c$.
\end{proposition}

\begin{remark}[What the proposition
  shows and does not show]
\label{rem:alpha-osc-status}
Proposition~\ref{prop:alpha-oscillation}
\emph{does not} prove (T\textsuperscript{*}).
It shows:
\begin{enumerate}[\rm(i)]
\item The integrated Toeplitz energy
  rewrites as a double transform
  involving $K(x)$ and
  $\widehat{g}(xk/c^{\theta_0})$.
\item A sufficient condition
  (\ref{eq:RL-condition}) is expressed
  entirely in terms of $K$, $g$,
  and the scaling $c^{\theta_0}$.
\item Condition~(\ref{eq:RL-condition})
  is of Riemann--Lebesgue type and
  is \emph{structurally accessible}:
  it depends on regularity of
  $g=1/|\theta''(\xi_0)|$,
  which comes from WKB structure,
  and decay of $K$,
  which comes from $\psi\in L^2$.
\end{enumerate}
Establishing~(\ref{eq:RL-condition})
quantitatively --- in particular,
controlling the $L^1$-uniformity in $c$
of the integrand in $x$ ---
is the decisive task for Paper~XVI.
\end{remark}

\begin{proposition}[Model reduction
  limit: exact status]
\label{prop:model-limit}
Let $\theta$ be the PSWF phase
\textup{(}Lemma~\ref{lem:WKB-phase}\textup{)}.
Then:
\begin{equation}\label{eq:model-limit}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}
    \Tcc(\psi(\cdot;\xi_0))
    \,d\xi_0
    + R(c),
\end{equation}
where $R(c)=O(c^{3\theta_0-2})=o(1)$.

\medskip
\noindent\emph{S3 vs.\ model-limit.}
Proposition~\ref{prop:S3} gives
the \textbf{frozen-$\xi_0$} asymptotics:
$\Tcc\sim c^{\theta_0}C>0$.
The present formula integrates
over all $\xi_0$;
the sampling interpretation
(Remark~\ref{rem:sampling-rewrite})
and the alpha-oscillation strategy
(Proposition~\ref{prop:alpha-oscillation})
show how the shift of the sampling
grid can produce cancellation
at the integrated level.

\medskip
\noindent\textbf{Generic model.}
For frozen $\alpha_0$:
$\Sigma_{\mathrm{model}}=\Theta(1)$.

\noindent\textbf{PSWF model.}
Variation of $\alpha_0(\xi_0)$
($\theta''\ne 0$) is the
PSWF-specific input;
whether it forces
$\Sigma_{\mathrm{model}}=o(1)$
is the content of~(T\textsuperscript{*}).
\end{proposition}

\begin{definition}[(T\textsuperscript{*})]
\label{def:Tstar}
\begin{equation}\label{eq:Tstar}
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \Tcc(\psi(\cdot;\xi_0))
  \,d\xi_0
  = o(1),
\end{equation}
or equivalently
$\Sigma_{\mathrm{model}}(c)=o(1)$.
\end{definition}

\begin{problem}[Critical Toeplitz
  Energy; primary open target]
\label{prob:Tstar}
\emph{Central open problem
of the programme after Paper~XV.}
Establish~(T\textsuperscript{*}).

\medskip
\noindent\textbf{Option~1
  (Alpha-oscillation / coarea).}
Proposition~\ref{prop:alpha-oscillation}
reduces~(T\textsuperscript{*}) to
establishing~(\ref{eq:RL-condition}):
Riemann--Lebesgue decay in $\alpha$
of the Dirichlet-kernel transform
of $K$ weighted by $g$.
Concretely:
\begin{itemize}[leftmargin=*]
\item Show $K\in L^1$ with
  sufficient decay
  (follows from $\psi\in\mathcal{S}$
  or $\psi\in H^1$).
\item Show
  $|\widehat{g}(\xi)|\to 0$
  as $|\xi|\to\infty$
  (follows from $g\in L^1$).
\item Show the $L^1$-uniformity
  in $c$ of the $x$-integrand:
  this is the main analytic task.
\end{itemize}

\noindent\textbf{Option~2
  (Microlocal splitting).}
Decompose $\xi_0$-space into level
sets of $\theta'$; quantitative
coarea with Jacobian $|\theta''|$.

\noindent\textbf{Option~3
  (Szeg\H{o}/Simon).}
Identify the limiting symbol of
$\Dc$ as $c\to\infty$;
(T\textsuperscript{*}) via
Szeg\H{o}'s theorem if symbol
vanishes.
\end{problem}

\begin{remark}[Two levels of reduction]
\label{rem:two-levels}
\begin{enumerate}[\rm(i)]
\item \emph{Analytic:}
  $\Sigma_{\mathrm{int}}=o(1)$
  from $\Sigma_{\mathrm{model}}=o(1)$
  under~(L).
\item \emph{Structural:}
  $\Sigma_{\mathrm{model}}=o(1)$
  $\Leftrightarrow$ (T\textsuperscript{*});
  accessible via
  Proposition~\ref{prop:alpha-oscillation}.
\end{enumerate}
\end{remark}

\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Assume (T\textsuperscript{*}),
(L), and (C).
Then:
\begin{equation}\label{eq:reduction-chain}
  \underbrace{
    \mathrm{(T^*)} + \mathrm{(L)}
  }_{\text{\rm Paper XVI}}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}} = o(1)
  \;\underset{\mathrm{(C)}}{\Longrightarrow}\;
  \text{\rm (D-strong)}
  \;\Rightarrow\;
  \text{\rm H3 fails}
  \;\Rightarrow\;
  \text{\rm H1 fails}.
\end{equation}
\end{theorem}

\begin{proof}
(T\textsuperscript{*}) gives
$\Sigma_{\mathrm{model}}=o(1)$.
(L) gives
$\Sigma_{\mathrm{amp}}
=O(c^{3\theta_0-2})=o(1)$.
Hence $\Sigma_{\mathrm{int}}=o(1)$.
Remainder: Paper~XIV chain under~(C).
\end{proof}

\begin{remark}[Dependency structure]
\label{rem:dependency}
(T\textsuperscript{*}) + (L)
jointly close
$\Sigma_{\mathrm{int}}=o(1)$;
(L) is not covered by the model
reduction alone.
(C) independently closes the
spectral density step.
All three conditions are independent.
\end{remark}

\begin{remark}[Programme status
  after Paper~XV]
\label{rem:series-status}
\[
  \underbrace{\mathrm{(T^*)}+\mathrm{(L)}}
  _{\text{Paper XVI}}
  +
  \underbrace{\mathrm{(C)}}
  _{\text{Paper XVII}}
  \;\Longrightarrow\;
  \text{H1 fails}.
\]
\textbf{XVI:} Establish~(L);
prove~(\ref{eq:RL-condition})
via Option~1 of
Problem~\ref{prob:Tstar}.
\textbf{XVII:} Establish~(C);
investigate Option~3.
\end{remark}
