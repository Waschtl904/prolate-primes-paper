\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to
$\Sigma_{\mathrm{int}}=o(1)$.
This paper identifies the core
structure and reduces the problem
to a sharply formulated scalar
cancellation question.

The main objects are
$I_c(k):=\int e^{ik\theta/c}w\,d\xi$
and $\Sigma_{\mathrm{model}}
=\sum_k|I_c(k)|^2$.
Pointwise tools fail;
the obstruction is
\emph{lack of cancellation
after phase freezing}.
A coarea substitution and WKB
analysis reduce
(T\textsuperscript{*}) to
the oscillatory sum
\begin{equation*}
  S(c):=\int_{\mathbb{R}} K(x)
  \sum_{k=1}^{K(c)}
  \widehat{g}(xk/c^{\theta_0})\,dx,
  \quad K\in\mathcal{S},\;
  |\widehat{g}(\xi)|=O(|\xi|^{-1/2}).
\end{equation*}
The exact equivalence is
(Lemma~\ref{lem:S-equivalence}):
\begin{equation*}
  \Sigma_{\mathrm{model}} = o(1)
  \;\iff\;
  S(c) = o(c^{-(2-2\theta_0)}).
\end{equation*}

Defining $\mathcal{C}(c):=S(c)/K(c)$
($K(c)=c^{\theta_0}$),
a Fubini--Tonelli argument shows
$\Psi(\lambda):=\int K\widehat{g}(\lambda\cdot)
\in L^1(d\lambda/\lambda)$
(Proposition~\ref{prop:Psi-L1}),
giving $\mathcal{C}(c)\to 0$,
i.e., $S(c)=o(c^{\theta_0})$.
However, the target is
$S(c)=o(c^{-(2-2\theta_0)})$:
a \emph{strictly stronger
cancellation} of order
$c^{\theta_0+(2-2\theta_0)}
=c^{2-\theta_0}$
beyond qualitative decay.

The primary remaining task for
Paper~XVI is to establish
this cancellation
(Problem~\ref{prob:S-cancellation})
by analysing $\sum_k\widehat{g}(xk/c^{\theta_0})$
as a \emph{non-uniform Fourier
sampling} / oscillatory kernel,
without taking absolute values.
The main contributions of this paper:
\begin{enumerate}[\rm(1)]
\item Scalar reduction to $S(c)$;
\item Exact equivalence
  $\Sigma_{\mathrm{model}}=o(1)\iff
  S(c)=o(c^{-(2-2\theta_0)})$;
\item Ces\`aro obstruction removed:
  $S(c)=o(c^{\theta_0})$;
\item Gap quantified:
  factor $c^{2-\theta_0}$ remaining;
\item Moving-symbol / ergodic
  structure and non-commuting
  limits identified.
\end{enumerate}
\end{abstract}

\label{lem:WKB-phase}
$\theta(\xi):=c\int_0^{\xi/c}p\,d\zeta$,
$w:=|A|^2/p$,

\label{lem:phase-nondeg}
(i)~$\theta'>0$;\;
(ii)~$\theta''\ne 0$ for $\xi\ne 0$;\;
(iii)~$\theta''(0)=0$, $\theta'''(0)\ne 0$;\;
(iv)~$\xi_0\mapsto\alpha_0:=\theta'(\xi_0)$
strictly monotone.
\end{lemma}

\end{corollary}

%%═══ 6. OBSTRUCTION AND REDUCTION
\section{The Critical Toeplitz
  Obstruction and Scalar Reduction}
\label{sec:toeplitz}

\begin{remark}[Core thesis]
\label{rem:core-thesis}
\begin{center}
\emph{Obstruction = frozen-phase
saturation.
Escape = global curvature averaging
via non-commuting limits.
Mathematical form: cancellation
in the oscillatory sum $S(c)$.}
\end{center}
\end{remark}

\begin{remark}[Moving-symbol / sampling]
\label{rem:moving-symbol}
$\Sigma_{\mathrm{model}}
\approx c^{2(1-\theta_0)}
\int\sum_k|\widehat{\psi}(k\alpha_0(\xi_0)
/c^{\theta_0})|^2\,d\xi_0$:
randomised sampling of $|\widehat\psi|^2$
along $\alpha_0(\xi_0)$;
three non-commuting limits
($k\to\infty$, $c\to\infty$, $\xi_0$-integration).
\end{remark}

\begin{lemma}[Singularity of $g$]
\label{lem:g-singularity}
Let $g(\alpha):=1/|\theta''(\xi_0(\alpha))|$,
$\alpha_*:=\theta'(0)$.
\begin{enumerate}[\rm(i)]
\item $g\in L^1$;
  $g\in C^\infty$ away from $\alpha_*$.
\item $g(\alpha)\sim C_*/|\alpha-\alpha_*|^{1/2}$
  near $\alpha_*$.
\item $|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$
  uniformly in sign.
\end{enumerate}
\end{lemma}

\begin{proof}
(ii): $\alpha_0'(0)=0$,
$\alpha_0''(0)\ne 0$ (Taylor expansion).
(iii): Fourier transform of
$|\cdot|^{-1/2}$.
\end{proof}

\begin{proposition}[Coarea reduction to $S(c)$]
\label{prop:alpha-oscillation}
Substitute $\xi_0\mapsto\alpha
=\alpha_0(\xi_0)$ with Jacobian $g$:
\begin{equation}\label{eq:S-def}
  S(c)
  := \int_{\mathbb{R}} K(x)
     \sum_{k=1}^{K(c)}
     \widehat{g}(xk/c^{\theta_0})
     \,dx,
  \quad
  K:=\psi*\widetilde{\psi}\in\mathcal{S}.
\end{equation}
Then $(T^*)$
$\Leftrightarrow$
$S(c)=o(K(c))=o(c^{\theta_0})$.
\end{proposition}

\begin{lemma}[Exact scaling equivalence]
\label{lem:S-equivalence}
Define $\mathcal{C}(c):=S(c)/K(c)$,
$K(c)=c^{\theta_0}$.
By Proposition~\ref{prop:model-limit}:
\begin{equation}\label{eq:sigma-expand}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}\cdot
    K(c)\cdot\mathcal{C}(c)
    + o(1)
  = c^{2-\theta_0}\cdot\mathcal{C}(c)
    + o(1).
\end{equation}
Since $S(c)=K(c)\cdot\mathcal{C}(c)
=c^{\theta_0}\mathcal{C}(c)$:
\begin{equation}\label{eq:sigma-S}
  \Sigma_{\mathrm{model}}(c)
  = c^{2-2\theta_0}\cdot S(c)
    + o(1).
\end{equation}
Therefore:
\begin{equation}\label{eq:exact-equiv}
  \boxed{\Sigma_{\mathrm{model}}(c)
  = o(1)
  \;\iff\;
  S(c) = o(c^{-(2-2\theta_0)}),}
\end{equation}
conditional on $o(1)$ error in
~\eqref{eq:sigma-S}.
\end{lemma}

\begin{proof}
~\eqref{eq:sigma-S} follows from
~\eqref{eq:sigma-expand} and
$S(c)=c^{\theta_0}\mathcal{C}(c)$:
$c^{2-\theta_0}\mathcal{C}(c)
=c^{2-\theta_0}\cdot S(c)/c^{\theta_0}
=c^{2-2\theta_0}S(c)$.
Equivalence~\eqref{eq:exact-equiv}
is immediate.
\end{proof}

\begin{remark}[The scaling gap,
  expressed in $S(c)$]
\label{rem:scaling-gap}
We have established:
\begin{equation}\label{eq:Sc-current}
  S(c) = o(c^{\theta_0})
  \qquad(\text{Corollary~\ref{cor:cesaro-closed}}).
\end{equation}
The target is:
\begin{equation}\label{eq:Sc-target}
  S(c) = o(c^{-(2-2\theta_0)}).
\end{equation}
The gap is a factor of
$c^{\theta_0}\cdot c^{2-2\theta_0}
=c^{2-\theta_0}$.
This is the \emph{same factor}
by which $\Sigma_{\mathrm{strip}}
=\Theta(c^{2-\theta_0})$ diverges
in the frozen-phase regime:
\begin{center}
\emph{The residual gap equals
exactly the frozen-phase saturation
rate.}
\end{center}
The gap cannot be closed by
Ces\`aro averaging alone;
it requires genuine cancellation
in the $k$-sum of $S(c)$,
without taking absolute values
on $\widehat{g}$.
\end{remark}

\subsection{What the Fubini--Ces\`aro
  argument gives}

\begin{lemma}[Weighted RL and rate]
\label{lem:weighted-RL}
Let $\Psi(\lambda)
:=\int K(x)\widehat{g}(\lambda x)\,dx$.
\begin{enumerate}[\rm(a)]
\item $\Psi(\lambda)\to 0$ (DCT).
\item $|\Psi(\lambda)|=O(\lambda^{-1/2})$,
  uniformly.
\end{enumerate}
\end{lemma}

\begin{lemma}[Ces\`aro borderline]
\label{lem:cesaro-decay}
$\delta=1/2$ gives $|\mathcal{C}(c)|
=O(1)$ from triangle inequality;
hence $S(c)=O(c^{\theta_0})$
(borderline).
If $\Psi\in L^1(d\lambda/\lambda)$:
$\mathcal{C}(c)\to 0$,
$S(c)=o(c^{\theta_0})$.
\end{lemma}

\begin{proposition}[$\Psi\in L^1(d\lambda/\lambda)$]
\label{prop:Psi-L1}
For $K\in\mathcal{S}$
and $|\widehat{g}|=O(|\xi|^{-1/2})$:
\begin{equation}\label{eq:L1-Psi}
  \int_1^\infty
  |\Psi(\lambda)|\,\frac{d\lambda}{\lambda}
  \lesssim
  \int_{\mathbb{R}}
  |K(x)|\,(1+|x|^{-1/2})\,dx
  < \infty.
\end{equation}
\end{proposition}

\begin{proof}
Fubini--Tonelli on truncated domain
$[1,R]$, then $R\to\infty$.
For fixed $x\ne 0$,
$\int_1^\infty|\widehat{g}(\lambda x)|
d\lambda/\lambda
=\int_{|x|}^\infty|\widehat{g}(u)|
du/u
\le 2C/\max(|x|,1)^{1/2}$.
Outer integral finite
since $\int_0^1|x|^{-1/2}dx=2$
and $K\in\mathcal{S}$.
\end{proof}

\begin{corollary}[$S(c)=o(c^{\theta_0})$]
\label{cor:cesaro-closed}
$\mathcal{C}(c)\to 0$,
hence $S(c)=o(c^{\theta_0})$.
\end{corollary}

\begin{proposition}[Model reduction limit]
\label{prop:model-limit}
\begin{equation}\label{eq:model-limit}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}
    \Tcc(\psi)\,d\xi_0
    + O(c^{3\theta_0-2}).
\end{equation}
\end{proposition}

%%═══ 7. OPEN PROBLEMS
\section{Open Problems for Paper~XVI}
\label{sec:open}

\begin{problem}[Primary: cancellation in
  $S(c)$ -- the scaling gap]
\label{prob:S-cancellation}
\emph{Primary target of Paper~XVI.}

\medskip
\noindent\textbf{Required:}
\begin{equation}\label{eq:S-target}
  S(c)
  = \int K(x)
    \sum_{k=1}^{K(c)}
    \widehat{g}(xk/c^{\theta_0})\,dx
  = o(c^{-(2-2\theta_0)}).
\end{equation}
Note: this is
$c^{2-\theta_0}$ stronger than
what Ces\`aro gives
($S(c)=o(c^{\theta_0})$,
Corollary~\ref{cor:cesaro-closed}).

\medskip
\noindent\textbf{Key structural point.}
The sum
$\sum_{k=1}^{K(c)}
\widehat{g}(xk/c^{\theta_0})$
is \emph{not} a monotone sum.
It is a \emph{non-uniform
Fourier sampling}
of $\widehat{g}$ at nodes
$\{xk/c^{\theta_0}\}_{k=1}^{K(c)}$.
In $x$-space, it is the discrete
convolution of $\widehat{g}$
with a Dirichlet-like
sampling measure.
Cancellation between terms
(for fixed $x$, varying $k$)
is the mechanism by which
$S(c)$ can be smaller than
its absolute-value bound.

\medskip
\noindent\textbf{Option~A
  ($k$-oscillation via
  Dirichlet-kernel cancellation).}
For fixed $x\ne 0$, write
$\widehat{g}(xk/c^{\theta_0})
=\int g(\alpha)
e^{ixk\alpha/c^{\theta_0}}d\alpha$.
Then:
\begin{equation}\label{eq:S-rewrite}
  \sum_{k=1}^{K(c)}
  \widehat{g}(xk/c^{\theta_0})
  = \int g(\alpha)
    D_{K(c)}^*(x\alpha/c^{\theta_0})
    \,d\alpha,
\end{equation}
where $D_N^*(t)
:=\sum_{k=1}^N e^{ikt}$
is a truncated exponential sum.
For generic $x\alpha/c^{\theta_0}$,
$D_N^*(t)=O(|t|^{-1})$;
after integrating in $\alpha$,
cancellation with the
$|\alpha-\alpha_*|^{-1/2}$ singularity
of $g$ may give
a bound much better than
$K(c)\cdot|\widehat{g}(x/c^{\theta_0})|$.

\medskip
\noindent\textbf{Option~B
  (stationary phase, 2nd order).}
The fold geometry at $\alpha_*$
(two branches $\xi_0>0$, $\xi_0<0$)
gives two contributions to
$\widehat{g}(\xi)$ near $\xi=0$
that may partially cancel.
Explicit PSWF-WKB structure
of the coefficient $C_*$ in
~\eqref{eq:g-sing}
may produce a cancellation
$C_{*,+}+C_{*,-}=0$
(anti-symmetry in $\xi_0$),
giving $\widehat{g}(\xi)=O(|\xi|^{-3/2})$
for $\xi$ near $0$.
This would improve
$|\Psi(\lambda)|=O(\lambda^{-3/2})$,
more than enough for the target.

\medskip
\noindent\textbf{Relation to existing
  theory.}
Option~A relates to:
Carleman-type estimates
for non-uniform sampling;
discrete Fourier analysis
on non-uniform grids;
semiclassical measures
for oscillatory sums.
Option~B relates to:
degenerate stationary phase
(Airy-type asymptotics).
\end{problem}

\begin{problem}[Rigorous model
  reduction]
\label{prob:rigorous-reduction}

\textbf{Priority~1: Sum--integral exchange.}
Justify~\eqref{eq:sigma-S}
with error $o(c^{-(2-2\theta_0)}\cdot c^{2-2\theta_0})=o(1)$:
the exchange
$\sum_k\int\leftrightarrow\int\sum_k$
must be \emph{quantitative},
not merely qualitative,
with error absorbed by the
scaling.

\textbf{Priority~2: Kernel error.}
$\sum_k|(\AcM)_{nm}-I_c(|n-m|)|^2
=o(1)$; naive bound fails for
$\theta_0\ge 1/3$.

\textbf{Priority~3: PSWF Lipschitz~(L).}
$\|\widehat{\Phi}_{n+k}^{(c)}
-\widehat{\Phi}_n^{(c)}\|
=O(k/c)$;
via WKB~\cite{Zworski2012}
or Sturm comparison.
\end{problem}

\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Conditional on
$S(c)=o(c^{-(2-2\theta_0)})$,
(L), (R), and~(C):
\begin{equation}\label{eq:chain}
  S(c)=o(c^{-(2-2\theta_0)})
  +(\mathrm{L})+(\mathrm{R})
  \Longrightarrow
  \Sigma_{\mathrm{int}} = o(1)
  \underset{(C)}{\Longrightarrow}
  \text{H3 fails}
  \Rightarrow\text{H1 fails.}
\end{equation}
\end{theorem}

\begin{remark}[Programme status
  after Paper~XV]
\[
  \underbrace{
    S(c)=o(c^{-(2-2\theta_0)})
    +(\mathrm{R})+(\mathrm{L})
  }_{\text{Paper XVI}}
  +\underbrace{(C)}_{\text{XVII}}
  \Longrightarrow\text{H1 fails.}
\]
The problem is fully reduced to:
\begin{center}
\emph{Cancellation of order $c^{2-\theta_0}$
in the oscillatory sum $S(c)$
(non-uniform Fourier sampling
of $\widehat{g}$ by nodes
$\{xk/c^{\theta_0}\}$).}
\end{center}
\end{remark}
