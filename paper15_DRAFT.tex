\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to a single
open problem: the intermediate-regime
Hilbert--Schmidt estimate
$\Sigma_{\mathrm{int}} = o(1)$
for $M(c)\sim\beta c$.
This paper identifies the core
analytic obstruction as a
\emph{critical-density Toeplitz
energy problem} and develops
the analytic toolkit required
for the reduction to this formulation.

The central object is the oscillatory
kernel family
\[
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi,
  \quad 1\le k\le c^{\theta_0},
\]
and the associated sum
$\Sigma_{\mathrm{model}}(c)
:= \sum_{k=1}^{\lfloor c^{\theta_0}\rfloor}
|I_c(k)|^2$.
The Dirichlet-kernel reformulation
(\S\ref{sec:model}) and mesoscopic
rescaling identify $\Sigma_{\mathrm{model}}$
with the critical Toeplitz energy
functional
$\Tcc(\psi) := \ip{\Dc\psi}{\psi}$
at the double-scaling
regime $K(c)\cdot h = 1$.

The WKB phase extraction
(Lemmas~\ref{lem:WKB-phase},
\ref{lem:phase-nondeg})
identifies $\theta(\xi)$ explicitly,
proves $\theta'\ne 0$ and
$\theta''\ne 0$ for $\xi\ne 0$,
and establishes that both naive
van~der~Corput
(Corollary~\ref{cor:curvature-available})
and the Weyl/UDPL/exponent-pair
combination
(Proposition~\ref{prop:W4-decoupling})
are insufficient to close the
$\Sigma_{\mathrm{model}}=o(1)$ claim:
the obstruction is not
\emph{analytic weakness} but
\emph{lack of cancellation after
phase freezing}
(\S\ref{sec:toeplitz},
Remark~\ref{rem:core-thesis}).

Proposition~\ref{prop:model-limit}
establishes the precise status:
$\Sigma_{\mathrm{model}}$ reduces
exactly to a critical Toeplitz energy
at frozen phase, which is
$\Theta(1)$ for generic phase
and of \emph{unknown asymptotics}
for the PSWF-specific phase.
Three resolution strategies are
formulated (\S\ref{sec:toeplitz}):
coarea/microlocal splitting,
phase-curvature averaging, and
spectral reinterpretation via
Szeg\H{o} asymptotics.
The Toeplitz decay property~(T\textsuperscript{*})
is identified as the decisive new
target of the programme.

The main structural output is the
\emph{Reduction Theorem}
(Theorem~\ref{thm:reduction}):
\[
  \mathrm{(T^*)} + \mathrm{(L)}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}}=o(1),
  \qquad
  \mathrm{(C)}
  \;\Longrightarrow\;
  \text{H1 fails},
\]
where~(L) (Lipschitz regularity,
Paper~XVI), (C) (local Weyl law,
Paper~XVII), and~(T\textsuperscript{*})
(PSWF Toeplitz decay)
are the three independent open
targets of the programme.
\end{abstract}

\section{Introduction}

\subsection{Inherited structure from Paper~XIV}

Paper~XIV established the following
reduction chain:
\begin{equation}\label{eq:chain-XIV}
  \underbrace{(C) + (D\text{-strong})}
  _{\text{Paper XIV target}}
  \;\Rightarrow\;
  S_N^{(M(c))}(c)\sim c_0\beta > 0
  \;\Rightarrow\;
  \text{H3 fails}
  \;\Rightarrow\;
  \text{H1 fails}.
\end{equation}
The HS decomposition
\begin{equation}\label{eq:HS-decomp}
  \norm{\AcM - D^{(M)}}_{\mathrm{HS}}^2
  = \Sigma_{\mathrm{near}}
  + \Sigma_{\mathrm{int}}
  + \Sigma_{\mathrm{far}}
\end{equation}
satisfies:
$\Sigma_{\mathrm{near}} = O(1/c) = o(1)$
(proved, Paper~XIV, conditional on~(C));
$\Sigma_{\mathrm{far}} = o(1)$
(accessible via stationary phase);
and the sole remaining obstruction:
\begin{equation}\label{eq:int-target}
  \Sigma_{\mathrm{int}}
  = \sum_{\substack{n\ne m\\
      C<|n-m|\le c^{\theta_0}}}
    \left|(\AcM)_{nm}\right|^2
  \stackrel{?}{=} o(1),
  \qquad M(c)\sim\beta c.
\end{equation}

\subsection{Three levels of decay}

\begin{remark}[Three levels of $o(1)$]
\label{rem:three-levels}
The symbol $o(1)$ appears at three
distinct levels throughout this paper;
these are \emph{not} interchangeable:
\begin{enumerate}[\rm(i)]
\item \emph{Pointwise decay}:
  $|(\AcM)_{nm}| = o(1/c)$
  for each fixed pair $(n,m)$
  with $|n-m|\ge C$.
  This is (D-weak); it does not
  imply HS-summability over
  $O(c^2)$ pairs.
\item \emph{HS-summated decay}:
  $\Sigma_{\mathrm{near}}=o(1)$,
  $\Sigma_{\mathrm{far}}=o(1)$,
  $\Sigma_{\mathrm{int}}=o(1)$.
  This is the operative level
  for (D-strong).
\item \emph{Uniform in $\theta_0$}:
  the crude bound
  $O(c^{\theta_0-1})=o(1)$
  holds for each fixed $\theta_0<1$
  but \emph{not} uniformly as
  $\theta_0\to 1^-$.
\end{enumerate}
\end{remark}

\subsection{The model reduction}

The Gram kernel takes the form
\begin{equation}\label{eq:gram-kernel}
  (\AcM)_{nm}
  = \int_{-c_0}^{c_0}
    \widehat{\Phi_n^{(c)}}(\xi)\,
    \overline{\widehat{\Phi_m^{(c)}}(\xi)}
    \,d\xi.
\end{equation}
Under assumption~(C) and the Lipschitz
bound
\begin{equation}\label{eq:Lipschitz-main}
  \norm{\widehat{\Phi_{n+k}^{(c)}}-\widehat{\Phi_n^{(c)}}}_{L^2([-c_0,c_0])}
  = O(k/c),
  \qquad |k|\le c^{\theta_0},
  \tag{L}
\end{equation}
the amplitude correction
$A_n\overline{A_m} - w$
contributes $O(k/c)$ per entry,
which is HS-summable.
The leading term reduces to
the model kernel:
\begin{equation}\label{eq:model-reduction}
  (\AcM)_{nm}
  \approx I_c(|n-m|)
  + O(|n-m|/c),
  \qquad
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi.
\end{equation}
Thus $\Sigma_{\mathrm{int}}=o(1)$
follows from
$\Sigma_{\mathrm{model}}(c)=o(1)$
once~\eqref{eq:Lipschitz-main} holds.
Condition~(L) is the subject of
Paper~XVI; the local Weyl law~(C)
is the subject of Paper~XVII.

%%═══ 2. THE MODEL PROBLEM
\section{The Intermediate-Regime
  Model Problem}
\label{sec:model}

\begin{problem}[Intermediate-regime
  kernel summation;
  model problem]
\label{prob:model-intermediate}

Let $\theta\in C^\infty([-c_0,c_0])$
satisfy $\theta'(\xi)\ne 0$,
and let $w\in C^\infty([-c_0,c_0])$,
$w>0$, $\int w\,d\xi=1$.

Define
\begin{equation}\label{eq:Ic-def}
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi,
  \qquad k\in\mathbb{Z}_{\ge 1}.
\end{equation}
Fix $\theta_0\in(0,1)$,
$K(c):=\lfloor c^{\theta_0}\rfloor$.

\medskip\noindent
\textbf{Open Target (Model-$\Sigma$).}
\begin{equation}\label{eq:model-sigma}
  \Sigma_{\mathrm{model}}(c)
  := \sum_{k=1}^{K(c)}|I_c(k)|^2
  \;\stackrel{?}{=}\; o(1).
\end{equation}
This is \emph{not} proved here.
It is reduced to~(T\textsuperscript{*})
in \S\ref{sec:toeplitz}.

\medskip\noindent
\textbf{Kernel reformulation.}
\begin{equation}\label{eq:kernel-rewrite}
  \Sigma_{\mathrm{model}}(c)
  = \int\!\int
    w(\xi)\,w(\eta)\,
    F_c(\xi,\eta)
    \,d\xi\,d\eta,
\end{equation}
\begin{equation}\label{eq:Fc}
  F_c(\xi,\eta)
  := \sum_{k=1}^{K(c)}
     e^{ik(\theta(\xi)-\theta(\eta))/c},
  \quad
  \phi:=\theta(\xi)-\theta(\eta).
\end{equation}
This is a \emph{bilinear form};
no separation into pointwise bounds
is admissible.

\medskip\noindent
\textbf{Two-scale structure.}
\begin{itemize}[leftmargin=*]
\item \textbf{Off-diagonal}
  ($|\xi-\eta|\gg c^{1-\theta_0}$):
  IBP gives $|I_c(k)|=O(c/k)$;
  $\Sigma_{\mathrm{off}}
  =O(c^{3\theta_0-2})=o(1)$
  for $\theta_0<2/3$.
\item \textbf{Diagonal strip}
  ($|\xi-\eta|\lesssim c^{1-\theta_0}$):
  strip area $\sim c^{2(1-\theta_0)}$,
  $|F_c|\le c^{\theta_0}$:
  contributes $O(1)$.
  This is the critical Toeplitz
  regime.
\end{itemize}

\medskip\noindent
\textbf{Mesoscopic rescaling.}
Fix $\xi_0$, set
$\xi=\xi_0+c^{1-\theta_0}u$,
$\eta=\xi_0+c^{1-\theta_0}v$:
\begin{equation}\label{eq:meso-F}
  F_c(u,v)
  = D_{K(c)}\!\big(
       h\,\theta'(\xi_0)(u-v)
     \big),
  \quad
  h:=c^{-\theta_0},
  \quad K(c)\cdot h = 1.
\end{equation}
This is the \emph{double-scaling
Szeg\H{o} regime}: neither
the classical Szeg\H{o} theorem
nor the semiclassical limit applies.
\end{problem}

%%═══ 3. CRITICAL MESOSCOPIC ORTHOGONALITY
\section{Critical Mesoscopic
  Orthogonality}
\label{sec:S3}

\begin{definition}[Toeplitz--Dirichlet
  operator; Critical Toeplitz Energy]
\label{def:TD-operator}
For fixed $\xi_0$ and $c$, define
the \emph{$c$-dependent}
Toeplitz--Dirichlet operator:
\begin{equation}\label{eq:Dc-def}
  (\Dc f)(u)
  := \int_{\mathbb{R}}
     D_{K(c)}\!\big(h\,\alpha_0(u-v)\big)
     \psi(v)\,f(v)\,dv,
\end{equation}
where $\alpha_0 := \theta'(\xi_0)$,
$h = c^{-\theta_0}$,
$K(c) = \lfloor c^{\theta_0}\rfloor$.
The \emph{critical Toeplitz energy
functional} (depending on $c$
through $K(c)$ and $h$) is
\begin{equation}\label{eq:Tc-def}
  \Tcc(\psi)
  := \ip{\Dc\psi}{\psi}
  = \sum_{k=1}^{K(c)}
    \left|\widehat{\psi}\!\left(
      \frac{k\alpha_0}{c^{\theta_0}}
    \right)\right|^2,
\end{equation}
so that
$\Sigma_{\mathrm{strip}}(c)
= c^{2(1-\theta_0)}\Tcc(\psi)$.
\end{definition}

\begin{proposition}[Critical
  mesoscopic saturation]
\label{prop:S3}
For fixed $\xi_0$ and frozen
$\alpha_0 = \theta'(\xi_0)$:

\medskip
\noindent\textbf{(i) Fourier representation.}
\begin{equation}\label{eq:Sigma-fourier}
  \Sigma_{\mathrm{strip}}(c)
  = c^{2(1-\theta_0)}
    \sum_{k=1}^{K(c)}
    \left|
      \widehat{\psi}\!\left(
        \frac{k\alpha_0}{c^{\theta_0}}
      \right)
    \right|^2.
\end{equation}

\noindent\textbf{(ii) Riemann sum
  saturation.}
\begin{equation}\label{eq:riemann}
  \frac{\Tcc(\psi)}{c^{\theta_0}}
  \;=\;
  \frac{1}{c^{\theta_0}}
  \sum_{k=1}^{K(c)}
  \left|\widehat{\psi}\!\left(
    \frac{k\alpha_0}{c^{\theta_0}}
  \right)\right|^2
  \;\longrightarrow\;
  \frac{\norm{\psi}_{L^2}^2}{|\alpha_0|}
  =: C(\psi,\alpha_0) > 0,
\end{equation}
hence
$\Tcc(\psi)
= c^{\theta_0}\cdot C(\psi,\alpha_0)
+ o(c^{\theta_0})$.

\noindent\textbf{(iii) Scaling
  conclusion.}
\begin{equation}\label{eq:strip-final}
  \Sigma_{\mathrm{strip}}(c)
  = c^{2-\theta_0}\cdot C(\psi,\alpha_0)
    + o(c^{2-\theta_0})
  = \Theta(c^{2-\theta_0}).
\end{equation}

\noindent\emph{Note:} this is the
\textbf{frozen-$\xi_0$} behaviour
at a single basepoint.
The full $\xi_0$-integrated result
is the subject of
Proposition~\ref{prop:model-limit};
cancellation between different
values of $\alpha_0(\xi_0)$ is
not captured here.
\end{proposition}

\begin{proof}
$\xi\mapsto|\widehat{\psi}(\xi)|^2$
is $L^1\cap C^0$ with compact support;
the sum is a Riemann sum with mesh
$|\alpha_0|/c^{\theta_0}\to 0$.
Part~(iii): $c^{2(1-\theta_0)}
\cdot c^{\theta_0}
= c^{2-\theta_0}$.
\end{proof}

\begin{corollary}[No-go for frozen phase;
  scale-balance obstruction]
\label{cor:S3-negative}
For any fixed
$\alpha_0 = \theta'(\xi_0)\ne 0$,
the strip contribution at that
$\xi_0$ satisfies
$\Sigma_{\mathrm{strip}}(c)
=\Theta(c^{2-\theta_0})\to+\infty$.
No oscillatory mechanism acting
\emph{pointwise} in $\alpha_0$ can
produce $o(1)$.
\end{corollary}

\begin{remark}[Scale-balance
  obstruction]
\label{rem:scale-balance}
The Weyl/UDPL bound from
Proposition~\ref{prop:W4-decoupling}
gives $|T(c)|=O(c^{3\theta_0/4})$,
but the strip-area prefactor is
$c^{2(1-\theta_0)}$.
Product:
$c^{2(1-\theta_0)}\cdot c^{3\theta_0/4}
= c^{2-5\theta_0/4} = \Omega(1)$
for all $\theta_0\in(0,1)$.
The Weyl/UDPL mechanism is correct
but cannot close
$\Sigma_{\mathrm{model}}=o(1)$ alone.
The missing input is variation of
$\alpha_0(\xi_0)=\theta'(\xi_0)$
across $\xi_0$ ---
a \emph{global}, not \emph{local},
property.
\end{remark}

\begin{remark}[Two escapes
  (revised)]
\label{rem:two-escapes}
\textbf{Escape~A
  (global phase variation):}
$\xi_0\mapsto\alpha_0(\xi_0)$
is strictly monotone
(Lemma~\ref{lem:phase-nondeg}(iv));
integration over $\xi_0$ may
produce cancellation in
$\int\Tcc(\psi(\cdot;\xi_0))\,d\xi_0$.

\textbf{Escape~B
  (hidden PSWF orthogonality):}
The WKB amplitude $w(\xi)$
encodes PSWF eigenstructure;
hidden cancellation not visible
in the model reduction may force
$\Tcc(\psi)=o(c^{\theta_0})$.

Both escapes require PSWF-specific
structure absent in the generic model.
\end{remark}

%%═══ 4. WKB PHASE EXTRACTION
\section{WKB Phase Extraction}
\label{sec:WKB}

\subsection{PSWF ODE and semiclassical
  scaling}

The PSWFs satisfy
\begin{equation}\label{eq:PSWF-ODE}
  \frac{d}{dx}\!\left[
    (1-x^2)\Phi_n^{(c)\prime}
  \right]
  + \left[\chi_n(c)-c^2x^2\right]
    \Phi_n^{(c)} = 0.
\end{equation}
Setting $n=\alpha c$, $h=1/c$,
$\xi=c\zeta$, the classical
momentum is
\begin{equation}\label{eq:momentum}
  p(\zeta;\alpha)
  :=
  \sqrt{\frac{\lambda(\alpha)-\zeta^2}
             {1-\zeta^2}},
  \qquad
  |\zeta|<\sqrt{\lambda(\alpha)}.
\end{equation}

\subsection{Phase function}

\begin{lemma}[WKB phase extraction]
\label{lem:WKB-phase}
In the classically allowed region:
\begin{equation}\label{eq:WKB-ansatz}
  \widehat{\Phi_n^{(c)}}(\xi)
  \approx_{\mathrm{WKB}}
  \frac{A(\xi/c;\alpha)}
       {\sqrt{p(\xi/c;\alpha)}}
  \exp\!\left(
    ic\int_0^{\xi/c}
    p(\zeta;\alpha)\,d\zeta
  \right).
\end{equation}
The phase and weight are:
\begin{equation}\label{eq:theta-def}
  \theta(\xi)
  := c\int_0^{\xi/c}
     p(\zeta;\alpha)\,d\zeta,
  \qquad
  w(\xi)
  := \frac{|A(\xi/c;\alpha)|^2}
          {p(\xi/c;\alpha)}.
\end{equation}
\end{lemma}

\subsection{Non-degeneracy}

\begin{lemma}[Phase non-degeneracy]
\label{lem:phase-nondeg}
For $\alpha\in(0,1)$ and
$|\xi|\le c_0<c\sqrt{\lambda(\alpha)}$:

\noindent\textbf{(i)}
$\theta'(\xi) = p(\xi/c;\alpha) > 0$.

\noindent\textbf{(ii)}
For $\xi\ne 0$:
$\theta''(\xi)
= \frac{-(\xi/c)(1-\lambda(\alpha))}
       {c(1-(\xi/c)^2)^{3/2}
         \sqrt{\lambda(\alpha)-(\xi/c)^2}}
\ne 0$.

\noindent\textbf{(iii)}
$|\theta''(\xi)|\ge c_*|\xi|/c$
for $|\xi|\ge\varepsilon$;
$\theta''(0)=0$,
$\theta'''(0)\ne 0$
(cubic non-degeneracy).

\noindent\textbf{(iv) Global variation:}
$\xi_0\mapsto\alpha_0(\xi_0)
:=\theta'(\xi_0)$ is strictly
monotone on
$[-c_0,c_0]\setminus\{0\}$
with $|\alpha_0'|=|\theta''|\ne 0$.
This non-trivial variation is
the structural basis for Escape~A.
\end{lemma}

\begin{proof}
Direct differentiation of $p(\xi/c;\alpha)$.
\end{proof}

\begin{remark}[$\lambda(\alpha)$]
\label{rem:lambda-alpha}
At WKB order \cite{Osipov2013}:
$\lambda(\alpha)
= \sin^2(\pi\alpha/2)+O(1/c)$,
confirming $\lambda(\alpha)\in(0,1)$
and $\theta''\ne 0$.
\end{remark}

\begin{remark}[Turning point
  at $\xi=0$]
\label{rem:turning-zero}
$\theta''(0)=0$ reflects PSWF symmetry;
cubic non-degeneracy
$\theta'''(0)\ne 0$ gives Airy control
in a neighbourhood of $0$.
\end{remark}

\subsection{van der Corput is
  insufficient}

\begin{corollary}[Curvature available
  but naive vdC insufficient]
\label{cor:curvature-available}
With $|\theta''|\sim\varepsilon/c$,
van~der~Corput gives
$|I_c(k)|\lesssim\sqrt{c/(k\varepsilon)}$,
hence
$\sum_k|I_c(k)|^2\lesssim
(c/\varepsilon)\log c = \Omega(c)$.
The correct tool is the bilinear
kernel~\eqref{eq:kernel-rewrite},
not pointwise bounds.
\end{corollary}

%%═══ 5. WEYL EQUIDISTRIBUTION
\section{Weyl Equidistribution and
  Decoupling}
\label{sec:Weyl}

\begin{remark}[Scope of the
  Weyl/UDPL argument]
\label{rem:what-proved}
Proposition~\ref{prop:W4-decoupling}
controls the \emph{linear} sum
$T(c)=\sum_k\int
e^{ik\theta'(\xi_0)/c^{\theta_0}}
\psi\,d\xi_0$.
It does \emph{not} directly control
$\Sigma_{\mathrm{model}}
=\sum_k|I_c(k)|^2$.
The Toeplitz saturation
(Proposition~\ref{prop:S3})
shows the step from $T$ to
$\Sigma$ carries an irreducible
prefactor $c^{2-5\theta_0/4}=\Omega(1)$.
\end{remark}

\subsection{Joint $k$/$\xi_0$
  decoupling}
\label{subsec:joint-decoupling}

\begin{lemma}[Uniform Degenerate
  Phase Lemma (UDPL)]
\label{lem:UDPL}
Let $h\in(0,1]$,
$\Phi_h\in C^3([-c_0,c_0])$, with:
\begin{enumerate}[\rm(P1)]
\item $|\Phi_h'|\ge\mu_1>0$
  for $|\xi_0|\ge\delta$
  (uniform in $h$).
\item $|\Phi_h''(\xi_0)|\ge\mu_2|\xi_0|$
  for $|\xi_0|\le r_0$
  (uniform in $h$).
\item $|\Phi_h'''|\ge\mu_3>0$
  on $[-c_0,c_0]$
  (uniform in $h$).
\end{enumerate}
For $\psi\in C_c^\infty((-c_0,c_0))$,
optimising over $\delta$:
\begin{equation}\label{eq:UDPL-opt}
  \left|
    \int e^{i\Phi_h}\psi\,d\xi_0
  \right|
  \le
  C_\psi'\,\mu_1^{-3/5}\mu_3^{-1/5},
  \qquad
  \text{uniformly in }h.
\end{equation}
\end{lemma}

\begin{proof}
Split at $\pm\delta$:
awy from $0$, first-derivative
vdC \cite[Ch.~VIII, Prop.~2]{Stein1993}
gives $C/(\mu_1\delta)$;
near $0$, cubic non-degeneracy
and Airy bound
\cite[Ch.~VIII, Thm.~1]{Stein1993}
give $C\mu_3^{-1/3}\delta^{2/3}$.
Balance at $(\mu_1\delta)^{-1}
=\mu_3^{-1/3}\delta^{2/3}$.
\end{proof}

\begin{remark}[No extra critical
  points]
\label{rem:no-extra-crit}
$\Phi_h'=0\Leftrightarrow\theta''=0$,
which holds only at $\xi_0=0$
by Lemma~\ref{lem:phase-nondeg}(ii).
\end{remark}

\begin{proposition}[Joint $k$/$\xi_0$
  Decoupling]
\label{prop:W4-decoupling}
For $T(c)
:=\sum_{k=1}^{K(c)}
\int e^{ik\theta'(\xi_0)/c^{\theta_0}}
\psi\,d\xi_0$
and any $\varepsilon>0$:
\begin{equation}\label{eq:W4-rate}
  |T(c)|
  =
  O\!\left(c^{3\theta_0/4+\varepsilon}\right)
  =
  o\!\left(c^{\theta_0}\right).
\end{equation}
\end{proposition}

\begin{proof}
\textbf{Weyl differencing.}
For $1\le H\le K(c)$:
\[
  |T(c)|^2
  \lesssim
  \frac{K(c)^2}{H}
  +
  \frac{K(c)}{H}
  \sum_{h=1}^{H}|J(h,c)|,
  \quad
  J(h,c)
  =\int e^{ih\theta'/c^{\theta_0}}
  \psi\,d\xi_0.
\]
\textbf{UDPL bound.}
With $\mu_1\sim h/c^{1+\theta_0}$,
$\mu_3\sim h/c^{\theta_0}$:
$|J(h,c)|\le C_\psi
c^{(1+\theta_0)/2}/h^{1/2}$.
\textbf{Optimisation.}
Choose $H=\lfloor c^{(1+\theta_0)/2}\rfloor$;
$\sum_{h=1}^H h^{-1/2}\le 2H^{1/2}$:
$|T(c)|^2\lesssim
c^{(3\theta_0-1)/2}+c^{3\theta_0/2}$.
Dominant term $c^{3\theta_0/2}$
gives $|T(c)|=O(c^{3\theta_0/4+\varepsilon})$.
\end{proof}

\begin{remark}[What W4 does not prove]
\label{rem:W4-scope}
W4 controls $|T(c)|=o(c^{\theta_0})$,
not $\Sigma_{\mathrm{model}}=o(1)$.
The gap exponent is
$2-\theta_0 - 3\theta_0/4
= 2-7\theta_0/4 > 0$
for $\theta_0<8/7$;
in particular for all
$\theta_0\in(0,1)$.
\end{remark}

\begin{remark}[Exponent-pair
  classification]
\label{rem:exp-pair}
$|T|=O(c^{3\theta_0/4})$:
$A$-process $(0,1)\to(1/2,1/2)$
then $B$-process $(1/2,1/2)\to(1/3,2/3)$.
Improvement requires a mean-value
theorem on a non-flat curve;
not needed for the programme.
\end{remark}

%%═══ 6. THE CRITICAL TOEPLITZ OBSTRUCTION
\section{The Critical Toeplitz
  Obstruction}
\label{sec:toeplitz}

\begin{remark}[Core thesis of
  Paper~XV]
\label{rem:core-thesis}
The central finding of this paper is:
\begin{center}
\emph{The obstruction to
$\Sigma_{\mathrm{model}}=o(1)$
is not analytic weakness,
but the absence of cancellation
after phase freezing.}
\end{center}
Specifically: every individual
analytic tool (vdC, IBP, Weyl,
UDPL) is correctly applied;
the failure is that these tools
operate \emph{pointwise in $\xi_0$},
while the necessary cancellation
is \emph{global in $\xi_0$},
requiring variation of
$\alpha_0(\xi_0)=\theta'(\xi_0)$
or hidden PSWF orthogonality.
\end{remark}

\begin{proposition}[Model reduction
  limit: exact status]
\label{prop:model-limit}
Let $\theta$ be the PSWF phase
\textup{(}Lemma~\ref{lem:WKB-phase}\textup{)}.
Then:
\begin{equation}\label{eq:model-limit}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}
    \Tcc(\psi(\cdot;\xi_0))
    \,d\xi_0
    + R(c),
\end{equation}
where $R(c)=O(c^{3\theta_0-2})=o(1)$
is the off-diagonal contribution.

\medskip
\noindent
\emph{Remark on the two levels.}
Proposition~\ref{prop:S3} describes
the \textbf{frozen-$\xi_0$} behaviour
of $\Tcc(\psi(\cdot;\xi_0))$
at a single basepoint:
$\Tcc\sim c^{\theta_0}C(\psi,\alpha_0)$.
The present formula performs the
full \textbf{$\xi_0$-integration}.
Cancellation between different
values of $\alpha_0(\xi_0)$ can
only arise at this integrated level,
not at the frozen level.

\medskip
\noindent\textbf{Generic model
  behaviour.}
For frozen $\alpha_0$,
$\Tcc\sim c^{\theta_0}C>0$,
so the integral is $\Theta(c^{\theta_0})$
and $\Sigma_{\mathrm{model}}=\Theta(1)$.
Thus $\Sigma_{\mathrm{model}}(c)\not\to 0$
for generic phase.

\medskip
\noindent\textbf{PSWF model.}
The weight $w(\xi)$ depends on $\xi$
through the WKB amplitude, and
$\alpha_0(\xi_0)=\theta'(\xi_0)$
varies with $\theta''\ne 0$.
Whether this forces
$\Sigma_{\mathrm{model}}=o(1)$
is the content of~(T\textsuperscript{*}).
\end{proposition}

\begin{definition}[PSWF Toeplitz
  Decay Property~(T\textsuperscript{*})]
\label{def:Tstar}
Property~(T\textsuperscript{*})
holds if, for the PSWF-specific
$\theta$ and $w$ of
Lemma~\ref{lem:WKB-phase}:
\begin{equation}\label{eq:Tstar}
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \Tcc(\psi(\cdot;\xi_0))
  \,d\xi_0
  = o(1)
  \qquad(c\to\infty),
\end{equation}
or equivalently $\Sigma_{\mathrm{model}}(c)=o(1)$
\textup{(}by Proposition~\ref{prop:model-limit}\textup{)}.
\end{definition}

\begin{problem}[Critical Toeplitz
  Energy; primary open target]
\label{prob:Tstar}
\emph{This is the central open
problem of the programme
after Paper~XV.}
Establish~(T\textsuperscript{*}).
Three strategies:

\medskip
\noindent\textbf{Option~1
  (Phase-variation averaging).}
The map $\xi_0\mapsto\alpha_0(\xi_0)$
is strictly monotone
(Lemma~\ref{lem:phase-nondeg}(iv)).
A coarea formula gives
\[
  \int_{-c_0}^{c_0}
  \Tcc(\psi(\cdot;\xi_0))\,d\xi_0
  = \int_{\alpha_{\min}}^{\alpha_{\max}}
    \Tcc(\psi(\cdot;\xi_0(\alpha)))
    \frac{d\xi_0}{d\alpha}\,d\alpha.
\]
If $\alpha\mapsto\Tcc$ oscillates
at scale $c^{\theta_0}$,
Riemann--Lebesgue gives $o(c^{\theta_0})$.

\medskip
\noindent\textbf{Option~2
  (Microlocal splitting).}
Decompose $\xi_0$-space into level
sets of $\theta'$; on each level set
$\Tcc$ saturates but contributes
measure zero. Quantitative coarea
with Jacobian $|\theta''(\xi_0)|$
may convert local saturation to
global decay.

\medskip
\noindent\textbf{Option~3
  (Szeg\H{o}/Simon reinterpretation).}
Regard $\Dc$ as a Toeplitz operator;
identify its limiting symbol as
$c\to\infty$. If the symbol vanishes
(spectral orthogonality of PSWFs
in $n$-space),
(T\textsuperscript{*}) follows
from Szeg\H{o}'s theorem.
\end{problem}

\begin{remark}[Two levels of
  reduction]
\label{rem:two-levels}
Paper~XV provides two reductions:
\begin{enumerate}[\rm(i)]
\item \emph{Analytic:}
  $\Sigma_{\mathrm{int}}=o(1)$
  follows from
  $\Sigma_{\mathrm{model}}=o(1)$
  under~(L).
\item \emph{Structural:}
  $\Sigma_{\mathrm{model}}=o(1)$
  $\Leftrightarrow$ (T\textsuperscript{*}):
  the integrated Toeplitz energy
  decays.
\end{enumerate}
The problem is precisely localised:
not ``prove oscillatory decay''
but ``identify the global
cancellation mechanism for
Toeplitz energy in the PSWF
system''.
\end{remark}

%%═══ 7. REDUCTION THEOREM
\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Assume:
\begin{enumerate}[\rm(i)]
\item \textup{(T\textsuperscript{*})}
  PSWF Toeplitz decay
  \textup{(}Definition~\ref{def:Tstar}\textup{)},
  i.e.\ $\Sigma_{\mathrm{model}}(c)=o(1)$.
\item \textup{(L)} Lipschitz
  regularity \textup{(}Paper~XVI\textup{)}:
  $\norm{\widehat{\Phi_{n+k}^{(c)}}
    -\widehat{\Phi_n^{(c)}}}_{L^2}
  = O(k/c)$,
  $|k|\le c^{\theta_0}$.
\item \textup{(C)} Local Weyl law
  \textup{(}Paper~XVII\textup{)}:
  $\mu_{n,c}([-c_0,c_0])
  = c_0/c + o(1/c)$
  uniformly.
\end{enumerate}
Then:
\begin{equation}\label{eq:reduction-chain}
  \underbrace{
    \mathrm{(T^*)} + \mathrm{(L)}
  }_{\text{\rm Paper XVI}}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}}(c) = o(1)
  \;\underset{\mathrm{(C)}}{\Longrightarrow}\;
  \text{\rm (D-strong)}
  \;\Rightarrow\;
  \text{\rm H3 fails}
  \;\Rightarrow\;
  \text{\rm H1 fails}.
\end{equation}
\end{theorem}

\begin{proof}
Under~(T\textsuperscript{*}),
$\Sigma_{\mathrm{model}}(c)=o(1)$.
Under~(L), the amplitude correction
$\Sigma_{\mathrm{amp}}
\le\sum_{k=1}^{K(c)}(k/c)^2
=O(c^{3\theta_0-2})=o(1)$.
Therefore
\[
  \Sigma_{\mathrm{int}}(c)
  \le
  \Sigma_{\mathrm{model}}(c)
  + \Sigma_{\mathrm{amp}}(c)
  = o(1).
\]
The remaining implications
$\Sigma_{\mathrm{int}}=o(1)
\underset{\mathrm{(C)}}{\Rightarrow}
\text{(D-strong)}
\Rightarrow\text{H3 fails}
\Rightarrow\text{H1 fails}$
are the chain of Paper~XIV.
\end{proof}

\begin{remark}[Dependency structure]
\label{rem:dependency}
\begin{itemize}[leftmargin=*]
\item (T\textsuperscript{*})
  + (L): jointly close
  $\Sigma_{\mathrm{int}}=o(1)$.
  (L) is needed even given
  (T\textsuperscript{*}): the
  amplitude error is not covered
  by the model reduction.
\item (C): independently closes
  the statistical spectral step
  $\Sigma_{\mathrm{int}}=o(1)
  \Rightarrow\text{(D-strong)}$.
\end{itemize}
All three conditions are independent.
\end{remark}

\begin{remark}[Programme status
  after Paper~XV]
\label{rem:series-status}
\[
  \underbrace{\mathrm{(T^*)}+\mathrm{(L)}}
  _{\text{Paper XVI}}
  +
  \underbrace{\mathrm{(C)}}
  _{\text{Paper XVII}}
  \;\Longrightarrow\;
  \text{H1 fails}.
\]
Paper~XV has precisely located
the core obstruction~(T\textsuperscript{*})
and provided the full reduction toolkit
(WKB, UDPL, Weyl, mesoscopic rescaling).
Papers~XVI and~XVII proceed to:
\begin{itemize}[leftmargin=*]
\item \textbf{XVI:} Establish~(L);
  investigate~(T\textsuperscript{*})
  via Options~1--2.
\item \textbf{XVII:} Establish~(C);
  investigate~(T\textsuperscript{*})
  via Option~3.
\end{itemize}
\end{remark}
