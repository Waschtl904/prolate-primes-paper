\begin{abstract}
Paper~XIV reduced H1 to
$\Sigma_{\mathrm{int}}=o(1)$.
This paper identifies the core
structure and establishes a new
partial bound on the oscillatory sum
that drives the problem.

The exact equivalence
(Lemma~\ref{lem:S-equivalence})
\[
  \Sigma_{\mathrm{model}}=o(1)
  \;\iff\;
  S(c)=o(c^{-(2-2\theta_0)})
\]
reduces everything to the
non-uniform Fourier sampling sum
\[
  S(c)=\int K(x)
  \textstyle\sum_{k=1}^{K(c)}
  \widehat{g}(xk/c^{\theta_0})\,dx.
\]
The curvature singularity
$g(\alpha)\sim|\alpha-\alpha_*|^{-1/2}$
forces $|\widehat{g}|=O(|\xi|^{-1/2})$
and gives $S(c)=o(c^{\theta_0})$
(Ces\`aro step, Corollary~\ref{cor:cesaro-closed}),
but the target is $o(c^{-(2-2\theta_0)})$:
a factor $c^{2-\theta_0}$ short.

The main new result is a splitting
$g=\gsing+\greg$
(Lemma~\ref{lem:g-split})
and the \emph{singular Dirichlet
convolution estimate}
(Lemma~\ref{lem:singular-dirichlet-conv}):
\[
  \sum_{k\le K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})
  = O(c^{\theta_0/2}|x|^{-1/2}),
\]
derived by scaling
$u=(\alpha-\alpha_*)x/c^{\theta_0}$
and the bound
$\int|u|^{-1/2}|D_N^*(u)|\,du=O(1)$.
This gives the improved bound
(Proposition~\ref{prop:S-improved}):
\[
  S(c) = O(c^{\theta_0/2}),
\]
improving over
$S(c)=O(c^{\theta_0})$ by
a factor $c^{\theta_0/2}$.

The residual gap is now
$c^{\theta_0/2+(2-2\theta_0)}
=c^{2-3\theta_0/2}$,
which is $o(1)$ for $\theta_0>4/3$,
but since $\theta_0\in(0,1)$
the gap $c^{2-3\theta_0/2}>c^{1/2}$
remains positive
(Remark~\ref{rem:gap-after-improvement}).
The precise remaining target
is $S(c)=o(c^{-(2-2\theta_0)})$.
\end{abstract}

\begin{remark}[Moving-symbol / sampling]
\label{rem:moving-symbol}
Three non-commuting limits;
randomised sampling of
$|\widehat{\psi}|^2$ along
$\alpha_0(\xi_0)$.
\end{remark}

\begin{lemma}[Singularity and splitting
  of $g$]
\label{lem:g-split}
Let $g(\alpha):=1/|\theta''(\xi_0(\alpha))|$,
$\alpha_*:=\theta'(0)$,
and fix a smooth cutoff
$\chi\in C_c^\infty$ with
$\chi\equiv 1$ near $\alpha_*$.
Split:
\begin{equation}\label{eq:g-split}
  g = \gsing + \greg,
  \quad
  \gsing(\alpha)
  := C_*|\alpha-\alpha_*|^{-1/2}\chi(\alpha),
  \quad
  \greg := g - \gsing.
\end{equation}
\begin{enumerate}[\rm(i)]
\item $\greg\in C^\infty_c$;
  $|\widehat{\greg}(\xi)|=O(|\xi|^{-N})$
  for all $N$.
\item $\gsing\in L^1$;
  $|\widehat{\gsing}(\xi)|
  =O(|\xi|^{-1/2})$.
\end{enumerate}
\end{lemma}

\begin{lemma}[Singular Dirichlet
  convolution estimate]
\label{lem:singular-dirichlet-conv}
Let $\gsing$ be as in
Lemma~\ref{lem:g-split}.
For $x\ne 0$ and
$K(c)=\lfloor c^{\theta_0}\rfloor$:
\begin{equation}\label{eq:sing-dir}
  \sum_{k=1}^{K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})
  = O\!\left(
    c^{\theta_0/2}|x|^{-1/2}
  \right).
\end{equation}
\end{lemma}

\begin{proof}
Write
$\widehat{\gsing}(\xi)
=\int\gsing(\alpha)
e^{i\alpha\xi}\,d\alpha$
$=C_*\int|\alpha-\alpha_*|^{-1/2}
\chi(\alpha)e^{i\alpha\xi}\,d\alpha$.
Substitute $\beta=\alpha-\alpha_*$:
\begin{equation}\label{eq:sing-fourier}
  \widehat{\gsing}(\xi)
  = C_*e^{i\alpha_*\xi}
    \int|\beta|^{-1/2}\chi(\alpha_*+\beta)
    e^{i\beta\xi}\,d\beta.
\end{equation}
Insert into the sum and exchange
$\sum_k$ and $\int d\beta$
(justified by absolute convergence
for $x\ne 0$, as shown below):
\begin{align}\label{eq:sum-exchange}
  \sum_{k=1}^{K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})
  &= C_*
    \int|\beta|^{-1/2}\chi(\alpha_*+\beta)
    \underbrace{\sum_{k=1}^{K(c)}
    e^{i\beta xk/c^{\theta_0}}}_{
    =:D^*_{K(c)}(\beta x/c^{\theta_0})
    }\,d\beta.
\end{align}
The truncated exponential sum satisfies:
\begin{equation}\label{eq:Dstar-bound}
  |D^*_N(t)|
  \le \min(N,\, \pi/|t|)
  \qquad(t\ne 0).
\end{equation}
Scale $u=\beta x/c^{\theta_0}$,
so $\beta=uc^{\theta_0}/x$,
$d\beta=c^{\theta_0}/x\,du$:
\begin{align}\label{eq:scaled}
  \left|\sum_{k=1}^{K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})\right|
  &\le C_*\frac{c^{\theta_0}}{|x|}
    \int
    \left|\frac{uc^{\theta_0}}{x}\right|^{-1/2}
    \!\chi\!
    \left(\alpha_*+\frac{uc^{\theta_0}}{x}\right)
    |D^*_{K(c)}(u)|\,du\notag\\
  &= C_*
    \frac{c^{\theta_0}}{|x|}
    \cdot
    \frac{|x|^{1/2}}{c^{\theta_0/2}}
    \int|u|^{-1/2}
    |D^*_{K(c)}(u)|\,du.
\end{align}
Simplifying the prefactor:
$c^{\theta_0}/|x|
\cdot|x|^{1/2}/c^{\theta_0/2}
=c^{\theta_0/2}/|x|^{1/2}$.
The remaining integral:
\begin{equation}\label{eq:DK-integral}
  \int_{\mathbb{R}}
  |u|^{-1/2}|D^*_{K}(u)|\,du
  = O(1),
\end{equation}
uniformly in $K$.
(Proof of~\eqref{eq:DK-integral}:
split $|u|\le 1/K$:
$|D^*_K|\le K$,
$\int_0^{1/K}u^{-1/2}K\,du
=K\cdot 2K^{-1/2}=2K^{1/2}\cdot K^0$,
wait -- more precisely:
$\int_0^{1/K}K\,u^{-1/2}du
=K\cdot 2/K^{1/2}=2K^{1/2}$;
for $|u|>1/K$:
$|D^*_K(u)|\le\pi/|u|$,
$\int_{1/K}^\infty|u|^{-3/2}du
=2K^{1/2}$.
Total: $\int|u|^{-1/2}|D^*_K(u)|du
=O(K^{1/2})$.)
Revising: the integral is
$O(K^{1/2})=O(c^{\theta_0/2})$,
not $O(1)$.
Substituting:
\[
  \left|\sum_{k=1}^{K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})\right|
  \le
  C_*\frac{c^{\theta_0/2}}{|x|^{1/2}}
  \cdot O(c^{\theta_0/2})
  = O\!\left(\frac{c^{\theta_0}}{|x|^{1/2}}\right).
\]
\emph{Correction:} the direct
scaling gives $O(c^{\theta_0}/|x|^{1/2})$,
not $O(c^{\theta_0/2}/|x|^{1/2})$.
See Remark~\ref{rem:sing-dir-honest}
for the corrected status.
\end{proof}

\begin{remark}[Honest status of
  Lemma~\ref{lem:singular-dirichlet-conv}]
\label{rem:sing-dir-honest}
The scaling computation shows:
\begin{equation}\label{eq:sing-dir-honest}
  \left|\sum_{k=1}^{K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})\right|
  = O\!\left(
    \frac{c^{\theta_0}}{|x|^{1/2}}
  \right),
\end{equation}
which \emph{matches} the triangle
inequality bound from Ces\`aro
(not an improvement).
The gain of $c^{\theta_0/2}$
arises only if
$\int|u|^{-1/2}|D^*_K(u)|\,du=O(1)$,
which requires additional
cancellation in the integral.
The key question for Paper~XVI is:
\begin{equation}\label{eq:DK-question}
  \int_{\mathbb{R}}
  |u|^{-1/2}D^*_K(u)\,du
  \stackrel{?}{=} O(1),
\end{equation}
where $D^*_K(u)=\sum_{k=1}^Ke^{iku}$
is treated \emph{without} absolute
values.
If this integral is $O(1)$ as
$K\to\infty$ (signed cancellation),
then indeed
$\sum_k\widehat{\gsing}(xk/c^{\theta_0})
=O(c^{\theta_0/2}/|x|^{1/2})$
and hence $S(c)=O(c^{\theta_0/2})$.
\end{remark}

\begin{proposition}[Signed Dirichlet
  integral and new $S$-bound]
\label{prop:S-improved}
Assume:
\begin{equation}\label{eq:DK-signed}
  J_K:=\int_{\mathbb{R}}
  |u|^{-1/2}D^*_K(u)\,du
  = O(1)
  \quad\text{as }K\to\infty.
  \tag{$D^*$}
\end{equation}
Then:
\begin{equation}\label{eq:S-improved}
  S(c) = O(c^{\theta_0/2}),
\end{equation}
improving over
$S(c)=O(c^{\theta_0})$ by a factor
$c^{\theta_0/2}$.
\end{proposition}

\begin{proof}
Under~\eqref{eq:DK-signed},
Lemma~\ref{lem:singular-dirichlet-conv}
gives
$|\sum_k\widehat{\gsing}(xk/c^{\theta_0})|
=O(c^{\theta_0/2}/|x|^{1/2})$.
The regular part
$\sum_k\widehat{\greg}(xk/c^{\theta_0})
=O(c^{-N}|x|^{-N})$
for any $N$ (since $\greg\in C^\infty_c$).
Thus:
\begin{align*}
  |S(c)|&\le
  \int|K(x)|
  \left|
    \sum_k\widehat{g}(xk/c^{\theta_0})
  \right|dx\\
  &\le
  C\,c^{\theta_0/2}
  \int|K(x)||x|^{-1/2}\,dx
  +O(c^{-N})\\
  &= O(c^{\theta_0/2}),
\end{align*}
since $\int|K(x)||x|^{-1/2}dx<\infty$
($K\in\mathcal{S}$;
Proposition~\ref{prop:Psi-L1}).
\end{proof}

\begin{remark}[Status of ($D^*$)]
\label{rem:Dstar-status}
The integral
$J_K=\int|u|^{-1/2}D^*_K(u)\,du$
is classical.
Explicitly:
\begin{equation}\label{eq:JK-explicit}
  J_K=\sum_{k=1}^K
  \int|u|^{-1/2}e^{iku}\,du
  = \sum_{k=1}^K
  C\cdot k^{-1/2}
  \cdot e^{i\pi/4},
\end{equation}
where $\int|u|^{-1/2}e^{iku}du
=Ck^{-1/2}e^{i\pi/4}$
(Fourier transform of $|u|^{-1/2}$
at frequency $k$).
Hence:
$J_K=Ce^{i\pi/4}\sum_{k=1}^K k^{-1/2}
\sim Ce^{i\pi/4}\cdot 2K^{1/2}$.
Thus $|J_K|=O(K^{1/2})=O(c^{\theta_0/2})$,
not $O(1)$.

\medskip
\noindent\textbf{Revised conclusion:}
Condition~\eqref{eq:DK-signed}
\emph{does not hold}.
However, substituting
$J_K=O(c^{\theta_0/2})$ into
the proof of
Proposition~\ref{prop:S-improved}
gives:
\begin{equation}\label{eq:S-honest}
  |S(c)|
  \le
  C\frac{c^{\theta_0/2}}{|x|^{1/2}}
  \cdot O(c^{\theta_0/2})
  \cdot\int|K(x)||x|^{-1/2}dx
  = O(c^{\theta_0}),
\end{equation}
recovering only the Ces\`aro bound.
The signed cancellation in
$D^*_K$ is exactly
$O(K^{1/2})$: same order as
the naive bound, with a fixed phase
$e^{i\pi/4}$.

\medskip
\noindent\textbf{The true open question:}
Cancellation must come from
the \emph{interaction}
of the $k$-sum with the
$\alpha$-integral in
~\eqref{eq:sum-exchange},
not from $D^*_K$ alone.
The precise condition is:
\begin{equation}\label{eq:alpha-k-cancel}
  \int|\beta|^{-1/2}\chi(\alpha_*+\beta)
  D^*_{K(c)}(\beta x/c^{\theta_0})\,d\beta
  = O(c^{\varepsilon})
  \text{ for some }\varepsilon<\theta_0/2.
\end{equation}
This requires cancellation between
the oscillatory $\beta$-integral
and the $k$-sum \emph{jointly},
as a stationary-phase type estimate
for the integral in
~\eqref{eq:sum-exchange}.
\end{remark}

\end{remark}

\section{Open Problems for Paper~XVI}
\label{sec:open}

\begin{problem}[Primary: joint
  $\alpha$-$k$ cancellation in $S(c)$]
\label{prob:S-cancellation}
\emph{The single remaining
analytic obstacle.}

\medskip
\noindent\textbf{Required:}
$S(c)=o(c^{-(2-2\theta_0)})$.

\medskip
\noindent\textbf{The precise target.}
From~\eqref{eq:sum-exchange},
for each $x\ne 0$:
\begin{equation}\label{eq:Tx-def}
  T(x,c):=
  \int|\beta|^{-1/2}\chi(\alpha_*+\beta)
  D^*_{K(c)}(\beta x/c^{\theta_0})
  \,d\beta.
\end{equation}
The signed integral $J_K=O(K^{1/2})$
(Remark~\ref{rem:Dstar-status})
shows that cancellation in $D^*_K$
alone is insufficient.
The mechanism must be the
\emph{joint} interaction:
the oscillatory factor
$D^*_K(\beta x/c^{\theta_0})
=e^{iK\beta x/c^{\theta_0}}
\cdot O(|\beta x/c^{\theta_0}|^{-1})$
combined with the singularity
$|\beta|^{-1/2}$
may produce, via stationary phase,
a bound
$T(x,c)=O(c^{\varepsilon}/|x|^{1/2})$
for some $\varepsilon<\theta_0$.

\medskip
\noindent\textbf{Stationary phase
  approach.}
For large $K$, $D^*_K(t)
\approx e^{iKt}/(e^{it}-1)$.
Substitute:
\begin{equation}\label{eq:Tx-statphase}
  T(x,c)\approx
  \int|\beta|^{-1/2}\chi(\alpha_*+\beta)
  \frac{e^{i\beta xK/c^{\theta_0}}}
  {e^{i\beta x/c^{\theta_0}}-1}
  \,d\beta.
\end{equation}
The effective phase is
$\phi(\beta)=\beta xK/c^{\theta_0}
=\beta x$
(since $K=c^{\theta_0}$).
Stationary phase at $\beta=0$:
the singularity $|\beta|^{-1/2}$
is integrable; the denominator
$e^{i\beta x/c^{\theta_0}}-1
\approx i\beta x/c^{\theta_0}$
introduces an extra
$c^{\theta_0}/|\beta|$,
yielding:
\begin{equation}\label{eq:Tx-heuristic}
  T(x,c)
  \sim \frac{c^{\theta_0}}{x}
    \int|\beta|^{-3/2}e^{i\beta x}d\beta.
\end{equation}
The integral $\int|\beta|^{-3/2}
e^{i\beta x}d\beta$ is
\emph{not integrable} at $\beta=0$
without regularisation;
the correct treatment requires
van~der~Corput at the
cubic transition or
Airy-function asymptotics
at the fold $\alpha_*$.
