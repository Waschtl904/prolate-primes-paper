  Mesoscopic Scaling, WKB Phase Extraction,
  and the Critical Toeplitz Obstruction}

\author{[Author]}
\address{[Address]}

\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to the
intermediate-regime Hilbert--Schmidt
estimate $\Sigma_{\mathrm{int}}=o(1)$
for $M(c)\sim\beta c$.
This paper identifies the core
analytic structure and develops the
reduction toolkit.

The main objects are the oscillatory
kernel family
$I_c(k):=\int e^{ik\theta(\xi)/c}w(\xi)\,d\xi$
and the model sum
$\Sigma_{\mathrm{model}}(c)
:=\sum_{k=1}^{K(c)}|I_c(k)|^2$.
The WKB analysis
(Lemmas~\ref{lem:WKB-phase},
\ref{lem:phase-nondeg})
shows that pointwise tools
(van~der~Corput, Weyl/UDPL)
are insufficient;
the obstruction is \emph{lack of
cancellation after phase freezing}.

A coarea substitution
(Proposition~\ref{prop:alpha-oscillation})
reduces (T\textsuperscript{*}) to
the weighted Riemann--Lebesgue
function
\[
  \Psi(\lambda)
  := \int K(x)\,\widehat{g}(\lambda x)\,dx,
\]
where $K=\psi*\widetilde{\psi}\in L^1$
and $g=1/|\theta''(\xi_0(\cdot))|
\in L^1$.
The singularity analysis
(Lemma~\ref{lem:g-singularity})
yields $|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$
and hence $|\Psi(\lambda)|=O(\lambda^{-1/2})$.

However, the reduction of
$\Sigma_{\mathrm{model}}=o(1)$ to
$\Psi(\lambda)\to 0$ requires
controlling the Ces\`aro average
$\frac{1}{K(c)}\sum_{k\le K(c)}
\Psi(k/c^{\theta_0})$,
not merely pointwise decay of
$\Psi$ (Lemma~\ref{lem:cesaro-decay}).
With $|\Psi(\lambda)|=O(\lambda^{-1/2})$,
this Ces\`aro average is $O(1)$:
the rate $\delta=1/2$ is \emph{exactly
borderline}.
Closing (T\textsuperscript{*})
therefore requires one of:
\begin{enumerate}[\rm(i)]
\item establishing $\Psi\in L^1(d\lambda/\lambda)$
  (sufficient for Ces\`aro decay);
\item showing $|\Psi(\lambda)|
  =O(\lambda^{-\delta})$ with $\delta>1$
  (requires better regularity of $g$);
\item exploiting $k$-oscillation
  in the sum.
\end{enumerate}
The main structural contribution
of this paper is the reduction to
this sharply-formulated Ces\`aro
problem, the identification of
the exact borderline exponent
$\delta=1/2$, and the conceptual
reinterpretation as a
\emph{moving-symbol Toeplitz /
randomised-sampling problem}
(Remark~\ref{rem:moving-symbol}),
which is the publishable core.
All remaining open problems
are collected in
\S\S\ref{sec:Tstar-open}--\ref{sec:open}.
\end{abstract}

\label{lem:WKB-phase}
$\theta(\xi):=c\int_0^{\xi/c}
p(\zeta;\alpha)\,d\zeta$,
$w:=|A|^2/p$,
$p(\zeta;\alpha):=\sqrt{(\lambda(\alpha)-\zeta^2)/(1-\zeta^2)}$.

\begin{lemma}[Phase non-degeneracy]
\label{lem:phase-nondeg}
(i) $\theta'>0$;\;
(ii) $\theta''(\xi)\ne 0$ for $\xi\ne 0$,
$\theta''\sim -\xi/c$ near $0$;\;
(iii) $\theta''(0)=0$, $\theta'''(0)\ne 0$;\;
(iv) $\xi_0\mapsto\alpha_0(\xi_0)
:=\theta'(\xi_0)$ strictly monotone.
\end{lemma}

\begin{corollary}[vdC insufficient]
\label{cor:curvature-available}
$\sum_k|I_c(k)|^2
\lesssim(c/\varepsilon)\log c=\Omega(c)$.
\end{corollary}

%%═══ 6. THE CRITICAL TOEPLITZ OBSTRUCTION
\section{The Critical Toeplitz
  Obstruction}
\label{sec:toeplitz}

\begin{remark}[Core thesis]
\label{rem:core-thesis}
\begin{center}
\emph{Obstruction = absence of
cancellation after phase freezing.
Escape = global curvature averaging.}
\end{center}
\end{remark}

\begin{remark}[Sampling reinterpretation]
\label{rem:sampling-rewrite}
$\Sigma_{\mathrm{model}}
\approx c^{2(1-\theta_0)}
\int\sum_k|\widehat{\psi}(k\alpha_0(\xi_0)
/c^{\theta_0})|^2\,d\xi_0$:
randomised sampling of $|\widehat\psi|^2$
along the curve $\alpha_0(\xi_0)$,
closer to ergodic averaging
than to classical Toeplitz theory.
\end{remark}

\begin{lemma}[Singularity of curvature weight]
\label{lem:g-singularity}
Let $g(\alpha):=1/|\theta''(\xi_0(\alpha))|$,
$\alpha_*:=\theta'(0)$.
\begin{enumerate}[\rm(i)]
\item $g\in C^\infty$ away from $\alpha_*$.
\item Near $\alpha_*$:
  $\alpha-\alpha_*\sim
  \tfrac12\theta'''(0)\xi_0^2/c$,
  so $|\xi_0|\sim C|\alpha-\alpha_*|^{1/2}$
  and
  $g(\alpha)\sim C_*/|\alpha-\alpha_*|^{1/2}$.
  Hence $g\in L^1$.
\item $|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$
  as $|\xi|\to\infty$.
\end{enumerate}
\end{lemma}

\begin{proof}
(ii): $\alpha_0'(0)=\theta''(0)=0$,
$\alpha_0''(0)=\theta'''(0)/c\ne 0$;
Taylor expansion gives the claim.
(iii): $\mathcal{F}[|\cdot-\alpha_*|^{-1/2}]
= C|\xi|^{-1/2}$;
remainder is $O(|\xi|^{-N})$ by
Riemann--Lebesgue.
\end{proof}

\begin{proposition}[Alpha-oscillation
  strategy]
\label{prop:alpha-oscillation}
Let $K:=\psi*\widetilde{\psi}\in L^1$.
Coarea substitution
$\xi_0\mapsto\alpha=\alpha_0(\xi_0)$:
\begin{equation}\label{eq:alpha-integral}
  \int_{-c_0}^{c_0}
  f_c(\alpha_0(\xi_0))\,d\xi_0
  = \int K(x)
    \sum_{k=1}^{K(c)}
    \widehat{g}(xk/c^{\theta_0})\,dx
  =: S(c).
\end{equation}
Sufficient condition for (T\textsuperscript{*}):
\begin{equation}\label{eq:RL-condition}
  \frac{S(c)}{K(c)}
  = \frac{1}{K(c)}
    \sum_{k=1}^{K(c)}
    \int|K(x)|\cdot|\widehat{g}(xk/c^{\theta_0})|
    \,dx = o(1).
\end{equation}
\end{proposition}

\begin{lemma}[Weighted Riemann--Lebesgue]
\label{lem:weighted-RL}
Let
$\Psi(\lambda):=\int K(x)\widehat{g}(\lambda x)\,dx$.

\begin{enumerate}[\rm(a)]
\item \emph{(Qualitative).}
  $g\in L^1\Rightarrow\widehat{g}\in C_0$;
  DCT gives $\Psi(\lambda)\to 0$.

\item \emph{(Quantitative rate).}
  By Lemma~\ref{lem:g-singularity}(iii)
  and $K\in\mathcal{S}$:
  \begin{equation}\label{eq:Psi-rate}
    |\Psi(\lambda)| = O(\lambda^{-1/2}).
  \end{equation}
\end{enumerate}
\end{lemma}

\begin{proof}
(a): Standard DCT.
(b): $|\widehat{g}(\lambda x)|
\le C(1+|\lambda x|)^{-1/2}$;
split at $|x|=\lambda^{-1}$,
both pieces $O(\lambda^{-1/2})$
for $K\in\mathcal{S}$.
\end{proof}

\begin{lemma}[Ces\`aro summation
  and the borderline exponent]
\label{lem:cesaro-decay}
Let $\Psi:\mathbb{R}_{>0}\to\mathbb{R}$
with $|\Psi(\lambda)|\le C\lambda^{-\delta}$
for some $\delta>0$.
For $K=K(c)=\lfloor c^{\theta_0}\rfloor$,
consider the Ces\`aro-type average
\begin{equation}\label{eq:cesaro-def}
  \mathcal{C}(c)
  := \frac{1}{K(c)}
     \sum_{k=1}^{K(c)}
     \Psi(k/c^{\theta_0}).
\end{equation}

\begin{enumerate}[\rm(i)]
\item \emph{(Riemann-sum estimate).}
  The sum is dominated by the
  integral:
  \[
    |\mathcal{C}(c)|
    \le \frac{C}{K}\sum_{k=1}^{K}
    (k/c^{\theta_0})^{-\delta}
    = \frac{C c^{\theta_0\delta}}{K}
      \sum_{k=1}^{K}k^{-\delta}.
  \]

\item \emph{(Case $\delta>1$).}
  $\sum_{k\ge 1}k^{-\delta}<\infty$,
  so $\mathcal{C}(c)=O(c^{\theta_0(\delta-1)}
  \cdot K^{-1})$.
  Since $K=c^{\theta_0}$:
  $\mathcal{C}(c)=O(c^{\theta_0(\delta-1)}
  \cdot c^{-\theta_0})
  =O(c^{-\theta_0(2-\delta)})$.
  For $\delta>1$: $2-\delta<1$ but
  $\mathcal{C}(c)\to 0$.
  \textbf{Sufficient for (T\textsuperscript{*}).}

\item \emph{(Case $\delta=1$).}
  $\sum_{k=1}^{K}k^{-1}\sim\log K$,
  so $\mathcal{C}(c)\sim
  c^{\theta_0}\log c/c^{\theta_0}
  = \log c\cdot c^{-\theta_0}
  \cdot c^{\theta_0}$.
  More carefully:
  $\mathcal{C}(c)
  =O(\log c\cdot c^{-\theta_0\cdot 0})
  = O(\log c/K)=O(\log c/c^{\theta_0})\to 0$.
  \textbf{Sufficient.}

\item \emph{(Case $\delta=1/2$; our situation).}
  $\sum_{k=1}^K k^{-1/2}\sim 2K^{1/2}$,
  so
  \begin{equation}\label{eq:borderline}
    |\mathcal{C}(c)|
    \lesssim
    \frac{c^{\theta_0/2}}{K}\cdot 2K^{1/2}
    = 2c^{\theta_0/2}\cdot K^{-1/2}
    = 2c^{\theta_0/2}\cdot c^{-\theta_0/2}
    = 2.
  \end{equation}
  \textbf{Borderline: $|\mathcal{C}(c)|=O(1)$,
  not $o(1)$.}
  This estimate is sharp:
  there exist functions with
  $|\Psi(\lambda)|\sim\lambda^{-1/2}$
  for which $\mathcal{C}(c)\not\to 0$.

\item \emph{(Sufficient condition for
  all $\delta>0$).}
  If additionally
  $\Psi\in L^1((1,\infty),d\lambda/\lambda)$,
  i.e.,
  $\int_1^\infty|\Psi(\lambda)|
  \frac{d\lambda}{\lambda}<\infty$,
  then $\mathcal{C}(c)\to 0$
  by the Hardy--Littlewood
  Ces\`aro theorem.
  For $|\Psi(\lambda)|=O(\lambda^{-\delta})$:
  $\int_1^\infty\lambda^{-\delta-1}\,d\lambda
  <\infty$ iff $\delta>0$.
  So \textbf{any $\delta>0$ with
  the $L^1(d\lambda/\lambda)$ condition
  suffices.}
\end{enumerate}
\end{lemma}

\begin{proof}
Parts~(i)--(iv): Direct Riemann-sum
estimation.
Part~(v): For the Ces\`aro theorem
with $L^1(d\lambda/\lambda)$
weighting, see e.g.~\cite{Stein1993},
Ch.~I.
\end{proof}

\begin{remark}[(T\textsuperscript{*}):
  exact gap]
\label{rem:Tstar-gap}
The rate $|\Psi(\lambda)|=O(\lambda^{-1/2})$
from Lemma~\ref{lem:weighted-RL}(b)
placed into the Ces\`aro estimate
yields $\mathcal{C}(c)=O(1)$
(Lemma~\ref{lem:cesaro-decay}(iv)):
the exponent is exactly borderline.
The claim $\mathcal{C}(c)\to 0$
therefore requires a \emph{strictly
sharper} input than
$|\Psi(\lambda)|=O(\lambda^{-1/2})$;
specifically, one needs either:
\begin{enumerate}[\rm(A)]
\item $\Psi\in L^1(d\lambda/\lambda)$:
  this holds if
  $|\Psi(\lambda)|=O(\lambda^{-\delta})$
  for \emph{any} $\delta>0$
  (Lemma~\ref{lem:cesaro-decay}(v));
  or
\item Cancellation in the $k$-sum
  beyond the triangle inequality:
  the sum $\sum_k\int K(x)
  \widehat{g}(xk/c^{\theta_0})\,dx$
  may have oscillating terms
  that sum to $o(K)$; or
\item $\theta_0$-restriction:
  if $\theta_0$ is small enough that
  the discrete sum can be replaced
  by a Riemann integral with
  better error bounds.
\end{enumerate}
This gap is the \emph{precise
remaining obstacle} for
(T\textsuperscript{*}), and is
the primary target of
Problem~\ref{prob:Tstar-cesaro}.
\end{remark}

\begin{proposition}[Model reduction limit]
\label{prop:model-limit}
\begin{equation}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}
    \Tcc(\psi)\,d\xi_0
    + O(c^{3\theta_0-2}).
\end{equation}
\end{proposition}

\begin{definition}[(T\textsuperscript{*})]
\label{def:Tstar}
$\Sigma_{\mathrm{model}}(c)=o(1)$,
or equivalently $\mathcal{C}(c)=o(1)$
(\emph{status: open;
see Problem~\ref{prob:Tstar-cesaro}}).
\end{definition}

\begin{remark}[Moving-symbol Toeplitz
  structure]
\label{rem:moving-symbol}
The problem is a \emph{moving-symbol
Toeplitz problem}: symbol
$\alpha_0(\xi_0)$ varies with basepoint;
the three limits $k\to\infty$,
$c\to\infty$, $\xi_0$-integration
do not commute.
Alternative view:
\emph{randomised sampling of
$|\widehat\psi|^2$ along
$\alpha_0(\xi_0)$},
closer to ergodic averaging
than to standard Szeg\H{o} theory
\cite{Simon2011,BottcherSilbermann1990}.
\end{remark}

%%═══ 7. OPEN PROBLEMS: (T*)
\section{Open Problems: (T\textsuperscript{*})
  and the Ces\`aro Step}
\label{sec:Tstar-open}

\begin{problem}[(T\textsuperscript{*}):
  closing the Ces\`aro step]
\label{prob:Tstar-cesaro}
\emph{Primary open target of
this programme after Paper~XV.}

\medskip
\noindent\textbf{Setup.}
By Proposition~\ref{prop:alpha-oscillation}
and Lemma~\ref{lem:weighted-RL},
(T\textsuperscript{*}) is equivalent to
$\mathcal{C}(c)\to 0$, where
$|\mathcal{C}(c)|=O(1)$
from the current $\delta=1/2$ bound.

\medskip
\noindent\textbf{Option~A
  (improve Psi-decay).}
Show $|\Psi(\lambda)|=O(\lambda^{-\delta})$
for some $\delta>1/2$.
Equivalently: improve the singularity
of $g$ near $\alpha_*$.
\begin{itemize}[leftmargin=*]
\item If $\theta''$ has a zero of
  order $m\ge 2$ at $\xi_0=0$,
  then $g(\alpha)\sim|\alpha-\alpha_*|^{-m/(m+1)}$
  and $\delta=1/(m+1)$.
  For $m=1$ (our case): $\delta=1/2$.
  For $m=2$: $\delta=1/3$.
  \emph{A higher-order zero of
  $\theta''$ does not help.}
\item Alternatively: use PSWF-specific
  structure of $g$ beyond the
  leading singularity;
  higher-order Taylor coefficients
  may improve $\widehat{g}$ via
  cancellation.
\end{itemize}

\medskip
\noindent\textbf{Option~B
  ($L^1(d\lambda/\lambda)$ condition).}
Show $\int_1^\infty|\Psi(\lambda)|
d\lambda/\lambda < \infty$
directly, without improving
the pointwise rate.
By Lemma~\ref{lem:cesaro-decay}(v),
this suffices for $\mathcal{C}(c)\to 0$.
\begin{itemize}[leftmargin=*]
\item This holds automatically if
  $|\Psi(\lambda)|=O(\lambda^{-\epsilon})$
  for any $\epsilon>0$;
  but it may also hold if
  $\Psi$ has enough oscillation.
\item Concrete goal:
  show $\int_1^\infty|\int K(x)
  \widehat{g}(\lambda x)\,dx|
  d\lambda/\lambda < \infty$
  using $K\in\mathcal{S}$.
\end{itemize}

\medskip
\noindent\textbf{Option~C
  ($k$-oscillation).}
The sum $S(c)=\int K(x)
\sum_k\widehat{g}(xk/c^{\theta_0})\,dx$
may have cancellation between terms
beyond what the triangle inequality
captures.
Concretely: the Dirichlet kernel
$\sum_k e^{ixk\alpha/c^{\theta_0}}$
oscillates in $k$; this oscillation,
combined with the decay of $K$,
may give $S(c)=o(K(c))$
without improving $|\Psi(\lambda)|$
pointwise.
\end{problem}

%%═══ 8. OPEN PROBLEMS: RIGOROUS REDUCTION
\section{Open Problems for Paper~XVI:
  Rigorous Reduction}
\label{sec:open}

\begin{problem}[Rigorous model reduction]
\label{prob:rigorous-reduction}
\textbf{Goal:}
$\Sigma_{\mathrm{int}}
=\Sigma_{\mathrm{model}}+o(1)$
uniform in $c$.

\medskip
\noindent\textbf{Priority~1:
  Kernel error.}
The HS-norm of the kernel error
\[
  \sum_{k\le c^{\theta_0}}
  |(\AcM)_{nm}-I_c(|n-m|)|^2
  = o(1)
\]
requires careful treatment:
the naive bound $O(c^{3\theta_0-1})$
is $o(1)$ only for $\theta_0<1/3$;
for $\theta_0\ge 1/3$,
cancellation must be exploited
(Remark~\ref{rem:kernel-critical}).

\medskip
\noindent\textbf{Priority~2:
  Sum--integral exchange.}
Uniform control of
$\sum_k\leftrightarrow\int\,d\xi_0$
in $k\le c^{\theta_0}$.

\medskip
\noindent\textbf{Priority~3: PSWF~(L).}
\begin{equation}\label{eq:L-goal}
  \norm{\widehat{\Phi_{n+k}^{(c)}}
  -\widehat{\Phi_n^{(c)}}}_{L^2}
  = O(k/c),\quad|k|\le c^{\theta_0}.
\end{equation}
Approaches: WKB stability
\cite{Zworski2012},
Sturm comparison for eigenvalues,
microlocal symbolic calculus.
\end{problem}

\begin{remark}[Kernel error
  is the critical bottleneck]
\label{rem:kernel-critical}
Naive estimate:
$\sum_{k\le c^{\theta_0}}c\cdot(k/c)^2
=O(c^{3\theta_0-1})$.
For $\theta_0\ge 1/3$:
this is $\Omega(1)$.
Rescuing this requires oscillation
in the off-diagonal kernel;
the same structure as the original
problem reappears one level down.
\end{remark}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Conditional on
(T\textsuperscript{*}), (L),

and~(C):
\begin{equation}\label{eq:reduction-chain}
  \underbrace{
    \mathrm{(T^*)}+\mathrm{(L)}
    +(\mathrm{R})
  }_{\text{Papers XVI--XVII}}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}} = o(1)
  \;\underset{\mathrm{(C)}}{\Longrightarrow}\;
  \text{H3 fails}
  \;\Rightarrow\;
  \text{H1 fails}.
\end{equation}
\begin{enumerate}[\rm(i)]
\item \textup{(T\textsuperscript{*})}:
  equivalent to $\mathcal{C}(c)\to 0$;
  reduced to Problem~\ref{prob:Tstar-cesaro}.
  Current status:

\item \textup{(L)} + \textup{(R)}:
  close $\Sigma_{\mathrm{int}}
  =\Sigma_{\mathrm{model}}+o(1)$;
  reduced to Problem~\ref{prob:rigorous-reduction}.
\item \textup{(C)}: closes the spectral step;
  Paper~XVII.
\end{enumerate}
\end{theorem}
