\begin{abstract}
Paper~XIV reduced H1 to
$\Sigma_{\mathrm{int}}=o(1)$.
This paper carries out a complete
scalar reduction and identifies
the precise remaining obstruction.

The exact equivalence
(Lemma~\ref{lem:S-equivalence}):
$\Sigma_{\mathrm{model}}=o(1)
\iff S(c)=o(c^{-(2-2\theta_0)})$
reduces everything to the
non-uniform Fourier sampling sum
$S(c)=\int K(x)
\sum_k\widehat{g}(xk/c^{\theta_0})dx$.
The Ces\`aro step gives
$S(c)=o(c^{\theta_0})$
(Corollary~\ref{cor:cesaro-closed}),
but the target is a factor
$c^{2-\theta_0}$ stronger.

The main new contributions:
\begin{enumerate}[\rm(1)]
\item \emph{Dirichlet obstruction}
  (Lemma~\ref{lem:dirichlet-obstruction}):
  signed $D^*_K$ alone gives
  $O(K^{1/2})$ with constant phase;
  no improvement over Ces\`aro
  from Dirichlet bounds alone.
\item \emph{Poisson bridge}
  (Lemma~\ref{lem:poisson-bridge}):
  Poisson summation converts
  the discrete $D^*_K$ into a
  continuous phase integral,
  rigorously justifying the
  transition from Dirichlet to
  Airy structure at the fold
  $\alpha_*$ ($\theta''(0)=0$,
  $\theta'''(0)\ne 0$).
\item \emph{Two routes for Paper~XVI}
  (Problem~\ref{prob:S-cancellation}):
  Route~A (Poisson~+~Airy,
  closes gap for $\theta_0>4/5$);
  Route~B (show all Poisson terms
  saturate at $O(c^{\theta_0})$,
  strong negative result).
\end{enumerate}
\end{abstract}

\begin{remark}[Core thesis]

\emph{All known mechanisms fail

\begin{lemma}[Singularity and
  splitting of $g$]
\label{lem:g-split}
With $g(\alpha):=1/|\theta''(\xi_0(\alpha))|$,
$\alpha_*:=\theta'(0)$,
$\chi\in C_c^\infty(\alpha_*\pm\delta)$:
\begin{equation}\label{eq:g-split}
  g = \gsing + \greg,\quad
  \gsing:= C_*|\alpha-\alpha_*|^{-1/2}\chi.
\end{equation}
$\greg\in C^\infty_c$ gives
$O(c^{-N})$ contributions.
The fold~\eqref{eq:fold} implies
$g\sim C_*/|\alpha-\alpha_*|^{1/2}$
and $|\widehat{\gsing}|=O(|\xi|^{-1/2})$.
\end{lemma}

\begin{lemma}[Exact scaling equivalence]
\label{lem:S-equivalence}
With $S(c):=\int K(x)\sum_k
\widehat{g}(xk/c^{\theta_0})dx$
and $K(c)=c^{\theta_0}$:
\begin{equation}\label{eq:exact-equiv}
  \boxed{\Sigma_{\mathrm{model}}=o(1)
  \;\iff\;
  S(c)=o(c^{-(2-2\theta_0)}).}
\end{equation}
Gap $=c^{2-\theta_0}$
$=$ frozen-phase saturation rate.
\end{lemma}

\begin{proposition}[$\Psi\in
  L^1(d\lambda/\lambda)$;
  Ces\`aro step]
\label{prop:Psi-L1}
Fubini--Tonelli:
$\int_1^\infty|\Psi(\lambda)|
d\lambda/\lambda
\lesssim\int|K(x)|(1+|x|^{-1/2})dx
<\infty$.
\end{proposition}

\begin{corollary}[$S(c)=o(c^{\theta_0})$]
\label{cor:cesaro-closed}
$S(c)=c^{\theta_0}\mathcal{C}(c)$,
$\mathcal{C}(c)\to 0$.
\end{corollary}

\subsection{The Dirichlet obstruction}

\begin{lemma}[Dirichlet obstruction]
\label{lem:dirichlet-obstruction}
Let $D^*_K(t):=\sum_{k=1}^K e^{ikt}$.
The signed integral:
\begin{equation}\label{eq:JK-def}
  J_K(x) :=
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  D^*_K(\beta x/c^{\theta_0})\,d\beta
  \;=\; Ce^{i\pi/4}
  \frac{c^{\theta_0/2}}{|x|^{1/2}}
  \sum_{k=1}^K k^{-1/2}.
\end{equation}
Hence $|J_K(x)|=\Theta(c^{\theta_0}/|x|^{1/2})$;
no improvement over Ces\`aro.
\end{lemma}

\begin{proof}
Fourier transform of $|\cdot|^{-1/2}$:
$\int|\beta|^{-1/2}e^{ik\beta x/c^{\theta_0}}
d\beta=C(c^{\theta_0}/kx)^{1/2}e^{i\pi/4}$.
Summing in $k$: $J_K(x)
=Ce^{i\pi/4}(c^{\theta_0}/|x|)^{1/2}
\sum_k k^{-1/2}
\sim 2Ce^{i\pi/4}c^{\theta_0/2}/|x|^{1/2}
\cdot K^{1/2}$.
With $K=c^{\theta_0}$:
$|J_K|=\Theta(c^{\theta_0}/|x|^{1/2})$.

\emph{Why no cancellation:}
the phase $e^{i\pi/4}$ is independent
of $k$; all modes add coherently.
The square-root singularity
$|\alpha-\alpha_*|^{-1/2}>0$
has no sign change near $\alpha_*$.
\end{proof}

\subsection{The Poisson bridge:
  from discrete to continuous
  phase}
\label{subsec:poisson}

\begin{lemma}[Poisson summation
  bridge]
\label{lem:poisson-bridge}
Let $f_c(t):=|t|^{-1/2}\chi(\alpha_*
+tc^{\theta_0}/x)
e^{i\alpha_*(xK/c^{\theta_0})}$.
Apply Poisson summation
$\sum_{k=1}^K f(k)
=\sum_{m\in\mathbb{Z}}
\int_1^K f(t)e^{2\pi imt}dt$
to convert the discrete sum
$\sum_k\widehat{\gsing}(xk/c^{\theta_0})$
into a continuous phase integral:
\begin{equation}\label{eq:poisson-expand}
  \sum_{k=1}^{K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})
  = \sum_{m\in\mathbb{Z}}
    I_m(x,c),
\end{equation}
where
\begin{equation}\label{eq:Im-def}
  I_m(x,c):=
  \int_1^{K(c)}
  \left(
    \int|\beta|^{-1/2}\chi(\alpha_*+\beta)
    e^{i\beta xt/c^{\theta_0}}d\beta
  \right)
  e^{2\pi imt}\,dt.
\end{equation}
The inner integral is the Fourier
transform of $\gsing$ at frequency
$xt/c^{\theta_0}$; using
$\widehat{\gsing}(xt/c^{\theta_0})
=O((c^{\theta_0}/xt)^{1/2})$:
\begin{equation}\label{eq:Im-bound}
  |I_m(x,c)|
  \le C\frac{c^{\theta_0/2}}{|x|^{1/2}}
  \int_1^{K}t^{-1/2}dt
  = O\!\left(\frac{c^{\theta_0}}{|x|^{1/2}}\right).
\end{equation}
This recovers the Ces\`aro bound
for each $m$ individually.
\end{lemma}

\begin{remark}[The $m=0$ term
  and degenerate stationary phase]
\label{rem:m0-airy}
For $m=0$:
\begin{equation}\label{eq:I0}
  I_0(x,c)=\int_1^{K(c)}
  \widehat{\gsing}(xt/c^{\theta_0})\,dt.
\end{equation}
This is a \emph{continuous}
integral of $\widehat{\gsing}$
along the ray $\xi=xt/c^{\theta_0}$.
The phase of $\widehat{\gsing}(\xi)$
comes from the Fourier transform of
the fold singularity:
\begin{equation}\label{eq:gsing-fourier}
  \widehat{\gsing}(\xi)
  = C_*e^{i\alpha_*\xi}
    \int|\beta|^{-1/2}
    \chi(\alpha_*+\beta)
    e^{i\beta\xi}\,d\beta.
\end{equation}
Near $\xi=0$ (i.e., $t\ll c^{\theta_0}/x$),
the fold condition
$\theta''(0)=0$, $\theta'''(0)\ne 0$
causes the phase of the inner
integral to degenerate
cubically in $\beta$:
\begin{equation}\label{eq:cubic-phase}
  \text{phase} \approx \beta\xi
  + \frac{\theta'''(0)}{6}\beta^3
  \cdot\text{(correction)}.
\end{equation}
At the Airy scale
$\xi\sim c^{-\theta_0/3}$
(i.e., $t\sim c^{2\theta_0/3}/x$),
the cubic term is $O(1)$
and $\widehat{\gsing}(\xi)$
behaves like an Airy function.
The integral $I_0(x,c)$
then has the structure of
a degenerate stationary-phase
integral in two variables
$(t,\beta)$, with degenerate
critical point at $(t,\beta)=(0,0)$.

\medskip
\noindent\textbf{Why this step
is the bridge:}
The Poisson summation converts
the discrete obstruction
(coherent phase $e^{i\pi/4}$
of Dirichlet modes) into a
continuous integral where
\emph{standard stationary-phase
methods apply}.
The fold geometry of $\theta$
is now directly visible
in the phase of the integrand.
This is the missing step
that rigorously connects
the Dirichlet structure
to the Airy mechanism.
\end{remark}

\begin{remark}[Status after
  Poisson bridge]
\label{rem:poisson-status}
The Poisson bridge is rigorous
(Euler--Maclaurin / Poisson
applied to smooth $f$;
the smoothness of the integrand
in $t$ follows from
$\widehat{\gsing}\in L^1$).
What remains for Paper~XVI:
\begin{itemize}[leftmargin=*]
\item \textbf{Route~A}:
  apply degenerate stationary phase
  to $I_0(x,c)$ in the regime
  $t\sim c^{2\theta_0/3}/x$
  (Airy scale);
  expected:
  $I_0(x,c)=O(c^{\theta_0/2}/|x|^{1/2})$;
  would close gap for $\theta_0>4/5$.
\item \textbf{Route~B}:
  show that $I_m$ for $m\ne 0$
  all saturate at
  $O(c^{\theta_0}/|x|^{1/2})$
  and that $I_0$ does too;
  this would establish
  $S(c)=\Theta(c^{\theta_0})$,
  a strong \emph{lower bound}
  (obstruction theorem).
\end{itemize}
\end{remark}

\subsection{$T(x,c)$ and the
  full structure}

\begin{definition}[$T(x,c)$]
\label{def:Txc}
The singular part of the
$k$-sum:
\begin{equation}\label{eq:Txc}
  T(x,c) :=
  \int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  D^*_{K(c)}(\beta x/c^{\theta_0})
  \,d\beta
  = \sum_{m\in\mathbb{Z}}
    I_m(x,c)
\end{equation}
(equality by
Lemma~\ref{lem:poisson-bridge}).
The Poisson decomposition
separates the $m=0$ Airy-type
contribution from the
high-frequency ($m\ne 0$) terms,
which are more rapidly oscillating
and thus more amenable to
standard stationary-phase bounds.
\end{definition}

%%══ 6. OPEN PROBLEMS
\section{Open Problems for Paper~XVI}
\label{sec:open}

\begin{problem}[Degenerate stationary
  phase for $I_0(x,c)$]
\label{prob:S-cancellation}
\emph{The single remaining
analytic obstacle.}

\medskip
\noindent\textbf{Object.}
The $m=0$ Poisson term:
\begin{equation}\label{eq:I0-problem}
  I_0(x,c)=\int_1^{K(c)}
  \widehat{\gsing}(xt/c^{\theta_0})\,dt.
\end{equation}

\medskip
\noindent\textbf{Route~A
  (Airy bound, positive result).}
\begin{enumerate}[\rm(A1)]
\item Expand
  $\widehat{\gsing}(\xi)$
  near $\xi=0$ using
  the cubic phase from
  $\theta'''(0)\ne 0$:
  $\widehat{\gsing}(\xi)
  \approx c_{1/2}|\xi|^{-1/2}
  e^{i\phi(\xi)}$,
  where $\phi(\xi)$ has a
  cubic term.
\item Split $[1,K]$ into:
  \emph{bulk} ($t\gg c^{2\theta_0/3}/x$):
  rapid oscillation, integrate
  by parts;
  \emph{Airy region}
  ($t\lesssim c^{2\theta_0/3}/x$):
  substitute $u=xt/c^{\theta_0}$,
  apply Airy function asymptotics.
\item Conclude
  $I_0(x,c)=O(c^{\theta_0/2}/|x|^{1/2})$;
  integrate in $x$:
  $S(c)=O(c^{\theta_0/2})$.
\item Check threshold:
  $c^{\theta_0/2}
  =o(c^{-(2-2\theta_0)})$
  $\iff\theta_0>4/5$.
\end{enumerate}

\medskip
\noindent\textbf{Route~B
  (strong obstruction,
  negative result).}
Show that the Airy approximation
produces a \emph{non-trivial
constant} in the integral
$\int|u|^{-1/2}\mathrm{Ai}(u)du
\ne 0$,
so that
$I_0(x,c)=\Theta(c^{\theta_0/2}/|x|^{1/2})$
(not $o$).
Then:
$S(c)=\Theta(c^{\theta_0/2})$;
combined with upper bound,
this gives
$S(c)=\Theta(c^{\theta_0/2})\ne o(c^{-(2-2\theta_0)})$
for $\theta_0\le 4/5$:
a \emph{conditional obstruction theorem}.

\medskip
\noindent\textbf{The key integral.}
In both routes, the central
object is:
\begin{equation}\label{eq:Ai-integral}
  \mathcal{A}:=
  \int_0^\infty u^{-1/2}
  \mathrm{Ai}(u)\,du.
\end{equation}
This is a classical integral;
$\mathrm{Ai}(u)$ decays super-exponentially
for $u>0$ (Schwartz),
so $\mathcal{A}<\infty$.
Numerically, $\mathcal{A}>0$
(Airy function is positive
for large $u$).
If $\mathcal{A}\ne 0$:
$I_0$ does not gain additional
phase cancellation,
and Route~B applies.
If there is cancellation
between $u<0$ and $u>0$
(oscillatory part of Ai):
Route~A may give strictly
better than $c^{\theta_0/2}$.
\end{problem}

\begin{problem}[Rigorous reduction]
\label{prob:rigorous-reduction}
(L), (R), (C) as before.
\end{problem}

\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Conditional on
$S(c)=o(c^{-(2-2\theta_0)})$,
(L), (R), (C):
$\Sigma_{\mathrm{int}}=o(1)
\Rightarrow\mathrm{H3\ fails}
\Rightarrow\mathrm{H1\ fails.}$
\end{theorem}

\begin{remark}[Conceptual summary]
\label{rem:concept-summary}
\begin{enumerate}[\rm(1)]
\item \emph{Dimension collapse}:
  operator $\to$ scalar $S(c)$.
\item \emph{Exact equivalence}
  and gap $=$ saturation rate.
\item \emph{Ces\`aro done}:
  $S(c)=o(c^{\theta_0})$.
\item \emph{Dirichlet obstruction}:
  signed modes add coherently;
  no improvement from $D^*_K$ alone.
\item \emph{Poisson bridge}:
  converts discrete sum
  $\to$ continuous phase integral;
  the fold geometry and Airy structure
  become visible in $I_0(x,c)$.
\item \emph{Paper~XVI}:
  decide Route~A vs.~B
  via the sign of
  $\mathcal{A}=\int_0^\infty
  u^{-1/2}\mathrm{Ai}(u)\,du$
  and the oscillatory contribution.
\end{enumerate}
\end{remark}
