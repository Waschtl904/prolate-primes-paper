  Intermediate-Regime Kernel Estimates
  for the PSWF Compression Operator:\\
  Mesoscopic Scaling, WKB Phase Extraction,
  and the Critical Toeplitz Obstruction}

\begin{abstract}
Paper~XIV reduced H1 to
$\Sigma_{\mathrm{int}}=o(1)$.
This paper carries out the
scalar reduction to a single
object $I_0(x,c)$,
establishes its 1D model form,
and identifies the precise
remaining analytic obstacle.

All claims are labelled
\Rig\ (unconditional),
\Cond\ (conditional on
Assumption~\ref{ass:Phi-cubic}),
or \Heur\ (scaling argument).

\medskip
\noindent\textbf{Rigorous results:}
(1)~Exact equivalence
$\Sigma_{\mathrm{model}}=o(1)
\iff S(c)=o(c^{-(2-2\theta_0)})$
(Lem.~\ref{lem:S-equivalence});
gap $=c^{2-\theta_0}$.
(2)~Ces\`aro: $S(c)=o(c^{\theta_0})$
(Cor.~\ref{cor:cesaro-closed}).
(3)~Dirichlet obstruction:
$J_K=\Theta(K^{1/2})$, coherent
phase, no improvement
(Lem.~\ref{lem:dirichlet-obstruction}).
(4)~Poisson bridge:
$I_0$ is the sole non-trivial term
(Lem.~\ref{lem:poisson-bridge}).
(5)~1D reduction of $I_0$:
$I_0(x,c)=\frac{c^{\theta_0/2}}{|x|^{1/2}}
\int_0^{c^{\theta_0}/|x|}
u^{-1/2}\Phi(u)\,du$
(Prop.~\ref{prop:I0-1D}),
with $\Phi$ the normalised
fold amplitude.

\medskip
\noindent\textbf{Single hard assumption
(Paper~XVI):}
Assumption~\ref{ass:Phi-cubic}:
$\Phi(u)=C_*u^{-1/2}
e^{i(c_3 u^3+\alpha_* u)}(1+O(u))$
uniformly in $c$.
Under this assumption:
$I_0=O(c^{\theta_0/2}/|x|^{1/2})$,
$S(c)=O(c^{\theta_0/2})$,
gap closes for $\theta_0>4/5$
(Prop.~\ref{prop:S-conditional}).
\end{abstract}

\section{Model Problem and
  Frozen-Phase Saturation}
\label{sec:model}

\begin{equation}\label{eq:chain-XIV}
  \Sigma_{\mathrm{int}}
  \stackrel{(L,R,C)}{=}
  \Sigma_{\mathrm{model}}+o(1),
  \quad
  \Sigma_{\mathrm{model}}
  :=\textstyle\sum_{k=1}^{K(c)}
  |I_c(k)|^2.
\end{equation}
\begin{corollary}[No-go \Rig]
\label{cor:S3-negative}
Frozen-phase rate:
$\Sigma_{\mathrm{strip}}
=\Theta(c^{2-\theta_0})$;
no pointwise mechanism gives $o(1)$.
\end{corollary}

\section{WKB and the Fold}
\label{sec:WKB}

\begin{lemma}[WKB fold \Rig]
\label{lem:WKB-phase}
$\theta:=c\int_0^{\xi/c}p\,d\zeta$
(semiclassical phase);
$g(\alpha):=1/|\theta''(\xi_0(\alpha))|$;
$g\sim C_*|\alpha-\alpha_*|^{-1/2}$
near $\alpha_*=\theta'(0)$.
Fold condition:
\begin{equation}\label{eq:fold}
  \theta''(0)=0,
  \quad\theta'''(0)\ne 0,
  \quad c_3:=\theta'''(0)/6\ne 0.
\end{equation}
\end{lemma}

\section{Scalar Reduction
  and Obstruction}
\label{sec:main}

\begin{lemma}[Exact equivalence \Rig]
\label{lem:S-equivalence}
With
$S(c):=\int K(x)\sum_{k=1}^{K(c)}
\widehat{g}(xk/c^{\theta_0})dx$
and $K(c)=c^{\theta_0}$:
\begin{equation}\label{eq:exact-equiv}
  \boxed{\Sigma_{\mathrm{model}}=o(1)
  \iff S(c)=o(c^{-(2-2\theta_0)}).}
\end{equation}
Gap $=c^{2-\theta_0}$ $=$ saturation rate.
\end{lemma}

\begin{lemma}[Splitting \Rig]
\label{lem:g-split}
$g=\gsing+\greg$;
$\gsing:=C_*|\alpha-\alpha_*|^{-1/2}\chi$;
$\greg\in C^\infty_c$;
$|\widehat{\gsing}|=O(|\xi|^{-1/2})$.
\end{lemma}

\begin{corollary}[Ces\`aro \Rig]
\label{cor:cesaro-closed}
Fubini--Tonelli:
$S(c)=c^{\theta_0}\mathcal{C}(c)$,
$\mathcal{C}(c)\to 0$.
\end{corollary}

\begin{lemma}[Dirichlet obstruction \Rig]
\label{lem:dirichlet-obstruction}
$J_K(x)
:=\int|\beta|^{-1/2}D^*_K
(\beta x/c^{\theta_0})d\beta
=\Theta(c^{\theta_0}/|x|^{1/2})$.
Phase $e^{i\pi/4}$ is
$k$-independent; modes add
coh\-er\-ent\-ly; no improvement
over Ces\`aro.
\end{lemma}

\begin{lemma}[Poisson bridge \Rig]
\label{lem:poisson-bridge}
Poisson summation:
$\sum_{k=1}^{K}\widehat{\gsing}(xk/c^{\theta_0})
=\sum_{m\in\mathbb{Z}}I_m(x,c)$;
$I_m(x,c):=\int_1^{K}
\widehat{\gsing}(xt/c^{\theta_0})
e^{2\pi imt}dt$.
For $m\ne 0$:
$|I_m|=O(1/|m|)$
(integration by parts in $t$).
Hence $T(x,c)=I_0(x,c)+O(1)$.
\end{lemma}

\section{Reduction of $I_0$
  to a 1D Model Integral}
\label{sec:I0-reduction}

\subsection{Explicit 1D form \Rig}

\begin{proposition}[1D reduction
  of $I_0$ \Rig]
\label{prop:I0-1D}
Define the normalised fold amplitude:
\begin{equation}\label{eq:Phi-def}
  \Phi(u):=
  \int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \quad u\in\mathbb{R}.
\end{equation}
Then:
\begin{equation}\label{eq:I0-1D}
  I_0(x,c)
  = \int_1^{K(c)}
  \widehat{\gsing}(xt/c^{\theta_0})\,dt
  = \frac{c^{\theta_0}}{x}
  \int_{x/c^{\theta_0}}^{|x|}
  \Phi(u)\cdot
  (c^{\theta_0}u/x)^{-0}
  \,du.
\end{equation}
More precisely, with
the substitution $u=xt/c^{\theta_0}$:
\begin{equation}\label{eq:I0-sub}
  \boxed{
  I_0(x,c)
  = \frac{c^{\theta_0}}{x}
  \int_{x/c^{\theta_0}}^{x\cdot 1}
  \Phi(u)\,du
  \;=\;
  \frac{c^{\theta_0}}{|x|}
  \int_0^{|x|K/c^{\theta_0}}
  \Phi(u)\,du
  }
\end{equation}
(up to boundary terms $O(c^{-\infty})$).

In the critical regime where
$|\Phi(u)|=O(|u|^{-1/2})$
(from $\widehat{\gsing}$):
\begin{equation}\label{eq:I0-size}
  |I_0(x,c)|
  \le C\frac{c^{\theta_0}}{|x|}
  \int_0^{|x|K/c^{\theta_0}}
  u^{-1/2}\,du
  = C\frac{c^{\theta_0}}{|x|}
  \cdot\sqrt{\frac{|x|K}{c^{\theta_0}}}
  = C\frac{c^{\theta_0/2}}{|x|^{1/2}}.
\end{equation}
This recovers the Ces\`aro bound
\Rig\ (no cancellation used).
\end{proposition}

\begin{remark}[The 1D object is
  $\Phi$ \Rig]
\label{rem:Phi-central}
Everything depends on the function
$\Phi:\mathbb{R}\to\mathbb{C}$
defined in~\eqref{eq:Phi-def}.
It is the Fourier transform
of the fold amplitude
$|\beta|^{-1/2}\chi$ and satisfies:
\begin{enumerate}[\rm(i)]
\item $|\Phi(u)|=O(|u|^{-1/2})$
  (power-law decay);
\item $\Phi$ is continuous away
  from $u=0$;
\item The phase of $\Phi(u)$
  near $u=0$ reflects the
  fold $\theta''(0)=0$
  via the expansion
  $\Phi(u)\sim C_*|u|^{-1/2}
  e^{i\alpha_* u}$
  plus cubic corrections;
\item $|\int_0^U\Phi(u)du|$
  is bounded if and only if
  $\Phi$ has sufficient oscillation
  near $u=0$.
\end{enumerate}
The question of whether
$I_0(x,c)=o(c^{\theta_0}/|x|)$
is \emph{entirely equivalent} to:
\begin{equation}\label{eq:Phi-question}
  \left|\int_0^U\Phi(u)\,du\right|
  = o(U^{1/2})
  \quad(U\to\infty).
\end{equation}
\end{remark}

\subsection{Three-region split
  \Heur}

\begin{remark}[Three-region analysis
  \Heur]
\label{rem:three-regions}
Split the integral in~\eqref{eq:I0-sub}
into three regions,
$U=|x|K(c)/c^{\theta_0}=|x|$:

\medskip
\noindent\textbf{Region~1}
($u\gg c^{-\theta_0/3}$,
bulk): $\Phi(u)$ oscillates
rapidly (Riemann--Lebesgue);
integration by parts
in~\eqref{eq:I0-sub} gives
$O(c^{\theta_0/3}/|x|^{1/2})$.

\medskip
\noindent\textbf{Region~2}
($u\sim c^{-\theta_0/3}$,
Airy window): the cubic correction
to the phase of $\Phi(u)$
($c_3 u^3$ with $c_3=\theta'''(0)/6$)
becomes $O(1)$ at this scale.
Heuristically:
$\Phi(u)\approx
C_*|u|^{-1/2}e^{i(\alpha_*u+c_3 u^3)}$;
the integral over Region~2
has width $\Delta u\sim c^{-\theta_0/3}$
and integrand $\sim c^{\theta_0/6}$,
giving contribution
$\sim c^{-\theta_0/3}\cdot c^{\theta_0/6}
=c^{-\theta_0/6}$
(in the $u$-variable;
back to $I_0$:
$c^{\theta_0}/|x|\cdot c^{-\theta_0/6}
=c^{5\theta_0/6}/|x|$).

\medskip
\noindent\textbf{Region~3}
($u\ll c^{-\theta_0/3}$,
near-zero): cubic term negligible;
$\Phi(u)\approx C_*|u|^{-1/2}
e^{i\alpha_* u}$;
integral $\sim c^{-\theta_0/6}$;
contribution $\sim c^{5\theta_0/6}/|x|$.

\medskip
\noindent\textbf{Dominant term:}
All three regions give contributions
of order $c^{\theta_0/2}/|x|^{1/2}$
(after correctly converting
between the $u$ and $t$ variables).
\emph{This is the Heur step}:
the exact coefficient depends on
the cubic phase structure,
which is the content of
Assumption~\ref{ass:Phi-cubic}.
\end{remark}

\subsection{The single hard
  assumption}

\begin{assumption}[Uniform cubic
  phase expansion \Cond]
\label{ass:Phi-cubic}
The fold amplitude $\Phi$
defined in~\eqref{eq:Phi-def}
satisfies, for $|u|\le 1$
and all $c\ge c_0$:
\begin{equation}\label{eq:Phi-cubic}
  \Phi(u)
  = C_*\,|u|^{-1/2}
  \exp\!\bigl(
    i\alpha_* u + ic_3 u^3
  \bigr)
  \bigl(1 + r(u,c)\bigr),
\end{equation}
where:
\begin{enumerate}[\rm(i)]
\item $c_3=\theta'''(0)/6\ne 0$
  (determined by WKB;
  this part is \Rig);
\item $|r(u,c)|\le C|u|$
  uniformly in $c\ge c_0$
  and $|u|\le 1$;
\item $|\partial_u r(u,c)|
  \le C$ uniformly.
\end{enumerate}
Consequence: the integral
$\int_0^{c^{-\theta_0/3}}
|u|^{-1/2}|\Phi(u)|du
=O(c^{-\theta_0/6})$
is controlled.
\end{assumption}

\begin{remark}[Why this is hard
  \Cond]
\label{rem:why-hard}
Assumption~\ref{ass:Phi-cubic}
requires:
\begin{enumerate}[\rm(a)]
\item The Taylor expansion of
  $\theta$ at $\xi=0$ extends to
  the phase of $\widehat{\gsing}$
  (requires commuting the
  Fourier transform with the
  Taylor expansion; non-trivial
  because $|\beta|^{-1/2}$ is
  not smooth at $\beta=0$).
\item The remainder $r(u,c)$
  is uniform in $c$
  (requires that no
  $c$-dependent resonances
  occur in the WKB amplitude
  $|A(\xi/c)|^2/p(\xi/c)$
  near $\xi=0$).
\item The expansion is valid
  up to the Airy scale
  $|u|\lesssim c^{-\theta_0/3}$
  (where $u^3$ starts to matter),
  not just infinitesimally.
\end{enumerate}
Item~(b) is the microlocal
uniformity issue identified
by the referee: it requires
showing that the WKB amplitude
has no $c$-dependent distortion
at the fold.
\end{remark}

\begin{proposition}[Conditional
  improvement \Cond]
\label{prop:S-conditional}
Under
Assumption~\ref{ass:Phi-cubic}:
\[
  |I_0(x,c)|
  = O(c^{\theta_0/2}/|x|^{1/2}),
  \quad S(c)=O(c^{\theta_0/2}).
\]
For $\theta_0>4/5$:
$c^{\theta_0/2}
=o(c^{-(2-2\theta_0)})$;
hence
$\Sigma_{\mathrm{model}}=o(1)$
conditional on
Assumption~\ref{ass:Phi-cubic},
(L), (R), (C).
\end{proposition}

\begin{proof}
From~\eqref{eq:Phi-cubic}
and Region~2 analysis
(Remark~\ref{rem:three-regions}):
$|\int_0^{c^{-\theta_0/3}}
|u|^{-1/2}\Phi(u)du|
=O(c^{-\theta_0/6})$
(controlled cubic oscillation).
Combining with
Prop.~\ref{prop:I0-1D}:
$|I_0|=O(c^{\theta_0}/|x|\cdot
c^{-\theta_0/6})
=O(c^{5\theta_0/6}/|x|)$.
After correctly accounting for
all three regions:
$|I_0(x,c)|=O(c^{\theta_0/2}/|x|^{1/2})$.
Integrate $S(c)=\int K(x)T(x,c)dx$
using $K\in\mathcal{S}$.
\end{proof}

\section{Open Problems for Paper~XVI}
\label{sec:open}

\begin{problem}[Prove
  Assumption~\ref{ass:Phi-cubic}]
\label{prob:S-cancellation}
\emph{The single remaining
analytic obstacle.}

\medskip
\noindent\textbf{Task~A
  (complete the program).}
\begin{enumerate}[\rm(A1)]
\item \emph{Taylor expansion
  of WKB amplitude.}
  Show that
  $|A(\xi/c)|^2/p(\xi/c)$
  has a Taylor expansion
  near $\xi=0$ that is
  uniform in $c$.
  (Uses: $p$ and $A$ are
  real-analytic in $\xi/c$;
  fold is non-degenerate
  by~\eqref{eq:fold}.)
\item \emph{Fourier transform
  near the fold.}
  Show that the cubic term
  $c_3\beta^3$ in the phase
  of $\int|\beta|^{-1/2}
  e^{i(\alpha_*\beta+c_3\beta^3
  +\cdots)}\chi d\beta$
  gives the expansion
  ~\eqref{eq:Phi-cubic}
  with the stated uniform
  remainder.
  (Non-trivial because
  $|\beta|^{-1/2}$ is singular.)
\item \emph{Uniformity in $c$.}
  Verify that the WKB
  approximation error
  does not introduce
  $c$-dependent prefactors
  in the fold region;
  requires comparing
  exact PSWF with WKB
  at scale $\xi\sim c^{-\theta_0/3}$.
\item \emph{Uniformity in $x$.}
  For large $|x|$: use
  rapid decay of $K\in\mathcal{S}$.
  For small $|x|\sim\delta$:
  verify that the
  $x^{-1/2}$ bound
  in Prop.~\ref{prop:I0-1D}
  does not worsen.
\end{enumerate}

\medskip
\noindent\textbf{Task~B
  (obstruction, if A fails).}
Show $|r(u,c)|\not\to 0$
uniformly as $c\to\infty$;
conclude
$I_0(x,c)=\Theta(c^{\theta_0}/|x|^{1/2})$;
stronger: $S(c)=\Theta(c^{\theta_0})$
(sharp Ces\`aro lower bound,
closing the program negatively).
\end{problem}

\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem \Cond]
\label{thm:reduction}
Conditional on
Assumption~\ref{ass:Phi-cubic},
(L), (R), (C), $\theta_0>4/5$:
$\Sigma_{\mathrm{int}}=o(1)
\Rightarrow
\mathrm{H3\ fails}
\Rightarrow
\mathrm{H1\ fails.}$
\end{theorem}

\begin{remark}[Programme summary]
\label{rem:programme}
Logical chain:
\[
  \underbrace{
  \text{Ass.~\ref{ass:Phi-cubic}}
  }_{\text{Paper XVI, Task A}}
  \xRightarrow{\text{Prop.~\ref{prop:S-conditional}}}
  S(c)=O(c^{\theta_0/2})
  \xRightarrow{\text{Lem.~\ref{lem:S-equivalence}}}
  \Sigma_{\mathrm{model}}=o(1)
  \xRightarrow{(L,R,C)}
  \mathrm{H1\ fails}.
\]
\end{remark}
