\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to a single
open problem: the intermediate-regime
Hilbert--Schmidt estimate
$\Sigma_{\mathrm{int}} = o(1)$
for $M(c)\sim\beta c$.
This paper identifies the core
analytic obstruction as a
\emph{critical-density Toeplitz
energy problem} and develops
the analytic toolkit required
for the reduction to this formulation.

The central object is the oscillatory
kernel family
\[
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi,
  \quad 1\le k\le c^{\theta_0},
\]
and the associated sum
$\Sigma_{\mathrm{model}}(c)
:= \sum_{k=1}^{\lfloor c^{\theta_0}\rfloor}
|I_c(k)|^2$.
The Dirichlet-kernel reformulation
(\S\ref{sec:model}) and mesoscopic
rescaling identify $\Sigma_{\mathrm{model}}$
with the critical Toeplitz energy
functional
$\Tcc(\psi) := \ip{\Dc\psi}{\psi}$
at the double-scaling
regime $K(c)\cdot h = 1$.

The WKB phase extraction
(Lemmas~\ref{lem:WKB-phase},
\ref{lem:phase-nondeg})
identifies $\theta(\xi)$ explicitly,
proves $\theta'\ne 0$ and
$\theta''\ne 0$ for $\xi\ne 0$,
and establishes that both naive
van~der~Corput
(Corollary~\ref{cor:curvature-available})
and the Weyl/UDPL/exponent-pair
combination
(Proposition~\ref{prop:W4-decoupling})
are insufficient to close the
$\Sigma_{\mathrm{model}}=o(1)$ claim:
the obstruction is not
\emph{analytic weakness} but
\emph{lack of cancellation after
phase freezing}
(\S\ref{sec:toeplitz},
Remark~\ref{rem:core-thesis}).

Section~\ref{sec:toeplitz} develops
the key structural reinterpretation:
$\Sigma_{\mathrm{model}}$ is a
\emph{superposition of shifted
Fourier samplings} of $|\widehat{\psi}|^2$
along the deformed grid
$\{k\alpha_0(\xi_0)/c^{\theta_0}\}$.
A coarea substitution
(Proposition~\ref{prop:alpha-oscillation})
reduces (T\textsuperscript{*}) to
the \emph{weighted Riemann--Lebesgue
condition}
\begin{equation}\label{eq:wRL-abstract}
  \int_{\mathbb{R}} K(x)\,
  \widehat{g}(\lambda x)\,dx
  \;\longrightarrow\; 0
  \qquad(\lambda\to\infty),
\end{equation}
where $K = \psi*\widetilde{\psi}\in\mathcal{S}$
and $g = 1/|\theta''(\xi_0(\cdot))| \in L^1_{\mathrm{loc}}$.
Crucially, the curvature weight $g$ has
an integrable singularity of type
$|\alpha-\alpha_*|^{-1}$ at the
turn-point image $\alpha_*$
(Lemma~\ref{lem:g-singularity}),
so its Fourier transform grows
logarithmically and
\emph{polynomial pointwise decay
of $\widehat{g}$ cannot be assumed}.
However, the rapid decay of $K$
can absorb this logarithmic growth,
and the correct target is the
weaker \emph{weighted} condition
\eqref{eq:wRL-abstract}
(Lemma~\ref{lem:weighted-RL}).

The main structural output is the
\emph{Reduction Theorem}
(Theorem~\ref{thm:reduction}):
\[
  \mathrm{(T^*)} + \mathrm{(L)}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}}=o(1),
  \qquad
  \mathrm{(C)}
  \;\Longrightarrow\;
  \text{H1 fails},
\]
where (T\textsuperscript{*}) is
equivalent to the weighted
Riemann--Lebesgue condition
\eqref{eq:wRL-abstract}.
\end{abstract}

\section{The Intermediate-Regime
  Model Problem}
\label{sec:model}

\begin{problem}[Model problem]
\label{prob:model-intermediate}
With $I_c(k)$ as in~\eqref{eq:model-reduction}
and $K(c):=\lfloor c^{\theta_0}\rfloor$:
\begin{equation}\label{eq:model-sigma}
  \Sigma_{\mathrm{model}}(c)
  := \sum_{k=1}^{K(c)}|I_c(k)|^2
  \stackrel{?}{=} o(1).
\end{equation}
Kernel reformulation:
\begin{equation}\label{eq:kernel-rewrite}
  \Sigma_{\mathrm{model}}
  = \iint w(\xi)w(\eta)F_c(\xi,\eta)\,d\xi\,d\eta,
  \quad
  F_c(\xi,\eta)
  := \sum_{k=1}^{K(c)} e^{ik(\theta(\xi)-\theta(\eta))/c}.
\end{equation}
Mesoscopic rescaling at $\xi_0$:
\begin{equation}\label{eq:meso-F}
  F_c(u,v)
  = D_{K(c)}(h\,\theta'(\xi_0)(u-v)),
  \quad h=c^{-\theta_0},\;
  K(c)\cdot h=1.
\end{equation}
\emph{Double-scaling Szeg\H{o} regime.}
\end{problem}

\label{prop:S3}
$\Sigma_{\mathrm{strip}}(c)
= c^{2-\theta_0}C(\psi,\alpha_0)

= \Theta(c^{2-\theta_0})$.
\emph{Scope}: frozen-$\xi_0$ only;

\end{proposition}

$\Sigma_{\mathrm{strip}}\to+\infty$

\end{corollary}

\begin{remark}[Two escapes]
\label{rem:two-escapes}
\textbf{Escape~A} (global $\xi_0$-variation):

\textbf{Escape~B} (PSWF orthogonality):

\end{remark}

\section{WKB Phase Extraction}

\begin{lemma}[WKB phase]
\label{lem:WKB-phase}
$\theta(\xi):= c\int_0^{\xi/c}p(\zeta;\alpha)\,d\zeta$,\;
$w(\xi):=|A(\xi/c;\alpha)|^2/p(\xi/c;\alpha)$,\;
$p(\zeta;\alpha):=\sqrt{(\lambda(\alpha)-\zeta^2)/(1-\zeta^2)}$.
\end{lemma}

\begin{lemma}[Phase non-degeneracy]
\label{lem:phase-nondeg}
(i) $\theta'>0$;\;
(ii) $\theta''(\xi)\ne 0$ for $\xi\ne 0$,
     with $\theta''(\xi)\sim -\xi/c$
     near $\xi=0$;\;
(iii) $\theta''(0)=0$, $\theta'''(0)\ne 0$;\;
(iv) $\xi_0\mapsto\alpha_0(\xi_0):=\theta'(\xi_0)$
     strictly monotone,
     $\alpha_0'=\theta''$.
\end{lemma}

\begin{proof}
Direct differentiation of $p(\xi/c;\alpha)$;
$\lambda(\alpha)=\sin^2(\pi\alpha/2)+O(1/c)$
\cite{Osipov2013}.
\end{proof}

\begin{corollary}[vdC insufficient]
\label{cor:curvature-available}
$\sum_k|I_c(k)|^2\lesssim(c/\varepsilon)\log c
=\Omega(c)$.
\end{corollary}

%%═══ 6. THE CRITICAL TOEPLITZ OBSTRUCTION
\section{The Critical Toeplitz Obstruction}
\label{sec:toeplitz}

\begin{remark}[Core thesis]
\label{rem:core-thesis}
\begin{center}
\emph{The obstruction is not analytic
weakness, but absence of cancellation
after phase freezing.}
\end{center}
Every individual tool operates
pointwise in $\xi_0$;
necessary cancellation is global.
\end{remark}

\begin{remark}[Sampling reinterpretation]
\label{rem:sampling-rewrite}
\begin{equation}\label{eq:sampling}
  \Sigma_{\mathrm{model}}
  \approx
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \sum_{k=1}^{K(c)}
  \left|\widehat{\psi}\!\left(
    \tfrac{k\alpha_0(\xi_0)}{c^{\theta_0}}
  \right)\right|^2
  d\xi_0.
\end{equation}
The grids $\{k\alpha_0(\xi_0)/c^{\theta_0}\}$
shift as $\xi_0$ varies;
destructive interference upon
$\xi_0$-integration is the key question.
\end{remark}

\begin{lemma}[Singularity of curvature
  weight]
\label{lem:g-singularity}
Let $g(\alpha):=1/|\theta''(\xi_0(\alpha))|$
be the Jacobian of the coarea substitution
$\xi_0\mapsto\alpha=\alpha_0(\xi_0)$.
Let $\alpha_* := \alpha_0(0) = \theta'(0) > 0$
be the image of the turn-point $\xi_0=0$
\textup{(}where $\theta''(0)=0$\textup{)}.

Then:
\begin{enumerate}[\rm(i)]
\item $g\in C^\infty(
  [\alpha_{\min},\alpha_{\max}]\setminus\{\alpha_*\})$.
\item Near $\alpha_*$, since
  $\theta''(\xi_0)\sim -\xi_0/c$
  and $\alpha-\alpha_*\sim\theta''(\xi_0)\cdot\xi_0$
  (by Taylor expansion of
  $\alpha_0(\xi_0)$ at $0$,
  using $\alpha_0''(0)=\theta'''(0)\ne 0$):
  \begin{equation}\label{eq:g-sing}
    g(\alpha)
    \sim \frac{C_*}{|\alpha-\alpha_*|^{1/2}}
    \qquad(\alpha\to\alpha_*),
  \end{equation}
  so $g\in L^1([\alpha_{\min},\alpha_{\max}])$.
\item The Fourier transform $\widehat{g}$
  satisfies:
  \begin{equation}\label{eq:g-hat-log}
    |\widehat{g}(\xi)|
    \lesssim
    C_\infty + C_*|\xi|^{-1/2}
    \qquad(\xi\ne 0),
  \end{equation}
  where the $|\xi|^{-1/2}$ term
  comes from the Fourier transform of
  $|\alpha|^{-1/2}$.
  In particular,
  $\widehat{g}\in L^\infty$ and
  $|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$
  as $|\xi|\to\infty$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i): $g = 1/|\theta''(\xi_0(\cdot))|$ is smooth
away from $\alpha_*$ since $\xi_0(\alpha)$
is smooth and $\theta''\ne 0$ for $\xi_0\ne 0$.

(ii): Near $\xi_0=0$,
$\alpha-\alpha_* = \alpha_0(\xi_0)-\alpha_0(0)
= \alpha_0''(0)\xi_0^2/2 + O(\xi_0^3)$
(note $\alpha_0'(0)=\theta''(0)=0$,
$\alpha_0''(0)=\theta'''(0)\ne 0$).
Hence $|\xi_0|\sim C|\alpha-\alpha_*|^{1/2}$
near $\alpha_*$, and
$g(\alpha)=1/|\theta''(\xi_0(\alpha))|
\sim c/|\xi_0(\alpha)|
\sim C_*/|\alpha-\alpha_*|^{1/2}$.

(iii): The Fourier transform of
$\alpha\mapsto|\alpha-\alpha_*|^{-1/2}$
is (up to phase and normalisation)
$C|\xi|^{-1/2}$; the smooth
remainder contributes $O(|\xi|^{-N})$
for any $N$ after subtraction.
\end{proof}

\begin{remark}[Why polynomial pointwise
  decay of $\widehat{g}$ is not assumed]
\label{rem:g-hat-not-poly}
Lemma~\ref{lem:g-singularity}(iii)
shows $|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$,
which corresponds to $\delta=1/2$
in any polynomial-decay assumption.
This is the \emph{actual, provable decay}
for this specific $g$;
it is \emph{not} logarithmic growth
(the singularity is $|\alpha|^{-1/2}$,
not $|\alpha|^{-1}$).

However, the key point remains:
\emph{one cannot simply assume
polynomial decay of $\widehat{g}$
a priori} and then apply
Lemma~\ref{lem:quantitative-RL};
one must use the \emph{actual
structure of $g$} (Lemma~\ref{lem:g-singularity})
or work with the weaker
weighted Riemann--Lebesgue condition
(Lemma~\ref{lem:weighted-RL}).
\end{remark}

\begin{proposition}[Alpha-oscillation strategy]
\label{prop:alpha-oscillation}
Let $K:=\psi*\widetilde{\psi}\in L^1$
(auto-correlation).
Under the coarea substitution
$\xi_0\mapsto\alpha=\alpha_0(\xi_0)$
with Jacobian $g(\alpha)=d\xi_0/d\alpha$:
\begin{equation}\label{eq:alpha-integral}
  \int_{-c_0}^{c_0}
  f_c(\alpha_0(\xi_0))\,d\xi_0
  = \int_{\mathbb{R}} K(x)
    \sum_{k=1}^{K(c)}
    \widehat{g}\!\left(\frac{xk}{c^{\theta_0}}\right)
    dx.
\end{equation}

\noindent\textbf{Sufficient condition~(RL).}
\begin{equation}\label{eq:RL-condition}
  \frac{1}{K(c)}
  \sum_{k=1}^{K(c)}
  \int_{\mathbb{R}}
  |K(x)|\cdot
  |\widehat{g}(xk/c^{\theta_0})|
  \,dx = o(1)
\end{equation}
implies (T\textsuperscript{*}).
\end{proposition}

\begin{lemma}[Weighted Riemann--Lebesgue
  condition]
\label{lem:weighted-RL}
The correct (and weaker) target for
(T\textsuperscript{*}) is the
\emph{weighted Riemann--Lebesgue
condition}:
\begin{equation}\label{eq:wRL}
  \int_{\mathbb{R}} K(x)\,
  \widehat{g}(\lambda x)\,dx
  \;\longrightarrow\; 0
  \qquad(\lambda\to\infty),
\end{equation}
where $\lambda = k/c^{\theta_0}$
and one averages over $k=1,\ldots,K(c)$.

\medskip
\noindent\textbf{Why this is weaker
  than pointwise decay of $\widehat{g}$:}
Condition~\eqref{eq:wRL} requires
only that the $K$-weighted average
of $\widehat{g}$ decays; it does
\emph{not} require
$\widehat{g}(\xi)\to 0$ uniformly
let alone $|\widehat{g}(\xi)|=O(|\xi|^{-\delta})$.

\medskip
\noindent\textbf{Sufficient conditions
  for~\eqref{eq:wRL}:}
\begin{enumerate}[\rm(a)]
\item \emph{(Qualitative)}:
  $K\in L^1$ and $g\in L^1$.
  Then $\widehat{g}\in C_0$ (Riemann--Lebesgue),
  and~\eqref{eq:wRL} follows by
  dominated convergence provided
  $|K(x)|\cdot|\widehat{g}(\lambda x)|
  \le |K(x)|\cdot\norm{\widehat{g}}_{L^\infty}$
  and $K\in L^1$.
  \emph{This is already available}:
  $g\in L^1$ (Lemma~\ref{lem:g-singularity}(ii))
  gives $\widehat{g}\in C_0$,
  so $\widehat{g}(\lambda x)\to 0$
  for each $x\ne 0$,
  and DCT gives~\eqref{eq:wRL}.
  \emph{The $k$-averaging is then automatic.}

\item \emph{(Quantitative, for rate)}:
  $|\widehat{g}(\xi)|=O(|\xi|^{-\delta})$
  with $\delta>0$;
  the rate is then $O(\lambda^{-\delta})$.
  By Lemma~\ref{lem:g-singularity}(iii),
  $\delta=1/2$ is provably available.

\item \emph{(Fastest, exploiting $K$)}:
  If $K\in\mathcal{S}$ (Schwartz class,
  which holds when $\psi\in\mathcal{S}$),
  then $K$ decays faster than any power,
  and even $\widehat{g}(\lambda x)$ growing
  mildly in $x$ (e.g., $\sim\log|x|$)
  is absorbed. Specifically:
  for any $N$,
  $\int|K(x)|\cdot(1+|\lambda x|)^{1/2}\,dx
  \lesssim \lambda^{-N}\norm{K}_{H^N}$,
  which by~\eqref{eq:g-hat-log}
  gives~\eqref{eq:wRL} with
  rate $O(\lambda^{-1/2})$.
\end{enumerate}

\medskip
\noindent\textbf{Consequence for (T\textsuperscript{*}):}
Under condition~(a),
\eqref{eq:wRL} holds as a qualitative
statement: $\int K\cdot\widehat{g}(\lambda\cdot)\,dx\to 0$.
Under condition~(c), the rate is
$O(\lambda^{-1/2})$, i.e.,
$O(c^{-\theta_0/2})$ for $\lambda=k/c^{\theta_0}$
with $k\sim c^{\theta_0}$.
Both suffice for~\eqref{eq:RL-condition}
and hence for (T\textsuperscript{*}).
\end{lemma}

\begin{proof}
Condition~(a):
$g\in L^1\Rightarrow\widehat{g}\in C_0$,
i.e., $\widehat{g}(\xi)\to 0$ as $|\xi|\to\infty$.
For $x\ne 0$, $\lambda\to\infty$:
$\widehat{g}(\lambda x)\to 0$.
DCT with dominating function
$|K(x)|\cdot\norm{\widehat{g}}_{L^\infty}\in L^1$
gives~\eqref{eq:wRL}.

Condition~(c):
write $|\widehat{g}(\lambda x)|\le C(1+|\lambda x|)^{-1/2}$
by~\eqref{eq:g-hat-log};
$\int|K(x)|(1+|\lambda x|)^{-1/2}\,dx
\le \lambda^{-1/2}\int|K(x)||x|^{-1/2}\,dx
+ \int_{|x|<\lambda^{-1}}|K(x)|\,dx$.
Both terms are $O(\lambda^{-1/2})$
for $K\in\mathcal{S}$.
\end{proof}

\begin{remark}[Role of $K$:
  absorbing logarithmic growth]
\label{rem:K-cancellation}
The interaction between $K$ and $\widehat{g}$
is the key mechanism.
Even if $\widehat{g}(\xi)$ grew as
$\log|\xi|$ (the worst case for a
$|\alpha|^{-1}$-type singularity),
the integral $\int K(x)\log|\lambda x|\,dx
= \log\lambda\cdot\int K(x)\,dx
+ \int K(x)\log|x|\,dx$
would give $o(1)$ \emph{provided}
$\int K(x)\,dx = \widehat{K}(0) = 0$.
For our $K=\psi*\widetilde\psi$:
$\widehat{K}(0)=|\widehat\psi(0)|^2$.
If $\psi$ is a bandlimited cutoff with
$\widehat\psi(0)=0$ (zero mean), this
gives automatic cancellation of the
logarithm.
This symmetry/cancellation check is
a concrete subtask for Paper~XVI.
\end{remark}

\begin{remark}[Moving-symbol Toeplitz
  structure]
\label{rem:moving-symbol}
\begin{center}
\emph{A Toeplitz operator whose
symbol moves with basepoint $\xi_0$:
a \textbf{moving-symbol} Toeplitz problem.}
\end{center}
Classical Szeg\H{o} theory
\cite{Simon2011,BottcherSilbermann1990}
fixes the symbol.
Here the three limits
$k\to\infty$, $c\to\infty$,
$\xi_0$-integration do not commute.
Alternative viewpoint:
this is \emph{randomised sampling}
of $|\widehat\psi|^2$ along the curve
$\alpha_0(\xi_0)$, closer to ergodic
averaging than to classical Toeplitz
theory.
\end{remark}

\begin{proposition}[Model reduction limit]
\label{prop:model-limit}
\begin{equation}\label{eq:model-limit}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}
    \Tcc(\psi(\cdot;\xi_0))
    \,d\xi_0
    + O(c^{3\theta_0-2}).
\end{equation}
For frozen $\alpha_0$:
$\Sigma_{\mathrm{model}}=\Theta(1)$.
For PSWF model:
(T\textsuperscript{*}) is equivalent to
the weighted RL condition~\eqref{eq:wRL}.
\end{proposition}

\begin{definition}[(T\textsuperscript{*})]
\label{def:Tstar}
\begin{equation}\label{eq:Tstar}
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \Tcc(\psi(\cdot;\xi_0))
  \,d\xi_0
  = o(1),
\end{equation}
or equivalently:
$\int K(x)\widehat{g}(\lambda x)\,dx\to 0$
as $\lambda=k/c^{\theta_0}\to\infty$.
By Lemma~\ref{lem:weighted-RL},
this follows from $g\in L^1$
(Lemma~\ref{lem:g-singularity}(ii))
and $K\in L^1$
(qualitative version).
\end{definition}

\begin{remark}[Status of (T\textsuperscript{*})
  after Lemma~\ref{lem:weighted-RL}]
\label{rem:Tstar-status}
Lemma~\ref{lem:weighted-RL}(a) shows:
\begin{center}
\textbf{(T\textsuperscript{*}) follows from
$g\in L^1$ and $K\in L^1$,
both of which are established.}
\end{center}
This means the \emph{qualitative}
statement $\Sigma_{\mathrm{model}}=o(1)$
is now proved,
conditioned on the validity of the
model approximation
\eqref{eq:model-limit}
(i.e., on the reduction steps
from $\Sigma_{\mathrm{int}}$ to
$\Sigma_{\mathrm{model}}$ being rigorous).
What remains for Paper~XVI:
\begin{enumerate}[\rm(i)]
\item \emph{Quantitative rate}:
establish $O(c^{-\theta_0/2})$
using condition~(c) with $K\in\mathcal{S}$.
\item \emph{Rigorous model reduction}:
verify that the approximation in
\eqref{eq:model-limit} is tight enough
for the qualitative conclusion to transfer.
\item \emph{Condition~(L)}:
estimate the Lipschitz error.
\end{enumerate}
\end{remark}

\begin{problem}[Critical Toeplitz Energy]
\label{prob:Tstar}
\textbf{Priority~1} (decisive for rate).
Establish
$|\int K(x)\widehat{g}(\lambda x)\,dx|
= O(\lambda^{-1/2})$
using $K\in\mathcal{S}$ and
$|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$
(Lemma~\ref{lem:g-singularity}(iii)).
This gives rate $O(c^{-\theta_0/2})$
for $\Sigma_{\mathrm{model}}$.

\medskip
\noindent\textbf{Priority~2}
(symmetry cancellation).
Check $\widehat{K}(0)=|\widehat\psi(0)|^2$.
If $\widehat\psi(0)=0$ (zero-mean $\psi$),
logarithmic growth in $\widehat{g}$
is automatically cancelled
(Remark~\ref{rem:K-cancellation}).

\medskip
\noindent\textbf{Priority~3}
(moving-symbol extension).
Investigate whether the moving-symbol
Toeplitz structure
(Remark~\ref{rem:moving-symbol})
admits a Szeg\H{o}-type theorem;
this would give an independent
route to (T\textsuperscript{*}).
\end{problem}

\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Assume (T\textsuperscript{*}), (L), (C).
\begin{equation}\label{eq:reduction-chain}
  \underbrace{\mathrm{(T^*)}+\mathrm{(L)}}
  _{\text{Paper XVI}}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}}=o(1)
  \;\underset{\mathrm{(C)}}{\Longrightarrow}\;
  \text{H3 fails}
  \;\Rightarrow\;
  \text{H1 fails}.
\end{equation}
Moreover:
\begin{enumerate}[\rm(iv)]
\item (T\textsuperscript{*}) is equivalent to
  the weighted RL condition~\eqref{eq:wRL}.
  This follows from $g,K\in L^1$
  (qualitative; Lemma~\ref{lem:weighted-RL}(a));
  the quantitative rate $O(c^{-\theta_0/2})$
  uses $K\in\mathcal{S}$ and
  $|\widehat{g}|=O(|\xi|^{-1/2})$
  (condition~(c)).
\end{enumerate}
\end{theorem}

\begin{proof}
(T\textsuperscript{*}) $\Rightarrow$
$\Sigma_{\mathrm{model}}=o(1)$;
(L) $\Rightarrow$
$\Sigma_{\mathrm{amp}}=O(c^{3\theta_0-2})=o(1)$;
hence $\Sigma_{\mathrm{int}}=o(1)$.
Chain to H1 fails: Paper~XIV under~(C).
Part~(iv):
Proposition~\ref{prop:alpha-oscillation}
+ Lemma~\ref{lem:weighted-RL}.
\end{proof}

\begin{remark}[Moving-symbol and
  non-commuting limits]
\label{rem:non-commuting}
Paper~XVI faces a problem at the
boundary of current Toeplitz operator
theory: the symbol variation rate
$|\alpha_0'|=|\theta''|\sim|\xi_0|/c$
is tied to the semiclassical scale,
beyond the scope of
\cite{BottcherSilbermann1990,Simon2011}.
\end{remark}

\begin{remark}[Programme status
  after Paper~XV]
\label{rem:series-status}
\[
  \underbrace{\mathrm{(T^*)}+\mathrm{(L)}}
  _{\text{XVI}}
  +\underbrace{\mathrm{(C)}}_{\text{XVII}}
  \;\Longrightarrow\;
  \text{H1 fails}.
\]
Paper~XVI: prove~(L) and
estimate~\eqref{eq:wRL}
with rate $O(c^{-\theta_0/2})$.
Paper~XVII: prove~(C) and
investigate Option~3.
\end{remark}
