\begin{abstract}
Paper~XIV reduced H1 to
$\Sigma_{\mathrm{int}}=o(1)$.
This paper carries out a complete
scalar reduction,
establishes two rigorous results
(exact equivalence and
Dirichlet obstruction),
and identifies the precise
remaining task.

All claims are explicitly labelled
\Rig, \Cond, or \Heur\ throughout;
the reader can immediately
identify the logical status
of each step.

The rigorous contributions:
\begin{enumerate}[\rm(1)]
\item \emph{Exact equivalence} \Rig:
  $\Sigma_{\mathrm{model}}=o(1)
  \iff S(c)=o(c^{-(2-2\theta_0)})$;
  gap $=c^{2-\theta_0}$
  $=$ frozen-phase saturation rate.
\item \emph{Ces\`aro step} \Rig:
  $\Psi\in L^1(d\lambda/\lambda)$
  via Fubini, giving $S(c)=o(c^{\theta_0})$.
\item \emph{Dirichlet obstruction} \Rig:
  $J_K=\Theta(K^{1/2})$ with constant
  phase $e^{i\pi/4}$; no improvement
  over Ces\`aro from $D^*_K$ alone.
\item \emph{Poisson bridge} \Rig:
  Poisson summation converts
  discrete $D^*_K$
  into a sum of continuous
  phase integrals $I_m(x,c)$;
  the $m=0$ term $I_0(x,c)$
  is identified as the
  sole non-trivial contribution.
\end{enumerate}
The conditional/heuristic part:
\begin{enumerate}[\rm(5)]
\item \emph{Airy scaling} \Heur:
  at scale $\xi\sim c^{-\theta_0/3}$,
  the fold $\theta''(0)=0$,
  $\theta'''(0)\ne 0$
  suggests an Airy-type structure
  for $I_0$;
  rigorously establishing this
  requires the Airy normal form
  (Paper~XVI, Task~A).
\item \emph{Conditional bound} \Cond:
  if the Airy normal form holds
  with uniform error control,
  then $S(c)=O(c^{\theta_0/2})$,
  closing the gap for $\theta_0>4/5$.
\end{enumerate}
\end{abstract}

\section{Frozen-Phase Saturation}
\label{sec:S3}

\begin{definition}[Toeplitz energy \Rig]
$\Tcc(\psi)
:=\sum_k|\widehat{\psi}(k\alpha_0/c^{\theta_0})|^2$;
$\Sigma_{\mathrm{strip}}
=c^{2(1-\theta_0)}\Tcc(\psi)
=\Theta(c^{2-\theta_0})$.
\end{definition}

\begin{corollary}[No-go \Rig]
\label{cor:S3-negative}
No pointwise mechanism gives $o(1)$;
frozen-phase rate $c^{2-\theta_0}$
is exactly the scaling gap.
\end{corollary}

\section{WKB and the Fold}
\label{sec:WKB}

\begin{lemma}[WKB phase and fold \Rig]
\label{lem:WKB-phase}
$\theta:=c\int_0^{\xi/c}p\,d\zeta$;
$\theta''(0)=0$, $\theta'''(0)\ne 0$:
degenerate (fold) critical point.
$g(\alpha):=1/|\theta''(\xi_0(\alpha))|$;
$g\sim C_*|\alpha-\alpha_*|^{-1/2}$
near $\alpha_*=\theta'(0)$.
\end{lemma}

\section{Scalar Reduction,
  Dirichlet Obstruction,
  and the Poisson Bridge}
\label{sec:main}

\begin{remark}[Core thesis \Rig+\Heur]
\label{rem:core-thesis}
\emph{\Rig: All known mechanisms
fail due to coherent Dirichlet
phase (Lemma~\ref{lem:dirichlet-obstruction}).
The Poisson bridge isolates
$I_0(x,c)$ as the sole
non-trivial object.
\Heur: The fold geometry of $\theta$
suggests Airy structure in $I_0$,
but this remains to be proved.}
\end{remark}

\begin{lemma}[Splitting of $g$ \Rig]
\label{lem:g-split}
$g=\gsing+\greg$,
$\gsing:=C_*|\alpha-\alpha_*|^{-1/2}\chi$,
$\greg\in C^\infty_c$.
$|\widehat{\gsing}|=O(|\xi|^{-1/2})$;
$|\widehat{\greg}|=O(|\xi|^{-N})$.
\end{lemma}

\begin{lemma}[Exact scaling
  equivalence \Rig]
\label{lem:S-equivalence}
$\Sigma_{\mathrm{model}}
=c^{2-2\theta_0}S(c)+o(1)$;
hence:
\begin{equation}\label{eq:exact-equiv}
\boxed{\Sigma_{\mathrm{model}}=o(1)
  \iff
  S(c)=o(c^{-(2-2\theta_0)}).}
\end{equation}
Gap $=c^{2-\theta_0}$
$=$ frozen-phase saturation rate.
\end{lemma}

\begin{proposition}[Ces\`aro step \Rig]
\label{prop:Psi-L1}
$\Psi(\lambda):=\int K\widehat{g}(\lambda\cdot)$;
Fubini--Tonelli:
$\int_1^\infty|\Psi|d\lambda/\lambda
\lesssim\int|K(x)|(1+|x|^{-1/2})dx
<\infty$.
\end{proposition}

\begin{corollary}[$S(c)=o(c^{\theta_0})$ \Rig]
\label{cor:cesaro-closed}
$S(c)=c^{\theta_0}\mathcal{C}(c)$,
$\mathcal{C}(c)\to 0$.
\end{corollary}

\subsection{Dirichlet obstruction \Rig}

\begin{lemma}[Dirichlet obstruction \Rig]
\label{lem:dirichlet-obstruction}
$D^*_K(t):=\sum_{k=1}^K e^{ikt}$.
The signed integral:
\begin{equation}\label{eq:JK-def}
  J_K(x):=\int|\beta|^{-1/2}
  D^*_K(\beta x/c^{\theta_0})d\beta
  = Ce^{i\pi/4}
  \frac{c^{\theta_0/2}}{|x|^{1/2}}
  \sum_{k=1}^K k^{-1/2}
  = \Theta\!\left(
  \frac{c^{\theta_0}}{|x|^{1/2}}\right).
\end{equation}
Consequence: signed $D^*_K$ gives
no improvement over Ces\`aro.
\end{lemma}

\begin{proof}
$\int|\beta|^{-1/2}e^{ik\beta x/c^{\theta_0}}d\beta
=C(c^{\theta_0}/(kx))^{1/2}e^{i\pi/4}$
(Fourier transform of $|\cdot|^{-1/2}$);
phase $e^{i\pi/4}$ independent of $k$;
$\sum_kk^{-1/2}\sim 2K^{1/2}$;
$K=c^{\theta_0}$.
\end{proof}

\begin{remark}[Why no cancellation \Rig]
\label{rem:no-cancel}
The phase $e^{i\pi/4}$ is
$k$-independent: all Dirichlet
modes add \emph{coherently}.
The singularity $|\alpha-\alpha_*|^{-1/2}>0$
has no sign change.
Cancellation requires the
\emph{full joint} $(\beta,k)$
integral---i.e., the continuous
phase after Poisson.
\end{remark}

\subsection{The Poisson bridge \Rig}

\begin{lemma}[Poisson bridge \Rig]
\label{lem:poisson-bridge}
Poisson summation applied to
$k\mapsto\widehat{\gsing}(xk/c^{\theta_0})$
(smooth in $k$ for fixed $x\ne 0$)
gives:
\begin{equation}\label{eq:poisson-expand}
  \sum_{k=1}^{K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})
  = \sum_{m\in\mathbb{Z}}
    I_m(x,c),
\end{equation}
\begin{equation}\label{eq:Im-def}
  I_m(x,c):=
  \int_1^{K(c)}
  \widehat{\gsing}(xt/c^{\theta_0})
  e^{2\pi imt}\,dt.
\end{equation}
The Ces\`aro bound holds termwise:
$|I_m(x,c)|=O(c^{\theta_0}/|x|^{1/2})$
for each $m$
(from $|\widehat{\gsing}(\xi)|=O(|\xi|^{-1/2})$
and the length $K(c)=c^{\theta_0}$).

The $m\ne 0$ terms oscillate
rapidly in $t$ (frequency $2\pi m$);
integration by parts gives
$|I_m|=O(1/|m|)$ for $m\ne 0$.
Thus $I_0$ dominates:
\begin{equation}\label{eq:I0-dominant}
  T(x,c):=
  \textstyle\sum_k
  \widehat{\gsing}(xk/c^{\theta_0})
  = I_0(x,c)+O(1).
\end{equation}
\end{lemma}

\begin{remark}[Why the bridge works \Rig]
\label{rem:bridge-works}
Poisson summation converts the
discrete coherent phases
($e^{i\pi/4}$, $k$-independent)
into a continuous integral
$I_0(x,c)=\int_1^{K}
\widehat{\gsing}(xt/c^{\theta_0})dt$
where the phase now depends
continuously on $t$.
Standard stationary-phase methods
(including degenerate ones)
are now applicable.
The fold condition
$\theta''(0)=0$, $\theta'''(0)\ne 0$
is encoded in the phase of
$\widehat{\gsing}(xt/c^{\theta_0})$
near $t\to 0$.
\end{remark}

\subsection{Airy structure of
  $I_0(x,c)$: heuristic and
  conditional results}

\begin{remark}[Airy scaling \Heur]
\label{rem:airy-heuristic}
For the integrand
$\widehat{\gsing}(\xi)$,
the fold~$\theta''(0)=0$,
$\theta'''(0)\ne 0$
causes the phase of
$\widehat{\gsing}$ to degenerate
cubically near $\xi=0$:
\[
  \widehat{\gsing}(\xi)
  \approx C_*|\xi|^{-1/2}
  \exp\!\left(
    i\alpha_*\xi
    +i\frac{\theta'''(0)}{6}\xi^3
    +\cdots
  \right).
\]
The scale at which the cubic
term becomes $O(1)$
(i.e., $\xi\sim c^{-\theta_0/3}$,
so $t\sim c^{2\theta_0/3}/x$)
is the \emph{Airy scale}.
Substituting $u=xt/c^{\theta_0}$
in $I_0$:
heuristically,
\[
  I_0(x,c)
  \sim \frac{c^{\theta_0}}{|x|}
  \int_0^{|x|/c^{0}}
  |u|^{-1/2}e^{i(\cdots)u^3}du
  = O\!\left(
    \frac{c^{\theta_0/2}}{|x|^{1/2}}
  \right),
\]
giving $S(c)=O(c^{\theta_0/2})$
(improvement by $c^{\theta_0/2}$
over Ces\`aro).

\textbf{This is a scaling argument,
not a theorem.}
The computation collapses if:
\begin{itemize}[leftmargin=*]
\item the cubic normal form
  requires a non-uniform
  coordinate change in $c$;
\item the Airy integral
  $\int|u|^{-1/2}e^{iu^3}du$
  has unexpected cancellation or
  divergence;
\item the uniformity in $x$
  fails (large or small $|x|$).
\end{itemize}
\end{remark}

\begin{assumption}[Airy normal form
  \Cond]
\label{ass:airy-normalform}
There exists a smooth, invertible
change of variables
$\beta = \Psi_c(s)$
(uniform in $c\ge c_0$,
$|x|\in[\delta,\delta^{-1}]$),
such that:
\begin{enumerate}[\rm(i)]
\item The phase of
  $\widehat{\gsing}(xs/c^{\theta_0})$
  is mapped to the Airy normal form
  $s\xi_0+s^3/3$
  (after rescaling $s\to
  c^{\theta_0/3}\cdot\tilde s$).
\item The Jacobian
  $|\Psi_c'(s)|$ is bounded
  uniformly in $c$.
\item The error in approximating
  $I_0(x,c)$ by the Airy integral
  is $o(c^{\theta_0/2}/|x|^{1/2})$,
  uniformly in $|x|\in[\delta,
  \delta^{-1}]$.
\end{enumerate}
\end{assumption}

\begin{proposition}[Conditional
  improvement \Cond]
\label{prop:S-conditional}
Under Assumption~\ref{ass:airy-normalform}:
\begin{equation}\label{eq:S-cond}
  S(c) = O(c^{\theta_0/2}).
\end{equation}
For $\theta_0>4/5$:
$c^{\theta_0/2}
=o(c^{-(2-2\theta_0)})$,
so $\Sigma_{\mathrm{model}}=o(1)$
conditional on~\ref{ass:airy-normalform},
(L), (R), (C).
\end{proposition}

\begin{proof}
Under Assumption~\ref{ass:airy-normalform}(iii),
$|I_0(x,c)|
=O(c^{\theta_0/2}/|x|^{1/2})$;
combined with
Lemma~\ref{lem:poisson-bridge}
($I_m=O(1/|m|)$ for $m\ne 0$):
$|T(x,c)|
=O(c^{\theta_0/2}/|x|^{1/2})$.
Integrate:
$|S(c)|\le C c^{\theta_0/2}
\int|K(x)||x|^{-1/2}dx
=O(c^{\theta_0/2})$.
\end{proof}

\section{Open Problems for Paper~XVI}
\label{sec:open}

\begin{problem}[Airy normal form
  (Paper~XVI, Task~A)]
\label{prob:S-cancellation}
Prove Assumption~\ref{ass:airy-normalform}.

\medskip
\noindent\textbf{Precise requirements.}
\begin{enumerate}[\rm(A1)]
\item \emph{Phase reduction.}
  Show that the phase function
  $\phi(\xi):=\mathrm{arg}
  (\widehat{\gsing}(\xi))$
  near $\xi=0$
  can be written as
  $\phi(\xi)=\alpha_*\xi
  +c_3\xi^3+O(\xi^4)$
  with $c_3=\theta'''(0)/6\ne 0$.
  This follows from the Taylor
  expansion of $\theta$ at $\xi=0$
  and is \textbf{rigorous}.
\item \emph{Normal form coordinate.}
  Construct $s=\Psi_c^{-1}(\xi)$
  such that the full phase
  of the integrand in $I_0$
  is exactly $as+bs^3$
  (Airy standard form).
  The existence of such a
  change of variables is guaranteed
  by the Malgrange preparation
  theorem, but the uniformity
  in $c$ requires explicit estimates
  on $\Psi_c$ and its derivatives.
\item \emph{Amplitude bound.}
  Show $|\Psi_c'(s)|$ and
  $|(|\Psi_c^{-1}(s)|)^{-1/2}
  \cdot|\Psi_c'(s)||$
  are bounded uniformly in $c$
  and $|x|\in[\delta,\delta^{-1}]$.
\item \emph{Uniformity in $x$.}
  Extend the estimate
  $|I_0(x,c)|
  =O(c^{\theta_0/2}/|x|^{1/2})$
  to all $|x|\ge\delta$
  (not just compact $x$-range);
  requires a separate argument
  for large $|x|$ using
  the rapid decay of $K\in\mathcal{S}$.
\end{enumerate}

\medskip
\noindent\textbf{Task~B
  (alternative: obstruction).}
If Task~A fails, show that
$I_0(x,c)=\Theta(c^{\theta_0}/|x|^{1/2})$
(no improvement); this would
establish $S(c)=\Theta(c^{\theta_0})$
as a \emph{matching lower bound},
proving that the Ces\`aro step
is sharp and the gap
$c^{2-\theta_0}$ cannot be closed
by any method based on $I_0$.
\end{problem}

(R): sum--integral exchange,

\end{problem}

\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem
  \Cond]
\label{thm:reduction}
Conditional on
Assumption~\ref{ass:airy-normalform},
(L), (R), (C), and $\theta_0>4/5$:
$\Sigma_{\mathrm{int}}=o(1)
\Rightarrow\mathrm{H3\ fails}
\Rightarrow\mathrm{H1\ fails.}$
\end{theorem}

\label{rem:concept-summary}
\begin{enumerate}[\rm(1)]
\item \textit{\Rig}\ Dimension
  collapse: operator $\to$ $S(c)$;
  exact equivalence;
  gap $=$ saturation rate.
\item \textit{\Rig}\ Ces\`aro step:
  $S(c)=o(c^{\theta_0})$.
\item \textit{\Rig}\ Dirichlet
  obstruction: coherent phase;
  no improvement.
\item \textit{\Rig}\ Poisson bridge:

\item \textit{\Cond}\ $S(c)=O(c^{\theta_0/2})$

\end{enumerate}
\end{remark}
