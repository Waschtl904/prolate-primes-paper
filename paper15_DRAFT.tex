  Intermediate-Regime Kernel Estimates

  Mesoscopic Scaling, WKB Phase Extraction,
  and the Critical Toeplitz Obstruction}

\author{[Author]}

\begin{abstract}
Paper~XIV reduced H1 to
$\Sigma_{\mathrm{int}}=o(1)$.
This paper carries out a complete
scalar reduction and identifies
the precise remaining obstruction.

The exact equivalence
(Lemma~\ref{lem:S-equivalence}):
\[
  \Sigma_{\mathrm{model}}=o(1)
  \;\iff\;
  S(c)=o(c^{-(2-2\theta_0)}),
\]
reduces everything to the
non-uniform Fourier sampling sum
\[
  S(c)=\int K(x)
  \textstyle\sum_{k=1}^{K(c)}
  \widehat{g}(xk/c^{\theta_0})\,dx,
  \quad K\in\mathcal{S},\;
  |{\widehat{g}}|=O(|\xi|^{-1/2}).
\]
The Ces\`aro step
(Corollary~\ref{cor:cesaro-closed})
gives $S(c)=o(c^{\theta_0})$
via Fubini--Tonelli, but the
target $o(c^{-(2-2\theta_0)})$
is a factor $c^{2-\theta_0}$ stronger.

The main new contributions are:
\begin{enumerate}[\rm(1)]
\item \emph{Exact equivalence}
  (Lemma~\ref{lem:S-equivalence}):
  the gap is precisely
  $c^{2-\theta_0}$,
  equal to the frozen-phase
  saturation rate.
\item \emph{Splitting}
  $g=\gsing+\greg$
  (Lemma~\ref{lem:g-split}):
  isolates the fold singularity
  $\gsing\sim|\alpha-\alpha_*|^{-1/2}$
  as the sole obstruction.
\item \emph{Dirichlet obstruction lemma}
  (Lemma~\ref{lem:dirichlet-obstruction}):
  signed Dirichlet-kernel cancellation
  alone gives only $O(K^{1/2})$
  (same order as the triangle
  inequality), proving that
  \emph{no improvement over
  Ces\`aro is possible from
  Dirichlet bounds alone}.
\item \emph{Identification of the
  last obstacle}
  (Problem~\ref{prob:S-cancellation}):
  the only remaining mechanism is
  a joint $\alpha$-$k$ stationary-phase
  estimate for
  $T(x,c):=\int|\beta|^{-1/2}
  D^*_{K(c)}(\beta x/c^{\theta_0})\,d\beta$
  at the fold $\alpha_*$,
  where $\theta''(0)=0$,
  $\theta'''(0)\ne 0$.
  The correct scaling is
  $\beta\sim c^{-\theta_0/3}$,
  giving an Airy-type kernel
  (not Dirichlet).
\end{enumerate}
\end{abstract}

\begin{remark}[Core thesis]

\emph{All known mechanisms fail

\end{center}
\end{remark}

\begin{lemma}[Singularity and
  splitting of $g$]
\label{lem:g-split}
Let $g(\alpha):=1/|\theta''(\xi_0(\alpha))|$,
$\alpha_*:=\theta'(0)$,
$\chi\in C_c^\infty$,
$\chi\equiv 1$ near $\alpha_*$:
\begin{equation}\label{eq:g-split}
  g = \gsing + \greg,\quad
  \gsing(\alpha)
  := C_*|\alpha-\alpha_*|^{-1/2}
     \chi(\alpha).
\end{equation}
\begin{enumerate}[\rm(i)]
\item $\greg\in C^\infty_c$;
  $|\widehat{\greg}(\xi)|
  =O(|\xi|^{-N})$ for all $N$.
\item $|\widehat{\gsing}(\xi)|
  =O(|\xi|^{-1/2})$.
\item $g\in L^1$;
  $g\in C^\infty$ away
  from $\alpha_*$;
  $g\sim C_*/|\alpha-\alpha_*|^{1/2}$
  near $\alpha_*$.
\end{enumerate}
\end{lemma}

\begin{proposition}[Coarea reduction]
\label{prop:alpha-oscillation}

  K:=\psi*\widetilde{\psi}\in\mathcal{S}.
\end{equation}
Then $(T^*)$

\end{proposition}

\begin{lemma}[Exact scaling equivalence]

\subsection{The Dirichlet obstruction}

\begin{lemma}[Dirichlet obstruction:
  no improvement from signed bounds]
\label{lem:dirichlet-obstruction}
Let $D^*_K(t):=\sum_{k=1}^K e^{ikt}$.
Define the signed integral:
\begin{equation}\label{eq:JK-def}
  J_K(x) :=
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  D^*_K(\beta x/c^{\theta_0})\,d\beta.
\end{equation}
Then:
\begin{equation}\label{eq:JK-bound}
  |J_K(x)|
  = O\!\left(
    \frac{c^{\theta_0/2}}{|x|^{1/2}}
    \cdot K^{1/2}
  \right)
  = O\!\left(
    \frac{c^{\theta_0}}{|x|^{1/2}}
  \right).
\end{equation}
In particular, the signed Dirichlet
integral gives the same order
as the triangle inequality.
\end{lemma}

\begin{proof}
Expand:
$J_K(x)=\sum_{k=1}^K
\int|\beta|^{-1/2}
e^{ik\beta x/c^{\theta_0}}d\beta$.
By the Fourier transform of
$|\cdot|^{-1/2}$:
\begin{equation}\label{eq:fourier-half}
  \int|\beta|^{-1/2}
  e^{ik\beta x/c^{\theta_0}}d\beta
  = C\cdot
  \left(\frac{c^{\theta_0}}{kx}\right)^{1/2}
  e^{i\pi/4}.
\end{equation}
Hence:
\begin{equation}\label{eq:JK-sum}
  J_K(x)
  = C e^{i\pi/4}
    \frac{c^{\theta_0/2}}{|x|^{1/2}}
    \sum_{k=1}^K k^{-1/2}
  \sim C e^{i\pi/4}
    \frac{c^{\theta_0/2}}{|x|^{1/2}}
    \cdot 2K^{1/2}.
\end{equation}
With $K=c^{\theta_0}$:
$|J_K(x)|
\sim C\frac{c^{\theta_0/2}}{|x|^{1/2}}
\cdot c^{\theta_0/2}
=O(c^{\theta_0}/|x|^{1/2})$.
This matches the triangle-inequality
bound $K\cdot|\widehat{\gsing}(x/c^{\theta_0})|
=c^{\theta_0}\cdot O(c^{-\theta_0/2}/|x|^{1/2})
=O(c^{\theta_0/2}/|x|^{1/2})$
up to a constant factor.

No cancellation of order $c^\varepsilon$
(for any $\varepsilon>0$) is available
from Dirichlet-kernel bounds alone.
\end{proof}

\begin{remark}[Why Dirichlet fails]
\label{rem:why-dirichlet-fails}
The sum $\sum_{k=1}^K k^{-1/2}
\sim 2K^{1/2}$
is a \emph{convergent partial sum
of a divergent series}.
The signed contributions
$C k^{-1/2}e^{i\pi/4}$ all point
in the \emph{same direction} (phase
$e^{i\pi/4}$ is independent of $k$):
they add coherently, not
cancellingly.
This is a structural consequence of
the square-root singularity:
$|\alpha-\alpha_*|^{-1/2}$
is \emph{positive} near $\alpha_*$,
so no phase oscillation appears
at leading order.
The only mechanism that can
produce genuine cancellation is
the \emph{interaction} of the
$\beta$-oscillation
with the Dirichlet phases---
i.e., the full integral
$T(x,c)$
(Problem~\ref{prob:S-cancellation}).
\end{remark}

\subsection{The integral $T(x,c)$
  and the fold geometry}

\begin{definition}[$T(x,c)$]
\label{def:Txc}
\begin{equation}\label{eq:Txc}
  T(x,c) :=
  \int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  D^*_{K(c)}(\beta x/c^{\theta_0})
  \,d\beta.
\end{equation}
Then $\sum_{k\le K}\widehat{\gsing}(xk/c^{\theta_0})
=C_* e^{i\alpha_* x/c^{\theta_0}}
T(x,c)$,
and
\begin{equation}\label{eq:S-via-T}
  S(c) = C_*\int K(x)
  e^{i\alpha_* x/c^{\theta_0}}
  T(x,c)\,dx
  + O(c^{-N}).
\end{equation}
\end{equation}
\end{definition}

\begin{remark}[Fold geometry and
  Airy scaling]
\label{rem:airy-scaling}
The fold at $\alpha_*$ is characterised
by $\theta''(0)=0$, $\theta'''(0)\ne 0$.
In the Dirichlet kernel:
\[
  D^*_K(t)=\frac{e^{iKt}-1}{e^{it}-1}
  \approx \frac{e^{iKt}}{it}
  \quad(t\ne 0),
\]
the approximation breaks down when
$|t|\lesssim 1/K$, i.e.,
$|\beta x/c^{\theta_0}|\lesssim 1/K
=c^{-\theta_0}$,
i.e., $|\beta|\lesssim c^{-\theta_0}/|x|$.

However, the fold introduces a
\emph{cubic} correction to the phase.
The correct scale at which
the fold geometry dominates is
$|\beta|\sim c^{-\theta_0/3}$
(Airy scale), not $c^{-\theta_0}$.
At this scale:
\begin{itemize}[leftmargin=*]
\item $D^*_K(\beta x/c^{\theta_0})
  \approx$ \emph{Airy kernel},
  not Dirichlet.
\item The integral $T(x,c)$
  receives its dominant contribution
  from $|\beta|\lesssim c^{-\theta_0/3}$.
\item Expected size:
  $T(x,c)=O(c^{-\theta_0/6}/|x|^{1/2})$
  (heuristic: $c^{-\theta_0/3}\cdot
  c^{-\theta_0/6}$ from the
  $|\beta|^{-1/2}$ weight).
\end{itemize}
If confirmed, this would give:
$S(c)=O(c^{-\theta_0/6})$,
closing the gap for
$\theta_0>12/13$
(since $-\theta_0/6<-(2-2\theta_0)$
$\Leftrightarrow \theta_0>12/13$).
This is a \emph{heuristic}; the
rigorous argument is the task of
Paper~XVI.
\end{remark}

%%ŌĢÉŌĢÉ 7. OPEN PROBLEMS
\section{Open Problems for Paper~XVI}
\label{sec:open}

\begin{problem}[Airy-type estimate
  for $T(x,c)$]
\label{prob:S-cancellation}
\emph{The single remaining obstacle.}

\medskip
\noindent\textbf{Object:}
\[
  T(x,c)=\int|\beta|^{-1/2}
  \chi(\alpha_*+\beta)
  D^*_{K(c)}(\beta x/c^{\theta_0})\,d\beta.
\]
\textbf{Target:}
$|T(x,c)|=o(c^{\theta_0}/|x|^{1/2})$,
uniformly in $|x|\ge\delta$.

\medskip
\noindent\textbf{Why Dirichlet fails}
(Lemma~\ref{lem:dirichlet-obstruction}):
the signed $D^*_K$ sum gives only
$O(K^{1/2})$ per mode, and the
modes add with constant phase
$e^{i\pi/4}$:
no cancellation.

\medskip
\noindent\textbf{The Airy mechanism.}
Near $\beta=0$, the phase in
$D^*_K(\beta x/c^{\theta_0})
=e^{iK\beta x/c^{\theta_0}}/(e^{i\beta x/c^{\theta_0}}-1)$
has a cubic correction from
$\theta'''(0)\ne 0$:
\[
  K\cdot\frac{\beta x}{c^{\theta_0}}
  +\frac{\theta'''(0)}{6}
  \frac{\beta^3 x^3}{c^{3\theta_0}}
  \cdot\text{(correction)}.
\]
At the Airy scale
$\beta\sim c^{-\theta_0/3}$
(where the cubic term is $O(1)$),
the Dirichlet kernel is replaced
by an Airy function:
\[
  D^*_K(\beta x/c^{\theta_0})
  \;\approx\;
  c^{\theta_0/3}\cdot
  \mathrm{Ai}(\beta x\cdot c^{-\theta_0/3}
  +\text{shift}).
\]
The integral becomes:
\[
  T(x,c)
  \approx c^{\theta_0/3}
  \int|\beta|^{-1/2}
  \mathrm{Ai}(\beta x c^{-\theta_0/3})
  \,d\beta.
\]
Substituting $u=\beta x c^{-\theta_0/3}$:
\[
  T(x,c)
  \approx \frac{c^{\theta_0/3}\cdot
  c^{\theta_0/6}}{|x|^{1/2}}
  \int|u|^{-1/2}\mathrm{Ai}(u)\,du
  =\frac{c^{\theta_0/2}}{|x|^{1/2}}
  \cdot C_{\mathrm{Ai}}.
\]
This gives $S(c)=O(c^{\theta_0/2})$---
a factor $c^{\theta_0/2}$ improvement
over Dirichlet.\;
\emph{Whether this closes the
full gap requires checking
$c^{\theta_0/2}<c^{-(2-2\theta_0)}$,
i.e., $\theta_0/2>2-2\theta_0$,
i.e., $\theta_0>4/5$.}

\medskip
\noindent\textbf{Task for Paper~XVI:}
\begin{enumerate}[\rm(A)]
\item Derive the Airy approximation
  for $D^*_K$ near $\beta=0$
  rigorously, using the WKB
  structure and
  $\theta'''(0)\ne 0$.
\item Bound
  $\int|u|^{-1/2}|\mathrm{Ai}(u)|du<\infty$
  (classical; $\mathrm{Ai}$
  is Schwartz for $u>0$
  and oscillatory for $u<0$).
\item Conclude $T(x,c)=O(c^{\theta_0/2}/|x|^{1/2})$
  and hence
  $S(c)=O(c^{\theta_0/2})$;
  close the remaining gap for
  $\theta_0>4/5$.
\end{enumerate}
\end{problem}

-\widehat{\Phi}_n\|=O(k/c)$.
(R): sum--integral exchange,

\end{problem}

\label{rem:concept-summary}
