\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to
$\Sigma_{\mathrm{int}}=o(1)$.
This paper identifies the core
structure and reduces the problem
to a sharply-formulated scalar
question.

The main objects are
$I_c(k):=\int e^{ik\theta/c}w\,d\xi$
and $\Sigma_{\mathrm{model}}
=\sum_k|I_c(k)|^2$.
Pointwise tools fail;
the obstruction is
\emph{lack of cancellation after
phase freezing}.
A coarea substitution and
WKB analysis reduce
(T\textsuperscript{*}) to the
Ces\`aro average
$\mathcal{C}(c)
=\frac{1}{K}\sum_{k\le K}
\Psi(k/c^{\theta_0})$
where
$\Psi(\lambda)=\int K\widehat{g}(\lambda\cdot)$.

The singularity
$g\sim|\alpha-\alpha_*|^{-1/2}$
gives $|\widehat{g}|=O(|\xi|^{-1/2})$,
hence $|\Psi(\lambda)|=O(\lambda^{-1/2})$.
This rate is \emph{exactly borderline}
for Ces\`aro decay.
A Fubini--Tonelli argument
(Proposition~\ref{prop:Psi-L1})
shows $\Psi\in L^1(d\lambda/\lambda)$,
and hence $\mathcal{C}(c)\to 0$
(Proposition~\ref{prop:L1-sufficient}).

However, since
$\Sigma_{\mathrm{model}}
\sim c^{2-\theta_0}\cdot\mathcal{C}(c)$
(Proposition~\ref{prop:model-limit})
and $c^{2-\theta_0}\to\infty$,
the Ces\`aro decay $\mathcal{C}(c)=o(1)$
does \emph{not} by itself imply
$\Sigma_{\mathrm{model}}=o(1)$.
Closing (T\textsuperscript{*}) requires
the \emph{rate}
$\mathcal{C}(c)=o(c^{-(2-\theta_0)})$
(Remark~\ref{rem:scaling-gap}).

The main contributions of this paper are:
\begin{enumerate}[\rm(1)]
\item \emph{Dimension collapse}:
reduction from operator problem
to scalar function $\Psi(\lambda)$.
\item \emph{Ces\`aro obstruction removed}:
$\Psi\in L^1(d\lambda/\lambda)$
via Fubini.
\item \emph{Scaling gap identified}:
$\mathcal{C}(c)=o(c^{-(2-\theta_0)})$
is the precise remaining target.
\item \emph{Moving-symbol / ergodic
structure}: the problem is
non-commuting limits, not classical
oscillation.
\end{enumerate}
\end{abstract}

\begin{corollary}[No-go]

\section{WKB Phase Extraction}
\label{sec:WKB}

\begin{lemma}[WKB phase]
\label{lem:WKB-phase}
$\theta(\xi):=c\int_0^{\xi/c}
p\,d\zeta$,
$w:=|A|^2/p$,
$p(\zeta;\alpha)
:=\sqrt{(\lambda(\alpha)-\zeta^2)/(1-\zeta^2)}$.
\end{lemma}

\begin{lemma}[Phase non-degeneracy]
\label{lem:phase-nondeg}
(i)~$\theta'>0$;\;
(ii)~$\theta''\ne 0$ for $\xi\ne 0$,
$\theta''\sim-\xi/c$ near $0$;\;
(iii)~$\theta''(0)=0$, $\theta'''(0)\ne 0$;\;
(iv)~$\xi_0\mapsto\alpha_0(\xi_0)
:=\theta'(\xi_0)$ strictly monotone.
\end{lemma}

\begin{corollary}[vdC insufficient]
\label{cor:curvature-available}
$\sum_k|I_c(k)|^2=\Omega(c)$;
vdC and Weyl/UDPL both fail.
\end{corollary}

%%═══ 6. OBSTRUCTION AND REDUCTION
\section{The Critical Toeplitz
  Obstruction and Its Reduction}
\label{sec:toeplitz}

\begin{remark}[Core thesis]
\label{rem:core-thesis}
\begin{center}
\emph{Obstruction = frozen-phase saturation.
Escape = global curvature averaging
(non-commuting limits).}
\end{center}
\end{remark}

\begin{remark}[Sampling / moving-symbol]
\label{rem:moving-symbol}
$\Sigma_{\mathrm{model}}
\approx c^{2(1-\theta_0)}
\int\sum_k|\widehat{\psi}(k\alpha_0(\xi_0)
/c^{\theta_0})|^2\,d\xi_0$:
randomised sampling of $|\widehat\psi|^2$
along $\alpha_0(\xi_0)$;
three non-commuting limits;
closer to ergodic averaging than
classical Szeg\H{o} theory
\cite{Simon2011,BottcherSilbermann1990}.
\end{remark}

\begin{lemma}[Singularity of $g$]
\label{lem:g-singularity}
Let $g(\alpha):=1/|\theta''(\xi_0(\alpha))|$,
$\alpha_*:=\theta'(0)$.
\begin{enumerate}[\rm(i)]
\item $g\in L^1$;
  $g\in C^\infty$ away from $\alpha_*$.
\item Near $\alpha_*$:
  $g(\alpha)\sim C_*/|\alpha-\alpha_*|^{1/2}$.
\item $|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$
  as $|\xi|\to\infty$,
  \emph{uniformly in sign}.
\end{enumerate}
\end{lemma}

\begin{proof}
(ii): $\alpha_0'(0)=\theta''(0)=0$,
$\alpha_0''(0)=\theta'''(0)/c\ne 0$;
Taylor expansion gives
$|\xi_0|\sim C|\alpha-\alpha_*|^{1/2}$.
(iii): $\mathcal{F}[|\cdot|^{-1/2}]=C|\xi|^{-1/2}$;
remainder $O(|\xi|^{-N})$ by RL.
\end{proof}

\begin{proposition}[Coarea reduction]
\label{prop:alpha-oscillation}
Substitute $\xi_0\mapsto\alpha
=\alpha_0(\xi_0)$:
\begin{equation}\label{eq:alpha-integral}
  \int_{-c_0}^{c_0}
  f_c(\alpha_0(\xi_0))\,d\xi_0
  = \int K(x)
    \sum_{k=1}^{K(c)}
    \widehat{g}(xk/c^{\theta_0})\,dx
  =: S(c).
\end{equation}
(T\textsuperscript{*}) is equivalent to
$S(c)=o(K(c))$,
i.e., $\mathcal{C}(c):=S(c)/K(c)\to 0$.
\end{proposition}

\begin{lemma}[Weighted RL and rate]
\label{lem:weighted-RL}
Let $\Psi(\lambda)
:=\int K(x)\widehat{g}(\lambda x)\,dx$.
\begin{enumerate}[\rm(a)]
\item $\Psi(\lambda)\to 0$
  (Riemann--Lebesgue + DCT).
\item $|\Psi(\lambda)|=O(\lambda^{-1/2})$
  uniformly, for $K\in\mathcal{S}$.
\end{enumerate}
\end{lemma}

\begin{lemma}[Ces\`aro analysis]
\label{lem:cesaro-decay}
Let $|\Psi(\lambda)|\le C\lambda^{-\delta}$,
$K(c)=c^{\theta_0}$.
\begin{enumerate}[\rm(i)]
\item $\delta>1$: $\mathcal{C}(c)\to 0$.
\item $\delta=1$: $\mathcal{C}(c)=O(\log c/c^{\theta_0})\to 0$.
\item $\delta=1/2$: $|\mathcal{C}(c)|=O(1)$
  (\textbf{borderline from
  triangle inequality alone}).
\item $\Psi\in L^1(d\lambda/\lambda)$:
  $\mathcal{C}(c)\to 0$
  (Ces\`aro theorem),
  but \emph{no rate}.
\end{enumerate}
\end{lemma}

\subsection{Weighted integrability
  and the Ces\`aro step}

\begin{proposition}[Ces\`aro criterion]
\label{prop:L1-sufficient}
If $\Psi\in L^1((1,\infty),d\lambda/\lambda)$
then $\mathcal{C}(c)\to 0$.
\end{proposition}

\begin{proof}
Standard: the Riemann sums
$\frac{1}{K}\sum_k\Psi(k/c^{\theta_0})$
converge to $0$ by the Ces\`aro
mean theorem for $L^1(d\lambda/\lambda)$
functions.
\end{proof}

\begin{proposition}[$\Psi\in L^1(d\lambda/\lambda)$]
\label{prop:Psi-L1}
Assume $K\in\mathcal{S}$ and
$|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$
for $|\xi|\ge 1$.
Then
\begin{equation}\label{eq:L1-Psi}
  \int_1^\infty
  |\Psi(\lambda)|\,\frac{d\lambda}{\lambda}
  \lesssim
  \int_{\mathbb{R}}
  |K(x)|\,(1+|x|^{-1/2})\,dx
  < \infty.
\end{equation}
\end{proposition}

\begin{proof}
\textbf{Fubini--Tonelli}\emph{
  (applied on truncated domain
  $\lambda\in[1,R]$, then $R\to\infty$)}:
\begin{align}\label{eq:fubini}
  \int_1^R|\Psi(\lambda)|\,\frac{d\lambda}{\lambda}
  &\le
  \int_1^R\int|K(x)||\widehat{g}(\lambda x)|
  \,dx\,\frac{d\lambda}{\lambda}\notag\\
  &=\int|K(x)|\!
  \left(\int_1^R|\widehat{g}(\lambda x)|
  \,\frac{d\lambda}{\lambda}\right)\!dx.
\end{align}
For fixed $x\ne 0$,
substitute $u=\lambda|x|$:
\begin{equation}\label{eq:inner}
  \int_1^R|\widehat{g}(\lambda x)|
  \,\frac{d\lambda}{\lambda}
  = \int_{|x|}^{R|x|}
  |\widehat{g}(u\,\mathrm{sgn}(x))|
  \,\frac{du}{u}.
\end{equation}
For $|u|\ge 1$:
$|\widehat{g}(u)|\le C|u|^{-1/2}$, so
\begin{equation}\label{eq:inner-bound}
  \int_{\max(|x|,1)}^{\infty}
  |\widehat{g}(u)|\,\frac{du}{u}
  \le C\int_{\max(|x|,1)}^\infty
  u^{-3/2}\,du
  = \frac{2C}{\max(|x|,1)^{1/2}}.
\end{equation}
Thus the inner integral is bounded
uniformly in $R$ by
$2C(1+|x|^{-1/2})$.
Monotone convergence
(or dominated convergence as $R\to\infty$)
passes the limit through.
Outer integral:
\begin{itemize}[leftmargin=*]
\item $|x|\le 1$:
  $|K(x)|\le\|K\|_{L^\infty}$
  and $\int_0^1|x|^{-1/2}\,dx=2$.
\item $|x|>1$:
  $|K(x)|=O(|x|^{-N})$;
  take $N=2$:
  $\int_1^\infty|x|^{-3/2}\,dx<\infty$.
\end{itemize}
Fubini--Tonelli is justified a
posterori by finiteness.
\end{proof}

\begin{corollary}[$\mathcal{C}(c)\to 0$]
\label{cor:cesaro-closed}
Under $K\in\mathcal{S}$ and the
PSWF WKB structure:
$\mathcal{C}(c)\to 0$.
\end{corollary}

\begin{proof}
Propositions~\ref{prop:Psi-L1}
and~\ref{prop:L1-sufficient}.
\end{proof}

\begin{remark}[Scaling gap:
  what $\mathcal{C}(c)\to 0$
  does \emph{not} give]
\label{rem:scaling-gap}
By Proposition~\ref{prop:model-limit}:
\begin{equation}\label{eq:sigma-model-scale}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}\Tcc(\psi)\,d\xi_0
    + o(1)
  \sim c^{2-\theta_0}\cdot\mathcal{C}(c).
\end{equation}
Since $c^{2-\theta_0}\to\infty$
for all $\theta_0\in(0,1)$,
the qualitative decay $\mathcal{C}(c)\to 0$
does \textbf{not} imply
$\Sigma_{\mathrm{model}}=o(1)$.

\medskip
\noindent\textbf{What is needed:}
\begin{equation}\label{eq:needed-rate}
  \mathcal{C}(c)
  = o\!\left(c^{-(2-\theta_0)}\right).
\end{equation}

\medskip
\noindent\textbf{What is proved:}
$\mathcal{C}(c)=o(1)$
(Corollary~\ref{cor:cesaro-closed}),
with no explicit rate.

\medskip
\noindent\textbf{The remaining issue
is not oscillatory, but scaling:}
The problem is that the
Ces\`aro theorem yields qualitative
decay, while the model sum
has a growing prefactor.
Closing this gap requires either:
\begin{enumerate}[\rm(A)]
\item A \emph{rate} for $\mathcal{C}(c)$,
  i.e., $|\mathcal{C}(c)|
  =O(c^{-(2-\theta_0)-\varepsilon})$
  for some $\varepsilon>0$;
\item A \emph{direct estimate}:
  $S(c)=o(K(c)\cdot c^{-(2-\theta_0)})
  =o(c^{-(2-2\theta_0)})$
  from the sum structure; or
\item The sum--integral exchange
  (R) in a \emph{quantitative} form
  that absorbs the $c^{2-\theta_0}$
  prefactor differently.
\end{enumerate}
\end{remark}

\begin{proposition}[Model reduction limit]
\label{prop:model-limit}
\begin{equation}\label{eq:model-limit}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}
    \Tcc(\psi)\,d\xi_0
    + O(c^{3\theta_0-2}).
\end{equation}
\end{proposition}

%%═══ 7. OPEN PROBLEMS
\section{Open Problems for Paper~XVI}
\label{sec:open}

\begin{problem}[Primary: scaling gap
  and rate for $\mathcal{C}(c)$]
\label{prob:rate-C}
\emph{This is now the primary
remaining obstacle.}

\medskip
\noindent\textbf{Required:}
$|\mathcal{C}(c)|=o(c^{-(2-\theta_0)})$.

\medskip
\noindent\textbf{Option~A
  (improve $\Psi$-decay).}
Show $|\Psi(\lambda)|=O(\lambda^{-\delta})$
for $\delta > 1/2$.
The qualitative barrier is:
the singularity
$g\sim|\alpha-\alpha_*|^{-1/2}$
has Fourier transform
exactly $O(|\xi|^{-1/2})$;
no power improvement comes from
the leading term.
However, the \emph{coefficient}
of the singularity at $\alpha_*$
has specific PSWF structure;
higher-order cancellation or
asymmetry may improve
the effective exponent.

\medskip
\noindent\textbf{Option~B
  (direct rate via oscillation).}
The sum
$S(c)=\int K(x)\sum_k
\widehat{g}(xk/c^{\theta_0})\,dx$
may admit cancellation between
terms beyond triangle inequality,
using the Dirichlet-like
oscillation of $\sum_ke^{ixk\alpha/c^{\theta_0}}$
and the decay of $K$.
Target:
$|S(c)|=O(c^{\theta_0(1-(2-\theta_0))})
= O(c^{2\theta_0-2})$.

\medskip
\noindent\textbf{Option~C
  (stationary phase, 2nd order).}
A second-order stationary phase
expansion near $\alpha_*$
may reveal additional decay:
the contribution near $\alpha_*$
to $\Psi(\lambda)$ may integrate
to $o(\lambda^{-1/2})$ after
cancellation between the two
branches $\xi_0>0$ and $\xi_0<0$
of the fold $\alpha_0(\xi_0)=\alpha_*$.
\end{problem}

\begin{problem}[Rigorous model reduction]
\label{prob:rigorous-reduction}
\textbf{Priority~1: Sum--integral exchange.}
Justify
$\sum_k\int\leftrightarrow\int\sum_k$
uniformly in $k\le c^{\theta_0}$
with quantitative error bound.
Note: the exchange must be
\emph{quantitative} (error
$o(c^{-(2-\theta_0)})$),
not merely qualitative,
to close the scaling gap.

\textbf{Priority~2: Kernel error.}
$\sum_k|(\AcM)_{nm}-I_c(|n-m|)|^2
=o(1)$ uniform in $c$;
naive bound fails for $\theta_0\ge 1/3$.

\textbf{Priority~3: PSWF Lipschitz~(L).}
$\|\widehat{\Phi}_{n+k}^{(c)}
-\widehat{\Phi}_n^{(c)}\|
=O(k/c)$.
Approaches: WKB
\cite{Zworski2012},
Sturm comparison,
microlocal calculus.
\end{problem}

%%═══ 8. REDUCTION THEOREM
\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Conditional on
$\mathcal{C}(c)=o(c^{-(2-\theta_0)})$,
(L), (R), and~(C):
\begin{equation}\label{eq:chain}
  \underbrace{
    \mathcal{C}=o(c^{-(2-\theta_0)})
    +(\mathrm{R})
    +(\mathrm{L})
  }_{\text{Paper XVI}}
  \Longrightarrow
  \Sigma_{\mathrm{int}} = o(1)
  \underset{(C)}{\Longrightarrow}
  \text{H3 fails}
  \Rightarrow\text{H1 fails.}
\end{equation}
\end{theorem}

\begin{remark}[Programme status]
\label{rem:series-status}
\[
  \underbrace{
    \text{rate for }\mathcal{C}
    +(\mathrm{R})+(\mathrm{L})
  }_{\text{Paper XVI}}
  +\underbrace{(C)}_{\text{XVII}}
  \Longrightarrow\text{H1 fails.}
\]
The oscillatory core is fully
identified and reduced.
The Ces\`aro obstruction is removed.
The remaining gap is the
\emph{scaling gap}:
the interaction between the
growing prefactor $c^{2-\theta_0}$
and the decay rate of $\mathcal{C}(c)$.
\end{remark}

\begin{remark}[Conceptual summary
  and publishable core]
\label{rem:concept-summary}
\begin{enumerate}[\rm(1)]
\item \emph{Dimension collapse}:
operator $\to$ Toeplitz energy
$\to$ scalar function $\Psi(\lambda)$.
\item \emph{Obstruction}:
frozen-phase saturation
$\Sigma_{\mathrm{strip}}
=\Theta(c^{2-\theta_0})$;
non-commuting limits.
\item \emph{Ces\`aro barrier}:
$\delta=1/2$ exactly critical;
Fubini argument removes it:
$\Psi\in L^1(d\lambda/\lambda)$
because $|x|^{-1/2}\in L^1_{\mathrm{loc}}$
and $K\in\mathcal{S}$.
\item \emph{Scaling gap precisely
identified}: need rate
$\mathcal{C}(c)=o(c^{-(2-\theta_0)})$,
not just $o(1)$.
\item \emph{Three paths to Paper~XVI}:
Option~A (better singularity),
Option~B ($k$-oscillation),
Option~C (stationary phase, 2nd order).
\end{enumerate}
\end{remark}
