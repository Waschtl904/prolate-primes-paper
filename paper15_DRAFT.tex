\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to a single
open problem: the intermediate-regime
Hilbert--Schmidt estimate
$\Sigma_{\mathrm{int}} = o(1)$
for $M(c)\sim\beta c$
(OP~\ref{prob:intermediate-XV}).
This paper isolates the core
analytic obstruction as a
\emph{mesoscopic Dirichlet kernel
summation problem}
(Problem~\ref{prob:model-intermediate}),
independent of PSWF structure,
and develops the tools required
for its resolution.

The central object is the oscillatory
kernel family
\[
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi,
  \quad 1\le k\le c^{\theta_0},
\]
and the associated sum
$\Sigma_{\mathrm{model}}(c)
:= \sum_{k=1}^{\lfloor c^{\theta_0}\rfloor}
|I_c(k)|^2$.
The two-scale structure of
$F_c(\xi,\eta)
:= \sum_{k=1}^{K(c)}
e^{ik(\theta(\xi)-\theta(\eta))/c}$
is identified: off-diagonal cancellation
is accessible by standard oscillatory
methods, while the diagonal mesoscopic
strip of width $\sim c^{1-\theta_0}$
requires a Dirichlet kernel analysis
at the critical scaling
$K(c)\cdot h = 1$ ($h := c^{-\theta_0}$).

The critical mesoscopic orthogonality
proposition (Proposition~\ref{prop:S3})
shows that the frozen-phase model
yields $\Sigma_{\mathrm{strip}}
= \Theta(c^{2-\theta_0})$,
establishing a \emph{no-go result}
for linear phase: the conjecture
(Model-$\Sigma$) requires genuinely
nonlinear phase ($\theta''\ne 0$).

The WKB phase extraction
(Lemma~\ref{lem:WKB-phase},
Lemma~\ref{lem:phase-nondeg})
identifies $\theta(\xi)$
explicitly from the PSWF ODE,
proves $\theta'\ne 0$ and
$\theta''\ne 0$ for $\xi\ne 0$
in the transition zone, and
shows that naive van~der~Corput
is insufficient
(Corollary~\ref{cor:curvature-available}).
The problem reduces to a
\emph{Weyl equidistribution}
problem for the phases
$\{k\theta'(\xi_0)/c^{\theta_0}\}$
(OP~\ref{prob:Weyl-equidist}),
with sub-case~(W1) already
giving $W(c)=O(c^{\theta_0/2})
=o(c^{\theta_0})$.
The final step is a decoupling
of the $k$-sum and $\xi_0$-integral
(W4).
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Inherited structure from Paper~XIV}

Paper~XIV established the following
reduction chain:
\begin{equation}\label{eq:chain-XIV}
  \underbrace{(C) + (D\text{-strong})}
  _{\text{Paper XV target}}
  \;\Rightarrow\;
  S_N^{(M(c))}(c)\sim c_0\beta > 0
  \;\Rightarrow\;
  \text{H3 fails}
  \;\Rightarrow\;
  \text{H1 fails}.
\end{equation}
The HS decomposition
\begin{equation}\label{eq:HS-decomp}
  \norm{\AcM - D^{(M)}}_{\mathrm{HS}}^2
  = \Sigma_{\mathrm{near}}
  + \Sigma_{\mathrm{int}}
  + \Sigma_{\mathrm{far}}
\end{equation}
satisfies:
$\Sigma_{\mathrm{near}} = O(1/c) = o(1)$
(proved, Paper~XIV, conditional on~(C));
$\Sigma_{\mathrm{far}} = o(1)$
(accessible via stationary phase);
and the sole remaining obstruction:
\begin{equation}\label{eq:int-target}
  \Sigma_{\mathrm{int}}
  = \sum_{\substack{n\ne m\\
      C<|n-m|\le c^{\theta_0}}}
    \left|(\AcM)_{nm}\right|^2
  \stackrel{?}{=} o(1),
  \qquad M(c)\sim\beta c.
\end{equation}

\subsection{Three levels of decay}

\begin{remark}[Three levels of $o(1)$]
\label{rem:three-levels}
The symbol $o(1)$ appears at three
distinct levels throughout this paper;
these are \emph{not} interchangeable:
\begin{enumerate}[\rm(i)]
\item \emph{Pointwise decay}:
  $|(\AcM)_{nm}| = o(1/c)$
  for each fixed pair $(n,m)$
  with $|n-m|\ge C$.
  This is (D-weak); it does not
  imply HS-summability over
  $O(c^2)$ pairs.
\item \emph{HS-summated decay}:
  $\Sigma_{\mathrm{near}}=o(1)$,
  $\Sigma_{\mathrm{far}}=o(1)$,
  $\Sigma_{\mathrm{int}}=o(1)$.
  This is the operative level
  for (D-strong).
\item \emph{Uniform in $\theta_0$}:
  the crude bound
  $O(c^{\theta_0-1})=o(1)$
  holds for each fixed $\theta_0<1$
  but \emph{not} uniformly as
  $\theta_0\to 1^-$.
  This is the structural gap
  of OP~\ref{prob:intermediate-XV}.
\end{enumerate}
The near-diagonal regime is controlled
at level~(ii) without passing
through~(i).
The intermediate regime is controlled
at~(ii) for each fixed $\theta_0$;
the gap to uniform control is
the core of this paper.
\end{remark}

\subsection{The model reduction}

The Gram kernel takes the form
\begin{equation}\label{eq:gram-kernel}
  (\AcM)_{nm}
  = \int_{-c_0}^{c_0}
    \widehat{\Phi_n^{(c)}}(\xi)\,
    \overline{\widehat{\Phi_m^{(c)}}(\xi)}
    \,d\xi.
\end{equation}
Under assumption~(C) and the Lipschitz
bound
\begin{equation}\label{eq:Lipschitz-main}
  \norm{A_{n+k}-A_n}_{L^2([-c_0,c_0])}
  = O(k/c),
  \qquad |k|\le c^{\theta_0},
\end{equation}
the amplitude correction
$A_n\overline{A_m} - w$
contributes $O(k/c)$ per entry,
which is HS-summable.
The leading term reduces to
the model kernel:
\begin{equation}\label{eq:model-reduction}
  (\AcM)_{nm}
  \approx I_c(|n-m|)
  + O(|n-m|/c),
  \qquad
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi.
\end{equation}
Thus $\Sigma_{\mathrm{int}}=o(1)$
follows from
$\Sigma_{\mathrm{model}}(c)=o(1)$
once~\eqref{eq:Lipschitz-main}
is established (\S\ref{sec:Lipschitz}).

%%═══ 2. THE MODEL PROBLEM
\section{The Intermediate-Regime
  Model Problem}
\label{sec:model}

\begin{problem}[Intermediate-regime
  kernel summation;
  model problem for Paper~XV]
\label{prob:model-intermediate}

Let $\theta\in C^\infty([-c_0,c_0])$
satisfy $\theta'(\xi)\ne 0$ on
$[-c_0,c_0]$, and let
$w\in C^\infty([-c_0,c_0])$,
$w>0$, with
$\int_{-c_0}^{c_0}w(\xi)\,d\xi=1$.

Define the oscillatory kernel
coefficients
\begin{equation}\label{eq:Ic-def}
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi,
  \qquad k\in\mathbb{Z}_{\ge 1}.
\end{equation}
Fix $\theta_0\in(0,1)$ and set
$K(c):=\lfloor c^{\theta_0}\rfloor$.

\medskip\noindent
\textbf{Conjecture (Model-$\Sigma$).}
\begin{equation}\label{eq:model-sigma}
  \Sigma_{\mathrm{model}}(c)
  := \sum_{k=1}^{K(c)}|I_c(k)|^2
  \;=\; o(1)
  \qquad(c\to\infty).
\end{equation}

\medskip\noindent
\textbf{Failure of pointwise estimates.}
Integration by parts gives for
each fixed $k\ge 1$:
\begin{equation}\label{eq:IBP}
  I_c(k)
  = \frac{c}{ik}
    \int_{-c_0}^{c_0}
    e^{ik\theta(\xi)/c}
    \partial_\xi\!\left(
      \frac{w(\xi)}{\theta'(\xi)}
    \right)d\xi,
  \qquad
  |I_c(k)|=O\!\left(\frac{c}{k}\right).
\end{equation}
Applying this pointwise:
$\sum_{k=1}^{K(c)}|I_c(k)|^2
\lesssim c^2\sum_{k=1}^{c^{\theta_0}}
k^{-2} = O(c^2)$.
Pointwise oscillatory decay is
therefore not the right tool.

\medskip\noindent
\textbf{Kernel reformulation.}
\begin{equation}\label{eq:kernel-rewrite}
  \Sigma_{\mathrm{model}}(c)
  = \int_{-c_0}^{c_0}
    \int_{-c_0}^{c_0}
    w(\xi)\,w(\eta)\,
    F_c(\xi,\eta)
    \,d\xi\,d\eta,
\end{equation}
where
\begin{equation}\label{eq:Fc}
  F_c(\xi,\eta)
  := \sum_{k=1}^{K(c)}
     e^{ik(\theta(\xi)-\theta(\eta))/c}
  = e^{i\phi/c}
    \cdot
    \frac{1-e^{iK(c)\phi/c}}
         {1-e^{i\phi/c}},
  \quad
  \phi:=\theta(\xi)-\theta(\eta).
\end{equation}
Thus $F_c$ is a rescaled Dirichlet
kernel in $\phi/c$.

\medskip\noindent
\textbf{Two-scale structure.}
Let $\delta:=|\xi-\eta|$.
\begin{itemize}[leftmargin=*]
\item \textbf{Off-diagonal}
  ($\delta\gg c^{1-\theta_0}$):
  oscillatory cancellation gives
  $|F_c(\xi,\eta)|\lesssim c/|\phi|$.
\item \textbf{Diagonal mesoscopic strip}
  ($\delta\lesssim c^{1-\theta_0}$):
  $|F_c(\xi,\eta)|\le K(c)=c^{\theta_0}$;
  no cancellation.
  Strip measure $\sim c^{1-\theta_0}$
  gives $O(1)$ contribution:
  new input is required.
\end{itemize}

\medskip\noindent
\textbf{Mesoscopic rescaling.}
Fix $\xi_0$ and set
$\xi=\xi_0+c^{1-\theta_0}u$,
$\eta=\xi_0+c^{1-\theta_0}v$:
\begin{equation}\label{eq:meso-F}
  F_c(u,v)
  = D_{K(c)}\!\big(
       h\,\theta'(\xi_0)(u-v)
     \big),
  \qquad
  h:=c^{-\theta_0},
  \quad K(c)\cdot h = 1.
\end{equation}
Both $K(c)\to\infty$ and $h\to 0$
simultaneously at critical product~$1$.

\medskip\noindent
\textbf{Reduction statement.}
Under~\eqref{eq:Lipschitz-main},
$\Sigma_{\mathrm{model}}(c)=o(1)$
$\Rightarrow$
$\Sigma_{\mathrm{int}}=o(1)$
$\Rightarrow$ (D-strong)
$\Rightarrow$ closes OP~XIV.\ref{prob:H1H3}.

\medskip\noindent
\textbf{Proposed strategy.}
\begin{enumerate}[\rm(S1)]
\item Off-diagonal stationary phase.
\item Mesoscopic rescaling
  to~\eqref{eq:meso-F}.
\item $L^2$-orthogonality / Weyl
  equidistribution across
  the diagonal strip.
\item Residual bound via
  Lipschitz~\eqref{eq:Lipschitz-main}.
\end{enumerate}
\end{problem}

\begin{remark}[Connection to
  Toeplitz/Szeg\H{o} theory]
\label{rem:toeplitz}
The kernel $D_{K(c)}(h\theta'(\xi_0)(u-v))$
is a finite Toeplitz section at
critical density $K(c)\cdot h=1$.
This is the double-scaling
Szeg\H{o} regime of
Definition~XIV.\ref{def:double-scaling}
instantiated at the mesoscopic level:
neither the classical Szeg\H{o} theorem
nor the semiclassical limit applies directly.
\end{remark}

%%═══ 3. CRITICAL MESOSCOPIC ORTHOGONALITY
\section{Critical Mesoscopic Orthogonality}
\label{sec:S3}

\subsection{The rescaled kernel operator}

After the mesoscopic rescaling
of \S\ref{sec:model}, the
diagonal-strip contribution is
\begin{equation}\label{eq:strip-contrib}
  \Sigma_{\mathrm{strip}}(c)
  := c^{2(1-\theta_0)}
     \int_{\mathbb{R}}
     \int_{\mathbb{R}}
     \psi(u)\,\overline{\psi(v)}\,
     D_{K(c)}\!\big(
       h\,\alpha_0(u-v)
     \big)
     \,du\,dv,
\end{equation}
where $\alpha_0:=\theta'(\xi_0)\ne 0$,
$\psi(u):=w(\xi_0+c^{1-\theta_0}u)
\cdot\chi_{|u|\le c^{\theta_0-1/2}}$,
$h=c^{-\theta_0}$, $K(c)=c^{\theta_0}$.

\begin{definition}[Toeplitz--Dirichlet
  operator at critical density]
\label{def:TD-operator}
\begin{equation}\label{eq:Dc-def}
  (\mathcal{D}_c f)(u)
  := \int_{\mathbb{R}}
     D_{K(c)}\!\big(h\,\alpha_0(u-v)\big)
     \psi(v)\,f(v)\,dv,
  \qquad
  \Sigma_{\mathrm{strip}}(c)
  = c^{2(1-\theta_0)}
    \ip{\mathcal{D}_c\psi}{\psi}.
\end{equation}
The kernel is a convolution with
Fourier spectrum
$\{\xi = k\alpha_0/c^{\theta_0}
: k=1,\ldots,K(c)\}$
at critical spacing $\alpha_0/c^{\theta_0}$.
\end{definition}

\subsection{The proposition}

\begin{proposition}[Critical
  mesoscopic orthogonality]
\label{prop:S3}
Let $\psi\in C_c^\infty(\mathbb{R})$,
$\alpha_0\ne 0$, $h=c^{-\theta_0}$,
$K(c)=\lfloor c^{\theta_0}\rfloor$.

\medskip
\noindent\textbf{(i) Fourier representation.}
\begin{equation}\label{eq:Sigma-fourier}
  \Sigma_{\mathrm{strip}}(c)
  = c^{2(1-\theta_0)}
    \sum_{k=1}^{K(c)}
    \left|
      \widehat{\psi}\!\left(
        \frac{k\alpha_0}{c^{\theta_0}}
      \right)
    \right|^2.
\end{equation}

\noindent\textbf{(ii) Riemann sum structure.}
\begin{equation}\label{eq:riemann}
  \frac{1}{c^{\theta_0}}
  \sum_{k=1}^{K(c)}
  \left|\widehat{\psi}\!\left(
    \frac{k\alpha_0}{c^{\theta_0}}
  \right)\right|^2
  \;\longrightarrow\;
  \frac{1}{|\alpha_0|}
  \int_0^{|\alpha_0|}
  |\widehat{\psi}(\xi)|^2\,d\xi
  \qquad(c\to\infty),
\end{equation}
hence
$\sum_{k=1}^{K(c)}
|\widehat{\psi}(k\alpha_0/c^{\theta_0})|^2
= c^{\theta_0}\cdot C(\psi,\alpha_0)
+ o(c^{\theta_0})$
with $C(\psi,\alpha_0)
:=\norm{\psi}^2_{L^2}/|\alpha_0|>0$.

\noindent\textbf{(iii) Scaling conclusion.}
\begin{equation}\label{eq:strip-final}
  \Sigma_{\mathrm{strip}}(c)
  = c^{2-\theta_0}\cdot C(\psi,\alpha_0)
    + o(c^{2-\theta_0})
  = \Theta(c^{2-\theta_0}).
\end{equation}
\end{proposition}

\begin{proof}
Since $D_K(t)=\sum_{k=1}^K e^{ikt}$:
\begin{align*}
  \Sigma_{\mathrm{strip}}(c)
  &= c^{2(1-\theta_0)}
     \sum_{k=1}^{K(c)}
     \left|\int\psi(u)
       e^{ikh\alpha_0 u}\,du
     \right|^2
  = c^{2(1-\theta_0)}
     \sum_{k=1}^{K(c)}
     \left|\widehat{\psi}
       (-k\alpha_0/c^{\theta_0})
     \right|^2.
\end{align*}
Part~(ii) follows since
$\xi\mapsto|\widehat{\psi}(\xi)|^2$
is in $C_c^\infty$ and the sum is a
Riemann sum with mesh
$|\alpha_0|/c^{\theta_0}\to 0$.
Part~(iii): $c^{2(1-\theta_0)}
\cdot c^{\theta_0}
= c^{2-\theta_0}$.
\end{proof}

\begin{corollary}[No-go for linear phase]
\label{cor:S3-negative}
Under the frozen-phase approximation
$\theta(\xi)\approx\alpha_0\xi$
(constant $\alpha_0$):
$\Sigma_{\mathrm{strip}}(c)
=\Theta(c^{2-\theta_0})\not\to 0$.
Strategy~(S3) as formulated
does not yield $\Sigma_{\mathrm{model}}=o(1)$
without additional input.
Conjecture~(Model-$\Sigma$) \emph{fails}
for linear $\theta$.
\end{corollary}

\begin{remark}[Two escapes]
\label{rem:two-escapes}
Decay of $\Sigma_{\mathrm{model}}$
requires one of:

\medskip
\noindent\textbf{Escape~A (phase curvature):}
$\theta''\ne 0$ induces varying mesh
$k|\theta''(\xi_0)|/c^{\theta_0}$
in the Riemann sum, leading to
stationary phase decay in the
$\xi_0$-integral.

\medskip
\noindent\textbf{Escape~B (phase mixing):}
Averaging over $\xi_0$ with
varying $\alpha_0=\theta'(\xi_0)$
produces Riemann--Lebesgue
cancellation, again requiring
$\theta''\ne 0$.

\medskip
\noindent\textbf{Common requirement:}
$\theta''\not\equiv 0$ on $[-c_0,c_0]$.
\end{remark}

%%═══ 4. WKB PHASE EXTRACTION
\section{WKB Phase Extraction}
\label{sec:WKB}

\subsection{The PSWF ODE and
  semiclassical scaling}

The PSWFs satisfy the
Sturm--Liouville equation
\begin{equation}\label{eq:PSWF-ODE}
  \frac{d}{dx}\!\left[
    (1-x^2)\Phi_n^{(c)\prime}
  \right]
  + \left[\chi_n(c)-c^2x^2\right]
    \Phi_n^{(c)} = 0,
  \quad x\in[-1,1].
\end{equation}
The Fourier transform
$\widehat{\Phi_n^{(c)}}$ satisfies
the dual ODE with $(c^2-\xi^2)$
in place of $(1-x^2)$.
Setting $n=\alpha c$, $h=1/c$,
$\xi=c\zeta$, the semiclassical
energy level is
$\lambda(\alpha):=
\lim_{c\to\infty}\chi_{\alpha c}(c)/c^2$,
with classical momentum
\begin{equation}\label{eq:momentum}
  p(\zeta;\alpha)
  :=
  \sqrt{\frac{\lambda(\alpha)-\zeta^2}
             {1-\zeta^2}},
  \qquad
  |\zeta|<\sqrt{\lambda(\alpha)}.
\end{equation}

\subsection{Phase function}

\begin{lemma}[WKB phase extraction]
\label{lem:WKB-phase}
In the classically allowed region
$|\xi|<c\sqrt{\lambda(\alpha)}$,
the leading WKB approximation gives
\begin{equation}\label{eq:WKB-ansatz}
  \widehat{\Phi_n^{(c)}}(\xi)
  \approx_{\mathrm{WKB}}
  \frac{A(\xi/c;\alpha)}
       {\sqrt{p(\xi/c;\alpha)}}
  \exp\!\left(
    ic\int_0^{\xi/c}
    p(\zeta;\alpha)\,d\zeta
  \right),
\end{equation}
with $A$ real, positive, smooth
away from turning points.
The phase function in the model
kernel $I_c(k)$ is
\begin{equation}\label{eq:theta-def}
  \theta(\xi)
  := c\int_0^{\xi/c}
     p(\zeta;\alpha)\,d\zeta
  = \int_0^{\xi/c}
    \sqrt{\frac{\lambda(\alpha)-\zeta^2}
               {1-\zeta^2}}\,
    c\,d\zeta,
\end{equation}
so that $k\theta(\xi)/c
= k\int_0^{\xi/c}p(\zeta;\alpha)\,d\zeta$.
\end{lemma}

\subsection{Non-degeneracy}

\begin{lemma}[Phase non-degeneracy]
\label{lem:phase-nondeg}
For $\alpha\in(0,1)$ and
$|\xi|\le c_0<c\sqrt{\lambda(\alpha)}$:

\medskip
\noindent\textbf{(i)} $\theta'(\xi)
= p(\xi/c;\alpha)
= \sqrt{(\lambda(\alpha)-(\xi/c)^2)
/(1-(\xi/c)^2)} > 0$.

\medskip
\noindent\textbf{(ii)} For $\xi\ne 0$:
\begin{equation}\label{eq:theta-pp}
  \theta''(\xi)
  = \frac{-(\xi/c)(1-\lambda(\alpha))}
         {c(1-(\xi/c)^2)^{3/2}
           \sqrt{\lambda(\alpha)-(\xi/c)^2}}
  \ne 0,
\end{equation}
since $\lambda(\alpha)\in(0,1)$
for all $\alpha\in(0,1)$.

\medskip
\noindent\textbf{(iii)} Quantitative bound:
$|\theta''(\xi)|\ge
c_*|\xi|/c$
for $|\xi|\ge\varepsilon>0$,
with $c_*=c_*(\alpha,c_0)>0$.
Near $\xi=0$: $\theta''(0)=0$
but $\theta'''(0)\ne 0$
(cubic non-degeneracy;
see Remark~\ref{rem:turning-zero}).
\end{lemma}

\begin{proof}
Differentiate $\theta'(\xi)
= p(\xi/c;\alpha)$ using the
quotient rule; $\lambda(\alpha)\in(0,1)$
gives the sign and non-vanishing.
\end{proof}

\begin{remark}[$\lambda(\alpha)$]
\label{rem:lambda-alpha}
At leading WKB order
\cite{Osipov2013}:
$\lambda(\alpha)
= \sin^2(\pi\alpha/2)+O(1/c)$,
so $\lambda(\alpha)\in(0,1)$
for $\alpha\in(0,1)$,
confirming $\theta''\ne 0$.
\end{remark}

\begin{remark}[Turning point at $\xi=0$]
\label{rem:turning-zero}
$\theta''(0)=0$ reflects the
even/odd symmetry of PSWFs
and is not a genuine obstruction:
a neighbourhood $|\xi|\le\varepsilon$
contributes $O(\varepsilon)$
to integrals, which is $o(1)$
for $\varepsilon\to 0$.
The cubic non-degeneracy
$\theta'''(0)\ne 0$ gives
Airy-type control in that region.
\end{remark}

\subsection{van der Corput is insufficient}

\begin{corollary}[Curvature available
  but naive vdC insufficient]
\label{cor:curvature-available}
The PSWF phase satisfies
$\theta''\ne 0$ (Lemma~\ref{lem:phase-nondeg}),
so both Escape~A and Escape~B
(Remark~\ref{rem:two-escapes}) are
in principle available.
However, applying van~der~Corput
with $|\theta''|\sim\varepsilon/c$:
\begin{equation}\label{eq:VdC-PSWF}
  |R(k,c)|
  \lesssim
  \sqrt{\frac{c^{1+\theta_0}}
             {k\varepsilon}},
  \qquad
  \sum_{k=1}^{c^{\theta_0}}|R(k,c)|
  \sim
  \frac{c^{(1+2\theta_0)/2}}
       {\sqrt{\varepsilon}}
  = \Omega(c^{\theta_0})
  \quad(\theta_0>0).
\end{equation}
The curvature scale $1/c$ exactly
matches the frequency scale
$1/c^{\theta_0}$, and no gain
results from naive vdC.
The missing ingredient is
the \emph{joint} summation in
$k$ and $\xi_0$.
\end{corollary}

%%═══ 5. WEYL EQUIDISTRIBUTION
\section{Reduction to Weyl
  Equidistribution}
\label{sec:Weyl}

\begin{remark}[From curvature to
  equidistribution]
\label{rem:what-needed}
After the mesoscopic rescaling,
$\Sigma_{\mathrm{strip}}(c)$
involves the sum
$\sum_{k=1}^{K(c)}
|\widehat{\psi}(k\theta'(\xi_0)/c^{\theta_0})|^2$
integrated over $\xi_0$.
For varying $\xi_0$ with
$\alpha_0(\xi_0)=\theta'(\xi_0)$
non-constant ($\theta''\ne 0$),
the phases $k\alpha_0(\xi_0)/c^{\theta_0}$
are equidistributed as $k$ varies
(Weyl), giving
\begin{equation}\label{eq:Weyl-sum}
  \sum_{k=1}^{K(c)}
  e^{2\pi i k\beta/c^{\theta_0}}
  = o(K(c))
  \quad\text{for irrational }\beta,
\end{equation}
with rate $O(c^{\theta_0/2})$
via Weyl differencing.
The problem is to make this
uniform in $\xi_0$ and decouple
the $k$-sum from the $\xi_0$-integral.
\end{remark}

\begin{problem}[Weyl equidistribution
  for PSWF phase;
  decisive open problem]
\label{prob:Weyl-equidist}
Let $\beta:=\theta'(\xi_0)
=p(\xi_0/c;\alpha)\sim\sqrt{\lambda(\alpha)}$.
Establish a useful bound for
\begin{equation}\label{eq:Weyl-target}
  W(c)
  :=
  \left|
    \sum_{k=1}^{K(c)}
    e^{2\pi ik\beta/c^{\theta_0}}
  \right|
  = o(K(c)).
\end{equation}

\medskip\noindent
\textbf{Four sub-cases:}
\begin{enumerate}[\rm(W1)]
\item \emph{Fixed irrational $\beta$:}
  $W(c)=O(c^{\theta_0/2})
  =o(c^{\theta_0})$
  by Weyl differencing.
  \textbf{Available.}

\item \emph{Slowly varying $\beta(c)$:}
  $\beta(c)=\sqrt{\lambda(\alpha)}+O(1/c)$.
  Perturbation shifts phases by
  $O(k/c)=O(c^{\theta_0-1})=o(1)$.
  Negligible for $\theta_0<1$.
  \textbf{Controllable.}

\item \emph{Averaging over $\xi_0$:}
  Van~der~Corput with phase
  $\phi(\xi_0)=k\theta'(\xi_0)/c^{\theta_0}$,
  $|\phi''|=k|\theta''(\xi_0)|/c^{\theta_0}
  \sim k/(c^{\theta_0+1})$ gives
  $\int e^{2\pi i k\theta'(\xi_0)/c^{\theta_0}}
  d\xi_0\lesssim c^{(\theta_0+1)/2}/k^{1/2}$.
  Summing: $\sim c^{\theta_0+1/2}$.
  Compare to main term $c^{\theta_0}$:
  ratio $c^{1/2}\to\infty$.
  \textbf{Not sufficient alone.}

\item \emph{Joint $k$-and-$\xi_0$
  decoupling (final step):}
  Treat $\sum_{k=1}^{K(c)}
  \int e^{ik\theta'(\xi_0)/c^{\theta_0}}
  \psi(\xi_0)\,d\xi_0$
  as a double exponential sum.
  For $\theta''\ne 0$: the curve
  $k\theta'(\xi_0)=\lambda c^{\theta_0}$
  has curvature $\sim k\theta''/c^{\theta_0}$,
  and Weyl differencing in both
  $k$ and $\xi_0$ simultaneously
  is expected to give the required gain.
  \textbf{Open; primary target of Paper~XV.}
\end{enumerate}

\medskip\noindent
\textbf{Key observation.}
Sub-case~(W1) already gives
$W(c)=O(c^{\theta_0/2})=o(c^{\theta_0})$
for each fixed irrational $\beta$.
If the $\xi_0$-integral can be
decoupled from the $k$-sum,
$\Sigma_{\mathrm{model}}=o(1)$ follows.
This decoupling is the
\emph{final technical step}.
\end{problem}

\begin{center}
\fbox{\parbox{0.88\linewidth}{
\textbf{Final logical status.}
\begin{itemize}[leftmargin=*]
\item $\theta(\xi)$ identified:
  \textbf{done} (Lem.~\ref{lem:WKB-phase}).
\item $\theta'\ne 0$:
  \textbf{proved} (Lem.~\ref{lem:phase-nondeg}(i)).
\item $\theta''\ne 0$ ($\xi\ne 0$):
  \textbf{proved} (Lem.~\ref{lem:phase-nondeg}(ii)).
\item Linear phase $\Rightarrow$
  Model-$\Sigma$ fails:
  \textbf{proved}
  (Cor.~\ref{cor:S3-negative}).
\item Naive vdC insufficient:
  \textbf{proved}
  (Cor.~\ref{cor:curvature-available}).
\item Weyl for fixed $\beta$:
  $O(c^{\theta_0/2})$
  \textbf{available} (W1).
\item Joint $k$/$\xi_0$ decoupling:
  \textbf{open; sole remaining step}
  (W4).
\item $\Sigma_{\mathrm{model}}=o(1)$:
  \textbf{conditional on (W4)}.
\end{itemize}}}
\end{center}

%%═══ 6. LIPSCHITZ REGULARITY
\section{Lipschitz Regularity of the
  PSWF Amplitude}
\label{sec:Lipschitz}

\begin{problem}[Lipschitz regularity]
\label{prob:Lipschitz}
\emph{Priority: critical.}

Establish uniformly in
$n\in\mathcal{T}_{\alpha,\kappa}(c)$,
$|k|\le c^{\theta_0}$:
\begin{equation}\label{eq:Lip-prob}
  \norm{
    \widehat{\Phi_{n+k}^{(c)}}
    -\widehat{\Phi_n^{(c)}}
  }_{L^2([-c_0,c_0])}
  = O(k/c).
\end{equation}
Follows at WKB level from
smooth $n/c$-dependence of the
amplitude $A(\zeta;\alpha)$;
rigorous justification requires
$L^2$-control of the semiclassical
transport equation.
\end{problem}

%%═══ 7. LOCAL WEYL LAW
\section{Local Weyl Law (C)}
\label{sec:WeylLaw}

\begin{problem}[Local Weyl law]
\label{prob:Weyl-XV}
\emph{Priority: critical.
Inherited from OP~XIV.\ref{prob:fourier-mass}.}

Prove uniformly in
$n\in\mathcal{T}_{\alpha,\kappa}(c)$:
\begin{equation}\label{eq:Weyl-uniform}
  \mu_{n,c}([-c_0,c_0])
  = \frac{c_0}{c}
  + o(1/c).
\end{equation}
Required for frozen weight
approximation, $\Sigma_{\mathrm{near}}=o(1)$,
and Prop.~XIV.\ref{prop:HS-reduction}.
Main difficulty: uniformity as
$n/c\to\alpha$ near the turning point.
\end{problem}
