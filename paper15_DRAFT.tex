\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to the
intermediate-regime Hilbert--Schmidt
estimate $\Sigma_{\mathrm{int}}=o(1)$
for $M(c)\sim\beta c$.
This paper identifies the core
analytic structure, develops the
reduction toolkit, and closes the
central oscillatory question.

The main objects are the oscillatory
kernel family
$I_c(k):=\int e^{ik\theta(\xi)/c}w(\xi)\,d\xi$
and the model sum
$\Sigma_{\mathrm{model}}(c)
:=\sum_{k=1}^{K(c)}|I_c(k)|^2$.
The mesoscopic rescaling identifies
$\Sigma_{\mathrm{model}}$ with the
critical Toeplitz energy
$\Tcc(\psi)=\ip{\Dc\psi}{\psi}$
at the double-scaling regime
$K(c)\cdot h=1$.

The WKB analysis
(Lemmas~\ref{lem:WKB-phase},
\ref{lem:phase-nondeg})
shows that pointwise oscillatory
tools (van~der~Corput, Weyl/UDPL)
are insufficient; the obstruction
is \emph{lack of cancellation
after phase freezing}.
A coarea substitution
(Proposition~\ref{prop:alpha-oscillation})
reduces the model target
(T\textsuperscript{*}) to the
weighted Riemann--Lebesgue condition
\begin{equation}\label{eq:wRL-abstract}
  \int_{\mathbb{R}} K(x)\,
  \widehat{g}(\lambda x)\,dx
  \longrightarrow 0
  \quad(\lambda\to\infty),
\end{equation}
where $K=\psi*\widetilde{\psi}\in L^1$
and $g=1/|\theta''(\xi_0(\cdot))|
\in L^1$
(Lemma~\ref{lem:g-singularity}).
Since $g\in L^1$ implies
$\widehat{g}\in C_0$,
condition~\eqref{eq:wRL-abstract}
follows by dominated convergence:
\emph{(T\textsuperscript{*}) is proved}
(Theorem~\ref{thm:Tstar},
Lemma~\ref{lem:weighted-RL}(a)).
With $K\in\mathcal{S}$ and
$|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$,
one obtains the quantitative rate
$\Sigma_{\mathrm{model}}=O(c^{-\theta_0/2})$
(Lemma~\ref{lem:weighted-RL}(c)).

The remaining open problems
for Paper~XVI are the
\emph{rigorous reduction steps},
not new oscillatory-integral ideas:
\begin{enumerate}[\rm(i)]
\item Kernel error control:
  $\Sigma_{\mathrm{int}}
  = \Sigma_{\mathrm{model}} + o(1)$
  (uniform in $k\le c^{\theta_0}$).
\item PSWF Lipschitz bound~(L):
  $\norm{\widehat{\Phi_{n+k}^{(c)}}
  - \widehat{\Phi_n^{(c)}}}=O(k/c)$.
\end{enumerate}
Conditional on these:
\[
  \mathrm{(L)}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}}=o(1)
  \;\underset{\mathrm{(C)}}{\Longrightarrow}\;
  \text{H1 fails}.
\]
\end{abstract}

%%═══ 2. THE MODEL PROBLEM
\section{The Intermediate-Regime
  Model Problem}
\label{sec:model}

\begin{problem}[Model problem]
\label{prob:model-intermediate}
With $K(c):=\lfloor c^{\theta_0}\rfloor$:
\begin{equation}\label{eq:model-sigma}
  \Sigma_{\mathrm{model}}(c)
  := \sum_{k=1}^{K(c)}|I_c(k)|^2.
\end{equation}
Kernel reformulation:
\begin{equation}\label{eq:kernel-rewrite}
  \Sigma_{\mathrm{model}}
  = \iint w(\xi)w(\eta)
    F_c(\xi,\eta)\,d\xi\,d\eta,
  \quad
  F_c(\xi,\eta)
  := \sum_{k=1}^{K(c)}
  e^{ik(\theta(\xi)-\theta(\eta))/c}.
\end{equation}
Mesoscopic rescaling at $\xi_0$
($h=c^{-\theta_0}$, $K(c)\cdot h=1$):
\begin{equation}\label{eq:meso-F}
  F_c(u,v)
  = D_{K(c)}(h\,\theta'(\xi_0)(u-v)).
\end{equation}
\emph{Double-scaling Szeg\H{o} regime:}
not covered by standard Szeg\H{o}
or semiclassical theory.
\end{problem}

%%═══ 3. CRITICAL MESOSCOPIC ORTHOGONALITY
\section{Critical Mesoscopic Orthogonality}
\label{sec:S3}

\begin{definition}[Toeplitz energy]
\label{def:TD-operator}
$\alpha_0:=\theta'(\xi_0)$,
$h=c^{-\theta_0}$:
\begin{equation}\label{eq:Tc-def}
  \Tcc(\psi)
  = \sum_{k=1}^{K(c)}
    |\widehat{\psi}(k\alpha_0/c^{\theta_0})|^2,
  \quad
  \Sigma_{\mathrm{strip}}
  = c^{2(1-\theta_0)}\Tcc(\psi).
\end{equation}
\end{definition}

\begin{proposition}[Frozen-$\xi_0$ saturation]
\label{prop:S3}
$\Sigma_{\mathrm{strip}}(c)
= c^{2-\theta_0}C(\psi,\alpha_0)
+ o(c^{2-\theta_0})
= \Theta(c^{2-\theta_0})$
for fixed $\alpha_0\ne 0$.
\emph{Scope:} frozen-$\xi_0$ only;
$\xi_0$-integration is the subject of
Proposition~\ref{prop:model-limit}.
\end{proposition}

\begin{corollary}[No-go for frozen phase]
\label{cor:S3-negative}
$\Sigma_{\mathrm{strip}}\to+\infty$;
no pointwise mechanism suffices.
\end{corollary}

\begin{remark}[Two escapes]
\label{rem:two-escapes}
\textbf{Escape~A} ($\xi_0$-variation):
$\alpha_0(\xi_0)$ strictly monotone.
\textbf{Escape~B} (PSWF orthogonality
in $w$). Both require PSWF structure.
\end{remark}

%%═══ 4. WKB PHASE EXTRACTION
\section{WKB Phase Extraction}
\label{sec:WKB}

\begin{lemma}[WKB phase]
\label{lem:WKB-phase}
$\theta(\xi):=c\int_0^{\xi/c}
p(\zeta;\alpha)\,d\zeta$,
$w(\xi):=|A(\xi/c;\alpha)|^2/p(\xi/c;\alpha)$,
$p(\zeta;\alpha):=\sqrt{(\lambda(\alpha)-\zeta^2)/(1-\zeta^2)}$.
\end{lemma}

\begin{lemma}[Phase non-degeneracy]
\label{lem:phase-nondeg}
(i) $\theta'>0$;\;
(ii) $\theta''(\xi)\ne 0$ for $\xi\ne 0$,
$\theta''(\xi)\sim -\xi/c$ near $0$;\;
(iii) $\theta''(0)=0$,
$\theta'''(0)\ne 0$;\;
(iv) $\xi_0\mapsto\alpha_0(\xi_0)
:=\theta'(\xi_0)$ strictly monotone.
\end{lemma}

\begin{proof}
Differentiation of $p(\xi/c;\alpha)$;
$\lambda(\alpha)=\sin^2(\pi\alpha/2)+O(1/c)$
\cite{Osipov2013}.
\end{proof}

\begin{corollary}[vdC insufficient]
\label{cor:curvature-available}
$\sum_k|I_c(k)|^2
\lesssim(c/\varepsilon)\log c
=\Omega(c)$;
both pointwise vdC and Weyl/UDPL fail.
\end{corollary}

Under uniform bounds (P1)--(P3)

\end{lemma}

\begin{proposition}[Joint Decoupling (W4)]
\label{prop:W4-decoupling}
$|T(c)|=O(c^{3\theta_0/4+\varepsilon})
=o(c^{\theta_0})$.
\end{proposition}

\begin{remark}[Scope of Weyl/UDPL]
\label{rem:what-proved}

Gap: $2-5\theta_0/4>0$ for all
$\theta_0\in(0,1)$.
\end{remark}

%%═══ 6. THE CRITICAL TOEPLITZ OBSTRUCTION
\section{The Critical Toeplitz
  Obstruction and Its Resolution}
\label{sec:toeplitz}

\begin{remark}[Core thesis]
\label{rem:core-thesis}
\begin{center}
\emph{The obstruction is not analytic
weakness, but absence of cancellation
after phase freezing.}
\end{center}
Every pointwise tool fails;
cancellation is global in $\xi_0$.
\end{remark}

\begin{remark}[Sampling reinterpretation]
\label{rem:sampling-rewrite}
\begin{equation}\label{eq:sampling}
  \Sigma_{\mathrm{model}}
  \approx
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \sum_{k=1}^{K(c)}
  \left|\widehat{\psi}\!\left(
    \tfrac{k\alpha_0(\xi_0)}{c^{\theta_0}}
  \right)\right|^2
  d\xi_0.
\end{equation}
The grids $\{k\alpha_0(\xi_0)/c^{\theta_0}\}$
shift as $\xi_0$ varies;
this is \emph{randomised sampling}
of $|\widehat\psi|^2$ along the
curve $\alpha_0(\xi_0)$,
closer to ergodic averaging
than to classical Toeplitz theory.
\end{remark}

\begin{lemma}[Singularity of curvature
  weight]
\label{lem:g-singularity}
Let $g(\alpha):=1/|\theta''(\xi_0(\alpha))|$
(Jacobian of $\xi_0\mapsto\alpha_0(\xi_0)$),
$\alpha_*:=\alpha_0(0)=\theta'(0)$.

\begin{enumerate}[\rm(i)]
\item $g\in C^\infty$ away from $\alpha_*$.
\item Near $\alpha_*$: since
  $\alpha_0'(0)=\theta''(0)=0$ and
  $\alpha_0''(0)=\theta'''(0)\ne 0$,
  one has
  $\alpha-\alpha_*\sim\theta'''(0)\xi_0^2/(2c)$,
  hence $|\xi_0|\sim C|\alpha-\alpha_*|^{1/2}$
  and
  \begin{equation}\label{eq:g-sing}
    g(\alpha)
    \sim \frac{C_*}{|\alpha-\alpha_*|^{1/2}},
    \qquad\alpha\to\alpha_*.
  \end{equation}
  Thus $g\in L^1([\alpha_{\min},\alpha_{\max}])$.
\item The Fourier transform satisfies:
  \begin{equation}\label{eq:g-hat-decay}
    |\widehat{g}(\xi)|
    = O(|\xi|^{-1/2})
    \qquad(|\xi|\to\infty),
  \end{equation}
  from the Fourier transform of
  $|\cdot-\alpha_*|^{-1/2}$;
  the smooth remainder is
  $O(|\xi|^{-N})$.
\end{enumerate}
\end{lemma}

\begin{proof}
(ii): Taylor expand $\alpha_0(\xi_0)$
at $\xi_0=0$:
$\alpha_0(\xi_0)-\alpha_0(0)
=\tfrac12\alpha_0''(0)\xi_0^2+O(\xi_0^3)$
(since $\alpha_0'(0)=\theta''(0)=0$).
Thus $|\xi_0|\sim C|\alpha-\alpha_*|^{1/2}$
and $g(\alpha)=c/|\xi_0(\alpha)|
\sim C_*/|\alpha-\alpha_*|^{1/2}$.
(iii): $\mathcal{F}[|\cdot|^{-1/2}](\xi)
= C|\xi|^{-1/2}$;
subtract local singularity and
apply Riemann--Lebesgue to remainder.
\end{proof}

\begin{proposition}[Alpha-oscillation
  strategy]
\label{prop:alpha-oscillation}
Let $K:=\psi*\widetilde{\psi}\in L^1$.
Under the coarea substitution
$\xi_0\mapsto\alpha=\alpha_0(\xi_0)$:
\begin{equation}\label{eq:alpha-integral}
  \int_{-c_0}^{c_0}
  f_c(\alpha_0(\xi_0))\,d\xi_0
  = \int_{\mathbb{R}} K(x)
    \sum_{k=1}^{K(c)}
    \widehat{g}\!\left(\frac{xk}{c^{\theta_0}}\right)
    dx.
\end{equation}

\noindent\textbf{Sufficient condition.}
\begin{equation}\label{eq:RL-condition}
  \frac{1}{K(c)}
  \sum_{k=1}^{K(c)}
  \int|K(x)|\cdot
  |\widehat{g}(xk/c^{\theta_0})|
  \,dx = o(1)
\end{equation}
implies (T\textsuperscript{*}).
\end{proposition}

\begin{lemma}[Weighted Riemann--Lebesgue]
\label{lem:weighted-RL}
The weighted Riemann--Lebesgue condition
\begin{equation}\label{eq:wRL}
  \Psi(\lambda)
  := \int_{\mathbb{R}} K(x)\,
  \widehat{g}(\lambda x)\,dx
  \longrightarrow 0
  \quad(\lambda\to\infty)
\end{equation}
suffices for~\eqref{eq:RL-condition}
and hence for (T\textsuperscript{*}).

\begin{enumerate}[\rm(a)]
\item \emph{(Qualitative, proved now).}
  $g\in L^1$ implies
  $\widehat{g}\in C_0$
  (Riemann--Lebesgue lemma).
  For each $x\ne 0$,
  $\widehat{g}(\lambda x)\to 0$
  as $\lambda\to\infty$.
  Dominated convergence
  with $|K(x)|\norm{\widehat{g}}_{L^\infty}
  \in L^1$ gives
  $\Psi(\lambda)\to 0$.
  \emph{This argument is complete.}

\item \emph{(Quantitative rate).}
  By Lemma~\ref{lem:g-singularity}(iii),
  $|\widehat{g}(\xi)|=O(|\xi|^{-1/2})$.
  For $K\in\mathcal{S}$:
  \begin{equation}\label{eq:rate}
    |\Psi(\lambda)|
    \lesssim \lambda^{-1/2}
    \int|K(x)||x|^{-1/2}\,dx
    = O(\lambda^{-1/2}).
  \end{equation}
  With $\lambda=k/c^{\theta_0}$,
  $k\sim c^{\theta_0}$:
  rate $O(c^{-\theta_0/2})$.

\item \emph{(Symmetry cancellation check).}
  If additionally $\widehat{K}(0)
  =|\widehat\psi(0)|^2=0$
  (zero-mean $\psi$), then even
  logarithmic growth in $\widehat{g}$
  (as would occur for a pure
  $|\cdot|^{-1}$ singularity)
  is automatically absorbed:
  $\int K(x)\log|\lambda x|\,dx
  =\log\lambda\cdot\widehat{K}(0)
  +\int K(x)\log|x|\,dx
  =0+O(1)\cdot o(1/\log\lambda)$.
\end{enumerate}
\end{lemma}

\begin{proof}
(a): Standard: $g\in L^1\Rightarrow
\widehat{g}\in C_0$; DCT applies
since $|K|\cdot\norm{\widehat{g}}_{L^\infty}\in L^1$.
(b): Write $|\widehat{g}(\lambda x)|
\le C(1+|\lambda x|)^{-1/2}$;
split at $|x|=\lambda^{-1}$:
both pieces give $O(\lambda^{-1/2})$
for $K\in\mathcal{S}$.
\end{proof}

\begin{theorem}[(T\textsuperscript{*})
  is proved]
\label{thm:Tstar}
Let $\psi\in L^2$ with
$K=\psi*\widetilde{\psi}\in L^1$,
and let $\theta$ be the PSWF WKB
phase \textup{(}Lemma~\ref{lem:WKB-phase}\textup{)}.
Then:
\begin{equation}\label{eq:Tstar-proved}
  \Sigma_{\mathrm{model}}(c)
  = o(1)
  \qquad(c\to\infty).
\end{equation}
With $K\in\mathcal{S}$ (e.g.,
$\psi\in\mathcal{S}$), the
quantitative rate is
\begin{equation}\label{eq:Tstar-rate}
  \Sigma_{\mathrm{model}}(c)
  = O(c^{-\theta_0/2}).
\end{equation}
\end{theorem}

\begin{proof}
$g\in L^1$
(Lemma~\ref{lem:g-singularity}(ii)),
$K\in L^1$:
Lemma~\ref{lem:weighted-RL}(a) gives
$\Psi(\lambda)\to 0$.
By Proposition~\ref{prop:alpha-oscillation},
\eqref{eq:RL-condition} holds,
hence (T\textsuperscript{*}):
$\int_{-c_0}^{c_0}\Tcc(\psi)\,d\xi_0
\cdot c^{2(1-\theta_0)} = o(1)$.
By Proposition~\ref{prop:model-limit}:
$\Sigma_{\mathrm{model}}=o(1)$.
Quantitative: Lemma~\ref{lem:weighted-RL}(b)
+ Proposition~\ref{prop:model-limit}.
\end{proof}

\begin{remark}[Moving-symbol Toeplitz
  structure]
\label{rem:moving-symbol}
\begin{center}
\emph{Moving-symbol Toeplitz problem:
symbol $\alpha_0(\xi_0)=\theta'(\xi_0)$
varies with basepoint.}
\end{center}
The three limits
$k\to\infty$, $c\to\infty$,
$\xi_0$-integration
do not commute;
the problem is closer to
\emph{ergodic/randomised sampling}
of $|\widehat\psi|^2$
than to standard Szeg\H{o} theory
\cite{Simon2011,BottcherSilbermann1990}.
\end{remark}

\begin{proposition}[Model reduction limit]
\label{prop:model-limit}
\begin{equation}\label{eq:model-limit}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}
    \Tcc(\psi(\cdot;\xi_0))\,d\xi_0
    + O(c^{3\theta_0-2}).
\end{equation}
The remainder $O(c^{3\theta_0-2})=o(1)$
for $\theta_0<2/3$.
\end{proposition}

\begin{definition}[(T\textsuperscript{*})]
\label{def:Tstar}
\begin{equation}\label{eq:Tstar}
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \Tcc(\psi(\cdot;\xi_0))\,d\xi_0
  = o(1),
\end{equation}
or equivalently
$\Sigma_{\mathrm{model}}(c)=o(1)$.
\emph{Status: proved}
(Theorem~\ref{thm:Tstar}).
\end{definition}

%%═══ 7. OPEN PROBLEMS FOR PAPER XVI
\section{Open Problems for Paper~XVI:
  Rigorous Reduction}
\label{sec:open}

\begin{problem}[Rigorous model reduction;
  primary open target]
\label{prob:rigorous-reduction}
\emph{This is the main remaining
obstacle for H1.}

\medskip
\noindent\textbf{Goal:}
\begin{equation}\label{eq:model-approx}
  \Sigma_{\mathrm{int}}(c)
  = \Sigma_{\mathrm{model}}(c)
  + o(1).
\end{equation}

\medskip
\noindent\textbf{Priority~1:
  Kernel error control.}
The true kernel is
$(\AcM)_{nm}\approx I_c(|n-m|)$
with error $O(|n-m|/c)$.
Required: show that the
\emph{HS norm of the error matrix}
is $o(1)$, uniformly in $c$:
\begin{equation}\label{eq:kernel-err}
  \sum_{\substack{n\ne m\\
    |n-m|\le c^{\theta_0}}}
  \left|(\AcM)_{nm} - I_c(|n-m|)\right|^2
  = o(1).
\end{equation}
The difficulty: there are
$O(c^{1+\theta_0})$ off-diagonal pairs;
each contributes $O((|n-m|/c)^2)$,
but the sum is
$O(c^{3\theta_0-1})=o(1)$
\emph{only if} $\theta_0<1/3$.
For $\theta_0\ge 1/3$, a more
refined estimate is required,
exploiting the oscillatory
structure of the kernel.

\medskip
\noindent\textbf{Priority~2:
  Exchange of sum and integral.}
The Toeplitz energy
$\Tcc(\psi)=\sum_{k}|\widehat\psi(k\alpha_0/c^{\theta_0})|^2$
is only defined for fixed $\xi_0$.
The exchange
\[
  \sum_k\int|\cdots|^2\,d\xi_0
  \leftrightarrow
  \int\sum_k|\cdots|^2\,d\xi_0
\]
requires uniform estimates in
$k\le c^{\theta_0}$ and $\xi_0\in[-c_0,c_0]$.

\medskip
\noindent\textbf{Priority~3:
  PSWF Lipschitz bound~(L).}
\begin{equation}\label{eq:L-goal}
  \norm{\widehat{\Phi_{n+k}^{(c)}}
  -\widehat{\Phi_n^{(c)}}}_{L^2([-c_0,c_0])}
  = O(k/c),
  \quad |k|\le c^{\theta_0}.
\end{equation}
Approaches:
\begin{enumerate}[\rm(i)]
\item WKB for PSWFs: semiclassical
  stability of amplitudes
  \cite{Zworski2012}.
\item Eigenvalue stability:
  $|\chi_{n+k}(c)-\chi_n(c)|
  =O(k)$ from the
  Sturm comparison theorem.
\item Microlocal: control of
  $\partial_n\widehat{\Phi_n^{(c)}}$
  via symbolic calculus.
\end{enumerate}
\end{problem}

\begin{remark}[Why kernel error
  is critical]
\label{rem:kernel-critical}
The naive bound gives
$\sum_{k\le c^{\theta_0}}(k/c)^2\cdot c
= O(c^{3\theta_0-1})$.
This is $o(1)$ for $\theta_0<1/3$,
but our regime requires
$\theta_0\in(0,1)$.
For $1/3\le\theta_0<1$,
cancellation in the kernel error
must be exploited;
this is the same type of difficulty
as the original
$\Sigma_{\mathrm{int}}$ problem,
and cannot be resolved by
pointwise estimates alone.
\end{remark}

\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Assume (L) and (C).
Then:
\begin{equation}\label{eq:reduction-chain}
  \underbrace{
    \mathrm{(T^*)}\text{ (proved)}
    + \mathrm{(L)} + \text{reduction}
  }_{\text{Paper XVI}}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}} = o(1)
  \;\underset{\mathrm{(C)}}{\Longrightarrow}\;
  \text{H3 fails}
  \;\Rightarrow\;
  \text{H1 fails}.
\end{equation}
\begin{enumerate}[\rm(i)]
\item (T\textsuperscript{*}) is proved:
  $\Sigma_{\mathrm{model}}=o(1)$
  with rate $O(c^{-\theta_0/2})$
  (Theorem~\ref{thm:Tstar}).
\item The transfer
  $\Sigma_{\mathrm{model}}=o(1)
  \Rightarrow\Sigma_{\mathrm{int}}=o(1)$
  requires the rigorous reduction
  of Problem~\ref{prob:rigorous-reduction}
  plus~(L).
\item The transfer
  $\Sigma_{\mathrm{int}}=o(1)
  \Rightarrow\text{H1 fails}$
  is Paper~XIV's chain under~(C).
\end{enumerate}
\end{theorem}

\begin{proof}
(T\textsuperscript{*}): Theorem~\ref{thm:Tstar}.
Under rigorous reduction
(Problem~\ref{prob:rigorous-reduction})
and~(L):
$\Sigma_{\mathrm{int}}
=\Sigma_{\mathrm{model}}+o(1)
=o(1)$.
Remainder: Paper~XIV chain under~(C).
\end{proof}

\begin{remark}[Conceptual summary
  of the programme]
\label{rem:concept-summary}
This paper establishes that
$\Sigma_{\mathrm{model}}=o(1)$
by identifying the correct structure:
\emph{randomised sampling}
of $|\widehat\psi|^2$
along the strictly monotone curve
$\alpha_0(\xi_0)=\theta'(\xi_0)$,
whose Jacobian $g=1/|\theta''|$
has an integrable
$|\alpha-\alpha_*|^{-1/2}$-singularity
that forces $\widehat{g}\in C_0$.
The classical tools (vdC, Weyl,
exponent pairs) are insufficient
because they operate pointwise;
the cancellation is global and
arises from the geometry of the
curve $\alpha_0(\xi_0)$,
not from phase oscillation.
This insight ---
\emph{obstruction $=$ frozen-phase
saturation, escape $=$ global
curvature averaging} ---
is the publishable core of
this paper.
\end{remark}
