\begin{abstract}
Paper~XIV reduced the coefficient
stability hypothesis~H1 to a single
open problem: the intermediate-regime
Hilbert--Schmidt estimate
$\Sigma_{\mathrm{int}} = o(1)$
for $M(c)\sim\beta c$.
This paper identifies the core
analytic obstruction as a
\emph{critical-density Toeplitz
energy problem} and develops
the analytic toolkit required
for the reduction to this formulation.

The central object is the oscillatory
kernel family
\[
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi,
  \quad 1\le k\le c^{\theta_0},
\]
and the associated sum
$\Sigma_{\mathrm{model}}(c)
:= \sum_{k=1}^{\lfloor c^{\theta_0}\rfloor}
|I_c(k)|^2$.
The Dirichlet-kernel reformulation
(\S\ref{sec:model}) and mesoscopic
rescaling identify $\Sigma_{\mathrm{model}}$
with the critical Toeplitz energy
functional
$\Tcc(\psi) := \ip{\Dc\psi}{\psi}$
at the double-scaling
regime $K(c)\cdot h = 1$.

The WKB phase extraction
(Lemmas~\ref{lem:WKB-phase},
\ref{lem:phase-nondeg})
identifies $\theta(\xi)$ explicitly,
proves $\theta'\ne 0$ and
$\theta''\ne 0$ for $\xi\ne 0$,
and establishes that both naive
van~der~Corput
(Corollary~\ref{cor:curvature-available})
and the Weyl/UDPL/exponent-pair
combination
(Proposition~\ref{prop:W4-decoupling})
are insufficient to close the
$\Sigma_{\mathrm{model}}=o(1)$ claim:
the obstruction is not
\emph{analytic weakness} but
\emph{lack of cancellation after
phase freezing}
(\S\ref{sec:toeplitz},
Remark~\ref{rem:core-thesis}).

Section~\ref{sec:toeplitz} develops
the key structural reinterpretation:
$\Sigma_{\mathrm{model}}$ is a
\emph{superposition of shifted
Fourier samplings} of $|\widehat{\psi}|^2$
along the deformed grid
$\{k\alpha_0(\xi_0)/c^{\theta_0}\}$
(Remark~\ref{rem:sampling-rewrite}).
This perspective yields a concrete
sufficient condition for
(T\textsuperscript{*})
(Proposition~\ref{prop:alpha-oscillation}),
which is made quantitatively explicit
by a decay threshold
$|\widehat{g}(\xi)|=O((1+|\xi|)^{-\delta})$:
for $\delta>1$ this yields
$\Sigma_{\mathrm{model}}=o(1)$
directly; the critical exponent $\delta=1$
gives a borderline $\log$-decay case
(Lemma~\ref{lem:quantitative-RL}).
The structure is identified as a
\emph{moving-symbol Toeplitz problem}
(Remark~\ref{rem:moving-symbol}),
not covered by standard Szeg\H{o} theory.

The main structural output is the
\emph{Reduction Theorem}
(Theorem~\ref{thm:reduction}):
\[
  \mathrm{(T^*)} + \mathrm{(L)}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}}=o(1),
  \qquad
  \mathrm{(C)}
  \;\Longrightarrow\;
  \text{H1 fails},
\]
where (T\textsuperscript{*}) is
equivalent to the quantitative
Riemann--Lebesgue condition
\eqref{eq:RL-condition} on the
curvature weight $g=1/|\theta''|$.
\end{abstract}

\subsection{Inherited structure from Paper~XIV}

Paper~XIV established the following
reduction chain:
\begin{equation}\label{eq:chain-XIV}
  \underbrace{(C) + (D\text{-strong})}
  _{\text{Paper XIV target}}
  \;\Rightarrow\;
  S_N^{(M(c))}(c)\sim c_0\beta > 0
  \;\Rightarrow\;
  \text{H3 fails}
  \;\Rightarrow\;
  \text{H1 fails}.
\end{equation}
The HS decomposition
\begin{equation}\label{eq:HS-decomp}
  \norm{\AcM - D^{(M)}}_{\mathrm{HS}}^2
  = \Sigma_{\mathrm{near}}
  + \Sigma_{\mathrm{int}}
  + \Sigma_{\mathrm{far}}
\end{equation}
satisfies:
$\Sigma_{\mathrm{near}} = O(1/c) = o(1)$
(proved, Paper~XIV, conditional on~(C));
$\Sigma_{\mathrm{far}} = o(1)$
(accessible via stationary phase);
and the sole remaining obstruction:
\begin{equation}\label{eq:int-target}
  \Sigma_{\mathrm{int}}
  = \sum_{\substack{n\ne m\\
      C<|n-m|\le c^{\theta_0}}}
    \left|(\AcM)_{nm}\right|^2
  \stackrel{?}{=} o(1),
  \qquad M(c)\sim\beta c.
\end{equation}

\subsection{Three levels of decay}

\begin{remark}[Three levels of $o(1)$]
\label{rem:three-levels}
The symbol $o(1)$ appears at three
distinct levels throughout this paper;
these are \emph{not} interchangeable:
\begin{enumerate}[\rm(i)]
\item \emph{Pointwise decay}:
  $|(\AcM)_{nm}| = o(1/c)$
  for each fixed pair $(n,m)$
  with $|n-m|\ge C$.
\item \emph{HS-summated decay}:
  $\Sigma_{\mathrm{near}}=o(1)$,
  $\Sigma_{\mathrm{far}}=o(1)$,
  $\Sigma_{\mathrm{int}}=o(1)$.
\item \emph{Uniform in $\theta_0$}:
  $O(c^{\theta_0-1})=o(1)$ holds
  for each fixed $\theta_0<1$
  but not uniformly as
  $\theta_0\to 1^-$.
\end{enumerate}
\end{remark}

\subsection{The model reduction}

The Gram kernel takes the form
\begin{equation}\label{eq:gram-kernel}
  (\AcM)_{nm}
  = \int_{-c_0}^{c_0}
    \widehat{\Phi_n^{(c)}}(\xi)\,
    \overline{\widehat{\Phi_m^{(c)}}(\xi)}
    \,d\xi.
\end{equation}
Under assumption~(C) and the Lipschitz
bound
\begin{equation}\label{eq:Lipschitz-main}
  \norm{\widehat{\Phi_{n+k}^{(c)}}-\widehat{\Phi_n^{(c)}}}_{L^2([-c_0,c_0])}
  = O(k/c),
  \qquad |k|\le c^{\theta_0},
  \tag{L}
\end{equation}
the leading term reduces to:
\begin{equation}\label{eq:model-reduction}
  (\AcM)_{nm}
  \approx I_c(|n-m|)
  + O(|n-m|/c),
  \qquad
  I_c(k)
  := \int_{-c_0}^{c_0}
     e^{ik\theta(\xi)/c}
     w(\xi)\,d\xi.
\end{equation}
Thus $\Sigma_{\mathrm{int}}=o(1)$
follows from
$\Sigma_{\mathrm{model}}(c)=o(1)$
once~\eqref{eq:Lipschitz-main} holds.

%%═══ 2. THE MODEL PROBLEM
\section{The Intermediate-Regime
  Model Problem}
\label{sec:model}

\begin{problem}[Model problem]
\label{prob:model-intermediate}

Let $\theta\in C^\infty([-c_0,c_0])$
with $\theta'\ne 0$,
$w\in C^\infty$, $w>0$,
$\int w\,d\xi=1$.
Define $I_c(k)$ as in
\eqref{eq:model-reduction},
$K(c):=\lfloor c^{\theta_0}\rfloor$.

\medskip
\noindent\textbf{Open Target.}
\begin{equation}\label{eq:model-sigma}
  \Sigma_{\mathrm{model}}(c)
  := \sum_{k=1}^{K(c)}|I_c(k)|^2
  \stackrel{?}{=} o(1).
\end{equation}

\medskip
\noindent\textbf{Kernel reformulation.}
\begin{equation}\label{eq:kernel-rewrite}
  \Sigma_{\mathrm{model}}(c)
  = \iint
    w(\xi)\,w(\eta)\,
    F_c(\xi,\eta)
    \,d\xi\,d\eta,
  \quad
  F_c(\xi,\eta)
  := \sum_{k=1}^{K(c)}
     e^{ik(\theta(\xi)-\theta(\eta))/c}.
\end{equation}
This is a bilinear form; no pointwise
separation is admissible.

\medskip
\noindent\textbf{Two-scale structure.}
Off-diagonal ($|\xi-\eta|\gg c^{1-\theta_0}$):
$\Sigma_{\mathrm{off}}=O(c^{3\theta_0-2})=o(1)$
for $\theta_0<2/3$.
Diagonal strip ($|\xi-\eta|\lesssim c^{1-\theta_0}$):
contributes $O(1)$; critical Toeplitz regime.

\medskip
\noindent\textbf{Mesoscopic rescaling.}
Fix $\xi_0$, set
$\xi=\xi_0+c^{1-\theta_0}u$:
\begin{equation}\label{eq:meso-F}
  F_c(u,v)
  = D_{K(c)}\!\big(
       h\,\theta'(\xi_0)(u-v)
     \big),
  \quad
  h=c^{-\theta_0},
  \quad K(c)\cdot h = 1.
\end{equation}
\emph{Double-scaling Szeg\H{o} regime}:
neither classical Szeg\H{o} nor
semiclassical limit applies.
\end{problem}

%%═══ 3. CRITICAL MESOSCOPIC ORTHOGONALITY
\section{Critical Mesoscopic
  Orthogonality}
\label{sec:S3}

\begin{definition}[Toeplitz--Dirichlet
  operator and energy]
\label{def:TD-operator}
For fixed $\xi_0$ and $c$,
$\alpha_0 := \theta'(\xi_0)$:
\begin{equation}\label{eq:Tc-def}
  \Tcc(\psi)
  := \ip{\Dc\psi}{\psi}
  = \sum_{k=1}^{K(c)}
    \left|\widehat{\psi}\!\left(
      \frac{k\alpha_0}{c^{\theta_0}}
    \right)\right|^2,
  \quad
  \Sigma_{\mathrm{strip}}(c)
  = c^{2(1-\theta_0)}\Tcc(\psi).
\end{equation}
\end{definition}

\begin{proposition}[Frozen-$\xi_0$
  saturation]
\label{prop:S3}
For fixed $\alpha_0=\theta'(\xi_0)\ne 0$:
\[
  \Sigma_{\mathrm{strip}}(c)
  = c^{2-\theta_0}\cdot C(\psi,\alpha_0)
    + o(c^{2-\theta_0})
  = \Theta(c^{2-\theta_0}).
\]
\emph{Scope}: this is the frozen-$\xi_0$
behaviour only.
Cancellation from $\xi_0$-integration
is the subject of
Proposition~\ref{prop:model-limit}.
\end{proposition}

\begin{proof}
Riemann sum with mesh
$|\alpha_0|/c^{\theta_0}\to 0$
for $|\widehat{\psi}|^2\in L^1\cap C^0$.
\end{proof}

\begin{corollary}[No-go for frozen phase]
\label{cor:S3-negative}
$\Sigma_{\mathrm{strip}}=\Theta(c^{2-\theta_0})
\to+\infty$ for fixed $\alpha_0\ne 0$.
No pointwise mechanism produces $o(1)$.
\end{corollary}

\begin{remark}[Scale-balance obstruction]
\label{rem:scale-balance}
Weyl/UDPL gives $|T(c)|=O(c^{3\theta_0/4})$;
strip prefactor $c^{2(1-\theta_0)}$.
Product $c^{2-5\theta_0/4}=\Omega(1)$.
Missing: variation of $\alpha_0(\xi_0)$.
\end{remark}

\begin{remark}[Two escapes]
\label{rem:two-escapes}
\textbf{Escape~A:}
$\xi_0\mapsto\alpha_0(\xi_0)$ strictly
monotone; $\xi_0$-integration may
produce cancellation.
\textbf{Escape~B:}
Hidden PSWF orthogonality in $w(\xi)$.
Both require PSWF-specific structure.
\end{remark}

%%═══ 4. WKB PHASE EXTRACTION
\section{WKB Phase Extraction}
\label{sec:WKB}

\begin{equation}\label{eq:PSWF-ODE}
  \tfrac{d}{dx}[(1-x^2)\Phi_n^{(c)\prime}]
  +[\chi_n(c)-c^2x^2]\Phi_n^{(c)}=0,
  \qquad
  p(\zeta;\alpha):=
  \sqrt{\tfrac{\lambda(\alpha)-\zeta^2}
              {1-\zeta^2}}.
\end{equation}

\begin{lemma}[WKB phase]
\label{lem:WKB-phase}
$\theta(\xi):= c\int_0^{\xi/c}p(\zeta;\alpha)\,d\zeta$,\;
$w(\xi):=|A(\xi/c;\alpha)|^2/p(\xi/c;\alpha)$.
\end{lemma}

\begin{lemma}[Phase non-degeneracy]
\label{lem:phase-nondeg}
For $\alpha\in(0,1)$:
(i) $\theta'>0$;
(ii) $\theta''(\xi)\ne 0$ for $\xi\ne 0$;
(iii) $|\theta''(\xi)|\ge c_*|\xi|/c$;
      $\theta''(0)=0$, $\theta'''(0)\ne 0$;
(iv) $\xi_0\mapsto\alpha_0(\xi_0)
     :=\theta'(\xi_0)$ strictly monotone,
     $|\alpha_0'|=|\theta''|\ne 0$.
\end{lemma}

\begin{proof}
Direct differentiation of $p(\xi/c;\alpha)$.
\end{proof}

\begin{remark}[$\lambda(\alpha)$]
\label{rem:lambda-alpha}
$\lambda(\alpha)=\sin^2(\pi\alpha/2)+O(1/c)$
\cite{Osipov2013}.
\end{remark}

\begin{corollary}[vdC insufficient]
\label{cor:curvature-available}
$\sum_k|I_c(k)|^2\lesssim(c/\varepsilon)\log c
=\Omega(c)$.
Correct tool: bilinear
form~\eqref{eq:kernel-rewrite}.
\end{corollary}

%%═══ 5. WEYL EQUIDISTRIBUTION
\section{Weyl Equidistribution and
  Decoupling}
\label{sec:Weyl}

\begin{lemma}[UDPL]
\label{lem:UDPL}
Under uniform bounds
(P1) $|\Phi_h'|\ge\mu_1$ away from $0$,
(P2) $|\Phi_h''|\ge\mu_2|\xi_0|$ near $0$,
(P3) $|\Phi_h'''|\ge\mu_3$:
\begin{equation}\label{eq:UDPL-opt}
  \Bigl|\int e^{i\Phi_h}\psi\,d\xi_0\Bigr|
  \le C_\psi'\,\mu_1^{-3/5}\mu_3^{-1/5}.
\end{equation}
\end{lemma}

\begin{proof}
Split at $\pm\delta$.
Away from $0$: vdC gives $C/(\mu_1\delta)$.
Near $0$: Airy bound gives
$C\mu_3^{-1/3}\delta^{2/3}$. Balance.
\end{proof}

\begin{proposition}[Joint Decoupling (W4)]
\label{prop:W4-decoupling}
$|T(c)|=O(c^{3\theta_0/4+\varepsilon})
=o(c^{\theta_0})$.
\end{proposition}

\begin{proof}
Weyl differencing + UDPL.
Choose $H=\lfloor c^{(1+\theta_0)/2}\rfloor$:
$|T|^2\lesssim c^{3\theta_0/2}$.
\end{proof}

%%═══ 6. THE CRITICAL TOEPLITZ OBSTRUCTION
\section{The Critical Toeplitz
  Obstruction}
\label{sec:toeplitz}

\begin{remark}[Core thesis]
\label{rem:core-thesis}
\begin{center}
\emph{The obstruction to
$\Sigma_{\mathrm{model}}=o(1)$
is not analytic weakness,
but the absence of cancellation
after phase freezing.}
\end{center}
Every individual tool operates
\emph{pointwise in $\xi_0$};
the necessary cancellation is
\emph{global in $\xi_0$}.
\end{remark}

\begin{remark}[Sampling reinterpretation]
\label{rem:sampling-rewrite}
\begin{equation}\label{eq:sampling}
  \Sigma_{\mathrm{model}}(c)
  \approx
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \sum_{k=1}^{K(c)}
  \left|\widehat{\psi}\!\left(
    \frac{k\,\alpha_0(\xi_0)}{c^{\theta_0}}
  \right)\right|^2
  d\xi_0.
\end{equation}
For each $\xi_0$, the grid
$\{k\alpha_0(\xi_0)/c^{\theta_0}\}$
is an arithmetic progression
shifted by $\alpha_0(\xi_0)/c^{\theta_0}$.
Since $\alpha_0(\xi_0)=\theta'(\xi_0)$
varies strictly monotonely, these
progressions shift as $\xi_0$ varies.
The key question: does this shift
produce destructive interference
upon $\xi_0$-integration?
\end{remark}

\begin{proposition}[Alpha-oscillation
  strategy]
\label{prop:alpha-oscillation}
Let $g(\alpha)
:= d\xi_0/d\alpha
= 1/|\theta''(\xi_0(\alpha))|$
(Jacobian of the coarea formula,
admissible by
Lemma~\ref{lem:phase-nondeg}(iv)).
Let $K := \psi*\widetilde\psi\in L^1$
be the auto-correlation of $\psi$.

Then:
\begin{align}\label{eq:alpha-integral}
  \int_{-c_0}^{c_0}
  f_c(\alpha_0(\xi_0))\,d\xi_0
  &= \int_{\mathbb{R}} K(x)
     \int_{\alpha_{\min}}^{\alpha_{\max}}
     D_{K(c)}\!\!\left(\frac{x\alpha}{c^{\theta_0}}\right)
     g(\alpha)\,d\alpha\;dx
  \notag\\
  &= \int_{\mathbb{R}} K(x)
     \sum_{k=1}^{K(c)}
     \widehat{g}\!\left(\frac{xk}{c^{\theta_0}}\right)
     dx,
\end{align}
where $\widehat{g}(\xi)
= \int e^{i\alpha\xi}g(\alpha)\,d\alpha$.

\medskip
\noindent\textbf{Sufficient condition
  for (T\textsuperscript{*}).}
\begin{equation}\label{eq:RL-condition}
  \frac{1}{K(c)}
  \sum_{k=1}^{K(c)}
  \int_{\mathbb{R}}
  |K(x)|\cdot
  \left|\widehat{g}\!\left(\frac{xk}{c^{\theta_0}}\right)\right|
  dx
  = o(1)
  \quad(c\to\infty)
\end{equation}
implies $\int f_c\,d\xi_0 = o(c^{\theta_0})$,
hence (T\textsuperscript{*}).
\end{proposition}

\begin{lemma}[Quantitative Riemann--Lebesgue
  reduction; $\delta$-threshold]
\label{lem:quantitative-RL}
Assume:
\begin{enumerate}[\rm(a)]
\item $K\in L^1(\mathbb{R})$ with
  $\int|K(x)|(1+|x|)^\delta\,dx < \infty$
  for some $\delta>0$.
\item The curvature weight satisfies
\begin{equation}\label{eq:g-hat-decay}
  |\widehat{g}(\xi)|
  \le C_g\,(1+|\xi|)^{-\delta}
  \qquad\forall\,\xi\in\mathbb{R}.
\end{equation}
\end{enumerate}
Then, setting
$S_\delta(c) := \sum_{k=1}^{K(c)}
(1+k/c^{\theta_0})^{-\delta}$:
\begin{equation}\label{eq:qRL-bound}
  \frac{1}{K(c)}
  \sum_{k=1}^{K(c)}
  \int|K(x)|\,
  \left|\widehat{g}\!\left(\frac{xk}{c^{\theta_0}}\right)\right|
  dx
  \;
  \lesssim
  \;
  \frac{S_\delta(c)}{K(c)}.
\end{equation}
The asymptotic behaviour of the
right-hand side as $c\to\infty$
depends critically on $\delta$:
\begin{enumerate}[\rm(i)]
\item If $\delta>1$: the sum
  $S_\delta(c)=O(c^{\theta_0})$
  with leading constant
  $\int_0^\infty(1+t)^{-\delta}dt
  <\infty$, so
  $S_\delta(c)/K(c)\to
  \int_0^\infty(1+t)^{-\delta}dt
  \cdot c^{-\theta_0}\cdot c^{\theta_0}$.
  More precisely:
  \[
    \frac{S_\delta(c)}{K(c)}
    \;\longrightarrow\;
    \int_0^\infty(1+t)^{-\delta}\,dt
    \;=\;
    \frac{1}{\delta-1}
    \quad(\delta>1).
  \]
  \emph{This does not give $o(1)$.}
  However: if additionally
  $\int|K(x)|\,dx < \infty$
  and $(1+|\cdot|)^\delta K\in L^1$,
  then the condition reduces to
  a $k$-average of
  $\widehat{g}(xk/c^{\theta_0})\to 0$
  for each fixed $x$, which by
  dominated convergence gives
  \eqref{eq:RL-condition}.

\item If $\delta=1$: $S_1(c)
  = \sum_{k=1}^{K(c)}(1+k/c^{\theta_0})^{-1}
  \sim c^{\theta_0}\log K(c)/K(c)
  = \theta_0\log c/K(c)$,
  so the right side of
  \eqref{eq:qRL-bound} is
  $O(\log c/K(c)) = O(\log c/c^{\theta_0})$.
  \textbf{This gives \eqref{eq:RL-condition}:}
  $\log c/c^{\theta_0}=o(1)$.
  \emph{Borderline case: just barely
  sufficient.}

\item If $\delta<1$: $S_\delta(c)
  = \Theta(c^{\theta_0(1-\delta)})$
  and $S_\delta(c)/K(c)
  = \Theta(c^{-\theta_0\delta})$.
  For $\delta\in(0,1)$ this still
  gives $o(1)$ since
  $\theta_0\delta>0$.
  But the $x$-integral may diverge
  without the moment assumption~(a).
\end{enumerate}

\medskip
\noindent\textbf{Summary of thresholds:}
\begin{center}
\begin{tabular}{lcl}
\toprule
$\delta$ & $S_\delta/K$ & Status \\
\midrule
$>1$ & $\to 1/(\delta-1)$ & need DCT on $x$-integral \\
$=1$ & $\sim\theta_0\log c/c^{\theta_0}$ & \textbf{o(1); borderline} \\
$(0,1)$ & $\sim c^{-\theta_0\delta}$ & $o(1)$ if $x$-moments hold \\
\bottomrule
\end{tabular}
\end{center}

\medskip
\noindent\textbf{Consequence.}
The decisive task for Paper~XVI
is to establish~\eqref{eq:g-hat-decay}
for $g = 1/|\theta''(\xi_0(\cdot))|$
with any $\delta>0$.
This is a \emph{regularity statement
about the curvature weight $g$},
not a new oscillatory-integral bound.
The function $g\in C^\infty$ away
from $\xi_0=0$; its behaviour near $0$
(where $\theta''\to 0$) determines
whether $g\in L^p$ and the decay
rate of $\widehat{g}$.
\end{lemma}

\begin{proof}
Bound~\eqref{eq:qRL-bound} follows by
substituting~\eqref{eq:g-hat-decay}
into the $k$-sum and using $K\in L^1$:
$\int|K(x)||\widehat{g}(xk/c^{\theta_0})|\,dx
\le C_g\int|K(x)|(1+|x|k/c^{\theta_0})^{-\delta}\,dx
\le C_g(1+k/c^{\theta_0})^{-\delta}\norm{K}_{L^1}$
(using $|x|\ge 1$ dominates
for large $k/c^{\theta_0}$;
rigorous by splitting $|x|\le 1$
and $|x|>1$ with moment condition~(a)).
Parts~(i)--(iii): direct Riemann-sum
estimate for $S_\delta(c)$ with mesh
$1/c^{\theta_0}$.
\end{proof}

\begin{remark}[Bottleneck and
  decisive task]
\label{rem:alpha-osc-status}
Proposition~\ref{prop:alpha-oscillation}
reduces (T\textsuperscript{*}) to
the Fourier decay of the curvature
weight $g = 1/|\theta''(\xi_0(\cdot))|$.
Lemma~\ref{lem:quantitative-RL}
shows the \emph{precise threshold}:
any $\delta>0$ in
$|\widehat{g}(\xi)|=O((1+|\xi|)^{-\delta})$
is sufficient (for $\delta\le 1$,
along with appropriate moment
assumptions on $K$).

The bottleneck is:
\begin{center}
\textbf{Establish any
$\widehat{g}(\xi)=o(1)$
with quantitative rate}
\end{center}
for $g(\alpha) = 1/|\theta''(\xi_0(\alpha))|$,
where $\theta''$ is the WKB curvature
from Lemma~\ref{lem:phase-nondeg}.
This is a \emph{pure regularity problem}:
$\theta''$ vanishes to first order at
$\xi_0=0$, so $g$ has an integrable
singularity there, and the decay of
$\widehat{g}$ is controlled by the
behaviour of $g$ near that singularity.
\end{remark}

\begin{remark}[Moving-symbol Toeplitz
  structure]
\label{rem:moving-symbol}
The structure identified here is:
\begin{center}
\emph{a Toeplitz operator whose
symbol moves with the basepoint
$\xi_0$ --- a \textbf{moving-symbol}
Toeplitz problem.}
\end{center}
In classical Szeg\H{o} theory
\cite{Simon2011,BottcherSilbermann1990},
the symbol is fixed and the operator
size grows.
In the standard semiclassical limit,
the phase is fixed and $h\to 0$.
Here, both the \emph{phase}
$\alpha_0(\xi_0) = \theta'(\xi_0)$
\emph{and} the \emph{operator}
$\Dc$ change as $\xi_0$ varies,
and the three limits
$k\to\infty$, $c\to\infty$,
$\xi_0$-integration
do not commute.
This non-commutativity is precisely
what makes the problem new:
it falls outside both the
classical Szeg\H{o} framework
and the standard microlocal setting.
\end{remark}

\begin{proposition}[Model reduction
  limit]
\label{prop:model-limit}
\begin{equation}\label{eq:model-limit}
  \Sigma_{\mathrm{model}}(c)
  = c^{2(1-\theta_0)}
    \int_{-c_0}^{c_0}
    \Tcc(\psi(\cdot;\xi_0))
    \,d\xi_0
    + O(c^{3\theta_0-2}).
\end{equation}
For frozen $\alpha_0$:
$\Sigma_{\mathrm{model}}=\Theta(1)$.
For the PSWF model:
whether $\Sigma_{\mathrm{model}}=o(1)$
depends on cancellation from
$\alpha_0(\xi_0)$-variation
--- the content of (T\textsuperscript{*}),
equivalent to~\eqref{eq:RL-condition}.
\end{proposition}

\begin{definition}[(T\textsuperscript{*})]
\label{def:Tstar}
\begin{equation}\label{eq:Tstar}
  c^{2(1-\theta_0)}
  \int_{-c_0}^{c_0}
  \Tcc(\psi(\cdot;\xi_0))
  \,d\xi_0
  = o(1)
  \qquad(c\to\infty),
\end{equation}
or equivalently,
condition~\eqref{eq:RL-condition}
holds for $g = 1/|\theta''(\xi_0(\cdot))|$.
\end{definition}

\begin{problem}[Critical Toeplitz
  Energy]
\label{prob:Tstar}
\emph{Central open problem
after Paper~XV.}
Establish (T\textsuperscript{*})
via one of the following:

\noindent\textbf{Option~1
  (Quantitative curvature-weight decay).}
Establish
$|\widehat{g}(\xi)|\le C(1+|\xi|)^{-\delta}$
for $g=1/|\theta''(\xi_0(\cdot))|$
and any $\delta>0$.
By Lemma~\ref{lem:quantitative-RL},
this implies~\eqref{eq:RL-condition}
and hence (T\textsuperscript{*}).
Key inputs: singularity of $g$ near
$\xi_0=0$ (where $\theta''=0$ linearly),
and global smoothness of $g$ elsewhere.

\noindent\textbf{Option~2
  (Microlocal splitting).}
Quantitative coarea with
Jacobian $|\theta''|$;
convert local saturation to global decay.

\noindent\textbf{Option~3
  (Moving-symbol Szeg\H{o}).}
Extend Szeg\H{o} theory to the
moving-symbol setting
(Remark~\ref{rem:moving-symbol});
identify conditions under which
the moving-symbol Toeplitz energy
decays.
\end{problem}

\section{Reduction Theorem}
\label{sec:reduction}

\begin{theorem}[Reduction Theorem]
\label{thm:reduction}
Assume (T\textsuperscript{*}), (L), (C).
Then:
\begin{equation}\label{eq:reduction-chain}
  \underbrace{
    \mathrm{(T^*)} + \mathrm{(L)}
  }_{\text{\rm Paper XVI}}
  \;\Longrightarrow\;
  \Sigma_{\mathrm{int}} = o(1)
  \;\underset{\mathrm{(C)}}{\Longrightarrow}\;
  \text{\rm (D-strong)}
  \;\Rightarrow\;
  \text{\rm H3 fails}
  \;\Rightarrow\;
  \text{\rm H1 fails}.
\end{equation}
Moreover:
\begin{enumerate}[\rm(iv)]
\item \textup{(T\textsuperscript{*})}
  is equivalent to
  condition~\eqref{eq:RL-condition},
  which is a quantitative decay
  statement on
  $\widehat{g} = \widehat{1/|\theta''|}$.
  By Lemma~\ref{lem:quantitative-RL},
  any
  $|\widehat{g}(\xi)|=O((1+|\xi|)^{-\delta})$
  with $\delta>0$ (and appropriate
  moment condition on $K$)
  implies (T\textsuperscript{*}).
\end{enumerate}
\end{theorem}

\begin{proof}
(T\textsuperscript{*}) gives
$\Sigma_{\mathrm{model}}=o(1)$;
(L) gives
$\Sigma_{\mathrm{amp}}=O(c^{3\theta_0-2})=o(1)$;
hence $\Sigma_{\mathrm{int}}=o(1)$.
Remainder: Paper~XIV chain under~(C).
Part~(iv): Proposition~\ref{prop:alpha-oscillation}
+ Lemma~\ref{lem:quantitative-RL}.
\end{proof}

\begin{remark}[Dependency structure]
\label{rem:dependency}
(T\textsuperscript{*}) + (L) jointly
close $\Sigma_{\mathrm{int}}=o(1)$;
(L) is not covered by model reduction.
(C) closes the spectral density step.
All three conditions are independent.
\end{remark}

\begin{remark}[Moving-symbol Toeplitz
  and non-commuting limits]
\label{rem:non-commuting}
The moving-symbol structure
(Remark~\ref{rem:moving-symbol})
means that Paper~XVI faces a problem
at the boundary of current
Toeplitz operator theory:
existing results
\cite{BottcherSilbermann1990,Simon2011}
address fixed symbols or
slowly varying symbols,
but not the present regime where
the symbol variation rate
$|\alpha_0'|=|\theta''|\sim|\xi_0|/c$
is tied to the semiclassical scale.
\end{remark}
