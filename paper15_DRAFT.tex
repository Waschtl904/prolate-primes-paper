\title[Intermediate-Regime Kernel Estimates]{%
  Intermediate-Regime Kernel Estimates
  for the PSWF Compression Operator:\\
  Mesoscopic Scaling, WKB Phase Extraction,
  and Reduction to the Fold Amplitude}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
This paper is a \emph{reduction paper}.
Its contribution is not a bound
on $\Sigma_{\mathrm{int}}$
beyond what Ces\`aro gives,
but the identification and
isolation of the single analytic
object that controls the gap:
the \emph{fold amplitude} $\Phi$
defined in~\eqref{eq:Phi-def}.

Four results are proved unconditionally:
\begin{enumerate}[\rm(1)]
\item \emph{Exact equivalence}
  (Lem.~\ref{lem:S-equivalence}):
  $\Sigma_{\mathrm{model}}=o(1)
  \iff S(c)=o(c^{-(2-2\theta_0)})$;
  the gap equals the
  frozen-phase saturation rate
  $c^{2-\theta_0}$.
\item \emph{Ces\`aro step}
  (Cor.~\ref{cor:cesaro-closed}):
  $S(c)=o(c^{\theta_0})$
  via Fubini--Tonelli;
  the gap is not closed.
\item \emph{Dirichlet barrier}
  (Lem.~\ref{lem:dirichlet-obstruction}):
  in the model where
  $\widehat{\gsing}$ is
  replaced by its leading-order
  Dirichlet structure,
  signed cancellation yields
  only $\Theta(K^{1/2})$,
  no improvement over Ces\`aro.
  This rules out
  Dirichlet-kernel methods
  in this model,
  but is not a universal no-go.
\item \emph{1D reduction}
  (Prop.~\ref{prop:I0-1D}):
  after Poisson summation
  (Lem.~\ref{lem:poisson-bridge}),
  the non-trivial part of $S(c)$
  is controlled by
  $I_0(x,c)=\frac{c^{\theta_0}}{|x|}
  \int_0^{|x|}\Phi(u)\,du$,
  with $\Phi$ the Fourier transform
  of the fold amplitude
  $|\beta|^{-1/2}\chi$.
  The remaining question is
  whether
  $\int_0^U\Phi(u)\,du=o(U^{1/2})$.
\end{enumerate}

\noindent
All further progress requires
establishing a \emph{uniform
cubic phase expansion}
for $\Phi$ near $u=0$
(Assumption~\ref{ass:Phi-cubic},
Stated but not proved here;
this is the content of Paper~XVI).
\end{abstract}

\section{Introduction}

\subsection{Scope of this paper}

Paper~XIV established the
logical chain
$(C)+(D\text{-strong})
\Rightarrow\text{H1 fails}$
and reduced the problem to:
\begin{equation}\label{eq:int-target}
  \Sigma_{\mathrm{int}}
  \stackrel{?}{=} o(1).
\end{equation}
Paper~XV does \emph{not} prove
$\Sigma_{\mathrm{int}}=o(1)$.
It reduces the problem to a
single 1D analytic question
about the fold amplitude $\Phi$,
states that question precisely,
and proves four unconditional
results along the way.

\section{Model Problem and
  Frozen-Phase Saturation}
\label{sec:model}

\begin{equation}\label{eq:model-def}
  I_c(k):=\int e^{ik\theta/c}
  w\,d\xi,
  \qquad
  \Sigma_{\mathrm{model}}
  :=\textstyle\sum_{k=1}^{K(c)}
  |I_c(k)|^2.
\end{equation}
Under assumptions (L), (R), (C):
$\Sigma_{\mathrm{int}}
=\Sigma_{\mathrm{model}}+o(1)$.

\begin{corollary}[No pointwise
  mechanism \Rig]
\label{cor:S3-negative}
$\Sigma_{\mathrm{strip}}
=\Theta(c^{2-\theta_0})\to+\infty$;
no pointwise (fixed-$k$) bound
can give $o(1)$.
\end{corollary}

\section{WKB and the Fold}
\label{sec:WKB}

\begin{lemma}[WKB phase and
  fold condition \Rig]
\label{lem:WKB-phase}
Let $\theta:=c\int_0^{\xi/c}p\,d\zeta$
be the semiclassical phase,
$g:=1/|\theta''(\xi_0(\cdot))|$.
Then:
$g\sim C_*|\alpha-\alpha_*|^{-1/2}$
near $\alpha_*=\theta'(0)$;
$|\widehat{g}|=O(|\xi|^{-1/2})$.
Fold condition at $\xi=0$:
\begin{equation}\label{eq:fold}
  \theta''(0)=0,
  \quad
  \theta'''(0)\ne 0,
  \quad
  c_3:=\tfrac{1}{6}\theta'''(0)\ne 0.
\end{equation}
\end{lemma}

\section{Four Unconditional Results}
\label{sec:main}

\begin{lemma}[Exact scaling
  equivalence \Rig]
\label{lem:S-equivalence}
With $S(c):=\int K(x)
\sum_{k=1}^{K(c)}
\widehat{g}(xk/c^{\theta_0})dx$,
$K(c)=c^{\theta_0}$:
\begin{equation}\label{eq:exact-equiv}
  \boxed{\Sigma_{\mathrm{model}}=o(1)
  \;\iff\;
  S(c)=o(c^{-(2-2\theta_0)}).}
\end{equation}
Gap $=c^{2-\theta_0}$
$=$ frozen-phase saturation rate.
\end{lemma}

\begin{lemma}[Splitting \Rig]
\label{lem:g-split}
$g=\gsing+\greg$;
$\gsing:=C_*|\alpha-\alpha_*|^{-1/2}\chi$
($\chi\in C^\infty_c$,
$\chi\equiv 1$ near $\alpha_*$);
$|\widehat{\gsing}|=O(|\xi|^{-1/2})$;
$|\widehat{\greg}|=O(|\xi|^{-N})$.
\end{lemma}

\begin{corollary}[Ces\`aro step \Rig]
\label{cor:cesaro-closed}
Fubini--Tonelli with
$\Psi(\lambda):=\int K(x)
\widehat{g}(\lambda x)dx$:
$S(c)=c^{\theta_0}\mathcal{C}(c)$
with $\mathcal{C}(c)\to 0$.
This gives $S(c)=o(c^{\theta_0})$;
the gap $c^{2-\theta_0}$ is
\emph{not} closed.
\end{corollary}

\begin{lemma}[Dirichlet barrier \Rig,
  model-specific]
\label{lem:dirichlet-obstruction}
In the model where the
$k$-sum is treated via
$D^*_K(t):=\sum_{k=1}^K e^{ikt}$,
the signed integral:
\begin{equation}\label{eq:JK}
  J_K(x):=\int|\beta|^{-1/2}
  D^*_K(\beta x/c^{\theta_0})d\beta
  = Ce^{i\pi/4}
  \frac{c^{\theta_0/2}}{|x|^{1/2}}
  \sum_{k=1}^K k^{-1/2}
  = \Theta\!\left(
  \frac{c^{\theta_0}}{|x|^{1/2}}\right).
\end{equation}
Phase $e^{i\pi/4}$
is $k$-independent;
Dirichlet modes add coherently.
Consequence: Dirichlet-kernel
methods cannot improve over
Ces\`aro \emph{in this model}.
\end{lemma}

\begin{remark}[Scope of the
  barrier \Rig]
\label{rem:barrier-scope}
Lemma~\ref{lem:dirichlet-obstruction}
is a \emph{model barrier},
not a universal no-go theorem.
It rules out methods that
rely on signed cancellation
within $D^*_K$ alone.
It does not rule out:
\begin{itemize}[nosep]
\item joint $(\beta,k)$ oscillation
  (which is what the
  Poisson--Airy route exploits);
\item other reformulations of
  the original operator problem
  that may have additional
  cancellation not present
  in this model.
\end{itemize}
\end{remark}

\begin{lemma}[Poisson bridge \Rig]
\label{lem:poisson-bridge}
Poisson summation
applied to
$k\mapsto\widehat{\gsing}(xk/c^{\theta_0})$
(smooth in $k$ for $x\ne 0$):
\begin{equation}\label{eq:poisson}
  \sum_{k=1}^{K(c)}
  \widehat{\gsing}(xk/c^{\theta_0})
  = \sum_{m\in\mathbb{Z}} I_m(x,c),
  \quad
  I_m:=\int_1^{K(c)}
  \widehat{\gsing}(xt/c^{\theta_0})
  e^{2\pi imt}\,dt.
\end{equation}
For $m\ne 0$, integration by
parts in $t$ gives
$|I_m(x,c)|
\le C_N(x)/(|m|\cdot 1)$
\emph{for each fixed $(x,c)$},
but \textbf{not uniformly in $c$}
as $K(c)=c^{\theta_0}\to\infty$.
The control of the tail
$\sum_{m\ne 0}I_m$
uniformly in $c$ is an open
sub-problem (see
Problem~\ref{prob:Im-tail}).
\end{lemma}

\begin{proposition}[1D reduction
  of $I_0$ \Rig]
\label{prop:I0-1D}
Define the fold amplitude:
\begin{equation}\label{eq:Phi-def}
  \Phi(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \quad u\in\mathbb{R}.
\end{equation}
Then $|\Phi(u)|=O(|u|^{-1/2})$
and:
\begin{equation}\label{eq:I0-1D}
  I_0(x,c)=
  \frac{c^{\theta_0}}{|x|}
  \int_0^{|x|}
  \Phi(u)\,du
  + O(c^{-\infty}).
\end{equation}
From $|\Phi|=O(u^{-1/2})$:
$|I_0(x,c)|\le C c^{\theta_0/2}/|x|^{1/2}$
(Ces\`aro bound, no cancellation).
Any improvement over this
is equivalent to:
\begin{equation}\label{eq:Phi-question}
  \left|\int_0^U\Phi(u)\,du\right|
  = o(U^{1/2})\quad(U\to\infty).
\end{equation}
\end{proposition}

\section{Open Problems for
  Paper~XVI}
\label{sec:open}

\begin{remark}[The single hard
  question]
\label{rem:single-question}
All four unconditional results
reduce the problem to a
single 1D question:
does~\eqref{eq:Phi-question} hold?
The answer depends entirely
on the oscillatory behaviour
of $\Phi(u)$ near $u=0$,
which in turn depends on whether
the fold phase can be put in
Airy normal form uniformly in $c$.
\end{remark}

\begin{assumption}[Uniform cubic
  phase expansion \Cond]
\label{ass:Phi-cubic}
For $|u|\le 1$,
uniformly in $c\ge c_0$:
\begin{equation}\label{eq:Phi-cubic}
  \Phi(u)=C_*|u|^{-1/2}
  e^{i(\alpha_* u+c_3 u^3)}
  (1+r(u,c)),
\end{equation}
$|r(u,c)|\le C|u|$ and
$|\partial_u r|\le C$
uniformly.
\end{assumption}

\begin{remark}[Why uniform
  control is hard]
\label{rem:hard}
Assumption~\ref{ass:Phi-cubic}
requires:
\begin{enumerate}[\rm(a)]
\item Taylor expansion of the
  WKB amplitude commutes
  with the Fourier transform
  despite the singularity
  $|\beta|^{-1/2}$ at $\beta=0$.
\item The WKB approximation
  error introduces no
  $c$-dependent distortion
  at the fold scale
  $\xi\sim c^{-\theta_0/3}$.
\item The expansion is valid
  up to $|u|\sim c^{-\theta_0/3}$
  (Airy scale), not merely
  infinitesimally.
\end{enumerate}
Item~(b) is the core difficulty:
it requires uniform microlocal
control of the PSWF at the
fold singularity,
which goes beyond standard
Morse-lemma / Malgrange arguments.
Without item~(b), the cubic
expansion may hold locally
but with $c$-dependent constants,
rendering the scaling argument
void.
\end{remark}

\begin{remark}[The $\theta_0>4/5$
  threshold is an artifact \Heur]
\label{rem:theta-threshold}
The conditional conclusion
$S(c)=O(c^{\theta_0/2})$
(under Ass.~\ref{ass:Phi-cubic})
leads to the threshold
$\theta_0>4/5$.
This threshold is an
\emph{artifact} of the
assumed Airy exponent $1/2$:
if Assumption~\ref{ass:Phi-cubic}
holds only with
$|r(u,c)|\le C|u|^{1-\varepsilon}$
(any $\varepsilon>0$),
the effective exponent
becomes $\theta_0/2-\delta$
for some $\delta>0$,
and the threshold shifts to
$\theta_0>4/5+\delta'$.
Logarithmic losses would
also shift it.
The number $4/5$ has no
special mathematical
status in this problem.
\end{remark}

\begin{problem}[Control of
  $I_m$-tail \Rig-gap]
\label{prob:Im-tail}
Show:
$\sum_{m\ne 0}|I_m(x,c)|
=O(c^{\theta_0/2}/|x|^{1/2})$
uniformly in $c$.
Note: the term-by-term bound
$|I_m|=O(1/|m|)$
does not close the sum
uniformly as $K(c)\to\infty$.
A separate argument
(e.g., Euler--Maclaurin
with explicit error)
is needed.
\end{problem}

\begin{problem}[Uniform cubic
  phase for $\Phi$]
\label{prob:Phi-cubic}
Prove or disprove
Assumption~\ref{ass:Phi-cubic}.
\begin{enumerate}[\rm(A)]
\item If true: conclude
  $|\int_0^U\Phi(u)du|
  =O(U^{1/4})$
  (from Airy decay),
  giving $I_0=O(c^{5\theta_0/8}/|x|^{5/8})$
  and $S(c)=O(c^{\theta_0/2})$
  (up to log corrections;
  see Rem.~\ref{rem:theta-threshold}).
\item If false: prove
  $\Phi$ has no uniform
  cubic expansion;
  this would likely give
  a matching lower bound
  $|\int_0^U\Phi(u)du|
  =\Theta(U^{1/2})$,
  establishing
  $S(c)=\Theta(c^{\theta_0})$
  (Ces\`aro is sharp).
\end{enumerate}
\end{problem}

\section{Programme Summary}
\label{sec:summary}

\begin{theorem}[Reduction theorem,
  conditional \Cond]
\label{thm:reduction}
Conditional on
Assumption~\ref{ass:Phi-cubic},
Problems~\ref{prob:Im-tail},
(L), (R), (C),
and $\theta_0>4/5$
(with the caveat of
Rem.~\ref{rem:theta-threshold}):
$\Sigma_{\mathrm{int}}=o(1)
\Rightarrow
\mathrm{H3\ fails}
\Rightarrow
\mathrm{H1\ fails.}$
\end{theorem}

\begin{remark}[What Paper~XV
  contributes]
\label{rem:contribution}
Paper~XV is a reduction paper.
Its contribution is:
\begin{enumerate}[\rm(1)]
\item Proving that the gap
  $c^{2-\theta_0}$ is\emph{exactly}
  the frozen-phase rate
  (Lem.~\ref{lem:S-equivalence}).
\item Proving that Dirichlet-kernel
  methods cannot close this gap
  in the model
  (Lem.~\ref{lem:dirichlet-obstruction},
  with stated scope limitations).
\item Reducing the problem
  rigorously to a single
  1D question about $\Phi$
  (Prop.~\ref{prop:I0-1D}).
\item Formulating precisely
  the single hard assumption
  needed for further progress
  (Ass.~\ref{ass:Phi-cubic}).
\end{enumerate}
What Paper~XV does \emph{not} do:
close the gap; prove $S(c)=o(c^{\theta_0})$
beyond Ces\`aro;
or establish any improvement
that is not conditional.
\end{remark}
