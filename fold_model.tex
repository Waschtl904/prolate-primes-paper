\title{Mini-Model for the Fold Amplitude:\\
  Two Competing Global Mechanisms,\\
  Saddle Dominance for $u<0$,\\
  and a Conditional Improvement\\
  over Ces\`aro}

\author{[Author]}

\begin{abstract}
This note studies
\[
  \Phi(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u + c_3\beta^3)}\,d\beta,
  \qquad c_3>0\text{ fixed},
\]
and $F(U):=\int_0^U\Phi(u)\,du$.

The correct stationary phase
calculation gives
$|\Phi(u)|\sim|u|^{-3/4}$
from the saddle for $u<0$,
\emph{not} $|u|^{-1/2}$
(the error in Versions~1--3).

Since $|u|^{-3/4}\ll|u|^{-1/2}$,
the saddle decays faster
than the singularity.
The genuine question is therefore:
\emph{does the saddle dominate
the singularity for $u<0$?}

If yes:
$|\Phi(u)|=O(|u|^{-3/4})$
for $u<0$,
giving $|F(U)|=O(U^{1/4})$
--- a genuine improvement
over Ces\`aro $O(U^{1/2})$.

If no (singularity dominates):
$|\Phi(u)|=\Theta(|u|^{-1/2})$
and Ces\`aro is sharp.

This dichotomy is the
exact open problem for Paper~XVI.
\end{abstract}

\maketitle

\section{Setup and Baseline}
\label{sec:setup}

\begin{definition}[Fold amplitude]
\label{def:Phi}
Fix $c_3>0$,
$\chi\in C^\infty_c$,
$\chi\equiv 1$ near $0$.
\begin{equation}\label{eq:Phi}
  \Phi(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u+c_3\beta^3)}\,d\beta.
\end{equation}
Baseline ($c_3=0$):
$|\Phi_0(u)|=O(|u|^{-1/2})$,
$|F_0(U)|=O(U^{1/2})$.
\end{definition}

\begin{lemma}[Unconditional
  upper bound \Rig]
\label{lem:rig}
$|\Phi(u)|=O(|u|^{-1/2})$
for all $u\ne 0$,
all $c_3>0$.
Proof: $|\beta|^{-1/2}\chi\in L^1$;
Riemann--Lebesgue applies
to the full integrand
(the cubic factor
$e^{ic_3\beta^3}$ does
not change $L^1$ membership).
Consequence:
$|F(U)|=O(U^{1/2})$.
\end{lemma}

\section{Two Global Mechanisms}
\label{sec:mechanisms}

\begin{remark}[Not layers,
  but two asymptotic mechanisms
  \Rig]
\label{rem:mechanisms}
The integral~\eqref{eq:Phi}
is governed by two globally
acting mechanisms:

\medskip
\noindent\textbf{Mechanism~S
  (Singularity):}
The factor $|\beta|^{-1/2}$
at $\beta=0$ contributes
via the Fourier transform
of a singular function.
For $c_3=0$, this gives
the full $|\Phi_0(u)|=\Theta(|u|^{-1/2})$.
For $c_3>0$, the
cubic perturbation modifies
the contribution of the
singularity; whether it
cancels or reinforces it
is the non-trivial question.

\medskip
\noindent\textbf{Mechanism~P
  (Saddle):}
For $u<0$, the phase
$\phi(\beta)=\beta u+c_3\beta^3$
has two real critical points
$\pm\beta_0$,
$\beta_0=\sqrt{|u|/(3c_3)}$.
Stationary phase at $\beta_0$
gives a contribution
calculated in the next section.

\medskip
Neither mechanism is
``local in $\beta$'':
both act on the full
integral~\eqref{eq:Phi}.
A partition-of-unity
decomposition is valid
\emph{only} if the two
regions ($|\beta|\ll\beta_0$
and $|\beta|\sim\beta_0$)
can be separated with
explicit error control;
this is an open
technical problem.
\end{remark}

\section{Correct Stationary Phase
  Calculation}
\label{sec:sp}

\begin{proposition}[Correct
  saddle contribution \Rig]
\label{prop:saddle}
For $u<0$, $|u|\to\infty$:
the saddle $\beta_0
=\sqrt{|u|/(3c_3)}$.

\noindent\textbf{Amplitude
  at the saddle:}
\[
  |\beta_0|^{-1/2}
  =\left(\frac{|u|}{3c_3}\right)^{-1/4}
  \sim C|u|^{-1/4}.
\]
\noindent\textbf{Second derivative:}
\[
  \phi''(\beta_0)=6c_3\beta_0
  =6c_3\cdot\sqrt{\frac{|u|}{3c_3}}
  \sim C'|u|^{1/2}.
\]
\noindent\textbf{Stationary phase
  formula:}
\[
  |\text{saddle contribution}|
  \sim|\beta_0|^{-1/2}\cdot
  |\phi''(\beta_0)|^{-1/2}
  \sim|u|^{-1/4}\cdot|u|^{-1/4}
  =|u|^{-1/2}.
\]

\medskip
\noindent\textbf{Wait---
  the correct exponent is}
\[
  \boxed{|\Phi_{\mathrm{saddle}}(u)|
  \sim|u|^{-1/4}\cdot|u|^{-1/4}
  =|u|^{-1/2}.}
\]

\noindent See Remark~\ref{rem:exponent}
for the reanalysis.
\end{proposition}

\begin{remark}[Careful recount
  of the exponent \Rig]
\label{rem:exponent}
Let us recount carefully.
$\beta_0=\sqrt{|u|/(3c_3)}$,
so:
\begin{align*}
  |\beta_0|^{-1/2}
  &=(|u|/(3c_3))^{-1/4}
  =C|u|^{-1/4},\\
  |\phi''(\beta_0)|^{-1/2}
  &=(6c_3\beta_0)^{-1/2}
  =(6c_3)^{-1/2}
  (|u|/(3c_3))^{-1/4}
  =C'|u|^{-1/4}.
\end{align*}
Therefore the standard
stationary phase estimate
gives:
\begin{equation}\label{eq:sp-correct}
  \boxed{|\Phi_{\mathrm{saddle}}(u)|
  \sim C|u|^{-1/4}\cdot|u|^{-1/4}
  =C|u|^{-1/2}.}
\end{equation}

\noindent\textbf{Key observation:}
the saddle contribution gives
$O(|u|^{-1/2})$---
\emph{the same order as
the singularity}.
There is no automatic gain
from the saddle:
both mechanisms contribute
at the same order $|u|^{-1/2}$.

\noindent\textbf{Retraction:}
The referee's claim that
the correct exponent is
$|u|^{-3/4}$ would require:
\[
  |\text{saddle}|
  \sim|\beta_0|^{-1/2}
  \cdot|\phi''|^{-1/2}
  \sim|u|^{-1/4}\cdot|u|^{-1/2}
  =|u|^{-3/4}.
\]
This would hold only if
$|\phi''(\beta_0)|^{-1/2}
\sim|u|^{-1/2}$,
i.e., $\phi''(\beta_0)
\sim|u|$.
But $\phi''(\beta_0)
=6c_3\beta_0\sim|u|^{1/2}$,
not $|u|$.
So $|\phi''|^{-1/2}
\sim|u|^{-1/4}$,
and the total is $|u|^{-1/2}$.

\textbf{The referee's exponent
$|u|^{-3/4}$ would require
$\phi''\sim|u|$, which would
hold only for a quartic
(not cubic) phase.}
For a cubic phase,
the correct stationary
phase exponent from the
saddle is $|u|^{-1/2}$.
\end{remark}

\begin{remark}[Genuine dichotomy
  between S and P \Open]
\label{rem:dichotomy}
Since both Mechanism~S
(singularity) and Mechanism~P
(saddle) give $O(|u|^{-1/2})$,
the leading-order behaviour
of $\Phi(u)$ is determined
by their \emph{interference}:
\begin{enumerate}[\rm(a)]
\item\textbf{Constructive
  (no cancellation):}
  $|\Phi(u)|=\Theta(|u|^{-1/2})$;
  Ces\`aro bound is sharp;
  $|F(U)|=\Theta(U^{1/2})$.
\item\textbf{Partial or full
  cancellation:}
  $|\Phi(u)|=o(|u|^{-1/2})$;
  genuine improvement;
  $|F(U)|=o(U^{1/2})$.
\end{enumerate}
Whether cancellation
occurs depends on the
relative phase between the
singularity and saddle
contributions.
The saddle phase is:
\[
  \phi(\beta_0)
  =-\tfrac{2}{3}
  \sqrt{\tfrac{|u|^3}{3c_3}}
  \to-\infty.
\]
The singularity contribution
has phase $\sim e^{i\pi/4}$
(Fresnel).
The relative phase
oscillates rapidly for
large $|u|$;
averaging effects may
produce partial cancellation.
\end{remark}

\section{What the Mini-Model
  Actually Proves}
\label{sec:results}

\begin{theorem}[Correct
  conclusions \Rig+\Open]
\label{thm:correct}
\emph{Proved unconditionally:}
\begin{enumerate}[\rm(1)]
\item $|\Phi(u)|=O(|u|^{-1/2})$
  for all $u\ne 0$
  (Riemann--Lebesgue;
  holds regardless of $c_3$).
\item The saddle contribution
  for $u<0$ also gives
  $O(|u|^{-1/2})$:
  \emph{same order as
  the singularity}.
  No automatic gain.
\item $|F(U)|=O(U^{1/2})$.
\end{enumerate}

\emph{Open (genuine dichotomy):}
\begin{enumerate}[\rm(4)]
\item Do Mechanisms~S and~P
  cancel or reinforce?
  This is the \emph{only}
  remaining question.
\item If cancellation:
  $|\Phi(u)|=o(|u|^{-1/2})$;
  improvement exists;
  the exponent and structure
  of the improvement must
  be determined.
\item If no cancellation:
  $|\Phi(u)|=\Theta(|u|^{-1/2})$;
  Ces\`aro is sharp;
  programme stalls.
\end{enumerate}

\emph{Retractions from
previous versions:}
\begin{itemize}[nosep]
\item $|u|^{-3/4}$ exponent:
  \textbf{wrong for cubic phase};
  requires quartic.
\item ``exponential decay
  for $u>0$'': too strong;
  correct bound is
  $O(u^{-1/2})$ with possible
  algebraic improvement.
\item Global $U^{2/3}$ law:
  retracted.
\item Three-layer decomposition:
  replaced by two-mechanism
  framework.
\end{itemize}
\end{theorem}

\begin{problem}[The one
  computation \Open]
\label{prob:one}
Compute explicitly
the leading asymptotic
of $\Phi(u)$ for
$u<0$, $|u|\to\infty$:
\begin{equation}\label{eq:Phi-goal}
  \Phi(u)
  =\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u+c_3\beta^3)}\,d\beta
  =C_*(u)\cdot|u|^{-1/2}+
  \text{lower order},
\end{equation}
where $C_*(u)$ collects
both the singularity
phase ($e^{i\pi/4}$)
and the saddle phase
($e^{i\phi(\beta_0)}$).
\begin{enumerate}[\rm(A)]
\item If $|C_*(u)|=\Theta(1)$:
  Ces\`aro is sharp.
\item If $C_*(u)\to 0$:
  the cancellation is
  genuine; determine
  the rate.
\end{enumerate}
The oscillation of $C_*(u)$
due to $e^{i\phi(\beta_0)}
=e^{-i\frac{2}{3}(|u|/3c_3)^{3/2}}$
suggests that $|C_*(u)|$
fluctuates but has no
reason to tend to $0$.
\emph{Proving or disproving
$C_*(u)\to 0$ is the
single decisive task.}
\end{problem}

\begin{remark}[Connection
  to Paper~XVI \Open]
\label{rem:XVI}
The assumption
assl:Phi-cubic in Paper~XV
assumed $\Phi(u)$ has a
clean cubic phase form.
The mini-model shows:
\begin{itemize}[nosep]
\item The cubic phase
  appears in the \emph{saddle
  contribution} (Mechanism~P).
\item But Mechanism~S
  (singularity) also
  contributes $O(|u|^{-1/2})$
  with a different phase.
\item The assumption is valid
  iff Mechanism~S is
  sub-leading, i.e.,
  iff the singularity is
  negligible compared
  to the saddle.
\item This is exactly the
  open Problem~\ref{prob:one}.
\end{itemize}
Paper~XVI's first task:
prove or disprove that
Mechanism~S is sub-leading
for $u<0$.
\end{remark}
