\title{Mini-Model for the Fold Amplitude:\\
  A Negative Result ---\\
  Ces\`aro is Numerically Sharp\\
  and the Cubic Term Does Not\\
  Improve the Scaling}

\author{[Author]}

\begin{abstract}
This note studies
\begin{equation*}
  \Phi(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u + c_3\beta^3)}\,d\beta,
  \quad c_3>0,
\end{equation*}
and $F(U):=\int_0^U\Phi(u)\,du$.

\textbf{Main result (negative):}
Numerical computation with
$c_3=1$ gives:
\[
  |\Phi(u)|\approx|u|^{-0.53},
  \quad
  |F(U)|\approx U^{0.48}.
\]
Both are consistent with the
Ces\`aro scaling $|u|^{-1/2}$,
$U^{1/2}$.
\emph{The cubic term changes
the phase of $\Phi$,
not its amplitude scaling.}

Consequences:
\begin{enumerate}[\rm(1)]
\item Assumption~ass:Phi-cubic
  from Paper~XV
  (uniform cubic improvement
  over Ces\`aro) is
  \emph{not supported}
  by this model.
\item The saddle contribution
  (Mechanism~P) does not
  dominate the singularity
  (Mechanism~S) in amplitude;
  both give $O(|u|^{-1/2})$
  and their interference
  is not systematic.
\item The correct conjecture
  for Paper~XVI changes:
  \emph{prove}
  $S(c)=\Theta(c^{\theta_0})$
  (Ces\`aro is sharp),
  not $S(c)=O(c^{\theta_0/2})$.
\end{enumerate}
\end{abstract}

\maketitle

\section{Setup and Mechanisms}
\label{sec:setup}

\begin{definition}[Model integral]
\label{def:Phi}
Fix $c_3>0$,
$\chi\in C^\infty_c$,
$\chi\equiv 1$ near $0$:
\begin{equation}\label{eq:Phi}
  \Phi(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u+c_3\beta^3)}\,d\beta.
\end{equation}
Baseline ($c_3=0$):
$|\Phi_0(u)|=\Theta(|u|^{-1/2})$;
$|F_0(U)|=\Theta(U^{1/2})$.
\end{definition}

\begin{remark}[Two mechanisms,
  same scale \Rig]
\label{rem:mechanisms}
The integral is governed by:
\begin{enumerate}[\rm(S)]
\item[(S)] \textbf{Singularity}
  at $\beta=0$:
  $|\beta|^{-1/2}$ contributes
  $|\Phi_S(u)|=\Theta(|u|^{-1/2})$
  (Fourier transform of a
  singular measure;
  independent of $c_3$
  to leading order).
\item[(P)] \textbf{Saddle} at
  $\beta_0=\sqrt{|u|/(3c_3)}$
  (for $u<0$):
  stationary phase gives
  $|\Phi_P(u)|\sim
  |\beta_0|^{-1/2}\cdot
  |\phi''(\beta_0)|^{-1/2}
  =|u|^{-1/4}\cdot|u|^{-1/4}
  =|u|^{-1/2}$.
\end{enumerate}
\textbf{Both give
$O(|u|^{-1/2})$.}
There is no automatic
amplitude gain from the saddle.
\end{remark}

\section{Numerical Evidence}
\label{sec:numerical}

\begin{proposition}[Numerical
  scaling \Num]
\label{prop:numerical}
Numerical computation
(trapezoidal rule,
$\beta\in[-50,50]$,
$10^4$ points,
Gaussian cutoff $\chi(\beta)
=e^{-\beta^2/100}$,
$c_3=1$):
\begin{align}
  |\Phi(u)|
  &\approx|u|^{-0.53}
  \quad(\text{fitted for }|u|>5),
  \label{eq:Phi-num}\\
  |F(U)|
  &\approx U^{0.48}
  \quad(\text{fitted for }U>5).
  \label{eq:F-num}
\end{align}
Both are consistent with
the Ces\`aro reference
$|u|^{-1/2}$, $U^{1/2}$
within the expected
numerical uncertainty
(deviation $<0.05$).
\end{proposition}

\begin{remark}[What the
  exponent $-0.53$ means \Num]
\label{rem:exponent-interp}
The fitted exponent
$-0.53$ lies between
$-1/2=-0.5$
(singularity baseline)
and $-1/2=-0.5$
(saddle baseline;
\emph{same value}).
There is no signature
of:
\begin{itemize}[nosep]
\item $-1/3\approx-0.33$
  (earlier erroneous
  saddle estimate);
\item $-3/4=-0.75$
  (referee's suggested
  exponent;
  requires quartic phase);
\item any exponent
  significantly below $-1/2$.
\end{itemize}
Conclusion: \emph{the
singularity at $\beta=0$
controls the scaling;
the cubic term changes
only the phase}.
\end{remark}

\begin{remark}[Phase vs.
  amplitude \Rig]
\label{rem:phase-vs-amp}
The cubic term
$e^{ic_3\beta^3}$:
\begin{itemize}[nosep]
\item strongly modifies
  the \emph{phase} of $\Phi(u)$
  (the saddle phase
  $\phi(\beta_0)=
  -\frac{2}{3}(|u|/3c_3)^{3/2}$
  oscillates rapidly);
\item does \emph{not} change
  the amplitude scaling
  $|\Phi(u)|=\Theta(|u|^{-1/2})$.
\end{itemize}
This is the central
negative result:
\[
  \boxed{|\Phi(u)|
  =\Theta(|u|^{-1/2})
  \text{ with or without }c_3.}
\]
The interference between
$\Phi_S$ and $\Phi_P$
is not systematic
(relative phase oscillates
with frequency $|u|^{3/2}$)
and does not produce
cancellation on average.
\end{remark}

\section{Consequences for
  Paper~XV and Paper~XVI}
\label{sec:consequences}

\begin{theorem}[Negative result
  \Num+\Rig]
\label{thm:negative}
In the mini-model:
\begin{enumerate}[\rm(1)]
\item $|\Phi(u)|=\Theta(|u|^{-1/2})$
  (numerically; analytically:
  the upper bound
  $O(|u|^{-1/2})$ is
  rigorous; the lower bound
  $\Omega(|u|^{-1/2})$
  is numerical).
\item $|F(U)|=\Theta(U^{1/2})$
  (same).
\item The cubic term in
  the phase does not
  improve the Ces\`aro
  scaling.
\item Assumption~ass:Phi-cubic
  from Paper~XV is
  \emph{not supported}
  by this model:
  there is no uniform
  cubic phase expansion
  that yields $o(U^{1/2})$.
\end{enumerate}
\end{theorem}

\begin{remark}[Revised
  programme \Open]
\label{rem:revised}
The negative result
changes the research
programme:
\begin{enumerate}[\rm(A)]
\item \textbf{Old conjecture
  (Paper~XV, Ass.~ass:Phi-cubic):}
  $S(c)=O(c^{\theta_0/2})$;
  Airy structure improves
  Ces\`aro.
  \emph{Not supported by
  mini-model.}
\item \textbf{New conjecture:}
  $S(c)=\Theta(c^{\theta_0})$;
  Ces\`aro is sharp;
  no improvement.
  If true, this would be
  a \emph{rigorous lower bound}
  result.
\item \textbf{Paper~XVI task:
  prove the lower bound.}
  Show
  $S(c)\ge Cc^{\theta_0}$
  for some $C>0$;
  this would close
  the problem definitively.
\end{enumerate}
\end{remark}

\begin{remark}[Limitation of
  the mini-model \Warn]
\label{rem:limitation}
The mini-model uses
a Gaussian cutoff
$\chi(\beta)=e^{-\beta^2/100}$
rather than the exact
PSWF amplitude.
The following are not
excluded:
\begin{itemize}[nosep]
\item the actual PSWF
  fold integral may have
  additional cancellation
  structure not present
  in this model;
\item the cutoff $\chi$
  affects the singularity
  structure near $\beta=0$;
\item the result is
  numerical, not
  a rigorous lower bound.
\end{itemize}
A rigorous proof of
$|\Phi(u)|=\Omega(|u|^{-1/2})$
would require:
\begin{enumerate}[\rm(i)]
\item showing the
  singularity contribution
  is not cancelled by
  the saddle;
\item this is an analytic
  computation of the
  coefficient $C_*(u)$
  in $\Phi(u)=C_*(u)
  |u|^{-1/2}+\text{l.o.t.}$
  and showing $\limsup
  |C_*(u)|>0$.
\end{enumerate}
\end{remark}

\begin{problem}[Lower bound:
  the new task]
\label{prob:lower}
Prove:
\[
  \limsup_{|u|\to\infty}
  |u|^{1/2}|\Phi(u)|>0.
\]
This would establish
$|\Phi(u)|=\Omega(|u|^{-1/2})$
and confirm Ces\`aro
is sharp.

Approach:
compute the coefficient
\[
  C_*(u):=|u|^{1/2}\Phi(u)
  =|u|^{1/2}\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u+c_3\beta^3)}
  \,d\beta.
\]
From Mechanism~S:
$C_*^{(S)}(u)\to
C_0:=\int_0^\infty
|\beta|^{-1/2}\chi(\beta)
e^{i\pi/4}d\beta\ne 0$
(Fresnel integral;
finite non-zero).
From Mechanism~P:
$C_*^{(P)}(u)\sim
e^{-i(2/3)(|u|/3c_3)^{3/2}}
\to 0$ \emph{weakly}
(rapid oscillation;
no pointwise limit).

Therefore:
$\limsup|C_*(u)|
\ge|C_0|-
\limsup|C_*^{(P)}(u)|$.
If $|C_0|>0$
(which it is: Fresnel)
and $|C_*^{(P)}|\le
|C_0|/2$ eventually,
then $\limsup|C_*(u)|
\ge|C_0|/2>0$.
\emph{This argument is
incomplete: it requires
showing $|C_*^{(P)}|
\le|C_0|/2$ for large $|u|$,
which is a quantitative
bound on the saddle
amplitude relative to
the Fresnel constant.}
\end{problem}
