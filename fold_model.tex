\title{Mini-Model for the Fold Amplitude:\\
  Two Regimes, One Rigorous Result,\\
  and the Correct Open Problem}

\author{[Author]}

\begin{abstract}
We study
\[
  \Phi_\alpha(u):=\int_0^\infty
  \beta^{-1/2}\chi\!\left(
    \tfrac{\beta}{|u|^\alpha}
  \right)
  e^{i(-|u|\beta + c_3\beta^3)}
  \,d\beta,
  \quad u<0,\;c_3>0,
\]
for $\alpha\ge 0$.
Two qualitatively
different behaviors:
\begin{enumerate}[\rm(A)]
\item $\alpha=0$
  (\textbf{fixed compact
  support}): the saddle
  $\beta_0\to\infty$
  exits the support;
  Erd\'elyi's theorem
  gives the\emph{unique,
  rigorous, numerically
  confirmed} leading term:
  \[
    \Phi_0(u)
    =\sqrt{\pi}\,e^{-i\pi/4}
    |u|^{-1/2}+O(|u|^{-3/2}).
  \]
  Ces\`aro is sharp:
  $|F(U)|=\Theta(U^{1/2})$.
\item $\alpha>0$
  (\textbf{growing support}):
  the saddle stays
  inside the support;
  Airy-type oscillations
  prevent any single
  power-law exponent;
  the amplitude of
  $|\Phi_\alpha(u)|$
  oscillates on a
  log-scale and no
  clean $|u|^{\gamma}$
  law holds.
\end{enumerate}
\textbf{The earlier
claim of a universal
exponent $-1/2$ for
all $\alpha\ge 0$
is retracted.}
It was based on
an incorrect stationary
phase computation
that failed to account
for the coalescence
of the saddle with
the singularity
for $\alpha>1/2$.
\end{abstract}

\maketitle

\section{The Fixed-Support
  Result: Rigorous}
\label{sec:erdelyi}

\begin{theorem}[Erd\'elyi,
  unique leading order
  \Rig]
\label{thm:erdelyi}
For $\alpha=0$,
$\chi\in C^\infty_c([0,R])$
fixed:
\[
  \Phi_0(u)
  =\sqrt{\pi}\,e^{-i\pi/4}
  |u|^{-1/2}+O(|u|^{-3/2}),
  \quad|u|\gg c_3 R^2.
\]
This is the complete
first-order asymptotic.
The cubic phase
enters only at
$O(c_3|u|^{-3/2})$.
\end{theorem}

\begin{corollary}[Ces\`aro
  sharp \Rig]
$\limsup|u|^{1/2}
|\Phi_0(u)|=\sqrt{\pi}>0$;
$|F_0(U)|=\Theta(U^{1/2})$.
\end{corollary}

\begin{proposition}[Numerical
  confirmation \Num]
With $c_3=R=1$:
\begin{center}
\begin{tabular}{c|c|c}
$|u|$ &
  $|\Phi_0|/
    (\sqrt{\pi}|u|^{-1/2})$
  & fitted $\gamma$
  (range)\\ \hline
$10$ & $0.971$ & $-0.502$
  \ ($16$--$50$)\\
$100$ & $1.000$ & $-0.500$
  \ ($50$--$200$)\\
$300$ & $1.000$ & $-0.500$
  \ ($200$--$1000$)
\end{tabular}
\end{center}
\end{proposition}

\section{The Growing-Support
  Problem: No Clean Power Law}
\label{sec:growing}

\begin{remark}[Numerical
  evidence \Num+\Warn]
\label{rem:growing}
For $\alpha>0$,
fitted exponents
in different $|u|$-ranges
are \emph{not stable}:
\begin{center}
\begin{tabular}{c|c|c|c}
$\alpha$ &
  $\gamma([16,50])$ &
  $\gamma([50,200])$ &
  $\gamma([200,1000])$
  \\ \hline
$0.00$ &
  $-0.502$ & $-0.500$ & $-0.500$ \\
$0.25$ &
  $-0.391$ & $-0.489$ & $+0.551$ \\
$0.50$ &
  $-0.299$ & $-1.235$ & $-0.071$ \\
$0.75$ &
  $-0.337$ & $-0.053$ & $+0.334$ \\
$1.00$ &
  $-0.558$ & $+0.175$ & $-0.343$
\end{tabular}
\end{center}
\textbf{For $\alpha>0$:
the exponent is
not converging;
$|\Phi_\alpha(u)|$
oscillates on a
log-scale.}
This is Airy-type
behavior: the saddle
stays inside the
support, generating
persistent oscillations
whose amplitude
cannot be described
by a single power law.
\end{remark}

\begin{remark}[Why the
  universal $-1/2$
  claim was wrong \Warn]
\label{rem:error}
In the dimensional
analysis of Case~III
($\alpha>1/2$):
the saddle $s_0
=\sqrt{|u|^{1-2\alpha}
/(3c_3)}\to 0$
approaches the
singularity $s=0$.
The standard stationary
phase formula requires
the amplitude to be
\emph{smooth} at
the saddle.
For $s_0\to 0$
and singular weight
$s^{-1/2}$,
this condition fails:
the amplitude
$s_0^{-1/2}\to\infty$.
The SP formula
$\sim s_0^{-1/2}
\cdot (\text{curvature})^{-1/2}$
cannot be applied.
The correct analysis
requires the
CFU/coalescence theory,
which gives an
Airy-modulated result,
\emph{not} a clean
power law.
\end{remark}

\section{What Is Known
  and What Is Open}
\label{sec:summary}

\begin{theorem}[Summary
  \Rig+\Num]
\label{thm:summary}
\begin{enumerate}[\rm(1)]
\item\textbf{Fixed support
  ($\alpha=0$):}
  $|\Phi_0(u)|
  =\Theta(|u|^{-1/2})$;
  rigorous;
  Ces\`aro sharp.
\item\textbf{Growing
  support ($\alpha>0$):}
  $|\Phi_\alpha(u)|$
  has no clean
  power-law exponent;
  it oscillates with
  Airy-type modulation.
  The \emph{envelope}
  of $|\Phi_\alpha|$
  likely decays,
  but the rate is
  not $|u|^{-1/2}$
  and is not a
  simple power.
\item\textbf{The exponent
  $-1/2$ is not
  universal.}
  It holds only
  for fixed support.
\end{enumerate}
\end{theorem}

\begin{problem}[PSWF
  regime \Open]
\label{prob:PSWF}
For the PSWF
fold integral
in Paper~XV:
\begin{enumerate}[\rm(i)]
\item Determine $\alpha$:
does the natural
bandwidth of the
PSWF spectral
representation
grow with the
spectral parameter
$c$, and if so,
at what rate?
\item If $\alpha=0$:
Erd\'elyi applies;
$|\Phi|\sim|u|^{-1/2}$;
Ces\`aro is sharp.
\item If $\alpha>0$:
the problem is
genuinely in the
Airy/CFU regime;
no single exponent;
the Ces\`aro bound
$O(U^{1/2})$
may or may not
be sharp depending
on cancellation
between Airy
oscillations.
\end{enumerate}
This is the
correct formulation
of the open problem.
\end{problem}
