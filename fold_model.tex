\title{Mini-Model for the Fold Amplitude:\\
  CFU-Type Endpoint--Saddle Coalescence\\
  with Singular Amplitude}

\author{[Author]}

\begin{abstract}
We study
\begin{equation*}
  \Phi(u):=\int_0^\infty
  \beta^{-1/2}\chi(\beta)
  e^{i(\beta u + c_3\beta^3)}\,d\beta,
  \quad u\in\mathbb{R},\;c_3>0.
\end{equation*}
Three facts are established
or identified:
\begin{enumerate}[\rm(1)]
\item \textbf{Upper bound}
  (rigorous):
  $|\Phi(u)|=O(|u|^{-1/2})$
  via Erd\'elyi's singular
  endpoint theorem.
  Riemann--Lebesgue gives
  only $o(1)$.
\item \textbf{Correct asymptotic
  class}:
  this is a
  \emph{CFU-type
  endpoint--saddle
  coalescence problem}
  (Chester--Friedman--Ursell
  1957)
  with a singular amplitude
  $\beta^{-1/2}$.
  The leading asymptotics
  for $u<0$ is \emph{not}
  a sum of two independent
  constants times phases;
  it is a single uniform
  Airy-type object:
  \[
    \Phi(u)\sim
    |u|^{-1/2}\cdot
    \mathcal{A}(u),
  \]
  where $\mathcal{A}(u)$
  is determined by the
  CFU normal form and
  the Jacobian of the
  coordinate change at
  $\beta=0$.
\item \textbf{True open
  problem}:
  is the integral
  \emph{CFU-reducible}
  despite the singularity
  $\beta^{-1/2}$ at the
  endpoint $\beta=0$?
  The singularity may
  obstruct the standard
  CFU coordinate change;
  this is the technical
  core of the problem.
\end{enumerate}
\end{abstract}

\maketitle

\section{Rigorous Bounds}
\label{sec:bounds}

\begin{lemma}[RL: $o(1)$ only
  \Rig]
\label{lem:RL}
$\beta^{-1/2}\chi\in L^1$
$\Rightarrow$ $\Phi(u)=o(1)$.
No rate follows from RL.
\end{lemma}

\begin{lemma}[Erd\'elyi:
  $O(|u|^{-1/2})$ \Rig]
\label{lem:Erdelyi}
For $u>0$, $u\to+\infty$,
or for $u<0$ restricted
to the region $|\beta|
\ll \beta_0(u)$
(endpoint contribution
only):
\begin{equation}\label{eq:Erdelyi}
  \int_0^\infty
  \beta^{-1/2}\chi(\beta)
  e^{i\beta u}\,d\beta
  = e^{i\pi/4}\sqrt{\pi}
  \cdot u^{-1/2}
  + O(u^{-3/2}).
\end{equation}
This requires the cubic
term $c_3\beta^3$ to be
negligible, i.e.~$c_3
\beta_0^2 \ll |u|$.
For $u<0$, the saddle
$\beta_0=\sqrt{|u|/(3c_3)}
\to\infty$, so the cubic
term is \emph{not}
negligible globally;
Eq.~\eqref{eq:Erdelyi}
applies only to the
near-endpoint contribution.
\end{lemma}

\begin{remark}[Summary \Rig]
\label{rem:bounds-summary}
The rigorous content is:
\[
  \Phi(u)=O(|u|^{-1/2}),
  \qquad
  |F(U)|=O(U^{1/2}).
\]
These are upper bounds.
The lower bound is open.
\end{remark}

\section{The Correct Asymptotic
  Class: CFU Coalescence}
\label{sec:CFU}

\begin{remark}[Why not additive
  $C_S + C_P$ \Warn]
\label{rem:not-additive}
A natural first attempt
is to write:
\[
  \Phi(u)
  \stackrel{?}{=}
  |u|^{-1/2}
  \bigl(
    C_S e^{i\pi/4}
    + C_P
    e^{i\phi(\beta_0)}
  \bigr)
  + \text{lower order},
\]
where $C_S$ comes from
the endpoint and $C_P$
from the saddle.
\textbf{This is not valid.}
Reason: in CFU-type
problems, the coordinate
change
\[
  \phi(\beta,u)
  :=\beta u+c_3\beta^3
  \longmapsto
  \pm\tfrac{1}{3}t^3
  + \lambda(u)\cdot t
\]
\emph{couples} the image
of $\beta=0$ and the
image of $\beta_0$ into
a single smooth function
of $t$.
The result is not a
sum of two independent
contributions but a
single uniform expression
involving the Airy
function.
Separate constants
$C_S, C_P$ do not
exist as independent
asymptotic objects.
\end{remark}

\begin{remark}[CFU structure
  in the non-singular case
  \Rig]
\label{rem:CFU-nonsingular}
For a smooth amplitude
$a(\beta)$ (no singularity
at $\beta=0$), the
CFU method (Chester,
Friedman, Ursell 1957;
Olver 1974, Ch.~11)
gives:
\begin{equation}\label{eq:CFU-standard}
  \int_0^\infty
  a(\beta)
  e^{i(\beta u+c_3\beta^3)}
  \,d\beta
  \sim
  e^{i\psi(u)}
  \bigl[
    A(u)\,
    \mathrm{Ai}(\lambda(u))
    + B(u)\,
    \mathrm{Ai}'(\lambda(u))
  \bigr],
\end{equation}
where $\lambda(u)\sim
-(3c_3)^{-1/3}|u|^{1/3}$
for $u\to-\infty$,
and $A(u), B(u)$ are
smooth functions of $u$
determined by the
coordinate change
$\beta\mapsto t$.
For $|u|\to\infty$:
$\mathrm{Ai}(\lambda)
\sim\lambda^{-1/4}
\cos(\tfrac{2}{3}
|\lambda|^{3/2}-\tfrac{\pi}{4})$,
which gives the
$|u|^{-1/12}$-type
growth of the Airy
envelope times the
$|u|^{1/3}$ scaling
of $\lambda$---yielding
the standard exponent
for non-singular
oscillatory integrals.
\end{remark}

\begin{remark}[Effect of
  singular amplitude
  $\beta^{-1/2}$ \Warn]
\label{rem:singular-amplitude}
In our case, the amplitude
is $a(\beta)=\beta^{-1/2}
\chi(\beta)$, singular
at $\beta=0$.
Two effects:
\begin{enumerate}[\rm(i)]
\item The endpoint $\beta=0$
  maps to some point
  $t_0$ in the CFU
  $t$-coordinate.
  If $t_0\ne 0$,
  the singularity appears
  as an interior
  singularity in the
  $t$-integral,
  and CFU applies
  modulo a singular
  weight $|t-t_0|^{-1/2}$.
\item If $t_0=0$
  (the singularity
  coincides with the
  Airy turning point),
  the standard CFU
  theory does not apply;
  a modified expansion
  is needed.
\end{enumerate}
Which case holds depends
on whether $\beta=0$
maps to the saddle
critical point in $t$,
which depends on the
geometry of $\phi$.
\end{remark}

\section{The True Open Problem}
\label{sec:open}

\begin{problem}[CFU
  reducibility with singular
  amplitude \Open]
\label{prob:CFU}
Is the integral
$\Phi(u)=\int_0^\infty
\beta^{-1/2}\chi(\beta)
e^{i(\beta u+c_3\beta^3)}
d\beta$ reducible to
a uniform Airy expansion
of the form~\eqref{eq:CFU-standard}
with the singular amplitude
carried through the
CFU coordinate change?

Specifically:
\begin{enumerate}[\rm(i)]
\item Compute the CFU
  map $\beta\mapsto t$
  for $\phi(\beta,u)
  =\beta u+c_3\beta^3$;
  determine where
  $\beta=0$ maps.
\item If $\beta=0\mapsto
  t_*\ne 0$:
  the singular amplitude
  $|t-t_*|^{-1/2}$
  appears in the $t$-integral;
  apply endpoint singular
  asymptotics to the
  $t$-integral.
  This gives a modified
  Airy expansion with
  coefficients determined
  by $t_*$ and the
  Jacobian $d\beta/dt$
  at $\beta=0$.
\item If $\beta=0\mapsto
  t_*=0$:
  standard CFU fails;
  a modified theory is
  required (see
  Temme 1985;
  Kaminski 1994).
\end{enumerate}
\textbf{This is the exact
technical question that
Assumption~ass:Phi-cubic
of Paper~XV was implicitly
assert\-ing without proof.}
\end{problem}

\begin{remark}[Consequence
  for the limsup conjecture
  \Open]
\label{rem:limsup}
If Problem~\ref{prob:CFU}
is resolved and gives
a uniform Airy expansion
$\Phi(u)\sim|u|^{-1/2}
\mathcal{A}(u)$,
then:
\begin{itemize}[nosep]
\item $\mathcal{A}(u)$
  is an \emph{Airy-type}
  function, not a sum
  of two constants;
\item whether
  $\limsup|\mathcal{A}(u)|>0$
  follows from the known
  asymptotics of Airy
  functions (non-vanishing
  oscillating envelope);
\item the three-scenario
  classification (A/B/C
  in v8) is replaced
  by: does the Airy
  coefficient in the
  CFU expansion vanish?
  For non-degenerate
  geometry, it does not.
\end{itemize}
\end{remark}

\begin{conjecture}[\Open]
\label{conj:sharp}
The integral is
CFU-reducible
(Problem~\ref{prob:CFU},
case~\rm(ii), $t_*\ne 0$),
and $\limsup_{u\to-\infty}
|u|^{1/2}|\Phi(u)|>0$.
\end{conjecture}

\section{Numerical Evidence}
\label{sec:num}

\begin{proposition}[\Num]
$|\Phi(u)|\approx|u|^{-0.53}$,
$|F(U)|\approx U^{0.48}$
(see above).
Consistent with
$O(|u|^{-1/2})$;
does not exclude
full Airy cancellation
at larger $|u|$.
\end{proposition}
