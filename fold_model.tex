\title{Mini-Model for the Fold Amplitude:\\
  Singular Endpoint Asymptotics,\\
  Equal-Order Interference,\\
  and the Ces\`aro Sharpness Conjecture}

\author{[Author]}

\begin{abstract}
We study
\[
  \Phi(u):=\int_0^\infty
  \beta^{-1/2}\chi(\beta)
  e^{i(\beta u + c_3\beta^3)}\,d\beta,
  \quad c_3>0,
\]
and $F(U):=\int_0^U\Phi(u)\,du$.

The bound $|\Phi(u)|=O(|u|^{-1/2})$
is established via the
\emph{singular endpoint
asymptotics} theorem
(Erd\'elyi 1956;
not from Riemann--Lebesgue,
which gives only $o(1)$).

For $u<0$, a saddle at
$\beta_0=\sqrt{|u|/(3c_3)}$
also contributes $O(|u|^{-1/2})$
--- the same order as
the endpoint.
\emph{This is an
equal-order interference
problem, not a perturbation.}

The single open question:
$\limsup_{|u|\to\infty}
|u|^{1/2}|\Phi(u)|>0$?
\end{abstract}

\maketitle

\section{Setup}
\label{sec:setup}

\begin{definition}[Fold amplitude]
\label{def:Phi}
Fix $c_3>0$,
$\chi\in C^\infty_c([0,\infty))$,
$\chi\equiv 1$ near $0$,
$\chi\ge 0$:
\begin{equation}\label{eq:Phi}
  \Phi(u)
  :=\int_0^\infty
  \beta^{-1/2}\chi(\beta)
  e^{i(\beta u+c_3\beta^3)}\,d\beta.
\end{equation}
Without cubic ($c_3=0$):
$\Phi_0(u)=\int_0^\infty
\beta^{-1/2}\chi(\beta)
e^{i\beta u}d\beta$.
\end{definition}

\begin{remark}[Why $[0,\infty)$
  instead of $\mathbb{R}$]
\label{rem:half-line}
The integral over
$\mathbb{R}$ with amplitude
$|\beta|^{-1/2}$
splits into two half-line
integrals; the structure
of the leading asymptotic
is the same.
Working on $[0,\infty)$
isolates the single
endpoint singularity
at $\beta=0$ and
avoids notational clutter.
\end{remark}

\section{The $O(|u|^{-1/2})$
  Bound: Correct Proof}
\label{sec:upper}

\begin{lemma}[Riemann--Lebesgue
  gives only $o(1)$ \Rig]
\label{lem:RL}
Since $\beta^{-1/2}\chi
\in L^1([0,\infty))$,
the Riemann--Lebesgue
lemma gives:
\[
  \Phi(u)\to 0
  \quad(|u|\to\infty).
\]
\textbf{This is all that
RL gives.}
In particular, RL does
not give the rate
$O(|u|^{-1/2})$.
\end{lemma}

\begin{lemma}[Singular endpoint
  asymptotics \Rig]
\label{lem:erdelyi}
Let $f(\beta)
=\beta^{\alpha-1}g(\beta)$
with $0<\alpha<1$
and $g\in C^\infty_c([0,\infty))$,
$g(0)\ne 0$.
Then (Erd\'elyi 1956,
Bleistein--Handelsman 1975):
\begin{equation}\label{eq:erdelyi}
  \int_0^\infty f(\beta)
  e^{i\lambda\beta}\,d\beta
  =e^{i\pi\alpha/2}
  \Gamma(\alpha)
  g(0)\lambda^{-\alpha}
  +O(\lambda^{-(\alpha+1)})
  \quad(\lambda\to+\infty).
\end{equation}
For our case:
$\alpha=1/2$,
$g(\beta):=\chi(\beta)
  e^{ic_3\beta^3/u}$
(treating the cubic
as a correction;
valid when $c_3\beta^3
\ll |u|\beta$,
i.e.~$\beta\ll|u|^{1/2}$):
\begin{equation}\label{eq:Phi0-asy}
  \Phi_0(u)
  =e^{i\pi/4}
  \Gamma(\tfrac{1}{2})
  \chi(0)u^{-1/2}
  +O(u^{-3/2})
  =\frac{\sqrt{\pi}
    e^{i\pi/4}}{u^{1/2}}
  +O(u^{-3/2})
  \quad(u\to+\infty).
\end{equation}
This gives
$|\Phi_0(u)|=\Theta(u^{-1/2})$
\textbf{rigorously}.

For $c_3>0$:
the cubic perturbation
$e^{ic_3\beta^3}$ modifies
the effective $g(0)$
to $g_c(0):=e^{0}=1$
(since the cubic vanishes
at $\beta=0$);
the leading term is
unchanged:
\begin{equation}\label{eq:Phi-endpoint}
  \Phi_{\mathrm{endpt}}(u)
  :=\text{endpoint contribution}
  =\frac{\sqrt{\pi}
    e^{i\pi/4}}{u^{1/2}}
  +O(u^{-3/2})
  \quad(u\to+\infty).
\end{equation}
\textbf{The cubic term does not
change the leading endpoint
asymptotics for $u>0$.}
\end{lemma}

\begin{remark}[Validity for
  $u<0$ \Rig]
\label{rem:u-neg}
For $u<0$, the endpoint
asymptotics apply to
the integral over
$[0,\varepsilon]$
(near $\beta=0$) with
the same leading term;
the saddle at
$\beta_0=\sqrt{|u|/(3c_3)}>0$
also contributes
(Section~\ref{sec:saddle}).
The endpoint and saddle
both contribute
$O(|u|^{-1/2})$;
see below.
\end{remark}

\section{The Saddle Contribution
  for $u<0$}
\label{sec:saddle}

\begin{lemma}[Saddle asymptotics
  \Rig]
\label{lem:saddle}
For $u<0$, $|u|\to\infty$:
the phase
$\phi(\beta)=\beta u+c_3\beta^3$
has a critical point
$\phi'(\beta_0)=0$ at
$\beta_0=\sqrt{|u|/(3c_3)}$.
Standard stationary phase
(assuming $\beta_0$ is
away from $0$, which
holds for $|u|\gg 1$):

\noindent\textbf{Amplitude:}
\[
  \beta_0^{-1/2}
  =(|u|/(3c_3))^{-1/4}
  =C_1|u|^{-1/4}.
\]
\noindent\textbf{Phase curvature:}
\[
  |\phi''(\beta_0)|
  =6c_3\beta_0
  =6c_3(|u|/(3c_3))^{1/2}
  =C_2|u|^{1/2}.
\]
\noindent\textbf{Stationary phase
  estimate:}
\begin{equation}\label{eq:saddle-est}
  |\Phi_{\mathrm{saddle}}(u)|
  \sim\beta_0^{-1/2}
  \cdot|\phi''(\beta_0)|^{-1/2}
  =C_1|u|^{-1/4}
  \cdot C_2^{-1/2}|u|^{-1/4}
  =C|u|^{-1/2}.
\end{equation}

\textbf{The saddle contributes
$O(|u|^{-1/2})$ ---
the same order as the
endpoint.}

The saddle phase:
\[
  \phi(\beta_0)
  =\beta_0 u+c_3\beta_0^3
  =-\tfrac{2}{3}
  \bigl(|u|^3/(3c_3)\bigr)^{1/2}
  \to-\infty.
\]
\end{lemma}

\begin{remark}[Not a perturbation
  problem \Rig]
\label{rem:not-perturbation}
Since both mechanisms
give $O(|u|^{-1/2})$:
\[
  \Phi(u)=
  \underbrace{C_S u^{-1/2}
    e^{i\pi/4}}_{
    \text{endpoint}}
  +\underbrace{C_P(u)
    |u|^{-1/2}
    e^{i\phi(\beta_0)}}_{
    \text{saddle}}
  +O(|u|^{-3/2}),
\]
where $C_S=\sqrt{\pi}$
and $|C_P(u)|=O(1)$.
This is an
\emph{equal-order
interference problem}:
not a leading term plus
a small perturbation,
but two contributions
of the same order with
rapidly varying relative
phase
$\phi(\beta_0)-\pi/4
=-\frac{2}{3}(|u|/3c_3)^{3/2}
-\pi/4$.

\textbf{The split
$\Phi=\Phi_{\text{endpt}}
+\Phi_{\text{saddle}}$
is not rigorous without
a partition of unity}
$\chi_S+\chi_P=1$
with $\mathrm{supp}(\chi_S)
\subset[0,2|u|^{1/3}]$
and $\mathrm{supp}(\chi_P)
\subset[|u|^{1/3},\infty)$,
and explicit
$O(|u|^{-3/2})$ error
bounds on each piece.
This has not been carried out.
\end{remark}

\section{The Interference
  Problem and the Conjecture}
\label{sec:interference}

\begin{definition}[Leading
  coefficient]
\label{def:Cstar}
For $u<0$, $|u|\gg 1$:
\begin{equation}\label{eq:Cstar}
  C_*(u):=|u|^{1/2}\Phi(u)
  \approx C_S e^{i\pi/4}
  +C_P\cdot e^{-i\frac{2}{3}
    (|u|/3c_3)^{3/2}}.
\end{equation}
(Formal; requires rigorous
partition to make precise.)
\end{definition}

\begin{remark}[Behaviour of
  $C_*(u)$]
\label{rem:Cstar}
The two components:
\begin{itemize}[nosep]
\item $C_S e^{i\pi/4}$:
  constant, non-zero
  ($C_S=\sqrt{\pi}\ne 0$).
\item $C_P e^{-i\frac{2}{3}
  (|u|/3c_3)^{3/2}}$:
  oscillates with
  frequency $\sim|u|^{1/2}$;
  if $|C_P|<|C_S|$,
  then $|C_*(u)|
  \ge|C_S|-|C_P|>0$
  for all large $|u|$.
\end{itemize}
The critical unknown:
\textbf{is $|C_P|<|C_S|$?}
If yes: $\limsup|C_*(u)|
\ge|C_S|-|C_P|>0$;
Ces\`aro is sharp.
If $|C_P|=|C_S|$:
the oscillating relative
phase means
$|C_*(u)|$ fluctuates
between $0$ and $2|C_S|$;
$\limsup|C_*(u)|=2|C_S|>0$
but $\liminf|C_*(u)|=0$.

\textbf{In either case,
$\limsup|C_*(u)|>0$
(as long as $C_S\ne 0$),
provided $|C_P|$ does
not cancel $C_S e^{i\pi/4}$
for all large $|u|$
simultaneously---which
would require a very
specific phase resonance.}
\end{remark}

\begin{conjecture}[Ces\`aro is
  sharp \Open]
\label{conj:sharp}
\[
  \limsup_{|u|\to\infty}
  |u|^{1/2}|\Phi(u)|>0.
\]
Equivalently:
$|F(U)|=\Theta(U^{1/2})$.

Heuristic support:
$C_S=\sqrt{\pi}\ne 0$
(rigorous);
$|C_P|$ is bounded
but not identically
equal to
$|C_S e^{i\pi/4}|$
for all $u$
(no phase resonance
expected for irrational
frequency $(|u|/3c_3)^{3/2}$).

What is needed for a
rigorous proof:
\begin{enumerate}[\rm(i)]
\item Rigorous partition
  of unity separating
  endpoint and saddle;
\item Explicit bound
  $|C_P|\le C_S-\varepsilon$
  for some $\varepsilon>0$
  and all $|u|\ge u_0$;
  OR
\item Weierstrass-type
  argument showing the
  phase
  $\frac{2}{3}(|u|/3c_3)^{3/2}$
  is never a rational
  multiple of $\pi/4$
  (to avoid exact cancellation
  on a dense sequence).
\end{enumerate}
\end{conjecture}

\section{Numerical Evidence}
\label{sec:numerical}

\begin{proposition}[\Num]
\label{prop:num}
Numerical computation
($c_3=1$, $u\in[-30,-0.5]$,
$\beta\in[-50,50]$,
Gaussian cutoff):
\[
  |\Phi(u)|\approx|u|^{-0.53},
  \quad
  |F(U)|\approx U^{0.48}.
\]
Consistent with
Conjecture~\ref{conj:sharp};
not a proof.
\end{proposition}

\section{Consequence for Paper~XV}
\label{sec:consequence}

\begin{remark}[\Open]
\label{rem:consequence}
If Conjecture~\ref{conj:sharp}
holds (and carries over
to the PSWF setting),
then:
\begin{itemize}[nosep]
\item Assumption~ass:Phi-cubic
  of Paper~XV is false;
\item $S(c)=\Theta(c^{\theta_0})$;
\item Ces\`aro is the sharp
  bound;
\item no improvement is
  possible via oscillatory
  methods in this regime;
\item the correct goal
  of Paper~XVI shifts
  to: prove the lower
  bound $S(c)\ge Cc^{\theta_0}$,
  i.e.,
  Ces\`aro is \emph{optimal}.
\end{itemize}
\end{remark}
