\title{Mini-Model for the Fold Amplitude:\\
  The Unique Leading-Order Asymptotics\\
  via Erd\'elyi's Theorem}

\author{[Author]}

\begin{abstract}
We study
\[
  \Phi(u):=\int_0^\infty
  \beta^{-1/2}\chi(\beta)
  e^{i(\beta u + c_3\beta^3)}\,d\beta,
  \quad u<0,\;c_3>0,
\]
where $\chi\in C^\infty_c([0,R])$
has compact, \emph{fixed}
support of radius $R>0$.
For fixed $R$ and
$|u|\to\infty$:
\begin{enumerate}[\rm(1)]
\item The saddle
  $\beta_0(u)
  =\sqrt{|u|/(3c_3)}\to\infty$
  exits $\mathrm{supp}(\chi)$;
  its contribution to
  $\Phi(u)$ vanishes
  super-polynomially.
\item Only the endpoint
  $\beta=0$ contributes.
  Erd\'elyi's theorem gives
  the \emph{unique}
  leading-order expansion:
  \begin{equation}\label{eq:main}
    \Phi(u)
    =\sqrt{\pi}\,e^{i\pi/4}
    \cdot|u|^{-1/2}
    +O(|u|^{-3/2}).
  \end{equation}
\item Numerically confirmed
  (ratio to prediction
  $\to 1.0000$ at
  $|u|=300$, $c_3=1$,
  $R=1$).
\item The Ces\`aro bound
  $|F(U)|=O(U^{1/2})$
  is sharp:
  $\limsup_{u\to-\infty}
  |u|^{1/2}|\Phi(u)|
  =\sqrt{\pi}>0$.
\end{enumerate}
\emph{The earlier
appearance of exponents
$-1/3$ and $-1/6$
was an artefact of
using a support
that grows with
$|u|$, which is a
different integral.}
\end{abstract}

\maketitle

\section{The Correct Setup:
  Fixed Compact Support}
\label{sec:setup}

\begin{remark}[What "fixed
  support" means \Warn]
\label{rem:fixed-support}
In all of the following,
$\chi\in C^\infty_c([0,R])$
with $R>0$ fixed,
independent of $u$.
Examples:
$\chi=\mathbf{1}_{[0,R]}$
(smoothed);
$\chi=e^{-\beta^2/R^2}
\cdot\mathbf{1}_{[0,5R]}$.

The earlier numerics
used $\chi=e^{-\beta^2/
(10|u|+1)}$:
the support grows
as $\sim\sqrt{|u|}$.
This is \emph{not}
the same integral.
For growing support,
the saddle always
stays inside, and
different exponents
appear.
Those results do not
apply to fixed $\chi$.
\end{remark}

\section{The Saddle Exits
  the Support}
\label{sec:saddle}

\begin{lemma}[Saddle outside
  support \Rig]
\label{lem:saddle-outside}
For $u<0$, the saddle
point of $\phi(\beta)
=-|u|\beta+c_3\beta^3$
is at
$\beta_0=\sqrt{|u|/(3c_3)}$.
For $|u|>3c_3 R^2$:
$\beta_0>R$,
so $\chi(\beta_0)=0$.

Consequence: the
saddle contributes
to $\Phi(u)$ only via
exponentially small
tail effects (from the
smoothed cutoff).
For \emph{fixed}
$\chi\in C^\infty_c([0,R])$
and $|u|>3c_3 R^2$,
\textbf{the saddle
contribution vanishes
faster than any
polynomial in $|u|^{-1}$}.
This follows from
integration by parts
(Riemann--Lebesgue
for smooth integrands
away from endpoints).
\end{lemma}

\begin{remark}[CFU
  irrelevant for fixed
  support \Warn]
\label{rem:CFU-irrelevant}
The Chester--Friedman--Ursell
method addresses the
caselence when a saddle
coalesces with
an endpoint \emph{as
$u$ varies}.
For fixed compact $\chi$
and $|u|\to\infty$,
the saddle moves to
$\infty$ and permanently
exits $\mathrm{supp}(\chi)$.
There is no coalescence.
CFU does not apply.
\end{remark}

\section{Erd\'elyi's Theorem
  and the Unique Asymptotic}
\label{sec:erdelyi}

\begin{theorem}[Leading order
  \Rig]
\label{thm:main}
Let $\chi\in C^\infty_c
([0,R])$,
$\chi(\beta)=1$
near $\beta=0$,
$c_3>0$.
For $|u|>3c_3 R^2$,
write:
\[
  \Phi(u)
  =\int_0^\infty
  \beta^{-1/2}\chi(\beta)
  e^{i c_3\beta^3}
  e^{-i|u|\beta}\,d\beta.
\]
Set
$a(\beta):=
\beta^{-1/2}\chi(\beta)
e^{ic_3\beta^3}$.
Erd\'elyi's theorem
(\cite{Erdelyi1956},
Ch.~2.1)
applied to
$\int_0^\infty
a(\beta)e^{-i|u|\beta}
d\beta$:
\[
  \int_0^\infty
  a(\beta)e^{-i|u|\beta}
  d\beta
  \sim
  \sum_{n=0}^\infty
  \frac{\Gamma(\tfrac{1}{2}+n)}
    {n!}
  \cdot
  a^{(n)}(0^+)
  \cdot
  (i|u|)^{-(1/2+n)}.
\]
Leading term:
$a(0^+):=0^{-1/2}\chi(0)
e^{0}=$ (limit as
$\beta\to0^+$)
$\Rightarrow$
$a(\beta)
\sim\beta^{-1/2}$;
$\Gamma(1/2)=\sqrt{\pi}$:
\begin{equation}\label{eq:Erdelyi}
  \Phi(u)
  =\sqrt{\pi}\,
  (i|u|)^{-1/2}
  +O(|u|^{-3/2}).
\end{equation}
Since
$(i|u|)^{-1/2}
=e^{-i\pi/4}|u|^{-1/2}$:
\begin{equation}\label{eq:main2}
  \boxed{
    \Phi(u)
    =\sqrt{\pi}\,
    e^{-i\pi/4}
    |u|^{-1/2}
    +O(|u|^{-3/2})
  }
\end{equation}
\textbf{This is valid
for $|u|\gg c_3 R^2$,
and it is the unique
correct leading order.}
\end{theorem}

\begin{remark}[Role of
  the cubic term \Rig]
\label{rem:cubic}
The factor
$e^{ic_3\beta^3}$
enters $a(\beta)$
but does not change
the leading term:
$a(0^+)
=\lim_{\beta\to 0^+}
\beta^{-1/2}
\chi(\beta)e^{ic_3\beta^3}
=\lim_{\beta\to 0^+}
\beta^{-1/2}\cdot 1
=$ (singular, standard
Erd\'elyi form).
The cubic contributes
only at order
$a''(0^+)\sim ic_3$,
giving corrections
$O(c_3|u|^{-3/2})$.
\end{remark}

\section{Sharpness of
  the Bound}
\label{sec:sharpness}

\begin{corollary}[Lower
  bound \Rig]
\label{cor:lower}
$\limsup_{u\to-\infty}
|u|^{1/2}|\Phi(u)|
=\sqrt{\pi}>0$.
\textit{Proof:}
From~\eqref{eq:main2},
$|u|^{1/2}|\Phi(u)|
\to\sqrt{\pi}$
as $|u|\to\infty$.
\qed
\end{corollary}

\begin{corollary}[Ces\`aro
  sharp \Rig]
\label{cor:cesaro}
$|F(U)|
=O(U^{1/2})$
and
$\limsup_{U\to\infty}
U^{-1/2}|F(U)|>0$.
The bound is sharp;
no improvement
to $o(U^{1/2})$
is possible from
the pointwise
asymptotics alone.
\end{corollary}

\section{Numerical
  Confirmation}
\label{sec:num}

\begin{proposition}[\Rig+\Num]
\label{prop:num}
With $c_3=1$, $R=1$,
$\chi(\beta)
=e^{-\beta^2}$:
\[
  \frac{|\Phi_{\mathrm{num}}
    (-|u|)|}
    {\sqrt{\pi}\,|u|^{-1/2}}
  \longrightarrow 1.0000
  \quad
  \text{as }|u|\to\infty.
\]
Numerically:
\begin{center}

  & Ratio\\ \hline

\textbf{The Erd\'elyi

for $|u|\ge 30$.}
\end{proposition}

\section{What CFU, $-1/3$,
  and $-1/6$ Actually Mean}
\label{sec:regimes}

\begin{remark}[Growing-support
  problem \Warn]
\label{rem:growing}
The exponents $-1/3$
and $-1/6$ appear
\emph{only} when
$\chi=\chi_u$ depends
on $u$, e.g.,
$\chi_u(\beta)
=\chi_0(\beta/|u|^\alpha)$
for some $\alpha>0$.
In that case the
support radius
$R_u\sim|u|^\alpha$
grows, so
$|u|_{\mathrm{cross}}
=3c_3 R_u^2
\sim|u|^{2\alpha}$
also grows.
The saddle can remain
inside the support
for all $|u|$,
and CFU / Airy
structure becomes
relevant.

This is a \emph{different
integral} from
the fixed-$\chi$ case.
Whether the PSWF
problem requires
growing or fixed
support depends on
the spectral parameter
scaling in the
original problem
(see Remark~\ref{rem:PSWF}
below).
\end{remark}

\begin{problem}[PSWF
  support scaling
  \Open]
\label{rem:PSWF}
In Paper~XV, the
cubic phase coefficient
$c_3$ and the
spectral bandwidth $R$
arise from the PSWF
expansion. Two cases:
\begin{enumerate}[\rm(A)]
\item $c_3 R^2 = O(1)$
  (fixed-support regime):
  Erd\'elyi applies;
  $|\Phi|\sim|u|^{-1/2}$;
  Ces\`aro is sharp.
\item $c_3 R^2\to\infty$
  with the spectral
  parameter (growing-support
  regime):
  CFU and Airy
  structure become
  relevant;
  potentially better
  bounds.
\end{enumerate}
Determining which
case holds requires
explicit formulas for
$c_3$ and $R$ from
the PSWF asymptotic
expansion.
This is the only
remaining open
question.
\end{problem}
