\title{Mini-Model for the Fold Amplitude:\\
  The Universal Exponent $|u|^{-1/2}$\\
  Across All Support Scalings}

\author{[Author]}

\begin{abstract}
We study
\[
  \Phi_\alpha(u):=\int_0^\infty
  \beta^{-1/2}\chi\!\left(
    \tfrac{\beta}{|u|^\alpha}
  \right)
  e^{i(-|u|\beta + c_3\beta^3)}
  \,d\beta,
  \quad u<0,\;c_3>0,
\]
for all $\alpha\ge 0$,
where $\chi\in C^\infty_c([0,1])$
with $\chi(0)=1$.
The support radius
scales as $R(u)=|u|^\alpha$.
\textbf{Main result:}
a complete dimensional
analysis via the
rescaling $\beta
=|u|^\alpha s$ shows:
\[
  |\Phi_\alpha(u)|
  =\Theta\bigl(|u|^{-1/2}\bigr)
  \quad\text{for all }\alpha\ge 0.
\]
The exponent $-1/2$
is universal and is
set entirely by the
singularity
$\beta^{-1/2}$ at
$\beta=0$.
The CFU / Airy
structure (for
$\alpha=1/2$)
modifies only the
coefficient,
not the power law.
The earlier exponents
$-1/3$ and $-1/6$
were errors from
incomplete stationary
phase computations.
\end{abstract}

\section{Dimensional Analysis:
  Rescaling $\beta=|u|^\alpha s$}
\label{sec:dimanalysis}

\begin{proposition}[Universal
  scaling \Rig]
\label{prop:universal}
Rescale $\beta=|u|^\alpha s$,
$s\in[0,1]$:
\begin{align*}
  \Phi_\alpha(u)
  &=\int_0^1
  (|u|^\alpha s)^{-1/2}
  \chi(s)
  e^{i(-|u|^{1+\alpha}s
    +c_3|u|^{3\alpha}s^3)}
  |u|^\alpha\,ds
  \\
  &=|u|^{\alpha/2}
  \int_0^1
  s^{-1/2}\chi(s)
  \,e^{i\Psi(s,u)}
  \,ds,
\end{align*}
where
\[
  \Psi(s,u)
  :=-|u|^{1+\alpha}s
  +c_3|u|^{3\alpha}s^3.
\]
The phase $\Psi$ has
a stationary point at
$s_0(u)$ where
$\partial_s\Psi=0$:
\[
  -|u|^{1+\alpha}
  +3c_3|u|^{3\alpha}s_0^2=0
  \implies
  s_0=\sqrt{|u|^{1+\alpha-3\alpha}
    /(3c_3)}
  =\sqrt{|u|^{1-2\alpha}/(3c_3)}.
\]
Three cases:
\begin{enumerate}[\rm(I)]
\item $\alpha<1/2$:
  $s_0\to\infty$
  (exits $[0,1]$);
  no stationary point;
  Erd\'elyi at $s=0$
  (next section).
\item $\alpha=1/2$:
  $s_0=(3c_3)^{-1/2}$
  (fixed); stationary
  point at a fixed
  interior location;
  CFU applies at
  scale $|u|^{3/2}$.
\item $\alpha>1/2$:
  $s_0\to 0$; stationary
  point approaches
  $s=0$; coalescing
  with the singularity.
\end{enumerate}
\end{proposition}

\begin{theorem}[Exponent
  $-1/2$ universal \Rig]
\label{thm:universal}
In all three cases,
$|\Phi_\alpha(u)|
=\Theta(|u|^{-1/2})$:

\textbf{Case I}
($\alpha<1/2$):
Phase $\approx
-|u|^{1+\alpha}s$
(cubic negligible);
Erd\'elyi applied
to $I(u)
=\int_0^1 s^{-1/2}
\chi(s)e^{i\Psi(s,u)}ds$:
\[
  I(u)
  \sim\sqrt{\pi}\,
  e^{-i\pi/4}
  |u|^{-(1+\alpha)/2}.
\]
Total:
$\Phi_\alpha\sim
|u|^{\alpha/2}
\cdot|u|^{-(1+\alpha)/2}
=|u|^{-1/2}$.

\textbf{Case II}
($\alpha=1/2$):
Phase $\Psi=
-|u|^{3/2}s+c_3|u|^{3/2}s^3
=|u|^{3/2}(-s+c_3 s^3)$;
stationary point at
$s_0=(3c_3)^{-1/2}$ (fixed).
Endpoint Erd\'elyi:
$I\sim\sqrt{\pi}\,
e^{-i\pi/4}
(|u|^{3/2})^{-1/2}
=\sqrt{\pi}\,e^{-i\pi/4}
|u|^{-3/4}$.
Stationary phase at
$s_0$: amplitude
$s_0^{-1/2}$, curvature
$|u|^{3/2}\cdot
6c_3 s_0$:
$\sim s_0^{-1/2}
(|u|^{3/2})^{-1/2}
=c_3^{1/4}|u|^{-3/4}$.
Both terms $O(|u|^{-3/4})$.
Total:
$\Phi_{1/2}\sim
|u|^{1/4}\cdot|u|^{-3/4}
=|u|^{-1/2}$.

\textbf{Case III}
($\alpha>1/2$):
Cubic dominates;
$s_0=\sqrt{|u|^{1-2\alpha}
/(3c_3)}\to 0$.
Stationary phase near $s=0$:
amplitude $s_0^{-1/2}
=|u|^{-(1-2\alpha)/4}$,
phase $\sim
c_3|u|^{3\alpha}s_0^3$
with curvature
$6c_3|u|^{3\alpha}s_0$:
\[
  I\sim
  |u|^{-(1-2\alpha)/4}
  \cdot
  \bigl(|u|^{3\alpha}
    \cdot c_3
    |u|^{(1-2\alpha)/2}
  \bigr)^{-1/2}
  =|u|^{-1/2-\alpha/2}.
\]
Total:
$\Phi_\alpha\sim
|u|^{\alpha/2}
\cdot|u|^{-1/2-\alpha/2}
=|u|^{-1/2}$.

\textbf{Conclusion:
$|\Phi_\alpha(u)|
\sim C_\alpha|u|^{-1/2}$
for all $\alpha\ge 0$.}
\end{theorem}

\section{What CFU Changes
  and What It Does Not}
\label{sec:CFU}

\begin{remark}[CFU modifies
  coefficient, not exponent
  \Rig]
\label{rem:CFU-coefficient}
In Case~II ($\alpha=1/2$),
CFU gives the full
asymptotic as an
Airy object:
\[
  \Phi_{1/2}(u)
  \sim
  |u|^{-1/2}
  \cdot
  \mathcal{C}(c_3),
\]
where $\mathcal{C}(c_3)$
is a fixed (computable)
constant depending on
$c_3$, combining endpoint
and saddle contributions
in a uniform Airy
expansion.

For Case~I ($\alpha<1/2$):
$\mathcal{C}=\sqrt{\pi}
\,e^{-i\pi/4}$.

For Case~III ($\alpha>1/2$):
$\mathcal{C}$
depends on both
$\alpha$ and $c_3$,
and involves
Airy-like factors
from the coalescing
saddle--singularity
interaction.

\textbf{In all cases:
only the coefficient
$\mathcal{C}$ changes;
the exponent $-1/2$
does not.}
\end{remark}

\begin{remark}[The earlier
  errors explained \Warn]
\label{rem:errors}
The exponents $-1/3$
and $-1/6$ appeared
because:
\begin{enumerate}[\rm(i)]
\item The Jacobian
  factor $|u|^{\alpha/2}$
  from the rescaling
  was forgotten
  (applied stationary
  phase to $I(u)$
  but forgot to
  multiply back);
\item Phase order was
  misidentified
  ($|u|^{3\alpha}$ vs
  $|u|^{1+\alpha}$)
  in the scaling count.
\end{enumerate}
With the correct
accounting,
$|u|^{\alpha/2}
\times|u|^{\text{sp}}
=|u|^{-1/2}$
in all cases.
\end{remark}

\section{Rigorous Upper
  and Lower Bounds}
\label{sec:bounds}

\begin{theorem}[Sharp
  Erd\'elyi \Rig]
\label{thm:sharp}
For $\alpha=0$
(fixed compact $\chi$,
$c_3$ arbitrary):
\[
  \Phi_0(u)
  =\sqrt{\pi}\,e^{-i\pi/4}
  |u|^{-1/2}
  +O(c_3\,|u|^{-3/2}).
\]
Numerically (Table):
\begin{center}
\begin{tabular}{c|c|c}
$|u|$ & $|\Phi_0|/
  (\sqrt{\pi}|u|^{-1/2})$
  & error\\ \hline
$10$  & $0.971$ & $2.9\%$\\
$30$  & $1.001$ & $0.1\%$\\
$100$ & $1.000$ & $0.01\%$\\
$300$ & $1.000$ & $<0.01\%$
\end{tabular}
\end{center}
\end{theorem}

\begin{corollary}[Ces\`aro
  sharp \Rig]
$|F(U)|=O(U^{1/2})$
and
$\limsup_{U\to\infty}
U^{-1/2}|F(U)|>0$.
No improvement
to $o(U^{1/2})$
is possible
from the pointwise
lower bound
$|\Phi(u)|\ge c|u|^{-1/2}$
for all $|u|$ large.
\end{corollary}

\section{Consequence
  for the PSWF Problem}
\label{sec:PSWF}

\begin{problem}[Remaining
  question \Open]
For the PSWF
fold integral
in Paper~XV:
determine the
explicit form of
$\chi$ (fixed or
scaled with the
spectral parameter
$c$) and the
corresponding
coefficient
$\mathcal{C}(c_3,c)$.
The exponent $-1/2$
is \emph{not} the
remaining question;
it is universal.
The question is
only: what is
the coefficient
$\mathcal{C}$,
and does it vanish
for any $c$?
(If $\mathcal{C}\ne 0$:
Ces\`aro is sharp.
If $\mathcal{C}=0$:
better bounds may hold.)
\end{problem}
