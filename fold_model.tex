\title{Mini-Model for the Fold Amplitude:\\
  Stationary Phase, Singularity Competition,\\
  and the Genuine Dichotomy}

\author{[Author]}

\begin{abstract}
This note studies the model integral
\[
  \Phi(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u + c_3\beta^3)}\,d\beta,
  \qquad c_3>0\text{ fixed},
\]
and $F(U):=\int_0^U\Phi(u)\,du$.

Four errors from the previous
version are corrected:
\begin{enumerate}[\rm(1)]
\item The claim
  ``$\Phi(u)$ decays exponentially
  for $u>0$'' is replaced by
  the correct statement:
  $\Phi(u)=O(u^{-N})$
  for all $N$ by integration
  by parts (Riemann--Lebesgue),
  \emph{without} contour
  deformation (which would
  require controlling
  Stokes lines and the
  singularity at $\beta=0$).
\item The product ansatz
  $|\Phi(u)|\sim
  |\beta_0|^{-1/2}\cdot
  |\phi''(\beta_0)|^{-1/2}$
  is flagged as heuristic:
  the singularity at $\beta=0$
  and the saddle at
  $\beta_0\sim|u|^{1/2}$
  are \emph{not independent}
  and require a split-integral
  analysis.
\item The global conclusion
  ``cubic term worsens decay''
  is retracted.
\item The genuine dichotomy
  is stated:
  for $u<0$ and $|u|\to\infty$,
  does the singularity at $\beta=0$
  or the saddle at $\beta_0$
  dominate $\Phi(u)$?
  This determines whether
  $|\Phi(u)|\sim|u|^{-1/2}$
  or $|\Phi(u)|\sim|u|^{-1/3}$.
\end{enumerate}
\end{abstract}

\maketitle

\section{Setup}
\label{sec:setup}

\begin{definition}[Fold amplitude]
\label{def:Phi}
Fix $c_3>0$, $\chi\in C^\infty_c$
with $\chi\equiv 1$ near $0$.
\begin{equation}\label{eq:Phi}
  \Phi(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u+c_3\beta^3)}\,d\beta.
\end{equation}
Without cubic ($c_3=0$):
$\Phi_0(u)=O(|u|^{-1/2})$
and $F_0(U):=
\int_0^U\Phi_0(u)\,du=O(U^{1/2})$
(Ces\`aro baseline).
\end{definition}

\section{Critical Points and
  Stationary Phase}
\label{sec:sp}

\begin{lemma}[Critical points \Rig]
\label{lem:sp}
Phase: $\phi(\beta)=\beta u+c_3\beta^3$;
$\phi'(\beta)=u+3c_3\beta^2$.
\begin{itemize}[nosep]
\item $u>0$:
  $\phi'(\beta)>0$ for all
  $\beta\in\mathbb{R}$;
  \emph{no real critical point}.
\item $u<0$:
  two real critical points
  $\pm\beta_0$,
  $\beta_0:=\sqrt{|u|/(3c_3)}>0$.
\item $\phi''(\pm\beta_0)
  =\pm 6c_3\beta_0\ne 0$:
  non-degenerate saddles.
\end{itemize}
\end{lemma}

\begin{remark}[What follows from
  no critical point \Rig]
\label{rem:u-pos}
For $u>0$:
since $\phi'\ne 0$ on $\mathbb{R}$
and the amplitude
$|\beta|^{-1/2}\chi$ is in $L^1$
away from $\beta=0$,
repeated integration by parts
in~\eqref{eq:Phi}
(splitting the integral away from
$\beta=0$) gives:
\begin{equation}\label{eq:u-pos-bound}
  \Phi(u)=O(u^{-N})
  \text{ for all }N\ge 1,
\end{equation}
provided the amplitude
$(|\beta|^{-1/2}\chi)/(\phi')^N$
remains integrable after
$N$ integrations by parts.
This requires careful treatment
of the singularity at $\beta=0$;
for $N=1$ this gives
$O(u^{-1})$ from the
non-singular part,
but the singular part
near $\beta=0$ contributes
$O(u^{-1/2})$ regardless
(Riemann--Lebesgue).

\textbf{Corrected statement:}
$\Phi(u)=O(u^{-1/2})$
for all $u>0$
(from the singularity at $\beta=0$
dominates for bounded $u$;
for large $u$ the integration
by parts competes).
\emph{The claim ``exponentially
small'' from the previous
version is not justified}
without contour deformation,
which would require:
(a) verifying no Stokes line
crosses the singularity
$\beta=0$;
(b) controlling the
contribution of $|\beta|^{-1/2}$
along the complex contour.
This is an open problem.
\end{remark}

\section{The Singularity--Saddle
  Competition for $u<0$}
\label{sec:competition}

\begin{remark}[Why the product
  ansatz fails \Warn]
\label{rem:product-fails}
The previous version claimed:
$|\Phi(u)|\sim
|\beta_0|^{-1/2}\cdot
|\phi''(\beta_0)|^{-1/2}
=|u|^{-1/4}\cdot|u|^{-1/6}
=|u|^{-1/3?}$.

This product ansatz fails because:
\begin{enumerate}[\rm(a)]
\item Stationary phase applies
  when the amplitude is smooth
  near the critical point.
  Here the amplitude is
  $|\beta|^{-1/2}$, which is
  \emph{singular at $\beta=0$}.
\item The critical point
  $\beta_0=\sqrt{|u|/(3c_3)}
  \to\infty$ as $|u|\to\infty$;
  for large $|u|$, $\beta_0$
  is far from $\beta=0$
  and the product ansatz
  may be valid.
  But for moderate $|u|\sim 1$,
  the saddle and singularity
  are both at scale $O(1)$
  and interact.
\item The correct procedure is
  to \emph{split the integral}:
  \[
    \Phi(u)=\int_0^{\beta_0/2}
    +\int_{\beta_0/2}^\infty
    +\int_{-\infty}^0 \cdots
  \]
  and treat each part
  separately.
\end{enumerate}
\end{remark}

\begin{proposition}[Split-integral
  analysis \Open]
\label{prop:split}
For $u<0$, $|u|\gg 1$,
define $\beta_0=\sqrt{|u|/(3c_3)}$.
Split:
\begin{equation}\label{eq:split}
  \Phi(u)
  = \underbrace{\int_0^{\beta_0/2}
    |\beta|^{-1/2}e^{i\phi(\beta)}
    d\beta}_{=:A(u)\;\text{singularity-dominated}}
  +\underbrace{\int_{\beta_0/2}^{2\beta_0}
    |\beta|^{-1/2}e^{i\phi(\beta)}
    d\beta}_{=:B(u)\;\text{saddle-dominated}}
  + \text{tail}.
\end{equation}
\begin{enumerate}[\rm(i)]
\item \textbf{Part $A(u)$}
  (singularity near $\beta=0$):
  on $[0,\beta_0/2]$,
  $\phi'(\beta)=u+3c_3\beta^2
  \approx u$ (constant $\approx u<0$);
  integration by parts gives:
  \[
    A(u)\approx
    \frac{e^{i\phi(0)}}{iu}
    \int_0^{\beta_0/2}
    \frac{d}{d\beta}
    [|\beta|^{-1/2}e^{i\phi(\beta)}]
    \cdots
  \]
  Leading term:
  $|A(u)|=O(|u|^{-1/2})$
  (from the singularity
  at $\beta=0$).
\item \textbf{Part $B(u)$}
  (near saddle $\beta_0$):
  on $[\beta_0/2,2\beta_0]$,
  amplitude $\sim\beta_0^{-1/2}
  =O(|u|^{-1/4})$;
  $|\phi''(\beta_0)|
  =6c_3\beta_0=O(|u|^{1/2})$;
  standard stationary phase:
  \[
    |B(u)|\sim
    \beta_0^{-1/2}\cdot
    |\phi''(\beta_0)|^{-1/2}
    =|u|^{-1/4}\cdot|u|^{-1/4}
    =|u|^{-1/2}.
  \]
\end{enumerate}
Both parts give $O(|u|^{-1/2})$.

\textbf{Tentative conclusion:}
$|\Phi(u)|=O(|u|^{-1/2})$
for $u<0$, $|u|\to\infty$:
the cubic term does
\emph{not} change the
leading power-law decay.
\emph{This is an open computation;
the constants and
oscillatory cancellation
between $A$ and $B$
are not determined here.}
\end{proposition}

\begin{remark}[Corrected exponent
  \Warn]
\label{rem:corrected-exponent}
The previous version claimed
$|\Phi(u)|\sim|u|^{-1/3}$
(from the erroneous product
ansatz $|u|^{-1/6}\cdot|u|^{-1/6}$).
The split-integral analysis
suggests instead
$|\Phi(u)|=O(|u|^{-1/2})$
(same as without the cubic term).

If this is correct, the cubic
term does not change the
leading decay rate,
only the oscillatory
phase and sub-leading constants.
The Ces\`aro bound
$F(U)=O(U^{1/2})$ would then
not be improved by the
cubic structure.

\textbf{This is the key question
for Paper~XVI.}
\end{remark}

\section{The Genuine Dichotomy}
\label{sec:dichotomy}

\begin{problem}[Singularity vs.
  saddle: the genuine dichotomy
  \Open]
\label{prob:dichotomy}
For $u<0$, $|u|\to\infty$:
which term dominates
in~\eqref{eq:split}?
\begin{enumerate}[\rm(A)]
\item \textbf{Singularity wins:}
  $|\Phi(u)|\sim C|u|^{-1/2}$;
  cubic term irrelevant at
  leading order;
  $|F(U)|=\Theta(U^{1/2})$;
  Ces\`aro is sharp;
  \emph{no Airy improvement}.
\item \textbf{Saddle wins:}
  $|\Phi(u)|\sim C|u|^{-?}$
  with exponent $<1/2$;
  possible improvement over
  Ces\`aro;
  \emph{Airy regime applies}.
\item \textbf{Cancellation between
  $A(u)$ and $B(u)$:}
  both contribute $O(|u|^{-1/2})$
  but with opposite sign;
  $|\Phi(u)|=o(|u|^{-1/2})$;
  this would be the
  most favourable case,
  giving genuine improvement.
\end{enumerate}
The correct answer depends on:
\begin{itemize}[nosep]
\item the exact phase
  $e^{i\phi(\beta_0)}$
  at the saddle;
\item whether $e^{i\phi(0)}$
  (singularity contribution)
  and $e^{i\phi(\beta_0)}$
  (saddle contribution)
  add or cancel;
\item the sub-leading structure
  of $\phi$ near $\beta=0$.
\end{itemize}
This is the \emph{single
critical computation}
for Paper~XVI.
\end{problem}

\begin{remark}[What we actually
  know \Rig]
\label{rem:what-we-know}
The following is rigorous:
\begin{enumerate}[\rm(1)]
\item $\Phi\in L^2(\mathbb{R})$;
  $|\Phi(u)|=O(|u|^{-1/2})$
  (from $|\cdot|^{-1/2}\chi
  \in L^2$, Plancherel;
  holds with or without cubic).
\item For $u>0$:
  no real critical point;
  $\Phi(u)=O(u^{-1/2})$
  from Riemann--Lebesgue;
  the question of whether
  $\Phi(u)$ decays faster
  is open (requires contour
  analysis).
\item For $u<0$:
  two real critical points;
  the leading behaviour is
  $O(|u|^{-1/2})$ from both
  the singularity and the
  saddle (split analysis above);
  whether they cancel
  or reinforce is the
  open problem.
\item $F(U)=\int_0^U\Phi(u)\,du
  =O(U^{1/2})$
  (rigorous upper bound;
  from $|\Phi|=O(u^{-1/2})$).
\item Whether $F(U)=o(U^{1/2})$
  is equivalent to
  $\Phi(u)=o(u^{-1/2})$
  and is entirely open.
\end{enumerate}
\end{remark}

\section{Consequence for Paper~XV
  and Paper~XVI}
\label{sec:consequence}

\begin{remark}[Revised status
  of Assumption~ass:Phi-cubic]
\label{rem:revised}
The mini-model shows:
\begin{itemize}[nosep]
\item The cubic term in the
  phase does NOT automatically
  improve the decay of $\Phi$.
\item The leading order
  $|\Phi(u)|=O(|u|^{-1/2})$
  is preserved under the
  cubic perturbation
  (from both singularity
  and saddle, each contributing
  $O(|u|^{-1/2})$).
\item Any improvement requires
  cancellation between
  singularity and saddle
  contributions.
  This cancellation is
  \emph{not guaranteed}
  and depends on the
  exact phase values.
\end{itemize}
For Paper~XV:
the Ces\`aro bound
$S(c)=o(c^{\theta_0})$ remains
the best unconditional result.

For Paper~XVI:
the task is to compute
exactly whether the
cancellation in
Problem~\ref{prob:dichotomy}
case~(C) occurs,
i.e., whether
$|A(u)+B(u)|=o(|u|^{-1/2})$.
\emph{This is the precise
one-integral computation
that decides the entire programme.}
\end{remark}

\begin{problem}[The deciding
  integral \Open]
\label{prob:deciding}
Compute explicitly:
\begin{equation}\label{eq:deciding}
  \Phi(u)
  =\int_{-\infty}^\infty
  |\beta|^{-1/2}\chi(\beta)
  e^{i(\beta u+c_3\beta^3)}\,d\beta
\end{equation}
for $u<0$, $|u|\to\infty$,
as an asymptotic series.
Specifically:
\begin{enumerate}[\rm(i)]
\item Compute the phase values
  $e^{i\phi(0)}=e^{0}=1$
  (singularity contribution)
  and $e^{i\phi(\beta_0)}
  =e^{i(\beta_0 u+c_3\beta_0^3)}$
  (saddle contribution).
\item Determine the relative
  sign / phase between
  $A(u)$ and $B(u)$.
\item Decide:
  \begin{itemize}[nosep]
  \item If $|A(u)+B(u)|
    =\Theta(|u|^{-1/2})$:
    Ces\`aro is sharp;
    the programme stalls.
  \item If $|A(u)+B(u)|
    =o(|u|^{-1/2})$:
    genuine improvement
    exists; characterise
    the exponent.
  \end{itemize}
\end{enumerate}
Note: $\phi(\beta_0)
=\beta_0 u+c_3\beta_0^3$
$=\sqrt{|u|/(3c_3)}\cdot(-|u|)
+c_3(|u|/(3c_3))^{3/2}$
$=-(2/3)\sqrt{|u|^3/(3c_3)}$
(classical cubic phase value);
this is real and negative,
so $e^{i\phi(\beta_0)}
=e^{-i\frac{2}{3}
\sqrt{|u|^3/(3c_3)}}$
oscillates rapidly for large $|u|$.
The phase of the singularity
contribution is $e^{i\phi(0)}=1$
(real, positive).
\textbf{They do not cancel
automatically:}
their relative phase is
$e^{-i\frac{2}{3}(|u|/3c_3)^{3/2}}$,
which oscillates.
The asymptotic cancellation
depends on the coefficient
structure.
\end{problem}
