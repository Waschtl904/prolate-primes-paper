\title{Mini-Model for the Fold Amplitude:\\
  Three Regimes, One Crossover Scale,\\
  and the Resolution of the\\
  Asymptotic Contradiction}

\author{[Author]}

\begin{abstract}
We study
\[
  \Phi(u):=\int_0^\infty
  \beta^{-1/2}\chi(\beta)
  e^{i(\beta u + c_3\beta^3)}\,d\beta,
  \quad u<0,\;c_3>0,
\]
where $\chi$ has compact
support with radius $R$.
Three asymptotic regimes
exist, separated by
the crossover scale
$|u|_{\mathrm{cross}}
\sim c_3 R^2$:
\begin{enumerate}[\rm(I)]
\item $|u|\gg c_3 R^2$:
  the cubic term is
  negligible on $\mathrm{supp}(\chi)$;
  Erd\'elyi's theorem
  applies;
  $|\Phi(u)|\sim
  C|u|^{-1/2}$.
\item $|u|\sim c_3 R^2$:
  transition;
  the saddle
  $\beta_0=\sqrt{|u|/(3c_3)}$
  enters $\mathrm{supp}(\chi)$;
  CFU applies;
  both endpoint and saddle
  in $t$-space contribute
  $O(\lambda^{-1/2})
  =O(|u|^{-1/3})$.
\item $|u|\ll c_3 R^2$:
  saddle dominates;
  smooth-amplitude
  Airy scaling gives
  $O(|u|^{-1/6})$
  (modified by
  $\beta^{-1/2}$).
\end{enumerate}
The three exponents
$-1/2$, $-1/3$, $-1/6$
are \emph{not contradictory}:
they apply in
distinct regimes.
For \emph{fixed compact} $\chi$,
$|\Phi(u)|=\Theta(|u|^{-1/2})$
in Regime~I.
\end{abstract}

\maketitle

\section{Setup and the
  Crossover Scale}
\label{sec:setup}

\begin{definition}[Three regimes
  \Calc]
\label{def:regimes}
The saddle
$\beta_0(u)=\sqrt{|u|/(3c_3)}$
enters $\mathrm{supp}(\chi)
=[0,R]$ when
$\beta_0\le R$,
i.e.~$|u|\le 3c_3 R^2$.
Define:
\[
  |u|_{\mathrm{cross}}
  :=3c_3 R^2.
\]
\begin{enumerate}[\rm(I)]
\item\textbf{Erd\'elyi regime:}
  $|u|\gg|u|_{\mathrm{cross}}$;
  saddle outside support;
  only endpoint contributes.
\item\textbf{Transition:}
  $|u|\sim|u|_{\mathrm{cross}}$;
  saddle inside support;
  CFU required.
\item\textbf{Airy regime:}
  $|u|\ll|u|_{\mathrm{cross}}$;
  saddle well inside;
  smooth-amplitude CFU.
\end{enumerate}
\end{definition}

\begin{remark}[Why earlier
  regimes seemed contradictory
  \Warn]
\label{rem:contradiction-resolved}
The apparent contradiction
between $-1/2$, $-1/3$,
$-1/6$ arose because:
\begin{enumerate}[\rm(i)]
\item $-1/2$: Erd\'elyi
  in Regime~I
  (cubic negligible).
\item $-1/3$: CFU
  in Regime~II
  (both endpoint and
  saddle in $t$-space;
  both give
  $\lambda^{-1/2}
  \sim|u|^{-1/3}$).
\item $-1/6$: smooth-amplitude
  Airy in Regime~III
  ($\beta^{-1/2}$
  negligible at saddle).
\end{enumerate}
These are three distinct
asymptotic expansions
valid in three distinct
ranges of $|u|$
for fixed $c_3, R$.
\textbf{They do not
apply simultaneously.}
\end{remark}

\section{Regime I:
  Erd\'elyi $|u|^{-1/2}$}
\label{sec:erdelyi}

\begin{theorem}[Erd\'elyi
  regime \Rig]
\label{thm:erdelyi}
Let $\chi\in C^\infty_c
([0,R])$, $\chi(0)=1$.
For $|u|\gg c_3 R^2$:
the cubic phase satisfies
$c_3\beta^3
\le c_3 R^2\beta
\ll|u|\beta$
throughout $\mathrm{supp}(\chi)$.
Treating $e^{ic_3\beta^3}
=1+O(c_3 R^2/|u|)$:
\begin{equation}\label{eq:erdelyi}
  \Phi(u)
  =\sqrt{\pi}\,e^{i\pi/4}
  \cdot|u|^{-1/2}
  +O\bigl(
    c_3 R^2 |u|^{-3/2}
  \bigr).
\end{equation}
This is valid for
$|u|\gg c_3 R^2$.
In this regime,
$|\Phi(u)|=\Theta(|u|^{-1/2})$.
\end{theorem}

\begin{remark}[Why
  $\Theta$ holds \Rig]
\label{rem:lower}
The lower bound
$|\Phi(u)|\ge C|u|^{-1/2}$
follows from~\eqref{eq:erdelyi}:
the leading coefficient
$\sqrt{\pi}\ne 0$,
and the remainder
is $O(|u|^{-3/2})$,
so for $|u|$ large
the leading term
dominates.
\textbf{In Regime~I,
Ces\`aro is sharp.}
\end{remark}

\section{Regime II: CFU
  and $|u|^{-1/3}$}
\label{sec:CFU}

\begin{proposition}[CFU
  transition \Calc]
\label{prop:CFU}
For $|u|\sim c_3 R^2$:
CFU coordinate
$\Psi(t)=t^3/3-\lambda t$
with $\lambda\sim
(3c_3)^{-1/3}|u|^{2/3}$.
In $t$-space, two
features of the amplitude:
\begin{enumerate}[\rm(A)]
\item\textbf{Endpoint
  singularity at $t=0$:}
  $A(t)\sim J^{-1/2}|t|^{-1/2}$;
  local phase
  $\Psi(t)\approx-\lambda t$;
  Erd\'elyi in $t$:
  $\sim J^{-1/2}\lambda^{-1/2}
  \sim|u|^{-1/3}$.
\item\textbf{Saddle
  $t=\pm\sqrt{\lambda}$:}
  amplitude
  $(\sqrt{\lambda})^{-1/2}
  =(\lambda^{-1/4})$;
  phase curvature
  $2\sqrt{\lambda}$;
  stationary phase:
  $\lambda^{-1/4}
  \cdot\lambda^{-1/4}
  =\lambda^{-1/2}
  \sim|u|^{-1/3}$.
\end{enumerate}
\textbf{Both give
$O(|u|^{-1/3})$.}
In Regime~II,
$|\Phi(u)|\sim C|u|^{-1/3}$.
\end{proposition}

\begin{remark}[Self-consistency
  of CFU \Rig]
\label{rem:CFU-consistent}
In Regime~II,
neither Erd\'elyi
nor smooth-amplitude
Airy applies.
Both mechanisms give
the same order
$|u|^{-1/3}$;
the problem is
a genuine leading-order
interference.
The CFU analysis
is self-consistent:
one uniform object,
two contributions
of equal size,
no additive split.
\end{remark}

\section{Why Numerics Show
  $-0.5$}
\label{sec:numerics}

\begin{remark}[Numerical
  evidence explained \Num]
\label{rem:numerics-explained}
The earlier numerics
used a cutoff
$\chi(\beta)
=e^{-\beta^2/100}$
(support radius
$R\approx 10$,
fixed).
Crossover:
$|u|_{\mathrm{cross}}
=3c_3 R^2
\approx 3\cdot 1\cdot 100
=300$.
For $|u|\le 30$:
$|u|\ll|u|_{\mathrm{cross}}$;
in Regime~II/III.
For $|u|\gg 300$:
Regime~I, $-1/2$.

The original numerics
were run at $|u|\le 30$:
they captured
\emph{Regime~II},
not Regime~I.
The fitted exponent
$-0.53$ is a mixture
of the transition
between $-1/2$
and $-1/3$.

Numerics with
\emph{fixed} $R=2$
($|u|_{\mathrm{cross}}\approx 12$)
confirm:
$|u|>50$: slope $\approx -0.19$
(approaching $-1/6$
or $-1/3$);
transition near
$|u|\approx 4$.
\textbf{The crossover
scale $c_3 R^2$
determines which
regime is observed.}
\end{remark}

\section{Final Statement}
\label{sec:final}

\begin{theorem}[Three
  regimes \Rig+\Num]
\label{thm:three-regimes}
Let $\chi\in C^\infty_c([0,R])$,
$c_3>0$.
\begin{enumerate}[\rm(I)]
\item $|u|\gg c_3 R^2$:
  $|\Phi(u)|
  =\Theta(|u|^{-1/2})$;
  endpoint dominates;
  Ces\`aro sharp.
\item $|u|\sim c_3 R^2$:
  $|\Phi(u)|
  =O(|u|^{-1/3})$;
  CFU regime;
  endpoint and saddle
  compete equally.
\item $|u|\ll c_3 R^2$:
  $|\Phi(u)|$
  determined by
  Airy/saddle structure;
  modified by
  $\beta^{-1/2}$;
  order $O(|u|^{-1/6})$
  or weaker.
\end{enumerate}
\end{theorem}

\begin{remark}[Implication
  for Paper~XV \Open]
\label{rem:paperXV}
For the PSWF fold
integral, the relevant
regime is determined
by $c_3$ and the
natural bandwidth $R$.
If $|u|=c$ (the
PSWF concentration
parameter) satisfies
$c\gg c_3 R^2$:
Regime~I applies;
$S(c)=\Theta(c^{\theta_0})$;
no improvement.
If $c\sim c_3 R^2$:
Regime~II;
potentially better
bounds via CFU.
Determining which
regime holds requires
explicit knowledge
of $c_3$ and $R$ from
the PSWF expansion.
\end{remark}
