\title{Mini-Model for the Fold Amplitude:\\
  Explicit CFU Map and the\\
  Location of the Endpoint Singularity\\
  in Normal-Form Coordinates}

\author{[Author]}

\begin{abstract}
We study
\[
  \Phi(u):=\int_0^\infty
  \beta^{-1/2}\chi(\beta)
  e^{i(\beta u + c_3\beta^3)}\,d\beta,
  \quad u<0,\;c_3>0.
\]
The Chester--Friedman--Ursell (CFU)
normal-form map is computed
explicitly.
\textbf{Main result:}
under the CFU map
$\phi(\beta)\mapsto
\frac{1}{3}t^3-\lambda t$,
the endpoint $\beta=0$
maps to $t_*=0$.
The Airy turning points
are at $t=\pm\sqrt{\lambda}$
with $\lambda\sim
(3c_3)^{-1/3}|u|^{2/3}\to\infty$.
Therefore:
\begin{enumerate}[\rm(1)]
\item $t_*=0$ is a
  \emph{regular point}
  of the normal form
  (not a turning point);
  the CFU reduction is
  \emph{not obstructed}.
\item The singular
  amplitude $\beta^{-1/2}$
  becomes a singularity
  in $t$ at $t_*=0$;
  its order is determined
  by the Jacobian
  $d\beta/dt$ at $t_*=0$.
\item The resulting
  $t$-integral has two
  contributions: Airy
  turning points at
  $t=\pm\sqrt{\lambda}$
  (saddle) and the
  interior singularity
  at $t_*=0$ (endpoint).
  These must be analysed
  jointly.
\end{enumerate}
\end{abstract}

\maketitle

\section{The CFU Map: Explicit Computation}
\label{sec:CFU-explicit}

\begin{definition}[Phase and
  critical points \Calc]
\label{def:phase}
For $u<0$, write $u=-|u|$:
\[
  \phi(\beta):=
  -|u|\beta+c_3\beta^3.
\]
Critical points
($\phi'=0$):
\[
  \beta_\pm
  =\pm\sqrt{|u|/(3c_3)}.
\]
Phase values:
\begin{align*}
  \phi(\beta_+)
  &=-\tfrac{2\sqrt{3}}{9}
    \frac{|u|^{3/2}}{\sqrt{c_3}},
  \\
  \phi(\beta_-)
  &=+\tfrac{2\sqrt{3}}{9}
    \frac{|u|^{3/2}}{\sqrt{c_3}}.
\end{align*}
Note: $\phi(0)=0$.
\end{definition}

\begin{definition}[CFU normal
  form \Calc]
\label{def:CFU-normalform}
The CFU normal form
(Chester--Friedman--Ursell
1957) associates to
$\phi$ the canonical
function
\[
  \Psi(t):=
  \tfrac{1}{3}t^3
  -\lambda t,
\]
where $\lambda>0$
is determined by:
\begin{equation}\label{eq:lambda-def}
  \tfrac{4}{3}\lambda^{3/2}
  =\phi(\beta_-)-\phi(\beta_+)
  =\tfrac{4\sqrt{3}}{9}
    \frac{|u|^{3/2}}{\sqrt{c_3}}.
\end{equation}
Solving:
\begin{equation}\label{eq:lambda}
  \lambda
  =\Bigl(
    \frac{3^{2/3}}{3c_3^{1/3}}
  \Bigr)|u|^{2/3}
  \sim (3c_3)^{-1/3}|u|^{2/3}.
\end{equation}
\textbf{$\lambda\to+\infty$
as $|u|\to\infty$};
the turning points
$t=\pm\sqrt{\lambda}\to\infty$.
\end{definition}

\begin{proposition}[Image of
  $\beta=0$ \Calc]
\label{prop:beta0-image}
Under the CFU map
$t=t(\beta)$ defined by
$\Psi(t(\beta))=\phi(\beta)$,
the image of $\beta=0$
satisfies:
\[
  \Psi(t_*)
  =\phi(0)=0
  \implies
  \tfrac{1}{3}t_*^3
  -\lambda t_*=0
  \implies
  t_*\bigl(t_*^2-3\lambda\bigr)=0.
\]
Solutions:
\[
  t_*=0,
  \quad
  t_*=\pm\sqrt{3\lambda}.
\]
By continuity of the
CFU map
(the map $\beta\mapsto t$
is smooth and monotone
near $\beta=0$ for
$|u|\gg c_3$):
\begin{equation}\label{eq:t-star}
  \boxed{\beta=0
  \longmapsto t_*=0.}
\end{equation}
The other solutions
$t_*=\pm\sqrt{3\lambda}$
correspond to
$\beta$-values away
from $0$.
\end{proposition}

\begin{corollary}[$t_*=0$ is
  not a turning point \Calc]
\label{cor:regular}
The Airy turning points
of $\Psi$ are at
$t=\pm\sqrt{\lambda}$.
Since $\lambda>0$,
we have $0\ne\pm\sqrt{\lambda}$.
Therefore:
\[
  t_*=0
  \text{ is a }
  \emph{regular point}
  \text{ of }
  \Psi(t).
\]
Consequence:
\textbf{the CFU reduction
is not obstructed by
the endpoint singularity.}
The singularity at
$t_*=0$ in the
$t$-integral is an
interior singularity
at a non-critical
location.
\end{corollary}

\section{Structure of the
  $t$-Integral}
\label{sec:t-integral}

\begin{proposition}[Singular
  amplitude in $t$-coordinates
  \Calc+\Warn]
\label{prop:t-amplitude}
After the CFU change
of variables $\beta\mapsto t$,
\[
  \Phi(u)=\int
  \beta(t)^{-1/2}
  \chi(\beta(t))
  \Bigl|\frac{d\beta}{dt}\Bigr|
  e^{i\Psi(t)-i\lambda t}
  \,dt
  =\int
  A(t)\cdot
  e^{i\Psi(t)}\,dt,
\]
where the new amplitude
is
\[
  A(t)
  =\beta(t)^{-1/2}
  \chi(\beta(t))
  \Bigl|\frac{d\beta}{dt}\Bigr|.
\]
At $t=t_*=0$:
$\beta(0)=0$, so
$A(t)\sim C|t|^{-\gamma}$
for some $\gamma>0$
determined by the
Jacobian.

To find $\gamma$:
near $\beta=0$,
$\phi(\beta)\approx
-|u|\beta$ and
$\Psi(t)\approx-\lambda t$,
so $t\approx|u|\beta/\lambda
=3c_3^{1/3}|u|^{1/3}\beta$.
Thus
$\beta\sim
C_J\lambda^{-1}|u|t
=(3c_3)^{1/3}|u|^{-1/3}t$,
and the Jacobian
$d\beta/dt\sim
(3c_3)^{1/3}|u|^{-1/3}$
(bounded).
Therefore:
\[
  \beta(t)^{-1/2}
  \Bigl|\frac{d\beta}{dt}\Bigr|
  \sim
  C\cdot
  \bigl((3c_3)^{1/3}
    |u|^{-1/3}t
  \bigr)^{-1/2}
  \cdot (3c_3)^{1/3}|u|^{-1/3}
  =C'|u|^{-1/6}|t|^{-1/2}.
\]
So \emph{the singularity
in $t$-space is}
$|t|^{-1/2}$---same
type as in $\beta$-space
---but now at an
interior point $t_*=0$
of the $t$-integral,
with an additional
amplitude factor
$|u|^{-1/6}$.
\end{proposition}

\begin{remark}[The two
  contributions in
  $t$-space \Calc]
\label{rem:two-contributions}
The $t$-integral now
has two distinct features:
\begin{enumerate}[\rm(A)]
\item\textbf{Airy turning
  points} at
  $t=\pm\sqrt{\lambda}
  \sim|u|^{1/3}$:
  these give the standard
  stationary phase
  contribution;
  the Airy function
  $\mathrm{Ai}(-\lambda)$
  for $\lambda\to\infty$
  decays as
  $\lambda^{-1/4}
  \sim|u|^{-1/12}$.
\item\textbf{Interior
  singularity} at $t_*=0$:
  the singular weight
  $|u|^{-1/6}|t|^{-1/2}$
  gives a Fourier-type
  contribution
  $\sim|u|^{-1/6}
  \cdot\lambda^{-1/2}
  \sim|u|^{-1/6}
  \cdot|u|^{-1/3}
  =|u|^{-1/2}$.
\end{enumerate}
\textbf{Both give the
same order
$O(|u|^{-1/2})$ in
$\Phi(u)$.}
This is consistent with
the Erd\'elyi upper
bound but now with
a clear mechanism
for both contributions
visible in $t$-space.
\end{remark}

\section{Consequence:
  Is CFU Sufficient?}
\label{sec:consequence}

\begin{remark}[What is now
  rigorous vs.\ open
  \Rig+\Open]
\label{rem:rigorous-vs-open}
\begin{enumerate}[\rm(1)]
\item\textbf{Rigorous:}
  $\lambda\sim(3c_3)^{-1/3}
  |u|^{2/3}$ (computed).
  $t_*=0$ is a regular
  point of $\Psi$
  (computed).
  $A(t)\sim C'|u|^{-1/6}
  |t|^{-1/2}$ near $t=0$
  (computed to leading
  order from Jacobian).
\item\textbf{Heuristic:}
  the Airy contribution
  (A) gives $O(|u|^{-1/12}
  \cdot\lambda^{-1/4})
  =O(|u|^{-1/3})$
  to $\Phi$;
  the interior singularity
  (B) gives $O(|u|^{-1/2})$.
  So contribution~(B)
  dominates asymptotically.
\item\textbf{Open:}
  rigorous asymptotic
  expansion of the
  $t$-integral combining
  Airy turning points
  and interior singular
  weight;
  this requires the
  theory of oscillatory
  integrals with
  coexisting turning
  points and singularities
  (Temme 1985;
  Kaminski 1994).
\end{enumerate}
\end{remark}

\begin{conjecture}[\Open]
\label{conj:sharp}
$\limsup_{u\to-\infty}
|u|^{1/2}|\Phi(u)|>0$.

Basis: contribution~(B)
(interior singularity
at $t_*=0$ in $t$-space)
has amplitude
$\sim C'|u|^{-1/2}$
with $C'=C'(c_3)\ne 0$
(the Jacobian does not
vanish at $t=0$).
If the $t$-integral
near $t=0$ is dominated
by this singularity,
then Ces\`aro is sharp.

This is \emph{not proved}:
the full $t$-integral
analysis is required.
\end{conjecture}

\section{Numerical Evidence}
\label{sec:num}

\begin{proposition}[\Num]
$|\Phi(u)|\approx|u|^{-0.53}$;
$|F(U)|\approx U^{0.48}$.
Consistent with
$O(|u|^{-1/2})$
dominating over
the Airy contribution
$O(|u|^{-1/3})$
(which would give
a larger exponent).
\end{proposition}
