\title[Completeness of the Limiting Slepian Eigenbasis]{%
  Completeness of the Limiting Slepian Eigenbasis in $L^2(\mathbb{R})$:
  A Classification of Stability Mechanisms}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
Let $\Hlim$ denote the bounded self-adjoint SOT-limit operator
of Paper~X and $\{\Phiinf{n}\}_{n \ge 0}$ the orthonormal
system of limit eigenfunctions of Paper~XI.
Define $\Kspace := \overline{\mathrm{span}}\{\Phiinf{n}\}$.
This paper classifies the mechanisms by which the completeness
problem $\Kspace = L^2(\mathbb{R})$ (Open Problem~7 of Paper~IX)
could be resolved.

\medskip
For finite $c$, the classical PSWFs $\{\Phic{c}{n}\}_{n \ge 0}$
form a complete orthonormal basis of $\PW_c^2$.
The central difficulty of the series is that this basis property
is not automatically stable under the SOT-limit $c \to \infty$:
SOT-convergence controls neither the spanning structure
nor the completeness of the limit system
(Paper~X, Remark; Reed--Simon \cite{ReedSimon1975}).
Completeness of $\{\Phiinf{n}\}$ is therefore
a \emph{stability problem for PSWF spanning structures
under the SOT-limit}.

\medskip
\textbf{Main results.}
We prove three structural theorems:
\begin{itemize}
\item \emph{(Theorem~A, Route~A.)} For any orthonormal system,
  the Parseval identity is logically equivalent to completeness.
  There is no frame-based shortcut.
\item \emph{(Theorem~B, Route~B.)} The completeness problem
  is equivalent to the family of local spanning conditions
  $\PW_{c_0}^2 \subseteq \Kspace$ for all $c_0 > 0$.
  This is a \emph{localisation theorem}, not a reduction:
  global completeness is rewritten as a stability
  question about individual Paley--Wiener bands.
\item \emph{(Theorem~C, Route~C.)} Under a compact-resolvent
  hypothesis, completeness follows from the spectral theorem;
  this route is structurally independent of Routes~A+B.
\end{itemize}

\medskip
The strictly weaker \emph{local spanning problem}
--- does $\{\Phiinf{n}\}$ span $\PW_{c_0}^2$
for a \emph{single} fixed $c_0$? ---
is identified as the genuine analytic open core
(Open Problem~\ref{prob:local-span}).
The coefficient-stability hypothesis H1
is the only statement in the paper strictly weaker
than completeness that is directly supported
by the series (Papers~VIII and~XII).
\end{abstract}

\medskip
\noindent
The Slepian concentration operator is
$(\Hc f)(x) = \int_{-\infty}^\infty \frac{\sin c(x-y)}{\pi(x-y)}
f(y)\,dy$.
For each finite $c$, $\{\Phic{c}{n}\}$ is a complete
orthonormal basis of $\PW_c^2$ \cite{Slepian1961}.
Paper~X establishes $\Hlim := \mathrm{SOT\text{-}lim}\,\Hc$
as a bounded self-adjoint operator, $\norm{\Hlim} \le 1$,
$\spc{\Hlim} \subseteq [0,1]$.
Paper~XI constructs $\{\Phiinf{n}\}$ with
\begin{equation}\label{eq:eigen}
  \Hlim\Phiinf{n} = \lambdainf{n}\Phiinf{n},\quad
  \ip{\Phiinf{m}}{\Phiinf{n}} = \delta_{mn},\quad
  \lambdainf{n} \in (0,1).
\end{equation}
The question is whether
\begin{equation}\label{eq:main}
  \Kspace := \overline{\mathrm{span}}\{\Phiinf{n}\}
  \stackrel{?}{=} L^2(\mathbb{R}).
\end{equation}

\subsection{Why SOT-convergence is not enough}
\label{sub:sot-not-enough}

The SOT-limit $\Hc \xrightarrow{\mathrm{SOT}} \Hlim$
is unconditional (Paper~X, Thread~A).
However:
\begin{itemize}
\item SOT-convergence does not imply
  norm-resolvent convergence \cite[Ex.~VIII.7.14]{ReedSimon1975}.
\item Eigenvalues can dissolve into continuous spectrum
  under SOT-limits \cite[Ex.~VIII.3.4]{ReedSimon1975}.
\item A uniformly bounded family of complete orthonormal
  bases can SOT-converge to a system that is not complete
  (see Remark~\ref{rem:sot-stability-failure}).
\end{itemize}
The completeness problem is therefore a genuinely new
difficulty, not resolved by the results of Papers~IX--XII.

\begin{remark}[SOT-limit can destroy completeness]
\label{rem:sot-stability-failure}
Let $\{e_n\}$ be the standard ONB of $\ell^2$.
Define $T_c e_n := \cos(c^{-1}) e_n + \sin(c^{-1}) e_{n+1}$
(a rotation mixing adjacent basis vectors).
Then $T_c \xrightarrow{\mathrm{SOT}} \mathrm{id}$,
but for any fixed truncation $N$, the system
$\{T_c e_n : n \le N\}$ does not span
$\mathrm{span}\{e_n : n \le N\}$ for $c < \infty$.
This illustrates that SOT-stability of individual
vectors does not imply stability of spanning properties.
In the PSWF setting, the situation is more subtle
because the ambient spaces $\PW_c^2$ also change
with $c$.
\end{remark}

\section{Inherited Data}
\label{sec:input}

\begin{lemma}[Classical spanning at finite $c$]
\label{lem:finite-c-basis}
For each finite $c > 0$,
$\{\Phic{c}{n}\}_{n \ge 0}$ is a complete orthonormal
basis of $\PW_c^2$ \cite{Slepian1961,Osipov2013}.
In particular:
$\sum_{n \ge 0}|\ip{g}{\Phic{c}{n}}|^2 = \norm{g}^2$
for all $g \in \PW_c^2$.
\end{lemma}

\begin{lemma}[Series results used]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item\label{A:sa}
  \emph{(Paper~X.)} $\Hlim$ bounded self-adjoint,
  $\norm{\Hlim}\le 1$, $\spc{\Hlim}\subseteq[0,1]$.
\item\label{A:ONS}
  \emph{(Paper~XI.)} $\{\Phiinf{n}\}$ satisfies
  \eqref{eq:eigen} and is orthonormal.
\item\label{A:nonvanish}
  \emph{(Paper~VIII.)} For $n\le\kappa c$:
  $|\widehat{\Phic{c}{n}}|\ge c_1 c^{-1/2}$ on $E_c$,
  $|E_c|\ge c_* c^{-1/2}$.
\item\label{A:precompact}
  \emph{(Paper~IX, Lem.~3.1.)} $\{\Phic{c}{n}\}_{c>0}$
  precompact in $L^2$.
\item\label{A:eval-rate}
  \emph{(Paper~VIII.)} $|\lambdac{c}{n}-\lambdainf{n}|
  \le C_\kappa c^{-1/4}$ for $n\le\kappa c$.
\item\label{A:evec-conv}
  \emph{(Paper~XII, cond.~Gap-S+SOT-Faster.)}
  $\norm{\Phic{c}{n}-\Phiinf{n}}\to 0$ for each fixed $n$.
\end{enumerate}
\end{lemma}

\begin{remark}[The stability gap]
\label{rem:stability-gap}
Lemma~\ref{lem:finite-c-basis} gives, for each finite $c$,
the complete Parseval identity on $\PW_c^2$.
Lemma~\ref{lem:inherited}\ref{A:evec-conv} gives
pointwise convergence $\Phic{c}{n}\to\Phiinf{n}$
(conditionally).
The gap is: \emph{pointwise convergence of basis vectors
does not imply preservation of the basis property}.
This is the stability problem at the heart of Paper~XIII.
\end{remark}

\begin{lemma}[Orthogonality characterisation]

%%ŌĢÉŌĢÉŌĢÉ 4. ROUTE A
\section{Route~A: Parseval is Tautologically Equivalent
  to Completeness}
\label{sec:routeA}

\begin{definition}[Gram operator]
\label{def:gram}
$Gf := \sum_{n\ge 0}\ip{f}{\Phiinf{n}}\Phiinf{n}$.
The partial sums form a Cauchy sequence in $L^2$
(Bessel: $\norm{S_Nf-S_Mf}^2 = \sum_{n=M+1}^N
|\ip{f}{\Phiinf{n}}|^2\to 0$).
Hence $G$ is well-defined, bounded, $\norm{Gf}\le\norm{f}$,
and is the orthogonal projection onto $\Kspace$.
(This uses only orthonormality of $\{\Phiinf{n}\}$,
not completeness.)
\end{definition}

\begin{theorem}[Route~A: Parseval $\Leftrightarrow$ completeness]
\label{thm:routeA}
For the orthonormal system $\{\Phiinf{n}\}$:
\[
  \Kspace = L^2(\mathbb{R})
  \iff G = \mathrm{id}
  \iff \textstyle\sum_n|\ip{f}{\Phiinf{n}}|^2 = \norm{f}^2\ \forall f
  \iff \exists A{>}0:\,
    \textstyle\sum_n|\ip{f}{\Phiinf{n}}|^2 \ge A\norm{f}^2\ \forall f.
\]
\end{theorem}
\begin{proof}
(i)${\Rightarrow}$(ii): $\Kspace=L^2$ makes $G$ project onto all
of $L^2$, so $G=\mathrm{id}$.
(ii)${\Rightarrow}$(iii): $\norm{Gf}^2 = \sum_n|\ip{f}{\Phiinf{n}}|^2
= \norm{f}^2$.
(iii)${\Rightarrow}$(iv): take $A=1$.
(iv)${\Rightarrow}$(i): if $f\perp\Phiinf{n}$ for all $n$
then $0\ge A\norm{f}^2$, so $f=0$;
hence $\{\Phiinf{n}\}^\perp=\{0\}$,
i.e.~$\Kspace=L^2$ by Lemma~\ref{lem:orth-char}.
For ONS, Bessel forces $A=1$, confirming
(iii)$\Leftrightarrow$(iv).
\end{proof}

\begin{remark}[Consequence for proof strategies]
\label{rem:routeA-consequence}
Theorem~\ref{thm:routeA} shows that the Parseval identity
and the frame lower bound are the same mathematical
fact as completeness, expressed in different language.
The typical correct logical direction is:
prove $\Kspace=L^2$ first (by density or spanning),
then deduce Parseval as a consequence.
The reverse direction --- trying to prove Parseval
in order to establish completeness --- requires
implicitly solving completeness already.
Any approach based on frame identities must therefore
work on a subspace level (individual $\PW_c^2$ spaces)
before passing to $L^2$ by density.
This is the entry point for Route~B.
\end{remark}

%%ŌĢÉŌĢÉŌĢÉ 5. ROUTE B
\section{Route~B: Completeness as a Stability Problem
  for PSWF Spanning Structures}
\label{sec:routeB}

\subsection{The localisation theorem}

\begin{lemma}[Paley--Wiener density]
\label{lem:pw-dense}
$\bigcup_{c>0}\PW_c^2$ is dense in $L^2(\mathbb{R})$.
\end{lemma}
\begin{proof}
Schwartz functions with compactly supported Fourier
transform lie in $\bigcup_c\PW_c^2$ and are dense
\cite[Thm.~7.14]{Rudin1991}.\end{proof}

\begin{theorem}[Route~B: localisation of completeness]
\label{thm:H2-equiv}
The following are equivalent:
\begin{enumerate}[\rm(i)]
\item $\Kspace = L^2(\mathbb{R})$;
\item $\PW_{c_0}^2 \subseteq \Kspace$ for every $c_0 > 0$.
\end{enumerate}
\end{theorem}
\begin{proof}
(i)${\Rightarrow}$(ii): trivial since $\PW_{c_0}^2\subset L^2=\Kspace$.
(ii)${\Rightarrow}$(i): if $\PW_{c_0}^2\subseteq\Kspace$
for all $c_0$, then $\bigcup_{c_0}\PW_{c_0}^2\subseteq\Kspace$.
Since $\Kspace$ is closed and
$\bigcup_{c_0}\PW_{c_0}^2$ is dense
(Lemma~\ref{lem:pw-dense}),
we get $L^2=\overline{\bigcup_{c_0}\PW_{c_0}^2}
\subseteq\Kspace$.
\end{proof}

\begin{remark}[Route~B is a localisation, not a reduction]
\label{rem:routeB-localisation}
Theorem~\ref{thm:H2-equiv} shows that global completeness
is equivalent to the family of local spanning conditions
$\PW_{c_0}^2\subseteq\Kspace$.
This is \emph{not a reduction} of the difficulty:
the quantification over all $c_0>0$ makes the local
statement exactly as strong as global completeness.

The \emph{conceptual value} of Route~B lies elsewhere:
\begin{itemize}
\item Each individual space $\PW_{c_0}^2$ is a
  \emph{concrete, analytically tractable}
  separable Hilbert space with the explicit
  finite-$c_0$ PSWF basis
  (Lemma~\ref{lem:finite-c-basis}).
\item The local question ---
  does $\{\Phiinf{n}\}$ span $\PW_{c_0}^2$? ---
  is a \emph{stability problem for PSWF basis
  structures under the SOT-limit}:
  for finite $c_0$, the answer is yes by
  Lemma~\ref{lem:finite-c-basis};
  the question is whether this survives $c\to\infty$.
\item This reframing identifies the correct
  \emph{analytic tools}: eigenvector convergence,
  coefficient stability, Fourier approximation in
  individual bandwidth windows.
\end{itemize}
Route~B translates global completeness into an
\emph{inductive-limit stability question};
it does not solve it.
\end{remark}

\subsection{The genuine weaker target}

\begin{hypothesis}[Local spanning: single bandwidth]
\label{hyp:local-span}
For a single fixed $c_0 > 0$:
\[
  \PW_{c_0}^2 \subseteq \Kspace,
  \quad\text{i.e.}\quad
  \{\Phiinf{n}\}_{n\ge 0}\text{ spans }\PW_{c_0}^2.
\]
\end{hypothesis}

\begin{remark}[Local H2$_{c_0}$ is strictly weaker than completeness]
\label{rem:local-H2-weaker}
Hypothesis~\ref{hyp:local-span} for a single $c_0$ is
strictly weaker than global completeness:
$\Kspace=L^2$ implies H2$_{c_0}$ for every $c_0$,
but H2$_{c_0}$ for one $c_0$ does not imply $\Kspace=L^2$.
It constitutes a \emph{non-trivial partial result}:
it says that the limit eigenfunctions span at least one
concrete Paley--Wiener band.
A proof of H2$_{c_0}$ for all $c_0$ would, by
Theorem~\ref{thm:H2-equiv}, complete the proof.
\end{remark}

\subsection{The coefficient stability hypothesis}

\begin{hypothesis}[H1: coefficient stability in $\ell^2$]
\label{hyp:coeff-stability}
For each $c_0>0$ and $g\in\PW_{c_0}^2$,
let $a_n^{(c)}:=\ip{g}{\Phic{c}{n}}$.
As $c\to\infty$: $a_n^{(c)}\to a_n^{(\infty)}:=
\ip{g}{\Phiinf{n}}$ in $\ell^2(\mathbb{N})$.
\end{hypothesis}

\begin{remark}[H1 is strictly weaker than completeness]
\label{rem:H1-weaker}
H1 concerns the $\ell^2$-stability of \emph{coordinates}
under the SOT-limit.
It does not assert that these coordinates reconstruct
$g$ in $L^2$; that is H2$_{c_0}$.
H1 is therefore strictly weaker than completeness and
does not imply H2$_{c_0}$.
It is the only hypothesis in this paper that is
(a) strictly weaker than completeness and
(b) directly supported by the series:
Lemma~\ref{lem:inherited}\ref{A:evec-conv}
(Paper~XII) gives pointwise convergence
$a_n^{(c)}\to a_n^{(\infty)}$;
Lemma~\ref{lem:inherited}\ref{A:nonvanish}
(Paper~VIII) prevents coefficient collapse;
the Bessel bound gives uniform $\ell^2$ control.
The open step in H1 is norm-convergence in $\ell^2$,
i.e., $\norm{(a_n^{(c)})}_{\ell^2}
\to\norm{(a_n^{(\infty)})}_{\ell^2}$
(see Remark~\ref{rem:H1-grounding}).
\end{remark}

\begin{remark}[H1 grounding: weak + norm $\Rightarrow$ strong]
\label{rem:H1-grounding}
For each $n$: $a_n^{(c)}\to a_n^{(\infty)}$ pointwise
(Lemma~\ref{lem:inherited}\ref{A:evec-conv}),
giving weak $\ell^2$-convergence.
For each $c$: $\sum_n|a_n^{(c)}|^2\le\norm{g}^2$
(Bessel), giving uniform norm bound.
Strong $\ell^2$-convergence would follow from
weak convergence plus norm convergence
(Hilbert-space identity
$\norm{x-y}^2=\norm{x}^2-2\,\mathrm{Re}\ip{x}{y}+\norm{y}^2$).
The missing step is:
\begin{equation}\label{eq:norm-H1}
  \textstyle\sum_n|a_n^{(c)}|^2
  \to\sum_n|a_n^{(\infty)}|^2.
\end{equation}
This requires a domination argument beyond the
Bessel majorant;
Lemma~\ref{lem:inherited}\ref{A:nonvanish}
(Paper~VIII) provides non-collapse of individual
coefficients for $n\le\kappa c$ and is the
natural starting point.
\end{remark}

\subsection{Conditional completeness via Route~B}

\begin{theorem}[Conditional completeness]
\label{thm:main-B}
\textup{(Conditional on H1 + H2$_{c_0}$ for all $c_0>0$.)}
If Hypothesis~\ref{hyp:coeff-stability} holds and
$\PW_{c_0}^2\subseteq\Kspace$ for all $c_0>0$,
then $\Kspace = L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Direct from Theorem~\ref{thm:H2-equiv}(ii)${\Rightarrow}$(i).
\end{proof}

\begin{remark}[Role of H1 in the proof]
\label{rem:H1-role}
Note that Theorem~\ref{thm:main-B} uses
H2$_{c_0}$ (for all $c_0$) as its sole hypothesis;
H1 does not appear in the proof.
H1 is relevant as \emph{supporting evidence}
for H2$_{c_0}$: coefficient stability (H1)
is a necessary consequence of H2$_{c_0}$
(if $g\in\PW_{c_0}^2$ is reconstructed by
$\{\Phiinf{n}\}$, then the coordinates must be stable),
but not sufficient for it.
H1 is therefore a \emph{verifiable partial step}
towards H2$_{c_0}$, and the only step in this
direction that is grounded in the current series.
\end{remark}

%%ŌĢÉŌĢÉŌĢÉ 6. ROUTE C
\section{Route~C: Conditional Compactness}
\label{sec:routeC}

\begin{remark}[Route~C is structurally independent]
\label{rem:routeC-indep}
Routes~A and~B operate within the PSWF geometry:
they use the Paley--Wiener spaces $\PW_c^2$,
the finite-$c$ basis property, and the spectral
data of Papers~VIII--XII.
Route~C bypasses all of this:
it posits a global compactness hypothesis on $\Hlim$
that, if true, gives completeness from the spectral
theorem alone, independently of PSWF geometry.
The two approaches reflect different operator regimes
and are alternatives, not competitors.
\end{remark}

\begin{hypothesis}[Compact resolvent]
\label{hyp:compact-resolvent}
$(\Hlim + I)^{-1}$ is compact on $L^2(\mathbb{R})$.
\end{hypothesis}

\begin{proposition}[Route~C]
\label{prop:compact-route}
\textup{(Conditional on Hypothesis~\ref{hyp:compact-resolvent}.)}
If $(\Hlim+I)^{-1}$ is compact, then $\Hlim$ has
purely discrete spectrum with finite multiplicities
accumulating only at $0$ or $1$,
and $\{\Phiinf{n}\}$ is an orthonormal basis;
hence $\Kspace = L^2(\mathbb{R})$.
\end{proposition}
\begin{proof}
$T:=(\Hlim+I)^{-1}$ compact self-adjoint
(self-adjoint since $\Hlim$ is).
By \cite[Thm.~VI.3.2]{ReedSimon1975},
$\spc{T}$ consists of eigenvalues accumulating at $0$.
Since the eigenvalues of $T$ are $(\lambdainf{n}+1)^{-1}$
(positive, as $\lambdainf{n}>0$),
$\spc{\Hlim}=\{\lambdainf{n}\}$ is purely discrete
with finite multiplicities;
$\spc{\Hlim}\subseteq[0,1]$ forces accumulation
only at $0$ or $1$.
The spectral theorem \cite[Thm.~VIII.6]{ReedSimon1975}
then gives the orthonormal basis property.
\end{proof}

\begin{remark}[Incompatibility with $\spc{\Hlim}=[0,1]$]
\label{rem:op4}
Paper~X, Open Problem~4 asks whether $\spc{\Hlim}=[0,1]$.
Continuous spectrum filling $[0,1]$ is incompatible
with compact resolvent
(compact self-adjoint operators have no non-zero
continuous spectrum).
Hypothesis~\ref{hyp:compact-resolvent} and
$\spc{\Hlim}=[0,1]$ are mutually exclusive;
Paper~XIII does not resolve this tension.
\end{remark}

%%ŌĢÉŌĢÉŌĢÉ 7. CLASSIFICATION
\section{Classification Theorem}
\label{sec:classification}

\begin{theorem}[Classification of stability mechanisms]
\label{thm:classification}
The completeness problem $\Kspace=L^2(\mathbb{R})$
admits three mechanisms of fundamentally different
logical character:
\begin{enumerate}[\rm(I)]
\item \emph{(Route~A.)} Parseval$\Leftrightarrow$completeness
  (Thm.~\ref{thm:routeA}).
  Frame identities are tautological reformulations,
  not proof tools.
\item \emph{(Route~B.)} Completeness
  $\Leftrightarrow$ $\PW_{c_0}^2\subseteq\Kspace$
  for all $c_0$ (Thm.~\ref{thm:H2-equiv}).
  This is a localisation theorem:
  global completeness is equivalent to the stability
  of PSWF spanning structures across all Paley--Wiener
  bands under the SOT-limit.
\item \emph{(Route~C.)} Compact resolvent
  $\Rightarrow$ completeness
  (Prop.~\ref{prop:compact-route}).
  Structurally independent of Routes~A+B.
\end{enumerate}

\end{theorem}

\begin{remark}[What Paper~XIII establishes]
Paper~XIII makes four contributions:
\begin{enumerate}[\rm(a)]
\item Theorem~\ref{thm:routeA}: no frame shortcut exists.
\item Theorem~\ref{thm:H2-equiv}: completeness is
  equivalent to a family of local PSWF stability
  conditions --- the correct localisation of the problem.
\item The identification of local H2$_{c_0}$
  (Hypothesis~\ref{hyp:local-span}) and H1
  (Hypothesis~\ref{hyp:coeff-stability})
  as the only statements in the paper that are
  strictly weaker than completeness and analytically
  grounded in the series.
\item Proposition~\ref{prop:compact-route}:
  an independent route via spectral collapse.
\end{enumerate}
\end{remark}

%%ŌĢÉŌĢÉŌĢÉ 8. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent
In order of difficulty and analytical accessibility:

\begin{problem}[H1: norm convergence of coefficients]
\label{prob:H1}
\emph{(Level~1: technical, most accessible.)}
Prove~\eqref{eq:norm-H1}.
Starting point: Paper~VIII nonvanishing +
Bessel majorant + Paper~XII eigenvector convergence.
\end{problem}

\begin{problem}[Local spanning: single bandwidth]
\label{prob:local-span}
\emph{(Level~2: local spectral stability, main gap.)}
For some (and then for all) fixed $c_0>0$,
prove that $\{\Phiinf{n}\}$ spans $\PW_{c_0}^2$,
i.e.\ establish Hypothesis~\ref{hyp:local-span}.
This is the \emph{minimal non-trivial partial result}
for completeness:
it is strictly weaker than $\Kspace=L^2$,
non-vacuous, and directly connected to the
finite-$c$ basis property
(Lemma~\ref{lem:finite-c-basis})
via the stability question of Remark~\ref{rem:stability-gap}.
Natural approach: exploit that $\{\Phic{c_0}{n}\}$
spans $\PW_{c_0}^2$ for finite $c_0$,
use Paper~XII eigenvector convergence to pass to
the limit for each $n$, and control the approximation
error via Papers~VI--VIII.
\end{problem}

\begin{problem}[Global spanning: all bandwidths]
\label{prob:global-span}
\emph{(Level~2b: equivalent to completeness.)}
Prove $\PW_{c_0}^2\subseteq\Kspace$ for all $c_0>0$.
By Theorem~\ref{thm:H2-equiv}, this is equivalent
to $\Kspace=L^2$.
\end{problem}

\begin{problem}[Spectral type of $\Hlim$]
\label{prob:spectral-type}
\emph{(Level~3: global, hardest.)}
Is $\spc{\Hlim}$ purely discrete, or does
$\spc{\Hlim}=[0,1]$?
Is $(\Hlim+I)^{-1}$ compact?
The last two are mutually exclusive
(Remark~\ref{rem:op4}).
\end{problem}

\begin{problem}[Operator identification]
\label{prob:bridge}
Prove $\Hlim = \overline{H_{\mathrm{spec}}}$
(Paper~IX, OP~7; Paper~X Bridge Theorem, conditional).
Requires completeness of $\{\Phiinf{n}\}$.
\end{problem}

\section{Summary}
\label{sec:summary}

Paper~XIII classifies the completeness problem
$\Kspace=L^2(\mathbb{R})$ as a \emph{stability problem
for PSWF spanning structures under the SOT-limit}.

\medskip

\begin{tabular}{llll}
\toprule
Mechanism & Strength & Analytic core & Status \\

Route~A & $=$ completeness & --- & Closed (tautology) \\
Route~B & $=$ completeness & SOT-stability
  of $\PW_c^2$-bases & Open \\
Local H2$_{c_0}$ & $<$ completeness
  & Basis stability at fixed $c_0$ & Open (main target) \\
H1 & $<$ completeness

Route~C & independent & Spectral type & Open \\

\end{center}

The local spanning problem (OP~\ref{prob:local-span})
is the \emph{minimal non-trivial open target}:
it is strictly weaker than completeness,
non-vacuous, and connected to the finite-$c$
basis structure via the stability gap
(Remark~\ref{rem:stability-gap}).
