\title[Completeness of the Limiting Slepian Eigenbasis]{%
  Completeness of the Limiting Slepian Eigenbasis in $L^2(\mathbb{R})$:
  A Structural Classification}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
Let $\Hlim$ denote the bounded self-adjoint SOT-limit operator
constructed in Paper~X, and let $\{\Phiinf{n}\}_{n \ge 0}$
be the orthonormal system of limit eigenfunctions from Paper~XI.
Define $\Kspace := \overline{\mathrm{span}}\{\Phiinf{n}\} \subseteq L^2(\mathbb{R})$.
This paper studies the completeness problem $\Kspace = L^2(\mathbb{R})$,
which is Open Problem~7 of Paper~IX and Open Problem~5 of Paper~X.

\medskip
\textbf{Main contribution: structural classification.}
We classify the completeness problem into three mechanisms
of fundamentally different logical character:

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Mechanism & Character & Status \\
\midrule
A (Frame identity) & \emph{Equivalence reformulation}
  & Not an independent proof route \\
B (Paley--Wiener transfer) & \emph{Genuine analytic reduction}
  & Conditional on Hyp.~\ref{hyp:pw-transfer} \\
C (Compact resolvent) & \emph{Structural collapse}
  & Conditional on Hyp.~\ref{hyp:compact-resolvent} \\
\bottomrule
\end{tabular}
\end{center}

\medskip
The central result (Theorem~\ref{thm:route-A-equivalence}) establishes
that the frame identity $\sum_n |\ip{f}{\Phiinf{n}}|^2 = \norm{f}^2$
is \emph{equivalent} to $\Kspace = L^2(\mathbb{R})$ for orthonormal
systems; it is therefore a reformulation, not a reduction.

\medskip
\textbf{Main theorem (Route~B).}
Under Hypothesis~\ref{hyp:pw-transfer} (coefficient stability
under SOT-limit, grounded in Papers~VIII--IX and XII),
the Paley--Wiener density argument yields $\Kspace = L^2(\mathbb{R})$
without any assumption on the spectral type of $\Hlim$.

\medskip
\textbf{Conditional result (Route~C).}
Under Hypothesis~\ref{hyp:compact-resolvent},
$\{\Phiinf{n}\}$ is an orthonormal basis.
This hypothesis is mutually exclusive with
$\spc{\Hlim} = [0,1]$ (Paper~X, Open Problem~4).
\end{abstract}

\section{Introduction}

\subsection{Setting and goal}

\begin{center}
\fbox{\parbox{0.88\linewidth}{%
\centering\itshape
The completeness problem $\Kspace = L^2(\mathbb{R})$ is not resolved
by SOT-convergence, spectral-type assumptions, or Sturm--Liouville theory.
This paper provides a precise classification of the mechanisms
by which completeness could be established, including a proof
that one apparent route (the frame lower bound) is in fact
an equivalent reformulation of the problem itself.}}
\end{center}

\medskip
\noindent
The Slepian concentration operator $\Hc$ acts on $L^2(\mathbb{R})$ via
\[
  (\Hc f)(x) = \int_{-\infty}^{\infty}
    \frac{\sin c(x-y)}{\pi(x-y)} f(y)\, dy.
\]
Paper~X establishes that $\Hlim := \mathrm{SOT}\text{-}\lim_{c \to \infty} \Hc$
exists as a bounded self-adjoint operator with $\norm{\Hlim} \le 1$
and $\spc{\Hlim} \subseteq [0,1]$.
Paper~XI constructs eigenfunctions $\{\Phiinf{n}\}$ satisfying
\begin{equation}\label{eq:eigen-relation}
  \Hlim \Phiinf{n} = \lambdainf{n} \Phiinf{n}, \qquad
  \ip{\Phiinf{m}}{\Phiinf{n}} = \delta_{mn},
  \qquad \lambdainf{n} \in (0,1).
\end{equation}
The central question is whether
\begin{equation}\label{eq:completeness-question}
  \Kspace := \overline{\mathrm{span}}\{\Phiinf{n}\}_{n \ge 0}
  \stackrel{?}{=} L^2(\mathbb{R}).
\end{equation}

\subsection{What this paper does not claim}

\begin{itemize}
\item We do not claim $\spc{\Hlim}$ is purely discrete
  (Paper~X, Open Problem~4 remains open).
\item We do not derive completeness from
  Sturm--Liouville theory of $\mathcal{L}_c$:
  the operators $\mathcal{L}_c$ and $\Hc$ are distinct;
  their eigenfunctions coincide classically \cite{Osipov2013}
  but the spectral structure of $\Hlim$ is independent.
\item We do not re-prove self-adjointness of $\Hlim$
  (unconditional in Paper~X).
\end{itemize}

%%═══ 3. INTERFACE LEMMA
\section{Interface Lemma: Completeness via Orthogonality}
\label{sec:interface}

\begin{lemma}[Orthogonality characterisation]
\label{lem:orth-char}
For an orthonormal system $\{\Phiinf{n}\}$ in $L^2(\mathbb{R})$:
\[
  \Kspace = L^2(\mathbb{R})
  \quad\Longleftrightarrow\quad
  \Bigl[\,\ip{f}{\Phiinf{n}} = 0\ \forall\, n
  \;\Rightarrow\; f = 0\,\Bigr].
\]
\end{lemma}
\begin{proof}
\cite[Theorem~3.3]{Conway1990}.
\end{proof}

\begin{remark}[Function of this lemma]
This is an \emph{interface lemma}: it converts the geometric
question~\eqref{eq:completeness-question} into an analytic condition
on inner products, which is the entry point for both Route~B
and Route~C.
\end{remark}

%%═══ 4. ROUTE A: FRAME IDENTITY AS EQUIVALENCE
\section{Route~A: The Frame Identity is an Equivalence, Not a Reduction}
\label{sec:riesz}

We record Route~A precisely in order to establish its correct
logical status: it is a reformulation of completeness,
not an independent proof strategy.

\begin{definition}[Gram operator]
\label{def:gram}
The Gram operator $G \colon L^2(\mathbb{R}) \to L^2(\mathbb{R})$ is defined by
\[
  Gf := \sum_{n \ge 0} \ip{f}{\Phiinf{n}} \Phiinf{n}.
\]
Bessel's inequality gives $\norm{Gf} \le \norm{f}$,
so $G$ is a well-defined bounded contraction.
$G$ is the orthogonal projection onto $\Kspace$.
\end{definition}

\begin{theorem}[Frame identity $\Longleftrightarrow$ completeness]
\label{thm:route-A-equivalence}
For the orthonormal system $\{\Phiinf{n}\}$,
the following are equivalent:
\begin{enumerate}[\rm(i)]
\item $\Kspace = L^2(\mathbb{R})$ (completeness);
\item $G = \mathrm{id}_{L^2(\mathbb{R})}$ (Gram operator is the identity);
\item The Parseval identity holds:
  $\displaystyle\sum_{n \ge 0}|\ip{f}{\Phiinf{n}}|^2 = \norm{f}^2$
  for all $f \in L^2(\mathbb{R})$;
\item The frame lower bound holds with $A = 1$:
  $\displaystyle\sum_{n \ge 0}|\ip{f}{\Phiinf{n}}|^2 \ge A\norm{f}^2$
  for some $A > 0$ and all $f \in L^2(\mathbb{R})$.
\end{enumerate}
\end{theorem}
\begin{proof}
(i)$\Rightarrow$(ii): If $\Kspace = L^2(\mathbb{R})$, then $G$ is the
orthogonal projection onto all of $L^2(\mathbb{R})$, hence $G = \mathrm{id}$.
(ii)$\Rightarrow$(iii): $\norm{Gf}^2 = \norm{f}^2$ expands as Parseval.
(iii)$\Rightarrow$(iv): Take $A = 1$.
(iv)$\Rightarrow$(i): The lower bound $\sum_n|\ip{f}{\Phiinf{n}}|^2 \ge A\norm{f}^2 > 0$
for $f \ne 0$ implies $Gf \ne 0$, hence $G$ is injective,
hence $\Kspace = \overline{\mathrm{ran}\,G} = L^2(\mathbb{R})$.
Finally (i)$\Rightarrow$(iii) gives $A = 1$ as the only admissible value,
so (iii)$\Leftrightarrow$(iv) for orthonormal systems.
\end{proof}

\begin{remark}[Route~A is not an independent proof strategy]
\label{rem:route-A-status}
Theorem~\ref{thm:route-A-equivalence} has the following critical
consequence: \emph{proving the frame lower bound
$\sum_n |\ip{f}{\Phiinf{n}}|^2 \ge A\norm{f}^2$ is exactly as
hard as proving completeness itself.}
Route~A is therefore a structural characterisation of the problem,
not a path to its resolution.
In particular:
\begin{itemize}
\item The nonvanishing estimates of Paper~VIII
  (Lemma~\ref{lem:inherited}\ref{A:nonvanish}) concern
  $|\widehat{\Phi}_n^{(c)}|$ for \emph{finite} $c$.
  They do not directly control $\ip{f}{\Phiinf{n}}$ for $f \notin PW_c^2$,
  and the passage $c \to \infty$ in the frame sum requires
  precisely the coefficient-stability that constitutes OP~\ref{prob:frame-stability}.
\item Any derivation of the frame lower bound from Paper~VIII data
  must implicitly solve the completeness problem;
  there is no shortcut via nonvanishing alone.
\end{itemize}
\end{remark}

\begin{problem}[Structural decomposition of completeness]
\label{prob:frame-stability}
Analyse whether $\Kspace = L^2(\mathbb{R})$ can be decomposed
into the following two independent statements:
\begin{enumerate}[\rm(a)]
\item \emph{(Limit-interchange stability.)}
\[
  \lim_{c \to \infty}
  \sum_{n \le \kappa c}
  |\ip{f}{\Phic{c}{n}}|^2
  = \sum_{n \ge 0}|\ip{f}{\Phiinf{n}}|^2
  \quad \forall\, f \in L^2(\mathbb{R}).
\]
\item \emph{(Frame isometry.)}
$\displaystyle\sum_{n \ge 0}|\ip{f}{\Phiinf{n}}|^2 = \norm{f}^2$.
\end{enumerate}
Note that (a) requires only dominated convergence via the
Bessel majorant and strong eigenvector convergence
(Lemma~\ref{lem:inherited}\ref{A:eigenvec-conv}),
while (b) is, by Theorem~\ref{thm:route-A-equivalence},
precisely the completeness statement.
Statement~(a) therefore does not reduce (b); it is a
\emph{necessary} consequence of completeness, not a sufficient one.
\end{problem}

%%═══ 5. ROUTE B: PALEY–WIENER TRANSFER (MAIN THEOREM)
\section{Route~B: Paley--Wiener Transfer --- Main Theorem}
\label{sec:pw}

Route~B is the primary analytic route.
It avoids circularity by attacking completeness directly:
if every $f$ orthogonal to all $\Phiinf{n}$ must be zero,
it suffices to show $f \perp \PW_c^2$ for all $c > 0$,
and then invoke density.

\begin{lemma}[Paley--Wiener density]
\label{lem:pw-dense}
$\bigcup_{c > 0} \PW_c^2$ is dense in $L^2(\mathbb{R})$.
\end{lemma}
\begin{proof}
$\bigcup_{c > 0} \PW_c^2$ contains all Schwartz functions with
compactly supported Fourier transform, which form a dense subspace
\cite[Theorem~7.14]{Rudin1991}.
\end{proof}

\begin{hypothesis}[Coefficient stability under SOT-limit]
\label{hyp:pw-transfer}
For each $c_0 > 0$ and $g \in \PW_{c_0}^2$,
expand $g = \sum_{n \ge 0} a_n^{(c)} \Phic{c}{n}$
in the PSWF basis of $\PW_c^2$ (for $c \ge c_0$).
Assume that as $c \to \infty$:
\begin{enumerate}[\rm(i)]
\item The coefficients converge: $a_n^{(c)} \to a_n^{(\infty)}$ in $\ell^2$.
\item The limit satisfies: $g = \sum_{n \ge 0} a_n^{(\infty)} \Phiinf{n}$
  in $L^2(\mathbb{R})$.
\item Dominated convergence applies via the Bessel majorant:
  $\sum_n |a_n^{(\infty)}|^2 \le \norm{g}^2$.
\end{enumerate}
\end{hypothesis}

\begin{remark}[Grounding of Hypothesis~\ref{hyp:pw-transfer}]
\label{rem:hyp-grounding}
Part~(i) of Hypothesis~\ref{hyp:pw-transfer} is related to
strong eigenvector convergence
$\norm{\Phic{c}{n} - \Phiinf{n}} \to 0$,
established conditionally in Paper~XII
(Lemma~\ref{lem:inherited}\ref{A:eigenvec-conv})
under Gap-S and SOT-Faster.
Part~(iii) follows from the orthonormality of $\{\Phiinf{n}\}$
(Lemma~\ref{lem:inherited}\ref{A:ONS}) unconditionally.
Part~(ii) is the substantive content: it says $g$ is recoverable
from its limit-coefficient expansion, which requires both
eigenvector convergence and a uniform approximation argument.
This is the genuine open step (Open Problem~\ref{prob:pw-transfer}).
\end{remark}

\begin{theorem}[Main theorem: Route~B]
\label{thm:main-B}
\textup{(Conditional on Hypothesis~\ref{hyp:pw-transfer}.)}
Assume Hypothesis~\ref{hyp:pw-transfer} holds.
Then $\Kspace = L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Suppose $f \in L^2(\mathbb{R})$ satisfies $\ip{f}{\Phiinf{n}} = 0$
for all $n \ge 0$; we show $f = 0$.

\textbf{Step~1.} Fix $c_0 > 0$ and $g \in \PW_{c_0}^2$.
By Hypothesis~\ref{hyp:pw-transfer}(ii),
\[
  g = \sum_{n \ge 0} a_n^{(\infty)} \Phiinf{n}
  \quad \text{in } L^2(\mathbb{R}).
\]

\textbf{Step~2.} Compute the inner product:
\[
  \ip{f}{g}
  = \ip*{f}{\sum_{n \ge 0} a_n^{(\infty)} \Phiinf{n}}
  = \sum_{n \ge 0} a_n^{(\infty)}\, \ip{f}{\Phiinf{n}}.
\]
The interchange of sum and inner product is justified by
dominated convergence:
$\sum_n |a_n^{(\infty)} \ip{f}{\Phiinf{n}}|
\le \norm{g} \cdot \norm{f} < \infty$
(Cauchy--Schwarz + Bessel for $\{a_n^{(\infty)}\}$).

\textbf{Step~3.} By hypothesis $\ip{f}{\Phiinf{n}} = 0$ for all $n$,
so $\ip{f}{g} = 0$.

\textbf{Step~4.} Since $c_0 > 0$ and $g \in \PW_{c_0}^2$ were arbitrary,
$f \perp \bigcup_{c_0 > 0} \PW_{c_0}^2$.
By density (Lemma~\ref{lem:pw-dense}), $f = 0$.

\textbf{Conclusion.} By Lemma~\ref{lem:orth-char},
$\Kspace = L^2(\mathbb{R})$.
\end{proof}

\begin{remark}[Why Route~B is not circular]
\label{rem:route-B-not-circular}
The argument uses Hypothesis~\ref{hyp:pw-transfer}(ii) to
represent $g \in \PW_{c_0}^2$ in the basis $\{\Phiinf{n}\}$.
This representation is the non-trivial content:
it asserts that $\{\Phiinf{n}\}$ \emph{spans} $\PW_{c_0}^2$,
which is a \emph{weaker} statement than spanning all of $L^2(\mathbb{R})$.
Closedness under the union over $c_0$ and density then
upgrade this to full completeness.
The logical direction is:
$\{\Phiinf{n}\}$ spans each $\PW_{c_0}^2$
$\Rightarrow$ $\{\Phiinf{n}\}$ spans $L^2(\mathbb{R})$.
This is a genuine reduction, not a reformulation.
\end{remark}

\begin{remark}[Relation to nonvanishing (Paper~VIII)]
\label{rem:paper8-in-B}
The nonvanishing estimates of Paper~VIII
(Lemma~\ref{lem:inherited}\ref{A:nonvanish}) enter Route~B
indirectly: they support the non-degeneracy of
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ via
$|\widehat{\Phi}_n^{(c)}| \ge c_1 c^{-1/2}$ on $E_c$,
which prevents coefficient collapse in the limit.
Making this precise is part of Open Problem~\ref{prob:pw-transfer}.
\end{remark}

%%═══ 6. ROUTE C: CONDITIONAL COMPACTNESS
\section{Route~C: Conditional Compactness}
\label{sec:compact}

\begin{remark}[Compactness is a genuine additional hypothesis]
\label{rem:compact-not-free}
Since $\spc{\Hlim} \subseteq [0,1]$, we have $-1 \notin \spc{\Hlim}$
and $(\Hlim + I)^{-1}$ is a well-defined bounded operator
(Lemma~\ref{lem:inherited}\ref{A:bounded-sa}).
Compactness of $(\Hlim + I)^{-1}$ is strictly stronger
and is not implied by boundedness, self-adjointness,
or SOT-convergence of $\Hc$ (Remark~\ref{rem:sot-limitation}).
\end{remark}

\begin{hypothesis}[Compact resolvent]
\label{hyp:compact-resolvent}
$(\Hlim + I)^{-1}$ is compact on $L^2(\mathbb{R})$.
\end{hypothesis}

\begin{proposition}[Conditional completeness via compact resolvent]
\label{prop:compact-route}
\textup{(Conditional on Hypothesis~\ref{hyp:compact-resolvent}.)}
If Hypothesis~\ref{hyp:compact-resolvent} holds, then:
\begin{enumerate}[\rm(i)]
\item $\spc{\Hlim} = \{\lambdainf{n}\}_{n \ge 0}$
  (purely discrete, accumulating only at $0$ or $1$);
\item $\{\Phiinf{n}\}$ is an orthonormal basis of $L^2(\mathbb{R})$;
\item $\Kspace = L^2(\mathbb{R})$.
\end{enumerate}
\end{proposition}
\begin{proof}
Compact resolvent implies purely discrete spectrum
\cite[Chapter~VIII]{ReedSimon1975}.
The spectral theorem for bounded self-adjoint operators
with purely discrete spectrum then gives (ii)--(iii)
\cite[Theorem~VIII.6]{ReedSimon1975}.
\end{proof}

\begin{remark}[Mutual exclusivity with Paper~X Open Problem~4]
\label{rem:op4-conflict}
Paper~X Open Problem~4 asks whether $\spc{\Hlim} = [0,1]$.
If $\spc{\Hlim} = [0,1]$, then $\Hlim$ has purely continuous
(or mixed) spectrum, and Hypothesis~\ref{hyp:compact-resolvent}
is \emph{false}: an operator with continuous spectrum
cannot have compact resolvent.
The two possibilities are mutually exclusive;
Paper~XIII does not resolve this dilemma.
\end{remark}

%%═══ 7. STRUCTURAL CLASSIFICATION THEOREM
\section{Structural Classification}
\label{sec:classification}

\begin{theorem}[Classification of completeness mechanisms]
\label{thm:classification}
The completeness problem $\Kspace = L^2(\mathbb{R})$
admits the following classification:
\begin{enumerate}[\rm(I)]
\item\label{class:A}
  \textup{(Mechanism~A: Equivalence reformulation.)}
  For the orthonormal system $\{\Phiinf{n}\}$,
  the frame identity
  $\sum_n |\ip{f}{\Phiinf{n}}|^2 = \norm{f}^2$
  is \emph{equivalent} to $\Kspace = L^2(\mathbb{R})$
  (Theorem~\ref{thm:route-A-equivalence}).
  It is a structural characterisation, not an independent proof route.

\item\label{class:B}
  \textup{(Mechanism~B: Genuine analytic reduction.)}
  Under Hypothesis~\ref{hyp:pw-transfer},
  $\Kspace = L^2(\mathbb{R})$
  (Theorem~\ref{thm:main-B}).
  Route~B is not circular: it reduces completeness to
  the strictly weaker statement that $\{\Phiinf{n}\}$
  spans each $\PW_c^2$, and closes via density.

\item\label{class:C}
  \textup{(Mechanism~C: Structural collapse.)}
  Under Hypothesis~\ref{hyp:compact-resolvent},
  $\{\Phiinf{n}\}$ is an orthonormal basis
  (Proposition~\ref{prop:compact-route}).
  Route~C is logically independent of Route~B;
  it depends on a hypothesis that may be \emph{false}
  if $\spc{\Hlim} = [0,1]$.
\end{enumerate}

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{llll}
\toprule
Route & Type & Hypothesis required & Circular? \\
\midrule
A & Equivalence & None (proven) & Yes (by def.) \\
B & Analytic reduction & Hyp.~\ref{hyp:pw-transfer} & No \\
C & Structural collapse & Hyp.~\ref{hyp:compact-resolvent} & No \\
\bottomrule
\end{tabular}
\end{center}
\end{theorem}

\begin{remark}[Consequence for the series]
Paper~XIII makes three contributions:
(a) Theorem~\ref{thm:route-A-equivalence}: Route~A is an
equivalence, not a reduction --- this is a negative structural result
that closes off a potential proof path;
(b) Theorem~\ref{thm:main-B}: Route~B is the primary analytic
route;
(c) the explicit identification of the two remaining hypotheses
(Hyp.~\ref{hyp:pw-transfer} and Hyp.~\ref{hyp:compact-resolvent})
as the precise obstructions to a complete proof.
\end{remark}

%%═══ 8. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Coefficient stability: Route~B core]
\label{prob:pw-transfer}
Prove Hypothesis~\ref{hyp:pw-transfer}:
for $g \in \PW_{c_0}^2$, the PSWF expansion coefficients
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ satisfy
$a_n^{(c)} \to a_n^{(\infty)}$ in $\ell^2$
and $g = \sum_n a_n^{(\infty)} \Phiinf{n}$ in $L^2(\mathbb{R})$.
This requires:
\begin{enumerate}[\rm(a)]
\item Strong eigenvector convergence
  $\norm{\Phic{c}{n} - \Phiinf{n}} \to 0$
  (Paper~XII, conditional on Gap-S + SOT-Faster);
\item Uniformity in $n$ for the approximation
  $g \approx \sum_{n \le N} a_n^{(c)} \Phic{c}{n}$
  as $c, N \to \infty$;
\item Non-degeneration of coefficients, supported by
  Paper~VIII nonvanishing
  (Lemma~\ref{lem:inherited}\ref{A:nonvanish}).
\end{enumerate}
\end{problem}

\begin{problem}[Spectral type of $\Hlim$]
\label{prob:spectral-type}
Determine whether $\spc{\Hlim}$ is discrete, continuous, or mixed.
\begin{enumerate}[\rm(a)]
\item Is $\spc{\Hlim} = \{\lambdainf{n}\}$? (discrete)
\item Is $\spc{\Hlim} = [0,1]$? (Paper~X, Open Problem~4)
\item Is $(\Hlim + I)^{-1}$ compact?
  (Hypothesis~\ref{hyp:compact-resolvent};
  would imply (a) and close Route~C)
\end{enumerate}
Note: (b) and (c) are mutually exclusive.
\end{problem}

\begin{problem}[Operator identification: bridge theorem]
\label{prob:bridge}
Prove or disprove $\Hlim = \overline{H_{\mathrm{spec}}}$,
where $H_{\mathrm{spec}} := \sum_n \lambdainf{n}\ip{\cdot}{\Phiinf{n}}\Phiinf{n}$.
Completeness of $\{\Phiinf{n}\}$ is necessary for
$H_{\mathrm{spec}}$ to be densely defined on all of $L^2(\mathbb{R})$
(Paper~IX, Open Problem~7).
\end{problem}

\begin{problem}[Spanning of individual $\PW_c^2$ spaces]
\label{prob:pswf-span}
Does $\{\Phiinf{n}\}$ span $\PW_c^2$ for each fixed $c > 0$?
This is the \emph{local} version of completeness and is the
substantive content of Hypothesis~\ref{hyp:pw-transfer}(ii).
A positive answer for all $c$ would complete Route~B.
\end{problem}

%%═══ 9. STRUCTURAL SUMMARY
\section{Structural Summary}
\label{sec:summary}

Paper~XIII identifies three mechanisms for the completeness problem
$\Kspace = L^2(\mathbb{R})$:

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{llll}
\toprule
Route & Character & Key step & Blocks on \\
\midrule
A & Equivalence & --- & Circular by Thm.~\ref{thm:route-A-equivalence} \\
B & Analytic reduction & OP~\ref{prob:pw-transfer} + OP~\ref{prob:pswf-span}
  & Coeff.\ stability \\
C & Structural collapse & OP~\ref{prob:spectral-type}(c)
  & Spectral-type question \\
\bottomrule
\end{tabular}
\end{center}

\medskip
\noindent
Route~B (Theorem~\ref{thm:main-B}) is the only non-circular
analytic route currently available.
Its proof skeleton is complete; the remaining open step
(Hypothesis~\ref{hyp:pw-transfer}) is grounded in
Papers~VIII, IX, and XII and does not require spectral-type
assumptions on $\Hlim$.
