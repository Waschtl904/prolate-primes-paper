  A Structural Classification}

\author{[Author]}

\begin{abstract}
Let $\Hlim$ denote the bounded self-adjoint strong-operator-topology
(SOT) limit operator constructed in Paper~X, and let
$\{\Phiinf{n}\}_{n \ge 0}$ be the orthonormal system of limit
eigenfunctions from Paper~XI.
Define $\Kspace := \overline{\mathrm{span}}\{\Phiinf{n}\} \subseteq L^2(\mathbb{R})$.
This paper studies the completeness problem $\Kspace = L^2(\mathbb{R})$,
which is Open Problem~7 of Paper~IX and Open Problem~5 of Paper~X.

\medskip
\textbf{Main contribution: structural classification.}
We classify the completeness problem into three mechanisms:

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Mechanism & Character & Status \\
\midrule
A (Frame identity) & Equivalence reformulation
  & Not an independent proof route \\
B (Paley--Wiener transfer) & Genuine analytic reduction
  & Conditional on H1 + H2 \\
C (Compact resolvent) & Structural collapse
  & Conditional on Hyp.~\ref{hyp:compact-resolvent} \\
\bottomrule
\end{tabular}
\end{center}

\medskip
Theorem~\ref{thm:route-A-equivalence} establishes that the
Parseval identity is \emph{tautologically equivalent} to
completeness for orthonormal systems:
it is the same statement in a different language,
not a proof tool.
Route~B reduces completeness to two independent hypotheses:
H1 (coefficient stability, supported by Papers~VIII+XII)
and H2 (local $\PW_c^2$-spanning, the genuine open core).
Route~C (compact resolvent) gives a purely discrete
spectrum with finite multiplicities accumulating at $0$ or $1$;
it is structurally disjoint from the PSWF geometry of Routes~A+B.
\end{abstract}

\medskip
\noindent
The Slepian concentration operator $\Hc$ acts on $L^2(\mathbb{R})$ via
\[
  (\Hc f)(x) = \int_{-\infty}^{\infty}
    \frac{\sin c(x-y)}{\pi(x-y)} f(y)\, dy.
\]
Paper~X establishes $\Hlim := \mathrm{SOT\text{-}lim}_{c \to \infty} \Hc$
as a bounded self-adjoint operator with $\norm{\Hlim} \le 1$
and $\spc{\Hlim} \subseteq [0,1]$.
Paper~XI constructs eigenfunctions satisfying
\begin{equation}\label{eq:eigen-relation}
  \Hlim \Phiinf{n} = \lambdainf{n} \Phiinf{n},\qquad
  \ip{\Phiinf{m}}{\Phiinf{n}} = \delta_{mn},\qquad
  \lambdainf{n} \in (0,1).
\end{equation}
The central question is
\begin{equation}\label{eq:completeness-question}
  \Kspace := \overline{\mathrm{span}}\{\Phiinf{n}\}_{n \ge 0}
  \stackrel{?}{=} L^2(\mathbb{R}).
\end{equation}

\begin{lemma}[Orthogonality characterisation]
\label{lem:orth-char}
For an orthonormal system $\{\Phiinf{n}\}$ in $L^2(\mathbb{R})$:
\[
  \Kspace = L^2(\mathbb{R})

\end{proof}

%%═══ 4. ROUTE A
\section{Route~A: The Parseval Identity is a Tautological
  Reformulation of Completeness}
\label{sec:riesz}

\begin{definition}[Gram operator]
\label{def:gram}
For the orthonormal system $\{\Phiinf{n}\}$, define
\[
  Gf := \sum_{n \ge 0} \ip{f}{\Phiinf{n}} \Phiinf{n}.
\]
\emph{Convergence:} By Bessel's inequality,
$\sum_{n=0}^{N} |\ip{f}{\Phiinf{n}}|^2 \le \norm{f}^2$
for all $N$, so the partial sums
$S_N f := \sum_{n=0}^{N} \ip{f}{\Phiinf{n}} \Phiinf{n}$
form a Cauchy sequence in $L^2$
(since $\norm{S_N f - S_M f}^2 = \sum_{n=M+1}^{N}
|\ip{f}{\Phiinf{n}}|^2 \to 0$ as $M,N \to \infty$).
Hence the series converges in $L^2$-norm and
$\norm{Gf} \le \norm{f}$.
\emph{Projection:} $G$ is a bounded self-adjoint idempotent
($G^2 = G = G^*$, verified directly from orthonormality),
so $G$ is the orthogonal projection of $L^2(\mathbb{R})$
onto $\mathrm{ran}\,G = \overline{\mathrm{span}}\{\Phiinf{n}\}
= \Kspace$.
\emph{Note:} this uses only orthonormality of
$\{\Phiinf{n}\}$, not completeness.
\end{definition}

\begin{theorem}[Parseval $\Longleftrightarrow$ completeness]
\label{thm:route-A-equivalence}
For the orthonormal system $\{\Phiinf{n}\}$,
the following are equivalent:
\begin{enumerate}[\rm(i)]
\item $\Kspace = L^2(\mathbb{R})$;
\item $G = \mathrm{id}_{L^2(\mathbb{R})}$;
\item $\displaystyle\sum_{n \ge 0}|\ip{f}{\Phiinf{n}}|^2
  = \norm{f}^2$ for all $f$;
\item $\exists\, A > 0$:
  $\displaystyle\sum_{n \ge 0}|\ip{f}{\Phiinf{n}}|^2
  \ge A\norm{f}^2$ for all $f$.
\end{enumerate}
\end{theorem}
\begin{proof}
\textbf{(i)$\Rightarrow$(ii).}
If $\Kspace = L^2$, then $G$ projects onto all of $L^2$,
so $G = \mathrm{id}$.

\textbf{(ii)$\Rightarrow$(iii).}
$\norm{f}^2 = \norm{Gf}^2 = \sum_n|\ip{f}{\Phiinf{n}}|^2$.

\textbf{(iii)$\Rightarrow$(iv).}
Take $A = 1$.

\textbf{(iv)$\Rightarrow$(i).}
If $f \perp \Phiinf{n}$ for all $n$,
then $\sum_n|\ip{f}{\Phiinf{n}}|^2 = 0$,
so (iv) forces $\norm{f}^2 \le 0$, hence $f = 0$.
Thus $\{\Phiinf{n}\}^\perp = \{0\}$,
which by Lemma~\ref{lem:orth-char} gives
$\Kspace = L^2(\mathbb{R})$.

\textbf{Remark on $A$.}
For orthonormal systems, Bessel gives the upper bound $A \le 1$.
If (iv) holds, then $A = 1$ and (iii) holds;
hence (iii)$\Leftrightarrow$(iv) for orthonormal systems.
\end{proof}

\begin{remark}[Structural consequence: tautology, not negativity]
\label{rem:route-A-status}
Theorem~\ref{thm:route-A-equivalence} shows that the
Parseval identity and the frame lower bound are
\emph{tautologically equivalent} to completeness:
they are the same mathematical fact stated in different
languages.
This is not a negative result in the sense of a
contradiction, but a structural one:
the typical path (prove density, deduce Parseval as
consequence) is the correct logical direction, not the
reverse.
\begin{itemize}
\item Any derivation of the frame lower bound
  from Paper~VIII data must implicitly solve
  completeness, since there is no shortcut via
  nonvanishing alone.
\item Frame-based arguments must be localised:
  establishing $\sum_n|\ip{f}{\Phiinf{n}}|^2 = \norm{f}^2$
  for $f \in \PW_c^2$ first, then passing to $L^2$ via
  density, is the only non-circular route ---
  and this is exactly Route~B.
\end{itemize}
\end{remark}

\begin{problem}[Independence of the two completeness components]
\label{prob:frame-stability}
Analyse whether $\Kspace = L^2$ decomposes into:
\begin{enumerate}[\rm(a)]
\item (limit-interchange)
  $\lim_{c \to \infty}\sum_{n \le \kappa c}
  |\ip{f}{\Phic{c}{n}}|^2
  = \sum_{n}|\ip{f}{\Phiinf{n}}|^2$ $\forall f$;
\item (frame isometry)
  $\sum_{n}|\ip{f}{\Phiinf{n}}|^2 = \norm{f}^2$.
\end{enumerate}
By Theorem~\ref{thm:route-A-equivalence},
(b) is exactly completeness.
(a) is a \emph{necessary} consequence of completeness,
not sufficient.
\end{problem}

%%═══ 5. ROUTE B
\section{Route~B: Paley--Wiener Transfer}
\label{sec:pw}

\begin{lemma}[Paley--Wiener density]
\label{lem:pw-dense}
$\bigcup_{c > 0} \PW_c^2$ is dense in $L^2(\mathbb{R})$.
\end{lemma}
\begin{proof}
Schwartz functions with compactly supported Fourier
transform lie in $\bigcup_c \PW_c^2$ and are dense
\cite[Thm.~7.14]{Rudin1991}.
\end{proof}

\begin{hypothesis}[H1: coefficient stability]
\label{hyp:coeff-stability}
For each fixed $c_0 > 0$ and $g \in \PW_{c_0}^2$,
let $a_n^{(c)} := \ip{g}{\Phic{c}{n}}$ for $c \ge c_0$.
As $c \to \infty$:
\[
  a_n^{(c)} \longrightarrow a_n^{(\infty)}
  := \ip{g}{\Phiinf{n}}
  \quad \text{in } \ell^2(\mathbb{N}).
\]
\end{hypothesis}

\begin{remark}[Justification of H1: weak + norm $\Rightarrow$ strong
  in $\ell^2$]
\label{rem:H1-grounding}
We indicate why H1 is plausible without claiming a proof.

\textbf{Step 1: pointwise convergence.}
For each fixed $n$, strong eigenvector convergence
$\norm{\Phic{c}{n} - \Phiinf{n}} \to 0$
(Lemma~\ref{lem:inherited}\ref{A:eigenvec-conv},
conditional on Gap-S + SOT-Faster, Paper~XII) gives
\[
  a_n^{(c)} = \ip{g}{\Phic{c}{n}}
  \longrightarrow \ip{g}{\Phiinf{n}} = a_n^{(\infty)}
\]
for each fixed $n$ by continuity of the inner product.
This is weak convergence in $\ell^2$:
$a_n^{(c)} \to a_n^{(\infty)}$ coordinatewise.

\textbf{Step 2: norm control.}
For each fixed $c$, Bessel's inequality for
the ONS $\{\Phic{c}{n}\}$ in $\PW_c^2$ gives
$\sum_n |a_n^{(c)}|^2 \le \norm{g}^2$;
analogously $\sum_n |a_n^{(\infty)}|^2 \le \norm{g}^2$.
Hence $\norm{(a_n^{(c)})}_{{\ell^2}} \le \norm{g}$
and $\norm{(a_n^{(\infty)})}_{{\ell^2}} \le \norm{g}$,
both uniformly in $c$.

\textbf{Step 3: weak + norm $\Rightarrow$ strong.}
If $a_n^{(c)} \rightharpoonup a_n^{(\infty)}$ weakly
in $\ell^2$ and additionally
$\norm{(a_n^{(c)})}_{{\ell^2}}
\to \norm{(a_n^{(\infty)})}_{{\ell^2}}$,
then by the Hilbert-space identity
$\norm{x - y}^2 = \norm{x}^2 - 2\,\mathrm{Re}\ip{x}{y}
+ \norm{y}^2$, strong convergence follows.
The missing ingredient is therefore:
\begin{equation}\label{eq:norm-conv-H1}
  \sum_n |a_n^{(c)}|^2
  \longrightarrow \sum_n |a_n^{(\infty)}|^2
  \quad \text{as } c \to \infty.
\end{equation}
This is the genuine open step in H1.
It would follow, for instance, from dominated convergence
in $\ell^2$: if there exists a sequence
$(b_n) \in \ell^1(\mathbb{N})$ with
$|a_n^{(c)}|^2 \le b_n$ uniformly in $c$,
then~\eqref{eq:norm-conv-H1} holds by Lebesgue.
A natural candidate is $b_n = |a_n^{(\infty)}|^2$
plus the eigenvalue gap estimate
(Lemma~\ref{lem:inherited}\ref{A:eigenvalue-rate}),
but making this uniform in $n$ and $c$
requires additional control from Paper~VIII
(nonvanishing + exponential decay of eigenvectors).
\end{remark}

\begin{hypothesis}[H2: local reconstruction in $\PW_c^2$]
\label{hyp:reconstruction}
For each $c_0 > 0$ and $g \in \PW_{c_0}^2$:
\[
  g = \sum_{n \ge 0} a_n^{(\infty)} \Phiinf{n}
  \quad \text{in } L^2(\mathbb{R}),
\]
where $a_n^{(\infty)} = \ip{g}{\Phiinf{n}}$.
Equivalently: $\PW_{c_0}^2 \subseteq \Kspace$.
\end{hypothesis}

\begin{remark}[H2 is strictly harder than H1]
\label{rem:H2-status}
H1 gives $\ell^2$-convergence of coordinates;
H2 asserts that these coordinates reconstruct $g$
in $L^2$-norm.
The implication H1 $\Rightarrow$ H2 is \emph{not}
automatic: knowing $(\ip{g}{\Phiinf{n}}) \in \ell^2$
and $g \in \Kspace^\perp + \Kspace$
does not recover $g$ from its projection coefficients
unless $\Kspace \supseteq \PW_{c_0}^2$.
H2 is a \emph{local completeness statement}:
$\{\Phiinf{n}\}$ spans $\PW_{c_0}^2$ for each fixed $c_0$.
This is structurally weaker in spectral support
than global completeness $\Kspace = L^2$,
but it is a genuine local completeness property
in the inductive limit of Paley--Wiener spaces,
not merely a coordinate identity.
\end{remark}

\begin{theorem}[Route~B: conditional completeness]
\label{thm:main-B}
\textup{(Conditional on H1 + H2.)}
If Hypotheses~\ref{hyp:coeff-stability}
and~\ref{hyp:reconstruction} hold,
then $\Kspace = L^2(\mathbb{R})$.
\end{theorem}
\begin{proof}
Let $f \in L^2$ with $\ip{f}{\Phiinf{n}} = 0$ for all $n$.

\textbf{Step~1.}
Fix $c_0 > 0$ and $g \in \PW_{c_0}^2$.
By H2, $g = \sum_{n} a_n^{(\infty)} \Phiinf{n}$
in $L^2$, with $a_n^{(\infty)} = \ip{g}{\Phiinf{n}}$.

\textbf{Step~2 (Hilbert-space dual pairing).}
Compute $\ip{f}{g}$ using the $L^2$-expansion from Step~1:
\[
  \ip{f}{g}
  = \ip*{f}{\sum_{n} a_n^{(\infty)} \Phiinf{n}}.
\]
The sum converges in $L^2$-norm (H2), so continuity
of the inner product gives:
\[
  \ip{f}{g} = \lim_{N\to\infty}
  \sum_{n=0}^{N} a_n^{(\infty)} \ip{f}{\Phiinf{n}}.
\]
This is a Hilbert-space dual-pairing limit;
no interchange of sum and inner product beyond
continuity is required.
Explicitly: by Cauchy--Schwarz in $\ell^2$,
\[
  \Bigl|\sum_{n=0}^{N} a_n^{(\infty)} \ip{f}{\Phiinf{n}}\Bigr|
  \le \norm{(a_n^{(\infty)})}_{\ell^2}
      \norm{(\ip{f}{\Phiinf{n}})}_{\ell^2}
  \le \norm{g}\norm{f} < \infty,
\]
where both $\ell^2$-norms are finite:
$(a_n^{(\infty)}) \in \ell^2$ by H1 (or directly by
Bessel for $g$),
and $(\ip{f}{\Phiinf{n}}) \in \ell^2$ by Bessel for $f$.

\textbf{Step~3.}
Since $\ip{f}{\Phiinf{n}} = 0$ for all $n$:
$\ip{f}{g} = 0$.

\textbf{Step~4.}
$c_0$ and $g$ were arbitrary, so
$f \perp \bigcup_{c_0} \PW_{c_0}^2$.
Density (Lemma~\ref{lem:pw-dense}) gives $f = 0$.

\textbf{Conclusion.}
By Lemma~\ref{lem:orth-char}, $\Kspace = L^2$.
\end{proof}

\begin{remark}[Why Route~B is not circular]
\label{rem:route-B-not-circular}
The argument uses H2 to represent $g \in \PW_{c_0}^2$
in the basis $\{\Phiinf{n}\}$.
H2 asserts $\PW_{c_0}^2 \subseteq \Kspace$:
this is a local statement about a specific
finite-bandwidth subspace, structurally different
from the global statement $\Kspace = L^2$.
The logical chain is:
\[
  \bigl[\PW_{c_0}^2 \subseteq \Kspace\;
  \forall\, c_0 > 0\bigr]
  \xRightarrow{\text{Lem.~\ref{lem:pw-dense}}}
  \Kspace = L^2(\mathbb{R}).
\]
A hypothetical proof of H2 for a single $c_0$
would not suffice; the density step requires all $c_0$.
Crucially, H2 for each $c_0$ does \emph{not} require
knowing $\Kspace = L^2$; it only requires that
the limit eigenfunctions $\{\Phiinf{n}\}$ span
the finite-bandwidth space $\PW_{c_0}^2$,
which is a priori a weaker spectral-support condition.
\end{remark}

%%═══ 6. ROUTE C
\section{Route~C: Conditional Compactness}
\label{sec:compact}

\begin{remark}[Route~C is structurally disjoint from Routes~A+B]
\label{rem:routeC-disjoint}
Routes~A and~B operate within the PSWF geometry:
they use the spectral structure of $\Hc$,
the Paley--Wiener spaces $\PW_c^2$,
and the eigenvector data from Papers~VIII--XII.
Route~C is structurally independent:
it posits a global compactness hypothesis on $\Hlim$
that, if true, gives completeness purely from
the spectral theorem, bypassing all PSWF geometry.
The two routes correspond to different operator
regimes (compactness vs.~SOT-geometry) and are
not competitors but alternatives.
\end{remark}

\begin{hypothesis}[Compact resolvent]
\label{hyp:compact-resolvent}
$(\Hlim + I)^{-1}$ is compact on $L^2(\mathbb{R})$.
\end{hypothesis}

\begin{proposition}[Route~C: conditional completeness]
\label{prop:compact-route}
\textup{(Conditional on Hypothesis~\ref{hyp:compact-resolvent}.)}
If $(\Hlim + I)^{-1}$ is compact, then:
\begin{enumerate}[\rm(i)]
\item $\Hlim$ has purely discrete spectrum
  $\spc{\Hlim} = \{\lambdainf{n}\}_{n \ge 0}$
  with finite multiplicities,
  accumulating only at $0$ or $1$;
\item $\{\Phiinf{n}\}$ is an orthonormal basis;
\item $\Kspace = L^2(\mathbb{R})$.
\end{enumerate}
\end{proposition}
\begin{proof}
Let $T := (\Hlim + I)^{-1}$.
By hypothesis, $T$ is compact and self-adjoint
(since $\Hlim$ is self-adjoint and $-1 \notin \spc{\Hlim}$).
By the spectral theorem for compact self-adjoint operators
\cite[Thm.~VI.3.2]{ReedSimon1975},
$\spc{T)$ consists of eigenvalues accumulating only at $0$.
Translating: $\lambdainf{n} = (\mu_n)^{-1} - 1$
where $\mu_n$ are eigenvalues of $T$ (all positive
since $T = (\Hlim+I)^{-1}$ and $\Hlim+I$ is
positive definite).
Hence $\spc{\Hlim}$ is purely discrete with
finite multiplicities; since $\spc{\Hlim} \subseteq [0,1]$,
accumulation can occur only at $0$ or $1$.
The eigenfunctions of $T$ are the same as those of
$\Hlim$, and the spectral theorem
\cite[Thm.~VIII.6]{ReedSimon1975}
yields the orthonormal basis property.
\end{proof}

\begin{remark}[Mutual exclusivity with $\spc{\Hlim} = [0,1]$]
\label{rem:op4-conflict}
If $\spc{\Hlim} = [0,1]$ (Paper~X, Open Problem~4),
the spectrum contains a continuous component,
which is incompatible with compact resolvent
(a compact self-adjoint operator has no continuous
spectrum other than possibly $0$).
Hypothesis~\ref{hyp:compact-resolvent} and
$\spc{\Hlim} = [0,1]$ are mutually exclusive.
\end{remark}

%%═══ 7. CLASSIFICATION
\section{Structural Classification}
\label{sec:classification}

\begin{theorem}[Classification of completeness mechanisms]
\label{thm:classification}
The completeness problem $\Kspace = L^2(\mathbb{R})$
admits the following classification:
\begin{enumerate}[\rm(I)]
\item \emph{(Route~A: tautological equivalence.)}
  Parseval $\Leftrightarrow$ completeness
  (Thm.~\ref{thm:route-A-equivalence}).
  Not a proof route.
\item \emph{(Route~B: analytic reduction.)}
  H1 + H2 $\Rightarrow$ completeness
  (Thm.~\ref{thm:main-B}).
  Non-circular; open core is H2.
\item \emph{(Route~C: structural collapse.)}
  Compact resolvent $\Rightarrow$ completeness
  (Prop.~\ref{prop:compact-route}).
  Structurally disjoint from Routes~A+B;
  mutually exclusive with $\spc{\Hlim} = [0,1]$.
\end{enumerate}

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lllll}
\toprule
Route & Type & Circular? & Open core & Depends on \\
\midrule
A & Tautology & Yes & --- & None \\
B & Analytic reduction & No & H2 (local $\PW_c^2$ span)
  & H1 + H2 \\
C & Structural collapse & No
  & Spectral type & Hyp.~\ref{hyp:compact-resolvent} \\
\bottomrule
\end{tabular}
\end{center}
\end{theorem}

%%═══ 8. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent
The open problems are hierarchised by difficulty and
logical dependence:

\medskip
\noindent
\textbf{Level~1 (technical convergence, most accessible):}

\begin{problem}[H1: norm convergence of
  coefficient sequences]
\label{prob:coeff-stability}
Prove Hypothesis~\ref{hyp:coeff-stability},
specifically the norm convergence
\eqref{eq:norm-conv-H1}.
Requires: pointwise convergence (from Paper~XII,
cond.), uniform tail control (Bessel majorant),
non-degeneracy of coefficients for $n \le \kappa c$
(Paper~VIII).
\end{problem}

\medskip
\noindent
\textbf{Level~2 (local spectral completeness, main gap):}

\begin{problem}[H2: local spanning of $\PW_c^2$]
\label{prob:pw-transfer}
Prove Hypothesis~\ref{hyp:reconstruction}:
$\{\Phiinf{n}\}$ spans $\PW_{c_0}^2$ for each
fixed $c_0 > 0$.
This is the genuinely hard step.
It does not follow from H1 alone.
Note: global completeness $\Kspace = L^2$ would
imply H2 (via Thm.~\ref{thm:route-A-equivalence}),
but H2 for all $c_0$ suffices for completeness
without circular reasoning.
\end{problem}

\medskip
\noindent
\textbf{Level~3 (global spectral type, hardest):}

\begin{problem}[Spectral type of $\Hlim$]
\label{prob:spectral-type}
Determine $\spc{\Hlim}$:
\begin{enumerate}[\rm(a)]
\item Is $\spc{\Hlim}$ purely discrete?
\item Is $\spc{\Hlim} = [0,1]$?
\item Is $(\Hlim+I)^{-1}$ compact?
\end{enumerate}
(b) and (c) are mutually exclusive
(Remark~\ref{rem:op4-conflict}).
\end{problem}

\begin{problem}[Operator identification]
\label{prob:bridge}
Prove $\Hlim = \overline{H_{\mathrm{spec}}}$,
where $H_{\mathrm{spec}} :=
\sum_n \lambdainf{n}\ip{\cdot}{\Phiinf{n}}\Phiinf{n}$.
Requires completeness of $\{\Phiinf{n}\}$
(Paper~IX, OP~7).
\end{problem}

%%═══ 9. SUMMARY
\section{Structural Summary}
\label{sec:summary}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{llll}
\toprule
Route & Character & Open core & Blocks on \\
\midrule
A & Tautology & --- & Circular \\
B & Analytic reduction & H2: local $\PW_c^2$ spanning
  & OP~\ref{prob:pw-transfer} \\
C & Structural collapse & Spectral type
  & OP~\ref{prob:spectral-type} \\
\bottomrule
\end{tabular}
\end{center}

\medskip
\noindent
Route~B is the only non-circular route;
its proof skeleton is complete.
The open core is H2 (OP~\ref{prob:pw-transfer}),
grounded in Papers~VIII, IX, XII,
requiring no spectral-type assumptions on $\Hlim$.
Route~C is a structurally independent alternative
that would resolve completeness if compactness
of $(\Hlim+I)^{-1}$ were established.
