\section{Introduction}
\label{sec:intro}
%%════════════════════════════════════════════════════════════════

\subsection{Context and goal}

This paper occupies a specific position in the series:
it provides the quantitative bridge between
the weak operator convergence established in Paper~X
and the completeness result targeted in Paper~XIII.

Paper~X establishes that the PSWF concentration
operators $\mathcal{H}_{c_k}$ converge in the
strong operator topology (SOT) to a bounded
self-adjoint limit $\Hlim$.
Paper~XI constructs a diagonal subsequence
along which the eigenfunctions
$\Phi_n^{(c_k)}$ converge weakly to limit
vectors $\Phi_n^{(\infty)}$, orthonormal and
satisfying $\Hlim\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textbf{The gap that this paper fills.}
SOT convergence of operators does \emph{not} imply
norm convergence of spectral projections
(see Rem.~\ref{rem:strong-vs-norm}).
Weak convergence of eigenfunctions does not imply
strong convergence.
The missing ingredient is a \emph{quantitative control}
on how the rank-1 spectral projections
$P_n^{(c_k)} = \ip{\cdot}{\Phi_n^{(c_k)}}\Phi_n^{(c_k)}$
behave as $c_k\to\infty$.
This control is precisely what
Theorem~\ref{thm:proj-stability}
(the main result of this paper) provides.

\subsection{Main result in context}

The central theorem (Sec.~\ref{sec:main}) is:
\[
  \|\Pnck - \Pninf\| \le
  \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)} \to 0,
\]
under a single structural hypothesis on the
spectral gap of $\mathcal{H}_c$ in the
transition window $W_c$
(\hyperref[hyp:gap-structural]{Hyp.~Gap-S}).
This estimate is explicit, depends only on
the eigenvalue convergence rate of Paper~VIII
and the gap hypothesis, and does \emph{not}
require norm resolvent convergence.

\subsection{What this paper does \emph{not} claim}

\begin{itemize}
\item \textbf{The Riesz-basis property} of
  $\{\Phi_n^{(\infty)}\}$ (stated as
  Cor.~\ref{cor:riesz}) \emph{is trivial}:
  it follows directly from the orthonormality
  established in Paper~XI (Sec.~\ref{sec:riesz-trivial}).
  It is recorded here for completeness
  and to close a loose end in Paper~XI,
  but it is \emph{not} the mathematical contribution
  of this paper.
\item \textbf{Completeness} $\mathcal{K}=L^2(\mathbb{R})$:
  Paper~XIII.
\item \textbf{Sharp form of the spectral gap}
  $g(c)$: Open Problem~\ref{prob:sharp-gc}.
  Section~\ref{sec:heuristic} provides a
  heuristic justification of
  \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
  via Widom asymptotics and the sigmoid
  transition profile of PSWF eigenvalues.
\end{itemize}

\begin{lemma}[Inherited data]
\label{lem:inherited}
The following are established in the

  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}$ for $n\le\kappa c$.
\item\label{A2} \emph{(X, Thread~A + Cor.~cor:resolvent)}

\item\label{A3} \emph{(XI, Lem.~3.1 + Thm.~5.1)}
  There exists $c_k\nearrow\infty$ such that
  $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}=\delta_{mn}$,
  $\Hlim\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$,

\end{enumerate}
\end{lemma}

\begin{remark}[Strong vs.\ norm resolvent convergence]
\label{rem:strong-vs-norm}
Paper~X establishes strong resolvent convergence
of $\mathcal{H}_{c_k}$.

this does \emph{not} imply norm convergence of
spectral projections.
The additional input required is the gap
hypothesis (Hyp.~\ref{hyp:gap-structural}),
which provides the spectral isolation needed for
the rank-1 projection stability argument
(Lem.~\ref{lem:rank1-stability}).
This is the essential new content of this paper.
\end{remark}

\section{Spectral Separation Hypothesis and WKB Layer}
\label{sec:gap}
%%════════════════════════════════════════════════════════════════

\begin{hypothesis}[Structural gap, Gap-S]
\label{hyp:gap-structural}
There exists $g\colon[1,\infty)\to(0,\infty)$
with $g(c)\to 0$ and $g(c)\gg c^{-1/4}$
such that for all $c\ge 1$ and $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S is the only hypothesis]
\label{rem:hyp-role}
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
is the single structural input of this paper
beyond the inherited data of Lem.~\ref{lem:inherited}.
All results (Thm.~\ref{thm:proj-stability},
Cor.~\ref{cor:nacc}, Cor.~\ref{cor:riesz})
follow from it.
Section~\ref{sec:heuristic} provides a
heuristic justification.
\end{remark}

\subsection{WKB layer and the transfer barrier}

The prolate differential operator
\begin{equation}\label{eq:Lc}
  \Lc := -\tfrac{d}{dx}[(1-x^2)\tfrac{d}{dx}]
  + c^2x^2
\end{equation}
has simple eigenvalues $\chi_n^{(c)}$.
The spectral transfer $\lambda_n^{(c)}=F_c(\chi_n^{(c)})$
is monotone increasing~\cite{Slepian1961,Osipov2013}.
A standard Airy reduction (Langer substitution
at the turning point $x_c=\sqrt{1-n^2/c^2}$)
yields a growing gap
$\chi_{n+1}^{(c)}-\chi_n^{(c)}\ge C\,\delta(c)$
with $\delta(c)\to\infty$.

\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of $\mathcal{L}_c$ alone
does not determine a canonical algebraic scaling
for $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
in $W_c$, without global kernel information
about $\mathcal{H}_c$.
\end{theorem}
\begin{proof}
The map $F_c$ near $\lambda=1/2$ has a
soft-exponential (sigmoid) profile
(Slepian--Widom, \cite{Slepian1961,Widom1964}).
Any formula $F_c'(\chi_n^{(c)})\asymp c^{-\beta}$
for fixed $\beta$ is therefore an unverified
global claim about $F_c$.
The $\lambda$-gap is positive (simplicity
of $\mathcal{H}_c$ + monotonicity of $F_c$)
but its rate is a global invariant of $\mathcal{H}_c$,
not determinable from $\Lc$ alone.
\end{proof}

\begin{remark}[Compatibility with Widom/Slepian]
Thm.~\ref{thm:wkb-indeterminacy} asserts
insufficiency of local ODE analysis,
not non-existence of an asymptotic scale.
Transition asymptotics of $F_c$ via global
kernel information would resolve
Prob.~\ref{prob:sharp-gc}.
\end{remark}

\section{Main Theorem: Spectral Projection Stability}
\label{sec:main}
%%════════════════════════════════════════════════════════════════

\begin{theorem}[Spectral projection stability]
\label{thm:proj-stability}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and the inherited data of Lem.~\ref{lem:inherited}.
For all $n\le\kappa c_k$:
\begin{equation}\label{eq:main-bound}
  \|\Pnck - \Pninf\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}
  \xrightarrow{k\to\infty} 0.
\end{equation}
In particular:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$ strongly in
  $L^2(\mathbb{R})$, uniformly for $n\le\kappa c_k$.
\item $\{\lambda_n^{(\infty)}\}$ has no accumulation
  point in $(0,1)$.
\item Synthesis operators satisfy
  $T_k\to T_\infty$ in operator norm,
  $\sup_k\|T_k\|=1$.
\end{enumerate}
\end{theorem}

\begin{remark}[Why this is not trivial]
\label{rem:why-nontrivial}
Eigenvalue convergence
$|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|\to 0$
does not alone imply
$\|\Pnck-\Pninf\|\to 0$.
The additional content is:
\begin{enumerate}
\item Gap-S provides spectral isolation,
  enabling use of the Riesz contour formula.
\item The specialized rank-1 projection
  stability lemma (Lem.~\ref{lem:rank1-stability})
  converts scalar eigenvalue control
  into operator norm control.
\item The combination of these two ingredients
  replaces norm resolvent convergence,
  which is \emph{not} available from Paper~X.
\end{enumerate}
\end{remark}

\section{Proof of the Main Theorem}
\label{sec:proj}
%%════════════════════════════════════════════════════════════════

\begin{lemma}[Rank-1 projection stability under
  scalar eigenvalue control]
\label{lem:rank1-stability}
Let $A$, $B$ be bounded self-adjoint operators
on a Hilbert space $\mathcal{H}$.
Suppose:
\begin{enumerate}[\rm(i)]
\item $\lambda$ is a simple eigenvalue of $A$
  isolated by spectral gap $\gamma>0$:
  $\mathrm{dist}(\lambda,\sigma(A)\setminus\{\lambda\})
  \ge\gamma$.
\item $\tilde\lambda$ is the corresponding simple
  eigenvalue of $B$ with
  $|\tilde\lambda-\lambda|\le\delta<\gamma/2$.
\end{enumerate}
Then the rank-1 spectral projections satisfy:
\begin{equation}\label{eq:rank1-bound}
  \|P_A(\lambda)-P_B(\tilde\lambda)\|
  \le \frac{C\,\delta}{\gamma},
\end{equation}
where $C>0$ is a universal constant.
\end{lemma}
\begin{proof}
Choose the Riesz contour $\Gamma$ of radius
$r\in(\delta,\gamma/2)$ centered at $\lambda$;
for large enough parameter, shift the center
to $\lambda$ uniformly.
Both projections are given by the Riesz formula:
$P = \frac{1}{2\pi i}\oint_\Gamma(H-z)^{-1}dz$.
On $\Gamma$:
$\mathrm{dist}(z,\sigma(A))\ge\gamma/2-r>0$;
$\mathrm{dist}(z,\sigma(B))>0$
(since $\tilde\lambda$ is inside $\Gamma$ and
all other eigenvalues of $B$ are
outside by a gap of at least $\gamma-\delta>\gamma/2$).
The rank-1 resolvent identity gives:
\[
  (A-z)^{-1}-(B-z)^{-1}
  =\frac{(\tilde\lambda-\lambda)
    \ip{\cdot}{\phi_B}\phi_A}
    {(\lambda-z)(\tilde\lambda-z)}
  +R(z),
\]
where $\phi_A$, $\phi_B$ are unit eigenvectors
and $\|R(z)\|=O(\delta^2/\gamma^3)$.
Integrating and bounding
$|\lambda-z|\ge r-\delta$ and $|\tilde\lambda-z|\ge r-\delta$
on $\Gamma$:
\[
  \|P_A-P_B\|\le
  \frac{|\Gamma|}{2\pi}\cdot
  \frac{\delta}{(r-\delta)^2}+O(\delta^2/\gamma^3).
\]
Choosing $r=\gamma/3$ and using $\delta<\gamma/2$
yields \eqref{eq:rank1-bound} with explicit $C$.
For full details see
\cite[\S\,II.1.4, Eq.~(1.17)]{Kato1966}.
\end{proof}

\begin{remark}[Comparison to general Kato theory]
\label{rem:kato-comparison}
The general Kato perturbation theory
(\cite[Ch.~II]{Kato1966})
deals with analytic families and
complex-variable eigenvalue tracking.
Lem.~\ref{lem:rank1-stability} is a
self-contained specialization to the
rank-1 case with explicit scalar eigenvalue
control~\eqref{A1}.
Its proof does not require
holomorphic families, only the Riesz formula,
the gap geometry of Gap-S,
and the eigenvalue rate of Paper~VIII.
\end{remark}

\begin{proof}[Proof of Thm.~\ref{thm:proj-stability}]
By \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
$\lambda_n^{(c_k)}$ is isolated in
$\sigma(\mathcal{H}_{c_k})$ with gap $\gamma=g(c_k)$.
By \eqref{A1},
$|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
\le C_\kappa c_k^{-1/4}=:\delta_k$.
For large $k$, $\delta_k<g(c_k)/2$
(since $g(c_k)\gg c_k^{-1/4}$),
so the hypotheses of Lem.~\ref{lem:rank1-stability}
are satisfied with $A=\mathcal{H}_{c_k}$,
$B=\Hlim$, $\gamma=g(c_k)$, $\delta=\delta_k$.
Apply \eqref{eq:rank1-bound}:
\[
  \|\Pnck-\Pninf\|
  \le\frac{C C_\kappa c_k^{-1/4}}{g(c_k)}\to 0.
\]
Part~(i): $\|P_n^{(c_k)}f-\Pninf f\|
\le\|\Pnck-\Pninf\|\cdot\|f\|\to 0$.
Apply with $f$ extracting $\Phi_n$.
Part~(ii): See Cor.~\ref{cor:nacc}.
Part~(iii): $\|T_k\|=1$ from Parseval;
$T_k\to T_\infty$ in norm from \eqref{eq:main-bound}.
\end{proof}

\begin{corollary}[NAcc: no accumulation]
\label{cor:nacc}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{A1},
$\{\lambda_n^{(\infty)}\}$ has no accumulation
point in $(0,1)$.
\end{corollary}
\begin{proof}
Fix $\mu\in(0,1)$ and $r>0$.
Choose $k_0$ with $C_\kappa c_{k_0}^{-1/4}<r/2$.
Then $\#\{n:\lambda_n^{(\infty)}\in(\mu-r,\mu+r)\}
\le\#\{n:\lambda_n^{(c_{k_0})}\in
(\mu-3r/2,\mu+3r/2)\}
\le\lceil 3r/g(c_{k_0})\rceil<\infty$.
\end{proof}

\section{The Riesz-Basis Corollary (Explicit Trivialization)}
\label{sec:riesz-trivial}
%%════════════════════════════════════════════════════════════════

\begin{remark}[This section is logically redundant]
\label{rem:riesz-redundant}
The Riesz-basis property of
$\{\Phi_n^{(\infty)}\}$ follows \emph{immediately}
from the orthonormality established in
Paper~XI (Lem.~\ref{lem:inherited}\eqref{A3})
and does not depend on any result of this paper.
It is stated here solely for completeness
and to close a loose end in Paper~XI.
The mathematical contribution of this paper
is Thm.~\ref{thm:proj-stability}.
\end{remark}

\begin{corollary}[Riesz basis — trivial]
\label{cor:riesz}
Let $\mathcal{K}:=\overline{\mathrm{span}}
\{\Phi_n^{(\infty)}\}$.
Then $\{\Phi_n^{(\infty)}\}$ is a Riesz basis
of $\mathcal{K}$ with constants $A=B=1$.
\end{corollary}
\begin{proof}
Orthonormality from \eqref{A3} gives
$\bigl\|\sum a_n\Phi_n^{(\infty)}\bigr\|^2=\sum|a_n|^2$
directly. Hence $A=B=1$.
\end{proof}

\begin{remark}[Non-trivial content]
\label{rem:nontrivial-content}
What is \emph{non-trivial} is not the Riesz property
but rather:
\begin{enumerate}
\item That the system $\{\Phi_n^{(\infty)}\}$
  exists and is orthonormal (Paper~XI).
\item That the projections $P_n^{(c_k)}$
  converge in operator norm to the
  limit projections (Thm.~\ref{thm:proj-stability}).
\item That this norm convergence is the
  technical input for Paper~XIII.
\end{enumerate}
\end{remark}

\section{Canonicalization (Conditional)}
\label{sec:canonical}
%%════════════════════════════════════════════════════════════════

\begin{lemma}[Spectral stability of $\mathcal{K}$,
  conditional]
\label{lem:same-K}
Assume additionally:
\begin{hypothesis}[Full projection family convergence]
\label{hyp:full-family}
The projections $\{\Pnck\}_{n\ge 0}$
converge as a family to
$\{\Pninf\}_{n\ge 0}$ in the
sense that the spectral measure of
$\mathcal{H}_{c_k}$ converges to that of
$\Hlim$ in the weak-$*$ sense.
\end{hypothesis}
Then for any two diagonal subsequences
$c_k$, $c_k'$ satisfying
\hyperref[hyp:gap-structural]{Hyp.~Gap-S},
the closed spans $\mathcal{K}$ and $\mathcal{K}'$
coincide.
\end{lemma}
\begin{proof}
Both $\mathcal{K}$ and $\mathcal{K}'$
are closed invariant subspaces of $\Hlim$
(SOT uniqueness, Paper~X Thread~A).
Under Hyp.~\ref{hyp:full-family},
both families generate the same
spectral measure of $\Hlim$,
hence the same closed span.
\end{proof}

\begin{remark}[Status of this lemma]
\label{rem:same-K-status}
Lem.~\ref{lem:same-K} is conditional on
Hyp.~\ref{hyp:full-family}.
Theorem~\ref{thm:proj-stability} gives
pointwise (in $n$) norm convergence of
spectral projections, but not convergence
of the full projection family as a spectral
measure.
Closing this gap would require uniform
control over the entire index range
$n\le\kappa c_k$ jointly,
not just for fixed $n$.
This is identified as the main remaining
structural gap in this paper.
\end{remark}

\begin{theorem}[Canonicalization, conditional]
\label{thm:canonical}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and Hyp.~\ref{hyp:full-family},
$\Hlim$ is a canonical self-adjoint operator
on $\mathcal{K}$, independent of diagonal
subsequence up to unitary equivalence.
\end{theorem}
\begin{proof}
(1) Riesz basis: Cor.~\ref{cor:riesz}.
(2) Unique $\mathcal{K}$: Lem.~\ref{lem:same-K}.
(3) Spectrum independent of subsequence:
SOT uniqueness + \eqref{A1}.
(4) Simple spectrum + orthonormal eigenbasis
    $\Rightarrow$ unitarily equivalent by
    spectral theorem.
\end{proof}

\section{Heuristic Justification of Gap-S}
\label{sec:heuristic}
%%════════════════════════════════════════════════════════════════

\textbf{This section is heuristic and non-rigorous.}
Its purpose is to explain why
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
is not ad hoc but is supported by
the known asymptotic structure of PSWF
eigenvalues.

\subsection{The sigmoid transition profile}

It is well known that the PSWF concentration
eigenvalues $\lambda_n^{(c)}$ exhibit a
sigmoidal transition from values near~$1$
to values near~$0$, centered at $n\approx 2c/\pi$
(the prolate regime boundary)~\cite{Slepian1961,Widom1964}.
The transition window has width
$O(\log c)$ in index space:
\begin{equation}\label{eq:transition}
  \lambda_n^{(c)} \approx
  \frac{1}{1+e^{\pi(n-2c/\pi)/\sigma(c)}},
  \quad \sigma(c) \asymp \log c.
\end{equation}
This is the \emph{Widom asymptotic formula}
(cf.~\cite{Widom1964,Osipov2013}).

\subsection{Local density and gap estimate}

The local density of eigenvalues in the
transition window is:
\begin{equation}\label{eq:density}
  \rho(c) := \#\{n : \lambda_n^{(c)}\in
  [1/4,3/4]\} \asymp \log c.
\end{equation}
By the mean value property,
the average spacing in the window is:
\begin{equation}\label{eq:spacing}
  \text{average gap} \sim \frac{1}{\rho(c)}
  \sim (\log c)^{-1}.
\end{equation}
Since $c^{-1/4}\ll (\log c)^{-1}$ for all
large $c$, this gives:
\begin{equation}\label{eq:gap-plausible}
  g(c) \sim (\log c)^{-1} \gg c^{-1/4},
\end{equation}
which is exactly the condition in
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}.

\subsection{What this heuristic means}

\begin{remark}[Isolation of the missing input]
\label{rem:missing-input}
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
is not an opaque global assumption.
It isolates a single piece of missing
global information:
the \emph{local density of concentration
eigenvalues in the transition window}.
The Widom formula \eqref{eq:transition}
strongly suggests that this density is
$O(\log c)$, which would give
$g(c)\sim(\log c)^{-1}$.
Making this rigorous via
Plancherel--Rotach asymptotics or
determinantal kernel estimates
would simultaneously prove
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and resolve Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{remark}[Comparison to Random Matrix Theory]
The spacing statistics of PSWF eigenvalues
in the transition window are expected to
resemble those of GUE random matrices
(Poisson-like in the bulk, Wigner-Dyson
at the transition edge).
In either regime, the minimum gap
grows faster than $c^{-1/4}$,
consistent with Gap-S.
This is further heuristic support,
not a proof.
\end{remark}

\section{Open Problems}
\label{sec:open}
%%════════════════════════════════════════════════════════════════

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Determine the rigorous form of $g(c)$
via global control of $F_c'$
(Plancherel--Rotach, resolvent asymptotics,
or determinantal kernel methods).
Expected: $g(c)\sim(\log c)^{-1}$.
\end{problem}

\begin{problem}[Full projection family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family}
(weak-$*$ convergence of spectral measures
of $\mathcal{H}_{c_k}$ to that of $\Hlim$)
using the pointwise convergence of
Thm.~\ref{thm:proj-stability}
and uniform index control.
This would close Lem.~\ref{lem:same-K}
unconditionally.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Is $\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII will use
Thm.~\ref{thm:proj-stability}
as the key input.
\end{problem}

\begin{problem}[Quantitative Riesz constants]
\label{prob:riesz-quant}
For a non-orthonormal perturbation
$\{\tilde\Phi_n^{(\infty)}\}$ of
$\{\Phi_n^{(\infty)}\}$,
determine Riesz constants $A$, $B$ explicitly
