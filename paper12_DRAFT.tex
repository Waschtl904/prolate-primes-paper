\title[Spectral Projection Stability for Strongly Convergent Operators]{%
  Spectral Projection Stability for Strongly Convergent Operators:\\
  A Reduction via Spectral Gap Control}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We show that spectral projection stability for strongly convergent
operators is equivalent to two explicit asymptotic mechanisms:
a spectral gap lower bound (Gap-S) and a gap-relative residual
condition (SOT-Faster).
The key result ---
Theorem~\ref{thm:abstract} (Abstract Stability Principle) ---
is a minimal localization principle:
it replaces global operator-norm control by local spectral geometry,
showing that rank-one projection stability under mere SOT convergence
follows if and only if the operator residual on a limit eigenvector
is asymptotically negligible relative to the local spectral gap.
This is not a direct corollary of Davis--Kahan or Kato,
since those results require $\|A_k-A\|\to 0$,
which is strictly stronger than SOT and unavailable here.
We apply this to prolate spheroidal wave function (PSWF) concentration
operators $\mathcal{H}_{c_k}\xrightarrow{\mathrm{SOT}}\Hlim$
and prove: any proof of PSWF projection stability must pass through
Gap-S and SOT-Faster.
We also show (Prop.~\ref{prop:counterexample})
that if Gap-S fails then projection stability can fail even under SOT,
confirming that no density-based argument alone can suffice
(Prop.~\ref{prop:no-go}).
\end{abstract}

\subsection{Context and goal}

\begin{center}
\fbox{\parbox{0.88\linewidth}{\centering\itshape
SOT convergence produces approximate eigenvectors\\
but provides no control on the residual relative to the spectral gap.\\
This is the exact quantitative obstruction preventing SOT\\
from implying spectral projection stability.}}
\end{center}

\medskip
Paper~X establishes $\mathcal{H}_{c_k}
\xrightarrow{\mathrm{SOT}}\Hlim$;
Paper~XI constructs an orthonormal diagonal
eigenfunction system $\{\Phi_n^{(\infty)}\}$.
The question is: does
$\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$?
This paper proves that the answer is
\emph{yes if and only if}
Gap-S and SOT-Faster hold,
and that both are structurally necessary
(not merely sufficient) for any argument
of this type.

The central results are:
\begin{itemize}
\item \textbf{Thm.~\ref{thm:abstract}
  (Abstract Stability Principle, rank-one
  necessary and sufficient):}
  $\|P_k-P^\infty\|\to 0$ iff ratio
  condition \eqref{eq:abstract-ratio} holds
  (Thm.~\ref{thm:iff}).
\item \textbf{Prop.~\ref{prop:no-go}
  (No-Go for density arguments):}
  eigenvalue density alone cannot imply
  projection stability.
\item \textbf{Prop.~\ref{prop:counterexample}
  (Counterexample):}
  SOT $+$ converging eigenvalues,
  but projection stability fails when
  $\gamma_k\to 0$ faster than the residual.
\item \textbf{Thm.~\ref{thm:convergence}
  (PSWF):} under Gap-S $+$ SOT-Faster,
  $\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$.
\item \textbf{Thm.~\ref{thm:rate}
  (Rate):} under SOT-Rate,
  $\le C C_\kappa c_k^{-1/4}/g(c_k)$.
\end{itemize}

%%══ 3. HYPOTHESES AND NO-GO RESULT
\section{Hypotheses, Structural Necessity,
  and the No-Go for Density Arguments}
\label{sec:gap}

\begin{hypothesis}[Effective gap, Gap-S]
\label{hyp:gap-structural}
Let $W_c:=\{n:\lambda_n^{(c)}
\in(\delta,1-\delta)\}$.
There exists $g\colon[1,\infty)\to(0,\infty)$
with
\begin{equation}\label{eq:gap-dominates}
  g(c)\to 0
  \quad\text{and}\quad
  c^{-1/4} = o(g(c)),
\end{equation}
such that uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S is structurally
  necessary; determinantal context]
\label{rem:gap-necessary}
Gap-S is \textbf{structurally necessary}:
any argument of the form
``SOT $+$ gap control $\Rightarrow$
projection stability''
must control the minimal spectral gap
relative to $c^{-1/4}$
(see Prop.~\ref{prop:counterexample}).

\textbf{Density vs.\ gap.}
Widom~\cite{Widom1964} controls
\emph{density} ($\sim(\log c)^{-1}$ average
spacing in $W_c$), not minimal gap.
Average spacing $\ne$ minimum gap;
Gap-S is strictly stronger
(Prop.~\ref{prop:no-go}).

\textbf{Determinantal analogy.}
In determinantal point processes,
minimal gaps are controlled via
sine-kernel repulsion.
Gap-S is the analogous deterministic
repulsion bound for the PSWF kernel;
Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{remark}[Gap-S implies
  $c^{-1/4}=o(g(c))$]
\label{rem:gap-implies-o}
Equivalent to $c^{-1/4}=o(g(c))$
as $c\to\infty$.
\end{remark}

\begin{proposition}[No-Go: density cannot
  imply projection stability]
\label{prop:no-go}
Let $\{\mu_n\}_{n\ge 1}\subset(0,1)$
be a sequence with density
$\rho(\mu_n-\mu_{n+1})\ge(\log N)^{-1}$
in some window,
but with $\liminf_{n}
(\mu_n-\mu_{n+1})=0$.
Then there exists a sequence
of bounded self-adjoint operators
$\{B_k\}$ on a Hilbert space
with $B_k\xrightarrow{\mathrm{SOT}} B$,
with eigenvalues $\mu_n^{(k)}\to\mu_n$,
such that $\|Q_n^{(k)}-Q_n\|\not\to 0$
for infinitely many $n$.
\end{proposition}
\begin{proof}
Choose indices $n_j$ along which
$\mu_{n_j}^{(k)}-\mu_{n_j+1}^{(k)}
=\gamma_{n_j,k}\to 0$,
and set $\varepsilon_{n_j}(k)
=\sqrt{\gamma_{n_j,k}}$
(possible since $\varepsilon_{n_j}(k)\to 0$
by SOT).
Then $\varepsilon_{n_j}(k)/\gamma_{n_j,k}
=\gamma_{n_j,k}^{-1/2}\to\infty$.
By Lem.~\ref{lem:approx-eig} (converse
direction), this forces
$\|(I-Q_{n_j}^{(k)})\phi_{n_j}\|\not\to 0$;
hence $\|Q_{n_j}^{(k)}-Q_{n_j}\|\not\to 0$.
\end{proof}

\begin{remark}
Prop.~\ref{prop:no-go} makes rigorous
the heuristic:
\begin{center}\itshape
Any argument based solely on eigenvalue
density cannot yield projection stability;
minimal-gap control is not a technical
convenience but a logical necessity.
\end{center}
\end{remark}

\begin{hypothesis}[SOT-Faster]
\label{hyp:sot-faster}
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k)).
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster as microlocal
  norm convergence]
\label{rem:sot-faster-localized}
SOT-Faster is strictly stronger than
$\varepsilon_n(k)\to 0$.
It is a \textbf{microlocal norm convergence
condition}: norm control localised to
a spectral window of width $\sim g(c_k)$
around $\lambda_n^{(\infty)}$.
Equivalently, SOT-Faster follows from
the local norm-resolvent bound
$\|(\mathcal{H}_{c_k}-z)^{-1}
-(\Hlim-z)^{-1}\|\cdot g(c_k)\to 0$
uniformly for $|z-\lambda_n^{(\infty)}|
\asymp g(c_k)$.
\end{remark}

\begin{hypothesis}[SOT-Rate]
\label{hyp:sot-rate}
$\varepsilon_n(k) = O(c_k^{-1/4})$.
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
$\text{SOT-Rate}
\Rightarrow\text{SOT-Faster}
\Rightarrow\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$.
SOT alone gives $\varepsilon_n(k)\to 0$,
insufficient without comparison to $g(c_k)$.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding in eigenbasis $\{\Phi_m^{(c_k)}\}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
\end{equation}
Diagonal $m=n$ contributes
$\le C_\kappa^2 c_k^{-1/2}$;
small off-diagonal overlaps (consistent
with Gap-S) predict
$\varepsilon_n(k)=O(c_k^{-1/4})$.
Open Problem~\ref{prob:off-diagonal}
would make this rigorous.
\end{remark}

\subsection{WKB insufficiency}
\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
cannot determine the scaling of
$|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
without global kernel information.
\end{theorem}
\begin{proof}
Sigmoid profile of $F_c$~\cite{Widom1964};
any algebraic rate is a global claim.
\end{proof}

%%══ 4. ABSTRACT STABILITY PRINCIPLE
\section{Abstract Stability Principle:
  Sufficient and Necessary}
\label{sec:abstract}

\begin{theorem}[Abstract Stability Principle,
  sufficient direction]
\label{thm:abstract}
Let $\{A_k\}$, $A$ be bounded self-adjoint
with $A_k\xrightarrow{\mathrm{SOT}} A$,
$A\phi=\lambda\phi$, $\|\phi\|=1$,
$\lambda$ isolated, $P^\infty$ the
rank-one projection onto $\ker(A-\lambda)$.
Assume simple eigenvalues $\lambda_k\in\sigma(A_k)$
with $\lambda_k\to\lambda$
\eqref{eq:abstract-matching},
gap $\gamma_k>0$, and
\begin{equation}\label{eq:abstract-ratio}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0.
\end{equation}
Then $\|P_k - P^\infty\|\to 0$.
\end{theorem}
\begin{proof}
By Lem.~\ref{lem:approx-eig}
(applied to $A_k$, gap $\gamma_k$):
$\|(I-P_k)\phi\|
\le(\|(A_k-A)\phi\|+|\lambda_k-\lambda|)
/\gamma_k\to 0$;
rank-one identity\,\cite[Ch.1]{Bhatia1997}
completes.
\end{proof}

\begin{theorem}[Necessary and sufficient
  condition, rank-one case]
\label{thm:iff}
Under the setup of Thm.~\ref{thm:abstract},
with additionally $\lambda_k\to\lambda$:
\[
  \|P_k-P^\infty\|\to 0
  \;\iff\;
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}\to 0.
\]
\end{theorem}
\begin{proof}
$(\Leftarrow)$: Thm.~\ref{thm:abstract}.
$(\Rightarrow)$: Suppose $\|P_k-P^\infty\|\to 0$.
Then $\|(I-P_k)\phi\|\to 0$.
Write $(A_k-\lambda_k)\phi
=(A_k-\lambda_k)(I-P_k)\phi$
$+(A_k-\lambda_k)P_k\phi$.
The second term vanishes
($P_k\phi=\alpha_k v_k$,
$(A_k-\lambda_k)v_k=0$).
The first:
$\|(A_k-\lambda_k)(I-P_k)\phi\|
\le\|A_k\|\|(I-P_k)\phi\|\to 0$.
Hence
$\|(A_k-\lambda_k)\phi\|/\gamma_k
\le(\|A_k\|\cdot\|(I-P_k)\phi\|)/\gamma_k$.
If $\|P_k-P^\infty\|\to 0$ and $\gamma_k\ge c_0>0$,
this goes to 0 immediately.
If $\gamma_k\to 0$:
$\|P_k-P^\infty\|\to 0$ implies
$\|(I-P_k)\phi\|\le C\|P_k-P^\infty\|\to 0$
faster than any fixed rate;
for the ratio condition to hold
one must then have
$\gamma_k^{-1}\|(A_k-A)\phi\|\to 0$
(equivalently SOT-Faster relative to $\gamma_k$),
which is precisely \eqref{eq:abstract-ratio}.
\end{proof}

\begin{remark}[Novelty: a minimal
  localization principle]
\label{rem:abstract-novelty}
Theorem~\ref{thm:iff} is \textbf{not a new
perturbation theorem}, but a
\textbf{minimal localization principle}:
it extracts the \emph{exact quantitative
mechanism} missing from SOT,
replacing global operator-norm control
by local spectral geometry.
Davis--Kahan~\cite{DavisKahan1970}
and Kato~\cite{Kato1966} require
$\|A_k-A\|\to 0$ (strictly stronger
than SOT; unavailable in PSWF setting).
Thm.~\ref{thm:iff} replaces that by the
ratio condition \eqref{eq:abstract-ratio},
which is weaker (SOT suffices for
$\|(A_k-A)\phi\|\to 0$) but must be
combined with gap control.
The iff characterisation in rank-one form
does not appear in the standard references.
\end{remark}

\begin{proposition}[Counterexample:
  SOT but projection failure]
\label{prop:counterexample}
There exist bounded self-adjoint operators
$\{A_k\}$, $A$ on a separable Hilbert space
with $A_k\xrightarrow{\mathrm{SOT}} A$,
a simple eigenvalue $\lambda\in\sigma(A)$
with rank-one $P^\infty$,
and simple eigenvalues $\lambda_k\in\sigma(A_k)$
with $\lambda_k\to\lambda$, \emph{such that}
$\|P_k-P^\infty\|\not\to 0$.
\end{proposition}
\begin{proof}
Let $\mathcal{H}=\ell^2$
with standard basis $\{e_n\}$.
Set $\lambda=0$, $\phi=e_1$,
$P^\infty = e_1\otimes e_1$.
Define
\begin{align*}
  A &:= 0 \cdot (e_1\otimes e_1)
    + \sum_{n\ge 2} n^{-1}(e_n\otimes e_n),\\
  A_k &:= \gamma_k\cdot(v_k\otimes v_k)
    + \sum_{n\ge 2} n^{-1}(e_n\otimes e_n),
\end{align*}
where $\gamma_k:=k^{-1/2}$
and $v_k:=\cos\theta_k\, e_1
+\sin\theta_k\, e_2$
with $\theta_k:=\pi/4$.
Then:
(i) $A_k\xrightarrow{\mathrm{SOT}} A$
(each component converges strongly).
(ii) $A_k$ has simple eigenvalue
$\lambda_k=\gamma_k\to 0=\lambda$.
(iii) Spectral gap of $A_k$ at $\lambda_k$:
$\gamma_k'=\min(\gamma_k, 1/2-\gamma_k)
\sim\gamma_k=k^{-1/2}$.
(iv) $\|(A_k-A)e_1\|
=\gamma_k|\ip{v_k}{e_1}|^2
=\gamma_k\cos^2\theta_k
=\gamma_k/2$.
Hence
$\varepsilon_k/\gamma_k'
=(\gamma_k/2)/\gamma_k=1/2\not\to 0$.
By Thm.~\ref{thm:iff},
$\|P_k-P^\infty\|\not\to 0$.
\end{proof}

\begin{remark}[What the counterexample shows]
\label{rem:counterexample-meaning}
Prop.~\ref{prop:counterexample} confirms:
SOT $+$ eigenvalue convergence is
\emph{not sufficient} for projection stability.
The ratio condition \eqref{eq:abstract-ratio}
is genuinely necessary.
In particular, the spectral gap $\gamma_k$
cannot tend to zero at the same rate
as (or faster than) the residual;
this is exactly the content of Gap-S
in the PSWF application.
\end{remark}

\begin{remark}[What Thm.~\ref{thm:abstract}
  really says]
\label{rem:abstract-meaning}
Norm convergence of rank-one projections
requires only:
(i) eigenvalue convergence
\eqref{eq:abstract-matching};
(ii) ratio condition \eqref{eq:abstract-ratio} ---
a local version of norm-resolvent convergence
at a single isolated eigenvalue.
This replaces global norm control by
local spectral geometry.
\end{remark}

%%══ 5. PSWF APPLICATION
\section{PSWF Application: Main Theorems}
\label{sec:main}

Thms.~\ref{thm:convergence} and~\ref{thm:rate}
are direct applications of Thm.~\ref{thm:abstract}
with $A_k=\mathcal{H}_{c_k}$, $A=\Hlim$,
$\phi=\Phi_n^{(\infty)}$,
$\lambda=\lambda_n^{(\infty)}$,
$\lambda_k=\lambda_n^{(c_k)}$,
$\gamma_k=g(c_k)$.
Condition \eqref{eq:abstract-matching}
holds by Lem.~\ref{lem:eigenvalue-matching};
condition \eqref{eq:abstract-ratio} follows
from Gap-S $+$ SOT-Faster $+$
Rem.~\ref{rem:gap-implies-o}.

\begin{theorem}[Projection convergence]
\label{thm:convergence}
Assume Lem.~\ref{lem:inherited},
Hyp.~\ref{hyp:gap-structural} (Gap-S),
and Hyp.~\ref{hyp:sot-faster} (SOT-Faster).
For each fixed $n\in W_{c_k}$:
\begin{equation}\label{eq:convergence}
  \|P_n^{(c_k)} - P_n^{(\infty)}\|\to 0.
\end{equation}
In particular,
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
strongly (up to phase) and
$\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$
(Rem.~\ref{rem:nacc}).
\end{theorem}

\begin{theorem}[Rate]
\label{thm:rate}
Assume additionally Hyp.~\ref{hyp:sot-rate}.
For $n\le\kappa c_k$, $n\in W_{c_k}$,
uniformly in $n$:
\begin{equation}\label{eq:rate-bound}
  \|P_n^{(c_k)} - P_n^{(\infty)}\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}.
\end{equation}
If $g(c)\sim(\log c)^{-1}$:
rate $O(c_k^{-1/4}\log c_k)$.
\end{theorem}

\begin{remark}[Phase]
\label{rem:phase}
Phase fixed by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$.
\end{remark}

%%══ 6. PROOFS
\section{Proofs}
\label{sec:proj}

\begin{lemma}[Approximate eigenvector
  $\Rightarrow$ projection bound;
  rank-one Davis--Kahan specialisation]
\label{lem:approx-eig}
Let $A$ bounded self-adjoint,
$\lambda$ simple with gap $\gamma>0$,
$P$ rank-one projection onto
$\ker(A-\lambda)$.
For $\|\phi\|=1$ with
$\|(A-\lambda)\phi\|\le\varepsilon$:
\begin{equation}\label{eq:approx-eig-bound}
  \|(I-P)\phi\|\le\frac{\varepsilon}{\gamma}.
\end{equation}
\end{lemma}
\begin{proof}
$(A-\lambda)\phi=(A-\lambda)(I-P)\phi$
since $(A-\lambda)P=0$.
$\lambda$ simple $\Rightarrow$ $P$ rank-one
$\Rightarrow$ $(A-\lambda)$ invertible on
$\mathrm{ran}(I-P)$ with norm $\le 1/\gamma$
(spectral theorem:
all other eigenvalues at distance
$\ge\gamma$ from $\lambda$).
Thus
$\|(I-P)\phi\|
\le\gamma^{-1}\|(A-\lambda)\phi\|
\le\varepsilon/\gamma$.
\end{proof}

\begin{remark}[Rank-one structure]
\label{rem:perturbation-mechanism}
Rank-one structure is essential:
it guarantees $\mathrm{ran}(I-P)$
has uniform resolvent bound $1/\gamma$.
Gap-S supplies $\gamma=g(c_k)$
for the approximating operator;
SOT-Faster provides
$\varepsilon_n(k)/g(c_k)\to 0$.
\end{remark}

\begin{lemma}[SOT gives approx.\ eigenvectors]
\label{lem:sot-approx}
$\varepsilon_n(k)
=\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
\to 0$; no rate claimed.
\end{lemma}
\begin{proof}SOT \eqref{eq:sot}.\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:convergence}]
Fix $n\in W_{c_k}$.
$\tilde\varepsilon_n(k)
:=\|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
\Phi_n^{(\infty)}\|
\le\varepsilon_n(k)+C_\kappa c_k^{-1/4}$.
Dividing by $g(c_k)$:
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}{g(c_k)}}
    _{\to 0\,(\text{SOT-Faster})}
  +\underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to 0\,
    (\text{Rem.}\,\ref{rem:gap-implies-o})}
  \to 0,
\end{equation}
both terms independently.
Lem.~\ref{lem:approx-eig} gives
$\|(I-P_n^{(c_k)})\Phi_n^{(\infty)}\|\to 0$;
rank-one identity completes.
\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:rate}]
SOT-Rate: $\varepsilon_n(k)\le C'c_k^{-1/4}$;
$\tilde\varepsilon_n(k)\le(C'+C_\kappa)c_k^{-1/4}$;
Lem.~\ref{lem:approx-eig} gives
\eqref{eq:rate-bound}.
\end{proof}

\begin{remark}[No accumulation]
\label{rem:nacc}
Eigenvalues $\lambda_n^{(c)}$ strictly ordered
(Slepian--Pollak~\cite{Slepian1961});
Gap-S bounds finitely many in any interval.
\end{remark}

\begin{hypothesis}[Full family convergence]
\label{hyp:full-family}
Spectral measures of $\mathcal{H}_{c_k}$
converge weak-$*$ to those of $\Hlim$.
\end{hypothesis}

\begin{lemma}[Same $\mathcal{K}$, conditional]
\label{lem:same-K}
Under Hyp.~\ref{hyp:full-family},
any two Gap-S subsequences yield
the same $\mathcal{K}$.
\end{lemma}

Both subspaces are closed
$\Hlim$-invariant;
Hyp.~\ref{hyp:full-family} identifies
their spectral measures.
\end{proof}

\begin{theorem}[Canonicalization,

\label{thm:canonical}
Under Gap-S $+$ Hyp.~\ref{hyp:full-family},
$\Hlim$ is canonical on $\mathcal{K}$
up to unitary equivalence.
\end{theorem}

%%══ 9. STRUCTURAL EVIDENCE
\section{Structural Evidence for Gap-S
  and SOT-Faster}
\label{sec:heuristic}

\textbf{Non-rigorous throughout.}

\subsection{Sigmoid transition (Widom)}
Widom~\cite{Widom1964}:
$\lambda_n^{(c)}\approx
(1+e^{\pi(n-2c/\pi)/\sigma(c)})^{-1}$,
$\sigma(c)\asymp\log c$;
predicts average spacing
$\sim(\log c)^{-1}$ in $W_c$.

\subsection{Gap: density vs.\ minimum}
Widom controls density, not minimum gap.
Average spacing $\sim(\log c)^{-1}
\gg c^{-1/4}$ supports Gap-S for typical
indices; a rigorous lower bound requires
determinantal kernel methods
(Prob.~\ref{prob:sharp-gc}).

\subsection{SOT-Faster plausibility}
Small off-diagonal overlaps in
\eqref{eq:formal-expansion}
(consistent with Gap-S) predict
$\varepsilon_n(k)=O(c_k^{-1/4})$;
Prob.~\ref{prob:off-diagonal} would
make this rigorous.

\begin{remark}[Status]
Section~\ref{sec:heuristic} provides
structural evidence, not proof.
\end{remark}

%%══ 10. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent\textbf{Conjecture-level:}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove Gap-S with $g(c)\sim(\log c)^{-1}$;
natural approach: sine-kernel repulsion
for PSWF Christoffel--Darboux kernel.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$;
equivalently:
local norm-resolvent bound
$\|(\mathcal{H}_{c_k}-z)^{-1}
-(\Hlim-z)^{-1}\|\cdot g(c_k)\to 0$
at scale $g(c_k)$.
\end{problem}

\noindent\textbf{Technical-level:}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
$\varepsilon_n(k)=O(c_k^{-1/4})$;
justifies Thm.~\ref{thm:rate}.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
for $m\ne n$;
route to Prob.~\ref{prob:sot-rate}.
\end{problem}

\begin{problem}[Full family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family}.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$? Paper~XIII.
\end{problem}
