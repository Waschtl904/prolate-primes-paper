\title[Spectral Projection Stability for Strongly Convergent Operators]{%
  Spectral Projection Stability for Strongly Convergent Operators:\\
  A Reduction via Spectral Gap Control}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We identify the sharp sufficient condition for rank-one spectral
projection stability under strong operator convergence,
and show that the same mechanism is necessary when the spectral
gap remains bounded away from zero.
The key result ---
Theorem~\ref{thm:abstract} (Abstract Stability Principle) ---
is a \emph{minimal localization principle}:
rank-one projection stability follows from mere SOT convergence
if and only if (when $\gamma_k\ge c_0>0$) the residual on the
limit eigenvector is asymptotically negligible relative to the
local spectral gap.
When the gap may vanish, the ratio condition
\eqref{eq:abstract-ratio} is sufficient;
its necessity in the vanishing-gap regime is a natural open
question (Rem.~\ref{rem:iff-open}).
This is not a corollary of Davis--Kahan or Kato,
since those results require $\|A_k-A\|\to 0$,
strictly stronger than SOT.
We also establish:
(i) a No-Go result (Prop.~\ref{prop:no-go})
showing eigenvalue density alone cannot imply
projection stability;
(ii) a counterexample (Prop.~\ref{prop:counterexample})
showing SOT $+$ eigenvalue convergence does not
sufficewithout gap control;
(iii) a reduction of PSWF projection stability
to two explicit asymptotic mechanisms
(Gap-S, SOT-Faster), both structurally necessary
for any argument of this type.
\end{abstract}

\subsection{Context and goal}

\begin{center}
\fbox{\parbox{0.88\linewidth}{\centering\itshape
SOT convergence produces approximate eigenvectors\\

This is the exact quantitative obstruction preventing SOT\\

\end{center}

\medskip
Paper~X establishes $\mathcal{H}_{c_k}
\xrightarrow{\mathrm{SOT}}\Hlim$;
Paper~XI constructs an orthonormal diagonal
eigenfunction system $\{\Phi_n^{(\infty)}\}$.
The question is: does
$\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$?
This paper proves:
\begin{itemize}
\item The ratio condition \eqref{eq:abstract-ratio}
  is the \emph{sharp sufficient mechanism}
  (Thm.~\ref{thm:abstract}).
\item When $\gamma_k\ge c_0>0$,
  it is also \emph{necessary}
  (Thm.~\ref{thm:necessary-bounded-gap}).
\item Both Gap-S and SOT-Faster are
  structurally necessary for any argument
  of this type (Props.~\ref{prop:no-go},
  \ref{prop:counterexample}).
\end{itemize}

The central results are:
\begin{itemize}
\item \textbf{Thm.~\ref{thm:abstract}
  (Abstract Stability Principle):}
  ratio condition \eqref{eq:abstract-ratio}
  $\Rightarrow$ $\|P_k-P^\infty\|\to 0$.
\item \textbf{Thm.~\ref{thm:necessary-bounded-gap}
  (Necessary mechanism, bounded gap):}
  $\|P_k-P^\infty\|\to 0$ $+$ $\gamma_k\ge c_0$
  $\Rightarrow$ ratio condition.
\item \textbf{Prop.~\ref{prop:no-go}
  (No-Go for density arguments):}
  eigenvalue density alone cannot yield
  projection stability.
\item \textbf{Prop.~\ref{prop:counterexample}
  (Counterexample):}
  SOT $+$ $\lambda_k\to\lambda$,
  but $\|P_k-P^\infty\|\not\to 0$
  when gap control fails.
\item \textbf{Thm.~\ref{thm:convergence}
  (PSWF):} under Gap-S $+$ SOT-Faster,
  $\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$.
\item \textbf{Thm.~\ref{thm:rate} (Rate).}
\end{itemize}

%%══ 3. HYPOTHESES AND NO-GO
\section{Hypotheses, Structural Necessity,
  and No-Go for Density Arguments}
\label{sec:gap}

\begin{hypothesis}[Effective gap, Gap-S]
\label{hyp:gap-structural}
Let $W_c:=\{n:\lambda_n^{(c)}
\in(\delta,1-\delta)\}$.
There exists $g\colon[1,\infty)\to(0,\infty)$
with
\begin{equation}\label{eq:gap-dominates}
  g(c)\to 0
  \quad\text{and}\quad
  c^{-1/4} = o(g(c)),
\end{equation}
such that uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S: structural necessity;
  determinantal context]
\label{rem:gap-necessary}
Gap-S is \textbf{structurally necessary}:
any argument of the form
``SOT $+$ gap $\Rightarrow$
projection stability''
must control the minimal gap
relative to the eigenvalue error $c^{-1/4}$
(confirmed by Prop.~\ref{prop:counterexample}).

\textbf{Density vs.\ gap.}
Widom~\cite{Widom1964} controls
average eigenvalue density
($\sim(\log c)^{-1}$ spacing in $W_c$),
not the minimal gap.
Average spacing $\ne$ minimum gap;
Gap-S is strictly stronger
(Prop.~\ref{prop:no-go}).

\textbf{Determinantal analogy.}
In determinantal point processes,
minimal gaps are controlled via
sine-kernel repulsion.
Gap-S is the analogous deterministic
repulsion bound for the PSWF kernel;
Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{remark}[Gap-S implies $c^{-1/4}=o(g(c))$]
\label{rem:gap-implies-o}
Equivalent to $c^{-1/4}=o(g(c))$,
used in Thm.~\ref{thm:convergence}.
\end{remark}

\begin{proposition}[No-Go: density cannot
  imply projection stability]
\label{prop:no-go}
Let $(\mu_n)$ be a sequence with
$\liminf_n(\mu_n-\mu_{n+1})=0$
but with Widom-type average density.
Then there exists a sequence of bounded
self-adjoint operators
$B_k\xrightarrow{\mathrm{SOT}} B$
with eigenvalues $\mu_n^{(k)}\to\mu_n$
such that $\|Q_n^{(k)}-Q_n\|\not\to 0$
for infinitely many $n$.
\end{proposition}
\begin{proof}
Choose indices $n_j$ along which
$\gamma_{n_j,k}:=
\mu_{n_j}^{(k)}-\mu_{n_j+1}^{(k)}\to 0$.
Set $\varepsilon_{n_j}(k):=\sqrt{\gamma_{n_j,k}}$
(achievable by an explicit SOT-convergent
construction).
Then
$\varepsilon_{n_j}(k)/\gamma_{n_j,k}
=\gamma_{n_j,k}^{-1/2}\to\infty$.
By Lem.~\ref{lem:approx-eig},
$\|(I-Q_{n_j}^{(k)})\phi_{n_j}\|
\not\to 0$, so
$\|Q_{n_j}^{(k)}-Q_{n_j}\|\not\to 0$.
\end{proof}

\begin{remark}
\begin{center}\itshape
Any argument based solely on eigenvalue
density cannot yield projection stability;
minimal-gap control is not a technical
convenience but a logical necessity.
\end{center}
\end{remark}

\begin{hypothesis}[SOT-Faster]
\label{hyp:sot-faster}
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k)).
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster as microlocal
  norm convergence]
\label{rem:sot-faster-localized}
SOT-Faster is strictly stronger than
$\varepsilon_n(k)\to 0$ (which SOT gives freely).
It is a \textbf{microlocal norm convergence
condition}: norm control localised to
width $\sim g(c_k)$ around $\lambda_n^{(\infty)}$.
Equivalently: SOT-Faster follows from
$\|(\mathcal{H}_{c_k}-z)^{-1}
-(\Hlim-z)^{-1}\|\cdot g(c_k)\to 0$
for $|z-\lambda_n^{(\infty)}|\asymp g(c_k)$.
\end{remark}

\begin{hypothesis}[SOT-Rate]
\label{hyp:sot-rate}
$\varepsilon_n(k) = O(c_k^{-1/4})$.
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
$\text{SOT-Rate}
\Rightarrow\text{SOT-Faster}
\Rightarrow\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$.
SOT alone gives $\varepsilon_n(k)\to 0$,
insufficient without gap comparison.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding in eigenbasis $\{\Phi_m^{(c_k)}\}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
\end{equation}
Diagonal term $m=n$:
$\le C_\kappa^2 c_k^{-1/2}$.
Small off-diagonal overlaps (consistent
with Gap-S) predict
$\varepsilon_n(k)=O(c_k^{-1/4})$;
Prob.~\ref{prob:off-diagonal} would
make this rigorous.
\end{remark}

\subsection{WKB insufficiency}
\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
cannot determine the scaling of
$|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
without global kernel information.
\end{theorem}
\begin{proof}
Sigmoid profile of $F_c$~\cite{Widom1964};
any algebraic rate is a global claim.
\end{proof}

%%══ 4. ABSTRACT STABILITY PRINCIPLE
\section{Abstract Stability Principle}
\label{sec:abstract}

\begin{theorem}[Abstract Stability Principle;
  sufficient condition]
\label{thm:abstract}
Let $\{A_k\}$, $A$ bounded self-adjoint,
$A_k\xrightarrow{\mathrm{SOT}} A$.
Let $\phi\in\mathcal{H}$, $\|\phi\|=1$,
$A\phi=\lambda\phi$, $\lambda$ isolated,
$P^\infty$ the rank-one projection
onto $\ker(A-\lambda)$.
Assume simple $\lambda_k\in\sigma(A_k)$,
$\lambda_k\to\lambda$
\begin{equation}\label{eq:abstract-matching}
  \lambda_k\to\lambda,
\end{equation}
with gaps $\gamma_k:=
\mathrm{dist}(\lambda_k,\sigma(A_k)
\setminus\{\lambda_k\})>0$,
and suppose
\begin{equation}\label{eq:abstract-ratio}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0.
\end{equation}
Then $\|P_k - P^\infty\|\to 0$.
\end{theorem}

\begin{proof}
Set $\varepsilon_k:=\|(A_k-A)\phi\|$,
$\delta_k:=|\lambda_k-\lambda|$.
$\|(A_k-\lambda_k)\phi\|
\le\varepsilon_k+\delta_k$.
Lem.~\ref{lem:approx-eig} gives
$\|(I-P_k)\phi\|
\le(\varepsilon_k+\delta_k)/\gamma_k\to 0$
by \eqref{eq:abstract-ratio}.
Rank-one identity~\cite[Ch.~1]{Bhatia1997}:
$\|P_k-P^\infty\|\to 0$.
\end{proof}

\begin{theorem}[Necessary condition;
  bounded-gap case]
\label{thm:necessary-bounded-gap}
Under the setup of Thm.~\ref{thm:abstract},
assume additionally $\gamma_k\ge c_0>0$
for some constant $c_0$.
If $\|P_k-P^\infty\|\to 0$, then
\begin{equation}\label{eq:ratio-necessary}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0.
\end{equation}
\end{theorem}

\begin{proof}
If $\gamma_k\ge c_0>0$:
$|\lambda_k-\lambda|/\gamma_k
\le|\lambda_k-\lambda|/c_0\to 0$
by \eqref{eq:abstract-matching}.
Since $\|P_k-P^\infty\|\to 0$,
we have $\|(I-P_k)\phi\|\to 0$.
Now write
\[
  (A_k-A)\phi
  = (A_k-\lambda_k)\phi
  -(\lambda_k-\lambda)\phi
  + (\lambda-A)\phi
  + (A_k-\lambda_k)^{-1}\text{[error]};
\]
more directly:
$\|(A_k-A)\phi\|
\le\|(A_k-A)(I-P_k)\phi\|
+\|(A_k-A)P_k\phi\|$.
Since $P_k\phi\to\phi$ in norm
(because $\|P_k-P^\infty\|\to 0$
and $P^\infty\phi=\phi$),
and $P_k\phi=\alpha_k v_k$
with $(A_k-\lambda_k)v_k=0$,
we get
$\|(A_k-A)P_k\phi\|
=|\alpha_k|\cdot\|(A_k-A)v_k\|
=|\alpha_k|\cdot\|(\lambda_k-A)v_k\|
\le(|\lambda_k|+\|A\|)\|(I-P^\infty)v_k\|$.
Since $\|P_k-P^\infty\|\to 0$
$\Rightarrow$ $v_k\to\phi$ strongly
$\Rightarrow$ $\|(I-P^\infty)v_k\|\to 0$,
this term $\to 0$.
The first term:
$\|(A_k-A)(I-P_k)\phi\|
\le(\|A_k\|+\|A\|)\|(I-P_k)\phi\|\to 0$
(operator norms are bounded;
$A_k\xrightarrow{\mathrm{SOT}}A$
and $\|A_k\|\le M$ by uniform boundedness).
Hence $\|(A_k-A)\phi\|\to 0$,
and dividing by $\gamma_k\ge c_0$
gives \eqref{eq:ratio-necessary}.
\end{proof}

\begin{remark}[Open: full iff when
  $\gamma_k\to 0$]
\label{rem:iff-open}
When $\gamma_k\to 0$,
the direction
``$\|P_k-P^\infty\|\to 0$
$\Rightarrow$
$\|(A_k-A)\phi\|/\gamma_k\to 0$''
requires a rate on $\|(A_k-A)\phi\|$
relative to $\gamma_k$.
From $\|P_k-P^\infty\|\to 0$
one obtains $\|(A_k-A)\phi\|\to 0$
(Thm.~\ref{thm:necessary-bounded-gap}
proof, independent of $\gamma_k$),
but whether this decays faster than
$\gamma_k$ depends on the specific
spectral geometry and is not determined
by convergence of projections alone.
This is a genuine open question
that we do not claim to resolve;
it is precisely the analytic difficulty
encoded in Hyp.~\ref{hyp:sot-faster}.
\end{remark}

\begin{remark}[Novelty: minimal
  localization principle]
\label{rem:abstract-novelty}
Thm.~\ref{thm:abstract} is
\textbf{not a new perturbation theorem},
but a \textbf{minimal localization principle}:
it extracts the exact quantitative mechanism
missing from SOT,
replacing global norm control
$\|A_k-A\|\to 0$
(Davis--Kahan~\cite{DavisKahan1970},
Kato~\cite{Kato1966})
by local spectral geometry.
The bounded-gap iff
(Thms.~\ref{thm:abstract}
$+$~\ref{thm:necessary-bounded-gap})
does not appear in standard references.
\end{remark}

\begin{remark}[What Thm.~\ref{thm:abstract}
  really says]
\label{rem:abstract-meaning}
Projection stability requires only:
(i)~eigenvalue convergence;
(ii)~the ratio condition
\eqref{eq:abstract-ratio},
a local norm-resolvent estimate
at a single eigenvalue.
This replaces global norm control
by local spectral geometry.
\end{remark}

\begin{proposition}[Counterexample:
  SOT but projection failure]
\label{prop:counterexample}
There exist bounded self-adjoint $\{A_k\}$,
$A$ on $\ell^2$ with
$A_k\xrightarrow{\mathrm{SOT}} A$,
a simple $\lambda\in\sigma(A)$,
and $\lambda_k\in\sigma(A_k)$
with $\lambda_k\to\lambda$,
such that $\|P_k-P^\infty\|\not\to 0$.
\end{proposition}
\begin{proof}
Let $\phi=e_1$, $\lambda=0$,
$P^\infty=e_1\otimes e_1$.
Set $\gamma_k:=k^{-1/2}$,
$v_k:=\tfrac{1}{\sqrt{2}}(e_1+e_2)$,
\[
  A:=\sum_{n\ge 2} n^{-1}(e_n\otimes e_n),
  \quad
  A_k:=\gamma_k(v_k\otimes v_k)
  +\sum_{n\ge 2} n^{-1}(e_n\otimes e_n).
\]
(i) $A_k\xrightarrow{\mathrm{SOT}} A$:
for any $f\in\ell^2$,
$\|(A_k-A)f\|=\gamma_k|\ip{v_k}{f}|
\|v_k\|\to 0$.
(ii) $\lambda_k=\gamma_k\to 0$.
(iii) Spectral gap of $A_k$ at $\lambda_k$:
$\gamma_k' \sim \gamma_k = k^{-1/2}$.
(iv) $\|(A_k-A)\phi\|
=\gamma_k|\ip{v_k}{e_1}|^2
=\gamma_k/2$.
Hence
$\varepsilon_k/\gamma_k'
\to 1/2\ne 0$,
so \eqref{eq:abstract-ratio} fails.
By Thm.~\ref{thm:abstract}
(contrapositive):
$\|P_k-P^\infty\|\not\to 0$.
\end{proof}

\begin{remark}[Sharpness]
\label{rem:counterexample-meaning}
Prop.~\ref{prop:counterexample} confirms:
SOT $+$ eigenvalue convergence
$\not\Rightarrow$ projection stability.
The ratio \eqref{eq:abstract-ratio}
with $\gamma_k\to 0$ is not
automatically satisfied;
this is exactly Gap-S $+$ SOT-Faster
in the PSWF application.
\end{remark}

%%══ 5. PSWF APPLICATION
\section{PSWF Application: Main Theorems}
\label{sec:main}

The following reduces PSWF projection stability
to two explicit analytic conditions.
Thms.~\ref{thm:convergence}
and~\ref{thm:rate} apply Thm.~\ref{thm:abstract}
with $A_k=\mathcal{H}_{c_k}$, $A=\Hlim$,
$\phi=\Phi_n^{(\infty)}$,
$\lambda=\lambda_n^{(\infty)}$,
$\lambda_k=\lambda_n^{(c_k)}$,
$\gamma_k=g(c_k)$.

\begin{theorem}[Projection convergence]
\label{thm:convergence}
Assume Lem.~\ref{lem:inherited},
Hyp.~\ref{hyp:gap-structural} (Gap-S),
and Hyp.~\ref{hyp:sot-faster} (SOT-Faster).
For each fixed $n\in W_{c_k}$:
\[
  \|P_n^{(c_k)} - P_n^{(\infty)}\|\to 0.
\]
In particular,
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
strongly (up to phase)
and $\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$
(Rem.~\ref{rem:nacc}).
\end{theorem}

\begin{theorem}[Rate]
\label{thm:rate}
Assume additionally Hyp.~\ref{hyp:sot-rate}.
For $n\le\kappa c_k$, $n\in W_{c_k}$,
uniformly in $n$:
\[
  \|P_n^{(c_k)} - P_n^{(\infty)}\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}.
\]
If $g(c)\sim(\log c)^{-1}$:
rate $O(c_k^{-1/4}\log c_k)$.
\end{theorem}

\begin{remark}[Phase]
Phase fixed by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
for large $k$.
\end{remark}

%%══ 6. PROOFS
\section{Proofs}
\label{sec:proj}

\begin{lemma}[Approx.\ eigenvector
  $\Rightarrow$ projection bound]
\label{lem:approx-eig}
Let $A$ bounded self-adjoint,
$\lambda$ simple with gap $\gamma>0$,
$P$ rank-one onto $\ker(A-\lambda)$.
For $\|\phi\|=1$,
$\|(A-\lambda)\phi\|\le\varepsilon$:
\[
  \|(I-P)\phi\|\le\varepsilon/\gamma.
\]
\end{lemma}
\begin{proof}
$(A-\lambda)\phi=(A-\lambda)(I-P)\phi$
since $(A-\lambda)P=0$.
$\lambda$ simple $\Rightarrow$
$(A-\lambda)$ invertible on $\mathrm{ran}(I-P)$
with $\|(A-\lambda)^{-1}\|\le 1/\gamma$
(spectral theorem).
Hence $\|(I-P)\phi\|\le\varepsilon/\gamma$.
\end{proof}

\begin{remark}[Rank-one structure]
\label{rem:perturbation-mechanism}
Rank-one structure gives a uniform
resolvent bound $1/\gamma$ on $\mathrm{ran}(I-P)$.
Gap-S supplies $\gamma=g(c_k)$ for $A_k$;
SOT-Faster gives $\varepsilon_n(k)/g(c_k)\to 0$.
\end{remark}

\begin{lemma}[SOT gives approx.\ eigenvectors]
\label{lem:sot-approx}
$\varepsilon_n(k)=
\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
\to 0$; no rate.
\end{lemma}
\begin{proof}SOT \eqref{eq:sot}.\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:convergence}]
Fix $n\in W_{c_k}$.
\[
  \tilde\varepsilon_n(k)
  :=\|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
  \Phi_n^{(\infty)}\|
  \le\varepsilon_n(k)+C_\kappa c_k^{-1/4}.
\]
Dividing by $g(c_k)$:
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}{g(c_k)}}
    _{\to 0\,(\text{SOT-Faster})}
  +\underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to 0\,
    (\text{Rem.}\,\ref{rem:gap-implies-o})}
  \to 0.
\end{equation}
Lem.~\ref{lem:approx-eig} gives
$\|(I-P_n^{(c_k)})\Phi_n^{(\infty)}\|\to 0$;
rank-one identity completes.
\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:rate}]
SOT-Rate: $\varepsilon_n(k)\le C'c_k^{-1/4}$;
Lem.~\ref{lem:approx-eig} with
$\tilde\varepsilon_n(k)\le(C'+C_\kappa)c_k^{-1/4}$
gives the bound.
\end{proof}

\begin{remark}[No accumulation]
\label{rem:nacc}
Gap-S bounds finitely many
$\lambda_n^{(\infty)}$ in any interval;
no accumulation in $(0,1)$.
\end{remark}

%%══ 9. STRUCTURAL EVIDENCE
\section{Structural Evidence for Gap-S
  and SOT-Faster}
\label{sec:heuristic}

\textbf{Non-rigorous throughout.}

\subsection{Sigmoid transition (Widom)}
Widom~\cite{Widom1964} predicts average
spacing $\sim(\log c)^{-1}$ in $W_c$,
consistent with Gap-S.

\subsection{Density vs.\ minimum gap}
Widom controls density, not minimum gap.
Typical spacing $\sim(\log c)^{-1}
\gg c^{-1/4}$ supports Gap-S;
rigorous lower bound needs
determinantal methods
(Prob.~\ref{prob:sharp-gc}).

\subsection{SOT-Faster plausibility}
Small off-diagonal overlaps in
\eqref{eq:formal-expansion}
predict $\varepsilon_n(k)=O(c_k^{-1/4})$;
Prob.~\ref{prob:off-diagonal}.

\begin{remark}[Status]
Section~\ref{sec:heuristic} gives
structural evidence only.
\end{remark}

%%══ 10. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent\textbf{Conjecture-level:}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove Gap-S with $g(c)\sim(\log c)^{-1}$
via sine-kernel repulsion.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$;
equivalently: local norm-resolvent bound
at scale $g(c_k)$.
\end{problem}

\begin{problem}[Full iff, vanishing gap]
\label{prob:full-iff}
Characterise when
$\|P_k-P^\infty\|\to 0$
$\Rightarrow$
$\|(A_k-A)\phi\|/\gamma_k\to 0$
when $\gamma_k\to 0$.
This is the precise analytic difficulty
encoded in Hyp.~\ref{hyp:sot-faster}.
\end{problem}

\noindent\textbf{Technical-level:}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
$\varepsilon_n(k)=O(c_k^{-1/4})$.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
Estimate
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$,
$m\ne n$.
\end{problem}

\begin{problem}[Full family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family}.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$? Paper~XIII.
\end{problem}
