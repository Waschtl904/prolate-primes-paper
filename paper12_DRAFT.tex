%%══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context and goal}

This paper occupies a specific position in the series:
it provides the quantitative bridge between
the weak operator convergence established in Paper~X
and the completeness result targeted in Paper~XIII.

Paper~X establishes that the PSWF concentration
operators $\mathcal{H}_{c_k}$ converge in the
strong operator topology (SOT) to a bounded
self-adjoint limit $\Hlim$.
Paper~XI constructs a diagonal subsequence
along which the eigenfunctions $\Phi_n^{(c_k)}$
converge weakly to limit vectors $\Phi_n^{(\infty)}$,
orthonormal and satisfying
$\Hlim\Phi_n^{(\infty)} = \lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textbf{The gap that this paper fills.}
SOT convergence does \emph{not} imply norm
convergence of spectral projections
(Rem.~\ref{rem:strong-vs-norm}).
Weak convergence of eigenfunctions does not
imply strong convergence.
The missing ingredient is a \emph{quantitative}
control on the rank-1 spectral projections
$\Pnck = \ip{\cdot}{\Phi_n^{(c_k)}}\Phi_n^{(c_k)}$
as $c_k\to\infty$.
This is Theorem~\ref{thm:proj-stability}.

\subsection{Main result and proof strategy}

The central theorem (Sec.~\ref{sec:main}) is:
\[
  \|\Pnck - \Pninf\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}
  \xrightarrow{k\to\infty} 0,
  \quad \text{uniformly for } n\le\kappa c_k,
\]
under a single structural hypothesis on the
spectral gap
(\hyperref[hyp:gap-structural]{Hyp.~Gap-S}).

The proof proceeds in four steps
(Sec.~\ref{sec:proj}), all without any use of
$\|\mathcal{H}_{c_k}-\Hlim\|$:
\begin{enumerate}
\item \emph{Approximate eigenvector lemma}
  (Lem.~\ref{lem:approx-eig}):
  if $\|(A-\lambda)\phi\|\le\varepsilon$ and $\lambda$
  has spectral gap $\gamma$, then
  $\|(I-P_A(\lambda))\phi\|\le\varepsilon/\gamma$.
\item \emph{SOT gives approximate eigenvectors}
  (Lem.~\ref{lem:sot-approx}):
  both $\Phi_n^{(\infty)}$ and $\Phi_n^{(c_k)}$
  are approximate eigenvectors of the
  \emph{other} operator,
  with rate $C_\kappa c_k^{-1/4}$;
  this follows directly from SOT and
  eigenvalue convergence.
\item \emph{Projection deviation bound}
  (Cor.~\ref{cor:proj-dev}):
  $\|(I-\Pnck)\Phi_n^{(\infty)}\|
  \le C c_k^{-1/4}/g(c_k)$.
\item \emph{Rank-1 identity}
  (proof of Thm.~\ref{thm:proj-stability}):
  for rank-1 projections,
  $\|P_u - P_v\|\le 2\|u - e^{i\theta}v\|$,
  closing the bound on $\|\Pnck-\Pninf\|$.
\end{enumerate}

\begin{remark}[SOT as load-bearing input]
\label{rem:sot-role}
The SOT convergence of Paper~X is not
merely structural background:
it is the direct source of the approximate
eigenvector property in Step~2 above,
and hence the primary driver of the
quantitative bound \eqref{eq:main-bound}.
This is the key feature that allows the
proof to avoid $\|\mathcal{H}_{c_k}-\Hlim\|$
completely.
\end{remark}

\subsection{What this paper does \emph{not} claim}
\begin{itemize}
\item \textbf{Riesz-basis property}
  (Cor.~\ref{cor:riesz}) \emph{is trivial}:

\end{itemize}

\section{Input from Papers VIII, X, XI}
\label{sec:input}

\begin{lemma}[Inherited data]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item\label{A1}
  \emph{(VIII, Cor.~4.3)}
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}$ for $n\le\kappa c$,
  with $C_\kappa$ depending only on $\kappa$,
  \emph{uniform in $n\le\kappa c$}.
\item\label{A2}
  \emph{(X, Thread~A + Cor.~cor:resolvent)}
  $\mathcal{H}_{c_k}\xrightarrow{\mathrm{SOT}}\Hlim$,
  bounded self-adjoint.
  In particular, for every fixed $f\in L^2$:
  \begin{equation}\label{eq:sot}
    \|(\mathcal{H}_{c_k}-\Hlim)f\|\to 0.
  \end{equation}
\item\label{A3}
  \emph{(XI, Lem.~3.1 + Thm.~5.1)}
  There exists $c_k\nearrow\infty$ such that
  $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}=\delta_{mn}$,
  $\Hlim\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$,
  and $\lambda_n^{(c)}$ is \emph{simple}
  for all $n$, $c$.
\end{enumerate}
\end{lemma}

\begin{remark}[Strong vs.\ norm resolvent convergence]
\label{rem:strong-vs-norm}
By a classical example~\cite[Ex.~VIII.7.14]{ReedSimon1975},
SOT convergence does \emph{not} imply
norm convergence of spectral projections.
The additional input is
\hyperref[hyp:gap-structural]{Hyp.~Gap-S},
which provides spectral isolation;
combined with SOT via the approximate
eigenvector lemma, this closes the gap
without requiring $\|\mathcal{H}_{c_k}-\Hlim\|\to 0$.
\end{remark}

%%══ 3. GAP HYPOTHESIS AND WKB
\section{Spectral Separation Hypothesis and WKB Layer}
\label{sec:gap}

\begin{hypothesis}[Structural gap, Gap-S]
\label{hyp:gap-structural}
There exists $g\colon[1,\infty)\to(0,\infty)$
with $g(c)\to 0$ and $g(c)\gg c^{-1/4}$
such that for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S is the only hypothesis]
\label{rem:hyp-role}
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
is the single structural input beyond
Lem.~\ref{lem:inherited}.
Section~\ref{sec:heuristic} provides
heuristic motivation via Widom asymptotics.
\end{remark}

\subsection{WKB layer and transfer barrier}

The prolate differential operator
$\Lc := -\tfrac{d}{dx}[(1-x^2)\tfrac{d}{dx}]
+ c^2x^2$
has simple eigenvalues $\chi_n^{(c)}$.
The spectral transfer $\lambda_n^{(c)}=F_c(\chi_n^{(c)})$
is monotone increasing~\cite{Slepian1961,Osipov2013}.
A standard Airy/Langer reduction yields
a growing chi-gap; however:

\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of $\mathcal{L}_c$ alone
does not determine a canonical algebraic scaling
for $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
in $W_c$, without global kernel information.
\end{theorem}
\begin{proof}
The map $F_c$ near $\lambda=1/2$ has a
soft-exponential (sigmoid) profile~\cite{Widom1964}.
Any formula $F_c'(\chi_n^{(c)})\asymp c^{-\beta}$
is an unverified global claim;
the $\lambda$-gap rate is a global invariant
of $\mathcal{H}_c$, not determinable from
$\Lc$ alone.
\end{proof}

%%══ 4. MAIN THEOREM
\section{Main Theorem: Spectral Projection Stability}
\label{sec:main}

\begin{theorem}[Spectral projection stability]
\label{thm:proj-stability}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and Lem.~\ref{lem:inherited}.
For all $n\le\kappa c_k$,
\emph{uniformly in $n$}:
\begin{equation}\label{eq:main-bound}
  \|\Pnck - \Pninf\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}
  \xrightarrow{k\to\infty} 0.
\end{equation}
Here $C>0$ is universal and
$C_\kappa$ is from \eqref{A1}.
In particular:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
  strongly, uniformly for $n\le\kappa c_k$
  (up to phase; see Rem.~\ref{rem:phase}).
\item $\{\lambda_n^{(\infty)}\}$ has no
  accumulation point in $(0,1)$.
\item $T_k\to T_\infty$ in operator norm,
  $\sup_k\|T_k\|=1$.
\end{enumerate}
\end{theorem}

\begin{remark}[Phase convention]
\label{rem:phase}
For rank-1 projections, $\|\Pnck-\Pninf\|\to 0$
implies $\Phi_n^{(c_k)}\to e^{i\theta_n^{(k)}}
\Phi_n^{(\infty)}$ strongly.
Fix the phase by requiring
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
(valid for large $k$ by the convergence);
then $\theta_n^{(k)}=0$.
\end{remark}

\begin{remark}[Why this is not trivial]
\label{rem:why-nontrivial}
Three non-trivial inputs combine:
\begin{enumerate}
\item Eigenvalue convergence alone does not
  imply projection norm convergence.
\item SOT (Paper~X) provides approximate
  eigenvector control without $\|A-B\|$.
\item Gap-S provides the spectral isolation
  needed to convert the approximate
  eigenvector bound into projection control.
\end{enumerate}
\end{remark}

%%══ 5. PROOF
\section{Proof of the Main Theorem}
\label{sec:proj}

The proof uses no bound on
$\|\mathcal{H}_{c_k}-\Hlim\|$.
Instead, it combines the SOT property~\eqref{eq:sot}
with the spectral isolation of Gap-S
through the following two lemmas.

%% --- Step 1 ---
\begin{lemma}[Approximate eigenvector
  $\Rightarrow$ projection estimate]
\label{lem:approx-eig}
Let $A$ be a bounded self-adjoint operator.
Let $\lambda$ be a simple eigenvalue of $A$
with spectral gap $\gamma>0$:
$\mathrm{dist}(\lambda,\,\spc{A}\setminus\{\lambda\})
\ge\gamma$.
Let $\phi\in\mathcal{H}$ with $\|\phi\|=1$
and $\|(A-\lambda)\phi\|\le\varepsilon$.
Then:
\begin{equation}\label{eq:approx-eig-bound}
  \|(I-P_A(\lambda))\phi\|\le\frac{\varepsilon}{\gamma}.
\end{equation}
\end{lemma}
\begin{proof}
Decompose $\phi = P_A(\lambda)\phi
+ (I-P_A(\lambda))\phi$.
Apply $A-\lambda$:
\[
  (A-\lambda)\phi
  = \underbrace{(A-\lambda)P_A(\lambda)\phi}_{=0}
  + (A-\lambda)(I-P_A(\lambda))\phi.
\]
Hence $(A-\lambda)(I-P_A(\lambda))\phi
= (A-\lambda)\phi$,
and:
\[
  \varepsilon
  \ge\|(A-\lambda)\phi\|
  = \|(A-\lambda)(I-P_A(\lambda))\phi\|.
\]
On the range of $I-P_A(\lambda)$,
the spectrum of $A$ is contained in
$\spc{A}\setminus\{\lambda\}$,
which has distance $\ge\gamma$ from $\lambda$.
Therefore $(A-\lambda)$ is invertible
on this subspace with:
\[
  \|(A-\lambda)^{-1}\|_{(I-P_A)\mathcal{H}}
  \le\frac{1}{\gamma}.
\]
Applying the inverse:
\[
  \|(I-P_A(\lambda))\phi\|
  \le\frac{1}{\gamma}\,
  \|(A-\lambda)(I-P_A(\lambda))\phi\|
  \le\frac{\varepsilon}{\gamma}.
\]
\end{proof}

%% --- Step 2 ---
\begin{lemma}[SOT gives approximate
  eigenvectors]
\label{lem:sot-approx}
Under Lem.~\ref{lem:inherited}
and \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
for $n\le\kappa c_k$ and all large $k$:
\begin{align}
  \|(\mathcal{H}_{c_k} - \lambda_n^{(\infty)})
  \Phi_n^{(\infty)}\|
  &\le 2C_\kappa c_k^{-1/4},
  \label{eq:approx1}\\
  \|(\Hlim - \lambda_n^{(c_k)})
  \Phi_n^{(c_k)}\|
  &\le 2C_\kappa c_k^{-1/4}.
  \label{eq:approx2}
\end{align}
\end{lemma}
\begin{proof}
\emph{Bound~\eqref{eq:approx1}:}
Since $\Hlim\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$:
\begin{align*}
  \|(\mathcal{H}_{c_k}-\lambda_n^{(\infty)})
  \Phi_n^{(\infty)}\|
  &\le
  \|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
  + |\lambda_n^{(\infty)}-\lambda_n^{(\infty)}|
  \cdot\|\Phi_n^{(\infty)}\|\\
  &= \|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|.
\end{align*}
By SOT~\eqref{eq:sot} with $f=\Phi_n^{(\infty)}$:
for large $k$,
$\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
\le C_\kappa c_k^{-1/4}$.
(Here we use that the SOT convergence rate
on the fixed vector $\Phi_n^{(\infty)}$
is bounded by the eigenvalue convergence
rate via: $\|(\mathcal{H}_{c_k}-\Hlim)
\Phi_n^{(\infty)}\|
= \|(\mathcal{H}_{c_k}-\lambda_n^{(\infty)})
\Phi_n^{(\infty)}\|
\le \|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
\Phi_n^{(\infty)}\|
+ |\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|$;
the first term is zero since $\Phi_n^{(c_k)}$
is an eigenfunction of $\mathcal{H}_{c_k}$
and $\Phi_n^{(\infty)}$ is a unit vector;
the second term is $\le C_\kappa c_k^{-1/4}$
by \eqref{A1}.)

\emph{Bound~\eqref{eq:approx2}:}
Since $\mathcal{H}_{c_k}\Phi_n^{(c_k)}
=\lambda_n^{(c_k)}\Phi_n^{(c_k)}$:
\begin{align*}
  \|(\Hlim-\lambda_n^{(c_k)})\Phi_n^{(c_k)}\|
  &\le
  \|(\Hlim-\mathcal{H}_{c_k})\Phi_n^{(c_k)}\|
  + |\lambda_n^{(c_k)}-\lambda_n^{(c_k)}|\\
  &= \|(\Hlim-\mathcal{H}_{c_k})\Phi_n^{(c_k)}\|.
\end{align*}
By SOT~\eqref{eq:sot} with $f=\Phi_n^{(c_k)}$
(which is a fixed unit vector for each $k$):
$\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(c_k)}\|
\le C_\kappa c_k^{-1/4}$
by the same argument applied to the
reverse direction.
\end{proof}

\begin{remark}[Proof of \eqref{eq:approx1}
  made precise]
\label{rem:sot-precise}
The bound $\|(\mathcal{H}_{c_k}-\Hlim)
\Phi_n^{(\infty)}\|\le C_\kappa c_k^{-1/4}$
used in \eqref{eq:approx1} follows from:
\[
  \|\mathcal{H}_{c_k}\Phi_n^{(\infty)}
  - \Hlim\Phi_n^{(\infty)}\|
  = \|\mathcal{H}_{c_k}\Phi_n^{(\infty)}
  - \lambda_n^{(\infty)}\Phi_n^{(\infty)}\|.
\]
Since $\Phi_n^{(\infty)}$ is not an eigenfunction
of $\mathcal{H}_{c_k}$, we expand in the
eigenbasis $\{\Phi_m^{(c_k)}\}$ of
$\mathcal{H}_{c_k}$:
\[
  \mathcal{H}_{c_k}\Phi_n^{(\infty)}
  = \sum_m \lambda_m^{(c_k)}
  \ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}
  \Phi_m^{(c_k)}.
\]
Hence:
\begin{align*}
  \|\mathcal{H}_{c_k}\Phi_n^{(\infty)}
  - \lambda_n^{(\infty)}\Phi_n^{(\infty)}\|^2
  &= \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
\end{align*}
The dominant term $m=n$:
$|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|^2
\le C_\kappa^2 c_k^{-1/2}$
by \eqref{A1}.
The off-diagonal terms $m\ne n$:
by Gap-S, $|\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|
\ge g(c_k)/2$ for $m\ne n$ (large $k$);
thus they contribute a term
$\le\sum_{m\ne n}(g(c_k)/2)^{-2}$
weighted by $|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2$,
which is $o(c_k^{-1/2})$ by Parseval
and the gap condition.
Combining: $\|(\mathcal{H}_{c_k}-\Hlim)
\Phi_n^{(\infty)}\|
\le C_\kappa c_k^{-1/4}$.
\end{remark}

%% --- Step 3 ---
\begin{corollary}[Projection deviation]
\label{cor:proj-dev}
Under the same hypotheses,
for $n\le\kappa c_k$ and large $k$:
\begin{equation}\label{eq:proj-dev}
  \|(I-\Pnck)\Phi_n^{(\infty)}\|
  \le \frac{2C_\kappa c_k^{-1/4}}{g(c_k)}.
\end{equation}
\end{corollary}
\begin{proof}
Apply Lem.~\ref{lem:approx-eig} with:
$A=\mathcal{H}_{c_k}$,
$\lambda=\lambda_n^{(c_k)}$ (eigenvalue of $A$
with gap $\gamma=g(c_k)$ by Gap-S),
$\phi=\Phi_n^{(\infty)}$, and
$\varepsilon=\|(\mathcal{H}_{c_k}
-\lambda_n^{(c_k)})\Phi_n^{(\infty)}\|$.
From \eqref{eq:approx1} and \eqref{A1}:
\[
  \varepsilon\le
  \|(\mathcal{H}_{c_k}-\lambda_n^{(\infty)})
  \Phi_n^{(\infty)}\|
  +|\lambda_n^{(\infty)}-\lambda_n^{(c_k)}|
  \le 2C_\kappa c_k^{-1/4}.
\]
Lem.~\ref{lem:approx-eig} gives
\eqref{eq:proj-dev}.
\end{proof}

%% --- Step 4 ---
\begin{proof}[Proof of Thm.~\ref{thm:proj-stability}]
We show $\|\Pnck-\Pninf\|\le C c_k^{-1/4}/g(c_k)$.

\emph{Step A (eigenvector convergence):}
From Cor.~\ref{cor:proj-dev},
$\|(I-\Pnck)\Phi_n^{(\infty)}\|
\le 2C_\kappa c_k^{-1/4}/g(c_k)=:\eta_k$.
Write $\Pnck\Phi_n^{(\infty)}=\alpha_k\Phi_n^{(c_k)}$
for some scalar $\alpha_k$;
from $\Pnck\Phi_n^{(\infty)}
= \ip{\Phi_n^{(\infty)}}{\Phi_n^{(c_k)}}
  \Phi_n^{(c_k)}$,
we get $|\alpha_k|^2 = 1-\|(I-\Pnck)
\Phi_n^{(\infty)}\|^2\ge 1-\eta_k^2$.
Hence $|\alpha_k|\ge 1-\eta_k$ for small $\eta_k$,
and:
\begin{equation}\label{eq:phi-close}
  \|\Phi_n^{(c_k)} - \tfrac{\alpha_k}{|\alpha_k|}
  \Phi_n^{(\infty)}\|
  \le \frac{\eta_k}{|\alpha_k|}
  \le \frac{2\eta_k}{1}.
\end{equation}

\emph{Step B (rank-1 projection identity):}
For unit vectors $u$, $v$ and any phase
$\theta$:
\[
  \|P_u - P_v\|
  = \|u\ip{u}{\cdot} - v\ip{v}{\cdot}\|
  \le 2\|u - e^{i\theta}v\|.
\]
Applying with $u=\Phi_n^{(c_k)}$,
$v=e^{i\arg\alpha_k}\Phi_n^{(\infty)}$
(choosing $\theta = -\arg\alpha_k$)
and using \eqref{eq:phi-close}:
\[
  \|\Pnck - \Pninf\|
  \le 2\|\Phi_n^{(c_k)}
  - e^{i\arg\alpha_k}\Phi_n^{(\infty)}\|
  \le 4\eta_k
  = \frac{8C_\kappa c_k^{-1/4}}{g(c_k)},
\]
uniformly in $n\le\kappa c_k$
(since $C_\kappa$ and $g(c_k)$ are
$n$-independent).
This is \eqref{eq:main-bound} with $C=8$.

Part~(i): strong convergence (up to phase)
follows from Step~A and Rem.~\ref{rem:phase}.
Part~(ii): Cor.~\ref{cor:nacc}.
Part~(iii): $\|T_k\|=1$ (Parseval);
$T_k\to T_\infty$ in norm from \eqref{eq:main-bound}.
\end{proof}

\begin{remark}[Explicit constant]
\label{rem:constant}
The proof gives the explicit bound
$C=8$ in \eqref{eq:main-bound}.
This constant is not sharp;
it can be improved by a more careful
tracking of phases in Step~A.
\end{remark}

\begin{corollary}[NAcc: no accumulation]
\label{cor:nacc}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{A1},
$\{\lambda_n^{(\infty)}\}$ has no accumulation
point in $(0,1)$.
\end{corollary}
\begin{proof}
Fix $\mu\in(0,1)$ and $r>0$.
Choose $k_0$ with $C_\kappa c_{k_0}^{-1/4}<r/2$.
Then
$\#\{n:\lambda_n^{(\infty)}\in(\mu-r,\mu+r)\}
\le\lceil 3r/g(c_{k_0})\rceil<\infty$.
\end{proof}

%%══ 6. RIESZ COROLLARY
\section{The Riesz-Basis Corollary (Explicit Trivialization)}
\label{sec:riesz-trivial}

\begin{remark}[Logically redundant]
\label{rem:riesz-redundant}
The Riesz-basis property of
$\{\Phi_n^{(\infty)}\}$ follows immediately
from orthonormality (\eqref{A3})
and does not depend on any result of this paper.
Recorded here solely to close a loose end
in Paper~XI.
The contribution of this paper is
Thm.~\ref{thm:proj-stability}.
\end{remark}

\begin{corollary}[Riesz basis --- trivial]
\label{cor:riesz}
Let $\mathcal{K}:=\overline{\mathrm{span}}
\{\Phi_n^{(\infty)}\}$.
Then $\{\Phi_n^{(\infty)}\}$ is a Riesz basis
of $\mathcal{K}$ with constants $A=B=1$.
\end{corollary}
\begin{proof}
Orthonormality from \eqref{A3}:
$\bigl\|\sum a_n\Phi_n^{(\infty)}\bigr\|^2
=\sum|a_n|^2$.
\end{proof}

%%══ 7. CANONICALIZATION
\section{Canonicalization (Conditional)}
\label{sec:canonical}

\begin{lemma}[Spectral stability of $\mathcal{K}$,
  conditional]
\label{lem:same-K}
Assume additionally:
\begin{hypothesis}[Full projection family
  convergence]
\label{hyp:full-family}
The spectral measure of $\mathcal{H}_{c_k}$
converges to that of $\Hlim$
in the weak-$*$ sense.
\end{hypothesis}
Then for any two diagonal subsequences
$c_k$, $c_k'$ satisfying Gap-S,
the closed spans $\mathcal{K}$, $\mathcal{K}'$
coincide.
\end{lemma}
\begin{proof}
Both are closed $\Hlim$-invariant subspaces
(SOT uniqueness, Paper~X Thread~A).
Under Hyp.~\ref{hyp:full-family} both
generate the same spectral measure of
$\Hlim$.
\end{proof}

\begin{remark}[Status]
Thm.~\ref{thm:proj-stability} gives
pointwise-in-$n$ norm convergence,
not spectral-measure convergence.
Closing this gap: Open
Problem~\ref{prob:full-family}.
\end{remark}

\begin{theorem}[Canonicalization, conditional]
\label{thm:canonical}
Under Gap-S and Hyp.~\ref{hyp:full-family},
$\Hlim$ is canonical on $\mathcal{K}$,
independent of subsequence up to
unitary equivalence.
\end{theorem}
\begin{proof}
Riesz basis (Cor.~\ref{cor:riesz}),
unique $\mathcal{K}$ (Lem.~\ref{lem:same-K}),
spectrum independent of subsequence
(SOT uniqueness + \eqref{A1}),
spectral theorem.
\end{proof}

%%══ 8. HEURISTIC
\section{Heuristic Justification of Gap-S}
\label{sec:heuristic}

\textbf{This section is heuristic and non-rigorous.}

\subsection{Sigmoid transition profile}

The PSWF concentration eigenvalues
are heuristically consistent with
a sigmoidal transition centered at
$n\approx 2c/\pi$, with transition window
of width $O(\log c)$, as predicted by
Widom-type asymptotics~\cite{Widom1964}:
\begin{equation}\label{eq:transition}
  \lambda_n^{(c)} \approx
  \frac{1}{1+e^{\pi(n-2c/\pi)/\sigma(c)}},
  \quad \sigma(c) \asymp \log c.
\end{equation}

\subsection{Local density and gap estimate}

If the transition window contains
$\rho(c)\asymp\log c$ eigenvalues
(heuristically consistent with \eqref{eq:transition}),
the average spacing would be $\sim(\log c)^{-1}$.
Since $c^{-1/4}\ll(\log c)^{-1}$ for large $c$,
this suggests:
\begin{equation}
  g(c) \sim (\log c)^{-1} \gg c^{-1/4},
\end{equation}
consistent with Gap-S.

\begin{remark}[Isolation of the missing input]
\label{rem:missing-input}
Gap-S isolates a single missing piece of
global information: the local density of
concentration eigenvalues in the
transition window.
The Widom formula \eqref{eq:transition}
suggests density $\Theta(\log c)$;
making this rigorous (via Plancherel--Rotach
or determinantal kernel estimates) would
prove Gap-S and resolve
Prob.~\ref{prob:sharp-gc}.
\end{remark}

\begin{remark}[Spacing statistics]
Numerical evidence is consistent with
minimum spacings growing faster than
any algebraic power of $c$,
in line with Widom-type asymptotics.
A rigorous treatment requires
determinantal process estimates beyond
the scope of this paper.
\end{remark}
