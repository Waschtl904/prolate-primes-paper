\begin{abstract}
We establish the Riesz-basis criterion for the limiting
orthonormal system $\{\Phi_n^{(\infty)}\}$ constructed in
Paper~XI along the diagonal subsequence $c_k$.
The central equivalence is:
\[
  \{\Phi_n^{(\infty)}\} \text{ is a Riesz basis of }\mathcal{K}
  \;\Longleftrightarrow\;
  \text{(Gap-S)} + \text{(Proj-Stab)}.
\]
The argument is organized in three layers.
Section~\ref{sec:gap} establishes the spectral
separation structure at the differential-operator
level (WKB/Airy analysis of $\mathcal{L}_c$),
and proves in Thm.~\ref{thm:wkb-indeterminacy}
that the WKB analysis of $\mathcal{L}_c$ alone
does not determine a canonical algebraic scaling
for the $\lambda$-gap in the transition regime,
without global kernel information about $\mathcal{H}_c$.
Section~\ref{sec:proj} introduces the central
technical tool: norm convergence of the spectral
projections $P_n^{(c_k)}\to\Pninf$
(Lem.~\ref{lem:proj-conv}),
which replaces all frame-perturbation arguments
and provides a direct structural route to
Riesz-basis stability.
All downstream results (NAcc, Riesz basis,
canonicalization) follow from the structural
gap hypothesis \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
via the projection framework.
\end{abstract}

      (iii) Riesz/canonicalization layer
      (Secs.~\ref{sec:riesz}--\ref{sec:canonical}).
\item \textbf{Central new lemma:}
      Lem.~\ref{lem:proj-conv}
      (norm convergence of spectral projections

      This replaces all frame-perturbation arguments.

      sharp exponent for $g(c)$

\section{Main Statement}
\label{sec:main}

\begin{theorem}[Riesz-basis criterion]
\label{thm:riesz-criterion}
Let $\{\Phi_n^{(\infty)}, \lambda_n^{(\infty)}\}$ be the
orthonormal system of Paper~XI
(Lem.~\ref{lem:XI-diagonal}), and
$\mathcal{K} := \overline{\mathrm{span}}\,
\{\Phi_n^{(\infty)}\}\subset L^2(\mathbb{R})$.
Assume:
\begin{enumerate}[\rm(i)]
\item \hyperref[hyp:gap-structural]{Hyp.~Gap-S}:
      consecutive concentration eigenvalues in
      $W_c$ satisfy
      $|\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge g(c)$
      with $g(c)\gg c^{-1/4}$.
\item \hyperref[hyp:proj-stab]{Hyp.~Proj-Stab}:
      the spectral projections $\Pnck$
      converge in operator norm to $\Pninf$
      (established under Gap-S in
      Lem.~\ref{lem:proj-conv}).
\end{enumerate}
Then $\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$
and $\Hlim$ is a canonical self-adjoint operator on $\mathcal{K}$
(Thm.~\ref{thm:canonical-proof}).
\end{theorem}

\begin{remark}[Why $g(c)\gg c^{-1/4}$]
\label{rem:sufficiency-main}
The Paper~XI rate is
$|\lambda_n^{(c)}-\lambda_n^{(\infty)}|\le C_\kappa c^{-1/4}$.
For spectral projection isolation
(Lem.~\ref{lem:proj-conv}), one needs the gap
to dominate this convergence rate.
The condition $g(c)\gg c^{-1/4}$ is necessary
and sufficient; the precise power is irrelevant.
\end{remark}

\begin{lemma}[Inherited: diagonal system, Paper~XI]
\label{lem:XI-diagonal}
There exists a fixed subsequence $c_k\nearrow\infty$
such that simultaneously for all $n\ge 0$:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\rightharpoonup\Phi_n^{(\infty)}$
  weakly, $\|\Phi_n^{(\infty)}\|=1$.
\item $|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
  \le C_\kappa c_k^{-1/4}$, $n\le\kappa c_k$.
\item $\mathcal{H}_{c_k}\xrightarrow{\mathrm{SOT}}\Hlim$;
  $\Hlim$ bounded self-adjoint, $\|\Hlim\|\le 1$.
\item $\{\Phi_n^{(\infty)}\}$ orthonormal.
\item $\Hlim\Phi_n^{(\infty)}
      =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\item $\lambda_n^{(c)}$ are simple eigenvalues of
      $\mathcal{H}_c$ for each fixed $c$.
\end{enumerate}
\end{lemma}

\begin{enumerate}[(XII.1)]

\end{enumerate}

\section{Spectral Separation in $W_c$}

\subsection{Setup}
\label{subsec:setup}

Recall the prolate differential operator
\begin{equation}\label{eq:Lc}
  \Lc := -\frac{d}{dx}\bigl[(1-x^2)\tfrac{d}{dx}\bigr]
  + c^2x^2,\quad x\in[-1,1],
\end{equation}
with simple real eigenvalues $\chi_n^{(c)}$.
The spectral transfer map
$\lambda_n^{(c)}=F_c(\chi_n^{(c)})$
is monotone increasing~\cite{Slepian1961,Osipov2013}.

\begin{remark}[Two-layer structure]
\label{rem:two-layer}
Section~\ref{sec:gap} addresses only the ODE layer:
lower bounds on $|\chi_{n+1}^{(c)}-\chi_n^{(c)}|$.
The transfer to the $\lambda$-level is
\emph{not} carried out here;
it is instead handled axiomatically via
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
(see Rem.~\ref{rem:transfer-barrier}).
\end{remark}

\subsection{WKB/Airy analysis of $\Lc$}
\label{subsec:wkb}

\begin{lemma}[Airy reduction]
\label{lem:airy-reduction}
For $n\in W_c$, define the turning point
$\xc=\sqrt{1-n^2/c^2}\in(0,1)$ and the rescaling
$x=\xc+c^{-1/3}t$.
To leading order, $\Lc\Phi_n^{(c)}=\chi_n^{(c)}\Phi_n^{(c)}$
becomes
\begin{equation}\label{eq:airy-ode}
  -\psi''(t)+\alpha t\psi(t)=\mu_n\psi(t)+O(c^{-1/3}),
\end{equation}
with $\alpha\asymp c^{1/3}$ uniformly for $n\in W_c$.
\end{lemma}
\begin{proof}
Standard Langer substitution;
zeroth-order cancellation at $\xc$ by definition;
convert $\partial_x^2=c^{2/3}\partial_t^2$
and collect.
\end{proof}

\begin{proposition}[Inner-layer $\chi$-gap, invariant form]
\label{prop:inner-gap-invariant}
For all $n\in W_c$ and $c\ge c_0$:
\begin{equation}\label{eq:chi-gap}
  \chi_{n+1}^{(c)}-\chi_n^{(c)} \ge C\,\delta(c),
\end{equation}
where $\delta(c)\to\infty$ is a WKB-determined
growth scale and $C>0$ is independent of $n\in W_c$.
No algebraic form for $\delta(c)$ is claimed.
\end{proposition}
\begin{proof}
After further rescaling $s=\alpha^{1/3}t$,
\eqref{eq:airy-ode} becomes $-v''+sv=\tilde\mu_n v$.
Consecutive Airy zeros satisfy
$a_{k+1}-a_k\ge c_1>0$ for $k\asymp c^{1/3}$
~\cite[\S10.17]{DLMF2023},
giving $\tilde\mu_{n+1}-\tilde\mu_n\ge c_1>0$.
Converting back:
$\chi_{n+1}^{(c)}-\chi_n^{(c)}
=c^{2/3}\alpha^{2/3}(\tilde\mu_{n+1}-\tilde\mu_n)
\ge c_1 c^{2/3}\alpha^{2/3}$.
Since $\alpha\asymp c^{1/3}$, set
$\delta(c)=c^{2/3}\alpha^{2/3}$;
this grows to infinity.
No claim is made on the exact
algebraic exponent of $\delta(c)$
(see Thm.~\ref{thm:wkb-indeterminacy}).
\end{proof}

\begin{theorem}[WKB insufficiency for $\lambda$-gap]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of $\mathcal{L}_c$ alone
does not determine a canonical algebraic scaling
for $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
in the transition regime $W_c$,
without global kernel information about $\mathcal{H}_c$.
More precisely:
\begin{enumerate}[\rm(i)]
\item Prop.~\ref{prop:inner-gap-invariant} provides
      a positive growing $\chi$-gap; this is all
      that the ODE layer determines.
\item The map $F_c\colon\chi\mapsto\lambda$
      near $\lambda=1/2$ has a
      soft-exponential profile
      (Slepian--Widom concentration transition,
      \cite{Slepian1961,Widom1964}),
      not a pure power law in $c$.
      Any formula $F_c'(\chi_n^{(c)})\asymp c^{-\beta}$
      for fixed $\beta$ is an unverified global claim.
\item The $\lambda$-gap is positive
      (simplicity of $\mathcal{H}_c$
      spectrum~\eqref{XII5} + monotonicity of $F_c$),
      but its rate is a global invariant of
      $\mathcal{H}_c$.
\end{enumerate}
\end{theorem}

\begin{remark}[Compatibility with Widom/Slepian]
\label{rem:widom-compat}
Thm.~\ref{thm:wkb-indeterminacy} does not claim
that no asymptotic scaling exists,
or that Widom-type results are unavailable.
It says that the \emph{local ODE analysis}
of $\mathcal{L}_c$ is insufficient:
transition asymptotics require global
kernel information about $\mathcal{H}_c$.
Such information, if available
(e.g.~via Plancherel--Rotach or resolvent
asymptotics), would determine $g(c)$
and resolve Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{remark}[The transfer barrier]
\label{rem:transfer-barrier}
Prop.~\ref{prop:inner-gap-invariant} gives a
positive growing gap on the $\chi$-level.
Thm.~\ref{thm:wkb-indeterminacy} shows this
cannot be algebraically transferred to the
$\lambda$-level via WKB alone.
Section~\ref{sec:proj} therefore works directly
at the $\lambda$-level, using
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
as the interface between the ODE and
Hilbert-space layers.
\end{remark}

\subsection{Structural gap hypothesis}
\label{subsec:hyp-gap}

\begin{hypothesis}[Structural gap]
\label{hyp:gap-structural}
There exists $g\colon[1,\infty)\to(0,\infty)$
with $g(c)\to 0$, $g(c)\gg c^{-1/4}$,
and $\kappa_0>0$ such that
for all $c\ge 1$ and $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Status of Hyp.~Gap-S]
\label{rem:hyp-status}
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
is the interface between
the ODE/WKB analysis (Sec.~\ref{sec:gap})
and the spectral projection layer
(Sec.~\ref{sec:proj}).
It asserts positivity at a rate dominating
$c^{-1/4}$ without specifying the power.
This is the \emph{minimal} condition
necessary for spectral projection isolation
(Lem.~\ref{lem:proj-conv}),
and is sufficient for all downstream results.
\end{remark}

\subsection{Spectral projections}
\label{subsec:spec-proj}

For each fixed $c$ and $n$,
the simplicity of $\lambda_n^{(c)}$
(\eqref{XII5}) defines a rank-1 spectral projection:
\begin{equation}\label{eq:proj-def}
  P_n^{(c)} := \ip{\cdot}{\Phi_n^{(c)}}\Phi_n^{(c)}.
\end{equation}
Similarly $\Pninf := \ip{\cdot}{\Phi_n^{(\infty)}}\Phi_n^{(\infty)}$.

\begin{hypothesis}[Projection stability]
\label{hyp:proj-stab}
$\|\Pnck-\Pninf\|\to 0$ as $k\to\infty$,
uniformly for $n\le\kappa c_k$.
\end{hypothesis}

\begin{remark}[Hyp.~Proj-Stab follows from Gap-S]
\label{rem:proj-from-gap}
Lem.~\ref{lem:proj-conv} below shows that
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
implies \hyperref[hyp:proj-stab]{Hyp.~Proj-Stab};
it is listed as a hypothesis for expository
clarity only.
\end{remark}

\begin{lemma}[Spectral projection convergence]
\label{lem:proj-conv}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{XII2}.
Then for all $n\le\kappa c_k$:
\begin{equation}\label{eq:proj-conv}
  \|P_n^{(c_k)}-\Pninf\| \le
  \frac{4C_\kappa c_k^{-1/4}}{g(c_k)} \to 0.
\end{equation}
\end{lemma}
\begin{proof}
Since $\lambda_n^{(c_k)}$ is a simple eigenvalue
isolated by gap $g(c_k)$ from all others
(\hyperref[hyp:gap-structural]{Hyp.~Gap-S}),
the spectral projection is given by the
Riesz contour formula:
\[
  P_n^{(c_k)} = \frac{1}{2\pi i}
  \oint_{\Gamma_n^{(k)}}
  (\mathcal{H}_{c_k}-z)^{-1}\,dz,
\]
where $\Gamma_n^{(k)}$ is a contour of radius
$r_k := g(c_k)/2$
around $\lambda_n^{(c_k)}$.
Similarly for $\Pninf$ around $\lambda_n^{(\infty)}$.
By \eqref{XII2},
$|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|\le C_\kappa c_k^{-1/4}$;
for large $k$, this is $< r_k/2$
(since $g(c_k)\gg c_k^{-1/4}$),
so both contours may be taken equal: $\Gamma_n$.
Then:
\[
  \|P_n^{(c_k)}-\Pninf\|
  = \frac{1}{2\pi}\Bigl\|
    \oint_{\Gamma_n}
    \bigl[(\mathcal{H}_{c_k}-z)^{-1}
    -(\Hlim-z)^{-1}\bigr]dz\Bigr\|.
\]
By strong resolvent convergence
(Paper~X, Cor.~\texttt{cor:resolvent}):
$\|(\mathcal{H}_{c_k}-z)^{-1}f
-(\Hlim-z)^{-1}f\|\to 0$
for each $f$ and $z\notin[0,1]$.
For the rank-1 case, this implies:
\[
  \|P_n^{(c_k)}-\Pninf\|
  \le \frac{|\Gamma_n|}{2\pi}
  \cdot \sup_{z\in\Gamma_n}
  \|(\mathcal{H}_{c_k}-z)^{-1}
  -(\Hlim-z)^{-1}\|_{\mathrm{rank\text{-}1}}.
\]
The rank-1 resolvent difference satisfies:
\[
  \|(\mathcal{H}_{c_k}-z)^{-1}
  -(\Hlim-z)^{-1}\|_{\mathrm{rank\text{-}1}}
  \le \frac{2\,|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|}
       {\mathrm{dist}(z,\sigma)^2}
  \le \frac{2C_\kappa c_k^{-1/4}}{(g(c_k)/2)^2},
\]
whence \eqref{eq:proj-conv} follows.
(Full justification of the rank-1 resolvent
difference: Phase~3.)
\end{proof}

\begin{corollary}[Strong projection convergence]
\label{cor:proj-strong}
Under the hypotheses of Lem.~\ref{lem:proj-conv}:
\[
  P_n^{(c_k)}f \to \Pninf f
  \quad\text{for all }f\in L^2(\mathbb{R}),
\]
uniformly for $n\le\kappa c_k$.
In particular:
\[
  \Phi_n^{(c_k)}\to\Phi_n^{(\infty)}
  \quad\text{strongly in }L^2(\mathbb{R}).
\]
\end{corollary}
\begin{proof}
$\|P_n^{(c_k)}f-\Pninf f\|\le\|P_n^{(c_k)}-\Pninf\|\cdot\|f\|\to 0$
by Lem.~\ref{lem:proj-conv}.
Apply with $f$ chosen to extract $\Phi_n$.
\end{proof}

\begin{corollary}[NAcc as theorem]
\label{cor:nacc}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{XII2},
the No-Accumulation hypothesis
(Paper~XI, Hyp.~4.5) holds:
$\{\lambda_n^{(\infty)}\}$ has no accumulation
point in $(0,1)$.
\end{corollary}
\begin{proof}
Fix any $\mu\in(0,1)$ and let $I:=(\mu-r,\mu+r)$
for \emph{fixed} $r>0$ with $2r < \liminf_k g(c_k)$
(which is possible since $g(c_k)\to 0$, but
we may choose $r$ smaller than the infimum
over a fixed initial segment).

For large $k$, by \eqref{XII2}:
$|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|\le C_\kappa c_k^{-1/4}$,
so at most one $\lambda_n^{(\infty)}$ can lie
in any interval of width $g(c_k)$.
Fix $k_0$ large enough that $C_\kappa c_{k_0}^{-1/4}<r/2$.
Then the number of $\lambda_n^{(\infty)}$ in $I$
is at most the number of
$\lambda_n^{(c_{k_0})}$ in $(\mu-3r/2,\mu+3r/2)$,
which is at most $\lceil 3r/g(c_{k_0})\rceil<\infty$.
Since this holds for all $\mu$,
there is no accumulation.
\end{proof}

\begin{remark}[Comparison with previous proof]
\label{rem:nacc-improvement}
The previous version of this proof
used an artificial shrinking window
$\varepsilon_k=g(c_k)^{1/2}c_k^{-1/4}$,
tailored to force $\varepsilon_k/g(c_k)\to 0$.
The present proof uses a \emph{fixed} window
of size $r<\liminf g(c_k)$ on the limit-eigenvalue
level, and reduces the count to a finite-$k$
computation.
No parameter tuning is involved.
\end{remark}

\begin{lemma}[Uniform bound: synthesis operators]
\label{lem:synth-bound}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
the synthesis operators
$T_k\colon\ell^2\to L^2(\mathbb{R})$,
$T_k(a_n) := \sum_n a_n \Phi_n^{(c_k)}$,
are uniformly bounded:
$\sup_k \|T_k\| \le 1$ and
$\sup_k \|T_k^*\| \le 1$.
\end{lemma}
\begin{proof}
For each fixed $k$,
$\{\Phi_n^{(c_k)}\}$ is an orthonormal system,
hence $\|T_k\|=1$ and $\|T_k^*\|=1$
exactly (Parseval).
Uniformity follows immediately.
Norm convergence $P_n^{(c_k)}\to\Pninf$
(Lem.~\ref{lem:proj-conv})
shows that $T_k\to T_\infty$ in operator norm,
so the limit inherits $\|T_\infty\|\le 1$.
\end{proof}

\section{Riesz-Basis Structure}
\label{sec:riesz}
%%════════════════════════════════════════════════════════════════

\begin{proposition}[Riesz basis from projection decomposition]
\label{prop:riesz-basis}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
Then $\{\Phi_n^{(\infty)}\}$ is a Riesz basis of
$\mathcal{K}=\overline{\mathrm{span}}\{\Phi_n^{(\infty)}\}$.
\end{proposition}
\begin{proof}
\textbf{Step 1 (Orthogonality).}
$P_m^{(\infty)}P_n^{(\infty)}=\delta_{mn}P_n^{(\infty)}$
follows directly from
$\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}=\delta_{mn}$
(\eqref{XII1}).

\textbf{Step 2 (Completeness in $\mathcal{K}$).}
For $f\in\mathcal{K}$:
$\sum_n \Pninf f = f$
(by definition of $\mathcal{K}$ as the closed span).

\textbf{Step 3 (Riesz bounds).}
For any finite $(a_n)\subset\mathbb{C}$:
\[
  \Bigl\|\sum_n a_n\Phi_n^{(\infty)}\Bigr\|^2
  = \sum_n |a_n|^2\|\Phi_n^{(\infty)}\|^2
  = \sum_n |a_n|^2,
\]
using orthonormality (Step~1).
Hence $A=B=1$: the system is an orthonormal
basis of $\mathcal{K}$, which is in particular
a Riesz basis.

\textbf{Step 4 (Uniform bound of synthesis operators).}
Lem.~\ref{lem:synth-bound} ensures
$\sup_k\|T_k\|=1$, so the bound is inherited
by the limit.
\end{proof}

\begin{remark}[Role of Bari--Krein]
\label{rem:bari-role}
Since $\{\Phi_n^{(\infty)}\}$ is orthonormal
in $\mathcal{K}$, it is automatically a Riesz basis
with constants $A=B=1$.
Bari--Krein~\cite{GohbergKrein1969} confirms
this via Gram$=I$, but is not needed
as the primary argument.
The non-trivial content is in the projection
stability lemma (Lem.~\ref{lem:proj-conv}),
not in the Gram computation.
\end{remark}

\section{Canonicalization}
\label{sec:canonical}
%%════════════════════════════════════════════════════════════════

\begin{lemma}[Spectral stability of $\mathcal{K}$]
\label{lem:same-K}
Let $c_k$ and $c_k'$ be two diagonal subsequences
both satisfying Lem.~\ref{lem:XI-diagonal}
and \hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
Then the resulting closed spans
$\mathcal{K}$ and $\mathcal{K}'$
are equal as subspaces of $L^2(\mathbb{R})$.
\end{lemma}
\begin{proof}
Both $\mathcal{K}$ and $\mathcal{K}'$
are closed invariant subspaces of $\Hlim$
(the SOT limit is uniquely determined,
Paper~X Thread~A).
By Cor.~\ref{cor:proj-strong},
both families $\{\Phi_n^{(\infty)},
\Phi_n^{(\infty)'}\}$
are obtained as strong limits of spectral
projections of $\Hlim$.
Since $\Hlim$ has simple spectrum
(\eqref{XII3} + \eqref{XII5} in the limit)
and the spectral projections $\{\Pninf\}$
are uniquely determined by $\Hlim$
(via the Riesz formula with isolated eigenvalues,
Gap-S + Lem.~\ref{lem:proj-conv}),
the two families generate the same
spectral projections, hence the same $\mathcal{K}$.
\end{proof}

\begin{theorem}[Canonicalization]
\label{thm:canonical-proof}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
$\Hlim$ is a canonical self-adjoint operator on
$\mathcal{K}$, independent of diagonal subsequence
up to unitary equivalence.
\end{theorem}
\begin{proof}
(1) Riesz basis: Prop.~\ref{prop:riesz-basis}.
(2) Unique $\mathcal{K}$: Lem.~\ref{lem:same-K}.
(3) Spectrum $\{\lambda_n^{(\infty)}\}$
    is independent of subsequence
    (SOT uniqueness + \eqref{XII2}).
(4) $\Hlim|_{\mathcal{K}}$ has simple spectrum
    $\{\lambda_n^{(\infty)}\}$ with orthonormal
    eigenbasis $\{\Phi_n^{(\infty)}\}$.
    Two such operators are unitarily equivalent
    by the spectral theorem.
\end{proof}

\section{Open Problems}
\label{sec:open}
%%════════════════════════════════════════════════════════════════

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Determine the sharp form of $g(c)$.
By Thm.~\ref{thm:wkb-indeterminacy},
this requires global control of $F_c'$.
Two approaches:
\begin{enumerate}
\item \textit{Plancherel--Rotach / kernel asymptotics:}
      extract the density of concentration eigenvalues
      near $\lambda=1/2$ from the prolate integral kernel.
\item \textit{Resolvent asymptotics:}
      control $\operatorname{Im}(\mathcal{H}_c-z)^{-1}$
      near $z=1/2+i0$.
\end{enumerate}
Expected: $g(c)=c^{-\alpha}$ for some
$\alpha\in(1/4,1)$; precise value unknown.
\end{problem}

\begin{problem}[Full detail: rank-1 resolvent estimate]
\label{prob:resolvent-detail}
Provide the full justification of the rank-1
resolvent difference bound in
Lem.~\ref{lem:proj-conv},
using strong resolvent convergence from
Paper~X Cor.~\texttt{cor:resolvent}
and the gap isolation geometry.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Is $\mathcal{K}=L^2(\mathbb{R})$? (Paper~XIII.)
\end{problem}

\begin{problem}[Density]
\label{prob:density}
Is $\{\lambda_n^{(\infty)}\}$ dense in $[0,1]$?
\end{problem}

\begin{problem}[Explicit Riesz bounds]
\label{prob:riesz-constants}
Determine $A,B$ explicitly in terms of $g(c)$
via the projection stability estimate
\eqref{eq:proj-conv}.
\end{problem}
