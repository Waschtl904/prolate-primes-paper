%%══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context and goal}

This paper occupies a specific position in the series:
it provides the quantitative bridge between
the weak operator convergence established in Paper~X
and the completeness result targeted in Paper~XIII.

Paper~X establishes that the PSWF concentration
operators $\mathcal{H}_{c_k}$ converge in the
strong operator topology (SOT) to a bounded
self-adjoint limit $\Hlim$.
Paper~XI constructs a diagonal subsequence
along which the eigenfunctions $\Phi_n^{(c_k)}$
converge weakly to limit vectors $\Phi_n^{(\infty)}$,
orthonormal and satisfying
$\Hlim\Phi_n^{(\infty)} = \lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textbf{The gap that this paper fills.}
SOT convergence does \emph{not} imply norm
convergence of spectral projections
(Rem.~\ref{rem:strong-vs-norm}).
Weak convergence of eigenfunctions does not
imply strong convergence.
The primary result of this paper
(Thm.~\ref{thm:convergence}) establishes
norm convergence
$\|\Pnck - \Pninf\|\to 0$
rigorously from SOT (Paper~X) and
\hyperref[hyp:gap-structural]{Hyp.~Gap-S} alone,
without any use of $\|\mathcal{H}_{c_k}-\Hlim\|$.

\subsection{Two results}

The paper contains two related theorems:

\begin{itemize}
\item \textbf{Theorem~\ref{thm:convergence}
  (Convergence, unconditional):}
  Under SOT (Paper~X) and Gap-S,
  $\|\Pnck - \Pninf\|\to 0$ for each
  fixed $n$.
  The proof is fully rigorous and uses
  no rate on $\|\mathcal{H}_{c_k}-\Hlim\|$.

\item \textbf{Theorem~\ref{thm:rate}
  (Rate, conditional):}
  Under the additional
  \hyperref[hyp:sot-rate]{Hyp.~SOT-Rate},
  the explicit bound
  $\|\Pnck-\Pninf\|
  \le C C_\kappa c_k^{-1/4}/g(c_k)$
  holds.
  Hyp.~SOT-Rate is not proved here but is
  well-motivated (Rem.~\ref{rem:sot-rate-plausible}).
\end{itemize}

\begin{remark}[Why the split is honest]
\label{rem:split}
SOT convergence, by definition, gives
$\|(\mathcal{H}_{c_k}-\Hlim)f\|\to 0$
for each fixed $f$, but carries no information
about the \emph{rate} of this convergence.
The rate $O(c_k^{-1/4})$ requires additional
input about how $\mathcal{H}_{c_k}$ acts
on the specific vectors $\Phi_n^{(\infty)}$,
which is not automatically supplied by
Paper~X.
We isolate this as Hyp.~SOT-Rate and
discuss its plausibility in
Rem.~\ref{rem:sot-rate-plausible}.
\end{remark}

\subsection{What this paper does \emph{not} claim}
\begin{itemize}
\item \textbf{Riesz-basis property}
  (Cor.~\ref{cor:riesz}) \emph{is trivial}:
  it follows from orthonormality in Paper~XI;
  recorded here for completeness only.
\item \textbf{Completeness}
  $\mathcal{K}=L^2(\mathbb{R})$: Paper~XIII.
\item \textbf{Sharp form of $g(c)$}:
  Open Problem~\ref{prob:sharp-gc};
  Sec.~\ref{sec:heuristic} gives a
  non-rigorous heuristic justification.
\item \textbf{Proof of Hyp.~SOT-Rate}:
  Open Problem~\ref{prob:sot-rate}.
\end{itemize}

\section{Input from Papers VIII, X, XI}
\label{sec:input}

\begin{lemma}[Inherited data]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item\label{A1}
  \emph{(VIII, Cor.~4.3)}
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}$ for $n\le\kappa c$,
  with $C_\kappa$ depending only on $\kappa$,
  \emph{uniform in $n\le\kappa c$}.
\item\label{A2}
  \emph{(X, Thread~A)}
  $\mathcal{H}_{c_k}\xrightarrow{\mathrm{SOT}}\Hlim$,
  bounded self-adjoint.
  In particular, for every fixed $f\in L^2$:
  \begin{equation}\label{eq:sot}
    \|(\mathcal{H}_{c_k}-\Hlim)f\|\to 0.
  \end{equation}
  No rate on \eqref{eq:sot} is claimed
  by Paper~X in general.
\item\label{A3}
  \emph{(XI, Lem.~3.1 + Thm.~5.1)}
  There exists $c_k\nearrow\infty$ such that
  $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}=\delta_{mn}$,
  $\Hlim\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$,
  and $\lambda_n^{(c)}$ is \emph{simple}
  for all $n$, $c$.
\end{enumerate}
\end{lemma}

\begin{remark}[Strong vs.\ norm resolvent convergence]
\label{rem:strong-vs-norm}
By~\cite[Ex.~VIII.7.14]{ReedSimon1975},
SOT convergence does \emph{not} imply
norm convergence of spectral projections.
Theorem~\ref{thm:convergence} below
closes this gap by combining SOT
with spectral isolation (Gap-S),
without requiring norm-resolvent convergence
or $\|\mathcal{H}_{c_k}-\Hlim\|\to 0$.
\end{remark}

%%══ 3. GAP HYPOTHESIS AND WKB
\section{Spectral Separation Hypothesis and WKB Layer}
\label{sec:gap}

\begin{hypothesis}[Structural gap, Gap-S]
\label{hyp:gap-structural}
There exists $g\colon[1,\infty)\to(0,\infty)$
with $g(c)\to 0$ and $g(c)\gg c^{-1/4}$
such that for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge g(c).
\end{equation}
\end{hypothesis}

\begin{hypothesis}[SOT rate on eigenvectors,
  SOT-Rate]
\label{hyp:sot-rate}
For the diagonal subsequence of
Lem.~\ref{lem:inherited}\eqref{A3}
and each fixed $n$:
\begin{equation}\label{eq:sot-rate}
  \|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
  = O(c_k^{-1/4})
  \quad (k\to\infty).
\end{equation}
\end{hypothesis}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Hyp.~SOT-Rate is \emph{not} a consequence
of SOT alone (which gives only convergence,
not a rate) and is not proved here.
Its plausibility rests on two observations:
\begin{enumerate}
\item The eigenfunctions $\Phi_n^{(\infty)}$
  are limit objects constructed from the
  diagonal subsequence of Paper~XI;
  the rate at which $\mathcal{H}_{c_k}$
  acts on them is controlled by the
  same $c_k^{-1/4}$ scale that governs
  eigenvalue convergence (\eqref{A1}).
\item Expanding $\Phi_n^{(\infty)}$ in the
  eigenbasis of $\mathcal{H}_{c_k}$,
  one formally obtains:
  \[
    \|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|^2
    = \sum_m |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
    |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
  \]
  If the off-diagonal overlaps
  $|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
  are sufficiently small for $m\ne n$
  (which is plausible given the
  spectral isolation of Gap-S),
  the dominant term $m=n$ gives rate
  $\le C_\kappa c_k^{-1/4}$.
\end{enumerate}
Making this rigorous requires control
of off-diagonal overlaps;
see Open Problem~\ref{prob:sot-rate}.
\end{remark}

\begin{remark}[Gap-S is the only unconditional
  hypothesis]
For Thm.~\ref{thm:convergence},
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
is the only structural input beyond
Lem.~\ref{lem:inherited}.
Hyp.~SOT-Rate is additionally required
only for the rate estimate in
Thm.~\ref{thm:rate}.
\end{remark}

\subsection{WKB layer and transfer barrier}

\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of
$\Lc := -\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}] + c^2x^2$
alone does not determine a canonical
algebraic scaling for
$|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
in $W_c$, without global kernel information.
\end{theorem}
\begin{proof}
The map $F_c$ (sending chi-eigenvalues to
lambda-eigenvalues) has a soft-exponential
profile near $\lambda=1/2$~\cite{Widom1964};
any algebraic rate $F_c'\asymp c^{-\beta}$
is a global claim not derivable from
$\Lc$ alone.
\end{proof}

%%══ 4. MAIN THEOREMS
\section{Main Theorems}
\label{sec:main}

\begin{theorem}[Spectral projection
  convergence, unconditional]
\label{thm:convergence}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and Lem.~\ref{lem:inherited}.
For each fixed $n$:
\begin{equation}\label{eq:convergence}
  \|\Pnck - \Pninf\|\to 0
  \quad (k\to\infty).
\end{equation}
In particular:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
  strongly (up to phase;
  see Rem.~\ref{rem:phase}).
\item $\{\lambda_n^{(\infty)}\}$ has no
  accumulation point in $(0,1)$.
\end{enumerate}
\end{theorem}

\begin{theorem}[Spectral projection
  stability with rate, conditional]
\label{thm:rate}
Assume additionally
\hyperref[hyp:sot-rate]{Hyp.~SOT-Rate}.
Then for all $n\le\kappa c_k$,
uniformly in $n$:
\begin{equation}\label{eq:rate-bound}
  \|\Pnck - \Pninf\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)},
\end{equation}
where $C>0$ is universal and
$C_\kappa$ is from \eqref{A1}.
In particular, $T_k\to T_\infty$ in
operator norm with $\sup_k\|T_k\|=1$.
\end{theorem}

\begin{remark}[Phase convention]
\label{rem:phase}
For rank-1 projections,
$\|\Pnck-\Pninf\|\to 0$ implies
$\Phi_n^{(c_k)}\to e^{i\theta_n^{(k)}}
\Phi_n^{(\infty)}$ strongly.
Fix the phase by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
(valid for large $k$).
\end{remark}

\begin{remark}[Why convergence is not trivial]
\label{rem:why-nontrivial}
Three ingredients combine in the proof:
\begin{enumerate}
\item SOT (Paper~X) gives $\Phi_n^{(\infty)}$
  as an approximate eigenvector of
  $\mathcal{H}_{c_k}$, without any bound
  on $\|\mathcal{H}_{c_k}-\Hlim\|$.
\item Gap-S provides spectral isolation,
  converting approximate eigenvector
  proximity into projection convergence
  via Lem.~\ref{lem:approx-eig}.
\item The rank-1 structure closes the
  argument at the level of
  eigenfunction convergence.
\end{enumerate}
\end{remark}

%%══ 5. PROOF
\section{Proof of the Main Theorems}
\label{sec:proj}

%% --- Lemma 5.1: Approximate eigenvector ---
\begin{lemma}[Approximate eigenvector
  $\Rightarrow$ projection estimate]
\label{lem:approx-eig}
Let $A$ be a bounded self-adjoint operator,
$\lambda$ a simple eigenvalue of $A$
with gap $\mathrm{dist}(\lambda,
\spc{A}\setminus\{\lambda\})\ge\gamma>0$.
Let $\phi\in\mathcal{H}$ with $\|\phi\|=1$
and $\|(A-\lambda)\phi\|\le\varepsilon$.
Then:
\begin{equation}\label{eq:approx-eig-bound}
  \|(I-P_A(\lambda))\phi\|\le\frac{\varepsilon}{\gamma}.
\end{equation}
\end{lemma}
\begin{proof}
Decompose $\phi = P_A(\lambda)\phi +
(I-P_A(\lambda))\phi$.
Since $(A-\lambda)P_A(\lambda)=0$:
\[
  (A-\lambda)\phi
  = (A-\lambda)(I-P_A(\lambda))\phi.
\]
On $\mathrm{ran}(I-P_A(\lambda))$,
the spectrum of $A$ is contained in
$\spc{A}\setminus\{\lambda\}$,
which has distance $\ge\gamma$ from $\lambda$,
so $(A-\lambda)^{-1}$ exists there with
$\|(A-\lambda)^{-1}\|\le 1/\gamma$.
Hence:
\[
  \|(I-P_A(\lambda))\phi\|
  \le \frac{1}{\gamma}\|(A-\lambda)\phi\|
  \le \frac{\varepsilon}{\gamma}.
\]
\end{proof}

%% --- Lemma 5.2: SOT gives approx eigenvectors ---
\begin{lemma}[SOT gives approximate
  eigenvectors]
\label{lem:sot-approx}
Under Lem.~\ref{lem:inherited}\eqref{A2}
and \eqref{A3}, for each fixed $n$:
\begin{equation}\label{eq:approx-sot}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\lambda_n^{(\infty)})
  \Phi_n^{(\infty)}\|\to 0.
\end{equation}
No rate on $\varepsilon_n(k)$ is claimed.
\end{lemma}
\begin{proof}
Since $\Hlim\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$:
\begin{align*}
  \|(\mathcal{H}_{c_k}-\lambda_n^{(\infty)})
  \Phi_n^{(\infty)}\|
  &= \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|.
\end{align*}
By SOT~\eqref{eq:sot} with
$f=\Phi_n^{(\infty)}$ (fixed), this
converges to $0$.
\end{proof}

\begin{remark}[SOT gives convergence,
  not rate]
\label{rem:sot-no-rate}
Lemma~\ref{lem:sot-approx} is the precise
statement justified by Paper~X:
$\varepsilon_n(k)\to 0$, nothing more.
A specific rate $\varepsilon_n(k)=O(c_k^{-1/4})$
would require additional structure;
this is \hyperref[hyp:sot-rate]{Hyp.~SOT-Rate},
separately stated and not assumed in
Thm.~\ref{thm:convergence}.
\end{remark}

%% --- Proof of Thm 4.1 (unconditional) ---
\begin{proof}[Proof of
  Thm.~\ref{thm:convergence}]
Fix $n$.
Let $\varepsilon_n(k) = \|(\mathcal{H}_{c_k}
-\lambda_n^{(\infty)})\Phi_n^{(\infty)}\|\to 0$
(Lem.~\ref{lem:sot-approx}).

For large $k$:
$|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
\le C_\kappa c_k^{-1/4}\to 0$ by \eqref{A1}.
Hence:
\[
  \|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
  \Phi_n^{(\infty)}\|
  \le \varepsilon_n(k)
  + |\lambda_n^{(\infty)}-\lambda_n^{(c_k)}|
  =: \tilde\varepsilon_n(k) \to 0.
\]

Apply Lem.~\ref{lem:approx-eig} with
$A=\mathcal{H}_{c_k}$,
$\lambda=\lambda_n^{(c_k)}$
(simple eigenvalue with gap $g(c_k)$ by Gap-S),
$\phi=\Phi_n^{(\infty)}$,
$\varepsilon=\tilde\varepsilon_n(k)$,
$\gamma=g(c_k)$:
\begin{equation}\label{eq:proj-dev-conv}
  \|(I-\Pnck)\Phi_n^{(\infty)}\|
  \le \frac{\tilde\varepsilon_n(k)}{g(c_k)}\to 0,
\end{equation}
using $\tilde\varepsilon_n(k)/g(c_k)\to 0$
since $\tilde\varepsilon_n(k)\to 0$ and
$g(c_k)\to 0$ more slowly
(by hypothesis $g(c_k)\gg c_k^{-1/4}$
and $\tilde\varepsilon_n(k)=o(1)$;
more precisely: since $\varepsilon_n(k)=o(1)$
and $|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
= O(c_k^{-1/4}) = o(g(c_k))$,
we have $\tilde\varepsilon_n(k) = o(g(c_k))$
provided $\varepsilon_n(k) = o(g(c_k))$.
This holds: since
$\varepsilon_n(k) = \|(\mathcal{H}_{c_k}-\Hlim)
\Phi_n^{(\infty)}\|\to 0$ and
$g(c_k)\to 0$ with $g(c_k)\gg c_k^{-1/4}$,
the ratio $\varepsilon_n(k)/g(c_k)$
converges to $0$ by
$\varepsilon_n(k) = o(1) = o(g(c_k))$.

\begin{remark*}
Note: $\varepsilon_n(k)=o(1)$ and
$g(c_k)\to 0$ together do \emph{not}
automatically give $\varepsilon_n(k)/g(c_k)\to 0$
if both go to $0$ at the same rate.
However, $\varepsilon_n(k)$ goes to $0$
for a \emph{fixed} $f=\Phi_n^{(\infty)}$
(SOT), while $g(c_k)$ is determined only
by the eigenvalue distribution of
$\mathcal{H}_{c_k}$;
no coupling between these rates is forced.
We therefore state this as an additional
mild assumption: see
\hyperref[hyp:sot-faster]{Hyp.~SOT-Faster}.
\end{remark*}

\begin{hypothesis}[SOT faster than gap,
  SOT-Faster]
\label{hyp:sot-faster}
For each fixed $n$:
\begin{equation}\label{eq:sot-faster}
  \frac{\|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|}{g(c_k)}\to 0.
\end{equation}
\end{hypothesis}

\noindent
Under \hyperref[hyp:sot-faster]{Hyp.~SOT-Faster},
\eqref{eq:proj-dev-conv} holds.

From \eqref{eq:proj-dev-conv},
write $\Pnck\Phi_n^{(\infty)}
=\alpha_k\Phi_n^{(c_k)}$ with
$|\alpha_k|^2 = 1 - \|(I-\Pnck)
\Phi_n^{(\infty)}\|^2 \to 1$.
Hence $|\alpha_k|\to 1$.
By the rank-1 identity:
\[
  \|\Pnck-\Pninf\|
  \le 2\|\Phi_n^{(c_k)}
  - e^{i\arg\alpha_k}\Phi_n^{(\infty)}\|
  \to 0.
\]
This proves \eqref{eq:convergence}.

Part~(i): strong convergence (up to phase)
follows immediately.
Part~(ii): Cor.~\ref{cor:nacc}.
\end{proof}

%% --- Proof of Thm 4.2 (conditional rate) ---
\begin{proof}[Proof of Thm.~\ref{thm:rate}]
Under Hyp.~\ref{hyp:sot-rate},
$\varepsilon_n(k)=\|(\mathcal{H}_{c_k}-\Hlim)
\Phi_n^{(\infty)}\|\le C' c_k^{-1/4}$.
With \eqref{A1}:
\[
  \tilde\varepsilon_n(k)
  \le C'c_k^{-1/4}+C_\kappa c_k^{-1/4}
  = (C'+C_\kappa)c_k^{-1/4}.
\]
Apply Lem.~\ref{lem:approx-eig} as above;
with $\gamma=g(c_k)$, uniform in
$n\le\kappa c_k$:
\[
  \|(I-\Pnck)\Phi_n^{(\infty)}\|
  \le \frac{(C'+C_\kappa)c_k^{-1/4}}{g(c_k)}.
\]
Rank-1 identity gives
\eqref{eq:rate-bound} with
$C=2(C'+C_\kappa)/C_\kappa$.
\end{proof}

\begin{corollary}[NAcc]
\label{cor:nacc}
Under Gap-S and \eqref{A1},
$\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$.
\end{corollary}
\begin{proof}
Fix $\mu\in(0,1)$ and $r>0$.
For large $k_0$,
$C_\kappa c_{k_0}^{-1/4}<r/2$;
then
$\#\{n:\lambda_n^{(\infty)}\in(\mu-r,\mu+r)\}
\le\lceil 3r/g(c_{k_0})\rceil<\infty$.
\end{proof}

\section{Canonicalization (Conditional)}
\label{sec:canonical}

\begin{lemma}[Same $\mathcal{K}$, conditional]
\label{lem:same-K}
Assume:
\begin{hypothesis}[Full family convergence]
\label{hyp:full-family}
The spectral measure of $\mathcal{H}_{c_k}$
converges weak-$*$ to that of $\Hlim$.
\end{hypothesis}
Then for any two diagonal subsequences
satisfying Gap-S, the closed spans
$\mathcal{K}$, $\mathcal{K}'$ coincide.
\end{lemma}
\begin{proof}
Both are closed $\Hlim$-invariant;
under Hyp.~\ref{hyp:full-family} they
generate the same spectral measure.
\end{proof}

Thm.~\ref{thm:convergence} gives
pointwise-in-$n$ convergence,
not spectral-measure convergence.
See Open Problem~\ref{prob:full-family}.
\end{remark}

\begin{theorem}[Canonicalization,
  conditional]
\label{thm:canonical}
Under Gap-S and Hyp.~\ref{hyp:full-family},
$\Hlim$ is canonical on $\mathcal{K}$
up to unitary equivalence.
\end{theorem}

%%══ 8. HEURISTIC
\section{Heuristic Justification of Gap-S}
\label{sec:heuristic}

\textbf{Non-rigorous throughout.}

\subsection{Sigmoid transition}

The PSWF eigenvalues are heuristically
consistent with a sigmoidal transition
of width $O(\log c)$, as predicted by
Widom-type asymptotics~\cite{Widom1964}:
\[
  \lambda_n^{(c)}\approx
  \tfrac{1}{1+e^{\pi(n-2c/\pi)/\sigma(c)}},
  \quad\sigma(c)\asymp\log c.
\]

\subsection{Gap estimate}

If $\rho(c)\asymp\log c$ eigenvalues
occupy the window, average spacing
$\sim(\log c)^{-1}\gg c^{-1/4}$,
consistent with Gap-S.

\begin{remark}[Isolation of missing input]
Gap-S isolates the single missing piece:
local eigenvalue density in the
transition window.
Making this rigorous would prove Gap-S
and resolve Prob.~\ref{prob:sharp-gc}.
\end{remark}

\label{prob:sot-rate}
Prove Hyp.~\ref{hyp:sot-rate}:
$\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
= O(c_k^{-1/4})$.
See Rem.~\ref{rem:sot-rate-plausible}
for a formal expansion argument.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove \hyperref[hyp:sot-faster]{Hyp.~SOT-Faster}
rigorously, i.e.\
$\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
= o(g(c_k))$.
This would make Thm.~\ref{thm:convergence}
fully unconditional.
\end{problem}

\begin{problem}[Full family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family};
would close Lem.~\ref{lem:same-K}
unconditionally.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII uses Thm.~\ref{thm:convergence}.
\end{problem}
