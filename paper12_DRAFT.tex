\title[A Localization Principle for Spectral Projection Stability]{%
  A Localization Principle for Spectral Projection Stability
  under Strong Operator Convergence}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
\textbf{Main result.}
For bounded self-adjoint operators with
$A_k\xrightarrow{\mathrm{SOT}} A$,
rank-one spectral projection stability
$\|P_k - P^\infty\|\to 0$
is equivalent, when the spectral gap
$\gamma_k\ge c_0>0$, to the gap-normalized
residual condition
\[
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}\to 0.
\]
This is a \emph{localization principle}
(by which we mean: control of the residual
relative to the local spectral gap)
that replaces global operator-norm control
(Davis--Kahan, Kato) by a single local
spectral-geometric condition,
and is not captured by standard
norm-based perturbation theory.

\medskip
We further establish:
a No-Go result (Prop.~\ref{prop:no-go})
showing eigenvalue density alone cannot
yield projection stability;
a counterexample (Prop.~\ref{prop:counterexample})
confirming sharpness when gap control fails;
and a reduction of PSWF projection stability
to two independent analytic mechanisms
(Gap-S and SOT-Faster).
\end{abstract}

%%══ 3. HYPOTHESES AND NO-GO
\section{Hypotheses, Structural Necessity,
  and No-Go}
\label{sec:gap}

\begin{hypothesis}[Effective gap, Gap-S]
\label{hyp:gap-structural}
Let $W_c:=\{n:\lambda_n^{(c)}
\in(\delta,1-\delta)\}$ for fixed $\delta>0$.
There exists $g\colon[1,\infty)\to(0,\infty)$
such that
\begin{equation}\label{eq:gap-dominates}
  \frac{c^{-1/4}}{g(c)}\to 0
  \quad\text{as }c\to\infty,
\end{equation}
and for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S: structural necessity;
  determinantal context]
\label{rem:gap-necessary}
Gap-S is \textbf{structurally necessary}
for any SOT-based argument for projection
stability that does not use additional
structure beyond the local spectral gap:
without \eqref{eq:gap-dominates},
the ratio $\tilde\varepsilon_n(k)/g(c_k)$
in \eqref{eq:ratio-to-zero} cannot be shown
to vanish (confirmed by Prop.~\ref{prop:counterexample}).
A different proof strategy might encode
an equivalent condition differently,
but must capture the same gap-relative
scale information.

\textbf{Density vs.\ gap.}
Widom~\cite{Widom1964} controls average
density ($\sim(\log c)^{-1}$ spacing),
not the minimal gap.
Average spacing $\ne$ minimum gap;
Gap-S is strictly stronger
(Prop.~\ref{prop:no-go}).

\textbf{Determinantal analogy.}
In determinantal point processes,
minimal gaps are controlled via
sine-kernel repulsion.
Gap-S is the analogous deterministic
bound for the PSWF kernel;
Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{remark}[$c^{-1/4}/g(c)\to 0$]
\label{rem:gap-implies-o}
Condition \eqref{eq:gap-dominates}
states explicitly that $c^{-1/4}/g(c)\to 0$,
i.e.\ the eigenvalue-matching error $C_\kappa c^{-1/4}$
is of strictly smaller order than the gap $g(c)$.
This ratio is used in \eqref{eq:ratio-to-zero}.
\end{remark}

The following result shows that any approach
based solely on eigenvalue density must fail,
even under SOT convergence and eigenvalue
convergence.

\begin{proposition}[No-Go: eigenvalue
  density cannot imply projection stability]
\label{prop:no-go}
Let $(\mu_n)$ be a sequence with
$\liminf_n(\mu_n-\mu_{n+1})=0$
but with Widom-type average density.
Then there exists a sequence
$B_k\xrightarrow{\mathrm{SOT}} B$
of bounded self-adjoint operators
with eigenvalues $\mu_n^{(k)}\to\mu_n$
such that $\|Q_n^{(k)}-Q_n\|\not\to 0$
for infinitely many $n$.
\end{proposition}
\begin{proof}
Choose indices $n_j$ along which
$\gamma_j := \mu_{n_j} - \mu_{n_j+1} \to 0$.
For each $j$, define on
$\mathrm{span}\{e_{n_j}, e_{n_j+1}\}$
the symmetric $2\times 2$ matrix
\[
  B_k^{(j)} :=
  \begin{pmatrix}
    \mu_{n_j}^{(k)} & \varepsilon_j(k) \\
    \varepsilon_j(k) & \mu_{n_j+1}^{(k)}
  \end{pmatrix},
  \quad
  \varepsilon_j(k) := \sqrt{\gamma_j^{(k)}},
\]
where $\gamma_j^{(k)}
:= \mu_{n_j}^{(k)} - \mu_{n_j+1}^{(k)}
\to \gamma_j \to 0$.
Set $B_k := \bigoplus_j B_k^{(j)}$
(block-diagonal, uniformly bounded);
then $B_k \xrightarrow{\mathrm{SOT}} B$.

For the $j$-th block,
the eigenvector of $B_k^{(j)}$
corresponding to the upper eigenvalue
makes an angle $\theta_j^{(k)}$
with $e_{n_j}$ satisfying
\[
  \tan\bigl(2\theta_j^{(k)}\bigr)
  = \frac{2\varepsilon_j(k)}{\gamma_j^{(k)}}
  = \frac{2}{\sqrt{\gamma_j^{(k)}}}
  \to +\infty.
\]
Hence $\theta_j^{(k)} \to \pi/4$,
so the upper eigenvector tends to
$\tfrac{1}{\sqrt{2}}(e_{n_j}+e_{n_j+1})$
rather than $e_{n_j}$.
Therefore
$\|Q_{n_j}^{(k)} - Q_{n_j}\|
\ge |\ip{Q_{n_j}^{(k)}e_{n_j}}{e_{n_j}}
- 1|
= |\cos^2\theta_j^{(k)} - 1|
\to \tfrac{1}{2}$,
so projection convergence fails for all $j$.
\end{proof}

\begin{remark}
\begin{center}\itshape
Any argument based solely on eigenvalue
density cannot yield projection stability;
minimal-gap control is logically necessary.
\end{center}
\end{remark}

\begin{hypothesis}[SOT-Faster]
\label{hyp:sot-faster}
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k)).
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster as microlocal
  norm convergence]
\label{rem:sot-faster-localized}
SOT-Faster is strictly stronger than
$\varepsilon_n(k)\to 0$ (SOT gives this freely).
It encodes norm control localised to a
spectral window of width $\sim g(c_k)$
around $\lambda_n^{(\infty)}$:
SOT-Faster follows from
$\|(\mathcal{H}_{c_k}-z)^{-1}
-(\Hlim-z)^{-1}\|\cdot g(c_k)\to 0$
for $|z-\lambda_n^{(\infty)}|\asymp g(c_k)$.
\end{remark}

\begin{hypothesis}[SOT-Rate]
\label{hyp:sot-rate}
$\varepsilon_n(k) = O(c_k^{-1/4})$.
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
SOT-Rate $\Rightarrow$ SOT-Faster
$\Rightarrow$ $\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$.
SOT alone gives $\varepsilon_n(k)\to 0$;
gap comparison is strictly additional.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding in eigenbasis $\{\Phi_m^{(c_k)}\}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
\end{equation}
Diagonal $m=n$: $\le C_\kappa^2 c_k^{-1/2}$.
Small off-diagonal overlaps
(consistent with Gap-S)
predict $\varepsilon_n(k)=O(c_k^{-1/4})$;
Prob.~\ref{prob:off-diagonal}.
\end{remark}

%%══ 4. ABSTRACT STABILITY PRINCIPLE
\section{Abstract Stability Principle}
\label{sec:abstract}

\begin{theorem}[Abstract Stability Principle;
  sufficient condition]
\label{thm:abstract}
Let $\{A_k\}$, $A$ be bounded self-adjoint
on a Hilbert space $\mathcal{H}$ with
$A_k\xrightarrow{\mathrm{SOT}} A$.
Let $\phi\in\mathcal{H}$, $\|\phi\|=1$,
$A\phi=\lambda\phi$, $\lambda$ isolated
and simple,
$P^\infty$ the rank-one projection
onto $\ker(A-\lambda)$.
Assume simple eigenvalues
$\lambda_k\in\sigma(A_k)$
with
\begin{equation}\label{eq:abstract-matching}
  \lambda_k\to\lambda,
\end{equation}
gap $\gamma_k:=\mathrm{dist}(\lambda_k,
\sigma(A_k)\setminus\{\lambda_k\})>0$,
and
\begin{equation}\label{eq:abstract-ratio}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0.
\end{equation}
Then $\|P_k - P^\infty\|\to 0$.
\end{theorem}
\begin{proof}
Set $\varepsilon_k:=\|(A_k-A)\phi\|$,
$\delta_k:=|\lambda_k-\lambda|$.
By triangle inequality:
$\|(A_k-\lambda_k)\phi\|
\le\varepsilon_k+\delta_k$.
Lem.~\ref{lem:approx-eig} gives
$\|(I-P_k)\phi\|
\le(\varepsilon_k+\delta_k)/\gamma_k\to 0$.
Rank-one identity~\cite[Ch.~1]{Bhatia1997}
yields $\|P_k-P^\infty\|\to 0$.
\end{proof}

\begin{lemma}[Rank-one projection convergence
  implies eigenvector convergence]
\label{lem:rank-one-proj}
Let $P_k = v_k\otimes v_k$ and
$P^\infty = \phi\otimes\phi$
be rank-one orthogonal projections
with $\|v_k\|=\|\phi\|=1$.
If $\|P_k - P^\infty\|\to 0$, then
there exist phases $\theta_k\in\mathbb{R}$
such that
\[
  \|v_k - e^{i\theta_k}\phi\|\to 0.
\]
\end{lemma}
\begin{proof}
Since $\|P_k - P^\infty\|\to 0$,
we have for all $f\in\mathcal{H}$:
$\|P_k f - P^\infty f\|\to 0$.
Applying this to $f=\phi$
(and using $P^\infty\phi=\phi$):
\[
  \|v_k\ip{v_k}{\phi} - \phi\|\to 0.
\]
Set $\alpha_k:=\ip{v_k}{\phi}$.
Then $\|v_k\alpha_k - \phi\|\to 0$
implies $|\alpha_k|\to 1$
(since $\|v_k\alpha_k\|=|\alpha_k|$
and $\|\phi\|=1$).
Choose $\theta_k$ such that
$e^{i\theta_k}\alpha_k=|\alpha_k|>0$.
Then
\begin{align*}
  \|v_k - e^{i\theta_k}\phi\|
  &\le \|v_k\alpha_k - \phi\|/|\alpha_k|
  + \|e^{i\theta_k}\phi\|\cdot|1-|\alpha_k||/|\alpha_k|
  \to 0,
\end{align*}
since $|\alpha_k|\to 1$ and
$\|v_k\alpha_k-\phi\|\to 0$.
\end{proof}

\begin{theorem}[Necessary condition;
  bounded-gap case]
\label{thm:necessary-bounded-gap}
Under the setup of Thm.~\ref{thm:abstract},
assume $\|P_k-P^\infty\|\to 0$
and $\gamma_k\ge c_0>0$.
Then \eqref{eq:abstract-ratio} holds.
\end{theorem}
\begin{proof}
\textbf{Eigenvalue term.}
$|\lambda_k-\lambda|/\gamma_k
\le|\lambda_k-\lambda|/c_0\to 0$.

\textbf{Residual term.}
Decompose $\phi=P_k\phi+(I-P_k)\phi$:
\begin{equation}\label{eq:decomp-necessary}
  (A_k-A)\phi
  = (A_k-A)P_k\phi
  + (A_k-A)(I-P_k)\phi.
\end{equation}
\textit{Term~(II).}
$\|(A_k-A)(I-P_k)\phi\|
\le(\sup_k\|A_k\|+\|A\|)\|(I-P_k)\phi\|$.
$\sup_k\|A_k\|\le M<\infty$
by uniform boundedness.
$\|(I-P_k)\phi\|\to 0$ since
$P^\infty\phi=\phi$ and $\|P_k-P^\infty\|\to 0$.
Hence Term~(II) $\to 0$.

\textit{Term~(I).}
Write $P_k\phi=\alpha_k v_k$,
$\alpha_k:=\ip{v_k}{\phi}$,
$A_k v_k=\lambda_k v_k$; so
$(A_k-A)P_k\phi
= \alpha_k(\lambda_k v_k - Av_k)$.
Decompose:
$\lambda_k v_k - Av_k
= (\lambda_k-\lambda)v_k
+ (\lambda-A)v_k$.
First part:
$\|(\lambda_k-\lambda)v_k\|
=|\lambda_k-\lambda|\to 0$.
Second part: since $(A-\lambda)\phi=0$,
\[
  (A-\lambda)v_k
  = (A-\lambda)(v_k - e^{i\theta_k}\phi);
\]
hence
$\|(A-\lambda)v_k\|
\le\|A-\lambda\|\cdot\|v_k-e^{i\theta_k}\phi\|\to 0$
by Lem.~\ref{lem:rank-one-proj}.
Since $|\alpha_k|\le 1$:
Term~(I) $\to 0$.

Combining: $\|(A_k-A)\phi\|\to 0$;
dividing by $\gamma_k\ge c_0$ gives
\eqref{eq:abstract-ratio}.
\end{proof}

\begin{remark}[Uniform gap is essential for necessity]
\label{rem:necessity-uniform-gap}
The assumption $\gamma_k\ge c_0>0$
cannot be dropped in
Thm.~\ref{thm:necessary-bounded-gap}.
Without a uniform lower bound on $\gamma_k$,
the normalization by $\gamma_k$ becomes
unstable: the quantity
$\|(A_k-A)\phi\|/\gamma_k$ may diverge
even when projection convergence holds,
because the numerator can vanish slower
than $\gamma_k$.
The precise analysis of the
vanishing-gap regime is encoded in
Hyp.~\ref{hyp:sot-faster}
and Open Problem~\ref{prob:full-iff}.
\end{remark}

\begin{remark}[Rank-one structure is essential]
\label{rem:rank-one-essential}
The necessity direction in
Thm.~\ref{thm:necessary-bounded-gap}
crucially uses the rank-one structure
of $P_k$ and $P^\infty$:
it allows the eigenvector convergence
$\|v_k - e^{i\theta_k}\phi\|\to 0$
to be extracted directly from
$\|P_k - P^\infty\|\to 0$
(Lem.~\ref{lem:rank-one-proj}).
For higher-multiplicity eigenspaces,
this argument fails without
additional assumptions
(e.g.\ convergence of reduced resolvents
or angular subspace bounds).
\end{remark}

\begin{remark}[Open: full iff when
  $\gamma_k\to 0$]
\label{rem:iff-open}
When $\gamma_k\to 0$,
the proof above shows $\|(A_k-A)\phi\|\to 0$
(independent of $\gamma_k$),
but whether the rate is faster than
$\gamma_k$ depends on the specific
spectral geometry and is not determined
by projection convergence alone.
This is precisely the content of
Hyp.~\ref{hyp:sot-faster}
and Open Problem~\ref{prob:full-iff}.
\end{remark}

\begin{remark}[Novelty: the localization
  principle as a one-vector perturbation
  theory]
\label{rem:abstract-novelty}
The residual-gap inequality underlying
Thm.~\ref{thm:abstract} appears implicitly
in classical spectral approximation theory
(Chatelin~\cite{Chatelin1983},
Osborn~\cite{Osborn1975},
Kato~\cite{Kato1966} Ch.~IV),
always within a framework that assumes
$\|A_k - A\| \to 0$.

\textbf{Classical theory}
(Davis--Kahan~\cite{DavisKahan1970},
Kato~\cite{Kato1966}):
$\|P_k - P^\infty\| \lesssim
\|A_k - A\|/\mathrm{gap}$,
controlled via the global operator norm.
\textbf{Present result:}
no access to $\|A_k - A\|$
(which may be infinite or diverging);
global control may fail completely,
yet projection stability is governed
purely by a single \emph{vector-local}
residual $\|(A_k-A)\phi\|$
relative to the gap.
We therefore call this a
\emph{one-vector perturbation principle}:
the only perturbation information needed
is the action of $A_k - A$ on the
single vector $\phi$.

\textbf{The novelty lies in showing that
this condition remains sufficient ---
and in the bounded-gap case necessary ---
under mere strong operator convergence,
where operator-norm control may
completely fail.}
To our knowledge, the exact iff formulation
in the bounded-gap regime
(Thms.~\ref{thm:abstract}
$+$~\ref{thm:necessary-bounded-gap})
is not explicitly isolated in the literature.
\end{remark}

\begin{proposition}[Counterexample:
  SOT but projection failure]
\label{prop:counterexample}
There exist bounded self-adjoint $\{A_k\}$,
$A$ on $\ell^2$ with
$A_k\xrightarrow{\mathrm{SOT}} A$,
simple $\lambda\in\sigma(A)$,
and $\lambda_k\in\sigma(A_k)$,
$\lambda_k\to\lambda$, such that
$\|P_k-P^\infty\|\not\to 0$.
\end{proposition}
\begin{proof}
Set $\phi=e_1$, $\lambda=0$,
$P^\infty=e_1\otimes e_1$,
$\gamma_k:=k^{-1/2}$,
$v_k:=\tfrac{1}{\sqrt{2}}(e_1+e_2)$,
\[
  A:=\sum_{n\ge 2} n^{-1}(e_n\otimes e_n),
  \quad
  A_k:=\gamma_k(v_k\otimes v_k)
  +\sum_{n\ge 2} n^{-1}(e_n\otimes e_n).
\]
(i)~$A_k\xrightarrow{\mathrm{SOT}} A$:
$\|(A_k-A)f\|=\gamma_k|\ip{v_k}{f}|\to 0$
for each $f\in\ell^2$.
(ii)~$\lambda_k=\gamma_k\to 0$.
(iii)~Spectral gap at $\lambda_k$:
$\gamma_k'\sim\gamma_k=k^{-1/2}$.
(iv)~$\|(A_k-A)\phi\|
=\gamma_k|\ip{v_k}{e_1}|^2
=\gamma_k/2$, so
$\|(A_k-A)\phi\|/\gamma_k'\to 1/2\ne 0$.

Since the ratio condition
\eqref{eq:abstract-ratio} fails,
Thm.~\ref{thm:abstract} cannot apply;
a direct computation using the explicit
matrix structure of $A_k$ confirms
$\|P_k - P^\infty\|\not\to 0$,
consistent with the necessity result
in Thm.~\ref{thm:necessary-bounded-gap}.
\end{proof}

\begin{remark}[Sharpness]
\label{rem:counterexample-meaning}
The ratio \eqref{eq:abstract-ratio}
is tight: SOT $+$ eigenvalue convergence
alone do not suffice.
\end{remark}

%%══ 5. PSWF APPLICATION
\section{PSWF Application: Main Theorems}
\label{sec:main}

This section reduces the PSWF projection
stability problem \emph{entirely} to two
independent analytic mechanisms:
gap control (Gap-S) and
localized SOT rate (SOT-Faster).
No further PSWF-specific input is needed
beyond Lem.~\ref{lem:inherited}.
The contribution of this section is a
\emph{complete reduction} of PSWF
projection stability to two independent
analytic problems, each of which can be
studied by separate methods
(Probs.~\ref{prob:sharp-gc},
\ref{prob:sot-faster}).

Thms.~\ref{thm:convergence}
and~\ref{thm:rate} apply Thm.~\ref{thm:abstract}
with
$A_k=\mathcal{H}_{c_k}$, $A=\Hlim$,
$\phi=\Phi_n^{(\infty)}$,
$\lambda=\lambda_n^{(\infty)}$,
$\lambda_k=\lambda_n^{(c_k)}$.

\begin{remark}[Gap inequality in application]
\label{rem:gap-inequality}
Gap-S supplies the lower bound
$|\lambda_n^{(c_k)}-\lambda_{n\pm 1}^{(c_k)}|
\ge g(c_k)$, hence $\gamma_k\ge g(c_k)$.
In the ratio condition \eqref{eq:abstract-ratio}
we use $g(c_k)$ as a lower bound for
$\gamma_k$; dividing by $g(c_k)$ is
therefore sufficient to verify
\eqref{eq:abstract-ratio}.
\end{remark}

\begin{theorem}[Projection convergence]
\label{thm:convergence}
Assume Lem.~\ref{lem:inherited},
Hyp.~\ref{hyp:gap-structural} (Gap-S),
and Hyp.~\ref{hyp:sot-faster} (SOT-Faster).
For each fixed $n\in W_{c_k}$:
\[
  \|P_n^{(c_k)} - P_n^{(\infty)}\|\to 0.
\]
In particular,
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
strongly (up to phase)
and $\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$
(Rem.~\ref{rem:nacc}).
\end{theorem}

\begin{theorem}[Rate]
\label{thm:rate}
Assume additionally Hyp.~\ref{hyp:sot-rate}.
For $n\le\kappa c_k$, $n\in W_{c_k}$,
uniformly in $n$:
\[
  \|P_n^{(c_k)} - P_n^{(\infty)}\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}.
\]
If $g(c)\sim(\log c)^{-1}$:
rate $O(c_k^{-1/4}\log c_k)$.
\end{theorem}

\begin{remark}[Phase]
Phase fixed by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
for large $k$.
\end{remark}

%%══ 6. PROOFS
\section{Proofs}
\label{sec:proj}

\begin{lemma}[Approx.\ eigenvector
  $\Rightarrow$ projection bound]
\label{lem:approx-eig}
Let $A$ bounded self-adjoint,
$\lambda$ isolated and simple,
$\gamma:=\mathrm{dist}(\lambda,
\spc{A}\setminus\{\lambda\})>0$,
$P$ rank-one onto $\ker(A-\lambda)$.
By the spectral theorem
(functional calculus),
the simplicity of $\lambda$ ensures
that $(A-\lambda)$ is invertible on
$\mathrm{ran}(I-P)$ with
$\|(A-\lambda)^{-1}|_{\mathrm{ran}(I-P)}\|
\le 1/\gamma$.
For $\|\phi\|=1$ and
$\|(A-\lambda)\phi\|\le\varepsilon$:
\[
  \|(I-P)\phi\|\le\varepsilon/\gamma.
\]
\end{lemma}
\begin{proof}
$(A-\lambda)\phi=(A-\lambda)(I-P)\phi$
since $(A-\lambda)P=0$.
By the spectral theorem, all spectral
values of $A$ on $\mathrm{ran}(I-P)$
lie at distance $\ge\gamma$ from $\lambda$;
hence $(A-\lambda)^{-1}$ exists on
$\mathrm{ran}(I-P)$ with norm $\le 1/\gamma$.
Therefore $\|(I-P)\phi\|\le\varepsilon/\gamma$.
\end{proof}

\begin{remark}[Rank-one structure]
\label{rem:perturbation-mechanism}
Rank-one gives uniform resolvent bound
$1/\gamma$ on $\mathrm{ran}(I-P)$.
Gap-S supplies $\gamma\ge g(c_k)$;
SOT-Faster gives
$\varepsilon_n(k)/g(c_k)\to 0$.
\end{remark}

\begin{lemma}[SOT: approx.\ eigenvectors,
  no rate]
\label{lem:sot-approx}
$\varepsilon_n(k)=
\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
\to 0$.
\end{lemma}
\begin{proof}SOT \eqref{eq:sot}.\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:convergence}]
Fix $n\in W_{c_k}$.
\[
  \tilde\varepsilon_n(k)
  :=\|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
  \Phi_n^{(\infty)}\|
  \le\varepsilon_n(k)+C_\kappa c_k^{-1/4}.
\]
By Rem.~\ref{rem:gap-inequality},
$\gamma_k\ge g(c_k)$; hence:
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to 0\,(\text{SOT-Faster})}
  +\underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to 0
    \,(\text{Rem.}\,\ref{rem:gap-implies-o})}
  \to 0.
\end{equation}
Lem.~\ref{lem:approx-eig} with
$\gamma_k\ge g(c_k)$ gives
$\|(I-P_n^{(c_k)})\Phi_n^{(\infty)}\|\to 0$;
rank-one identity completes.
\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:rate}]
SOT-Rate: $\varepsilon_n(k)\le C'c_k^{-1/4}$;
$\tilde\varepsilon_n(k)\le(C'+C_\kappa)c_k^{-1/4}$;
Lem.~\ref{lem:approx-eig} with
$\gamma_k\ge g(c_k)$ gives the bound.
\end{proof}

\begin{remark}[No accumulation]
\label{rem:nacc}
Gap-S bounds finitely many
$\lambda_n^{(\infty)}$ in any interval.
\end{remark}

%%══ 9. STRUCTURAL EVIDENCE
\section{Structural Evidence for Gap-S
  and SOT-Faster}
\label{sec:heuristic}

\textbf{Non-rigorous throughout.}

\subsection{Sigmoid transition (Widom)}
Widom~\cite{Widom1964} predicts
$\lambda_n^{(c)}\approx
(1+e^{\pi(n-2c/\pi)/\sigma(c)})^{-1}$
with $\sigma(c)\asymp\log c$;
average spacing $\sim(\log c)^{-1}$ in $W_c$,
consistent with Gap-S.

\subsection{Density vs.\ minimum gap}
Widom controls density, not minimum gap.
Typical spacing $\gg c^{-1/4}$ supports
Gap-S for typical indices;
rigorous lower bound needs
determinantal methods
(Prob.~\ref{prob:sharp-gc}).

\subsection{SOT-Faster plausibility}
Small off-diagonal overlaps in
\eqref{eq:formal-expansion}
predict $\varepsilon_n(k)=O(c_k^{-1/4})$;
Prob.~\ref{prob:off-diagonal}.

\begin{remark}[Status]
Section~\ref{sec:heuristic} provides
structural evidence, not proof.
\end{remark}

%%══ 10. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent\textbf{Conjecture-level:}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove Gap-S with $g(c)\sim(\log c)^{-1}$
via sine-kernel repulsion.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$;
equivalently: local norm-resolvent bound
at scale $g(c_k)$.
\end{problem}

\begin{problem}[Full iff, vanishing gap]
\label{prob:full-iff}
Characterise when
$\|P_k-P^\infty\|\to 0$
$\Rightarrow$
$\|(A_k-A)\phi\|/\gamma_k\to 0$
for $\gamma_k\to 0$.
This is the precise analytic difficulty
encoded in Hyp.~\ref{hyp:sot-faster}.
\end{problem}

\noindent\textbf{Technical-level:}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
$\varepsilon_n(k)=O(c_k^{-1/4})$.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
for $m\ne n$.
\end{problem}

\begin{problem}[Full family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family}.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$? Paper~XIII.
\end{problem}
