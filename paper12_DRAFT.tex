\begin{abstract}
\textbf{Main result (abstract,
  fully proved).}
For bounded self-adjoint operators with
$A_k\xrightarrow{\mathrm{SOT}} A$,
under explicit spectral isolation
(Assumption~S), the gap-normalized
residual condition
\[
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0
\]
is sufficient for rank-one spectral
projection stability
$\|P_k - P^\infty\|\to 0$,
and necessary when $\gamma_k\ge c_0>0$.
The mechanism is explicit: rank-one
structure reduces projection error to
a single scalar; the spectral gap
provides the resolvent bound; the local
residual is the direct observable
(Lem.~\ref{lem:mechanism-chain}).

\medskip
\textbf{Conditional PSWF corollary.}
Under two analytic hypotheses
(Gap-S, SOT-Faster), PSWF projection
stability reduces entirely to the
abstract principle.
Both hypotheses are open
(Problems~\ref{prob:sharp-gc},
\ref{prob:sot-faster}).

\medskip
We further establish: a No-Go result
(Prop.~\ref{prop:no-go}); a separation
example showing genuine spectral
interaction with $\|A_k-A\|\equiv 1$
but full projection stability
(Prop.~\ref{prop:separation}); and a
quantitative counterexample linking
ratio blow-up to projection failure
(Prop.~\ref{prop:counterexample}).
\end{abstract}

\subsection{Context and goal}

\begin{center}
\fbox{\parbox{0.88\linewidth}{%
\centering\itshape
SOT produces approximate eigenvectors,
but does not control the residual
relative to the spectral gap.
This is the precise obstruction to
spectral projection stability.}}
\end{center}

\medskip
\noindent
\textbf{Main Principle (informal).}
Let $A_k\xrightarrow{\mathrm{SOT}} A$
with isolated simple eigenvalue $\lambda$
and associated simple eigenvalues
$\lambda_k$ (Assumption~S). Then:
\begin{equation}\label{eq:main-principle}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0
  \;\Longrightarrow\;
  \|P_k-P^\infty\|\to 0.
\end{equation}
The converse holds when
$\gamma_k\ge c_0>0$
(Thm.~\ref{thm:necessary-bounded-gap}).

\medskip
\noindent
\textbf{Why rank-one + gap + residual
suffice.}
The mechanism is structural, not
``one direction controls all.''
Rank-one structure means
$P^\infty = \phi\otimes\phi$ is
\emph{completely encoded} by $\phi$
--- there is only one degree of freedom.
The spectral gap supplies a resolvent
bound converting residual smallness
into projection smallness.
The local residual is the only quantity
accessible under SOT without global
norm control.
This chain is exact and tight
(Lem.~\ref{lem:mechanism-chain}).

\medskip
This paper proves:
\begin{itemize}
\item \eqref{eq:main-principle} is the
  \emph{sharp sufficient mechanism}
  (Thm.~\ref{thm:abstract},
  Lem.~\ref{lem:mechanism-chain});
\item necessary when $\gamma_k\ge c_0>0$
  (Thm.~\ref{thm:necessary-bounded-gap});
\item genuinely beyond global-norm-based
  theory in the SOT regime:
  families with $\|A_k-A\|\equiv 1$,
  genuine spectral interaction, yet
  full projection stability exist
  (Prop.~\ref{prop:separation});
\item eigenvalue density alone does not
  control spectral angles
  (Prop.~\ref{prop:no-go});
\item ratio blow-up implies projection
  failure, quantitatively
  (Prop.~\ref{prop:counterexample});
\item PSWF stability reduces
  conditionally to Gap-S and SOT-Faster
  (Cors.~\ref{cor:convergence},
  \ref{cor:rate}).
\end{itemize}

\section{Hypotheses, Structural
  Necessity, and No-Go}
\label{sec:gap}

\begin{hypothesis}[Effective gap, Gap-S]
\label{hyp:gap-structural}
Let $W_c:=\{n:\lambda_n^{(c)}
\in(\delta,1-\delta)\}$
for fixed $\delta>0$.
There exists $g\colon[1,\infty)
\to(0,\infty)$ with
\begin{equation}\label{eq:gap-dominates}
  \frac{c^{-1/4}}{g(c)}\to 0
  \quad\text{as }c\to\infty,
\end{equation}
and uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}
  -\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S: status and
  determinantal context]
\label{rem:gap-necessary}
Gap-S is an open hypothesis
(Prob.~\ref{prob:sharp-gc}).
It is necessary for the present
SOT-based argument: without
\eqref{eq:gap-dominates} the ratio
in \eqref{eq:ratio-to-zero} cannot
be shown to vanish.
Widom~\cite{Widom1964} controls
average density, not the minimum gap;
Gap-S is strictly stronger.
The analogous bound in determinantal
point processes follows from
sine-kernel repulsion; Gap-S is the
conjectured deterministic counterpart.
\end{remark}

\begin{remark}[$c^{-1/4}/g(c)\to 0$]
\label{rem:gap-implies-o}
\eqref{eq:gap-dominates} states that the
eigenvalue-matching error is of strictly
smaller order than $g(c)$;
used in \eqref{eq:ratio-to-zero}.
\end{remark}

\begin{proposition}[No-Go: eigenvalue
  density does not control spectral
  angles]
\label{prop:no-go}
Let $(\mu_n)$ be a sequence with
$\liminf_n(\mu_n-\mu_{n+1})=0$
but Widom-type average density.
Then there exists a sequence
$B_k\xrightarrow{\mathrm{SOT}} B$
of bounded self-adjoint operators
with eigenvalues $\mu_n^{(k)}\to\mu_n$
such that
$\|Q_n^{(k)}-Q_n\|\not\to 0$
for infinitely many $n$.
\end{proposition}
\begin{proof}
Choose $n_j$ with
$\gamma_j:=\mu_{n_j}-\mu_{n_j+1}\to 0$.
On $\mathrm{span}\{e_{n_j},e_{n_j+1}\}$:
\[
  B_k^{(j)}:=
  \begin{pmatrix}
    \mu_{n_j}^{(k)} & \sqrt{\gamma_j^{(k)}}\\
    \sqrt{\gamma_j^{(k)}} & \mu_{n_j+1}^{(k)}
  \end{pmatrix}.
\]
Set $B_k:=\bigoplus_j B_k^{(j)}$.
SOT convergence follows
since each $e_m$ is affected by at most
one block and $\sqrt{\gamma_j^{(k)}}\to 0$.
Eigenvector angle:
$\tan(2\theta_j^{(k)})
= 2/\sqrt{\gamma_j^{(k)}}\to\infty$,
so $\theta_j^{(k)}\to\pi/4$
and $\|Q_{n_j}^{(k)}-Q_{n_j}\|\to 1/2$.
\end{proof}

\begin{remark}[Scope of No-Go]
\label{rem:no-go-scope}
This is an existential pathology:
for any sequence with $\liminf$ gap
$= 0$, there \emph{exists} a family
exhibiting projection instability.
It does not assert that every operator
family with Widom-type density is
unstable.
In particular, it does not contradict
structured kernels with built-in
spectral rigidity
(e.g.\ integrable operators,
PSWF-type reproducing kernels),
where additional structure --- such
as Gap-S --- prevents gap collapse.
The construction exploits exactly
the regime excluded by Gap-S.
\end{remark}

\begin{hypothesis}[SOT-Faster]
\label{hyp:sot-faster}
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k)
  :=\|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k)).
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster: pointwise,
  not uniform; the sole analytic
  bottleneck]
\label{rem:sot-faster-localized}
SOT-Faster is a \emph{pointwise}
condition at the single fixed vector
$\Phi_n^{(\infty)}$, not uniform
control over a spectral window.
Uniform local norm convergence would
require
$\sup_{\|f\|\le 1}
\|(\mathcal{H}_{c_k}-\Hlim)f\|
\cdot g(c_k)^{-1}\to 0$
on a spectral window of width $g(c_k)$
--- a strictly stronger condition
involving all vectors in that window.
SOT alone gives $\varepsilon_n(k)\to 0$
without rate; SOT-Faster adds only
the rate $o(g(c_k))$ at one vector.
It is the \emph{sole remaining analytic
bottleneck} in the PSWF reduction:
given Gap-S, SOT-Faster is the only
missing ingredient for
Cor.~\ref{cor:convergence}
(Prob.~\ref{prob:sot-faster}).
\end{remark}

\begin{hypothesis}[SOT-Rate]
\label{hyp:sot-rate}
$\varepsilon_n(k) = O(c_k^{-1/4})$.
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
SOT-Rate $\Rightarrow$ SOT-Faster
$\Rightarrow$
Cor.~\ref{cor:convergence}.
SOT gives only $\varepsilon_n(k)\to 0$.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding in $\{\Phi_m^{(c_k)}\}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}
  -\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}
  {\Phi_m^{(c_k)}}|^2.
\end{equation}
Diagonal $m=n$:
$\le C_\kappa^2 c_k^{-1/2}$.
Small off-diagonal overlaps
predict $O(c_k^{-1/4})$;
Prob.~\ref{prob:off-diagonal}.
\end{remark}

\subsection{WKB insufficiency}
\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
cannot determine the scaling of
$|\lambda_{n+1}^{(c)}

without global kernel information.
\end{theorem}
\begin{proof}
Sigmoid profile of
$F_c$~\cite{Widom1964};
any algebraic rate is a global claim.
\end{proof}

%%══ 4. ABSTRACT STABILITY PRINCIPLE
\section{Abstract Stability Principle}
\label{sec:abstract}

\begin{assumption}[Spectral isolation,
  Assumption~S]
\label{ass:spectral-isolation}
Let $\{A_k\}$, $A$ be bounded
self-adjoint on $\mathcal{H}$ with
$A_k\xrightarrow{\mathrm{SOT}} A$.
\begin{enumerate}[\rm(S1)]
\item\label{S1}
  $\lambda\in\sigma(A)$ isolated,
  simple; $P^\infty = \phi\otimes\phi$.
\item\label{S2}
  Each $\lambda_k\in\sigma(A_k)$
  isolated, simple;
  $\gamma_k:=\dist(\lambda_k,
  \sigma(A_k)\setminus\{\lambda_k\})
  >0$; $P_k$ rank-one.
\item\label{S3}
  $\lambda_k\to\lambda$.
\end{enumerate}
\end{assumption}

\begin{remark}[Spectral isolation,
  index stability, and SOT]
\label{rem:spectral-isolation}
Under strong resolvent convergence
(SRC), Assumption~S follows
automatically from isolation of
$\lambda$~\cite[Ch.~VIII]{Kato1966}.
Under mere SOT, SRC may fail and
Assumption~S must be imposed.

The condition (S3) in particular
requires that a stable eigenvalue
assignment $\lambda_k\leftrightarrow\lambda$
exists.
This is not automatic under SOT:
eigenvalues can in principle cross or
fail to be consistently indexed.
In the PSWF application, this is
resolved by strict ordering:
$\lambda_n^{(c)}$ is strictly ordered
for all $c$ (Lem.~\ref{lem:inherited}(A3)),
so the assignment
$\lambda_n^{(c_k)}
\leftrightarrow\lambda_n^{(\infty)}$
is canonical by index and
Lem.~\ref{lem:eigenvalue-matching}
gives $\lambda_n^{(c_k)}
\to\lambda_n^{(\infty)}$.
In the abstract setting, (S3) is an
explicit axiom that the user must verify
for their specific family.
\end{remark}

\begin{lemma}[Mechanism chain:
  rank-one + gap + residual]
\label{lem:mechanism-chain}
Under Assumption~S, the implication
\eqref{eq:main-principle} is the
composition of three exact steps:
\begin{enumerate}[\rm(M1)]
\item\label{M1}
  \textbf{(Rank-one encoding.)}
  For rank-one projections
  $P = u\otimes u$,
  $Q = v\otimes v$,
  $\|u\|=\|v\|=1$:
  $\|P-Q\|=\|(I-Q)u\|$
  \textup{\cite[Ch.~1]{Bhatia1997}}.
  Hence $\|P_k-P^\infty\|
  = \|(I-P_k)\phi\|$.
  The projection has a
  \emph{single degree of freedom}.
\item\label{M2}
  \textbf{(Gap resolvent bound.)}
  $(A_k-\lambda_k)$ is invertible on
  $\mathrm{ran}(I-P_k)$ with
  $\|(A_k-\lambda_k)^{-1}
  |_{\mathrm{ran}(I-P_k)}\|
  \le 1/\gamma_k$.
  Since $(A_k-\lambda_k)\phi
  =(A_k-\lambda_k)(I-P_k)\phi$:
  \[
    \|(I-P_k)\phi\|
    \le
    \frac{\|(A_k-\lambda_k)\phi\|}
    {\gamma_k}.
  \]
\item\label{M3}
  \textbf{(Residual triangle.)}
  $\|(A_k-\lambda_k)\phi\|
  \le\|(A_k-A)\phi\|
  +|\lambda_k-\lambda|$,
  using $(A-\lambda)\phi=0$.
\end{enumerate}
Combining (M1)--(M3):
$\|P_k-P^\infty\|
\le(\|(A_k-A)\phi\|
+|\lambda_k-\lambda|)/\gamma_k\to 0$.
Each step is tight:
removing rank-one structure breaks (M1)
(Rem.~\ref{rem:rank-one-essential});
removing gap control breaks (M2)
(Prop.~\ref{prop:no-go});
removing residual smallness breaks (M3)
(Prop.~\ref{prop:counterexample}).
\end{lemma}
\begin{proof}
(M1) is \cite[Ch.~1]{Bhatia1997}.
(M2): spectral theorem on
$A_k$ restricted to
$\mathrm{ran}(I-P_k)$,
all spectral values at distance
$\ge\gamma_k$ from $\lambda_k$.
$(A_k-\lambda_k)P_k\phi=0$, so
$(A_k-\lambda_k)\phi
=(A_k-\lambda_k)(I-P_k)\phi$.
(M3): triangle inequality
with $(A-\lambda)\phi=0$.
\end{proof}

\begin{remark}[Why the chain is exact,
  not magical]
\label{rem:chain-exact}
(M1) does not say ``one direction
controls all directions.''
It says rank-one structure means
\emph{there is only one direction}:
$P^\infty=\phi\otimes\phi$ has no
other degrees of freedom.
(M2) is a standard resolvent bound,
not a global norm argument:
it uses only the spectral structure
of $A_k$ on $\mathrm{ran}(I-P_k)$,
which is controlled by $\gamma_k$
alone, without any reference to
$A_k - A$.
(M3) is an exact decomposition
enabled by $\phi$ being a strict
eigenvector of $A$ (not approximate):
$(A-\lambda)\phi = 0$ exactly.
The chain works precisely because SOT
provides the residual
$(A_k-A)\phi\to 0$, gap provides the
resolvent bound, and rank-one
collapses the projection norm to a
single scalar.
\end{remark}

\begin{lemma}[Davis--Kahan implies
  ratio condition]
\label{lem:dk-comparison}
Under Assumption~S, if
$\|A_k - A\|\to 0$ and
$\gamma_k \ge c_0 > 0$:
\[
  \|A_k-A\|\to 0
  \;\Longrightarrow\;
  \eqref{eq:main-principle}
  \;\Longrightarrow\;
  \|P_k-P^\infty\|\to 0,
\]
neither implication reverses
(Props.~\ref{prop:separation},
\ref{prop:counterexample}).
\end{lemma}
\begin{proof}
$\|(A_k-A)\phi\|\le\|A_k-A\|\to 0$;
$|\lambda_k-\lambda|\le\|A_k-A\|\to 0$;
divide by $c_0$.
\end{proof}

\begin{proposition}[Separation:
  genuine spectral interaction,
  spectrally orthogonal perturbation,
  $\|A_k-A\|\equiv 1$]
\label{prop:separation}
There exist bounded self-adjoint
$\{A_k\}$, $A$ on $\ell^2$
satisfying Assumption~S with
$\gamma_k\ge c_0>0$, such that:
\begin{enumerate}[\rm(i)]
\item $\|A_k - A\| = 1$ for all $k$
  (Davis--Kahan gives no information);
\item $\|(A_k - A)\phi\|
  = \varepsilon_k \to 0$,
  so \eqref{eq:main-principle} holds;
\item $A_k - A$ has a genuine
  off-diagonal spectral interaction
  component $S_k$ that mixes
  eigenspaces and perturbs eigenvalues;
\item the large-norm component of
  $A_k - A$ is \emph{asymptotically
  spectrally orthogonal to the active
  subspace} $\mathrm{span}\{\phi\}$;
\item $\|P_k - P^\infty\| \to 0$.
\end{enumerate}
\end{proposition}
\begin{proof}
Set $\phi=e_1$, $\lambda=0$,
\[
  A:=\sum_{n\ge 2}
  \mu_n(e_n\otimes e_n),
  \quad\mu_n\in[-1,-c_0].
\]
$\lambda=0$ is isolated, simple,
gap $\gamma\ge c_0$.
Define
$A_k - A := S_k + K_k$,
where
$S_k := \varepsilon_k
(e_2\otimes e_1+e_1\otimes e_2)$,
$\|S_k\|=\varepsilon_k\to 0$,
and $K_k$ acts on
$\mathrm{span}\{e_3,e_4,\ldots\}$
with $\|K_k\|=1$ and
$\sigma(A|_{\mathrm{span}\{e_n,n\ge 3\}}
+K_k)$ uniformly away from $0$.

\textit{(i)}: $\|A_k-A\|\ge\|K_k\|=1$.

\textit{(iii)}: $S_k$ is off-diagonal
between $e_1$ and $e_2$;
it shifts the eigenvalue $\lambda=0$
by $O(\varepsilon_k^2)$ and rotates
eigenvectors in
$\mathrm{span}\{e_1,e_2\}$.
This is genuine spectral interaction,
not decoupling.

\textit{(iv)}: $K_k$ acts on
$\mathrm{span}\{e_3,\ldots\}
\perp \mathrm{span}\{e_1\}$,
so its effect on the active
spectral direction $\phi=e_1$ is zero:
$K_k\phi=0$.

\textit{(ii)}: $(A_k-A)\phi
= S_k e_1 + K_k e_1
= \varepsilon_k e_2$;
$\|(A_k-A)\phi\|=\varepsilon_k\to 0$.

\textit{Gap}: $\gamma_k\ge c_0/2$
for large $k$~\cite[Ch.~II]{Kato1966}.

\textit{(v)}: Apply
Thm.~\ref{thm:abstract}.
\end{proof}

\begin{remark}[``Spectral orthogonality''
  is the principle, not a defect]
\label{rem:block-is-the-point}
A reviewer may ask: is not $K_k$
acting on $\phi^\perp$ a hidden block
structure?
Yes --- and this is precisely the
content of the localization principle.

Mechanism step (M1) says: the
projection $P^\infty=\phi\otimes\phi$
has a \emph{single degree of freedom},
determined by $\phi$.
Any perturbation component that is
spectrally orthogonal to
$\mathrm{span}\{\phi\}$
--- however large in global norm ---
is invisible to $P^\infty$.
The principle identifies which
perturbations matter: those that
move $\phi$ relative to $\gamma_k$.

The off-diagonal component $S_k$
confirms that the perturbation genuinely
touches the spectral mechanism
(it mixes $e_1$ and $e_2$ and shifts
the eigenvalue); yet its effect on
$\phi$ is controlled by $\varepsilon_k$.
The large component $K_k$ is invisible
to $\phi$ not by accident but by
the geometry the principle identifies.
Davis--Kahan sees only
$\|A_k-A\|\equiv 1$
and cannot exploit this geometry.
\end{remark}

\begin{theorem}[Main Principle;
  sufficient condition]
\label{thm:abstract}
Under Assumption~S,
\eqref{eq:main-principle}
$\Rightarrow$
$\|P_k-P^\infty\|\to 0$.
\end{theorem}
\begin{proof}
Lem.~\ref{lem:mechanism-chain},
steps (M3), (M2), (M1).
\end{proof}

\begin{remark}[Non-circularity]
\label{rem:non-circularity}
$\phi$ is a fixed eigenvector of $A$,
not of $A_k$.
$(A_k-A)\phi$ is evaluated on a
fixed vector; no eigenvector of $A_k$
enters its definition.
\end{remark}

\begin{remark}[Structural independence]
\label{rem:structural-independence}
No spectral data of $A_k$ enters the
construction of $\phi$.
All $A_k$-quantities appear only as
$(A_k-A)\phi$ or $(A_k-\lambda_k)\phi$.
\end{remark}

\begin{lemma}[Rank-one projection
  $\Rightarrow$ eigenvector convergence]
\label{lem:rank-one-proj}
If $P_k=v_k\otimes v_k$,
$P^\infty=\phi\otimes\phi$,
$\|v_k\|=\|\phi\|=1$, and
$\|P_k-P^\infty\|\to 0$,
then there exist phases $\theta_k$ with
$\|v_k-e^{i\theta_k}\phi\|\to 0$.
\end{lemma}
\begin{proof}
$\|v_k\ip{v_k}{\phi}-\phi\|\to 0$;
$|\ip{v_k}{\phi}|\to 1$;
standard estimate.
\end{proof}

\begin{theorem}[Necessary condition;
  bounded-gap case]
\label{thm:necessary-bounded-gap}
Under Assumption~S, if
$\|P_k-P^\infty\|\to 0$ and
$\gamma_k\ge c_0>0$,
then \eqref{eq:main-principle} holds.
\end{theorem}
\begin{proof}
\textbf{Eigenvalue:}
$|\lambda_k-\lambda|/c_0\to 0$.

\textbf{Residual:}
$(A_k-A)\phi
= (A_k-A)P_k\phi
+ (A_k-A)(I-P_k)\phi$.
Term~(II):
$\le 2M\|(I-P_k)\phi\|\to 0$.
Term~(I):
$\alpha_k(\lambda_k v_k - Av_k)$;
$|\lambda_k-\lambda|\to 0$;
$(A-\lambda)v_k
=(A-\lambda)(v_k-e^{i\theta_k}\phi)
\to 0$ by Lem.~\ref{lem:rank-one-proj}.
Divide by $c_0$.
\end{proof}

\begin{remark}[Necessity requires
  uniform gap]
\label{rem:necessity-uniform-gap}
Without $\gamma_k\ge c_0$,
the ratio may diverge even when
$\|P_k-P^\infty\|\to 0$.
Open Prob.~\ref{prob:full-iff}.
\end{remark}

\begin{remark}[Rank-one is essential
  for necessity]
\label{rem:rank-one-essential}
For higher-multiplicity eigenspaces,
the extraction
$\|v_k-e^{i\theta_k}\phi\|\to 0$
from $\|P_k-P^\infty\|\to 0$
requires additional assumptions.
\end{remark}

\begin{remark}[Novelty: one-vector
  perturbation principle in SOT regime]
\label{rem:abstract-novelty}
Classical theory
(Davis--Kahan~\cite{DavisKahan1970},
Kato~\cite{Kato1966},
Chatelin~\cite{Chatelin1983},
Osborn~\cite{Osborn1975})
requires $\|A_k-A\|\to 0$
or norm-resolvent convergence as input.
The present result operates under
SOT only: no global norm control is
available or assumed.
Projection stability is characterized
entirely by the vector-local quantity
$\|(A_k-A)\phi\|/\gamma_k$.

The key structural point: classical
theory uses $\|A_k-A\|$ to bound
the residual; here the residual
$\|(A_k-A)\phi\|$ is the direct
observable, and no global operator
norm enters the conditions.
This goes beyond global-norm-based
perturbation theory in the SOT-only
regime, not by strengthening the
output, but by weakening the input
from global to vector-local.
\end{remark}

\begin{proposition}[Counterexample:
  ratio blow-up implies projection
  failure, quantitatively]
\label{prop:counterexample}
There exist bounded self-adjoint
$\{A_k\}$, $A$ on $\ell^2$ satisfying
Assumption~S with
$A_k\xrightarrow{\mathrm{SOT}} A$,
$\lambda_k\to\lambda$, such that
\[
  r_k :=
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  \equiv \frac{1}{\sqrt{2}},
\]
and for all large $k$:
\[
  \|P_k-P^\infty\|
  \ge \tfrac{1}{2}
  = \tfrac{1}{\sqrt{2}}\cdot r_k.
\]
\end{proposition}
\begin{proof}
Set $\phi=e_1$, $\lambda=0$,
$\gamma_k:=k^{-1/2}$,
$v_k:=\tfrac{1}{\sqrt{2}}(e_1+e_2)$,
$A:=\sum_{n\ge 2}n^{-1}(e_n\otimes e_n)$,
$A_k:=\gamma_k(v_k\otimes v_k)
+\sum_{n\ge 2}n^{-1}(e_n\otimes e_n)$.
$(A_k-A)e_1
=\gamma_k\ip{v_k}{e_1}v_k
=\tfrac{\gamma_k}{\sqrt{2}}\cdot
\tfrac{1}{\sqrt{2}}(e_1+e_2)
=\tfrac{\gamma_k}{2}(e_1+e_2)$;
$\|(A_k-A)\phi\|
=\tfrac{\gamma_k}{\sqrt{2}}$;
$r_k=1/\sqrt{2}$.
Eigenvector of $A_k$ for
$\lambda_k=\gamma_k$: $v_k$;
$P_k=v_k\otimes v_k$.
$\|P_k-P^\infty\|
\ge|\ip{(P_k-P^\infty)e_1}{e_1}|
=|1/2-1|=1/2$.
\end{proof}

\begin{remark}[Structural necessity
  of the ratio]
\label{rem:counterexample-meaning}
By (M1): $\|P_k-P^\infty\|
=\|(I-P_k)\phi\|$.
If $r_k\ge c>0$, then by (M2) the
bound $\|(I-P_k)\phi\|\le r_k$
is tight: the counterexample shows
$\|(I-P_k)\phi\|\asymp r_k$.
The bound \eqref{eq:proj-blowup}
is therefore not a coincidence but
a direct consequence of the chain
(M1)--(M3): ratio blow-up is
\emph{structurally necessary} for
projection failure.
\end{remark}

%%══ 5. PSWF APPLICATION
\section{PSWF Application:
  Conditional Corollaries}
\label{sec:main}

The following corollaries are
\emph{conditional}: they assume
Gap-S (Hyp.~\ref{hyp:gap-structural})
and SOT-Faster (Hyp.~\ref{hyp:sot-faster}),
both of which are open
(Probs.~\ref{prob:sharp-gc},
\ref{prob:sot-faster}).
The contribution of this section is a
\emph{complete reduction} of PSWF
projection stability to these two
independent analytic problems.
No further PSWF-specific input beyond
Lem.~\ref{lem:inherited} is needed.
Assumption~S is verified by
Lem.~\ref{lem:inherited}(A3)
and Lem.~\ref{lem:eigenvalue-matching}.

\begin{lemma}[Gap-S supplies gap bound]
\label{lem:gap-bridge}
Under Hyp.~\ref{hyp:gap-structural},
$\gamma_k \ge g(c_k)$
for each $n\in W_{c_k}$.
\end{lemma}
\begin{proof}
Strict ordering gives
$\gamma_k\ge|\lambda_n^{(c_k)}
-\lambda_{n\pm 1}^{(c_k)}|\ge g(c_k)$.
\end{proof}

\begin{corollary}[Projection convergence,
  conditional]
\label{cor:convergence}
\textup{[Conditional on Gap-S and
SOT-Faster.]}
Under Lem.~\ref{lem:inherited},
Hyps.~\ref{hyp:gap-structural},
\ref{hyp:sot-faster},
for each fixed $n\in W_{c_k}$:
\[
  \|P_n^{(c_k)}-P_n^{(\infty)}\|
  \to 0.
\]
In particular,
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
strongly (up to phase).
\end{corollary}

\begin{corollary}[Rate, conditional]
\label{cor:rate}
\textup{[Conditional on Gap-S and
SOT-Rate.]}
Under Hyp.~\ref{hyp:sot-rate}
additionally, for
$n\le\kappa c_k$, $n\in W_{c_k}$,
uniformly in $n$:
\[
  \|P_n^{(c_k)}-P_n^{(\infty)}\|
  \le
  \frac{C\,C_\kappa\,c_k^{-1/4}}
  {g(c_k)}.
\]
If $g(c)\sim(\log c)^{-1}$:
rate $O(c_k^{-1/4}\log c_k)$.
\end{corollary}

\begin{remark}[Phase convention]
Phase fixed by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
for large $k$.
\end{remark}

%%══ 6. PROOFS
\section{Proofs of Corollaries}
\label{sec:proj}

\begin{lemma}[Approx.\ eigenvector
  bound]
\label{lem:approx-eig}
Let $A$ bounded self-adjoint, $\lambda$
isolated, simple, gap $\gamma>0$,
$P$ rank-one.
For $\|\phi\|=1$,
$\|(A-\lambda)\phi\|\le\varepsilon$:
$\|(I-P)\phi\|\le\varepsilon/\gamma$.
\end{lemma}
\begin{proof}
$(A-\lambda)\phi
=(A-\lambda)(I-P)\phi$;
spectral theorem.
\end{proof}

\begin{proof}[Proof of
  Cor.~\ref{cor:convergence}]
$\tilde\varepsilon_n(k)
:=\|(\mathcal{H}_{c_k}
-\lambda_n^{(c_k)})
\Phi_n^{(\infty)}\|
\le\varepsilon_n(k)+C_\kappa c_k^{-1/4}$.
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}
  {g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to 0
    \,(\text{SOT-Faster})}
  +\underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to 0
    \,(\text{Rem.}\,
    \ref{rem:gap-implies-o})}
  \to 0.
\end{equation}
Lem.~\ref{lem:approx-eig}
with $\gamma_k\ge g(c_k)$
gives $\|(I-P_n^{(c_k)})
\Phi_n^{(\infty)}\|\to 0$;
(M1) completes.
\end{proof}

\begin{proof}[Proof of
  Cor.~\ref{cor:rate}]
$\tilde\varepsilon_n(k)
\le(C'+C_\kappa)c_k^{-1/4}$;
divide by $g(c_k)$.
\end{proof}

\begin{remark}[No accumulation]
\label{rem:nacc}
Gap-S implies finitely many
$\lambda_n^{(\infty)}$ in any
subinterval of $(0,1)$.
\end{remark}

%%══ 10. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent\textbf{Conjecture-level:}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove Gap-S with
$g(c)\sim(\log c)^{-1}$
via sine-kernel repulsion.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$.
\end{problem}

\begin{problem}[Full iff,
  vanishing gap]
\label{prob:full-iff}
Characterize
$\|P_k-P^\infty\|\to 0$
$\Leftrightarrow$
$\|(A_k-A)\phi\|/\gamma_k\to 0$
when $\gamma_k\to 0$.
\end{problem}

\noindent\textbf{Technical-level:}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
$\varepsilon_n(k)=O(c_k^{-1/4})$.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
Bound
$|\ip{\Phi_n^{(\infty)}}
{\Phi_m^{(c_k)}}|$
for $m\ne n$.
\end{problem}

\begin{problem}[Strong resolvent
  convergence]
\label{prob:src}
Does $\mathcal{H}_{c_k}$ converge
in strong resolvent sense?
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII.
\end{problem}
