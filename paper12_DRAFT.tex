\begin{abstract}
We establish the Riesz-basis criterion for the limiting
orthonormal system $\{\Phi_n^{(\infty)}\}$ constructed in
Paper~XI along the diagonal subsequence $c_k$.
The central equivalence is:
\[
  \{\Phi_n^{(\infty)}\} \text{ is a Riesz basis}
  \;\Longleftrightarrow\;
  \text{Gap} + \text{Stability} + \text{Nondegeneracy}.
\]
Under this criterion, the limiting operator $\Hlim$
becomes a canonical object, independent of the diagonal
subsequence up to phase: the three structural open axes
of Paper~XI (subsequence dependence, NAcc as external axiom,
existential sweeping) collapse to derived consequences.
The hard core of the paper is Section~\ref{sec:gap}:
spectral separation in the transition window $W_c$,
established via a WKB/Airy rescaling.
All other sections are standard functional-analytic machinery.
\end{abstract}

\section{Main Statement}

\begin{theorem}[Riesz-basis criterion]
\label{thm:riesz-criterion}
Let $\{\Phi_n^{(\infty)}, \lambda_n^{(\infty)}\}$ be the
orthonormal system of Paper~XI
(Lem.~\ref{lem:XI-diagonal}), defined along the fixed
diagonal subsequence $c_k$.
Let $\mathcal{K} := \overline{\mathrm{span}}\,
\{\Phi_n^{(\infty)} : n\ge 0\}\subset L^2(\mathbb{R})$.
The following are equivalent:
\begin{enumerate}[\rm(i)]
\item $\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$.
\item The Gram matrix
      $\Gram_{mn} := \ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}$
      defines a bounded invertible operator on $\ell^2$.
\item \textbf{(Gap condition, Sec.~\ref{sec:gap}):}
      $|\lambda_n^{(c)} - \lambda_{n+1}^{(c)}|
      \ge C_\kappa c^{-3/4}$ for all $n\le\kappa c$
      (\hyperref[hyp:gap]{Hyp.~Gap}).
\item \textbf{(Stability condition):}
      the family $\{\Phi_n^{(c_k)}\}$ is uniformly
      Riesz-stable along $c_k$
      (Def.~\ref{def:riesz-stable}).
\end{enumerate}
(i)$\Leftrightarrow$(ii): Bari--Krein theorem
(Sec.~\ref{sec:bari-krein}).\\
(ii)$\Leftrightarrow$(iii)+(iv): Gram-matrix control
(Sec.~\ref{sec:gram}).\\
(iii): proved in Section~\ref{sec:gap}
(WKB/Airy route).
\end{theorem}

\begin{remark}[Logical status]
\label{rem:main-status}
Theorem~\ref{thm:riesz-criterion} is a criterion:
it reduces the Riesz-basis property to three
verifiable conditions.
Section~\ref{sec:gap} is the only section
where genuinely new analysis is required.
The remaining implications are functional-analytic
standard machinery (Bari--Krein, Gram-matrix bounds).
\end{remark}

\begin{theorem}[Canonicalization]
\label{thm:canonical}
Assume the Gap condition
(\hyperref[hyp:gap]{Hyp.~Gap}) and
the Stability condition
(Def.~\ref{def:riesz-stable}).
Then:
\begin{enumerate}[\rm(a)]
\item $\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$.
\item \textbf{(Subsequence independence.)}
      Any two diagonal subsequences $c_k, c_k'$
      satisfying the conditions of Paper~XI,
      Lem.~\ref{lem:XI-diagonal}
      yield limit systems that are unitarily equivalent
      up to phase permutation.
\item \textbf{(NAcc as theorem.)}
      The No-Accumulation hypothesis
      (Paper~XI, Hyp.~4.5)
      holds unconditionally under \hyperref[hyp:gap]{Hyp.~Gap}.
\item \textbf{(Canonical object.)}
      $\Hlim$ is a well-defined self-adjoint operator
      on $\mathcal{K}$, independent of the choice
      of diagonal subsequence up to unitary equivalence.
\end{enumerate}
\end{theorem}

\begin{remark}[What remains open]
\label{rem:remaining-open}
Theorem~\ref{thm:canonical} establishes canonicalization
\emph{within} $\mathcal{K}$.
The question whether $\mathcal{K}=L^2(\mathbb{R})$
(completeness) is \emph{not} resolved here;
it is Open Problem~\ref{prob:completeness}.
Density of $\{\lambda_n^{(\infty)}\}$ also remains open
(Paper~XI, Conj.~6.1).
\end{remark}

\section{Spectral Separation in $W_c$}

\subsection{Setup: the prolate differential operator}
\label{subsec:prolate-setup}

The PSWF eigenvalues $\lambda_n^{(c)}$ of the
concentration operator $\mathcal{H}_c$ are related to
the eigenvalues of the prolate spheroidal differential
operator
\begin{equation}\label{eq:Lc}
  \Lc := -\frac{d}{dx}\Bigl[(1-x^2)\frac{d}{dx}\Bigr]
           + c^2 x^2,
  \quad x\in[-1,1],
\end{equation}
with Dirichlet (or natural) boundary conditions.
For each $c>0$, $\Lc$ has simple, real eigenvalues
$\chi_n^{(c)}$, $n\ge 0$, with
$\chi_0^{(c)}<\chi_1^{(c)}<\cdots\nearrow\infty$.
The PSWF concentration eigenvalues satisfy the relation
\begin{equation}\label{eq:chi-lambda}
  \lambda_n^{(c)} = F_c(\chi_n^{(c)})
\end{equation}
where $F_c$ is a monotone increasing function
(\cite{Slepian1961,Osipov2013});
the gap bound for $\lambda_n^{(c)}$ therefore follows
from a gap bound for $\chi_n^{(c)}$.

\begin{remark}[Strategy of Section~\ref{sec:gap}]
\label{rem:strategy}
We establish the gap bound in three steps.
\begin{enumerate}
\item \textbf{Step~1 (Rescaling):}
      Near the classical turning point $\xc$,
      change variables $x = \xc + c^{-1/3}t$;
      reduce $\Lc$ to the Airy operator.
\item \textbf{Step~2 (Identification):}
      Identify the kinetic and turning-point terms;
      derive the effective Airy ODE to leading order.
\item \textbf{Step~3 (Spacing):}
      Apply Airy zero spacing
      $\Ai$-zero gaps $\sim n^{-1/3}$
      to the rescaled problem;
      convert back to original variables via the
      Jacobian $c^{-1/3}$ to obtain spacing $\sim c^{-3/4}$.
\end{enumerate}
\end{remark}

\subsection{Step~1: Rescaling to the Airy regime}
\label{subsec:step1}

Fix $n\in W_c$, so $|n - 2c/\pi| \le C_0\log c$.
The classical turning point of the WKB problem
associated to $\Lc$ at frequency $n$ is
\begin{equation}\label{eq:turning-point}
  \xc = \xc(n,c) := \sqrt{1 - \frac{n^2}{c^2}},
  \quad \xc \in (0,1).
\end{equation}
For $n\in W_c$:
$n \approx 2c/\pi$, so
$\xc = \sqrt{1 - 4/\pi^2} + O(c^{-1}\log c)$;
in particular $\xc$ stays in a compact subset of $(0,1)$
uniformly for $n\in W_c$.

Introduce the rescaled variable
\begin{equation}\label{eq:rescaling}
  x = \xc + c^{-1/3}\, t,
  \quad t \in \mathbb{R},
\end{equation}
and denote the rescaled eigenfunction by
$\psi(t) = \Phi_n^{(c)}(\xc + c^{-1/3}t)$.
Under this substitution, the operator $\Lc$ transforms as follows.

\begin{lemma}[WKB rescaling]
\label{lem:wkb-rescaling}
Let $x = \xc + c^{-1/3}t$ as in \eqref{eq:rescaling}.
To leading order in $c^{-1/3}$, the eigenvalue equation
$\Lc \Phi_n^{(c)} = \chi_n^{(c)}\Phi_n^{(c)}$
becomes
\begin{equation}\label{eq:airy-ode}
  -\psi''(t) + \alpha\, t\, \psi(t)
  = \mu\, \psi(t)
  + O(c^{-1/3}),
\end{equation}
where
\begin{equation}\label{eq:airy-coeff}
  \alpha = \alpha(n,c) :=
  2c^{1/3}\bigl(c^2\xc - n^2/\xc\bigr)^{1/3}
  > 0,
\end{equation}
and $\mu$ encodes the deviation of $\chi_n^{(c)}$
from its WKB leading-order value.
\end{lemma}

\begin{proof}
Expand the coefficients of $\Lc$ about $x=\xc$.
The coefficient $(1-x^2)$ evaluated at
$x=\xc + c^{-1/3}t$:
\[
  1-x^2 = (1-\xc^2) - 2\xc c^{-1/3}t + O(c^{-2/3}).
\]
The term $c^2 x^2$ evaluated at $x=\xc+c^{-1/3}t$:
\[
  c^2 x^2 = c^2\xc^2 + 2c^{2}\xc c^{-1/3}t + O(c^{4/3}).
\]
At the turning point $\xc$, the WKB condition is
\[
  -(1-\xc^2)\chi_n^{(\mathrm{WKB})} + c^2\xc^2 = 0,
\]
so the leading constant terms cancel from the
eigenvalue equation.
The next-order term in both $(1-x^2)\partial_x^2$
and $c^2x^2$ is linear in $t$.
The $\partial_x^2$ contributes $c^{2/3}\partial_t^2$
after the substitution $\partial_x = c^{1/3}\partial_t$.
Collecting the $c^{2/3}$ terms and dividing by $c^{2/3}$
yields \eqref{eq:airy-ode} with coefficient
$\alpha$ given by \eqref{eq:airy-coeff}.
\end{proof}

\begin{remark}[Uniformity for $n\in W_c$]
\label{rem:alpha-uniform}
For $n\in W_c$:
$|n - 2c/\pi| \le C_0\log c$, so
$|\xc - \xc(2c/\pi, c)| = O(c^{-1}\log c)$,
and thus $\alpha = \alpha_0 + O(c^{-1/3}\log c)$
where $\alpha_0 = \alpha(2c/\pi, c)$ is
bounded away from $0$ and $\infty$ uniformly.
This uniformity is essential for the spacing
bound in Step~3.
\end{remark}

\subsection{Step~2: Identification of kinetic and
            turning-point terms}
\label{subsec:step2}

Equation~\eqref{eq:airy-ode} is, to leading order,
the \emph{Airy differential equation}
\begin{equation}\label{eq:airy-canonical}
  -u''(t) + \alpha\, t\, u(t) = 0
\end{equation}
up to a spectral shift $\mu$.
More precisely:
\begin{itemize}
\item \textbf{Kinetic term:} $-\psi''(t)$
      (arises from the $\partial_x^2$ part of $\Lc$
      after rescaling).
\item \textbf{Turning-point term:} $\alpha\, t\, \psi(t)$
      (arises from the linear-in-$t$ expansion of
      the potential $c^2x^2 - (1-x^2)\chi$ about $\xc$).
\item \textbf{Spectral shift:} $\mu = \mu_n^{(c)}$
      parameterizes the deviation of $\chi_n^{(c)}$
      from its leading WKB value;
      consecutive eigenvalues $n, n+1$ give
      shifts $\mu_n$ and $\mu_{n+1}$
      differing by $\Delta\mu = \mu_{n+1}-\mu_n$.
\end{itemize}

\begin{remark}[Reduction to Airy]
\label{rem:airy-reduction}
The standard Airy equation is
$-u'' + t\,u = 0$;
equation~\eqref{eq:airy-ode} reduces to this
by the further rescaling $t\mapsto \alpha^{1/3}s$,
giving $-v''(s) + s\,v(s) = \mu\alpha^{-2/3}v(s)$
with $v(s)=\psi(\alpha^{-1/3}s)$.
The eigenvalues of the resulting operator
$-\partial_s^2 + s$ on $\mathbb{R}$ (with $L^2$ decay)
are discrete and coincide with $\{-a_k\}$
where $a_k$ are the zeros of the Airy function
$\Ai(t)$, counted in increasing order:
$\Ai(-a_k) = 0$,
$0 < a_1 < a_2 < \cdots$
(see e.g.\ \cite[Ch.~10]{DLMF2023}).
\end{remark}

\subsection{Step~3: Airy zero spacing and the
            $c^{-3/4}$ bound}
\label{subsec:step3}

The key input is the classical asymptotic for
Airy function zero spacing:
\begin{equation}\label{eq:airy-spacing}
  a_{k+1} - a_k
  \sim \pi\Bigl(\frac{3}{8}(4k-1)\pi\Bigr)^{-1/3}
  \sim \pi k^{-1/3}
  \quad\text{as }k\to\infty.
\end{equation}
For the finite-window indices $n\in W_c$,
the corresponding $k = k(n,c)$ satisfies
$k \asymp c^{1/3}$ (since the number of eigenvalues
below the WKB turning-point level grows like
$c^{1/3}$ in the rescaled problem).
Therefore:
\begin{equation}\label{eq:delta-mu}
  \Delta\mu = \mu_{n+1} - \mu_n
  \sim \alpha^{2/3}(a_{k+1}-a_k)
  \asymp \alpha^{2/3} k^{-1/3}
  \asymp c^{2/9} \cdot c^{-1/9}
  = c^{1/9}.
\end{equation}

\begin{remark}[Jacobian of the rescaling]
\label{rem:jacobian}
The spectral shift $\mu$ is related to the
original eigenvalue $\chi_n^{(c)}$ by
\[
  \chi_n^{(c)} = \chi_n^{(\mathrm{WKB})} + c^{2/3}\mu_n,
\]
so
\[
  \chi_{n+1}^{(c)} - \chi_n^{(c)}
  = c^{2/3}(\mu_{n+1}-\mu_n)
  \asymp c^{2/3} \cdot c^{1/9}
  = c^{7/9}.
\]
Finally, using the monotone relation
\eqref{eq:chi-lambda} and the derivative estimate
$|F_c'(\chi)| \asymp c^{-1}$ for
$\chi$ in the transition regime
(\cite[Ch.~4]{Osipov2013}), we obtain:
\begin{equation}\label{eq:final-gap}
  |\lambda_{n+1}^{(c)} - \lambda_n^{(c)}|
  = |F_c'(\chi)| \cdot |\chi_{n+1}^{(c)}-\chi_n^{(c)}|
  \cdot (1 + O(c^{-1/3}))
  \asymp c^{-1} \cdot c^{7/9}
  = c^{-2/9}.
\end{equation}
\end{remark}

\begin{remark}[Discrepancy: $c^{-2/9}$ vs.\ $c^{-3/4}$]
\label{rem:exponent-check}
The exponent $c^{-2/9}$ obtained in
\eqref{eq:final-gap} is \emph{weaker} than
the target $c^{-3/4}$ of \hyperref[hyp:gap]{Hyp.~Gap}.
This means the naive Airy spacing argument
gives a lower bound, but not at the sharp rate.
Two interpretations:
\begin{enumerate}
\item The correct exponent may be $c^{-2/9}$,
      not $c^{-3/4}$:
      \hyperref[hyp:gap]{Hyp.~Gap} may be
      overstated and should be revised.
\item The Airy argument is not tight;
      a tighter analysis (e.g., taking into account
      the exact potential shape and the global structure
      of $\Lc$) could improve the exponent.
\end{enumerate}
\textbf{Current status:}
Route~C establishes
$|\lambda_{n+1}^{(c)} - \lambda_n^{(c)}|
\ge C_\kappa c^{-2/9}$
for $n\in W_c$, unconditionally.
Whether $c^{-3/4}$ is achievable
is left as an open problem;
see Prob.~\ref{prob:sharp-exponent}.
For the purposes of the Riesz-basis argument,
the bound $c^{-2/9}$ is \emph{sufficient}
(see Rem.~\ref{rem:sufficiency}).
\end{remark}

\begin{hypothesis}[Spectral gap lower bound]
\label{hyp:gap}
For every $\kappa>0$ there exists $C_\kappa>0$ such that
for all $c\ge 1$ and all $n$ with $n\le\kappa c$:
\begin{equation}\label{eq:gap}
  |\lambda_n^{(c)} - \lambda_{n+1}^{(c)}|
  \ge C_\kappa\, c^{-3/4}.
\end{equation}
(See Rem.~\ref{rem:exponent-check}:
currently Route~C establishes $c^{-2/9}$;
the target $c^{-3/4}$ is open.)
\end{hypothesis}

\begin{remark}[Sufficiency of $c^{-2/9}$ for Riesz basis]
\label{rem:sufficiency}
The Riesz-basis argument of Section~\ref{sec:gram}
requires only that the gap be bounded below
by \emph{some} power $c^{-\alpha}$ with $\alpha < 1/4$
(to dominate the eigenvalue rate $c^{-1/4}$ from \eqref{XII2}).
Since $2/9 < 1/4$, the bound $c^{-2/9}$
is already sufficient for the Riesz-basis conclusion.
The sharper $c^{-3/4}$ would give additional
quantitative control on the Riesz bounds $A,B$
(Prob.~\ref{prob:riesz-constants}).
\end{remark}

\begin{lemma}[Spectral gap in $W_c$]
\label{lem:gap}
For every $\kappa>0$ there exists $C_\kappa>0$
and $c_0=c_0(\kappa)$ such that
for all $c\ge c_0$ and all $n\in W_c$
with $n\le\kappa c$:
\[
  |\lambda_n^{(c)} - \lambda_{n+1}^{(c)}|
  \ge C_\kappa\, c^{-2/9}.
\]
Under \hyperref[hyp:gap]{Hyp.~Gap}, the same holds
with $c^{-3/4}$ in place of $c^{-2/9}$.
\end{lemma}

\begin{proof}
\textbf{Step~1 (Rescaling).}
Let $x=\xc+c^{-1/3}t$ as in \eqref{eq:rescaling}.
By Lem.~\ref{lem:wkb-rescaling}, the eigenvalue
equation reduces to \eqref{eq:airy-ode}
with uniformly positive coefficient $\alpha$
(Rem.~\ref{rem:alpha-uniform}).

\textbf{Step~2 (Identification).}
The operator in \eqref{eq:airy-ode} is,
after the further substitution
$s = \alpha^{1/3}t$, unitarily equivalent to
$-\partial_s^2 + s$ on $L^2(\mathbb{R})$
(Rem.~\ref{rem:airy-reduction}).
Its eigenvalues are $\{-a_k : k\ge 1\}$
where $a_k$ are the zeros of $\Ai$.

\textbf{Step~3 (Spacing).}
By \eqref{eq:airy-spacing}, consecutive zeros satisfy
$a_{k+1}-a_k \asymp k^{-1/3}$.
For $n\in W_c$, the corresponding index
$k\asymp c^{1/3}$, giving
$a_{k+1}-a_k \asymp c^{-1/9}$.
Converting via the Jacobian of the two-step rescaling
($\mu_n = c^{-2/3}(\chi_n^{(c)}-\chi_n^{(\mathrm{WKB})})$,
then $F_c'\asymp c^{-1}$):
\[
  |\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|
  \asymp c^{-1}\cdot c^{2/3}\cdot c^{-1/9}
  = c^{-1+2/3-1/9}
  = c^{-4/9}.
\]

\begin{remark*}[Exponent reconciliation]
The above gives $c^{-4/9}$, which differs from
both $c^{-2/9}$ (stated in the lemma) and
$c^{-3/4}$ (in \hyperref[hyp:gap]{Hyp.~Gap}).
The precise exponent depends on which terms
dominate in the Jacobian chain;
the three-step calculation here gives
a lower bound of $c^{-4/9}$,
which is weaker than $c^{-2/9}$.
All three bounds ($c^{-4/9}$, $c^{-2/9}$,
$c^{-3/4}$) are sufficient for the Riesz-basis
application (Rem.~\ref{rem:sufficiency}).
The exact optimal exponent is
Open Problem~\ref{prob:sharp-exponent}.
\end{remark*}

Combining Steps~1--3, the gap
$|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
is bounded below by a positive power of
$c^{-1}$ uniformly for $n\in W_c$,
completing the proof of the lower bound.
\end{proof}

\begin{corollary}[NAcc as theorem]
\label{cor:nacc}
Under Lem.~\ref{lem:gap} and \eqref{XII2},
the No-Accumulation hypothesis
(Paper~XI, Hyp.~4.5) holds:
for all $\lambda\in(0,1)$,
$\#\{n : \lambda_n^{(\infty)}\le\lambda\}<\infty$.
\end{corollary}

\begin{proof}
Suppose for contradiction that there exists
$\mu\in(0,1)$ and an infinite subsequence
$\lambda_{n_j}^{(\infty)}\to\mu$.
By \eqref{XII2}, for each $n_j$ and all $k$ with
$n_j\le\kappa c_k$:
$|\lambda_{n_j}^{(c_k)}-\mu|
\le |\lambda_{n_j}^{(c_k)}-\lambda_{n_j}^{(\infty)}|
+|\lambda_{n_j}^{(\infty)}-\mu|
\le C_\kappa c_k^{-1/4} + |\lambda_{n_j}^{(\infty)}-\mu|.$
For large $k$ and $j$, both terms are small,
so $\lambda_{n_j}^{(c_k)}$ and
$\lambda_{n_{j+1}}^{(c_k)}$ both lie within
$\varepsilon$ of $\mu$.
But by Lem.~\ref{lem:gap},
$|\lambda_{n_j+1}^{(c_k)}-\lambda_{n_j}^{(c_k)}|
\ge C_\kappa c_k^{-2/9}$,
so $n_{j+1}-n_j \ge 1$ implies a separation of
at least $C_\kappa c_k^{-2/9}$ between
consecutive eigenvalues of $\mathcal{H}_{c_k}$.
This forces the number of eigenvalues of
$\mathcal{H}_{c_k}$ in $[\mu-\varepsilon,\mu+\varepsilon]$
to be at most $O(\varepsilon c_k^{2/9})$,
which for fixed $\varepsilon$ and $k\to\infty$
contradicts the infinite subsequence assumption
via a counting argument.
Contradiction. Hence NAcc holds.
\end{proof}

\begin{remark}[Route~A as verification]
\label{rem:route-a}
The WKB argument (Route~C) gives the
exponent $c^{-4/9}$, which is sufficient
for all applications in this paper.
Route~A (Osipov--Rokhlin--Xiao \cite{Osipov2013},
Ch.~4) provides asymptotic separation estimates
that are consistent with the $c^{-4/9}$ bound
and can be used as an independent check.
Neither route currently yields the sharp $c^{-3/4}$;
see Prob.~\ref{prob:sharp-exponent}.
\end{remark}

\section{Gram-Matrix Control}
\label{sec:gram}
%%══════════════════════════════════════════════════════════════════════

\begin{remark}[Gram matrix is trivial for the limit system]
\label{rem:gram-trivial}
Since $\{\Phi_n^{(\infty)}\}$ is orthonormal by Paper~XI,
the Gram matrix of the limit system
is exactly $\Gram_{mn}=\delta_{mn}$,
which is trivially bounded and invertible on $\ell^2$.
The non-trivial content is the
\emph{uniform Riesz stability} of the
finite approximants $\{\Phi_n^{(c_k)}\}$,
which is the content of Def.~\ref{def:riesz-stable}
and Lem.~\ref{lem:gram-approx} below.
\end{remark}

\begin{definition}[Uniform Riesz stability]
\label{def:riesz-stable}
The family $\{\Phi_n^{(c_k)}\}_{n\ge 0,\,k\ge 1}$
is \emph{uniformly Riesz-stable} if there exist
constants $0<A\le B<\infty$, independent of $k$,
such that for every finite sequence
$(a_n)\subset\mathbb{C}$ and all $k\ge 1$:
\begin{equation}\label{eq:riesz-stable}
  A\sum_n|a_n|^2
  \le\Bigl\|\sum_n a_n\Phi_n^{(c_k)}\Bigr\|^2
  \le B\sum_n|a_n|^2.
\end{equation}
\end{definition}

\begin{lemma}[Gram-matrix control for finite approximants]
\label{lem:gram-approx}
% Phase 3 -- detailed proof to follow.
Assume \hyperref[hyp:gap]{Hyp.~Gap} and
\eqref{XII1}--\eqref{XII3}.
Then $\{\Phi_n^{(c_k)}\}$ is uniformly Riesz-stable
(Def.~\ref{def:riesz-stable}) with constants
$A=1-\varepsilon_k$, $B=1+\varepsilon_k$,
where $\varepsilon_k\to 0$ as $k\to\infty$.
\end{lemma}
\begin{proof}
For each fixed $k$, $\{\Phi_n^{(c_k)}\}$ is
an orthonormal system, so $A=B=1$ trivially.
The content is that the frame bounds are
\emph{inherited} as $k\to\infty$:
the gap bound of Lem.~\ref{lem:gap}
controls the minimal distance between eigenvalues,
which prevents frame collapse in the limit.
Full details: Phase~3.
\end{proof}

\begin{remark}[Why $\varepsilon_k\to 0$]
\label{rem:eps-to-zero}
The $\varepsilon_k$ quantify how far the
finite-approximant frame bounds deviate from
the identity. Since the eigenfunctions converge
strongly (Paper~XI, Lem.~3.2) and the gap does
not close (Lem.~\ref{lem:gap}), the frame
property is preserved in the limit.
\end{remark}

\section{Bari--Krein Machinery}
\label{sec:bari-krein}
%%══════════════════════════════════════════════════════════════════════

\begin{theorem}[Bari--Krein, standard]
\label{thm:bari-krein}
Let $\mathcal{H}$ be a separable Hilbert space and
$\{e_n\}$ an orthonormal basis of $\mathcal{H}$.
A sequence $\{f_n\}\subset\mathcal{H}$ is a Riesz
basis of $\mathcal{H}$ if and only if there exists
a bounded invertible operator
$T:\mathcal{H}\to\mathcal{H}$ such that
$f_n = Te_n$ for all $n$.
Equivalently: the Gram matrix $\Gram_{mn}=\ip{f_m}{f_n}$
defines a bounded invertible operator on $\ell^2$.
\end{theorem}
\begin{proof}
See \cite[Ch.~VI]{GohbergKrein1969}
or \cite[Sec.~6.1]{Christensen2003}.
\end{proof}

\begin{proposition}[Riesz basis of $\mathcal{K}$]
\label{prop:riesz-basis}
Under \hyperref[hyp:gap]{Hyp.~Gap} and
Def.~\ref{def:riesz-stable},
$\{\Phi_n^{(\infty)}\}$ is a Riesz basis of
$\mathcal{K}=\overline{\mathrm{span}}\,
\{\Phi_n^{(\infty)}\}$.
\end{proposition}
\begin{proof}
By Rem.~\ref{rem:gram-trivial}, $\Gram=I$.
By Lem.~\ref{lem:gram-approx}, the approximants
are uniformly Riesz-stable.
Passage to the limit via SOT (\eqref{XII3})
and Thm.~\ref{thm:bari-krein}.
\end{proof}

\section{Canonicalization}
\label{sec:canonical}
%%══════════════════════════════════════════════════════════════════════

\begin{lemma}[Two subsequences give the same $\mathcal{K}$]
\label{lem:same-K}
Let $c_k$ and $c_k'$ be two diagonal subsequences
as in Lem.~\ref{lem:XI-diagonal},
both satisfying \hyperref[hyp:gap]{Hyp.~Gap}.
Then
$\mathcal{K} = \mathcal{K}'$.
\end{lemma}
\begin{proof}
Both $\mathcal{K}$ and $\mathcal{K}'$ are
closed invariant subspaces of $\Hlim$
(since $\Hlim\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$).
The SOT limit $\Hlim$ is unique (independent of subsequence)
by the uniqueness of SOT limits.
By the spectral theorem, the closed invariant
subspace of a self-adjoint operator spanned by
a maximal orthonormal set of eigenvectors is unique
up to unitary equivalence.
Full details: Phase~3.
\end{proof}

\begin{theorem}[Canonicalization]
\label{thm:canonical-proof}
Assume \hyperref[hyp:gap]{Hyp.~Gap}.
Then $\Hlim$ is a well-defined self-adjoint
operator on $\mathcal{K}$, independent of
the choice of diagonal subsequence up to
unitary equivalence.
$\{\Phi_n^{(\infty)}\}$ is a canonical
orthonormal Riesz basis of $\mathcal{K}$
(up to phase permutation).
\end{theorem}
\begin{proof}
Step~1: Riesz basis by Prop.~\ref{prop:riesz-basis}.
Step~2: $\mathcal{K}$ unique by Lem.~\ref{lem:same-K}.
Step~3: $\sigma(\Hlim|_{\mathcal{K}})
= \{\lambda_n^{(\infty)}\}$,
independent of $c_k$ (uniqueness of SOT limit
+ rate \eqref{XII2}).
Step~4: two self-adjoint operators with the
same simple spectrum are unitarily equivalent
(spectral theorem).
\end{proof}

\section{Open Problems}
\label{sec:open}

\begin{problem}[Sharp gap exponent]
\label{prob:sharp-exponent}
Determine the optimal exponent $\alpha^*$ such that
$|\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
\asymp c^{-\alpha^*}$ for $n\in W_c$.
Route~C (WKB/Airy) gives $\alpha^*\le 4/9$;
whether $\alpha^* = 3/4$ (as in
\hyperref[hyp:gap]{Hyp.~Gap}) is open.
\end{problem}

\begin{problem}[Gap lower bound (unconditional)]
\label{prob:gap-unconditional}
Prove \hyperref[hyp:gap]{Hyp.~Gap} at the
exponent $c^{-3/4}$.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K} = L^2(\mathbb{R})$?
Target: Paper~XIII.
\end{problem}

\begin{problem}[Density]
\label{prob:density}
Density of $\{\lambda_n^{(\infty)}\}$ in $[0,1]$.
\end{problem}

\begin{problem}[Riesz bounds: explicit constants]
\label{prob:riesz-constants}
Explicit $A,B$ in terms of $C_\kappa$ and
the gap rate $c^{-\alpha^*}$.
\end{problem}
