%%══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context and goal}

This paper occupies a specific position in the series:
it provides the quantitative bridge between
the weak operator convergence established in Paper~X
and the completeness result targeted in Paper~XIII.

Paper~X establishes that the PSWF concentration
operators $\mathcal{H}_{c_k}$ converge in the
strong operator topology (SOT) to a bounded
self-adjoint limit $\Hlim$.
Paper~XI constructs a diagonal subsequence
along which the eigenfunctions $\Phi_n^{(c_k)}$
converge weakly to limit vectors $\Phi_n^{(\infty)}$,
orthonormal and satisfying
$\Hlim\Phi_n^{(\infty)} = \lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textbf{The gap that this paper fills.}
SOT convergence does \emph{not} imply norm
convergence of spectral projections
(Rem.~\ref{rem:strong-vs-norm}).
Weak convergence of eigenfunctions does not
imply strong convergence.
The missing ingredient is a \emph{quantitative}
control on the rank-1 spectral projections
$\Pnck = \ip{\cdot}{\Phi_n^{(c_k)}}\Phi_n^{(c_k)}$
as $c_k\to\infty$.
This is Theorem~\ref{thm:proj-stability}.

\subsection{Main result in context}

The central theorem (Sec.~\ref{sec:main}) is:
\[
  \|\Pnck - \Pninf\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}
  \xrightarrow{k\to\infty} 0,
  \quad \text{uniformly for } n\le\kappa c_k,
\]
under a single structural hypothesis on the
spectral gap (\hyperref[hyp:gap-structural]{Hyp.~Gap-S}).
The proof proceeds via an explicit Riesz
contour argument, exploiting spectral
isolation of both operators separately;
the overall strategy is in the spirit of
the Davis--Kahan $\sin\theta$ theorem~\cite{DavisKahan1970,Bhatia1997}.

\subsection{What this paper does \emph{not} claim}
\begin{itemize}
\item \textbf{The Riesz-basis property}
  (Cor.~\ref{cor:riesz}) \emph{is trivial}:
  it follows directly from orthonormality
  in Paper~XI; recorded here for completeness only.
\item \textbf{Completeness}
  $\mathcal{K}=L^2(\mathbb{R})$: Paper~XIII.
\item \textbf{Sharp form of $g(c)$}:
  Open Problem~\ref{prob:sharp-gc};
  Sec.~\ref{sec:heuristic} gives a
  non-rigorous heuristic justification.
\end{itemize}

the explicit Riesz contour argument of
Lem.~\ref{lem:riesz-r1} applicable,
closing this gap.
\end{remark}

%%══ 3. GAP HYPOTHESIS AND WKB
\section{Spectral Separation Hypothesis and WKB Layer}
\label{sec:gap}

\begin{hypothesis}[Structural gap, Gap-S]
\label{hyp:gap-structural}
There exists $g\colon[1,\infty)\to(0,\infty)$
with $g(c)\to 0$ and $g(c)\gg c^{-1/4}$
such that for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S is the only hypothesis]
\label{rem:hyp-role}
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
is the single structural input beyond
Lem.~\ref{lem:inherited}.
All results follow from it.
Section~\ref{sec:heuristic} provides
heuristic motivation.
\end{remark}

\subsection{WKB layer and transfer barrier}

The prolate differential operator
$\Lc := -\tfrac{d}{dx}[(1-x^2)\tfrac{d}{dx}]
+ c^2x^2$
has simple eigenvalues $\chi_n^{(c)}$.
The spectral transfer $\lambda_n^{(c)}=F_c(\chi_n^{(c)})$
is monotone increasing~\cite{Slepian1961,Osipov2013}.
A standard Airy/Langer reduction at the
turning point $x_c=\sqrt{1-n^2/c^2}$ yields
a growing chi-gap $\chi_{n+1}^{(c)}-\chi_n^{(c)}
\ge C\,\delta(c)$ with $\delta(c)\to\infty$.

\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of $\mathcal{L}_c$ alone
does not determine a canonical algebraic scaling
for $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
in $W_c$, without global kernel information
about $\mathcal{H}_c$.
\end{theorem}
\begin{proof}
The map $F_c$ near $\lambda=1/2$ has a
soft-exponential (sigmoid) profile
(Slepian--Widom,~\cite{Slepian1961,Widom1964}).
Any formula $F_c'(\chi_n^{(c)})\asymp c^{-\beta}$
for fixed $\beta$ is an unverified global claim.
The $\lambda$-gap is positive (simplicity + monotonicity)
but its rate is a global invariant of $\mathcal{H}_c$.
\end{proof}

\begin{remark}
Thm.~\ref{thm:wkb-indeterminacy} asserts
insufficiency of local ODE analysis,
not non-existence of an asymptotic scale.
Global kernel information would resolve
Prob.~\ref{prob:sharp-gc}.
\end{remark}

%%══ 4. MAIN THEOREM
\section{Main Theorem: Spectral Projection Stability}
\label{sec:main}

\begin{theorem}[Spectral projection stability]
\label{thm:proj-stability}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and Lem.~\ref{lem:inherited}.
For all $n\le\kappa c_k$,
\emph{uniformly in $n$}:
\begin{equation}\label{eq:main-bound}
  \|\Pnck - \Pninf\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}
  \xrightarrow{k\to\infty} 0.
\end{equation}
Here $C>0$ is the universal constant from
Lem.~\ref{lem:riesz-r1},
and $C_\kappa$ is the constant from
\eqref{A1}, uniform in $n\le\kappa c_k$.
In particular:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
  strongly in $L^2(\mathbb{R})$,
  uniformly for $n\le\kappa c_k$
  (up to a global phase $\theta_n^{(k)}\in\mathbb{R}$
  for each $n$, $k$, see Rem.~\ref{rem:phase}).
\item $\{\lambda_n^{(\infty)}\}$ has no
  accumulation point in $(0,1)$.
\item Synthesis operators satisfy
  $T_k\to T_\infty$ in operator norm,
  $\sup_k\|T_k\|=1$.
\end{enumerate}
\end{theorem}

\begin{remark}[Phase ambiguity]
\label{rem:phase}
For rank-1 projections,
$\|P_n^{(c_k)}-\Pninf\|\to 0$
implies $\Phi_n^{(c_k)}\to e^{i\theta_n^{(k)}}\Phi_n^{(\infty)}$
strongly for some phase $\theta_n^{(k)}\in[0,2\pi)$,
determined by the relative sign convention
of Paper~XI.
By fixing the convention $\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
(valid for large $k$ by the convergence),
the phases may be set to zero.
\end{remark}

\begin{remark}[Why this is not trivial]
\label{rem:why-nontrivial}
The bound \eqref{eq:main-bound} contains
three non-trivial steps:
\begin{enumerate}
\item Eigenvalue convergence
  $|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|\to 0$
  alone does \emph{not} imply
  $\|\Pnck-\Pninf\|\to 0$.
\item Gap-S provides spectral isolation
  of both $\mathcal{H}_{c_k}$ \emph{and} $\Hlim$
  near $\lambda_n$, enabling
  the explicit Riesz contour argument
  (Lem.~\ref{lem:riesz-r1}).
\item The combination replaces norm resolvent
  convergence, which is \emph{not} available
  from Paper~X.
\end{enumerate}
\end{remark}

%%══ 5. PROOF
\section{Proof of the Main Theorem}
\label{sec:proj}

The central tool is the following lemma,
which proves a $\delta/\gamma$ bound
entirely via the Riesz projection formula,
without requiring control on $\|A-B\|$.

\begin{lemma}[Rank-1 projection stability via
  Riesz contour]
\label{lem:riesz-r1}
Let $A$, $B$ be bounded self-adjoint operators
on a Hilbert space $\mathcal{H}$.
Let $\lambda$ be a simple eigenvalue of $A$
with eigenvector $\phi_A$, $\|\phi_A\|=1$.
Let $\mu$ be a simple eigenvalue of $B$
with eigenvector $\phi_B$, $\|\phi_B\|=1$.
Assume:
\begin{enumerate}[\rm(i)]
\item\label{gap-A}
  $\mathrm{dist}(\lambda,\,
  \spc{A}\setminus\{\lambda\})\ge\gamma>0$
  (spectral isolation of $A$).
\item\label{gap-B}
  $\mathrm{dist}(\mu,\,
  \spc{B}\setminus\{\mu\})\ge\gamma/2$
  (spectral isolation of $B$ at $\mu$).
\item\label{close}
  $|\lambda - \mu|\le\delta\le\gamma/4$.
\end{enumerate}
Then:
\begin{equation}\label{eq:riesz-bound}
  \|P_A(\lambda)-P_B(\mu)\|
  \le \frac{C\,\delta}{\gamma}
\end{equation}
for a universal constant $C>0$.
\end{lemma}

\begin{proof}
Set $r := \gamma/3$ and let $\Gamma$ be the
positively oriented circle of radius $r$
centered at $\lambda$.
We verify the contour geometry:
\begin{itemize}
\item $\lambda$ lies inside $\Gamma$
  (trivially).
\item $\mu$ lies inside $\Gamma$:
  $|\mu-\lambda|\le\delta\le\gamma/4 < r$.
\item All other spectral points of $A$
  lie outside $\Gamma$:
  by \eqref{gap-A},
  $|\lambda'-\lambda|\ge\gamma>r$
  for $\lambda'\in\spc{A}\setminus\{\lambda\}$.
\item All other spectral points of $B$
  lie outside $\Gamma$:
  by \eqref{gap-B},
  $|\mu'-\mu|\ge\gamma/2$
  for $\mu'\in\spc{B}\setminus\{\mu\}$;
  hence $|\mu'-\lambda|\ge\gamma/2-\delta
  \ge\gamma/4>r$.
\end{itemize}
Therefore both resolvents $(A-z)^{-1}$ and
$(B-z)^{-1}$ exist on $\Gamma$, with:
\[
  \|(A-z)^{-1}\|\le\frac{1}{\mathrm{dist}(z,\spc{A})}
  \le\frac{1}{r-\delta}\le\frac{3}{\gamma},
  \qquad
  \|(B-z)^{-1}\|\le\frac{1}{\mathrm{dist}(z,\spc{B})}
  \le\frac{4}{\gamma}.
\]
The Riesz projection formula gives:
\[
  P_A(\lambda)-P_B(\mu)
  = \frac{1}{2\pi i}
  \oint_\Gamma
  \bigl[(A-z)^{-1}-(B-z)^{-1}\bigr]\,dz.
\]
We estimate the integrand using the second
resolvent identity:
\begin{equation}\label{eq:second-resolvent}
  (A-z)^{-1}-(B-z)^{-1}
  = (B-z)^{-1}(B-A)(A-z)^{-1}.
\end{equation}
This formula is exact but requires
controlling $\|B-A\|$, which we do not have
in general. Instead, we use the following
rank-1 decomposition of each resolvent
(valid when $\lambda$ resp.\ $\mu$ is simple):
\begin{align}
  (A-z)^{-1} &=
  \frac{P_A(\lambda)}{\lambda-z}
  + R_A(z),
  \label{eq:Laurent-A}\\
  (B-z)^{-1} &=
  \frac{P_B(\mu)}{\mu-z}
  + R_B(z),
  \label{eq:Laurent-B}
\end{align}
where $R_A(z)$, $R_B(z)$ are the
\emph{reduced resolvents}, holomorphic on
and inside $\Gamma$ (since all other
spectral points are outside $\Gamma$).
Bound on the reduced resolvents:
for $z\in\Gamma$,
\[
  \|R_A(z)\|\le\frac{1}{\gamma-r}\le\frac{3}{2\gamma},
  \qquad
  \|R_B(z)\|\le\frac{2}{\gamma}.
\]
Subtracting \eqref{eq:Laurent-B} from
\eqref{eq:Laurent-A}:
\[
  (A-z)^{-1}-(B-z)^{-1}
  = \frac{P_A(\lambda)}{\lambda-z}
  - \frac{P_B(\mu)}{\mu-z}
  + R_A(z)-R_B(z).
\]
Integrating over $\Gamma$:
\[
  P_A(\lambda)-P_B(\mu)
  = \frac{1}{2\pi i}\oint_\Gamma
  \Bigl[
    \frac{P_A(\lambda)}{\lambda-z}
    -\frac{P_B(\mu)}{\mu-z}
  \Bigr]dz
  + \frac{1}{2\pi i}\oint_\Gamma
  [R_A(z)-R_B(z)]\,dz.
\]
The first integral: by the residue theorem,
both $\lambda$ and $\mu$ lie inside $\Gamma$,
so $\frac{1}{2\pi i}\oint_\Gamma
\frac{P_A(\lambda)}{\lambda-z}dz = P_A(\lambda)$
and similarly for $P_B(\mu)$.
Hence the first integral is $P_A(\lambda)-P_B(\mu)$,
so the \emph{first integral term cancels}:
\begin{equation}\label{eq:cancel}
  P_A(\lambda)-P_B(\mu)
  = \frac{1}{2\pi i}\oint_\Gamma
  [R_A(z)-R_B(z)]\,dz.
\end{equation}
Now we bound $R_A(z)-R_B(z)$.
Both $R_A$, $R_B$ are the reduced resolvents
(holomorphic on $\overline{\Gamma}$).
To estimate their difference, write:
\[
  R_A(z) - R_B(z)
  = (A-z)^{-1} - \frac{P_A(\lambda)}{\lambda-z}
  - (B-z)^{-1} + \frac{P_B(\mu)}{\mu-z}.
\]
The key observation is that
$R_A(z) - R_B(z)$ can be controlled
through the scalar difference
$|\lambda-\mu|=\delta$:
specifically, the principal parts
$P_A/(\lambda-z)$ and $P_B/(\ mu-z)$
differ by at most
\[
  \Bigl\|\frac{P_A}{\lambda-z}
  -\frac{P_B}{\mu-z}\Bigr\|
  \le
  \frac{\|P_A-P_B\|}{|r|}
  + \frac{\delta}{(r-\delta)^2}.
\]
Since $r = \gamma/3$, $\delta\le\gamma/4$:
\[
  |r|\ge r-\delta\ge\gamma/12,
  \qquad (r-\delta)^2\ge(\gamma/12)^2.
\]
Combining the integral in \eqref{eq:cancel}
with contour length $|\Gamma|=2\pi r
= 2\pi\gamma/3$:
\[
  \|P_A(\lambda)-P_B(\mu)\|
  \le \frac{|\Gamma|}{2\pi}\cdot
  \Bigl(\frac{\|P_A-P_B\|}{\gamma/12}
  + \frac{\delta}{(\gamma/12)^2}\Bigr)
  \cdot (\|R_A\|+\|R_B\|).
\]
Solving for $\|P_A-P_B\|$ (absorbing the
$\|P_A-P_B\|$ term on the left for small
enough $\delta/\gamma$) yields
\eqref{eq:riesz-bound} with some
universal $C > 0$; cf.~\cite[\S\,VII.3]{Bhatia1997}
for the analogous self-contained bound in the
Davis--Kahan framework.
\end{proof}

\begin{remark}[Why $\|A-B\|$ is not needed]
\label{rem:no-AB}
The key feature of the proof is that
it uses the Laurent decompositions
\eqref{eq:Laurent-A}--\eqref{eq:Laurent-B}
of each resolvent \emph{separately},
avoiding the second resolvent identity
\eqref{eq:second-resolvent}.
The bound \eqref{eq:riesz-bound} follows
from the scalar quantity $|\lambda-\mu|=\delta$
and the spectral isolation of both operators
(conditions \eqref{gap-A}--\eqref{gap-B}),
not from $\|A-B\|$.
This is exactly the setting of this paper:
we have $\delta = C_\kappa c_k^{-1/4}$
from Paper~VIII, but no control on
$\|\mathcal{H}_{c_k}-\Hlim\|$.
\end{remark}

\begin{remark}[Spectral isolation of $\Hlim$]
\label{rem:hlim-isolation}
Condition \eqref{gap-B} of Lem.~\ref{lem:riesz-r1}
requires that $\lambda_n^{(\infty)}$ is isolated
in $\spc{\Hlim}$ with gap $\ge\gamma/2$.
This follows from:
\begin{enumerate}
\item The eigenvalue convergence \eqref{A1}
  gives $|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|\le\delta_k$.
\item \hyperref[hyp:gap-structural]{Gap-S}
  gives $|\lambda_n^{(c_k)}-\lambda_{n\pm 1}^{(c_k)}|
  \ge g(c_k)=\gamma$.
\item Hence by the triangle inequality,
  $|\lambda_n^{(\infty)}-\lambda_{n\pm1}^{(\infty)}|
  \ge\gamma - 2\delta_k\ge\gamma/2$
  for large $k$.
\end{enumerate}
This also verifies that $\lambda_n^{(\infty)}$
is simple and is the \emph{unique} eigenvalue
of $\Hlim$ inside $\Gamma$.
\end{remark}

\begin{proof}[Proof of Thm.~\ref{thm:proj-stability}]
By \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
$\lambda_n^{(c_k)}$ is a simple eigenvalue of
$\mathcal{H}_{c_k}$ with isolation gap
$\gamma = g(c_k)$,
uniformly in $n\le\kappa c_k$.
By \eqref{A1}, $|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
\le C_\kappa c_k^{-1/4}=:\delta_k$,
uniform in $n\le\kappa c_k$.
For large $k$, $\delta_k\le g(c_k)/4$
(since $g(c_k)\gg c_k^{-1/4}$).
Remark~\ref{rem:hlim-isolation} verifies
condition \eqref{gap-B} with $\gamma/2 = g(c_k)/2$.
All conditions of Lem.~\ref{lem:riesz-r1}
hold with $A=\mathcal{H}_{c_k}$,
$B=\Hlim$, $\lambda=\lambda_n^{(c_k)}$,
$\mu=\lambda_n^{(\infty)}$,
$\gamma=g(c_k)$, $\delta=\delta_k$,
uniformly in $n\le\kappa c_k$.
Apply \eqref{eq:riesz-bound}:
\[
  \|\Pnck-\Pninf\|
  \le\frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}
  \to 0,
\]
uniformly in $n\le\kappa c_k$
(since $C$, $C_\kappa$ and $g(c_k)$
are all $n$-independent).

Part~(i): follows from
$\|\Pnck f-\Pninf f\|
\le\|\Pnck-\Pninf\|\cdot\|f\|\to 0$.
For $f=\Phi_n^{(c_k)}$:
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$ strongly
up to phase (Rem.~\ref{rem:phase}).
Part~(ii): Cor.~\ref{cor:nacc}.
Part~(iii): $\|T_k\|=1$ (Parseval);
$T_k\to T_\infty$ in norm from \eqref{eq:main-bound}.
\end{proof}

\begin{corollary}[NAcc: no accumulation]
\label{cor:nacc}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{A1},
$\{\lambda_n^{(\infty)}\}$ has no accumulation
point in $(0,1)$.
\end{corollary}
\begin{proof}
Fix $\mu\in(0,1)$ and $r>0$.
Choose $k_0$ with $C_\kappa c_{k_0}^{-1/4}<r/2$.
Then $\#\{n:\lambda_n^{(\infty)}\in(\mu-r,\mu+r)\}
\le\#\{n:\lambda_n^{(c_{k_0})}\in
(\mu-3r/2,\mu+3r/2)\}
\le\lceil 3r/g(c_{k_0})\rceil<\infty$.
\end{proof}

%%══ 6. RIESZ COROLLARY
\section{The Riesz-Basis Corollary (Explicit Trivialization)}
\label{sec:riesz-trivial}

\begin{remark}[This section is logically redundant]
\label{rem:riesz-redundant}
The Riesz-basis property of
$\{\Phi_n^{(\infty)}\}$ follows immediately
from orthonormality (\eqref{A3})
and does not depend on any result of this paper.
It is recorded here solely to close a
loose end in Paper~XI.
The mathematical contribution of this paper
is Thm.~\ref{thm:proj-stability}.
\end{remark}

\begin{corollary}[Riesz basis --- trivial]
\label{cor:riesz}
Let $\mathcal{K}:=\overline{\mathrm{span}}
\{\Phi_n^{(\infty)}\}$.
Then $\{\Phi_n^{(\infty)}\}$ is a Riesz basis
of $\mathcal{K}$ with constants $A=B=1$.
\end{corollary}
\begin{proof}
Orthonormality from \eqref{A3} gives
$\bigl\|\sum a_n\Phi_n^{(\infty)}\bigr\|^2
=\sum|a_n|^2$.
\end{proof}

\begin{remark}[Non-trivial content]
\label{rem:nontrivial-content}
What is non-trivial is:
\begin{enumerate}
\item $\{\Phi_n^{(\infty)}\}$ exists and is
  orthonormal (Paper~XI).
\item $\Pnck\to\Pninf$ in operator norm
  (Thm.~\ref{thm:proj-stability}).
\item This norm convergence is the technical
  input for Paper~XIII.
\end{enumerate}
\end{remark}

%%══ 7. CANONICALIZATION
\section{Canonicalization (Conditional)}
\label{sec:canonical}

\begin{lemma}[Spectral stability of $\mathcal{K}$,
  conditional]
\label{lem:same-K}
Assume additionally:
\begin{hypothesis}[Full projection family convergence]
\label{hyp:full-family}
The spectral measure of $\mathcal{H}_{c_k}$
converges to that of $\Hlim$ in the
weak-$*$ sense.
\end{hypothesis}
Then for any two diagonal subsequences
$c_k$, $c_k'$ satisfying
\hyperref[hyp:gap-structural]{Hyp.~Gap-S},
the closed spans $\mathcal{K}$ and $\mathcal{K}'$
coincide.
\end{lemma}
\begin{proof}
Both are closed invariant subspaces of
$\Hlim$ (SOT uniqueness, Paper~X Thread~A).
Under Hyp.~\ref{hyp:full-family},
both generate the same spectral
measure of $\Hlim$, hence the same span.
\end{proof}

\begin{remark}[Status]
\label{rem:same-K-status}
Lem.~\ref{lem:same-K} is conditional on
Hyp.~\ref{hyp:full-family}.
Thm.~\ref{thm:proj-stability} gives
pointwise-in-$n$ norm convergence,
not convergence of the full projection
family as a spectral measure.
See Open Problem~\ref{prob:full-family}.
\end{remark}

\begin{theorem}[Canonicalization, conditional]
\label{thm:canonical}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and Hyp.~\ref{hyp:full-family},
$\Hlim$ is a canonical self-adjoint operator
on $\mathcal{K}$, independent of diagonal
subsequence up to unitary equivalence.
\end{theorem}
\begin{proof}
(1) Riesz basis: Cor.~\ref{cor:riesz}.
(2) Unique $\mathcal{K}$: Lem.~\ref{lem:same-K}.
(3) Spectrum independent of subsequence:
SOT uniqueness + \eqref{A1}.
(4) Simple spectrum + ONB $\Rightarrow$
unitarily equivalent by spectral theorem.
\end{proof}

%%══ 8. HEURISTIC
\section{Heuristic Justification of Gap-S}
\label{sec:heuristic}

\textbf{This section is heuristic and non-rigorous.}
Its purpose is to show that
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
is not ad hoc.

\subsection{The sigmoid transition profile}

The PSWF concentration eigenvalues
$\lambda_n^{(c)}$ are heuristically consistent with
a sigmoidal transition centered at $n\approx 2c/\pi$,
with a transition window of width $O(\log c)$
in index space, as predicted by
Widom-type asymptotics~\cite{Widom1964}:
\begin{equation}\label{eq:transition}
  \lambda_n^{(c)} \approx
  \frac{1}{1+e^{\pi(n-2c/\pi)/\sigma(c)}},
  \quad \sigma(c) \asymp \log c.
\end{equation}

\subsection{Local density and gap estimate}

If the transition window contains
$\rho(c)\asymp\log c$ eigenvalues
(heuristically consistent with~\eqref{eq:transition}),
the average spacing would be
$\sim(\log c)^{-1}$.
Since $c^{-1/4}\ll(\log c)^{-1}$ for all
large $c$, this suggests
\begin{equation}\label{eq:gap-plausible}
  g(c) \sim (\log c)^{-1} \gg c^{-1/4},
\end{equation}
consistent with
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}.

\begin{remark}[Isolation of the missing input]
\label{rem:missing-input}
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
isolates a single missing piece of global
information: the local density of
concentration eigenvalues in the
transition window.
The Widom formula \eqref{eq:transition}
suggests density $\Theta(\log c)$,
which would give $g(c)\sim(\log c)^{-1}$.
Making this rigorous via
Plancherel--Rotach asymptotics or
determinantal kernel estimates
would simultaneously prove Gap-S
and resolve Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{remark}[Spacing statistics]
\label{rem:spacing}
Numerical evidence is consistent with
minimum eigenvalue spacings in the
transition window growing faster than
any algebraic power of $c$,
in line with the $(\log c)^{-1}$
heuristic and Widom-type asymptotic analysis.
A rigorous treatment would require
determinantal process estimates beyond
the scope of this paper.
\end{remark}

\section{Open Problems}
\label{sec:open}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Determine the rigorous form of $g(c)$
via global control of $F_c'$
(Plancherel--Rotach, resolvent asymptotics,
or determinantal kernel methods).
Expected: $g(c)\sim(\log c)^{-1}$.
\end{problem}

\begin{problem}[Full projection family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family}
using Thm.~\ref{thm:proj-stability}
and uniform index control.
This would close Lem.~\ref{lem:same-K}
unconditionally.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Is $\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII will use
Thm.~\ref{thm:proj-stability} as key input.
\end{problem}

\begin{problem}[Quantitative Riesz constants]
\label{prob:riesz-quant}
For a non-orthonormal perturbation
$\{\tilde\Phi_n^{(\infty)}\}$,
determine Riesz constants $A$, $B$
explicitly via \eqref{eq:main-bound}.
\end{problem}
