\title[Spectral Projection Stability for Strongly Convergent Operators]{%
  Spectral Projection Stability for Strongly Convergent Operators:\\
  A Reduction via Spectral Gap Control}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We identify the precise conditions under which strong operator convergence
$A_k \xrightarrow{\mathrm{SOT}} A$ upgrades to norm convergence of
rank-one spectral projections.
The key mechanism ---
a rank-one specialisation of the Davis--Kahan $\sin\Theta$ theorem
under mere SOT convergence ---
requires only that the operator residual on a limit eigenvector
decay faster than the local spectral gap of the approximating operator;
no operator-norm bound is needed.
We apply this to the prolate spheroidal wave function (PSWF) concentration
operators $\mathcal{H}_{c_k} \xrightarrow{\mathrm{SOT}} \Hlim$,
reducing the projection stability problem to two explicit asymptotic
hypotheses (Gap-S and SOT-Faster), both identified as structurally
necessary conditions of this type of argument.
This paper does not prove these hypotheses;
it isolates them as the precise missing mechanism in the chain
$\mathrm{SOT} \Rightarrow \text{projection stability}
\Rightarrow \text{completeness}$.
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Context and goal}

This paper occupies a specific position in the series:
it \emph{identifies the precise missing mechanism}
and reduces the problem of spectral projection stability
to two explicit asymptotic hypotheses.

\begin{center}\itshape
SOT convergence produces approximate eigenvectors\\
but provides no control on the residual relative to the spectral gap.\\
This is the exact obstruction this paper identifies and reduces to
explicit hypotheses.
\end{center}

Paper~X establishes that the PSWF concentration
operators $\mathcal{H}_{c_k}$ converge in the
strong operator topology (SOT) to a bounded
self-adjoint limit $\Hlim$.
Paper~XI constructs a diagonal subsequence
along which the eigenfunctions $\Phi_n^{(c_k)}$
converge weakly to limit vectors $\Phi_n^{(\infty)}$,
orthonormal and satisfying
$\Hlim\Phi_n^{(\infty)} = \lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textbf{The gap that this paper fills.}
SOT convergence does \emph{not} imply norm
convergence of spectral projections
(Rem.~\ref{rem:strong-vs-norm}).
SOT produces approximate eigenvectors
--- with $\|(\mathcal{H}_{c_k}-\lambda_n^{(\infty)})
\Phi_n^{(\infty)}\|\to 0$ ---
but no control on this residual
\emph{relative to the spectral gap}.
Without such scale control,
spectral isolation cannot be invoked,
and projection convergence cannot be concluded.
This paper identifies the precise conditions
to close this gap, reducing the problem to
two explicit hypotheses.

The central results are:
\begin{itemize}
\item \textbf{Thm.~\ref{thm:abstract}
  (Abstract Stability Principle):}
  operator-theoretic implication with minimal
  assumptions; isolates an explicit ratio
  condition for rank-one projection stability
  under SOT convergence.
\item \textbf{Thm.~\ref{thm:convergence}
  (PSWF application):}
  under Gap-S and SOT-Faster,
  $\|P_n^{(c_k)} - P_n^{(\infty)}\|\to 0$.
\item \textbf{Thm.~\ref{thm:rate}
  (Rate):} under SOT-Rate,
  $\|P_n^{(c_k)}-P_n^{(\infty)}\|
  \le C C_\kappa c_k^{-1/4}/g(c_k)$.
\end{itemize}

\begin{remark}[Scope: reduction, not resolution]
\label{rem:scope}
This paper does \emph{not} prove projection
stability unconditionally.
It identifies the precise missing mechanism
and reduces the problem to two hypotheses
(Gap-S, SOT-Faster), both currently unproved.
Theorem~\ref{thm:abstract} is fully rigorous
as an operator-theoretic statement;
all PSWF-specific content is encoded in
Hypotheses~\ref{hyp:gap-structural}
and~\ref{hyp:sot-faster}.
\end{remark}

%%══ 3. HYPOTHESES
\section{Hypotheses and Motivation}
\label{sec:gap}

\begin{hypothesis}[Effective asymptotic
  gap in transition window, Gap-S]
\label{hyp:gap-structural}
Let $W_c\subset\mathbb{N}$ denote the
\emph{transition window}: indices $n$
for which $\lambda_n^{(c)}\in(\delta,1-\delta)$
for some fixed $\delta>0$.
There exists $g\colon[1,\infty)\to(0,\infty)$
with
\begin{equation}\label{eq:gap-dominates}
  g(c)\to 0
  \quad\text{and}\quad
  \lim_{c\to\infty}\frac{c^{-1/4}}{g(c)} = 0,
\end{equation}
such that for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S is structurally
  necessary]
\label{rem:gap-necessary}
Gap-S is \emph{not} a plausible
simplifying assumption;
it is a \textbf{structurally necessary condition}.
Any argument of the type
``SOT $+$ gap control $\Rightarrow$
projection stability''
must control the minimal spectral gap
relative to the eigenvalue convergence rate.
Without an explicit lower bound of the
form \eqref{eq:gap-s} with
$c^{-1/4}=o(g(c))$, the ratio
$\tilde\varepsilon_n(k)/g(c_k)$
in \eqref{eq:ratio-to-zero} cannot be
shown to vanish, and the entire argument
collapses.
Identifying this as the key missing input
is one of the main contributions of this paper.

\textbf{Density vs.\ gap.}
Gap-S is \emph{not} implied by
Widom-type asymptotics~\cite{Widom1964}:
Widom's theorem controls eigenvalue
\emph{density} in the transition window
(average spacing $\sim(\log c)^{-1}$),
not the \emph{minimal gap}.
Average spacing $\ne$ minimum gap;
a uniform lower bound requires
a separate, stronger argument
(e.g., via determinantal repulsion bounds;
see Open Problem~\ref{prob:sharp-gc}).
\end{remark}

\begin{remark}[Gap-S implies $c^{-1/4}=o(g(c))$]
\label{rem:gap-implies-o}
Condition \eqref{eq:gap-dominates} is
equivalent to $c^{-1/4}=o(g(c))$
as $c\to\infty$, used explicitly in
the proof of Thm.~\ref{thm:convergence}.
\end{remark}

\begin{hypothesis}[Gap-dominated residual
  condition, SOT-Faster]
\label{hyp:sot-faster}
For each fixed $n$, along the diagonal
subsequence of Lem.~\ref{lem:inherited}:
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k))
  \quad\text{as }k\to\infty.
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster as localized
  norm information]
\label{rem:sot-faster-localized}
SOT-Faster is strictly stronger than
pointwise convergence ($\varepsilon_n(k)\to 0$,
which SOT supplies).
It encodes \textbf{partial norm-convergence
information localised to the $n$-th
spectral scale}: the residual must be
asymptotically negligible relative to
the local gap $g(c_k)$.
Equivalently, it is a one-sided local
norm-resolvent bound:
SOT-Faster would follow from
$\|(\mathcal{H}_{c_k}-z)^{-1}
-(\Hlim-z)^{-1}\|\cdot g(c_k)\to 0$
uniformly for $z$ at distance
$\asymp g(c_k)$ from $\lambda_n^{(\infty)}$.
This reformulation makes precise why
SOT-Faster is \emph{not} a technical
side condition but the central
analytic difficulty of the problem.
\end{remark}

\begin{remark}[SOT-Faster is the
  core difficulty]
\label{rem:sot-faster-core}
Proving SOT-Faster rigorously is
Open Problem~\ref{prob:sot-faster}.
Section~\ref{sec:heuristic} presents
evidence consistent with it in the
PSWF setting, but no rigorous proof.
\end{remark}

\begin{hypothesis}[SOT rate on
  eigenvectors, SOT-Rate]
\label{hyp:sot-rate}
For each fixed $n$:
\begin{equation}\label{eq:sot-rate}
  \varepsilon_n(k) = O(c_k^{-1/4})
  \quad\text{as }k\to\infty.
\end{equation}
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
\[
  \text{SOT-Rate}
  \underset{\eqref{eq:gap-dominates}}{\Longrightarrow}
  \text{SOT-Faster}
  \Longrightarrow
  \|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0.
\]
SOT alone gives only $\varepsilon_n(k)\to 0$,
insufficient without comparison to $g(c_k)$.
If $g(c)\sim(\log c)^{-1}$,
Thm.~\ref{thm:rate} gives rate
$O(c_k^{-1/4}\log c_k)$.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding $\Phi_n^{(\infty)}$ in the
eigenbasis $\{\Phi_m^{(c_k)}\}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
\end{equation}
The diagonal term ($m=n$) contributes
$\le C_\kappa^2 c_k^{-1/2}$.
If off-diagonal overlaps are small
(consistent with Gap-S),
the sum is dominated by $m=n$,
predicting $\varepsilon_n(k)=O(c_k^{-1/4})$.
This is formally consistent with SOT-Rate;
Open Problem~\ref{prob:off-diagonal}
would make it rigorous.
\end{remark}

\subsection{WKB layer}
\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
does not determine a canonical scaling
for $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
without global kernel information.
\end{theorem}
\begin{proof}
The transfer map $F_c$ has a sigmoid
profile near $\lambda=1/2$~\cite{Widom1964};
any algebraic rate $F_c'\asymp c^{-\beta}$
is a global claim not derivable from
$\Lc$ alone.
\end{proof}

%%══ 4. ABSTRACT STABILITY PRINCIPLE
\section{Abstract Stability Principle}
\label{sec:abstract}

The following theorem is purely
operator-theoretic.
It is the main general contribution
of this paper.

\begin{theorem}[Abstract Stability Principle]
\label{thm:abstract}
Let $\mathcal{H}$ be a Hilbert space.
Let $\{A_k\}$, $A$ be bounded self-adjoint
operators on $\mathcal{H}$ with
$A_k\xrightarrow{\mathrm{SOT}} A$.
Let $\phi\in\mathcal{H}$, $\|\phi\|=1$,
with $A\phi = \lambda\phi$, and assume
$\lambda$ is isolated in $\sigma(A)$.
Let $P^\infty$ be the rank-one spectral
projection of $A$ onto $\ker(A-\lambda)$.

Assume there exists a sequence of
simple eigenvalues $\lambda_k\in\sigma(A_k)$
such that:
\begin{equation}\label{eq:abstract-matching}
  \lambda_k\to\lambda
  \quad\text{as }k\to\infty.
\end{equation}
Let $\gamma_k:=\mathrm{dist}(\lambda_k,
\sigma(A_k)\setminus\{\lambda_k\})>0$
and let $P_k$ be the
corresponding rank-one spectral projection.

Suppose:
\begin{equation}\label{eq:abstract-ratio}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0
  \quad\text{as }k\to\infty.
\end{equation}
Then:
\[
  \|P_k - P^\infty\|\to 0
  \quad\text{as }k\to\infty.
\]
\end{theorem}

\begin{proof}
Set $\varepsilon_k:=\|(A_k-A)\phi\|$
and $\delta_k:=|\lambda_k-\lambda|$.
Triangle inequality:
$\|(A_k-\lambda_k)\phi\|
\le \varepsilon_k + \delta_k$.
Since $\lambda_k$ is simple with gap $\gamma_k$,
Lem.~\ref{lem:approx-eig} gives
$\|(I-P_k)\phi\|
\le(\varepsilon_k+\delta_k)/\gamma_k\to 0$.
Writing $P_k\phi=\alpha_k v_k$ with unit
eigenvector $v_k$ and
$|\alpha_k|^2=1-\|(I-P_k)\phi\|^2\to 1$,
the rank-one identity (cf.~\cite[Ch.~1]{Bhatia1997})
gives $\|P_k-P^\infty\|\to 0$.
\end{proof}

\begin{remark}[Novelty of Thm.~\ref{thm:abstract}]
\label{rem:abstract-novelty}
The Davis--Kahan theorem~\cite{DavisKahan1970}
and Kato's perturbation theory~\cite{Kato1966}
consider eigenspace stability under
\emph{norm} perturbations.
Theorem~\ref{thm:abstract} differs in a
precise and useful way:
it replaces the norm-perturbation
hypothesis by the explicit
\emph{ratio condition}
\eqref{eq:abstract-ratio},
which requires only SOT convergence
(yielding $\|(A_k-A)\phi\|\to 0$)
and a spectral gap lower bound on the
approximating operators.
The combination ---
rank-one projection stability under
mere SOT convergence, controlled by
a gap-relative residual ratio ---
does not appear explicitly in the
standard references and is the
specific contribution of this theorem.
\end{remark}

\begin{remark}[What Thm.~\ref{thm:abstract}
  really says]
\label{rem:abstract-meaning}
Norm convergence of rank-one spectral
projections does not require operator
norm convergence:
it requires only (i)~eigenvalue
convergence \eqref{eq:abstract-matching}
and (ii)~the ratio condition
\eqref{eq:abstract-ratio},
a local version of norm-resolvent
convergence restricted to a single
isolated eigenvalue.
\end{remark}

%%══ 5. PSWF APPLICATION: MAIN THEOREMS
\section{PSWF Application: Main Theorems}
\label{sec:main}

Theorems~\ref{thm:convergence}
and~\ref{thm:rate} are direct applications
of Thm.~\ref{thm:abstract} with:
$A_k=\mathcal{H}_{c_k}$,
$A=\Hlim$,
$\phi=\Phi_n^{(\infty)}$,
$\lambda=\lambda_n^{(\infty)}$,
$\lambda_k=\lambda_n^{(c_k)}$,
$\gamma_k=g(c_k)$.
Condition~\eqref{eq:abstract-matching}
holds by Lem.~\ref{lem:eigenvalue-matching}.
Condition~\eqref{eq:abstract-ratio}
is verified by Gap-S (for $\gamma_k$)
together with SOT-Faster and
Rem.~\ref{rem:gap-implies-o}
(for the ratio to zero).

\begin{theorem}[Spectral projection
  convergence]
\label{thm:convergence}
Assume Lem.~\ref{lem:inherited},
\hyperref[hyp:gap-structural]{Hyp.~Gap-S},
and \hyperref[hyp:sot-faster]{Hyp.~SOT-Faster}.
For each fixed $n$ with $n\in W_{c_k}$
for all large $k$:
\begin{equation}\label{eq:convergence}
  \|P_n^{(c_k)} - P_n^{(\infty)}\|\to 0
  \quad\text{as }k\to\infty.
\end{equation}
In particular, $\Phi_n^{(c_k)}\to
\Phi_n^{(\infty)}$ strongly (up to phase)
and $\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$
(Rem.~\ref{rem:nacc}).
\end{theorem}

\begin{theorem}[Rate]
\label{thm:rate}
Assume additionally
\hyperref[hyp:sot-rate]{Hyp.~SOT-Rate}.
For all $n\le\kappa c_k$ with
$n\in W_{c_k}$, uniformly in $n$:
\begin{equation}\label{eq:rate-bound}
  \|P_n^{(c_k)} - P_n^{(\infty)}\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)},
\end{equation}
where $C>0$ is universal.
If $g(c)\sim(\log c)^{-1}$,
this gives $O(c_k^{-1/4}\log c_k)$.
\end{theorem}

\begin{remark}[Phase convention]
\label{rem:phase}
$\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$
implies $\Phi_n^{(c_k)}
\to e^{i\theta_n^{(k)}}\Phi_n^{(\infty)}$
strongly;
phase fixed by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
for all large $k$.
\end{remark}

%%══ 6. PROOFS
\section{Proofs}
\label{sec:proj}

\begin{lemma}[Approximate eigenvector
  $\Rightarrow$ projection estimate;
  rank-one specialisation of
  Davis--Kahan $\sin\Theta$;
  cf.\ \cite{DavisKahan1970,Kato1966}]
\label{lem:approx-eig}
Let $A$ be bounded self-adjoint,
$\lambda$ a \emph{simple} eigenvalue
with gap
$\gamma:=\mathrm{dist}(\lambda,
\spc{A}\setminus\{\lambda\})>0$.
Since $\lambda$ is simple,
$P_n^A$ is rank-one and the
reduced resolvent on $\mathrm{ran}(I-P_n^A)$
has norm $\le 1/\gamma$.
For $\phi\in\mathcal{H}$, $\|\phi\|=1$,
with $\|(A-\lambda)\phi\|\le\varepsilon$:
\begin{equation}\label{eq:approx-eig-bound}
  \|(I-P_n^A)\phi\|\le\frac{\varepsilon}{\gamma}.
\end{equation}
\end{lemma}
\begin{proof}
$(A-\lambda)\phi=(A-\lambda)(I-P_n^A)\phi$
since $(A-\lambda)P_n^A=0$.
The resolvent bound gives the result.
\end{proof}

\begin{remark}[Perturbation mechanism]
\label{rem:perturbation-mechanism}
Lemma~\ref{lem:approx-eig} is a rank-one
specialisation of Davis--Kahan
$\sin\Theta$~\cite{DavisKahan1970}.
The gap $\gamma$ is applied to $A=\mathcal{H}_{c_k}$
(not $\Hlim$): Gap-S supplies the gap
of the approximating operator.
SOT-Faster ensures $\varepsilon_n(k)/g(c_k)\to 0$,
activating the mechanism.
\end{remark}

\begin{lemma}[SOT gives approximate
  eigenvectors, no rate]
\label{lem:sot-approx}
Under \eqref{A2}--\eqref{A3}:
$\varepsilon_n(k)
=\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
\to 0$ as $k\to\infty$.
No rate is claimed.
\end{lemma}
\begin{proof}
SOT~\eqref{eq:sot} applied to
$f=\Phi_n^{(\infty)}$.
\end{proof}

\begin{proof}[Proof of
  Thm.~\ref{thm:convergence}]
Fix $n\in W_{c_k}$.
By Lems.~\ref{lem:sot-approx}
and~\ref{lem:eigenvalue-matching}:
\begin{equation}\label{eq:eps-tilde}
  \tilde\varepsilon_n(k)
  :=\|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
  \Phi_n^{(\infty)}\|
  \le \underbrace{\varepsilon_n(k)}_{\text{SOT}}
  + \underbrace{|\lambda_n^{(\infty)}
    -\lambda_n^{(c_k)}|}_{\text{ev.\ shift}}
  \le \varepsilon_n(k) + C_\kappa c_k^{-1/4}.
\end{equation}
Dividing by $g(c_k)$:
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to 0\;\text{(SOT-Faster)}}
  +
  \underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to 0\;\text{(Rem.~\ref{rem:gap-implies-o})}}
  \to 0,
\end{equation}
both terms vanishing independently.
This verifies \eqref{eq:abstract-ratio}
for $\gamma_k=g(c_k)$.
Lem.~\ref{lem:approx-eig} then gives
$\|(I-P_n^{(c_k)})\Phi_n^{(\infty)}\|
\le\tilde\varepsilon_n(k)/g(c_k)\to 0$,
and the rank-one identity completes
the proof (as in Thm.~\ref{thm:abstract}).
\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:rate}]
SOT-Rate gives $\varepsilon_n(k)\le C' c_k^{-1/4}$,
so $\tilde\varepsilon_n(k)
\le(C'+C_\kappa)c_k^{-1/4}$;
with $\gamma_k=g(c_k)$,
Lem.~\ref{lem:approx-eig} and the
rank-one identity yield \eqref{eq:rate-bound}.
\end{proof}

\begin{remark}[No accumulation
  of $\lambda_n^{(\infty)}$]
\label{rem:nacc}
Under Gap-S and \eqref{A1},
$\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$.
\emph{Proof sketch.}
Eigenvalues $\lambda_n^{(c)}$ are strictly
ordered (Slepian--Pollak~\cite{Slepian1961}).
Fix $\mu$, $r>0$; for large $k_0$,
eigenvalue matching bounds
$\lambda_n^{(\infty)}\in(\mu-r,\mu+r)$
to $\lambda_n^{(c_{k_0})}$ in a slightly
larger interval.
By Gap-S, at most
$\lceil 3r/g(c_{k_0})\rceil<\infty$
eigenvalues lie there.
\end{remark}

\section{Canonicalization
  (Optional Extension)}
\label{sec:canonical}

\begin{remark}[Status of this section]
This section is independent of the main
result and may be skipped on a first
reading.
It records a conditional canonicalization
result under an additional strong hypothesis
(Hyp.~\ref{hyp:full-family}),
included for completeness of the series.
\end{remark}

\begin{lemma}[Same $\mathcal{K}$,
  conditional]
\label{lem:same-K}
Assume additionally:
\begin{hypothesis}[Full family convergence]
\label{hyp:full-family}
The spectral measures of $\mathcal{H}_{c_k}$
converge weak-$*$ to those of $\Hlim$.
\end{hypothesis}
Then any two diagonal subsequences
satisfying Gap-S yield the same
closed span $\mathcal{K}$.
\end{lemma}
\begin{proof}
Both subspaces are closed
$\Hlim$-invariant;
under Hyp.~\ref{hyp:full-family}
they generate the same spectral measure.
\end{proof}

\begin{theorem}[Canonicalization,
  conditional]
\label{thm:canonical}
Under Gap-S and Hyp.~\ref{hyp:full-family},
$\Hlim$ is canonical on $\mathcal{K}$
up to unitary equivalence.
\end{theorem}

%%══ 9. HEURISTIC
\section{Heuristic Motivation of Gap-S
  and SOT-Faster}
\label{sec:heuristic}

\textbf{Non-rigorous throughout.}

\subsection{Sigmoid transition (Widom)}

Widom-type asymptotics~\cite{Widom1964}
predict a sigmoidal eigenvalue profile:
$\lambda_n^{(c)}\approx
(1+e^{\pi(n-2c/\pi)/\sigma(c)})^{-1}$
with $\sigma(c)\asymp\log c$.
This predicts average spacing
$\sim(\log c)^{-1}$ in $W_c$,
consistent with Gap-S.

\subsection{Gap estimate and the
  density--gap distinction}

Widom controls \emph{density},
not \emph{minimum gap};
Gap-S requires the latter.
If $\rho(c)\asymp\log c$ eigenvalues
occupy $W_c$, average spacing
$\sim(\log c)^{-1}\gg c^{-1/4}$
supports Gap-S at the level of
typical behavior.
A rigorous deterministic lower bound
requires determinantal kernel methods
(Open Problem~\ref{prob:sharp-gc}).

\subsection{SOT-Faster plausibility}

From \eqref{eq:formal-expansion},
small off-diagonal overlaps
(consistent with Gap-S)
predict $\varepsilon_n(k)=O(c_k^{-1/4})$,
consistent with SOT-Rate.
Making this rigorous requires
Open Problem~\ref{prob:off-diagonal}.

\begin{remark}[Heuristic status]
Section~\ref{sec:heuristic} provides
evidence consistent with Gap-S and
SOT-Faster; it does not constitute
a proof of either.
\end{remark}

%%══ 10. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\medskip
\noindent\textbf{Conjecture-level:}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove $|\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
\ge g(c)$ for $n\in W_c$,
$g(c)\sim(\log c)^{-1}$;
would establish Gap-S as a theorem.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$;
would make Thm.~\ref{thm:convergence}
fully unconditional.
Equivalently: prove local resolvent
stability at the $n$-th eigenvalue
scale (Rem.~\ref{rem:sot-faster-localized}).
\end{problem}

\medskip
\noindent\textbf{Technical-level:}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
Prove $\varepsilon_n(k)=O(c_k^{-1/4})$;
implies SOT-Faster and justifies
Thm.~\ref{thm:rate}.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
Estimate
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
for $m\ne n$;
route to Prob.~\ref{prob:sot-rate}.
\end{problem}

\begin{problem}[Full family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family};
makes Lem.~\ref{lem:same-K}
unconditional.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$? Paper~XIII.
\end{problem}
