\title[Spectral Projection Stability for Strongly Convergent Operators]{%
  Spectral Projection Stability for Strongly Convergent Operators:\\
  A Reduction via Spectral Gap Control}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
We identify the precise conditions under which strong operator convergence
$A_k \xrightarrow{\mathrm{SOT}} A$ upgrades to norm convergence of
rank-one spectral projections.
The key mechanism is a rank-one specialisation of the Davis--Kahan
$\sin\Theta$ theorem: if the operator residual on a limit eigenvector
decays faster than the local spectral gap of the approximating operator,
then the corresponding spectral projections converge in norm.
We apply this to the prolate spheroidal wave function (PSWF) concentration
operators $\mathcal{H}_{c_k} \xrightarrow{\mathrm{SOT}} \Hlim$,
reducing the projection stability problem to two explicit asymptotic
hypotheses (Gap-S and SOT-Faster).
This paper does not prove these hypotheses; it identifies them as the
precise missing mechanism in the chain
$\mathrm{SOT} \Rightarrow \text{projection stability}
\Rightarrow \text{completeness}$.
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Context and goal}

This paper occupies a specific position in the series:
it \emph{identifies the precise missing mechanism}
and reduces the problem of spectral projection stability
to two explicit asymptotic hypotheses.

\begin{center}\itshape
SOT convergence produces approximate eigenvectors\\
but provides no control on the residual relative to the spectral gap.\\
This is the exact obstruction this paper identifies and bridges.
\end{center}

Paper~X establishes that the PSWF concentration
operators $\mathcal{H}_{c_k}$ converge in the
strong operator topology (SOT) to a bounded
self-adjoint limit $\Hlim$.
Paper~XI constructs a diagonal subsequence
along which the eigenfunctions $\Phi_n^{(c_k)}$
converge weakly to limit vectors $\Phi_n^{(\infty)}$,
orthonormal and satisfying
$\Hlim\Phi_n^{(\infty)} = \lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textbf{The gap that this paper fills.}
SOT convergence does \emph{not} imply norm
convergence of spectral projections
(Rem.~\ref{rem:strong-vs-norm}).
The difficulty is precise:
SOT convergence produces approximate eigenvectors
--- the vectors $\Phi_n^{(\infty)}$ satisfy
$\|(\mathcal{H}_{c_k}-\lambda_n^{(\infty)})
\Phi_n^{(\infty)}\|\to 0$ as $k\to\infty$ ---
but provides no control on the residual
\emph{relative to the spectral gap}.
Without such control, spectral isolation
cannot be invoked, and projection convergence
cannot be concluded.
This paper identifies the precise additional
conditions that close this gap, reducing the
problem to two explicit hypotheses.

The central results are:
\begin{itemize}
\item \textbf{Thm.~\ref{thm:abstract}
  (Abstract Stability Principle):}
  an operator-theoretic implication valid for
  any sequence of bounded self-adjoint operators
  satisfying approximate eigenvector $+$
  gap-dominated residual conditions.
\item \textbf{Thm.~\ref{thm:convergence}
  (PSWF application):}
  under Gap-S and SOT-Faster,
  $\|P_n^{(c_k)} - P_n^{(\infty)}\|\to 0$.
\item \textbf{Thm.~\ref{thm:rate}
  (Rate):} under the additional
  \hyperref[hyp:sot-rate]{Hyp.~SOT-Rate},
  $\|P_n^{(c_k)}-P_n^{(\infty)}\|
  \le C C_\kappa c_k^{-1/4}/g(c_k)$.
\end{itemize}

\begin{remark}[Scope: reduction, not resolution]
\label{rem:scope}
This paper does \emph{not} prove projection
stability unconditionally.
It identifies the precise missing mechanism
and reduces the problem to two hypotheses
(Gap-S, SOT-Faster), both currently unproved.
The Abstract Stability Principle
(Thm.~\ref{thm:abstract}) is fully rigorous
as an operator-theoretic statement;
the PSWF content is encoded entirely in
Hypotheses~\ref{hyp:gap-structural}
and~\ref{hyp:sot-faster}.
\end{remark}

%%══ 3. HYPOTHESES
\section{Hypotheses and Motivation}
\label{sec:gap}

\begin{hypothesis}[Effective asymptotic
  gap in transition window, Gap-S]
\label{hyp:gap-structural}
Let $W_c\subset\mathbb{N}$ denote the
\emph{transition window}: indices $n$
for which $\lambda_n^{(c)}\in(\delta,1-\delta)$
for some fixed $\delta>0$.
There exists $g\colon[1,\infty)\to(0,\infty)$
with
\begin{equation}\label{eq:gap-dominates}
  g(c)\to 0
  \quad\text{and}\quad
  \lim_{c\to\infty}\frac{c^{-1/4}}{g(c)} = 0,
\end{equation}
such that for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Scope and interpretation
  of Gap-S]
\label{rem:gap-scope}
Gap-S is a uniform lower bound on the
spectral gap \emph{along the chosen
diagonal subsequence}, not a pointwise
bound for all indices.
The restriction to $W_c$ is essential:
bulk eigenvalues (near $0$ or $1$)
have exponentially small individual gaps
and do not appear in the diagonal
subsequence of Paper~XI.
Within $W_c$, Widom-type asymptotics
\cite{Widom1964} predict average spacing
$\sim(\log c)^{-1}$;
Gap-S with $g(c)=(\log c)^{-1}$
is thus an effective asymptotic condition
consistent with known PSWF spectral theory.
A rigorous deterministic lower bound
remains Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{remark}[Gap-S implies $c^{-1/4}=o(g(c))$]
\label{rem:gap-implies-o}
Condition \eqref{eq:gap-dominates} is
equivalent to $c^{-1/4}=o(g(c))$
as $c\to\infty$, used explicitly in
the proof of Thm.~\ref{thm:convergence}.
\end{remark}

\begin{hypothesis}[Gap-dominated residual
  condition, SOT-Faster]
\label{hyp:sot-faster}
For each fixed $n$, along the diagonal
subsequence of Lem.~\ref{lem:inherited}:
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k))
  \quad\text{as }k\to\infty.
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster as local
  resolvent stability and scale condition]
\label{rem:sot-faster-resolvent}
SOT-Faster is quantitatively stronger
than SOT in a precise sense:
it requires the operator residual on the
fixed limit eigenvector $\Phi_n^{(\infty)}$
to be \emph{asymptotically negligible
relative to the local spectral scale}
of the approximating operator
$\mathcal{H}_{c_k}$.
This is a scale condition, not merely
a smallness condition:
$\varepsilon_n(k)\to 0$ is not sufficient;
the rate must dominate the gap $g(c_k)$.
In spirit, SOT-Faster is a
\emph{local resolvent stability condition}
at the $n$-th eigenvalue:
it would follow from uniform control of
$\|(\mathcal{H}_{c_k}-z)^{-1}
-(\Hlim-z)^{-1}\|$
restricted to the $n$-th spectral scale,
which is norm-resolvent information
localised to a single eigenvalue.
Proving this rigorously is
Open Problem~\ref{prob:sot-faster}.
\end{remark}

\begin{remark}[SOT-Faster is the
  core difficulty]
\label{rem:sot-faster-core}
SOT-Faster encapsulates \emph{the entire
difficulty} of upgrading SOT convergence
to norm convergence of rank-one
spectral projections.
SOT (Paper~X) alone gives
$\varepsilon_n(k)\to 0$,
but no comparison with $g(c_k)$.
\end{remark}

\begin{hypothesis}[SOT rate on
  eigenvectors, SOT-Rate]
\label{hyp:sot-rate}
For each fixed $n$:
\begin{equation}\label{eq:sot-rate}
  \varepsilon_n(k) = O(c_k^{-1/4})
  \quad\text{as }k\to\infty.
\end{equation}
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
\[
  \text{SOT-Rate}
  \underset{\eqref{eq:gap-dominates}}{\Longrightarrow}
  \text{SOT-Faster}
  \Longrightarrow
  \|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0.
\]
SOT alone gives only $\varepsilon_n(k)\to 0$,
insufficient without comparison to $g(c_k)$.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding $\Phi_n^{(\infty)}$ in the
eigenbasis $\{\Phi_m^{(c_k)}\}$
of $\mathcal{H}_{c_k}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
\end{equation}
The diagonal term $m=n$ contributes
$\le C_\kappa^2 c_k^{-1/2}$ by \eqref{A1}.
If off-diagonal overlaps are small
for $m\ne n$ (consistent with Gap-S),
the sum is dominated by $m=n$,
predicting $\varepsilon_n(k)=O(c_k^{-1/4})$.
This is formally consistent with SOT-Rate
but not rigorous evidence;
it requires off-diagonal overlap estimates
(Open Problem~\ref{prob:off-diagonal}).
\end{remark}

\subsection{WKB layer}
\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
does not determine a canonical scaling
for $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
without global kernel information.
\end{theorem}
\begin{proof}
The transfer map $F_c$ has a sigmoid
profile near $\lambda=1/2$~\cite{Widom1964};
any algebraic rate $F_c'\asymp c^{-\beta}$
is a global claim not derivable from
$\Lc$ alone.
\end{proof}

%%══ 4. ABSTRACT STABILITY PRINCIPLE
\section{Abstract Stability Principle}
\label{sec:abstract}

The following theorem is purely
operator-theoretic and requires no
knowledge of PSWFs.
It is stated in full generality to
make explicit that the PSWF application
(Thm.~\ref{thm:convergence}) is an
instance of a universal mechanism.

\begin{theorem}[Abstract Stability Principle]
\label{thm:abstract}
Let $\mathcal{H}$ be a Hilbert space and
let $\{A_k\}$, $A$ be bounded self-adjoint
operators on $\mathcal{H}$ with
$A_k\xrightarrow{\mathrm{SOT}} A$.
Let $\phi\in\mathcal{H}$, $\|\phi\|=1$,
with $A\phi = \lambda\phi$ for some
$\lambda\in\mathbb{R}$, simple in $\sigma(A)$.
Let $P^\infty$ denote the rank-one
spectral projection of $A$ onto $\ker(A-\lambda)$.

Suppose there exist $\gamma_k>0$ such that
$\lambda_k:=\lambda_k(A_k)$ is a simple
eigenvalue of $A_k$ with
\begin{equation}\label{eq:abstract-gap}
  \mathrm{dist}\!\bigl(\lambda_k,\,
  \sigma(A_k)\setminus\{\lambda_k\}\bigr)
  \ge\gamma_k > 0,
\end{equation}
and suppose:
\begin{equation}\label{eq:abstract-sot-faster}
  \frac{\|(A_k - A)\phi\|}{\gamma_k}
  + \frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0
  \quad\text{as }k\to\infty.
\end{equation}
Let $P_k$ denote the rank-one spectral
projection of $A_k$ onto $\ker(A_k-\lambda_k)$.
Then:
\[
  \|P_k - P^\infty\| \to 0
  \quad\text{as }k\to\infty.
\]
\end{theorem}
\begin{proof}
Set $\varepsilon_k:=\|(A_k-A)\phi\|$
and $\delta_k:=|\lambda_k-\lambda|$.
By the triangle inequality:
\[
  \|(A_k-\lambda_k)\phi\|
  \le \varepsilon_k + \delta_k.
\]
Since $\lambda_k$ is a simple eigenvalue
of $A_k$ with spectral gap $\gamma_k$,
Lem.~\ref{lem:approx-eig} gives
$\|(I-P_k)\phi\|\le(\varepsilon_k+\delta_k)/\gamma_k\to 0$.
The rank-one identity then yields
$\|P_k-P^\infty\|\to 0$
(see proof of Thm.~\ref{thm:convergence}
for the algebraic details).
\end{proof}

\begin{remark}[What Thm.~\ref{thm:abstract}
  really says]
\label{rem:abstract-meaning}
Theorem~\ref{thm:abstract} makes precise:
\emph{norm convergence of spectral projections
does not require operator norm convergence ---
only local control of the residual and
eigenvalue shift relative to the spectral gap
of the approximating operator.}
This is a local version of
norm-resolvent convergence restricted to
a single eigenvalue.
\end{remark}

%%══ 5. PSWF APPLICATION: MAIN THEOREMS
\section{PSWF Application: Main Theorems}
\label{sec:main}

\begin{theorem}[Spectral projection
  convergence]
\label{thm:convergence}
Assume Lem.~\ref{lem:inherited},
\hyperref[hyp:gap-structural]{Hyp.~Gap-S},
and \hyperref[hyp:sot-faster]{Hyp.~SOT-Faster}.
For each fixed $n$ with $n\in W_{c_k}$
for all large $k$:
\begin{equation}\label{eq:convergence}
  \|P_n^{(c_k)} - P_n^{(\infty)}\|\to 0
  \quad\text{as }k\to\infty.
\end{equation}
In particular:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
  strongly as $k\to\infty$, up to phase.
\item $\{\lambda_n^{(\infty)}\}$ has no
  accumulation point in $(0,1)$.
\end{enumerate}
\end{theorem}

\begin{theorem}[Rate]
\label{thm:rate}
Assume additionally
\hyperref[hyp:sot-rate]{Hyp.~SOT-Rate}.
For all $n\le\kappa c_k$ with
$n\in W_{c_k}$, uniformly in $n$:
\begin{equation}\label{eq:rate-bound}
  \|P_n^{(c_k)} - P_n^{(\infty)}\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)},
\end{equation}
where $C>0$ is universal.
\end{theorem}

\begin{remark}[Phase convention]
\label{rem:phase}
$\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$
implies $\Phi_n^{(c_k)}
\to e^{i\theta_n^{(k)}}\Phi_n^{(\infty)}$
strongly.
Phase is fixed by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$,
valid for all large $k$.
\end{remark}

%%══ 6. PROOFS
\section{Proofs}
\label{sec:proj}

\begin{lemma}[Approximate eigenvector
  $\Rightarrow$ projection estimate;
  rank-one specialisation of
  Davis--Kahan $\sin\Theta$;
  cf.\ \cite{DavisKahan1970,Kato1966}]
\label{lem:approx-eig}
Let $A$ be a bounded self-adjoint operator
and $\lambda$ a \emph{simple} eigenvalue
of $A$ with spectral gap
$\gamma:=\mathrm{dist}(\lambda,
\spc{A}\setminus\{\lambda\})>0$.
Since $\lambda$ is simple,
$P_n^A$ is rank-one and the
reduced resolvent
$(A-\lambda)^{-1}\big|_{\mathrm{ran}(I-P_n^A)}$
is bounded with norm $\le 1/\gamma$.
Let $\phi\in\mathcal{H}$, $\|\phi\|=1$,
with $\|(A-\lambda)\phi\|\le\varepsilon$.
Then:
\begin{equation}\label{eq:approx-eig-bound}
  \|(I-P_n^A)\phi\|
  \le\frac{\varepsilon}{\gamma}.
\end{equation}
\end{lemma}
\begin{proof}
Write $\phi = P_n^A\phi + (I-P_n^A)\phi$.
Since $(A-\lambda)P_n^A=0$:
$(A-\lambda)\phi = (A-\lambda)(I-P_n^A)\phi$.
On $\mathrm{ran}(I-P_n^A)$,
$\|(A-\lambda)^{-1}\|\le 1/\gamma$
(by simplicity of $\lambda$),
so $\|(I-P_n^A)\phi\|
\le\gamma^{-1}\|(A-\lambda)\phi\|
\le\varepsilon/\gamma$.
\end{proof}

\begin{remark}[Perturbation mechanism]
\label{rem:perturbation-mechanism}
Lemma~\ref{lem:approx-eig} is a rank-one
specialisation of the Davis--Kahan
$\sin\Theta$ theorem~\cite{DavisKahan1970}:
\emph{an approximate eigenvector whose
residual is smaller than the spectral gap
must concentrate on the true eigenspace}.
The gap $\gamma$ is applied to
$A=\mathcal{H}_{c_k}$ (not $\Hlim$):
Gap-S supplies the gap of the
\emph{approximating} operator,
which is the relevant quantity for
spectral isolation.
SOT-Faster then ensures
$\varepsilon_n(k)/g(c_k)\to 0$,
activating the mechanism.
\end{remark}

\begin{lemma}[SOT gives approximate
  eigenvectors, no rate]
\label{lem:sot-approx}
Under \eqref{A2}--\eqref{A3},
for each fixed $n$:
\begin{equation}\label{eq:approx-sot}
  \varepsilon_n(k)
  = \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|\to 0
  \quad\text{as }k\to\infty.
\end{equation}
No rate on $\varepsilon_n(k)$ is claimed.
\end{lemma}
\begin{proof}
Since $\Hlim\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$,
SOT~\eqref{eq:sot} applied to
$f=\Phi_n^{(\infty)}$ gives the result.
\end{proof}

\begin{proof}[Proof of
  Thm.~\ref{thm:convergence}]
Fix $n\in W_{c_k}$ for all large $k$.
By Lem.~\ref{lem:sot-approx} and
Lem.~\ref{lem:eigenvalue-matching}:
\begin{equation}\label{eq:eps-tilde}
  \tilde\varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
  \Phi_n^{(\infty)}\|
  \le \underbrace{\varepsilon_n(k)}_{\text{SOT term}}
  + \underbrace{|\lambda_n^{(\infty)}
    -\lambda_n^{(c_k)}|}_{\text{eigenvalue shift}}
  \le \varepsilon_n(k) + C_\kappa c_k^{-1/4}.
\end{equation}
Dividing by $g(c_k)>0$:
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to\,0
    \;\text{(SOT-Faster)}}
  +
  \underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to\,0
    \;\text{(Rem.~\ref{rem:gap-implies-o})}}
  \to 0
  \quad\text{as }k\to\infty,
\end{equation}
where both terms vanish independently
as $k\to\infty$.
Since $\lambda_n^{(c_k)}$ is a simple
eigenvalue of $\mathcal{H}_{c_k}$
with spectral gap $\ge g(c_k)$
(Gap-S for $n\in W_{c_k}$),
Lem.~\ref{lem:approx-eig} gives:
\begin{equation}\label{eq:proj-dev}
  \|(I-P_n^{(c_k)})\Phi_n^{(\infty)}\|
  \le\frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \to 0
  \quad\text{as }k\to\infty.
\end{equation}
Write $P_n^{(c_k)}\Phi_n^{(\infty)}
=\alpha_k\Phi_n^{(c_k)}$ with
$\alpha_k
=\ip{\Phi_n^{(\infty)}}{\Phi_n^{(c_k)}}$;
then $|\alpha_k|^2
=1-\|(I-P_n^{(c_k)})\Phi_n^{(\infty)}\|^2
\to 1$.
By the rank-one projection identity
$\|P_u-P_v\|\le 2\|u-e^{i\theta}v\|$
(cf.~\cite[Ch.~1]{Bhatia1997}):
\[
  \|P_n^{(c_k)}-P_n^{(\infty)}\|
  \le 2\|\Phi_n^{(c_k)}
  -e^{i\arg\alpha_k}\Phi_n^{(\infty)}\|
  \to 0
  \quad\text{as }k\to\infty.
\]
\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:rate}]
Under SOT-Rate \eqref{eq:sot-rate}:
$\varepsilon_n(k)\le C' c_k^{-1/4}$,
so $\tilde\varepsilon_n(k)
\le(C'+C_\kappa)c_k^{-1/4}$.
With $\gamma=g(c_k)$,
Lem.~\ref{lem:approx-eig} and the
rank-one identity yield \eqref{eq:rate-bound}
uniformly for $n\le\kappa c_k$,
$n\in W_{c_k}$.
\end{proof}

\begin{corollary}[No accumulation
  of $\lambda_n^{(\infty)}$]
\label{cor:nacc}
Under Gap-S and \eqref{A1},
$\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$.
\end{corollary}
\begin{proof}
Since eigenvalues $\lambda_n^{(c)}$ are
strictly ordered in $n$
(Slepian--Pollak~\cite{Slepian1961}),
fix $\mu\in(0,1)$, $r>0$;
choose $k_0$ with
$C_\kappa c_{k_0}^{-1/4}<r/2$.
Then $\lambda_n^{(\infty)}\in(\mu-r,\mu+r)$
implies $\lambda_n^{(c_{k_0})}
\in(\mu-3r/2,\mu+3r/2)$.
By Gap-S on $W_{c_{k_0}}$,
at most $\lceil 3r/g(c_{k_0})\rceil<\infty$
eigenvalues lie in this interval.
\end{proof}

%%══ 8. CANONICALIZATION
\section{Canonicalization (Conditional)}
\label{sec:canonical}

\begin{lemma}[Same $\mathcal{K}$,
  conditional]
\label{lem:same-K}
Assume additionally:
\begin{hypothesis}[Full family convergence]
\label{hyp:full-family}
The spectral measures of $\mathcal{H}_{c_k}$
converge weak-$*$ to those of $\Hlim$.
\end{hypothesis}
Then any two diagonal subsequences
satisfying Gap-S yield the same
closed span $\mathcal{K}$.
\end{lemma}
\begin{proof}
Both subspaces are closed
$\Hlim$-invariant;
under Hyp.~\ref{hyp:full-family}
they generate the same spectral measure.
\end{proof}

\begin{theorem}[Canonicalization,
  conditional]
\label{thm:canonical}
Under Gap-S and Hyp.~\ref{hyp:full-family},
$\Hlim$ is canonical on $\mathcal{K}$
up to unitary equivalence.
\end{theorem}

%%══ 9. HEURISTIC
\section{Heuristic Motivation of Gap-S
  and SOT-Faster}
\label{sec:heuristic}

\textbf{Non-rigorous throughout.}

\subsection{Sigmoid transition (Widom)}

Widom-type asymptotics~\cite{Widom1964}
predict a sigmoidal eigenvalue profile
in the transition window:
\[
  \lambda_n^{(c)}\approx
  \tfrac{1}{1+e^{\pi(n-2c/\pi)/\sigma(c)}},
  \quad\sigma(c)\asymp\log c.
\]
This predicts qualitative behavior
consistent with Gap-S at scale
$g(c)\sim(\log c)^{-1}$.

\subsection{Gap estimate}

If $\rho(c)\asymp\log c$ eigenvalues
occupy $W_c$,
the average spacing is
$\sim(\log c)^{-1}\gg c^{-1/4}$.
This supports Gap-S at the level of
\emph{typical spacing},
but does not imply a uniform lower bound:
average spacing $\ne$ minimum gap.
A rigorous deterministic lower bound
requires Plancherel--Rotach or
determinantal kernel methods
\cite{Widom1964,Osipov2013}.

\subsection{SOT-Faster plausibility}

From \eqref{eq:formal-expansion},
if off-diagonal overlaps
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
are small for $m\ne n$
(consistent with spectral isolation
under Gap-S),
the sum is dominated by $m=n$,
predicting $\varepsilon_n(k)=O(c_k^{-1/4})$.
This is formally consistent with SOT-Rate
but not rigorous evidence
(Open Problem~\ref{prob:off-diagonal}).

\begin{remark}[Heuristic status]
Section~\ref{sec:heuristic} provides
evidence consistent with Gap-S and
SOT-Faster;
it does not constitute a rigorous proof
of either.
\end{remark}

%%══ 10. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent The open problems below are
tiered by perceived depth.

\medskip
\noindent\textbf{Conjecture-level
(require new asymptotic/spectral methods):}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove $|\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
\ge g(c)$ for $n\in W_c$
with $g(c)\sim(\log c)^{-1}$ rigorously;
would establish Gap-S as a theorem.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$
as $k\to\infty$;
would make Thm.~\ref{thm:convergence}
fully unconditional.
Equivalently (Rem.~\ref{rem:sot-faster-resolvent}):
prove local resolvent stability of
$\mathcal{H}_{c_k}$ at the $n$-th
eigenvalue scale.
\end{problem}

\medskip
\noindent\textbf{Technical-level
(more tractable, conditional on above):}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
Prove $\varepsilon_n(k)=O(c_k^{-1/4})$;
stronger than SOT-Faster
(implies it via Rem.~\ref{rem:hyp-hierarchy})
and would justify the rate in
Thm.~\ref{thm:rate}.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
Estimate
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
for $m\ne n$ rigorously;
route to Open Problem~\ref{prob:sot-rate}.
\end{problem}

\begin{problem}[Full family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family};
would make Lem.~\ref{lem:same-K}
unconditional.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII.
\end{problem}
