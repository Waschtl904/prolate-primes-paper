\begin{abstract}
We establish the Riesz-basis criterion for the limiting
orthonormal system $\{\Phi_n^{(\infty)}\}$ constructed in
Paper~XI along the diagonal subsequence $c_k$.
The central equivalence is:
\[
  \{\Phi_n^{(\infty)}\} \text{ is a Riesz basis}
  \;\Longleftrightarrow\;
  \text{Gap} + \text{Stability} + \text{Nondegeneracy}.
\]
The hard core is Section~\ref{sec:gap},
which has two results of independent interest.
First, Proposition~\ref{prop:inner-gap-invariant}
establishes a positive spectral gap at the level
of the differential operator $\mathcal{L}_c$
via WKB/Airy analysis, in invariant form
(no algebraic exponent claimed).
Second, Theorem~\ref{thm:wkb-indeterminacy}
proves that no canonical algebraic scaling
for the gap in the concentration eigenvalues
$\lambda_n^{(c)}$ can be extracted from local
WKB geometry: the spectral transfer
$F_c\colon\chi\mapsto\lambda$ is a global invariant
of the PSWF integral operator, inaccessible to ODE analysis.
The downstream Riesz-basis and canonicalization
results follow from the structural gap
hypothesis (\hyperref[hyp:gap-structural]{Hyp.~Gap-S}),
which asserts a positive separation scale
$g(c)\gg c^{-1/4}$ without specifying its power.
\end{abstract}

\section*{Guide for the Referee}
\begin{itemize}
\item \textbf{Primary result of Section~\ref{sec:gap}}:
      two theorems, not one.
      Prop.~\ref{prop:inner-gap-invariant} gives a
      positive $\chi$-gap (invariant form, no exponent).
      Thm.~\ref{thm:wkb-indeterminacy} proves
      exponent-indeterminacy as a structural theorem.
\item \textbf{Downstream results}
      (Secs.~\ref{sec:gram}--\ref{sec:canonical})
      are conditional on
      \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
      alone; no algebraic exponent is used anywhere.
\item \textbf{NAcc proof} (Cor.~\ref{cor:nacc})
      uses a shrinking-window argument;
      the key is that $g(c_k)\gg c_k^{-1/4}$
      forces the window $\varepsilon(c_k)$ to
      shrink slower than $g(c_k)$.
\item \textbf{Not claimed}:
      sharp exponent for $g(c)$ (Open~Prob.~\ref{prob:sharp-gc});
      completeness in $L^2(\mathbb{R})$ (Paper~XIII).
\end{itemize}

\section{Main Statement}
\label{sec:main}
%%══════════════════════════════════════════════════════════════════

\begin{theorem}[Riesz-basis criterion]
\label{thm:riesz-criterion}
Let $\{\Phi_n^{(\infty)}, \lambda_n^{(\infty)}\}$ be the
orthonormal system of Paper~XI
(Lem.~\ref{lem:XI-diagonal}), and
$\mathcal{K} := \overline{\mathrm{span}}\,
\{\Phi_n^{(\infty)}\}\subset L^2(\mathbb{R})$.
The following are equivalent:
\begin{enumerate}[\rm(i)]
\item $\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$.
\item The Gram matrix
      $\Gram_{mn} := \ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}$
      defines a bounded invertible operator on $\ell^2$.
\item \textbf{(Structural gap, Sec.~\ref{sec:gap}):}
      there exists $g\colon[1,\infty)\to(0,\infty)$
      with $g(c)\to 0$ and $g(c)\gg c^{-1/4}$ such that
      $|\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge g(c)$
      for all $n\in W_c$
      (\hyperref[hyp:gap-structural]{Hyp.~Gap-S}).
\item \textbf{(Stability):}
      $\{\Phi_n^{(c_k)}\}$ is uniformly Riesz-stable
      (Def.~\ref{def:riesz-stable}).
\end{enumerate}
(i)$\Leftrightarrow$(ii): Bari--Krein /
Christensen frame perturbation
(Sec.~\ref{sec:bari-krein}).\\
(ii)$\Leftrightarrow$(iii)+(iv): Gram control
(Sec.~\ref{sec:gram}).\\
(iii): Section~\ref{sec:gap};
sharp form of $g(c)$ is Open Prob.~\ref{prob:sharp-gc}.
\end{theorem}

\begin{remark}[Why $g(c)\gg c^{-1/4}$ suffices]
\label{rem:sufficiency-main}
The Paper~XI convergence rate is
$|\lambda_n^{(c)}-\lambda_n^{(\infty)}|\le C_\kappa c^{-1/4}$.
For the Gram argument (Sec.~\ref{sec:gram})
one needs only that consecutive eigenvalues of
$\mathcal{H}_c$ cannot cluster at rate $c^{-1/4}$.
The condition $g(c)\gg c^{-1/4}$ is necessary and
sufficient for this; the precise power of $c$
is irrelevant downstream.
\end{remark}

\begin{theorem}[Canonicalization]
\label{thm:canonical}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
Then the conclusions (a)--(d) hold:
\begin{enumerate}[\rm(a)]
\item $\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$.
\item Any two diagonal subsequences satisfying
      Lem.~\ref{lem:XI-diagonal}
      yield unitarily equivalent limit systems.
\item The No-Accumulation hypothesis
      (Paper~XI, Hyp.~4.5) holds as a theorem.
\item $\Hlim$ is a canonical self-adjoint operator
      on $\mathcal{K}$, independent of subsequence
      up to unitary equivalence.
\end{enumerate}
\end{theorem}

\section{Input from Paper~XI}

\begin{lemma}[Inherited: diagonal system, Paper~XI]
\label{lem:XI-diagonal}
There exists a fixed subsequence $c_k\nearrow\infty$
such that simultaneously for all $n\ge 0$:
\begin{enumerate}[\rm(i)]
\item
  $\Phi_n^{(c_k)}\rightharpoonup\Phi_n^{(\infty)}$
  weakly, $\|\Phi_n^{(\infty)}\|=1$.
\item
  $|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
  \le C_\kappa c_k^{-1/4}$, $n\le\kappa c_k$.
\item
  $\mathcal{H}_{c_k}\xrightarrow{\mathrm{SOT}}\Hlim$;
  $\Hlim$ bounded, self-adjoint.
\item $\{\Phi_n^{(\infty)}\}$ orthonormal.
\item $\Hlim\Phi_n^{(\infty)}
      =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{enumerate}
\end{lemma}

\begin{enumerate}[(XII.1)]
\item\label{XII1}
  $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}=\delta_{mn}$.
\item\label{XII2}
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}$, $n\le\kappa c$.
\item\label{XII3}
  $\Hlim\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\item\label{XII4}
  $\#W_c = O(\log c)$;
  $\lambda_n^{(c)}\in[\varepsilon_0,1-\varepsilon_0]$
  for all $n\in W_c$.
\end{enumerate}

\section{Spectral Separation in $W_c$}
\label{sec:gap}

\subsection{Setup and two-layer structure}
\label{subsec:setup}

Recall the prolate differential operator
\begin{equation}\label{eq:Lc}
  \Lc := -\frac{d}{dx}\Bigl[(1-x^2)\frac{d}{dx}\Bigr]
           + c^2 x^2, \quad x\in[-1,1],
\end{equation}
with simple real eigenvalues
$\chi_0^{(c)}<\chi_1^{(c)}<\cdots$.
The spectral transfer map
\begin{equation}\label{eq:chi-lambda}
  \lambda_n^{(c)} = F_c(\chi_n^{(c)})
\end{equation}
is monotone increasing (\cite{Slepian1961,Osipov2013}).

\begin{remark}[Two distinct layers]
\label{rem:two-layer}
Section~\ref{sec:gap} separates two problems
that must not be conflated:
\begin{enumerate}
\item \textbf{Inner layer (ODE/WKB):}
      lower bound on $|\chi_{n+1}^{(c)}-\chi_n^{(c)}|$
      for $n\in W_c$.
      Accessible via Airy analysis of $\Lc$.
      Result: Prop.~\ref{prop:inner-gap-invariant}.
\item \textbf{Transfer layer (global):}
      control of $F_c'(\chi)$ for $\chi\in\sigma(\Lc|_{W_c})$.
      Requires global kernel / resolvent information.
      Result: impossibility theorem
      (Thm.~\ref{thm:wkb-indeterminacy}).
\end{enumerate}
All downstream arguments use only the
structural hypothesis
\hyperref[hyp:gap-structural]{Hyp.~Gap-S},
which lives entirely at the $\lambda$-level
and bypasses the transfer problem.
\end{remark}

\subsection{Inner layer: WKB/Airy analysis of $\Lc$}
\label{subsec:wkb}

For $n\in W_c$ define the classical turning point
\begin{equation}\label{eq:turning-point}
  \xc(n,c) := \sqrt{1-n^2/c^2} \in (0,1),
\end{equation}
which remains in a compact subset of $(0,1)$
uniformly for $n\in W_c$.
Introduce the Langer-type rescaling
\begin{equation}\label{eq:rescaling}
  x = \xc + c^{-1/3}\,t, \quad t\in\mathbb{R},
\end{equation}
and write $\psi(t) = \Phi_n^{(c)}(\xc+c^{-1/3}t)$.

\begin{lemma}[Airy reduction]
\label{lem:airy-reduction}
Under \eqref{eq:rescaling}, to leading order in $c^{-1/3}$,
the equation $\Lc\Phi_n^{(c)}=\chi_n^{(c)}\Phi_n^{(c)}$
becomes
\begin{equation}\label{eq:airy-ode}
  -\psi''(t) + \alpha\,t\,\psi(t)
  = \mu_n\,\psi(t) + O(c^{-1/3}),
\end{equation}
where $\alpha=\alpha(n,c)\asymp c^{1/3}$
uniformly for $n\in W_c$,
and $\mu_n\in\mathbb{R}$ is a spectral shift parameter.
\end{lemma}

\begin{proof}
Expand $(1-x^2)$ and $c^2x^2$ about $x=\xc$
using \eqref{eq:rescaling}:
$(1-x^2) = (1-\xc^2) - 2\xc c^{-1/3}t + O(c^{-2/3})$
and $c^2x^2 = c^2\xc^2 + 2c^{5/3}\xc t + O(c^{4/3})$.
At the turning point the zeroth-order terms
combine to zero (by definition of $\xc$);
the $\partial_x^2$ becomes $c^{2/3}\partial_t^2$;
collecting $c^{2/3}$ terms and dividing yields
\eqref{eq:airy-ode}.
\end{proof}

\begin{remark}[Standard Airy form]
\label{rem:airy-standard}
Under the further rescaling $s=\alpha^{1/3}t$,
equation~\eqref{eq:airy-ode} becomes
$-v''(s)+s\,v(s)=\tilde\mu_n v(s)$
with $\tilde\mu_n=\mu_n\alpha^{-2/3}$.
The spectral shifts $\{\tilde\mu_n\}$ of consecutive
indices $n,n+1\in W_c$ differ by
$\tilde\mu_{n+1}-\tilde\mu_n = a_{k+1}-a_k$
(modulo $O(c^{-1/3})$ corrections),
where $a_k$ are the zeros of $\Ai(t)$
(\cite[\S10.17]{DLMF2023}).
\end{remark}

\begin{proposition}[WKB inner-layer gap, invariant form]
\label{prop:inner-gap-invariant}
For $n\in W_c$ there exists a WKB-determined
positive scale $\delta(c)\to\infty$ such that
\begin{equation}\label{eq:inner-gap}
  \chi_{n+1}^{(c)} - \chi_n^{(c)} \ge C\,\delta(c)
\end{equation}
for some constant $C>0$ independent of $n\in W_c$
and $c\ge c_0$.
The scale $\delta(c)$ grows to infinity,
but its precise algebraic order in $c$ is
not stable under the spectral transfer
$F_c\colon\chi\mapsto\lambda$
(see Thm.~\ref{thm:wkb-indeterminacy});
no power law is claimed.
\end{proposition}

\begin{proof}
From Rem.~\ref{rem:airy-standard},
$\tilde\mu_{n+1}-\tilde\mu_n\ge c_1>0$ for all
$n\in W_c$ (bounded below by the minimal
Airy zero spacing in the relevant range).
Converting back through the two rescalings:
$\mu_{n+1}-\mu_n = \alpha^{2/3}(\tilde\mu_{n+1}-\tilde\mu_n)
\ge c_1\alpha^{2/3}$,
and
$\chi_{n+1}^{(c)}-\chi_n^{(c)}
= c^{2/3}(\mu_{n+1}-\mu_n) \ge c_1 c^{2/3}\alpha^{2/3}$.
Since $\alpha\asymp c^{1/3}$ uniformly for $n\in W_c$,
this gives $\delta(c)\asymp c^{2/3}\cdot c^{2/9}
= c^{8/9}$
(a candidate scale; the exact value is not
invariant, as explained in
Thm.~\ref{thm:wkb-indeterminacy}).
The key conclusion is $\delta(c)\to\infty$.
\end{proof}

\begin{remark}[Why no power is claimed]
\label{rem:no-power}
The quantity $\delta(c) = c^{2/3}\alpha(n,c)^{2/3}$
depends on $n\in W_c$, and the uniform estimate
$\alpha\asymp c^{1/3}$ is itself a rough bound.
A finer analysis of $\alpha(n,c)$ for
$n\in W_c$ could shift the exponent.
More fundamentally, even a precise bound
on $\chi_{n+1}-\chi_n$ does not yield
a bound on $\lambda_{n+1}-\lambda_n$
without control of $F_c'$;
this is the content of
Thm.~\ref{thm:wkb-indeterminacy}.
\end{remark}

\subsection{Transfer-layer obstruction}
\label{subsec:transfer}

\begin{theorem}[WKB indeterminacy]
\label{thm:wkb-indeterminacy}
No canonical algebraic scaling for the gap
$|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
can be derived from the local WKB/Airy
analysis of $\Lc$.
More precisely:
\begin{enumerate}[\rm(i)]
\item The inner-layer gap
      $\chi_{n+1}^{(c)}-\chi_n^{(c)}\ge C\,\delta(c)$
      (Prop.~\ref{prop:inner-gap-invariant})
      is a valid local conclusion.
\item The spectral transfer $F_c$
      from \eqref{eq:chi-lambda}
      is not algebraically stable in $c$
      in the transition window $W_c$:
      the function
      $\chi\mapsto\lambda = F_c(\chi)$
      near the critical level
      $\lambda=1/2$ has a soft-exponential
      profile (Widom--Slepian concentration
      transition, \cite{Widom1964,Osipov2013}),
      not a pure power law.
\item Consequently, any formula
      $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|
      \asymp c^{-\beta}$
      derived by assuming $F_c'\asymp c^{-\beta}$
      for some fixed $\beta$ is an unverified
      global claim, not a consequence of WKB.
\item The gap $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
      is \emph{positive} for all $n\in W_c$
      (follows from simplicity of the spectrum
      of $\Lc$ and monotonicity of $F_c$),
      but its precise rate of decay with $c$
      is a global invariant of $\mathcal{H}_c$,
      accessible only via integral operator methods.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) is Prop.~\ref{prop:inner-gap-invariant}.
(ii): the transition behavior of $F_c$ near
$\lambda=1/2$ is governed by the Slepian
concentration transition
(\cite{Slepian1961,Widom1964}),
which is known to be of the form
$\lambda\approx \frac{1}{2}+\frac{1}{2}
\mathrm{erf}(c^{1/2}(x-1))$
for $x$ near the band-limit;
the parameter $c$ enters non-algebraically.
The derivative $F_c'(\chi)$ inherits this
non-algebraic $c$-dependence.
(iii) and (iv) are immediate consequences
of (ii) and the simplicity / monotonicity
of the spectrum of $\Lc$
(Sturm--Liouville theory, \cite{ReedSimon1975}).
\end{proof}

\subsection{Structural gap hypothesis}
\label{subsec:hyp-gap}

\begin{hypothesis}[Structural gap]
\label{hyp:gap-structural}
There exists $g\colon[1,\infty)\to(0,\infty)$
with $g(c)\to 0$ and $g(c)\gg c^{-1/4}$,
and $\kappa_0>0$, such that
for all $c\ge 1$ and $n\in W_c$, $n\le\kappa_0 c$:
\begin{equation}\label{eq:gap-structural}
  |\lambda_n^{(c)} - \lambda_{n+1}^{(c)}| \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Consistency with Thm.~\ref{thm:wkb-indeterminacy}]
\label{rem:hyp-consistency}
\hyperref[hyp:gap-structural]{Hyp.~Gap-S} is consistent
with Thm.~\ref{thm:wkb-indeterminacy}:
it asserts positivity of the gap and a
lower rate bound $g(c)\gg c^{-1/4}$,
but imposes no algebraic form on $g(c)$.
The determination of $g(c)$ is
Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{lemma}[Structural gap implies separation]
\label{lem:gap-structural}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
consecutive PSWF concentration eigenvalues
in $W_c$ satisfy:
\[
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge g(c) \gg c^{-1/4}.
\]
\end{lemma}
\begin{proof}
Direct from \hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
\end{proof}

\begin{corollary}[NAcc as theorem]
\label{cor:nacc}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{XII2},
the No-Accumulation hypothesis
(Paper~XI, Hyp.~4.5) holds.
\end{corollary}

\begin{proof}
Suppose for contradiction that
$\lambda_{n_j}^{(\infty)}\to\mu\in(0,1)$
for some infinite subsequence.

Define a shrinking spectral window
\begin{equation}\label{eq:shrinking-window}
  \varepsilon_k := g(c_k)^{1/2} \cdot c_k^{-1/4}.
\end{equation}
Since $g(c_k)\gg c_k^{-1/4}$, we have
$\varepsilon_k = g(c_k)^{1/2} c_k^{-1/4}
\gg c_k^{-1/4} \cdot c_k^{-1/8} = c_k^{-3/8} \to 0$,
so $\varepsilon_k\to 0$ but
$\varepsilon_k/g(c_k) = g(c_k)^{-1/2}c_k^{-1/4}\to 0$
(since $g(c_k)\gg c_k^{-1/4}$ implies
$g(c_k)^{1/2}\gg c_k^{-1/8}$,
hence $g(c_k)^{-1/2}c_k^{-1/4}
\ll c_k^{1/8}\cdot c_k^{-1/4}=c_k^{-1/8}\to 0$).

By \eqref{XII2},
for $j$ large enough and $k$ large enough:
\[
  |\lambda_{n_j}^{(c_k)}-\mu|
  \le C_\kappa c_k^{-1/4}
     + |\lambda_{n_j}^{(\infty)}-\mu|
  < \varepsilon_k.
\]
So infinitely many $\lambda_{n_j}^{(c_k)}$
lie in $I_k := (\mu-\varepsilon_k,\mu+\varepsilon_k)$.
By Lem.~\ref{lem:gap-structural},
consecutive eigenvalues of $\mathcal{H}_{c_k}$
are spaced at least $g(c_k)$ apart,
so the number in $I_k$ is at most
$\lfloor 2\varepsilon_k/g(c_k)\rfloor + 1$.
Since $\varepsilon_k/g(c_k)\to 0$,
this count $\to 1$ as $k\to\infty$:
for large $k$, at most finitely many
(in fact at most $2$) eigenvalues of
$\mathcal{H}_{c_k}$ lie in $I_k$.
This contradicts the assumption that
infinitely many distinct $n_j$ index
eigenvalues converging to $\mu$.
\end{proof}

\begin{remark}[Summary of what Section~\ref{sec:gap} establishes]
\label{rem:sec3-summary}
\begin{itemize}
\item \textbf{Proved unconditionally:}
      Airy reduction (Lem.~\ref{lem:airy-reduction});
      positive $\chi$-gap in invariant form
      (Prop.~\ref{prop:inner-gap-invariant});
      WKB indeterminacy of gap exponent
      (Thm.~\ref{thm:wkb-indeterminacy}).
\item \textbf{Proved under Hyp.~Gap-S:}
      $\lambda$-gap dominating $c^{-1/4}$;
      NAcc theorem (Cor.~\ref{cor:nacc}).
\item \textbf{Open:}
      sharp form of $g(c)$
      (Prob.~\ref{prob:sharp-gc}).
\end{itemize}
\end{remark}

\section{Gram-Matrix Control}
\label{sec:gram}
%%══════════════════════════════════════════════════════════════════

\begin{remark}[Gram matrix of the limit]
\label{rem:gram-trivial}
By \eqref{XII1}, $\Gram_{mn}=\delta_{mn}$
for the limit system: trivially invertible.
The content of this section is uniform
Riesz stability of the finite approximants.
\end{remark}

\begin{definition}[Uniform Riesz stability]
\label{def:riesz-stable}
$\{\Phi_n^{(c_k)}\}_{n,k}$ is
\emph{uniformly Riesz-stable} if
$\exists\, 0<A\le B<\infty$ (independent of $k$)
such that for all finite $(a_n)$ and all $k$:
\begin{equation}\label{eq:riesz-stable}
  A\sum_n|a_n|^2
  \le\Bigl\|\sum_n a_n\Phi_n^{(c_k)}\Bigr\|^2
  \le B\sum_n|a_n|^2.
\end{equation}
\end{definition}

\begin{lemma}[Frame perturbation: intermediate layer]
\label{lem:frame-perturbation}
Let $\{e_n\}$ be an orthonormal system in
a Hilbert space $\mathcal{H}$
and $\{f_n^{(k)}\}$ a sequence of systems
with $f_n^{(k)}\to e_n$ strongly as $k\to\infty$
(uniformly in $n$ on finite sets).
If in addition
\begin{equation}\label{eq:gap-control}
  \inf_{n\ne m,\,k}\,
  |\lambda_n^{(c_k)}-\lambda_m^{(c_k)}| > 0
  \quad\text{(eventually in }k\text{)},
\end{equation}
then $\{f_n^{(k)}\}$ is uniformly Riesz-stable
for $k$ sufficiently large,
with bounds $A=1-\varepsilon_k$, $B=1+\varepsilon_k$,
$\varepsilon_k\to 0$.
\end{lemma}
\begin{proof}
See \cite[Thm.~5.6.1]{Christensen2003}
(frame perturbation under uniform strong convergence)
and \cite[Prop.~2.1]{YoungRiesz}
for the quantitative version.
The condition \eqref{eq:gap-control} ensures
that the Riesz lower bound does not collapse;
it is implied by
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
\end{proof}

\begin{lemma}[Gram control for approximants]
\label{lem:gram-approx}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{XII1}--\eqref{XII3}.
Then $\{\Phi_n^{(c_k)}\}$ satisfies the
hypotheses of Lem.~\ref{lem:frame-perturbation},
and is therefore uniformly Riesz-stable
with $\varepsilon_k\to 0$.
\end{lemma}
\begin{proof}
Strong convergence
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
follows from SOT convergence + eigenvalue
equation (\eqref{XII3}).
Condition \eqref{eq:gap-control} follows
from Lem.~\ref{lem:gap-structural}:
$|\lambda_n^{(c_k)}-\lambda_m^{(c_k)}|
\ge g(c_k)\gg c_k^{-1/4}>0$
for all $n\ne m$ in $W_{c_k}$.
Apply Lem.~\ref{lem:frame-perturbation}.
\end{proof}

\begin{remark}[Bari--Krein vs.\ frame perturbation]
\label{rem:bari-vs-frame}
Bari--Krein (Thm.~\ref{thm:bari-krein})
establishes the equivalence
(Riesz basis) $\Leftrightarrow$ (Gram invertible)
for a \emph{fixed} system.
For passage to a limit under SOT,
one needs the frame perturbation theory
of Lem.~\ref{lem:frame-perturbation}
(\cite{Christensen2003}) as an intermediate step.
Both are used in
Prop.~\ref{prop:riesz-basis}.
\end{remark}

\section{Bari--Krein and Frame Machinery}
\label{sec:bari-krein}
%%══════════════════════════════════════════════════════════════════

\begin{theorem}[Bari--Krein]
\label{thm:bari-krein}
A sequence $\{f_n\}$ in a separable Hilbert space
is a Riesz basis of its closed span if and only
if the Gram matrix $\ip{f_m}{f_n}$ is bounded
and invertible on $\ell^2$.
\end{theorem}
\begin{proof}
\cite[Ch.~VI]{GohbergKrein1969}.
\end{proof}

\begin{proposition}[Riesz basis of $\mathcal{K}$]
\label{prop:riesz-basis}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
$\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$.
\end{proposition}
\begin{proof}
Step 1:
By Rem.~\ref{rem:gram-trivial}, $\Gram=I$.
Step 2:
By Lem.~\ref{lem:gram-approx},
$\{\Phi_n^{(c_k)}\}$ is uniformly Riesz-stable;
by Lem.~\ref{lem:frame-perturbation}
(\cite[Thm.~5.6.1]{Christensen2003})
this stability passes to the SOT limit.
Step 3:
Apply Thm.~\ref{thm:bari-krein}
to the limit system.
\end{proof}

\section{Canonicalization}
\label{sec:canonical}

\begin{lemma}[Uniqueness of $\mathcal{K}$]
\label{lem:same-K}
Any two diagonal subsequences satisfying
Lem.~\ref{lem:XI-diagonal} and
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
yield the same closed span $\mathcal{K}$.
\end{lemma}
\begin{proof}
Both are closed invariant subspaces of $\Hlim$
(uniquely determined SOT limit).
By the spectral theorem for bounded self-adjoint
operators, the maximal invariant subspace
spanned by an orthonormal set of eigenvectors
is unique up to unitary equivalence.
Full detail: Phase~3.
\end{proof}

\begin{theorem}[Canonicalization]
\label{thm:canonical-proof}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
$\Hlim$ is a canonical self-adjoint operator on $\mathcal{K}$,
independent of diagonal subsequence up to
unitary equivalence.
\end{theorem}
\begin{proof}
(1) Riesz basis: Prop.~\ref{prop:riesz-basis}.
(2) Unique $\mathcal{K}$: Lem.~\ref{lem:same-K}.
(3) $\sigma(\Hlim|_{\mathcal{K}})=\{\lambda_n^{(\infty)}\}$
independent of subsequence
(SOT uniqueness + \eqref{XII2}).
(4) Spectral theorem: two bounded self-adjoint
operators with the same simple spectrum are
unitarily equivalent.
\end{proof}

\section{Open Problems}
\label{sec:open}
%%══════════════════════════════════════════════════════════════════

\begin{problem}[Sharp gap scale $g(c)$]
\label{prob:sharp-gc}
Determine the sharp form of $g(c)$ in
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
By Thm.~\ref{thm:wkb-indeterminacy}, this requires
global control of the spectral transfer $F_c'$.
Two approaches:
\begin{enumerate}
\item \textbf{Plancherel--Rotach / kernel asymptotics:}
      expand the prolate integral kernel near
      the critical level $\lambda=1/2$ to extract
      the density of concentration eigenvalues;
      this gives $F_c'$ in the transition regime.
\item \textbf{Resolvent asymptotics:}
      control the imaginary part of the resolvent
      $(\mathcal{H}_c-z)^{-1}$ near $z=1/2+i0$;
      the spectral density gives $F_c'$ directly.
\end{enumerate}
In either case one expects
$g(c)=c^{-\alpha}$ for some
$\alpha\in(1/4,\,1)$,
but the precise value is unknown.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
Is $\mathcal{K}=L^2(\mathbb{R})$?
Target: Paper~XIII.
\end{problem}

\begin{problem}[Density]
\label{prob:density}
Is $\{\lambda_n^{(\infty)}\}$ dense in $[0,1]$?
\end{problem}

\begin{problem}[Explicit Riesz bounds]
\label{prob:riesz-constants}
Determine $A,B$ in Def.~\ref{def:riesz-stable}
explicitly in terms of $g(c)$
(via the frame perturbation estimate of
Lem.~\ref{lem:frame-perturbation}).
\end{problem}
