%%══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context and goal}

This paper occupies a specific position in the series:
it provides the bridge between
the strong operator convergence of Paper~X
and the completeness result of Paper~XIII.

Paper~X establishes that the PSWF concentration
operators $\mathcal{H}_{c_k}$ converge in the
strong operator topology (SOT) to a bounded
self-adjoint limit $\Hlim$.
Paper~XI constructs a diagonal subsequence
along which the eigenfunctions $\Phi_n^{(c_k)}$
converge weakly to limit vectors $\Phi_n^{(\infty)}$,
orthonormal and satisfying
$\Hlim\Phi_n^{(\infty)} = \lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textbf{The gap that this paper fills.}
SOT convergence does \emph{not} imply norm
convergence of spectral projections
(Rem.~\ref{rem:strong-vs-norm}).
The difficulty is precise:
SOT convergence produces approximate eigenvectors
--- the vectors $\Phi_n^{(\infty)}$ satisfy
$\|(\mathcal{H}_{c_k}-\lambda_n^{(\infty)})
\Phi_n^{(\infty)}\|\to 0$ as $k\to\infty$ ---
but provides no control on the residual
\emph{relative to the spectral gap}.
Without such control, spectral isolation
cannot be invoked, and projection convergence
cannot be concluded.
This is exactly the gap this paper identifies
and bridges.

The central result (Thm.~\ref{thm:convergence})
establishes $\|\Pnck - \Pninf\|\to 0$
under two well-motivated conditions:
\begin{itemize}
\item \hyperref[hyp:gap-structural]{Gap-S}:
  in the transition window $W_c$,
  $\lim_{c\to\infty} c^{-1/4}/g(c)=0$
  (spectral gaps dominate the eigenvalue
  convergence rate);
\item \hyperref[hyp:sot-faster]{SOT-Faster}:
  $\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
  =o(g(c_k))$ as $k\to\infty$
  (residual decays faster than the gap).
\end{itemize}
Neither condition requires any bound on
$\|\mathcal{H}_{c_k}-\Hlim\|$.

\subsection{Main results}

\begin{itemize}
\item \textbf{Theorem~\ref{thm:convergence}
  (Convergence):}
  Under Gap-S and SOT-Faster,
  $\|\Pnck - \Pninf\|\to 0$.

\item \textbf{Theorem~\ref{thm:rate}
  (Rate):}
  Under the additional
  \hyperref[hyp:sot-rate]{Hyp.~SOT-Rate},
  $\|\Pnck-\Pninf\|
  \le C C_\kappa c_k^{-1/4}/g(c_k)$,
  uniformly in $n\le\kappa c_k$.
\end{itemize}

\begin{remark}[Core contribution]
\label{rem:conditions}
The contribution of this paper is the
identification of the exact minimal conditions
under which SOT convergence upgrades to
norm convergence of rank-one spectral projections.
The key mechanism
--- approximate eigenvector $+$ spectral isolation
$\Rightarrow$ projection convergence ---
is a rank-one specialisation of the
Davis--Kahan $\sin\Theta$ theorem
(see Lem.~\ref{lem:approx-eig}
and Rem.~\ref{rem:perturbation-mechanism}).
SOT-Faster encapsulates precisely the missing
quantitative link: the operator residual on
$\Phi_n^{(\infty)}$ must decay faster than
the spectral gap.
\end{remark}

%%══ 3. HYPOTHESES
\section{Hypotheses and Motivation}
\label{sec:gap}

\begin{hypothesis}[Structural gap in
  transition window, Gap-S]
\label{hyp:gap-structural}
Let $W_c\subset\mathbb{N}$ denote the
\emph{transition window} of indices $n$
for which $\lambda_n^{(c)}\in(\delta,1-\delta)$
for some fixed $\delta>0$.
There exists $g\colon[1,\infty)\to(0,\infty)$
with
\begin{equation}\label{eq:gap-dominates}
  g(c)\to 0
  \quad\text{and}\quad
  \lim_{c\to\infty}\frac{c^{-1/4}}{g(c)} = 0,
\end{equation}
such that for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Scope of Gap-S]
\label{rem:gap-scope}
The restriction to the transition window
$W_c$ is essential: PSWF eigenvalues
exhibit near-$0$ and near-$1$ clusters
(bulk regions) where individual gaps
are not uniformly bounded below by $g(c)$.
However, bulk-region gaps are generically
\emph{larger} than transition-window gaps,
so restricting to $W_c$ is the binding
constraint.
All indices $n$ appearing in the
diagonal subsequence of Paper~XI satisfy
$n\in W_{c_k}$ for $k$ large;
this is the only case required.
\end{remark}

\begin{remark}[Gap-S implies $c^{-1/4}=o(g(c))$]
\label{rem:gap-implies-o}
Condition \eqref{eq:gap-dominates} is
equivalent to $c^{-1/4}=o(g(c))$
as $c\to\infty$, cited in the proof
of Thm.~\ref{thm:convergence}.
\end{remark}

\begin{hypothesis}[SOT faster than gap,
  SOT-Faster]
\label{hyp:sot-faster}
For each fixed $n$, along the diagonal
subsequence of Lem.~\ref{lem:inherited}:
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k))
  \quad\text{as }k\to\infty.
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster is the
  core difficulty]
\label{rem:sot-faster-core}
SOT (Paper~X) gives $\varepsilon_n(k)\to 0$
as $k\to\infty$, but no comparison with
$g(c_k)$.
SOT-Faster is not a technical side condition:
it encapsulates \emph{the entire difficulty}
of upgrading SOT convergence to norm
convergence of rank-one spectral projections.
Proving it rigorously is
Open Problem~\ref{prob:sot-faster};
Sec.~\ref{sec:heuristic} presents
evidence consistent with it holding
in the PSWF setting.
\end{remark}

\begin{hypothesis}[SOT rate on
  eigenvectors, SOT-Rate]
\label{hyp:sot-rate}
For each fixed $n$:
\begin{equation}\label{eq:sot-rate}
  \varepsilon_n(k) = O(c_k^{-1/4})
  \quad\text{as }k\to\infty.
\end{equation}
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
\[
  \text{SOT-Rate}
  \underset{\eqref{eq:gap-dominates}}{\Longrightarrow}
  \text{SOT-Faster}
  \Longrightarrow
  \|\Pnck-\Pninf\|\to 0.
\]
SOT alone gives only $\varepsilon_n(k)\to 0$,
insufficient without comparison to $g(c_k)$.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding $\Phi_n^{(\infty)}$ in the
eigenbasis $\{\Phi_m^{(c_k)}\}$
of $\mathcal{H}_{c_k}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
\end{equation}
The diagonal term $m=n$ contributes
$\le C_\kappa^2 c_k^{-1/2}$ by \eqref{A1}.
If off-diagonal overlaps
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
are small for $m\ne n$ (suggested by Gap-S;
see Sec.~\ref{sec:heuristic}),
then \eqref{eq:formal-expansion}
is dominated by $m=n$,
giving $\varepsilon_n(k)=O(c_k^{-1/4})$.
This is formally consistent with SOT-Rate,
but does not constitute rigorous evidence;
making it precise requires off-diagonal
overlap estimates
(Open Problem~\ref{prob:sot-rate}).
\end{remark}

\subsection{WKB layer}
\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
does not determine a canonical scaling
for $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
without global kernel information.
\end{theorem}
\begin{proof}
The transfer map $F_c$ has a sigmoid
profile near $\lambda=1/2$~\cite{Widom1964};
any algebraic rate $F_c'\asymp c^{-\beta}$
is a global claim not derivable from
$\Lc$ alone.
\end{proof}

%%══ 4. MAIN THEOREMS
\section{Main Theorems}
\label{sec:main}

\begin{theorem}[Spectral projection
  convergence]
\label{thm:convergence}
Assume Lem.~\ref{lem:inherited},
\hyperref[hyp:gap-structural]{Hyp.~Gap-S},
and \hyperref[hyp:sot-faster]{Hyp.~SOT-Faster}.
For each fixed $n$:
\begin{equation}\label{eq:convergence}
  \|\Pnck - \Pninf\|\to 0
  \quad\text{as }k\to\infty.
\end{equation}
In particular:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
  strongly as $k\to\infty$, up to phase.
\item $\{\lambda_n^{(\infty)}\}$ has no
  accumulation point in $(0,1)$.
\end{enumerate}
\end{theorem}

\begin{theorem}[Rate]
\label{thm:rate}
Assume additionally
\hyperref[hyp:sot-rate]{Hyp.~SOT-Rate}.
For all $n\le\kappa c_k$,
uniformly in $n$:
\begin{equation}\label{eq:rate-bound}
  \|\Pnck - \Pninf\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)},
\end{equation}
where $C>0$ is universal.
\end{theorem}

\begin{remark}[Phase convention]
\label{rem:phase}
For rank-one projections,
$\|\Pnck-\Pninf\|\to 0$ implies
$\Phi_n^{(c_k)}\to e^{i\theta_n^{(k)}}
\Phi_n^{(\infty)}$ strongly.
Phase is fixed by the convention
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$,
valid for all large $k$.
\end{remark}

%%══ 5. PROOF
\section{Proof of the Main Theorems}
\label{sec:proj}

\begin{lemma}[Approximate eigenvector
  $\Rightarrow$ projection estimate;
  rank-one specialisation of
  Davis--Kahan $\sin\Theta$;
  cf.\ \cite{DavisKahan1970,Kato1966}]
\label{lem:approx-eig}
Let $A$ be a bounded self-adjoint operator
and $\lambda$ a simple eigenvalue of $A$
with
$\gamma:=\mathrm{dist}(\lambda,
\spc{A}\setminus\{\lambda\})>0$.
Let $\phi\in\mathcal{H}$, $\|\phi\|=1$,
with $\|(A-\lambda)\phi\|\le\varepsilon$.
Then:
\begin{equation}\label{eq:approx-eig-bound}
  \|(I-P_n^A)\phi\|
  \le\frac{\varepsilon}{\gamma},
\end{equation}
where $P_n^A$ denotes the
rank-one spectral projection of $A$
onto the eigenspace of $\lambda$.
\end{lemma}
\begin{proof}
Write $\phi = P_n^A\phi + (I-P_n^A)\phi$.
Since $(A-\lambda)P_n^A=0$:
$(A-\lambda)\phi = (A-\lambda)(I-P_n^A)\phi$.
On $\mathrm{ran}(I-P_n^A)$,
$\|(A-\lambda)^{-1}\|\le 1/\gamma$,
so
$\|(I-P_n^A)\phi\|
\le\gamma^{-1}\|(A-\lambda)\phi\|
\le\varepsilon/\gamma$.
\end{proof}

\begin{remark}[Perturbation mechanism]
\label{rem:perturbation-mechanism}
Lemma~\ref{lem:approx-eig} is a rank-one
specialisation of the Davis--Kahan
$\sin\Theta$ theorem~\cite{DavisKahan1970}.
It expresses a fundamental principle:
\emph{an approximate eigenvector whose
residual $\varepsilon_n(k)$ is smaller
than the spectral gap $g(c_k)$
must concentrate on the true eigenspace,
with $\|(I-\Pnck)\Phi_n^{(\infty)}\|
\le\varepsilon_n(k)/g(c_k)$}.
SOT-Faster ensures $\varepsilon_n(k)/g(c_k)\to 0$
as $k\to\infty$, making this principle effective.
Lemma~\ref{lem:sot-approx} shows that SOT
alone produces approximate eigenvectors
($\varepsilon_n(k)\to 0$), but without the
gap comparison that Lemma~\ref{lem:approx-eig}
requires.
\end{remark}

\begin{lemma}[SOT gives approximate
  eigenvectors, no rate]
\label{lem:sot-approx}
Under \eqref{A2}--\eqref{A3},
for each fixed $n$:
\begin{equation}\label{eq:approx-sot}
  \varepsilon_n(k)
  = \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|\to 0
  \quad\text{as }k\to\infty.
\end{equation}
No rate on $\varepsilon_n(k)$ is claimed.
\end{lemma}
\begin{proof}
Since $\Hlim\Phi_n^{(\infty)}
=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$,
we have $\varepsilon_n(k)
=\|(\mathcal{H}_{c_k}-\Hlim)
\Phi_n^{(\infty)}\|\to 0$
by SOT~\eqref{eq:sot} applied
to the fixed vector $f=\Phi_n^{(\infty)}$.
\end{proof}

\begin{proof}[Proof of
  Thm.~\ref{thm:convergence}]
Fix $n$.
By Lem.~\ref{lem:sot-approx} and the
triangle inequality, using
Lem.~\ref{lem:eigenvalue-matching}:
\begin{equation}\label{eq:eps-tilde}
  \tilde\varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
  \Phi_n^{(\infty)}\|
  \le \varepsilon_n(k)
  + |\lambda_n^{(\infty)}-\lambda_n^{(c_k)}|
  \le \varepsilon_n(k) + C_\kappa c_k^{-1/4}.
\end{equation}
Dividing by $g(c_k)$ and applying
SOT-Faster \eqref{eq:sot-faster}
and Rem.~\ref{rem:gap-implies-o}:
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to\,0
    \;\text{(SOT-Faster)}}
  +
  \underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to\,0
    \;\text{(Rem.~\ref{rem:gap-implies-o})}}
  \to 0
  \quad\text{as }k\to\infty.
\end{equation}
Apply Lem.~\ref{lem:approx-eig}
(with $A=\mathcal{H}_{c_k}$,
$\lambda=\lambda_n^{(c_k)}$,
gap $\ge g(c_k)$ by Gap-S restricted to
$n\in W_{c_k}$,
$\phi=\Phi_n^{(\infty)}$,
$\varepsilon=\tilde\varepsilon_n(k)$;
and Rem.~\ref{rem:perturbation-mechanism}):
\begin{equation}\label{eq:proj-dev}
  \|(I-\Pnck)\Phi_n^{(\infty)}\|
  \le\frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \to 0
  \quad\text{as }k\to\infty.
\end{equation}
Write $\Pnck\Phi_n^{(\infty)}
=\alpha_k\Phi_n^{(c_k)}$ with
$\alpha_k=\ip{\Phi_n^{(\infty)}}{\Phi_n^{(c_k)}}$;
then $|\alpha_k|^2=1
-\|(I-\Pnck)\Phi_n^{(\infty)}\|^2\to 1$.
By the rank-one projection identity
$\|P_u-P_v\|\le 2\|u-e^{i\theta}v\|$
(cf.~\cite[Ch.~1]{Bhatia1997}):
\[
  \|\Pnck-\Pninf\|
  \le 2\|\Phi_n^{(c_k)}
  -e^{i\arg\alpha_k}\Phi_n^{(\infty)}\|
  \to 0
  \quad\text{as }k\to\infty.
\]
\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:rate}]
Under SOT-Rate \eqref{eq:sot-rate}:
$\varepsilon_n(k)\le C' c_k^{-1/4}$,
so $\tilde\varepsilon_n(k)
\le(C'+C_\kappa)c_k^{-1/4}$.
With gap $\gamma=g(c_k)$,
Lem.~\ref{lem:approx-eig} and the
rank-one identity give \eqref{eq:rate-bound}
uniformly for $n\le\kappa c_k$.
\end{proof}

\begin{corollary}[No accumulation
  of $\lambda_n^{(\infty)}$]
\label{cor:nacc}
Under Gap-S and \eqref{A1},
$\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$.
\end{corollary}
\begin{proof}
The eigenvalues $\lambda_n^{(c)}$ are
strictly ordered in $n$~\cite{Slepian1961}.
Fix $\mu\in(0,1)$, $r>0$;
choose $k_0$ with
$C_\kappa c_{k_0}^{-1/4}<r/2$.
Then $\lambda_n^{(\infty)}\in(\mu-r,\mu+r)$
implies $\lambda_n^{(c_{k_0})}
\in(\mu-3r/2,\mu+3r/2)$.
By Gap-S (restricted to $W_{c_{k_0}}$),
at most $\lceil 3r/g(c_{k_0})\rceil<\infty$
eigenvalues lie in this interval.
\end{proof}

%%══ 8. HEURISTIC
\section{Heuristic Motivation of Gap-S
  and SOT-Faster}
\label{sec:heuristic}

\textbf{Non-rigorous throughout.}

\subsection{Sigmoid transition (Widom)}

Widom-type asymptotics~\cite{Widom1964}
predict a sigmoidal eigenvalue profile
in the transition window:
\[
  \lambda_n^{(c)}\approx
  \tfrac{1}{1+e^{\pi(n-2c/\pi)/\sigma(c)}},
  \quad\sigma(c)\asymp\log c.
\]
This predicts the correct qualitative
behavior and is consistent with Gap-S
having scale $g(c)\sim(\log c)^{-1}$.

\subsection{Gap estimate}

If $\rho(c)\asymp\log c$ eigenvalues
occupy the transition window $W_c$,
average spacing is $\sim(\log c)^{-1}
\gg c^{-1/4}$,
strongly suggesting Gap-S with
$g(c)\sim(\log c)^{-1}$.
Rigorous proof requires
Plancherel--Rotach or determinantal
kernel methods~\cite{Widom1964,Osipov2013}.

\subsection{SOT-Faster plausibility}

From \eqref{eq:formal-expansion},
if off-diagonal overlaps satisfy
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|
\to 0$ as $k\to\infty$ for each $m\ne n$
(which Gap-S makes plausible,
as spectral isolation concentrates mass
on the diagonal),
then \eqref{eq:formal-expansion}
is dominated by the $m=n$ term,
predicting $\varepsilon_n(k)=O(c_k^{-1/4})$.
This is formally consistent with SOT-Rate,
but does not constitute rigorous evidence;
making it precise requires control
of off-diagonal overlaps
(Open Problem~\ref{prob:sot-rate}).

\begin{remark}[Heuristic status]
Sections~\ref{sec:heuristic} predict the
correct qualitative scaling and provide
evidence consistent with Gap-S and
SOT-Faster; they do not constitute
rigorous proofs.
\end{remark}

%%══ 9. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Determine $g(c)$ rigorously;
expected $g(c)\sim(\log c)^{-1}$.
\end{problem}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
Prove $\varepsilon_n(k)=O(c_k^{-1/4})$
as $k\to\infty$;
would imply SOT-Faster and justify
the rate in Thm.~\ref{thm:rate}.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$
as $k\to\infty$;
would make Thm.~\ref{thm:convergence}
fully unconditional.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
Estimate
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
for $m\ne n$ rigorously;
would supply evidence for
Open Problem~\ref{prob:sot-rate}.
\end{problem}

\begin{problem}[Full family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family};
would make Lem.~\ref{lem:same-K}
unconditional.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII.
\end{problem}
