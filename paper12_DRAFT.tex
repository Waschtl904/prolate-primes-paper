\begin{abstract}
\textbf{Part~I: Abstract principle
  (fully proved).}
Spectral projection stability in the
SOT regime admits a sharp scalar
control via the invariant
\[
  R_k :=
  \frac{\|(A_k-A)\phi\|
  + |\lambda_k-\lambda|}{\gamma_k}.
\]
For bounded self-adjoint operators
with $A_k\xrightarrow{\mathrm{SOT}} A$,
under explicit spectral isolation
(Assumption~S):
$R_k\to 0$ is sufficient for
$\|P_k - P^\infty\|\to 0$,
and necessary when $\gamma_k\ge c_0>0$
(Thm.~\ref{thm:necessary-bounded-gap}).
The mechanism is an exact factorization
of responsibility into three
independent structures
(Thm.~\ref{thm:mechanism}):
rank-one structure effects scalar
reduction; the spectral gap provides
the resolvent bound; the local residual
is the direct observable under SOT.
Classical perturbation theory cannot
isolate this factorization in explicit
form because it operates with
$\|A_k-A\|$, which is insensitive
to the direction $\phi$.
The term \emph{localization} refers
throughout to restriction of the
perturbation analysis to the single
direction $\phi$ --- not to spatial
or semiclassical localization.

\medskip
\textbf{Part~II: PSWF obstruction
  set (conditional).}
We identify the complete obstruction
set for PSWF spectral projection
stability: two analytic conditions
(Gap-S, SOT-Faster) are the sole
remaining hypotheses; no further
PSWF-specific mechanism is involved
beyond these two conditions.
This reduces the problem to exactly
two open analytic problems
(Problems~\ref{prob:sharp-gc},
\ref{prob:sot-faster}).

\medskip
We further establish: a No-Go result
showing that any attempt to bypass
a gap condition is provably impossible
(Prop.~\ref{prop:no-go}); a separation
example with $\|A_k-A\|\equiv 1$
but full projection stability
(Prop.~\ref{prop:separation}); and a
quantitative counterexample linking
ratio blow-up to projection failure
(Prop.~\ref{prop:counterexample}).
\end{abstract}

\textbf{Core Theorem (informal).}
\emph{Spectral projection stability
in the SOT regime admits a sharp
scalar control via the invariant}
\begin{equation}\label{eq:main-principle}
  R_k :=
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
\end{equation}
\emph{which provides a sharp sufficient
condition, and a necessary one under
a uniform gap\/}
(Thms.~\ref{thm:abstract},
\ref{thm:necessary-bounded-gap}):
\[
  R_k\to 0
  \;\Longrightarrow\;
  \|P_k-P^\infty\|\to 0.
\]

\medskip
\noindent
\textbf{Why this factorization was
not isolated before.}
Classical perturbation theory
(Davis--Kahan, Kato) operates with
$\|A_k-A\|$, which is rotation-invariant
and insensitive to the direction $\phi$.
It cannot distinguish between
perturbations that interact with $\phi$
and those that do not, and therefore
cannot isolate the three-way
factorization in explicit form.
The novelty is not the inequality
itself, but the exact factorization
of responsibility: each of the three
structures (rank-one, gap, residual)
plays a distinct and non-substitutable
role.

\medskip
\noindent
\textbf{Why rank-one + gap + residual
suffice.}
Rank-one structure means
$P^\infty=\phi\otimes\phi$ has a single
degree of freedom --- scalar reduction.
The spectral gap supplies the resolvent
bound. The local residual is the only
quantity accessible under SOT without
global norm control. This chain is
exact and tight
(Thm.~\ref{thm:mechanism}).

\begin{lemma}[Eigenvalue matching
  and index stability]
\label{lem:eigenvalue-matching}
For each fixed $n$:
\[
  |\lambda_n^{(c_k)}
  -\lambda_n^{(\infty)}|
  \le C_\kappa c_k^{-1/4},
\]
so $\lambda_n^{(c_k)}
\to\lambda_n^{(\infty)}$.
The index $n$ is stable: strict ordering
of $\lambda_n^{(c)}$ for all $c$
(Lem.~\ref{lem:inherited}(A3)) ensures
that the correspondence
$\lambda_n^{(c_k)}
\leftrightarrow\lambda_n^{(\infty)}$
is unambiguous for all large $k$
and cannot cross.
Since eigenvalues depend continuously
and strictly monotonically on $c$
(classical Sturm--Liouville oscillation
theory~\cite[Sec.~3.2]{Osipov2013}),
crossings between distinct branches
$\lambda_n^{(c)}$ and
$\lambda_m^{(c)}$ for $n\ne m$
are excluded for all $c$.
Hence the labeling by index is globally
consistent in $c$, and the assignment
$\lambda_n^{(c_k)}\leftrightarrow
\lambda_n^{(\infty)}$ is canonical.
\end{lemma}
\begin{proof}
Convergence from \eqref{A1}.
Index stability: strict ordering is
preserved in the limit; canonical
assignment by index.
Crossing exclusion: strictly monotone
eigenvalue branches
(Sturm--Liouville character of
$\mathcal{L}_c$) cannot intersect.
\end{proof}

\section{Hypotheses, Structural
  Necessity, and No-Go}
\label{sec:gap}

\begin{hypothesis}[Effective gap, Gap-S]
\label{hyp:gap-structural}
Let $W_c:=\{n:\lambda_n^{(c)}
\in(\delta,1-\delta)\}$
for fixed $\delta>0$.
There exists $g\colon[1,\infty)
\to(0,\infty)$ with
\begin{equation}\label{eq:gap-dominates}
  \frac{c^{-1/4}}{g(c)}\to 0
  \quad\text{as }c\to\infty,
\end{equation}
and uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}
  -\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S: structural
  necessity via No-Go, and
  determinantal context]
\label{rem:gap-necessary}
Gap-S is an open hypothesis
(Prob.~\ref{prob:sharp-gc}).
Without \eqref{eq:gap-dominates}
the ratio $R_k$ in
\eqref{eq:ratio-to-zero}
cannot be shown to vanish.
Widom~\cite{Widom1964} controls
average density, not the minimum gap;
Gap-S is strictly stronger.

Gap-S is moreover \emph{structurally
necessary}: Prop.~\ref{prop:no-go}
shows that any attempt to bypass a
gap condition is provably impossible
--- without a positive gap lower bound,
projection instability is unavoidable
for some SOT-convergent family.
Gap-S is precisely the hypothesis that
excludes this pathological regime.
The analogous bound in determinantal
point processes follows from
sine-kernel repulsion.
\end{remark}

\begin{remark}[Gap-S is logically
  independent of the stability principle]
\label{rem:gap-independent}
Gap-S is not implied by the stability
principle (Thm.~\ref{thm:abstract}),
nor does it imply it.
Gap-S supplies the gap lower bound
$\gamma_k\ge g(c_k)$ needed in the
PSWF instantiation
(Lem.~\ref{lem:gap-bridge}),
but is not logically entailed by
the abstract mechanism.
It is an independent analytic
hypothesis about the PSWF eigenvalue
spectrum.
\end{remark}

\begin{remark}[$c^{-1/4}/g(c)\to 0$]
\label{rem:gap-implies-o}
\eqref{eq:gap-dominates} states that
eigenvalue-matching error is of strictly
smaller order than $g(c)$;
used in \eqref{eq:ratio-to-zero}.
\end{remark}

\begin{proposition}[No-Go: bypassing
  a gap condition is impossible]
\label{prop:no-go}
Let $(\mu_n)$ have
$\liminf_n(\mu_n-\mu_{n+1})=0$
but Widom-type average density.
Then there exists
$B_k\xrightarrow{\mathrm{SOT}} B$
with $\|Q_n^{(k)}-Q_n\|\not\to 0$
for infinitely many $n$.
In particular, no condition on
eigenvalue density alone can guarantee
spectral projection stability.
\end{proposition}
\begin{proof}
Choose $n_j$ with
$\gamma_j:=\mu_{n_j}-\mu_{n_j+1}\to 0$.
On $\mathrm{span}\{e_{n_j},e_{n_j+1}\}$:
\[
  B_k^{(j)}:=
  \begin{pmatrix}
    \mu_{n_j}^{(k)} & \sqrt{\gamma_j^{(k)}}\\
    \sqrt{\gamma_j^{(k)}} & \mu_{n_j+1}^{(k)}
  \end{pmatrix}.
\]
Set $B_k:=\bigoplus_j B_k^{(j)}$;
SOT convergence follows since each $e_m$
lies in at most one block.
Eigenvector angle:
$\tan(2\theta_j^{(k)})
=2/\sqrt{\gamma_j^{(k)}}\to\infty$,
so $\|Q_{n_j}^{(k)}-Q_{n_j}\|\to 1/2$.
\end{proof}

\begin{remark}[Scope of No-Go]
\label{rem:no-go-scope}
This is an existential result:
for any sequence with $\liminf$ gap $=0$,
there \emph{exists} a SOT-convergent
family exhibiting projection instability.
It does not assert instability for
every such family, and does not
contradict structured kernels with
spectral rigidity (e.g.\ PSWF-type
kernels), where Gap-S prevents gap
collapse.
The result shows that a gap lower bound
is not a technical artifact but a
structural necessity.
\end{remark}

\begin{hypothesis}[SOT-Faster]
\label{hyp:sot-faster}
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k)
  :=\|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k)).
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster: pointwise,
  not uniform; genuine open hypothesis]
\label{rem:sot-faster-localized}
SOT-Faster is a pointwise condition
at the single fixed vector
$\Phi_n^{(\infty)}$, not uniform control
over a spectral window.
SOT gives only $\varepsilon_n(k)\to 0$;
SOT-Faster adds rate $o(g(c_k))$
at one vector --- strictly weaker than
local norm convergence over a window.
There is no general reason why SOT
convergence should be compatible with
any prescribed rate relative to
$g(c_k)$: this is precisely why
SOT-Faster is an open hypothesis and
not a consequence of SOT.
SOT-Faster is the sole remaining
analytic bottleneck in the PSWF
reduction
(Prob.~\ref{prob:sot-faster}).
\end{remark}

\begin{remark}[Logical status of
  SOT-Faster]
\label{rem:sot-faster-logical}
SOT-Faster plays \emph{no role} in
the abstract stability principle
(Thm.~\ref{thm:abstract},
Thm.~\ref{thm:mechanism}).
It is used exclusively in the PSWF
instantiation (Sec.~\ref{sec:main})
to compare two independent asymptotic
regimes.
The abstract theorem remains valid
without any rate assumption on SOT.
\end{remark}

\begin{remark}[Logical hierarchy]

\section{Abstract Stability Principle}
\label{sec:abstract}

\begin{assumption}[Spectral isolation,
  Assumption~S]
\label{ass:spectral-isolation}
Let $\{A_k\}$, $A$ bounded self-adjoint
on $\mathcal{H}$ with
$A_k\xrightarrow{\mathrm{SOT}} A$.
\begin{enumerate}[\rm(S1)]
\item\label{S1}
  $\lambda\in\sigma(A)$ isolated,
  simple; $P^\infty=\phi\otimes\phi$.
\item\label{S2}
  There exists an open interval
  $I\ni\lambda$ with
  $\sigma(A)\cap I=\{\lambda\}$.
  For each $k$, $\lambda_k$ is the
  \emph{unique} eigenvalue of $A_k$
  in $I$, isolated and simple;
  $\gamma_k:=\dist(\lambda_k,
  \sigma(A_k)\setminus\{\lambda_k\})
  >0$; $P_k$ rank-one.
\item\label{S3}
  $\lambda_k\to\lambda$.
\end{enumerate}
\end{assumption}

\begin{remark}[Spectral isolation,
  index stability, and SOT]
\label{rem:spectral-isolation}
Under strong resolvent convergence,
Assumption~S follows automatically
from isolation of
$\lambda$~\cite[Ch.~VIII]{Kato1966}.
Under mere SOT, eigenvalues can in
principle fail to be consistently
indexed; (S3) must be verified
explicitly.
The uniqueness clause in (S2) prevents
eigenvalue jumping: no other branch
of $\sigma(A_k)$ can enter $I$ and
be mistakenly identified as $\lambda_k$.

In the PSWF application, strict
ordering of $\lambda_n^{(c)}$ for
all $c$ is not an assumption but a
classical consequence of the
Sturm--Liouville character of
$\mathcal{L}_c$
(Lem.~\ref{lem:eigenvalue-matching},
\cite[Sec.~3.2]{Osipov2013}).
In the abstract setting, (S3) is an
explicit axiom the user must verify.
\end{remark}

\begin{remark}[Role of SOT as minimal
  topology for residual observability]
\label{rem:sot-role}
The only contribution of SOT in this
paper is pointwise vanishing:
\[
  (A_k-A)\phi\to 0
  \quad\text{for each fixed }\phi.
\]
No uniform control is used or required.
The spectral gap and isolation are
independent structural inputs,
not consequences of SOT.

SOT is the minimal topology ensuring
pointwise residual observability:
any weaker mode of convergence that
does not ensure $(A_k-A)\phi\to 0$
immediately destroys step~(M3) of
the mechanism, and the factorization
chain breaks irreparably.
SOT is therefore not doing ``hidden
norm work''; it is precisely the
minimal structure the mechanism
requires.

This is non-trivial:
SOT alone is not sufficient
(Rem.~\ref{rem:strong-vs-norm},
Prop.~\ref{prop:no-go}),
and the gap alone cannot substitute
for residual control
(Prop.~\ref{prop:counterexample}).
\end{remark}

\begin{theorem}[Exact Mechanism
  Factorization]
\label{thm:mechanism}
Under Assumption~S, the scalar
control $R_k\to 0
\Rightarrow\|P_k-P^\infty\|\to 0$
is the composition of three exact,
independent, and non-substitutable
steps:
\begin{enumerate}[\rm(M1)]
\item\label{M1}
  \textbf{(Rank-one scalar reduction.)}
  $\|P-Q\|=\|(I-Q)u\|$
  for rank-one projections
  $P=u\otimes u$, $Q=v\otimes v$
  \textup{\cite[Ch.~1]{Bhatia1997}}.
  Hence $\|P_k-P^\infty\|
  =\|(I-P_k)\phi\|$.
  The projection reduces to a single
  scalar degree of freedom.
\item\label{M2}
  \textbf{(Spectral gap resolvent
  bound.)}
  $\|(A_k-\lambda_k)^{-1}
  |_{\mathrm{ran}(I-P_k)}\|
  \le 1/\gamma_k$.
  Since $(A_k-\lambda_k)\phi
  =(A_k-\lambda_k)(I-P_k)\phi$:
  $\|(I-P_k)\phi\|
  \le\|(A_k-\lambda_k)\phi\|/\gamma_k$.
\item\label{M3}
  \textbf{(SOT residual triangle.)}
  $\|(A_k-\lambda_k)\phi\|
  \le\|(A_k-A)\phi\|
  +|\lambda_k-\lambda|$,
  using $(A-\lambda)\phi=0$.
\end{enumerate}
Combined:
\[
  \|P_k-P^\infty\| \le R_k \to 0.
\]
Each step is exact and tight.
Removing any single structure
breaks the chain independently:
(M1) fails for higher rank;
(M2) fails without a gap lower bound
(Prop.~\ref{prop:no-go});
(M3) fails without pointwise residual
control, i.e.\ without SOT
(Prop.~\ref{prop:counterexample}).
\end{theorem}
\begin{proof}
(M1): \cite[Ch.~1]{Bhatia1997}.
(M2): Spectral theorem on
$\mathrm{ran}(I-P_k)$;
$(A_k-\lambda_k)P_k\phi=0$.
(M3): Triangle with $(A-\lambda)\phi=0$.
\end{proof}

\begin{remark}[Each structure is
  non-substitutable]
\label{rem:chain-exact}
(M1): rank-one collapses the projection
to one direction --- not ``one direction
controls all,'' but ``there is only
one direction.''
(M2): a standard resolvent bound using
only $\gamma_k$, independent of $A_k-A$.
(M3): exact because $(A-\lambda)\phi=0$
strictly.
No two of the three structures suffice
without the third.
\end{remark}

\begin{remark}[Davis--Kahan and the
  present principle operate on
  incompatible information structures]
\label{rem:input-comparison}
Davis--Kahan~\cite{DavisKahan1970}
is driven by $\|A_k-A\|$: a
rotation-invariant quantity insensitive
to the direction $\phi$ and to whether
the perturbation interacts with the
active subspace.
The present principle depends on
$\|(A_k-A)\phi\|$: a direction-sensitive
quantity that measures exactly this
interaction.
These operate on fundamentally
incompatible information structures:
$\|A_k-A\|$ cannot be used to compute
$\|(A_k-A)\phi\|$ in either direction
without additional structure.
Prop.~\ref{prop:separation} makes
this concrete: $\|A_k-A\|=1$ but
$\|(A_k-A)\phi\|\to 0$, with full
projection stability.
Davis--Kahan gives no information here;
the present principle gives a complete
answer.
\end{remark}

\begin{remark}[Non-embeddability into
  norm-based perturbation theory]
\label{rem:nonreducibility}
There is no function $F$ such that
$\|(A_k-A)\phi\| = F(\|A_k-A\|)$
or vice versa, even under Assumption~S.
Hence the hypothesis cannot be
embedded into norm-resolvent
perturbation theory without loss of
precisely the information that makes
the principle work in the SOT regime.
\end{remark}

\begin{lemma}[Davis--Kahan implies
  ratio condition]
\label{lem:dk-comparison}
Under Assumption~S, if
$\|A_k-A\|\to 0$ and
$\gamma_k\ge c_0>0$:
$\|A_k-A\|\to 0
\Rightarrow R_k\to 0
\Rightarrow\|P_k-P^\infty\|\to 0$;
neither implication reverses
(Props.~\ref{prop:separation},
\ref{prop:counterexample}).
\end{lemma}
\begin{proof}
$\|(A_k-A)\phi\|\le\|A_k-A\|\to 0$;
$|\lambda_k-\lambda|\le\|A_k-A\|\to 0$;
divide by $c_0$.
\end{proof}

\begin{proposition}[Separation:
  incompatible regimes,
  $\|A_k-A\|\equiv 1$]
\label{prop:separation}
There exist bounded self-adjoint
$\{A_k\}$, $A$ on $\ell^2$
satisfying Assumption~S with
$\gamma_k\ge c_0>0$, such that:
\begin{enumerate}[\rm(i)]
\item $\|A_k-A\|=1$ for all $k$;
\item $\|(A_k-A)\phi\|
  =\varepsilon_k\to 0$;
\item $A_k-A$ has a genuine
  off-diagonal spectral interaction
  component;
\item the large-norm component is
  asymptotically spectrally orthogonal
  to $\mathrm{span}\{\phi\}$;
\item $\|P_k-P^\infty\|\to 0$.
\end{enumerate}
\end{proposition}
\begin{proof}
Set $\phi=e_1$, $\lambda=0$,
$A:=\sum_{n\ge 2}\mu_n(e_n\otimes e_n)$,
$\mu_n\in[-1,-c_0]$.
Define $A_k-A:=S_k+K_k$ where
$S_k:=\varepsilon_k
(e_2\otimes e_1+e_1\otimes e_2)$,
$\|S_k\|=\varepsilon_k\to 0$,
and $K_k$ acts on
$\mathrm{span}\{e_3,e_4,\ldots\}$
with $\|K_k\|=1$ and spectrum of
$A|_{\mathrm{span}\{e_n,n\ge 3\}}+K_k$
uniformly away from $0$.

\textit{(i)}: $\|A_k-A\|\ge\|K_k\|=1$.
\textit{(iii)}: $S_k$ couples
$e_1,e_2$ off-diagonally.
\textit{(iv)}: $K_ke_1=0$.
\textit{(ii)}: $(A_k-A)\phi
=\varepsilon_k e_2$;
$\|(A_k-A)\phi\|=\varepsilon_k\to 0$.
\textit{Gap}: $\gamma_k\ge c_0/2$
for large $k$~\cite[Ch.~II]{Kato1966}.
\textit{(v)}: Thm.~\ref{thm:abstract}.
\end{proof}

\begin{remark}[Direction-sensitivity
  is the mechanism, not a defect]
\label{rem:block-is-the-point}
Step~(M1) says $P^\infty=\phi\otimes\phi$
has a single degree of freedom.
Any perturbation component spectrally
orthogonal to $\mathrm{span}\{\phi\}$
is invisible to $P^\infty$,
regardless of its norm.
Davis--Kahan sees $\|A_k-A\|\equiv 1$
and gives no information.
The present principle identifies the
geometrically relevant quantity.
Direction-sensitivity is the mechanism.
\end{remark}

\begin{theorem}[Main Principle;
  sufficient condition]
\label{thm:abstract}
Under Assumption~S,
$R_k\to 0$
$\Rightarrow$
$\|P_k-P^\infty\|\to 0$.
\end{theorem}
\begin{proof}
Thm.~\ref{thm:mechanism},
steps (M3), (M2), (M1).
\end{proof}

\begin{remark}[Non-circularity]
\label{rem:non-circularity}
$\phi$ is a fixed eigenvector of $A$,
not of $A_k$. No spectral data of
$A_k$ enters the construction of $\phi$.
\end{remark}

\begin{remark}[Structural independence]
\label{rem:structural-independence}
All $A_k$-quantities appear only as
$(A_k-A)\phi$ or $(A_k-\lambda_k)\phi$.
\end{remark}

\begin{remark}[Rank-one: canonical
  minimal model case]
\label{rem:rank-one-scope}
Rank-one is not an artificial
restriction but the \emph{canonical
minimal model case} in which the
direction-sensitive mechanism is
sharp and complete.
The abstract mechanism is not
intrinsically rank-one.
What is rank-one specific is the
scalar reduction in step~(M1):
\[
  \|P_k - P^\infty\|
  = \|(I-P_k)\phi\|,
\]
which holds only for rank-one
projections and makes the proof
fully one-dimensional at the level
of estimates, allowing $R_k$ to
serve as a complete scalar invariant.

For higher-multiplicity eigenspaces,
scalar reduction fails: rotations
within the eigenspace introduce new
degrees of freedom requiring
additional geometric input beyond
a single test vector.
Rank-one is the minimal nontrivial
case with complete scalar control.
Extension to higher rank is a separate
problem
(Prob.~\ref{prob:higher-rank}).
\end{remark}

\begin{remark}[Fixed test vector
  convention]
\label{rem:fixed-test-vector}
The vector $\Phi_n^{(\infty)}$ is
fixed in $L^2$ and is \emph{not}
transported along the $c_k$-dependent
spectral bases $\{\Phi_m^{(c_k)}\}$.
All residual computations are
performed in the ambient $L^2$ space.
This is consistent with the abstract
setting, where $\phi$ is the fixed
eigenvector of $A$
(Rem.~\ref{rem:non-circularity}).
\end{remark}

\begin{lemma}[Rank-one projection
  $\Rightarrow$ eigenvector convergence]
\label{lem:rank-one-proj}
If $P_k=v_k\otimes v_k$,
$P^\infty=\phi\otimes\phi$,
$\|P_k-P^\infty\|\to 0$, then
$\exists$ phases $\theta_k$:
$\|v_k-e^{i\theta_k}\phi\|\to 0$.
\end{lemma}
\begin{proof}
Standard rank-one estimate.
\end{proof}

\begin{theorem}[Necessary condition;
  bounded-gap case]
\label{thm:necessary-bounded-gap}
Under Assumption~S, if
$\|P_k-P^\infty\|\to 0$ and
$\gamma_k\ge c_0>0$,
then $R_k\to 0$.
\end{theorem}
\begin{proof}
\textbf{Eigenvalue:}
$|\lambda_k-\lambda|/c_0\to 0$.

\textbf{Residual:}
$(A_k-A)\phi
=(A_k-A)P_k\phi
+(A_k-A)(I-P_k)\phi$.
Term~(II):
$\le 2M\|(I-P_k)\phi\|\to 0$.
Term~(I): $\alpha_k(\lambda_kv_k-Av_k)$;
$|\lambda_k-\lambda|\to 0$;
$(A-\lambda)v_k\to 0$ by
Lem.~\ref{lem:rank-one-proj}.
Divide by $c_0$.
\end{proof}

\begin{remark}[Necessity requires
  uniform gap]
\label{rem:necessity-uniform-gap}
Without $\gamma_k\ge c_0$, $R_k$
may diverge even when
$\|P_k-P^\infty\|\to 0$.
The full iff characterization
for vanishing gap remains open
(Prob.~\ref{prob:full-iff}).
\end{remark}

\begin{remark}[Rank-one essential
  for necessity]
\label{rem:rank-one-essential}
For higher-multiplicity eigenspaces,
additional assumptions are required
to extract eigenvector convergence
from projection convergence.
\end{remark}

\begin{remark}[Novelty: exact
  factorization in the SOT regime]
\label{rem:abstract-novelty}
Classical theory
(Davis--Kahan~\cite{DavisKahan1970},
Kato~\cite{Kato1966},
Chatelin~\cite{Chatelin1983},
Osborn~\cite{Osborn1975})
requires $\|A_k-A\|\to 0$ or
norm-resolvent convergence.
Such global-norm input conflates
what rank-one does, what the gap does,
and what SOT provides, and therefore
cannot isolate the factorization in
explicit form.

The novelty of the present result is
not the inequality $R_k\to 0
\Rightarrow\|P_k-P^\infty\|\to 0$
itself, but the exact factorization
of responsibility:
rank-one effects scalar reduction;
gap supplies the resolvent bound;
SOT delivers the observable residual.
Each structure is necessary and
plays a distinct, non-substitutable
role (Thm.~\ref{thm:mechanism}).

This factorization is invisible to
norm-based theory because $\|A_k-A\|$
conflates all three structures.
\end{remark}

\begin{proposition}[Counterexample:
  ratio blow-up implies projection
  failure]
\label{prop:counterexample}
There exist $\{A_k\}$, $A$
satisfying Assumption~S with
$A_k\xrightarrow{\mathrm{SOT}} A$,
$\lambda_k\to\lambda$, such that
$R_k\equiv 1/\sqrt{2}$
and
$\|P_k-P^\infty\|\ge 1/2$.
\end{proposition}
\begin{proof}
$\phi=e_1$, $\lambda=0$,
$\gamma_k:=k^{-1/2}$,
$v_k:=\tfrac{1}{\sqrt{2}}(e_1+e_2)$,
$A:=\sum_{n\ge 2}n^{-1}(e_n\otimes e_n)$,
$A_k:=\gamma_k(v_k\otimes v_k)
+\sum_{n\ge 2}n^{-1}(e_n\otimes e_n)$.
$(A_k-A)e_1
=\tfrac{\gamma_k}{2}(e_1+e_2)$;
$\|(A_k-A)\phi\|=\gamma_k/\sqrt{2}$;
$R_k=1/\sqrt{2}$.
$P_k=v_k\otimes v_k$;
$\|P_k-P^\infty\|
\ge|\tfrac{1}{2}-1|=\tfrac{1}{2}$.
\end{proof}

\begin{remark}[Structural necessity
  of $R_k$]
\label{rem:counterexample-meaning}
By (M1): $\|P_k-P^\infty\|
=\|(I-P_k)\phi\|$.
If $R_k\ge c>0$, then
$\|(I-P_k)\phi\|\asymp R_k$ is tight;
the scalar invariant $R_k$ is
structurally necessary, not merely
sufficient.
\end{remark}

%%══ 5. PSWF APPLICATION
\section{PSWF Application:
  Conditional Corollaries}
\label{sec:main}

% FIX C: Obstruction Set zuerst,
% Reduktion danach
\begin{center}
\fbox{\parbox{0.88\linewidth}{%
\centering\itshape
\textbf{Main contribution of this
section:}
We identify the complete obstruction
set for PSWF spectral projection
stability.
Two analytic conditions --- Gap-S
(Hyp.~\ref{hyp:gap-structural})
and SOT-Faster
(Hyp.~\ref{hyp:sot-faster}) ---
are the \emph{sole} remaining
hypotheses; no further PSWF-specific
mechanism is involved beyond the
inherited data
(Lem.~\ref{lem:inherited}).
This reduces the problem to exactly
two open analytic problems.
Both are open
(Probs.~\ref{prob:sharp-gc},
\ref{prob:sot-faster}).
The following corollaries are
\textbf{conditional} on these
two hypotheses.}}
\end{center}

\medskip
Assumption~S is verified by
Lem.~\ref{lem:inherited}(A3)
and Lem.~\ref{lem:eigenvalue-matching};
index stability follows from the
Sturm--Liouville character of
$\mathcal{L}_c$
(Rem.~\ref{rem:spectral-isolation}).
The vector $\Phi_n^{(\infty)}$ serves
as fixed test vector in the ambient
$L^2$ space throughout
(Rem.~\ref{rem:fixed-test-vector}).

\begin{lemma}[Gap-S supplies gap bound]
\label{lem:gap-bridge}
Under Hyp.~\ref{hyp:gap-structural},
$\gamma_k\ge g(c_k)$
for each $n\in W_{c_k}$.
\end{lemma}
\begin{proof}
Strict ordering gives
$\gamma_k\ge g(c_k)$.
\end{proof}

\begin{corollary}[Projection convergence,
  conditional]
\label{cor:convergence}
\textup{[Conditional on Gap-S and
SOT-Faster.]}
Under Lem.~\ref{lem:inherited},
Hyps.~\ref{hyp:gap-structural},
\ref{hyp:sot-faster},
for fixed $n\in W_{c_k}$:
$\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$.
In particular,
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
strongly (up to phase).
\end{corollary}

\begin{corollary}[Rate, conditional]
\label{cor:rate}
\textup{[Conditional on Gap-S and
SOT-Rate.]}
Under Hyp.~\ref{hyp:sot-rate},
for $n\le\kappa c_k$, $n\in W_{c_k}$,
uniformly in $n$:
\[
  \|P_n^{(c_k)}-P_n^{(\infty)}\|
  \le
  \frac{C\,C_\kappa\,c_k^{-1/4}}
  {g(c_k)}.
\]
If $g(c)\sim(\log c)^{-1}$:
rate $O(c_k^{-1/4}\log c_k)$.
\end{corollary}

\begin{remark}[Phase convention]
Phase fixed by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
for large $k$.
\end{remark}

%%══ 6. PROOFS
\section{Proofs of Corollaries}
\label{sec:proj}

\begin{lemma}[Approx.\ eigenvector
  bound]
\label{lem:approx-eig}
Let $A$ bounded self-adjoint,
$\lambda$ isolated, simple, gap
$\gamma>0$, $P$ rank-one,
$\|\phi\|=1$,
$\|(A-\lambda)\phi\|\le\varepsilon$:
$\|(I-P)\phi\|\le\varepsilon/\gamma$.
\end{lemma}
\begin{proof}
$(A-\lambda)\phi=(A-\lambda)(I-P)\phi$;
spectral theorem.
\end{proof}

\begin{proof}[Proof of
  Cor.~\ref{cor:convergence}]
$\tilde\varepsilon_n(k)
:=\|(\mathcal{H}_{c_k}
-\lambda_n^{(c_k)})
\Phi_n^{(\infty)}\|
\le\varepsilon_n(k)+C_\kappa c_k^{-1/4}$.
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}
  {g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to 0
    \,(\text{SOT-Faster})}
  +\underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to 0
    \,(\text{Rem.}\,
    \ref{rem:gap-implies-o})}
  \to 0.
\end{equation}
Lem.~\ref{lem:approx-eig};
(M1) completes.
\end{proof}

\begin{proof}[Proof of
  Cor.~\ref{cor:rate}]
$\tilde\varepsilon_n(k)
\le(C'+C_\kappa)c_k^{-1/4}$;
divide by $g(c_k)$.
\end{proof}

\begin{remark}[No accumulation]
\label{rem:nacc}
Gap-S implies finitely many
$\lambda_n^{(\infty)}$
in any subinterval of $(0,1)$.
\end{remark}

%%══ 10. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent\textbf{Conjecture-level:}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove Gap-S with
$g(c)\sim(\log c)^{-1}$
via sine-kernel repulsion.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$.
\end{problem}

\begin{problem}[Full iff,
  vanishing gap]
\label{prob:full-iff}
Characterize $R_k\to 0$
$\Leftrightarrow$
$\|P_k-P^\infty\|\to 0$
when $\gamma_k\to 0$.
\end{problem}

\noindent\textbf{Technical-level:}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
$\varepsilon_n(k)=O(c_k^{-1/4})$.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
Bound
$|\ip{\Phi_n^{(\infty)}}
{\Phi_m^{(c_k)}}|$
for $m\ne n$.
\end{problem}

\begin{problem}[Strong resolvent
  convergence]
\label{prob:src}
Does $\mathcal{H}_{c_k}$ converge
in strong resolvent sense?
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII.
\end{problem}

\begin{problem}[Higher-rank extension]
\label{prob:higher-rank}
Extend the localization principle
to higher-multiplicity eigenspaces,
where scalar reduction (M1) fails
and additional geometric structure
is required.
\end{problem}
