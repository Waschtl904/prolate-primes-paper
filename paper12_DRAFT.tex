\begin{abstract}
\textbf{Main result (abstract,
  fully proved).}
For bounded self-adjoint operators with
$A_k\xrightarrow{\mathrm{SOT}} A$,
under explicit spectral isolation
(Assumption~S), the gap-normalized
residual condition
\[
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0
\]
is sufficient for rank-one spectral
projection stability
$\|P_k - P^\infty\|\to 0$,
and necessary when $\gamma_k\ge c_0>0$.
The mechanism is explicit
(Lem.~\ref{lem:mechanism-chain}):
rank-one structure reduces projection
error to a single scalar; the spectral
gap provides the resolvent bound;
the local residual is the direct
observable under SOT.
The hypothesis structure is strictly
different from Davis--Kahan: it depends
on a direction-sensitive quantity
$\|(A_k-A)\phi\|$, not the
rotation-invariant $\|A_k-A\|$
(Rem.~\ref{rem:input-comparison},
Prop.~\ref{prop:separation}).

\medskip
\textbf{Conditional PSWF corollary.}
Under two analytic hypotheses
(Gap-S, SOT-Faster), PSWF projection
stability reduces entirely to the
abstract principle.
Both hypotheses are open
(Problems~\ref{prob:sharp-gc},
\ref{prob:sot-faster}).

\medskip
We further establish: a No-Go result
(Prop.~\ref{prop:no-go}); a separation
example with $\|A_k-A\|\equiv 1$
but full projection stability
(Prop.~\ref{prop:separation}); and a
quantitative counterexample linking
ratio blow-up to projection failure
(Prop.~\ref{prop:counterexample}).
\end{abstract}

\subsection{Context and goal}

\begin{center}
\fbox{\parbox{0.88\linewidth}{%
\centering\itshape
SOT produces approximate eigenvectors,
but does not control the residual
relative to the spectral gap.
This is the precise obstruction to
spectral projection stability.}}
\end{center}

\medskip
\noindent
\textbf{Main Principle (informal).}
Let $A_k\xrightarrow{\mathrm{SOT}} A$
with isolated simple eigenvalue $\lambda$
and associated simple eigenvalues
$\lambda_k$ (Assumption~S). Then:
\begin{equation}\label{eq:main-principle}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0
  \;\Longrightarrow\;
  \|P_k-P^\infty\|\to 0.
\end{equation}
Converse when $\gamma_k\ge c_0>0$
(Thm.~\ref{thm:necessary-bounded-gap}).

\medskip
\noindent
\textbf{Why rank-one + gap + residual
suffice.}
The mechanism is structural, not
``one direction controls all.'' Rank-one
structure means $P^\infty=\phi\otimes\phi$
has a single degree of freedom. The
spectral gap supplies the resolvent
bound. The local residual is the only
quantity accessible under SOT without
global norm control. This chain is
exact and tight
(Lem.~\ref{lem:mechanism-chain}).

\medskip
This paper proves:
\begin{itemize}
\item \eqref{eq:main-principle} is the
  sharp sufficient mechanism
  (Thm.~\ref{thm:abstract},
  Lem.~\ref{lem:mechanism-chain});
\item necessary when $\gamma_k\ge c_0>0$
  (Thm.~\ref{thm:necessary-bounded-gap});
\item the hypothesis is direction-sensitive
  and not reducible to Davis--Kahan
  (Rem.~\ref{rem:input-comparison},
  Prop.~\ref{prop:separation});
\item SOT is the minimal access condition
  for the local residual; gap and
  isolation are independent structural
  inputs (Rem.~\ref{rem:sot-role});
\item eigenvalue density alone does not
  control spectral angles
  (Prop.~\ref{prop:no-go});
\item ratio blow-up implies projection
  failure quantitatively
  (Prop.~\ref{prop:counterexample});
\item PSWF stability reduces
  conditionally to Gap-S and SOT-Faster
  (Cors.~\ref{cor:convergence},
  \ref{cor:rate}).
\end{itemize}

%%══ 3. HYPOTHESES AND NO-GO
\section{Hypotheses, Structural
  Necessity, and No-Go}
\label{sec:gap}

\begin{hypothesis}[Effective gap, Gap-S]
\label{hyp:gap-structural}
Let $W_c:=\{n:\lambda_n^{(c)}
\in(\delta,1-\delta)\}$
for fixed $\delta>0$.
There exists $g\colon[1,\infty)
\to(0,\infty)$ with
\begin{equation}\label{eq:gap-dominates}
  \frac{c^{-1/4}}{g(c)}\to 0
  \quad\text{as }c\to\infty,
\end{equation}
and uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}
  -\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S: status and
  determinantal context]
\label{rem:gap-necessary}
Gap-S is an open hypothesis
(Prob.~\ref{prob:sharp-gc}).
Without \eqref{eq:gap-dominates}
the ratio in \eqref{eq:ratio-to-zero}
cannot be shown to vanish.
Widom~\cite{Widom1964} controls
average density, not the minimum gap;
Gap-S is strictly stronger.
The analogous bound in determinantal
point processes follows from
sine-kernel repulsion.
\end{remark}

\begin{remark}[$c^{-1/4}/g(c)\to 0$]
\label{rem:gap-implies-o}
\eqref{eq:gap-dominates} states that
eigenvalue-matching error is of strictly
smaller order than $g(c)$;
used in \eqref{eq:ratio-to-zero}.
\end{remark}

\begin{proposition}[No-Go: eigenvalue
  density does not control spectral
  angles]
\label{prop:no-go}
Let $(\mu_n)$ have
$\liminf_n(\mu_n-\mu_{n+1})=0$
but Widom-type average density.
Then there exists
$B_k\xrightarrow{\mathrm{SOT}} B$
with $\|Q_n^{(k)}-Q_n\|\not\to 0$
for infinitely many $n$.
\end{proposition}
\begin{proof}
Choose $n_j$ with
$\gamma_j:=\mu_{n_j}-\mu_{n_j+1}\to 0$.
On $\mathrm{span}\{e_{n_j},e_{n_j+1}\}$:
\[
  B_k^{(j)}:=
  \begin{pmatrix}
    \mu_{n_j}^{(k)} & \sqrt{\gamma_j^{(k)}}\\
    \sqrt{\gamma_j^{(k)}} & \mu_{n_j+1}^{(k)}
  \end{pmatrix}.
\]
Set $B_k:=\bigoplus_j B_k^{(j)}$;
SOT convergence follows since each $e_m$
lies in at most one block.
Eigenvector angle:
$\tan(2\theta_j^{(k)})
=2/\sqrt{\gamma_j^{(k)}}\to\infty$,
so $\|Q_{n_j}^{(k)}-Q_{n_j}\|\to 1/2$.
\end{proof}

\begin{remark}[Scope of No-Go]
\label{rem:no-go-scope}
This is an existential pathology:
for any sequence with $\liminf$ gap $=0$,
there \emph{exists} a family exhibiting
projection instability.
It does not assert instability for
every such family, and does not
contradict structured kernels with
spectral rigidity (e.g.\ integrable
operators, PSWF-type kernels), where
Gap-S prevents gap collapse.
The construction exploits exactly
the regime excluded by Gap-S.
\end{remark}

\begin{hypothesis}[SOT-Faster]
\label{hyp:sot-faster}
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k)
  :=\|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k)).
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster: pointwise,
  not uniform; sole analytic bottleneck]
\label{rem:sot-faster-localized}
SOT-Faster is a pointwise condition
at the single fixed vector
$\Phi_n^{(\infty)}$, not uniform control
over a spectral window.
SOT gives only $\varepsilon_n(k)\to 0$;
SOT-Faster adds rate $o(g(c_k))$
at one vector --- strictly weaker than
local norm convergence over a window.
It is the sole remaining analytic
bottleneck in the PSWF reduction
(Prob.~\ref{prob:sot-faster}).
\end{remark}

\begin{hypothesis}[SOT-Rate]
\label{hyp:sot-rate}
$\varepsilon_n(k) = O(c_k^{-1/4})$.
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
SOT-Rate $\Rightarrow$ SOT-Faster
$\Rightarrow$
Cor.~\ref{cor:convergence}.
SOT alone: $\varepsilon_n(k)\to 0$.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}
  -\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}
  {\Phi_m^{(c_k)}}|^2.
\end{equation}
Diagonal: $\le C_\kappa^2 c_k^{-1/2}$.
Small off-diagonal overlaps predict
$O(c_k^{-1/4})$;
Prob.~\ref{prob:off-diagonal}.
\end{remark}

\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
cannot determine the scaling of

\end{theorem}

$F_c$~\cite{Widom1964};
any algebraic rate is a global claim.
\end{proof}

%%══ 4. ABSTRACT STABILITY PRINCIPLE
\section{Abstract Stability Principle}
\label{sec:abstract}

\begin{assumption}[Spectral isolation,
  Assumption~S]
\label{ass:spectral-isolation}
Let $\{A_k\}$, $A$ bounded self-adjoint
on $\mathcal{H}$ with
$A_k\xrightarrow{\mathrm{SOT}} A$.
\begin{enumerate}[\rm(S1)]
\item\label{S1}
  $\lambda\in\sigma(A)$ isolated,
  simple; $P^\infty=\phi\otimes\phi$.
\item\label{S2}
  Each $\lambda_k\in\sigma(A_k)$
  isolated, simple;
  $\gamma_k:=\dist(\lambda_k,
  \sigma(A_k)\setminus\{\lambda_k\})
  >0$; $P_k$ rank-one.
\item\label{S3}
  $\lambda_k\to\lambda$.
\end{enumerate}
\end{assumption}

\begin{remark}[Spectral isolation,
  index stability, and SOT]
\label{rem:spectral-isolation}
Under strong resolvent convergence,
Assumption~S follows automatically
from isolation of
$\lambda$~\cite[Ch.~VIII]{Kato1966}.
Under mere SOT, eigenvalues can in
principle fail to be consistently
indexed; (S3) must be verified
explicitly.
In the PSWF application, strict ordering
of $\lambda_n^{(c)}$ for all $c$
(Lem.~\ref{lem:inherited}(A3))
ensures canonical assignment
$\lambda_n^{(c_k)}
\leftrightarrow\lambda_n^{(\infty)}$,
and Lem.~\ref{lem:eigenvalue-matching}
gives $\lambda_n^{(c_k)}
\to\lambda_n^{(\infty)}$.
In the abstract setting, (S3) is an
explicit axiom the user must verify.
\end{remark}

\begin{remark}[Role of SOT]
\label{rem:sot-role}
The only contribution of strong operator
convergence in this paper is the
pointwise vanishing of residuals:
\[
  (A_k-A)\phi\to 0
  \quad\text{for each fixed }\phi.
\]
No uniform control is used or required.
The spectral gap and isolation are
independent structural inputs,
not consequences of SOT.

The theorem asserts: this minimal
SOT output, when combined with a gap
structure, is sufficient to recover
spectral projection stability.
This is a non-trivial statement:
SOT alone is not sufficient
(Rem.~\ref{rem:strong-vs-norm},
Prop.~\ref{prop:no-go}),
and the gap cannot substitute for
the residual control
(Prop.~\ref{prop:counterexample}).

SOT is therefore not doing ``hidden
norm work''; it is the minimal
access condition that makes
$\|(A_k-A)\phi\|$ a well-defined,
vanishing observable.
\end{remark}

\begin{lemma}[Mechanism chain:
  rank-one + gap + residual]
\label{lem:mechanism-chain}
Under Assumption~S,
\eqref{eq:main-principle}
is the composition of three exact steps:
\begin{enumerate}[\rm(M1)]
\item\label{M1}
  \textbf{(Rank-one encoding.)}
  $\|P-Q\|=\|(I-Q)u\|$
  for rank-one projections
  $P=u\otimes u$, $Q=v\otimes v$
  \textup{\cite[Ch.~1]{Bhatia1997}}.
  Hence $\|P_k-P^\infty\|
  =\|(I-P_k)\phi\|$.
  The projection has a single degree
  of freedom.
\item\label{M2}
  \textbf{(Gap resolvent bound.)}
  $\|(A_k-\lambda_k)^{-1}
  |_{\mathrm{ran}(I-P_k)}\|
  \le 1/\gamma_k$.
  Since $(A_k-\lambda_k)\phi
  =(A_k-\lambda_k)(I-P_k)\phi$:
  $\|(I-P_k)\phi\|
  \le\|(A_k-\lambda_k)\phi\|/\gamma_k$.
\item\label{M3}
  \textbf{(Residual triangle.)}
  $\|(A_k-\lambda_k)\phi\|
  \le\|(A_k-A)\phi\|
  +|\lambda_k-\lambda|$,
  using $(A-\lambda)\phi=0$.
\end{enumerate}
Combined:
$\|P_k-P^\infty\|
\le(\|(A_k-A)\phi\|
+|\lambda_k-\lambda|)/\gamma_k\to 0$.
Removing any of the three conditions
breaks the chain
(Rems.~\ref{rem:rank-one-essential},
\ref{rem:necessity-uniform-gap};
Props.~\ref{prop:no-go},
\ref{prop:counterexample}).
\end{lemma}
\begin{proof}
(M1): \cite[Ch.~1]{Bhatia1997}.
(M2): Spectral theorem on
$\mathrm{ran}(I-P_k)$;
$(A_k-\lambda_k)P_k\phi=0$.
(M3): Triangle with $(A-\lambda)\phi=0$.
\end{proof}

\begin{remark}[Why the chain is exact]
\label{rem:chain-exact}
(M1): rank-one means there is
\emph{only one direction}, not that
one direction controls all.
(M2): a standard resolvent bound
using only $\gamma_k$, no reference
to $A_k-A$.
(M3): exact because
$(A-\lambda)\phi=0$ strictly, not
approximately.
The chain works because SOT provides
the residual, gap provides the
resolvent bound, and rank-one collapses
the projection norm to a scalar.
\end{remark}

\begin{remark}[Structural difference
  to Davis--Kahan]
\label{rem:input-comparison}
Davis--Kahan~\cite{DavisKahan1970}
is driven by $\|A_k-A\|$, which is
invariant under redistributions of
spectral mass across all directions.

The present principle depends instead
on $\|(A_k-A)\phi\|$, which is
\emph{direction-sensitive}: it measures
how the perturbation interacts with
the specific direction $\phi$,
and is not controlled by $\|A_k-A\|$
in either direction.

Prop.~\ref{prop:separation} exhibits
families with $\|A_k-A\|=1$
but $\|(A_k-A)\phi\|\to 0$,
showing the hypotheses are not
reducible to each other.

The structural difference is therefore:
Davis--Kahan is invariant under
global spectral redistributions of
the perturbation;
the present principle is
intrinsically direction-sensitive.
This is not a re-labelling but a
strictly different hypothesis
structure, applicable precisely where
Davis--Kahan gives no information.
\end{remark}

\begin{lemma}[Davis--Kahan implies
  ratio condition]
\label{lem:dk-comparison}
Under Assumption~S, if
$\|A_k-A\|\to 0$ and
$\gamma_k\ge c_0>0$:
$\|A_k-A\|\to 0
\Rightarrow\eqref{eq:main-principle}
\Rightarrow\|P_k-P^\infty\|\to 0$;
neither implication reverses
(Props.~\ref{prop:separation},
\ref{prop:counterexample}).
\end{lemma}
\begin{proof}
$\|(A_k-A)\phi\|\le\|A_k-A\|\to 0$;
$|\lambda_k-\lambda|\le\|A_k-A\|\to 0$;
divide by $c_0$.
\end{proof}

\begin{proposition}[Separation:
  direction-sensitive perturbation,
  $\|A_k-A\|\equiv 1$]
\label{prop:separation}
There exist bounded self-adjoint
$\{A_k\}$, $A$ on $\ell^2$
satisfying Assumption~S with
$\gamma_k\ge c_0>0$, such that:
\begin{enumerate}[\rm(i)]
\item $\|A_k-A\|=1$ for all $k$;
\item $\|(A_k-A)\phi\|
  =\varepsilon_k\to 0$;
\item $A_k-A$ has a genuine
  off-diagonal spectral interaction
  component;
\item the large-norm component is
  asymptotically spectrally orthogonal
  to the active subspace
  $\mathrm{span}\{\phi\}$;
\item $\|P_k-P^\infty\|\to 0$.
\end{enumerate}
\end{proposition}
\begin{proof}
Set $\phi=e_1$, $\lambda=0$,
$A:=\sum_{n\ge 2}\mu_n(e_n\otimes e_n)$,
$\mu_n\in[-1,-c_0]$.
Define $A_k-A:=S_k+K_k$ where
$S_k:=\varepsilon_k
(e_2\otimes e_1+e_1\otimes e_2)$,
$\|S_k\|=\varepsilon_k\to 0$,
and $K_k$ acts on
$\mathrm{span}\{e_3,e_4,\ldots\}$
with $\|K_k\|=1$ and spectrum of
$A|_{\mathrm{span}\{e_n,n\ge 3\}}+K_k$
uniformly away from $0$.

\textit{(i)}: $\|A_k-A\|\ge\|K_k\|=1$.
\textit{(iii)}: $S_k$ couples
$e_1,e_2$ off-diagonally; genuine
eigenvalue splitting and rotation.
\textit{(iv)}: $K_ke_1=0$,
so $K_k$ is spectrally orthogonal to
$\phi=e_1$.
\textit{(ii)}: $(A_k-A)\phi
=\varepsilon_k e_2$;
$\|(A_k-A)\phi\|=\varepsilon_k\to 0$.
\textit{Gap}: $\gamma_k\ge c_0/2$
for large $k$~\cite[Ch.~II]{Kato1966}.
\textit{(v)}: Thm.~\ref{thm:abstract}.
\end{proof}

\begin{remark}[Direction-sensitivity
  is the principle, not a defect]
\label{rem:block-is-the-point}
The large component $K_k$ acts on
$\phi^\perp$ --- is this hidden block
structure? Yes, and this is exactly
the content of the principle.

Step (M1) says the projection
$P^\infty=\phi\otimes\phi$ has a
single degree of freedom, determined
by $\phi$. Any perturbation component
spectrally orthogonal to
$\mathrm{span}\{\phi\}$ is invisible
to $P^\infty$, regardless of its norm.

Davis--Kahan cannot exploit this:
it sees only $\|A_k-A\|\equiv 1$ and
gives no information. The present
principle identifies the geometrically
relevant quantity: the perturbation
restricted to the active direction.

The off-diagonal $S_k$ confirms the
perturbation genuinely touches the
spectral mechanism (it mixes $e_1$
and $e_2$), yet is controlled by
$\varepsilon_k$. The direction-sensitivity
is the mechanism, not an artifact.
\end{remark}

\begin{theorem}[Main Principle;
  sufficient condition]
\label{thm:abstract}
Under Assumption~S,
\eqref{eq:main-principle}
$\Rightarrow$
$\|P_k-P^\infty\|\to 0$.
\end{theorem}
\begin{proof}
Lem.~\ref{lem:mechanism-chain},
steps (M3), (M2), (M1).
\end{proof}

\begin{remark}[Non-circularity]
\label{rem:non-circularity}
$\phi$ is a fixed eigenvector of $A$,
not of $A_k$. No spectral data of
$A_k$ enters the construction of $\phi$.
\end{remark}

\begin{remark}[Structural independence]
\label{rem:structural-independence}
All $A_k$-quantities appear only as
$(A_k-A)\phi$ or $(A_k-\lambda_k)\phi$.
\end{remark}

\begin{lemma}[Rank-one projection
  $\Rightarrow$ eigenvector convergence]
\label{lem:rank-one-proj}
If $P_k=v_k\otimes v_k$,
$P^\infty=\phi\otimes\phi$,
$\|P_k-P^\infty\|\to 0$, then
$\exists$ phases $\theta_k$:
$\|v_k-e^{i\theta_k}\phi\|\to 0$.
\end{lemma}
\begin{proof}
Standard rank-one estimate.
\end{proof}

\begin{theorem}[Necessary condition;
  bounded-gap case]
\label{thm:necessary-bounded-gap}
Under Assumption~S, if
$\|P_k-P^\infty\|\to 0$ and
$\gamma_k\ge c_0>0$,
then \eqref{eq:main-principle} holds.
\end{theorem}
\begin{proof}
\textbf{Eigenvalue:}
$|\lambda_k-\lambda|/c_0\to 0$.

\textbf{Residual:}
$(A_k-A)\phi
=(A_k-A)P_k\phi
+(A_k-A)(I-P_k)\phi$.
Term~(II):
$\le 2M\|(I-P_k)\phi\|\to 0$.
Term~(I): $\alpha_k(\lambda_kv_k-Av_k)$;
$|\lambda_k-\lambda|\to 0$;
$(A-\lambda)v_k\to 0$ by
Lem.~\ref{lem:rank-one-proj}.
Divide by $c_0$.
\end{proof}

\begin{remark}[Necessity requires
  uniform gap]
\label{rem:necessity-uniform-gap}
Without $\gamma_k\ge c_0$, the ratio
may diverge even when
$\|P_k-P^\infty\|\to 0$.
Open Prob.~\ref{prob:full-iff}.
\end{remark}

\begin{remark}[Rank-one essential
  for necessity]
\label{rem:rank-one-essential}
For higher-multiplicity eigenspaces,
additional assumptions are required
to extract eigenvector convergence
from projection convergence.
\end{remark}

\begin{remark}[Novelty: direction-sensitive
  perturbation principle in SOT regime]
\label{rem:abstract-novelty}
Classical theory
(Davis--Kahan~\cite{DavisKahan1970},
Kato~\cite{Kato1966},
Chatelin~\cite{Chatelin1983},
Osborn~\cite{Osborn1975})
requires $\|A_k-A\|\to 0$ or
norm-resolvent convergence.
The present result operates under
SOT only, with a direction-sensitive
hypothesis $\|(A_k-A)\phi\|/\gamma_k\to 0$
that is not controlled by global norm.

This goes beyond global-norm-based
perturbation theory not by strengthening
the output, but by replacing a
rotation-invariant global input with
a direction-sensitive local one.
The replacement is enabled precisely
by rank-one structure, which collapses
the projection norm to a single scalar.
\end{remark}

\begin{proposition}[Counterexample:
  ratio blow-up implies projection
  failure]
\label{prop:counterexample}
There exist $\{A_k\}$, $A$
satisfying Assumption~S with
$A_k\xrightarrow{\mathrm{SOT}} A$,
$\lambda_k\to\lambda$, such that
$r_k:=\|(A_k-A)\phi\|/\gamma_k
\equiv 1/\sqrt{2}$
and
$\|P_k-P^\infty\|\ge 1/2
=(1/\sqrt{2})\cdot r_k$.
\end{proposition}
\begin{proof}
$\phi=e_1$, $\lambda=0$,
$\gamma_k:=k^{-1/2}$,
$v_k:=\tfrac{1}{\sqrt{2}}(e_1+e_2)$,
$A:=\sum_{n\ge 2}n^{-1}(e_n\otimes e_n)$,
$A_k:=\gamma_k(v_k\otimes v_k)
+\sum_{n\ge 2}n^{-1}(e_n\otimes e_n)$.
$(A_k-A)e_1
=\tfrac{\gamma_k}{\sqrt{2}}v_k
=\tfrac{\gamma_k}{2}(e_1+e_2)$;
$\|(A_k-A)\phi\|=\gamma_k/\sqrt{2}$;
$r_k=1/\sqrt{2}$.
$P_k=v_k\otimes v_k$;
$\|P_k-P^\infty\|
\ge|\tfrac{1}{2}-1|=\tfrac{1}{2}$.
\end{proof}

\begin{remark}[Structural necessity
  of the ratio]
\label{rem:counterexample-meaning}
By (M1): $\|P_k-P^\infty\|
=\|(I-P_k)\phi\|$.
If $r_k\ge c>0$, then
$\|(I-P_k)\phi\|\asymp r_k$ is tight;
ratio blow-up is structurally necessary
for projection failure.
\end{remark}

%%══ 5. PSWF APPLICATION
\section{PSWF Application:
  Conditional Corollaries}
\label{sec:main}

\begin{center}
\fbox{\parbox{0.88\linewidth}{%
\centering\itshape
The following are
\textbf{conditional corollaries},
not theorems. They assume Gap-S
(Hyp.~\ref{hyp:gap-structural})
and SOT-Faster
(Hyp.~\ref{hyp:sot-faster}),
both of which are open
(Probs.~\ref{prob:sharp-gc},
\ref{prob:sot-faster}).
The contribution of this section is a
complete reduction of PSWF projection
stability to these two independent
analytic hypotheses.
No further PSWF-specific input beyond
Lem.~\ref{lem:inherited} is needed.}}
\end{center}

\medskip
Assumption~S is verified by
Lem.~\ref{lem:inherited}(A3)
and Lem.~\ref{lem:eigenvalue-matching}.

\begin{lemma}[Gap-S supplies gap bound]
\label{lem:gap-bridge}
Under Hyp.~\ref{hyp:gap-structural},
$\gamma_k\ge g(c_k)$
for each $n\in W_{c_k}$.
\end{lemma}
\begin{proof}
Strict ordering gives
$\gamma_k\ge g(c_k)$.
\end{proof}

\begin{corollary}[Projection convergence,
  conditional]
\label{cor:convergence}
\textup{[Conditional on Gap-S and
SOT-Faster.]}
Under Lem.~\ref{lem:inherited},
Hyps.~\ref{hyp:gap-structural},
\ref{hyp:sot-faster},
for fixed $n\in W_{c_k}$:
$\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$.
In particular,
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
strongly (up to phase).
\end{corollary}

\begin{corollary}[Rate, conditional]
\label{cor:rate}
\textup{[Conditional on Gap-S and
SOT-Rate.]}
Under Hyp.~\ref{hyp:sot-rate},
for $n\le\kappa c_k$, $n\in W_{c_k}$,
uniformly in $n$:
\[
  \|P_n^{(c_k)}-P_n^{(\infty)}\|
  \le
  \frac{C\,C_\kappa\,c_k^{-1/4}}
  {g(c_k)}.
\]
If $g(c)\sim(\log c)^{-1}$:
rate $O(c_k^{-1/4}\log c_k)$.
\end{corollary}

\begin{remark}[Phase convention]
Phase fixed by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
for large $k$.
\end{remark}

%%══ 6. PROOFS
\section{Proofs of Corollaries}
\label{sec:proj}

\begin{lemma}[Approx.\ eigenvector
  bound]
\label{lem:approx-eig}
Let $A$ bounded self-adjoint,
$\lambda$ isolated, simple, gap
$\gamma>0$, $P$ rank-one,
$\|\phi\|=1$,
$\|(A-\lambda)\phi\|\le\varepsilon$:
$\|(I-P)\phi\|\le\varepsilon/\gamma$.
\end{lemma}
\begin{proof}
$(A-\lambda)\phi=(A-\lambda)(I-P)\phi$;
spectral theorem.
\end{proof}

\begin{proof}[Proof of
  Cor.~\ref{cor:convergence}]
$\tilde\varepsilon_n(k)
:=\|(\mathcal{H}_{c_k}
-\lambda_n^{(c_k)})
\Phi_n^{(\infty)}\|
\le\varepsilon_n(k)+C_\kappa c_k^{-1/4}$.
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}
  {g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to 0
    \,(\text{SOT-Faster})}
  +\underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to 0
    \,(\text{Rem.}\,
    \ref{rem:gap-implies-o})}
  \to 0.
\end{equation}
Lem.~\ref{lem:approx-eig};
(M1) completes.
\end{proof}

\begin{proof}[Proof of
  Cor.~\ref{cor:rate}]
$\tilde\varepsilon_n(k)
\le(C'+C_\kappa)c_k^{-1/4}$;
divide by $g(c_k)$.
\end{proof}

\begin{remark}[No accumulation]
\label{rem:nacc}
Gap-S implies finitely many
$\lambda_n^{(\infty)}$
in any subinterval of $(0,1)$.
\end{remark}

%%══ 10. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\noindent\textbf{Conjecture-level:}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove Gap-S with
$g(c)\sim(\log c)^{-1}$
via sine-kernel repulsion.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$.
\end{problem}

\begin{problem}[Full iff,
  vanishing gap]
\label{prob:full-iff}
Characterize
$\|P_k-P^\infty\|\to 0$
$\Leftrightarrow$
$\|(A_k-A)\phi\|/\gamma_k\to 0$
when $\gamma_k\to 0$.
\end{problem}

\noindent\textbf{Technical-level:}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
$\varepsilon_n(k)=O(c_k^{-1/4})$.
\end{problem}

\begin{problem}[Off-diagonal overlaps]
\label{prob:off-diagonal}
Bound
$|\ip{\Phi_n^{(\infty)}}
{\Phi_m^{(c_k)}}|$
for $m\ne n$.
\end{problem}

\begin{problem}[Strong resolvent
  convergence]
\label{prob:src}
Does $\mathcal{H}_{c_k}$ converge
in strong resolvent sense?
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII.
\end{problem}
