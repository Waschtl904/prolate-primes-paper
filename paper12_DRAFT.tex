%%══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context and goal}

This paper occupies a specific position in the series:
it provides the bridge between
the strong operator convergence of Paper~X
and the completeness result of Paper~XIII.

Paper~X establishes that the PSWF concentration
operators $\mathcal{H}_{c_k}$ converge in the
strong operator topology (SOT) to a bounded
self-adjoint limit $\Hlim$.
Paper~XI constructs a diagonal subsequence
along which the eigenfunctions $\Phi_n^{(c_k)}$
converge weakly to limit vectors $\Phi_n^{(\infty)}$,
orthonormal and satisfying
$\Hlim\Phi_n^{(\infty)} = \lambda_n^{(\infty)}\Phi_n^{(\infty)}$.

\textbf{The gap that this paper fills.}
SOT convergence does \emph{not} imply norm
convergence of spectral projections
(Rem.~\ref{rem:strong-vs-norm}).
The central result of this paper
(Thm.~\ref{thm:convergence}) establishes:
\[
  \|\Pnck - \Pninf\|\to 0,
\]
under two well-motivated conditions:
\begin{itemize}
\item \hyperref[hyp:gap-structural]{Gap-S}:
  spectral gaps of $\mathcal{H}_{c_k}$
  dominate the eigenvalue convergence rate
  (precisely: $\lim_{c\to\infty}
  c^{-1/4}/g(c)=0$);
\item \hyperref[hyp:sot-faster]{SOT-Faster}:
  $\|(\mathcal{H}_{c_k}-\Hlim)\Phi_n^{(\infty)}\|
  =o(g(c_k))$.
\end{itemize}
Neither condition requires any bound on
$\|\mathcal{H}_{c_k}-\Hlim\|$.

\subsection{Main results}

\begin{itemize}
\item \textbf{Theorem~\ref{thm:convergence}
  (Convergence):}
  Under Gap-S and SOT-Faster,
  $\|\Pnck - \Pninf\|\to 0$.

\item \textbf{Theorem~\ref{thm:rate}
  (Rate):}
  Under the additional
  \hyperref[hyp:sot-rate]{Hyp.~SOT-Rate},
  $\|\Pnck-\Pninf\|
  \le C C_\kappa c_k^{-1/4}/g(c_k)$,
  uniformly in $n\le\kappa c_k$.
\end{itemize}

\begin{remark}[Core contribution]
\label{rem:conditions}
SOT-Faster captures \emph{exactly}
the missing quantitative link between
operator convergence and spectral isolation:
\begin{quote}
  \textit{The entire difficulty of upgrading
  SOT convergence to norm convergence of
  spectral projections is captured by
  Hypothesis~SOT-Faster.}
\end{quote}
This paper's contribution is the
reduction of projection stability to
the two isolated, well-motivated conditions
Gap-S and SOT-Faster.
Sec.~\ref{sec:heuristic} provides
heuristic motivation for both.
\end{remark}

\subsection{What this paper does \emph{not} claim}
\begin{itemize}
\item \textbf{Riesz-basis property}
  (Cor.~\ref{cor:riesz}): trivial from
  orthonormality in Paper~XI.
\item \textbf{Completeness}
  $\mathcal{K}=L^2(\mathbb{R})$: Paper~XIII.
\item \textbf{Proof of Gap-S}:
  Open Problem~\ref{prob:sharp-gc}.
\item \textbf{Proof of SOT-Faster
  or SOT-Rate}:
  Open Problems~\ref{prob:sot-faster},
  \ref{prob:sot-rate}.
\end{itemize}

\section{Input from Papers VIII, X, XI}
\label{sec:input}

\begin{lemma}[Inherited data]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item\label{A1}
  \emph{(VIII, Cor.~4.3)}
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}$
  for $n\le\kappa c$, uniform in $n$.
\item\label{A2}
  \emph{(X, Thread~A)}
  $\mathcal{H}_{c_k}\xrightarrow{\mathrm{SOT}}\Hlim$,
  bounded self-adjoint:
  \begin{equation}\label{eq:sot}
    \|(\mathcal{H}_{c_k}-\Hlim)f\|\to 0
    \quad\forall\, f\in L^2.
  \end{equation}
  Paper~X provides no rate on \eqref{eq:sot}.
\item\label{A3}
  \emph{(XI, Lem.~3.1 + Thm.~5.1)}
  There exists $c_k\nearrow\infty$:
  $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}=\delta_{mn}$,
  $\Hlim\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$,
  $\lambda_n^{(c)}$ simple for all $n$, $c$.
\end{enumerate}
\end{lemma}

\begin{remark}[SOT does not imply
  projection norm convergence]
\label{rem:strong-vs-norm}
By~\cite[Ex.~VIII.7.14]{ReedSimon1975},
SOT convergence does not in general imply
norm convergence of spectral projections.
Gap-S and SOT-Faster supply the
additional structure that closes this gap.
\end{remark}

%%══ 3. HYPOTHESES
\section{Hypotheses and Motivation}
\label{sec:gap}

\begin{hypothesis}[Structural gap, Gap-S]
\label{hyp:gap-structural}
There exists $g\colon[1,\infty)\to(0,\infty)$
with
\begin{equation}\label{eq:gap-dominates}
  g(c)\to 0
  \quad\text{and}\quad
  \lim_{c\to\infty}\frac{c^{-1/4}}{g(c)} = 0,
\end{equation}
such that for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|\ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S implies $c^{-1/4}=o(g(c))$]
\label{rem:gap-implies-o}
Condition \eqref{eq:gap-dominates} states
precisely that $c^{-1/4}=o(g(c))$
as $c\to\infty$.
This is used explicitly in the proof of
Thm.~\ref{thm:convergence}
(eq.~\eqref{eq:ratio-to-zero})
to ensure $C_\kappa c_k^{-1/4}/g(c_k)\to 0$.
\end{remark}

\begin{hypothesis}[SOT faster than gap,
  SOT-Faster]
\label{hyp:sot-faster}
For each fixed $n$, along the diagonal
subsequence of Lem.~\ref{lem:inherited}:
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k))
  \quad(k\to\infty).
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster is the core difficulty]
\label{rem:sot-faster-core}
SOT (Paper~X) gives $\varepsilon_n(k)\to 0$,
but no comparison with $g(c_k)$.
SOT-Faster requires precisely:
\[
  \varepsilon_n(k) = o(g(c_k)),
\]
i.e., the operator residual on
$\Phi_n^{(\infty)}$ decays faster than
the spectral gap.
This is \emph{not} a technical side condition:
it encapsulates the entire difficulty of
upgrading SOT convergence to norm
convergence of rank-one spectral projections.
Proving it rigorously is
Open Problem~\ref{prob:sot-faster}.
\end{remark}

\begin{hypothesis}[SOT rate on eigenvectors,
  SOT-Rate]
\label{hyp:sot-rate}
For each fixed $n$:
\begin{equation}\label{eq:sot-rate}
  \varepsilon_n(k)
  = O(c_k^{-1/4}).
\end{equation}
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
\[
  \text{SOT-Rate}
  \underset{\eqref{eq:gap-dominates}}{\Longrightarrow}
  \text{SOT-Faster}
  \Longrightarrow
  \|\Pnck-\Pninf\|\to 0.
\]
The implication SOT-Rate $\Rightarrow$
SOT-Faster follows from
$c_k^{-1/4}=o(g(c_k))$
(Rem.~\ref{rem:gap-implies-o}).
SOT alone gives only $\varepsilon_n(k)\to 0$,
which is insufficient without comparison
to $g(c_k)$.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding $\Phi_n^{(\infty)}$ in the
eigenbasis $\{\Phi_m^{(c_k)}\}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|^2.
\end{equation}
Diagonal term $m=n$:
$\le C_\kappa^2 c_k^{-1/2}$ by \eqref{A1}.
Off-diagonal $m\ne n$: controlled if
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|$
are small, plausible from Gap-S
(spectral isolation concentrates mass
on the diagonal).
Making this rigorous requires
off-diagonal overlap estimates;
see Open Problem~\ref{prob:sot-rate}.
\end{remark}

\subsection{WKB layer}
\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
The WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
does not determine a canonical scaling
for $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
without global kernel information.
\end{theorem}
\begin{proof}
The transfer map $F_c$ has a sigmoid
profile near $\lambda=1/2$~\cite{Widom1964};
any rate $F_c'\asymp c^{-\beta}$ is a
global claim not derivable from
$\Lc$ alone.
\end{proof}

%%══ 4. MAIN THEOREMS
\section{Main Theorems}
\label{sec:main}

\begin{theorem}[Spectral projection
  convergence]
\label{thm:convergence}
Assume Lem.~\ref{lem:inherited},
\hyperref[hyp:gap-structural]{Hyp.~Gap-S},
and \hyperref[hyp:sot-faster]{Hyp.~SOT-Faster}.
For each fixed $n$:
\begin{equation}\label{eq:convergence}
  \|\Pnck - \Pninf\|\to 0
  \quad (k\to\infty).
\end{equation}
In particular:
\begin{enumerate}[\rm(i)]
\item $\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
  strongly, up to phase
  (Rem.~\ref{rem:phase}).
\item $\{\lambda_n^{(\infty)}\}$ has no
  accumulation point in $(0,1)$
  (Cor.~\ref{cor:nacc}).
\end{enumerate}
\end{theorem}

\begin{theorem}[Spectral projection
  stability with explicit rate]
\label{thm:rate}
Assume additionally
\hyperref[hyp:sot-rate]{Hyp.~SOT-Rate}.
For all $n\le\kappa c_k$,
uniformly in $n$:
\begin{equation}\label{eq:rate-bound}
  \|\Pnck - \Pninf\|
  \le \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}
  \to 0,
\end{equation}
where $C>0$ is universal.
\end{theorem}

\begin{remark}[Phase convention]
\label{rem:phase}
For rank-one projections,
$\|\Pnck-\Pninf\|\to 0$ implies
$\Phi_n^{(c_k)}\to e^{i\theta_n^{(k)}}
\Phi_n^{(\infty)}$ strongly.
Fix phase by
$\ip{\Phi_n^{(c_k)}}{\Phi_n^{(\infty)}}>0$
(valid for large $k$).
\end{remark}

\begin{remark}[Structure of the proof]
\label{rem:why-nontrivial}
Three inputs combine:
\begin{enumerate}
\item SOT + SOT-Faster: $\Phi_n^{(\infty)}$
  is a \emph{controlled} approximate
  eigenvector of $\mathcal{H}_{c_k}$
  (residual $o(g(c_k))$).
\item Gap-S: spectral isolation converts
  the residual bound into projection
  convergence via Lem.~\ref{lem:approx-eig}.
\item Rank-one structure closes the
  argument at the eigenfunction level.
\end{enumerate}
\end{remark}

%%══ 5. PROOF
\section{Proof of the Main Theorems}
\label{sec:proj}

\begin{lemma}[Approximate eigenvector
  $\Rightarrow$ projection estimate;
  cf.\ \cite{DavisKahan1970,Kato1966}]
\label{lem:approx-eig}
Let $A$ be a bounded self-adjoint operator,
$\lambda$ a simple eigenvalue of $A$ with gap
$\gamma:=\mathrm{dist}(\lambda,
\spc{A}\setminus\{\lambda\})>0$.
Let $\phi\in\mathcal{H}$, $\|\phi\|=1$,
$\|(A-\lambda)\phi\|\le\varepsilon$.
Then:
\begin{equation}\label{eq:approx-eig-bound}
  \|(I-P_A(\lambda))\phi\|\le
  \frac{\varepsilon}{\gamma}.
\end{equation}
\end{lemma}
\begin{proof}
Write $\phi = P_A\phi + (I-P_A)\phi$.
Since $(A-\lambda)P_A=0$:
$(A-\lambda)\phi = (A-\lambda)(I-P_A)\phi$.
On $\mathrm{ran}(I-P_A)$, the spectrum
of $A$ lies at distance $\ge\gamma$
from $\lambda$, so
$\|(A-\lambda)^{-1}\|\le 1/\gamma$ there.
Hence
$\|(I-P_A)\phi\|\le
\gamma^{-1}\|(A-\lambda)\phi\|
\le\varepsilon/\gamma$.
\end{proof}

\begin{lemma}[SOT gives approximate
  eigenvectors (no rate)]
\label{lem:sot-approx}
Under \eqref{A2}--\eqref{A3},
for each fixed $n$:
\begin{equation}\label{eq:approx-sot}
  \varepsilon_n(k) =
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|\to 0.
\end{equation}
No rate on $\varepsilon_n(k)$ is claimed.
\end{lemma}
\begin{proof}
$\varepsilon_n(k)
= \|(\mathcal{H}_{c_k}-\Hlim)
\Phi_n^{(\infty)}\|\to 0$
by SOT~\eqref{eq:sot} with
$f=\Phi_n^{(\infty)}$ fixed.
\end{proof}

\begin{proof}[Proof of
  Thm.~\ref{thm:convergence}]
Fix $n$.
By Lem.~\ref{lem:sot-approx} and \eqref{A1}:
\begin{equation}\label{eq:eps-tilde}
  \tilde\varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
  \Phi_n^{(\infty)}\|
  \le \varepsilon_n(k)
  + |\lambda_n^{(\infty)}-\lambda_n^{(c_k)}|
  \to 0.
\end{equation}
By SOT-Faster \eqref{eq:sot-faster}
and Rem.~\ref{rem:gap-implies-o}:
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to\,0
    \text{ (SOT-Faster)}}
  +
  \underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to\,0
    \text{ (Rem.~\ref{rem:gap-implies-o})}}
  \to 0.
\end{equation}
Apply Lem.~\ref{lem:approx-eig} with
$A=\mathcal{H}_{c_k}$,
$\lambda=\lambda_n^{(c_k)}$ (simple,
gap $\ge g(c_k)$ by Gap-S),
$\phi=\Phi_n^{(\infty)}$,
$\varepsilon=\tilde\varepsilon_n(k)$:
\begin{equation}\label{eq:proj-dev}
  \|(I-\Pnck)\Phi_n^{(\infty)}\|
  \le\frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \to 0.
\end{equation}
Write $\Pnck\Phi_n^{(\infty)}
=\alpha_k\Phi_n^{(c_k)}$ with
$\alpha_k=\ip{\Phi_n^{(\infty)}}{\Phi_n^{(c_k)}}$;
then $|\alpha_k|^2=1-\|(I-\Pnck)
\Phi_n^{(\infty)}\|^2\to 1$.
Rank-one projection identity
($\|P_u-P_v\|\le 2\|u-e^{i\theta}v\|$):
\[
  \|\Pnck-\Pninf\|
  \le 2\|\Phi_n^{(c_k)}
  -e^{i\arg\alpha_k}\Phi_n^{(\infty)}\|
  \to 0.
\]
\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:rate}]
Under SOT-Rate \eqref{eq:sot-rate}:
$\varepsilon_n(k)\le C' c_k^{-1/4}$,
so $\tilde\varepsilon_n(k)
\le(C'+C_\kappa)c_k^{-1/4}$.
With $\gamma=g(c_k)$, Lem.~\ref{lem:approx-eig}
and the rank-one identity yield
\eqref{eq:rate-bound}
uniformly for $n\le\kappa c_k$.
\end{proof}

\begin{corollary}[No accumulation
  of $\lambda_n^{(\infty)}$]
\label{cor:nacc}
Under Gap-S and \eqref{A1},
$\{\lambda_n^{(\infty)}\}$ has no
accumulation point in $(0,1)$.
\end{corollary}
\begin{proof}
The eigenvalues $\lambda_n^{(c)}$ are
strictly ordered in $n$~\cite{Slepian1961}.
Fix $\mu\in(0,1)$, $r>0$;
choose $k_0$ with
$C_\kappa c_{k_0}^{-1/4}<r/2$.
Then $\lambda_n^{(\infty)}\in(\mu-r,\mu+r)$
implies $\lambda_n^{(c_{k_0})}
\in(\mu-3r/2,\mu+3r/2)$.
By Gap-S, at most
$\lceil 3r/g(c_{k_0})\rceil<\infty$
eigenvalues lie in this interval.
\end{proof}

\begin{proof}

%%══ 7. CANONICALIZATION
\section{Canonicalization (Conditional)}
\label{sec:canonical}

\begin{lemma}[Same $\mathcal{K}$,
  conditional]
\label{lem:same-K}
Assume:
\begin{hypothesis}[Full family convergence]
\label{hyp:full-family}
The spectral measures of $\mathcal{H}_{c_k}$
converge weak-$*$ to those of $\Hlim$.
\end{hypothesis}
Then any two diagonal subsequences
satisfying Gap-S yield the same
closed span $\mathcal{K}$.
\end{lemma}
\begin{proof}
Both subspaces are closed
$\Hlim$-invariant; under
Hyp.~\ref{hyp:full-family}
they generate the same spectral measure.
\end{proof}

\begin{theorem}[Canonicalization,
  conditional]
\label{thm:canonical}
Under Gap-S and Hyp.~\ref{hyp:full-family},
$\Hlim$ is canonical on $\mathcal{K}$
up to unitary equivalence.
\end{theorem}

%%══ 8. HEURISTIC
\section{Heuristic Motivation of Gap-S
  and SOT-Faster}
\label{sec:heuristic}

\textbf{Non-rigorous throughout.}

\subsection{Sigmoid transition (Widom)}

Widom-type asymptotics~\cite{Widom1964}
suggest the plausibility of a sigmoidal
profile:
\[
  \lambda_n^{(c)}\approx
  \tfrac{1}{1+e^{\pi(n-2c/\pi)/\sigma(c)}},
  \quad\sigma(c)\asymp\log c.
\]
This is heuristically consistent with,
but does not prove, Gap-S or SOT-Faster.

\subsection{Gap estimate}

If $\rho(c)\asymp\log c$ eigenvalues
occupy the transition window,
average spacing $\sim(\log c)^{-1}
\gg c^{-1/4}$,
suggesting the plausibility of Gap-S.
Rigorous proof requires
Plancherel--Rotach or determinantal
kernel methods.

\subsection{SOT-Faster plausibility}

From \eqref{eq:formal-expansion},
if off-diagonal overlaps
$|\ip{\Phi_n^{(\infty)}}{\Phi_m^{(c_k)}}|
=o(1)$ for $m\ne n$ (plausible from Gap-S),
then \eqref{eq:formal-expansion}
is dominated by $m=n$,
giving $\varepsilon_n(k)=O(c_k^{-1/4})$
(SOT-Rate), which implies SOT-Faster.

\begin{remark}[Plausibility $\ne$ proof]
This section suggests plausibility;
it does not constitute a rigorous argument.
The open problems remain
\ref{prob:sharp-gc}--\ref{prob:sot-faster}.
\end{remark}

\section{Open Problems}
\label{sec:open}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Determine $g(c)$ rigorously;
expected $g(c)\sim(\log c)^{-1}$.
\end{problem}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
Prove $\varepsilon_n(k)=O(c_k^{-1/4})$;
would imply SOT-Faster and justify
the rate in Thm.~\ref{thm:rate}.
\end{problem}

\begin{problem}[SOT-Faster]
\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$;
would make Thm.~\ref{thm:convergence}
fully unconditional.
\end{problem}

\begin{problem}[Full family convergence]
\label{prob:full-family}
Establish Hyp.~\ref{hyp:full-family};
would make Lem.~\ref{lem:same-K}
unconditional.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K}=L^2(\mathbb{R})$?
Paper~XIII.
\end{problem}
