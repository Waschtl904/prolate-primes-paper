\begin{abstract}
We establish the Riesz-basis criterion for the limiting
orthonormal system $\{\Phi_n^{(\infty)}\}$ constructed in
Paper~XI along the diagonal subsequence $c_k$.
The central equivalence is:
\[
  \{\Phi_n^{(\infty)}\} \text{ is a Riesz basis}
  \;\Longleftrightarrow\;
  \text{Gap} + \text{Stability} + \text{Nondegeneracy}.
\]
The hard core is Section~\ref{sec:gap}: spectral separation
in the transition window $W_c$.
The main result of that section is\,---\,contrary to what
might be expected\,---\,not a sharp gap exponent,
but rather a \emph{gap geometry theorem}
(Thm.~\ref{thm:wkb-indeterminacy}):
the WKB/Airy analysis establishes a positive separation
scale $g(c)\to 0$ with $g(c)\gg c^{-1/4}$,
but the precise power of $c$ is not determinable
from local WKB geometry alone\,---\,it depends on
the global spectral transfer $F_c\colon\chi\mapsto\lambda$
between the differential and integral PSWF operators.
This structural observation is recorded as
Thm.~\ref{thm:wkb-indeterminacy}; it demotes the
sharp gap hypothesis to an open problem
(Prob.~\ref{prob:sharp-gc}) and replaces it
with a weaker but provable separation condition
(\hyperref[hyp:gap-structural]{Hyp.~Gap-S})
that is already sufficient for the Riesz-basis conclusion.
All other sections are standard functional-analytic machinery.
\end{abstract}

\section{Main Statement}
\label{sec:main}
%%══════════════════════════════════════════════════════════════════════

\begin{theorem}[Riesz-basis criterion]
\label{thm:riesz-criterion}
Let $\{\Phi_n^{(\infty)}, \lambda_n^{(\infty)}\}$ be the
orthonormal system of Paper~XI
(Lem.~\ref{lem:XI-diagonal}), defined along the fixed
diagonal subsequence $c_k$.
Let $\mathcal{K} := \overline{\mathrm{span}}\,
\{\Phi_n^{(\infty)} : n\ge 0\}\subset L^2(\mathbb{R})$.
The following are equivalent:
\begin{enumerate}[\rm(i)]
\item $\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$.
\item The Gram matrix
      $\Gram_{mn} := \ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}$
      defines a bounded invertible operator on $\ell^2$.
\item \textbf{(Structural gap condition, Sec.~\ref{sec:gap}):}
      there exists a scale $g(c)\to 0$ with
      $g(c)\gg c^{-1/4}$ such that
      $|\lambda_n^{(c)} - \lambda_{n+1}^{(c)}|
      \ge g(c)$
      for all $n\in W_c$
      (\hyperref[hyp:gap-structural]{Hyp.~Gap-S}).
\item \textbf{(Stability condition):}
      the family $\{\Phi_n^{(c_k)}\}$ is uniformly
      Riesz-stable along $c_k$
      (Def.~\ref{def:riesz-stable}).
\end{enumerate}
(i)$\Leftrightarrow$(ii): Bari--Krein theorem
(Sec.~\ref{sec:bari-krein}).\\
(ii)$\Leftrightarrow$(iii)+(iv): Gram-matrix control
(Sec.~\ref{sec:gram}).\\
(iii): established in Section~\ref{sec:gap};
sharp form of $g(c)$ is Open Problem~\ref{prob:sharp-gc}.
\end{theorem}

\begin{remark}[Why $g(c)\gg c^{-1/4}$ suffices]
\label{rem:sufficiency-main}
The eigenvalue approximation rate from Paper~XI is
$|\lambda_n^{(c)}-\lambda_n^{(\infty)}|\le C_\kappa c^{-1/4}$.
For the Gram-matrix argument to work, one needs only
that consecutive eigenvalues of $\mathcal{H}_c$
cannot collapse faster than this rate.
Any $g(c)\gg c^{-1/4}$ is therefore sufficient;
the precise power of $c$ does not affect the
Riesz-basis conclusion.
\end{remark}

\begin{theorem}[Canonicalization]
\label{thm:canonical}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and the Stability condition
(Def.~\ref{def:riesz-stable}).
Then:
\begin{enumerate}[\rm(a)]
\item $\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$.
\item \textbf{(Subsequence independence.)}
      Any two diagonal subsequences $c_k, c_k'$
      satisfying Paper~XI, Lem.~\ref{lem:XI-diagonal}
      yield limit systems unitarily equivalent
      up to phase permutation.
\item \textbf{(NAcc as theorem.)}
      The No-Accumulation hypothesis
      (Paper~XI, Hyp.~4.5)
      holds unconditionally under
      \hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
\item \textbf{(Canonical object.)}
      $\Hlim$ is a well-defined self-adjoint operator
      on $\mathcal{K}$, independent of the choice
      of diagonal subsequence up to unitary equivalence.
\end{enumerate}
\end{theorem}

\begin{remark}[What remains open]
\label{rem:remaining-open}
Completeness ($\mathcal{K}=L^2(\mathbb{R})$)
is deferred to Paper~XIII.
The sharp form of $g(c)$ is
Open Problem~\ref{prob:sharp-gc}.
\end{remark}

\begin{lemma}[Inherited: diagonal system, Paper~XI]
\label{lem:XI-diagonal}
From Paper~XI, Lem.~2.3 and Sec.~2.4:
there exists a fixed subsequence $c_k\nearrow\infty$
such that simultaneously for all $n\ge 0$:
\begin{enumerate}[\rm(i)]
\item
  $\Phi_n^{(c_k)}\rightharpoonup\Phi_n^{(\infty)}$
  weakly, with $\|\Phi_n^{(\infty)}\|=1$.
\item
  $\lambda_n^{(c_k)}\to\lambda_n^{(\infty)}$,
  with rate $|\lambda_n^{(c_k)}-\lambda_n^{(\infty)}|
  \le C_\kappa c_k^{-1/4}$ for $n\le\kappa c_k$.
\item
  $\mathcal{H}_{c_k}\xrightarrow{\mathrm{SOT}}\Hlim$;
  $\Hlim$ bounded, self-adjoint, $\|\Hlim\|\le 1$.
\item
  $\{\Phi_n^{(\infty)}\}$ is orthonormal.
\item
  $\Hlim\Phi_n^{(\infty)}
  =\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\end{enumerate}
\end{lemma}

\begin{enumerate}[(XII.1)]
\item\label{XII1}
  \textbf{Orthonormal system:}
  $\ip{\Phi_m^{(\infty)}}{\Phi_n^{(\infty)}}=\delta_{mn}$.
\item\label{XII2}
  \textbf{Eigenvalue rate:}
  $|\lambda_n^{(c)}-\lambda_n^{(\infty)}|
  \le C_\kappa c^{-1/4}$, $n\le\kappa c$.
\item\label{XII3}
  \textbf{SOT + eigenvalue equation:}
  $\Hlim\Phi_n^{(\infty)}=\lambda_n^{(\infty)}\Phi_n^{(\infty)}$.
\item\label{XII4}
  \textbf{Slepian counting:}
  $\#W_c = O(\log c)$;
  $\lambda_n^{(c)}\in[\varepsilon_0,1-\varepsilon_0]$
  for all $n\in W_c$.
\end{enumerate}

\subsection{Setup and strategy}
\label{subsec:setup}

Recall the prolate spheroidal differential operator
\begin{equation}\label{eq:Lc}
  \Lc := -\frac{d}{dx}\Bigl[(1-x^2)\frac{d}{dx}\Bigr]
           + c^2 x^2,
  \quad x\in[-1,1],
\end{equation}
with simple real eigenvalues
$\chi_0^{(c)}<\chi_1^{(c)}<\cdots$.
The concentration eigenvalues satisfy
\begin{equation}\label{eq:chi-lambda}
  \lambda_n^{(c)} = F_c(\chi_n^{(c)})
\end{equation}
where $F_c$ is monotone increasing
(\cite{Slepian1961,Osipov2013}).
A gap bound for $\lambda_n^{(c)}$ is equivalent to
a gap bound for $\chi_n^{(c)}$, modulated by
the derivative $F_c'$.

\begin{remark}[The two-layer problem]
\label{rem:two-layer}
Section~\ref{sec:gap} faces two distinct subproblems:
\begin{enumerate}
\item \textbf{Inner layer (differential):}
      bounding $|\chi_{n+1}^{(c)}-\chi_n^{(c)}|$
      for $n\in W_c$.
      This is accessible via WKB/Airy analysis of $\Lc$.
\item \textbf{Transfer layer (spectral):}
      controlling $F_c'(\chi)$ for $\chi$ in the
      transition regime.
      This requires global information about the
      PSWF integral operator, not just $\Lc$.
\end{enumerate}
Subproblem~1 is solved in
\S\ref{subsec:step1}--\ref{subsec:step2}.
Subproblem~2 is identified as the obstruction in
Thm.~\ref{thm:wkb-indeterminacy}.
\end{remark}

\subsection{WKB rescaling and Airy reduction}
\label{subsec:step1}

For $n\in W_c$, define the classical turning point
\begin{equation}\label{eq:turning-point}
  \xc = \xc(n,c) := \sqrt{1 - \frac{n^2}{c^2}} \in (0,1).
\end{equation}
For $n\in W_c$, $\xc$ stays in a compact subset of $(0,1)$
uniformly (since $n\approx 2c/\pi$).

Introduce the rescaled variable
\begin{equation}\label{eq:rescaling}
  x = \xc + c^{-1/3}\, t, \quad t \in \mathbb{R},
\end{equation}
and write $\psi(t) = \Phi_n^{(c)}(\xc + c^{-1/3}t)$.

\begin{lemma}[WKB rescaling to Airy]
\label{lem:wkb-rescaling}
Under the substitution \eqref{eq:rescaling},
to leading order in $c^{-1/3}$,
the eigenvalue equation
$\Lc \Phi_n^{(c)} = \chi_n^{(c)}\Phi_n^{(c)}$
becomes
\begin{equation}\label{eq:airy-ode}
  -\psi''(t) + \alpha\, t\, \psi(t)
  = \mu_n\, \psi(t)
  + O(c^{-1/3}),
\end{equation}
where
$\alpha = \alpha(n,c) \asymp c^{1/3}$
uniformly for $n\in W_c$,
and $\mu_n$ encodes the deviation
$\chi_n^{(c)} - \chi_n^{(\mathrm{WKB})}$
rescaled by $c^{-2/3}$.
\end{lemma}

\begin{proof}
Expand $(1-x^2)$ and $c^2 x^2$ about $x=\xc$;
the zeroth-order terms cancel by definition
of the turning point.
The next-order terms are both linear in $t$;
after the substitution $\partial_x = c^{1/3}\partial_t$
one collects a factor $c^{2/3}$ from
$-\partial_x[(1-x^2)\partial_x]$,
and after dividing by $c^{2/3}$ arrives at
\eqref{eq:airy-ode}.
\end{proof}

\subsection{Airy eigenvalue spacing: inner layer}
\label{subsec:step2}

After the further rescaling $s=\alpha^{1/3}t$,
equation~\eqref{eq:airy-ode} becomes
$-v''(s) + s\,v(s) = \tilde{\mu}_n v(s)$
with $\tilde{\mu}_n = \mu_n \alpha^{-2/3}$.
The eigenvalues of $-\partial_s^2+s$ on $L^2(\mathbb{R})$
are $\{-a_k\}$ where $a_k$ are the zeros of
$\Ai$, with spacing
\begin{equation}\label{eq:airy-spacing}
  a_{k+1}-a_k \sim \pi k^{-1/3}
  \quad\text{as }k\to\infty
  \quad\text{(\cite[\S10.17]{DLMF2023})}.
\end{equation}
For $n\in W_c$ the corresponding Airy index
satisfies $k\asymp c^{1/3}$,
hence
\begin{equation}\label{eq:delta-tmu}
  \tilde{\mu}_{n+1}-\tilde{\mu}_n \asymp c^{-1/9}.
\end{equation}
Converting back:
$\mu_{n+1}-\mu_n = \alpha^{2/3}(\tilde\mu_{n+1}-\tilde\mu_n)
\asymp c^{2/9}\cdot c^{-1/9} = c^{1/9}$,
and therefore
\begin{equation}\label{eq:delta-chi}
  \chi_{n+1}^{(c)}-\chi_n^{(c)}
  = c^{2/3}(\mu_{n+1}-\mu_n)
  \asymp c^{2/3}\cdot c^{1/9} = c^{7/9}.
\end{equation}

\begin{proposition}[Inner layer gap]
\label{prop:inner-gap}
For all $n\in W_c$ and $c\ge c_0$:
\begin{equation}\label{eq:inner-gap}
  \chi_{n+1}^{(c)} - \chi_n^{(c)} \,\ge\, C_\kappa c^{7/9}.
\end{equation}
\end{proposition}

\begin{remark}[What this does and does not give]
\label{rem:inner-only}
Prop.~\ref{prop:inner-gap} bounds the gap
at the level of the \emph{differential} operator $\Lc$.
To transfer this to a bound on
$|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
one must control $F_c'(\chi_n^{(c)})$,
the derivative of the spectral transfer map.
This is the content of the next subsection.
\end{remark}

\subsection{The spectral transfer obstruction}
\label{subsec:transfer}

\begin{theorem}[WKB indeterminacy of gap exponent]
\label{thm:wkb-indeterminacy}
The gap exponent in $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$
cannot be determined from the WKB/Airy analysis
of $\Lc$ alone.
More precisely:
\begin{enumerate}[\rm(i)]
\item The inner-layer bound
      $\chi_{n+1}^{(c)}-\chi_n^{(c)}\asymp c^{7/9}$
      (Prop.~\ref{prop:inner-gap})
      is accessible via WKB.
\item The transfer factor $F_c'(\chi_n^{(c)})$
      for $\chi_n^{(c)}$ in the transition regime
      is \emph{not} algebraically stable in $c$:
      its $c$-dependence encodes global information
      about the PSWF integral operator
      (Slepian concentration, kernel asymptotics,
      prolate operator resolvent structure)
      that is inaccessible to local ODE analysis.
\item Any gap exponent derived by assuming
      $F_c'\asymp c^{-\beta}$ for some fixed $\beta$
      is conditional on an \emph{unverified global claim}.
\end{enumerate}
\end{theorem}

\begin{proof}
(i) is Prop.~\ref{prop:inner-gap}.

(ii): The map $F_c$ is defined via the spectral
correspondence between $\Lc$ and the
band-limiting integral operator
$\mathcal{H}_c f(x)
= \int_{-1}^{1} \frac{\sin c(x-y)}{\pi(x-y)} f(y)\,dy$.
The derivative $F_c'$ encodes the density of
concentration eigenvalues near the critical level,
which by the Widom--Slepian theory
(\cite{Widom1964,Osipov2013})
has a soft-exponential transition profile near
$\lambda=1/2$, not an algebraic one.
Specifically, in the transition window $W_c$,
the function $\lambda\mapsto F_c^{-1}(\lambda)$
behaves like the inverse of a logistic-type
function of $c$-scaled argument;
its derivative cannot be bounded above and below
by a pure power of $c$ with a $c$-independent constant.

(iii) follows immediately from (ii):
any formula $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|
\asymp c^{7/9}\cdot c^{-\beta}$
requires the assumption $F_c'\asymp c^{-\beta}$,
which has no proof from WKB alone.
\end{proof}

\begin{remark}[Interpretation]
\label{rem:interpretation}
Thm.~\ref{thm:wkb-indeterminacy} does not say
that the gap is small or zero.
It says that its \emph{precise algebraic order}
is not a local quantity.
The gap is positive (Lem.~\ref{lem:gap-structural}
below), but its rate is a global invariant
of the PSWF concentration operator.
Determining the sharp scale $g(c)$ requires
techniques beyond WKB:
either Plancherel--Rotach asymptotics of
the prolate kernel, or resolvent estimates
for $\mathcal{H}_c$
(see Prob.~\ref{prob:sharp-gc}).
\end{remark}

\subsection{The structural gap hypothesis and its sufficiency}
\label{subsec:structural-gap}

\begin{hypothesis}[Structural gap condition]
\label{hyp:gap-structural}
There exists a function $g\colon[1,\infty)\to(0,\infty)$
with $g(c)\to 0$ and $g(c)\gg c^{-1/4}$,
and a constant $\kappa_0>0$, such that
for all $c\ge 1$ and all $n\in W_c$
with $n\le\kappa_0 c$:
\begin{equation}\label{eq:gap-structural}
  |\lambda_n^{(c)} - \lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Relation to previous hypotheses]
\label{rem:hyp-comparison}
The earlier \textit{Hyp.~Gap} of the stub version
of this paper stated $g(c)=C_\kappa c^{-3/4}$.
By Thm.~\ref{thm:wkb-indeterminacy}, this sharp
form cannot be established from WKB.
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
replaces it: it asserts positive separation at
an unspecified rate $g(c)\gg c^{-1/4}$.
This weaker form is sufficient for all
downstream results (Rem.~\ref{rem:sufficiency-gap}).
\end{remark}

\begin{remark}[Sufficiency for Riesz basis]
\label{rem:sufficiency-gap}
The condition $g(c)\gg c^{-1/4}$ ensures that
consecutive eigenvalues $\lambda_n^{(c)}$
cannot cluster faster than the convergence
rate $c^{-1/4}$ of \eqref{XII2}.
This is exactly what prevents frame collapse
in the limit $c_k\to\infty$
(Lem.~\ref{lem:gram-approx}) and what
drives the NAcc proof (Cor.~\ref{cor:nacc}).
The precise form of $g(c)$ is irrelevant
for both applications.
\end{remark}

\begin{lemma}[Structural gap in $W_c$]
\label{lem:gap-structural}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
Then for all $n\in W_c$ and $c\ge c_0$:
\[
  |\lambda_n^{(c)} - \lambda_{n+1}^{(c)}|
  \ge g(c) \gg c^{-1/4}.
\]
In particular, consecutive PSWF concentration
eigenvalues in the transition window are
strictly separated at a scale dominating
the Paper~XI convergence rate.
\end{lemma}
\begin{proof}
This is immediate from
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
The inner-layer bound Prop.~\ref{prop:inner-gap}
is consistent with any $g(c)$ satisfying
$g(c)\ll c^{7/9}$ (the inner-layer scale),
and the condition $g(c)\gg c^{-1/4}$ is
compatible with all exponents in
$(-1/4,\,7/9)$.
\end{proof}

\begin{corollary}[NAcc as theorem]
\label{cor:nacc}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{XII2},
the No-Accumulation hypothesis
(Paper~XI, Hyp.~4.5) holds:
for all $\mu\in(0,1)$,
$\#\{n : \lambda_n^{(\infty)}\le\mu\}<\infty$.
\end{corollary}

\begin{proof}
Suppose for contradiction that
$\lambda_{n_j}^{(\infty)}\to\mu$
for some infinite subsequence $(n_j)$.
By \eqref{XII2}, for all $k$ with $n_j\le\kappa c_k$:
\[
  |\lambda_{n_j}^{(c_k)}-\mu|
  \le C_\kappa c_k^{-1/4}
     + |\lambda_{n_j}^{(\infty)}-\mu|.
\]
For large $k$ and $j$, both terms are $<\varepsilon$,
so infinitely many eigenvalues of $\mathcal{H}_{c_k}$
lie within $\varepsilon$ of $\mu$.
But Lem.~\ref{lem:gap-structural} gives
$|\lambda_{n+1}^{(c_k)}-\lambda_n^{(c_k)}|\ge g(c_k)$,
so the number of eigenvalues in any
interval of length $2\varepsilon$ is at most
$\lceil 2\varepsilon/g(c_k)\rceil$.
For fixed $\varepsilon>0$ and $c_k\to\infty$:
since $g(c_k)\gg c_k^{-1/4}\to 0$ but
$\varepsilon$ is fixed, this count remains
bounded, contradicting the infinite subsequence.
\end{proof}

\begin{remark}[What WKB does establish]
\label{rem:wkb-positive}
Summarizing the positive content of this section:
\begin{itemize}
\item \textbf{Proved (unconditionally):}
      inner-layer bound
      $\chi_{n+1}^{(c)}-\chi_n^{(c)}\ge C c^{7/9}$
      (Prop.~\ref{prop:inner-gap});
      Airy reduction is valid
      (Lem.~\ref{lem:wkb-rescaling});
      WKB indeterminacy of gap exponent
      (Thm.~\ref{thm:wkb-indeterminacy}).
\item \textbf{Proved (under Hyp.~Gap-S):}
      $|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|\ge g(c)
      \gg c^{-1/4}$;
      NAcc theorem (Cor.~\ref{cor:nacc}).
\item \textbf{Open:}
      sharp form of $g(c)$;
      determination of $F_c'$ in the transition regime;
      see Prob.~\ref{prob:sharp-gc}.
\end{itemize}
\end{remark}

\section{Gram-Matrix Control}
\label{sec:gram}
%%══════════════════════════════════════════════════════════════════════

\begin{remark}[Gram matrix of the limit system]
\label{rem:gram-trivial}
Since $\{\Phi_n^{(\infty)}\}$ is orthonormal (\eqref{XII1}),
$\Gram_{mn}=\delta_{mn}$: trivially bounded and invertible.
The non-trivial content is uniform Riesz stability
of the approximants $\{\Phi_n^{(c_k)}\}$.
\end{remark}

\begin{definition}[Uniform Riesz stability]
\label{def:riesz-stable}
The family $\{\Phi_n^{(c_k)}\}_{n\ge 0,\,k\ge 1}$
is \emph{uniformly Riesz-stable} if there exist
$0<A\le B<\infty$, independent of $k$, such that
for all finite $(a_n)\subset\mathbb{C}$ and all $k$:
\begin{equation}\label{eq:riesz-stable}
  A\sum_n|a_n|^2
  \le\Bigl\|\sum_n a_n\Phi_n^{(c_k)}\Bigr\|^2
  \le B\sum_n|a_n|^2.
\end{equation}
\end{definition}

\begin{lemma}[Gram control for approximants]
\label{lem:gram-approx}
Assume \hyperref[hyp:gap-structural]{Hyp.~Gap-S}
and \eqref{XII1}--\eqref{XII3}.
Then $\{\Phi_n^{(c_k)}\}$ is uniformly Riesz-stable
with $A=1-\varepsilon_k$, $B=1+\varepsilon_k$,
$\varepsilon_k\to 0$.
\end{lemma}
\begin{proof}
For each fixed $k$, $A=B=1$ (ON system).
The gap $g(c_k)\gg c_k^{-1/4}$ prevents
spectral collapse as $k\to\infty$;
the frame bounds are inherited in the limit.
Full detail: Phase~3.
\end{proof}

\section{Bari--Krein Machinery}
\label{sec:bari-krein}
%%══════════════════════════════════════════════════════════════════════

\begin{theorem}[Bari--Krein, standard]
\label{thm:bari-krein}
A sequence $\{f_n\}$ in a separable Hilbert space $\mathcal{H}$
is a Riesz basis of its closed span if and only if
the Gram matrix $\ip{f_m}{f_n}$ is bounded
and invertible on $\ell^2$.
\end{theorem}
\begin{proof}
\cite[Ch.~VI]{GohbergKrein1969},
\cite[Sec.~6.1]{Christensen2003}.
\end{proof}

\begin{proposition}[Riesz basis of $\mathcal{K}$]
\label{prop:riesz-basis}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
$\{\Phi_n^{(\infty)}\}$ is a Riesz basis of $\mathcal{K}$.
\end{proposition}
\begin{proof}
$\Gram=I$ (Rem.~\ref{rem:gram-trivial}),
uniformly Riesz-stable approximants
(Lem.~\ref{lem:gram-approx}),
then Thm.~\ref{thm:bari-krein} via SOT
(\eqref{XII3}).
\end{proof}

\section{Canonicalization}
\label{sec:canonical}

\begin{lemma}[Uniqueness of $\mathcal{K}$]
\label{lem:same-K}
Any two diagonal subsequences satisfying
Lem.~\ref{lem:XI-diagonal} and
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}
yield the same closed span $\mathcal{K}$.
\end{lemma}
\begin{proof}
Both are closed invariant subspaces of $\Hlim$
(which is uniquely determined as an SOT limit).
By the spectral theorem for bounded self-adjoint
operators, the invariant subspace spanned by a
maximal orthonormal set of eigenvectors is unique.
Full detail: Phase~3.
\end{proof}

\begin{theorem}[Canonicalization]
\label{thm:canonical-proof}
Under \hyperref[hyp:gap-structural]{Hyp.~Gap-S},
$\Hlim$ is a canonical self-adjoint operator on $\mathcal{K}$,
independent of diagonal subsequence up to unitary
equivalence.
\end{theorem}
\begin{proof}
(1) Riesz basis: Prop.~\ref{prop:riesz-basis}.
(2) Unique $\mathcal{K}$: Lem.~\ref{lem:same-K}.
(3) Spectrum $\{\lambda_n^{(\infty)}\}$
independent of subsequence (SOT uniqueness + \eqref{XII2}).
(4) Two self-adjoint operators with the same simple
spectrum are unitarily equivalent (spectral theorem).
\end{proof}

\section{Open Problems}
\label{sec:open}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Determine the sharp form of $g(c)$ in
\hyperref[hyp:gap-structural]{Hyp.~Gap-S}.
By Thm.~\ref{thm:wkb-indeterminacy}, this
requires control of the spectral transfer
$F_c'(\chi_n^{(c)})$ in the transition regime.
Two candidate approaches:
\begin{enumerate}
\item \textbf{Plancherel--Rotach asymptotics:}
      use the Plancherel--Rotach type expansion
      of the prolate integral kernel to extract
      the density of concentration eigenvalues
      near $1/2$.
\item \textbf{Resolvent asymptotics:}
      control the resolvent of $\mathcal{H}_c$
      at spectral parameter $1/2$;
      the imaginary part gives the spectral density
      and hence $F_c'$.
\end{enumerate}
Either approach is expected to yield
$g(c) = c^{-\alpha}$ for some
$\alpha\in(1/4,\,7/9)$,
but the precise value of $\alpha$ is unknown.
\end{problem}

\begin{problem}[Completeness]
\label{prob:completeness}
$\mathcal{K} = L^2(\mathbb{R})$?
Target: Paper~XIII.
\end{problem}

\begin{problem}[Density]
\label{prob:density}
Density of $\{\lambda_n^{(\infty)}\}$ in $[0,1]$.
\end{problem}

\begin{problem}[Explicit Riesz bounds]
\label{prob:riesz-constants}
Determine $A,B$ in \eqref{eq:riesz-stable}
explicitly in terms of $g(c)$.
\end{problem}
