\title[A Localization Principle for Spectral Projection Stability]{%
  A Localization Principle for Spectral Projection Stability
  under Strong Operator Convergence}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
\textbf{Main result.}
For bounded self-adjoint operators with
$A_k\xrightarrow{\mathrm{SOT}} A$,
under an explicit spectral isolation
assumption (Assumption~S),
the gap-normalized residual condition
\[
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}\to 0
\]
is sufficient for rank-one spectral
projection stability
$\|P_k - P^\infty\|\to 0$,
and necessary in the bounded-gap regime
$\gamma_k\ge c_0>0$.

The sufficiency rests on an explicit
three-step mechanism
(Lem.~\ref{lem:mechanism-chain}):
rank-one structure encodes the full
projection in a single vector;
the spectral gap provides a resolvent
bound; and the local residual controls
the only remaining degree of freedom.
This is a \emph{localization principle}
--- dependence only on the residual
evaluated on a single eigenvector,
normalized by the local spectral gap ---
that goes beyond global-norm-based
perturbation theory in the SOT-only regime
(Lem.~\ref{lem:dk-comparison},
Prop.~\ref{prop:separation}).

\medskip
We further establish:
a No-Go result (Prop.~\ref{prop:no-go})
showing eigenvalue density alone does not
control spectral angles;
a quantitative counterexample
(Prop.~\ref{prop:counterexample})
linking ratio blow-up to projection
failure; and a reduction of PSWF
projection stability to two independent
analytic mechanisms (Gap-S and
SOT-Faster).
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Context and goal}

\begin{center}
\fbox{\parbox{0.88\linewidth}{\centering\itshape
SOT produces approximate eigenvectors,
but does not control the residual
relative to the spectral gap.
This is the precise obstruction to
spectral projection stability.}}
\end{center}

\medskip
\noindent
\textbf{Main Principle (informal).}
Let $A_k\xrightarrow{\mathrm{SOT}} A$
with isolated simple eigenvalue $\lambda$
of $A$ and associated simple eigenvalues
$\lambda_k$ of $A_k$
(Assumption~S below). Then:
\begin{equation}\label{eq:main-principle-informal}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0
  \;\Longrightarrow\;
  \|P_k-P^\infty\|\to 0.
\end{equation}
The converse holds when $\gamma_k\ge c_0>0$
(Thm.~\ref{thm:necessary-bounded-gap}).

\medskip
\noindent
\textbf{Why rank-one + gap + residual suffice.}
The mechanism is not that smallness in
one direction ``propagates'' to all
directions --- this would be false in
general.
Rather, rank-one structure means
the projection $P^\infty = \phi\otimes\phi$
is \emph{entirely encoded} by $\phi$;
there are no other degrees of freedom.
The spectral gap then supplies a resolvent
bound that converts residual smallness
into projection smallness
(Lem.~\ref{lem:mechanism-chain}).
This chain is exact and fails at each
step if any of the three conditions
is removed.

\medskip
Paper~X establishes $\mathcal{H}_{c_k}
\xrightarrow{\mathrm{SOT}}\Hlim$;
Paper~XI constructs an orthonormal
diagonal eigenfunction system
$\{\Phi_n^{(\infty)}\}$.
This paper proves:
\begin{itemize}
\item \eqref{eq:main-principle-informal}
  is the \emph{sharp sufficient mechanism}
  (Thm.~\ref{thm:abstract},
  Lem.~\ref{lem:mechanism-chain}).
\item When $\gamma_k\ge c_0>0$,
  it is also \emph{necessary}
  (Thm.~\ref{thm:necessary-bounded-gap}).
\item The condition operates beyond
  global-norm-based perturbation theory:
  there exist families with
  $\|A_k-A\|\equiv 1$, genuine spectral
  interaction, yet
  \eqref{eq:main-principle-informal}
  holds and $\|P_k-P^\infty\|\to 0$
  (Prop.~\ref{prop:separation}).
\item The ``block structure'' in that
  example is not a hidden assumption
  but the \emph{principle itself}:
  whether $\phi$ sees the perturbation
  is the only question that matters
  (Rem.~\ref{rem:block-is-the-point}).
\item Eigenvalue density alone does not
  control spectral angles
  (Prop.~\ref{prop:no-go}).
\item SOT $+$ eigenvalue convergence
  is insufficient; ratio blow-up implies
  projection failure quantitatively
  (Prop.~\ref{prop:counterexample}).
\item PSWF projection stability reduces
  to Gap-S and SOT-Faster
  (Thms.~\ref{thm:convergence},
  \ref{thm:rate}).
\end{itemize}

%%══ 3. HYPOTHESES AND NO-GO
\section{Hypotheses, Structural Necessity,
  and No-Go}
\label{sec:gap}

\begin{hypothesis}[Effective gap, Gap-S]
\label{hyp:gap-structural}
Let $W_c:=\{n:\lambda_n^{(c)}
\in(\delta,1-\delta)\}$
for fixed $\delta>0$.
There exists $g\colon[1,\infty)\to(0,\infty)$
such that
\begin{equation}\label{eq:gap-dominates}
  \frac{c^{-1/4}}{g(c)}\to 0
  \quad\text{as }c\to\infty,
\end{equation}
and for all $c\ge 1$,
uniformly in $n\in W_c$:
\begin{equation}\label{eq:gap-s}
  |\lambda_n^{(c)}-\lambda_{n+1}^{(c)}|
  \ge g(c).
\end{equation}
\end{hypothesis}

\begin{remark}[Gap-S: necessity and
  determinantal context]
\label{rem:gap-necessary}
Gap-S is necessary for any argument
based solely on local spectral gap
control and SOT-type information
(Prop.~\ref{prop:no-go}).
Widom~\cite{Widom1964} controls average
density ($\sim(\log c)^{-1}$ spacing),
not the minimal gap; Gap-S is strictly
stronger.
In determinantal point processes,
minimal gaps are controlled via
sine-kernel repulsion; Gap-S is the
analogous deterministic bound for the
PSWF kernel
(Open Problem~\ref{prob:sharp-gc}).
\end{remark}

\begin{remark}[$c^{-1/4}/g(c)\to 0$]
\label{rem:gap-implies-o}
Condition \eqref{eq:gap-dominates}
states that the eigenvalue-matching
error $C_\kappa c^{-1/4}$ is of strictly
smaller order than $g(c)$.
This ratio is used in
\eqref{eq:ratio-to-zero}.
\end{remark}

\begin{proposition}[No-Go: eigenvalue
  density does not control spectral angles]
\label{prop:no-go}
Let $(\mu_n)$ be a sequence with
$\liminf_n(\mu_n-\mu_{n+1})=0$
but with Widom-type average density.
Then there exists a sequence
$B_k\xrightarrow{\mathrm{SOT}} B$
of bounded self-adjoint operators
with eigenvalues $\mu_n^{(k)}\to\mu_n$
such that
$\|Q_n^{(k)}-Q_n\|\not\to 0$
for infinitely many $n$.
\end{proposition}
\begin{proof}
Choose indices $n_j$ with
$\gamma_j := \mu_{n_j} - \mu_{n_j+1}
\to 0$.
For each $j$ define on
$\mathrm{span}\{e_{n_j}, e_{n_j+1}\}$:
\[
  B_k^{(j)} :=
  \begin{pmatrix}
    \mu_{n_j}^{(k)} & \varepsilon_j(k) \\
    \varepsilon_j(k) & \mu_{n_j+1}^{(k)}
  \end{pmatrix},
  \quad
  \varepsilon_j(k) := \sqrt{\gamma_j^{(k)}},
\]
$\gamma_j^{(k)}
:= \mu_{n_j}^{(k)} - \mu_{n_j+1}^{(k)}
\to \gamma_j \to 0$.
Set $B_k := \bigoplus_j B_k^{(j)}$
(block-diagonal, uniformly bounded);
SOT convergence follows since each $e_m$
is affected by at most one block and
$\varepsilon_j(k)\to 0$.
The eigenvector angle satisfies
\[
  \tan\bigl(2\theta_j^{(k)}\bigr)
  = \frac{2}{\sqrt{\gamma_j^{(k)}}}
  \to +\infty,
\]
so $\theta_j^{(k)}\to\pi/4$ and
$\|Q_{n_j}^{(k)}-Q_{n_j}\|
\ge|\cos^2\theta_j^{(k)}-1|
\to\tfrac{1}{2}$ for all $j$.
\end{proof}

\begin{remark}[Scope of No-Go]
\label{rem:no-go-scope}
This is a constructed pathology, not
a general impossibility over all
operator families.
The claim is precise: for any sequence
with $\liminf$ gap $= 0$, there
\emph{exists} a family exhibiting
projection instability.
The construction exploits exactly the
gap-collapse regime excluded by Gap-S.
\end{remark}

\begin{hypothesis}[SOT-Faster]
\label{hyp:sot-faster}
\begin{equation}\label{eq:sot-faster}
  \varepsilon_n(k) :=
  \|(\mathcal{H}_{c_k}-\Hlim)
  \Phi_n^{(\infty)}\|
  = o(g(c_k)).
\end{equation}
\end{hypothesis}

\begin{remark}[SOT-Faster: pointwise,
  not uniform; strictly weaker than
  local norm convergence]
\label{rem:sot-faster-localized}
SOT-Faster asserts
$\varepsilon_n(k) = o(g(c_k))$,
a \emph{pointwise} condition at the
single fixed vector $\Phi_n^{(\infty)}$.
This is strictly weaker than local
norm convergence, which would require
\[
  \sup_{\substack{\|f\|\le 1}}
  \|(\mathcal{H}_{c_k}-\Hlim)f\|
  \cdot g(c_k)^{-1} \to 0
  \quad \text{on a spectral window},
\]
i.e.\ uniform control over all vectors
in a spectral window of width $g(c_k)$.
SOT-Faster requires only the ratio
$\varepsilon_n(k)/g(c_k)\to 0$
at one test vector --- a strictly weaker,
one-vector condition.
It is also not implied by mere SOT,
which gives only $\varepsilon_n(k)\to 0$
without rate relative to $g(c_k)$.
SOT-Faster is therefore the minimal
vector-local rate condition needed
to activate the localization principle,
and is the sole remaining analytic
bottleneck in the PSWF reduction
(Open Problem~\ref{prob:sot-faster}).
\end{remark}

\begin{hypothesis}[SOT-Rate]
\label{hyp:sot-rate}
$\varepsilon_n(k) = O(c_k^{-1/4})$.
\end{hypothesis}

\begin{remark}[Logical hierarchy]
\label{rem:hyp-hierarchy}
SOT-Rate $\Rightarrow$ SOT-Faster
$\Rightarrow$
$\|P_n^{(c_k)}-P_n^{(\infty)}\|\to 0$.
SOT alone gives $\varepsilon_n(k)\to 0$;
gap comparison is strictly additional.
\end{remark}

\begin{remark}[Plausibility of SOT-Rate]
\label{rem:sot-rate-plausible}
Expanding in eigenbasis
$\{\Phi_m^{(c_k)}\}$:
\begin{equation}\label{eq:formal-expansion}
  \varepsilon_n(k)^2
  = \sum_m
  |\lambda_m^{(c_k)}-\lambda_n^{(\infty)}|^2
  |\ip{\Phi_n^{(\infty)}}
  {\Phi_m^{(c_k)}}|^2.
\end{equation}
Diagonal $m=n$:
$\le C_\kappa^2 c_k^{-1/2}$.
Small off-diagonal overlaps
(consistent with Gap-S)
predict $\varepsilon_n(k)=O(c_k^{-1/4})$;
Prob.~\ref{prob:off-diagonal}.
\end{remark}

\subsection{WKB insufficiency}
\begin{theorem}[WKB insufficiency]
\label{thm:wkb-indeterminacy}
WKB/Airy analysis of
$\Lc=-\tfrac{d}{dx}[(1-x^2)
\tfrac{d}{dx}]+c^2x^2$
cannot determine the scaling of
$|\lambda_{n+1}^{(c)}-\lambda_n^{(c)}|$

\end{theorem}
\begin{proof}
Sigmoid profile of
$F_c$~\cite{Widom1964};
any algebraic rate is a global claim.
\end{proof}

%%══ 4. ABSTRACT STABILITY PRINCIPLE
\section{Abstract Stability Principle}
\label{sec:abstract}

\begin{assumption}[Spectral isolation,
  Assumption~S]
\label{ass:spectral-isolation}
Let $\{A_k\}$ and $A$ be bounded
self-adjoint on $\mathcal{H}$ with
$A_k\xrightarrow{\mathrm{SOT}} A$.
\begin{enumerate}[\rm(S1)]
\item\label{S1}
  $\lambda\in\sigma(A)$ is isolated and
  simple; $P^\infty$ is the rank-one
  projection onto $\ker(A-\lambda)$.
\item\label{S2}
  For each $k$, $\lambda_k\in\sigma(A_k)$
  is isolated and simple, with
  $\gamma_k :=
  \dist(\lambda_k,\sigma(A_k)
  \setminus\{\lambda_k\})>0$
  and rank-one projection $P_k$.
\item\label{S3}
  $\lambda_k\to\lambda$.
\end{enumerate}
\end{assumption}

\begin{remark}[Spectral isolation under SOT]
\label{rem:spectral-isolation}
Under strong resolvent convergence,
Assumption~S follows automatically
from isolation of $\lambda$ via
Riesz projections~\cite[Ch.~VIII]{Kato1966}.
Under mere SOT, it must be imposed
explicitly.
In the PSWF application,
(S1)--(S3) follow from
Lem.~\ref{lem:inherited}(A3)
and Lem.~\ref{lem:eigenvalue-matching}.
\end{remark}

\begin{lemma}[Mechanism chain:
  why rank-one + gap + residual suffice]
\label{lem:mechanism-chain}
Under Assumption~S,
the implication
$\|(A_k-A)\phi\|/\gamma_k\to 0
\Rightarrow \|P_k-P^\infty\|\to 0$
is the composition of three steps,
each of which is exact and tight:
\begin{enumerate}[\rm(M1)]
\item\label{M1}
  \textbf{(Rank-one encoding)}
  $P^\infty = \phi\otimes\phi$
  is completely determined by $\phi$;
  the only degree of freedom for
  $\|P_k - P^\infty\|$ is
  $\|(I-P^\infty)(P_k - P^\infty)\|
  = \|(I-P_k)\phi\|$
  (up to normalization,
  by the rank-one identity
  $\|P-Q\| = \|(I-P)Qf\|/\|f\|$
  for rank-one $P$, $Q$,
  $f\in\mathrm{ran}(Q)$).
\item\label{M2}
  \textbf{(Gap resolvent bound)}
  The spectral gap $\gamma_k$ controls
  the resolvent on
  $\mathrm{ran}(I-P_k)$:
  \[
    \|(A_k-\lambda_k)^{-1}
    |_{\mathrm{ran}(I-P_k)}\|
    \le \frac{1}{\gamma_k}.
  \]
  This converts residual smallness into
  projection smallness:
  $\|(I-P_k)\phi\|
  \le\|(A_k-\lambda_k)\phi\|/\gamma_k$.
\item\label{M3}
  \textbf{(Residual triangle inequality)}
  $\|(A_k-\lambda_k)\phi\|
  \le\|(A_k-A)\phi\|
  +|\lambda_k-\lambda|$,
  so the ratio condition
  \eqref{eq:abstract-ratio}
  controls the residual.
\end{enumerate}
Each step is an equality or tight bound
(see Prop.~\ref{prop:counterexample}
for tightness of the combined estimate).
Removing rank-one structure
(Rem.~\ref{rem:rank-one-essential}),
removing gap control
(Prop.~\ref{prop:no-go}),
or removing residual smallness
(Prop.~\ref{prop:counterexample})
each breaks the chain.
\end{lemma}
\begin{proof}
(M1): For rank-one projections
$P = u\otimes u$, $Q = v\otimes v$
with $\|u\|=\|v\|=1$:
$\|P-Q\| = \|(I-Q)u\|$
(classical; \cite[Ch.~1]{Bhatia1997}).
Hence $\|P_k-P^\infty\|
= \|(I-P_k)\phi\|$.

(M2): By the spectral theorem,
$A_k - \lambda_k$ is invertible on
$\mathrm{ran}(I-P_k)$ with all spectral
values at distance $\ge\gamma_k$
from zero.
The decomposition
$\phi = P_k\phi + (I-P_k)\phi$
and $(A_k-\lambda_k)P_k\phi = 0$
give
$(A_k-\lambda_k)\phi
= (A_k-\lambda_k)(I-P_k)\phi$;
hence
$\|(I-P_k)\phi\|
\le\|(A_k-\lambda_k)\phi\|/\gamma_k$.

(M3): Triangle inequality.
Combining (M1)--(M3):
$\|P_k-P^\infty\|
= \|(I-P_k)\phi\|
\le (\|(A_k-A)\phi\|
+ |\lambda_k-\lambda|)/\gamma_k \to 0$.
\end{proof}

\begin{remark}[Why the chain is exact,
  not magical]
\label{rem:chain-exact}
Step (M1) shows that rank-one structure
\emph{collapses the problem}: measuring
projection error reduces entirely to
measuring how well $\phi$ is captured
by $P_k$, which is a scalar.
Step (M2) shows that the spectral gap
is the \emph{only} mechanism by which
operator information reaches projection
information when no global norm is
available.
Step (M3) is the \emph{local observable}:
the only quantity accessible under SOT
without global norm control.
The chain is not ``one direction controls
all''; it is ``rank-one structure means
there is only one direction.''
\end{remark}

\begin{lemma}[Davis--Kahan implies
  ratio condition]
\label{lem:dk-comparison}
Under Assumption~S,
if $\|A_k - A\|\to 0$, then
\eqref{eq:abstract-ratio} holds
provided $\gamma_k \ge c_0 > 0$.
In particular:
\[
  \|A_k-A\|\to 0
  \;\Longrightarrow\;
  \eqref{eq:abstract-ratio}
  \;\Longrightarrow\;
  \|P_k-P^\infty\|\to 0,
\]
but neither implication reverses
(Props.~\ref{prop:separation},
\ref{prop:counterexample}).
\end{lemma}
\begin{proof}
$\|(A_k-A)\phi\|\le\|A_k-A\|\to 0$;
$|\lambda_k-\lambda|\le\|A_k-A\|\to 0$;
divide by $\gamma_k\ge c_0$.
\end{proof}

\begin{proposition}[Separation: genuine
  spectral interaction with
  $\|A_k-A\|\equiv 1$]
\label{prop:separation}
There exist bounded self-adjoint
$\{A_k\}$, $A$ on $\ell^2$
satisfying Assumption~S with
$\gamma_k \ge c_0 > 0$, such that:
\begin{enumerate}[\rm(i)]
\item $\|A_k - A\| = 1$ for all $k$;
\item $\|(A_k - A)\phi\|
  = \varepsilon_k \to 0$,
  so \eqref{eq:abstract-ratio} holds;
\item $A_k - A$ has a genuine
  off-diagonal spectral interaction
  component;
\item $\|P_k - P^\infty\| \to 0$.
\end{enumerate}
\end{proposition}
\begin{proof}
Set $\phi = e_1$, $\lambda = 0$,
\[
  A := \sum_{n \ge 2}
  \mu_n (e_n \otimes e_n),
  \quad \mu_n \in [-1, -c_0]
  \text{ uniformly},
\]
so $\lambda = 0$ is isolated and simple
with gap $\gamma \ge c_0 > 0$.
Define:
\[
  A_k - A :=
  \underbrace{\varepsilon_k
  (e_2 \otimes e_1 + e_1 \otimes e_2)}_{S_k}
  + \underbrace{K_k}_{}
\]
where $S_k$ is a rank-two off-diagonal
perturbation with $\|S_k\|=\varepsilon_k\to 0$,
and $K_k$ acts on
$\mathrm{span}\{e_3,e_4,\ldots\}$
only, with $\|K_k\|=1$, with
$\sigma(A|_{\mathrm{span}\{e_n,n\ge3\}}
+K_k)$ uniformly bounded away from $0$.

\textit{(i)}: $\|A_k-A\|\ge\|K_k\|=1$.

\textit{(iii)}: $S_k$ couples $e_1$ and
$e_2$ off-diagonally,
producing genuine eigenvalue splitting
and eigenvector mixing between
$e_1$ and $e_2$.
This is not a block decoupling:
$S_k$ acts non-trivially on
$\mathrm{span}\{e_1,e_2\}$
and $e_1\in\mathrm{dom}(S_k)$.

\textit{(ii)}: $(A_k-A)\phi
= S_k e_1 + K_k e_1
= \varepsilon_k e_2 + 0$
(since $K_k e_1 = 0$).
So $\|(A_k-A)\phi\|=\varepsilon_k\to 0$.

\textit{Gap}: $\gamma_k
\ge c_0 - C\varepsilon_k^2\ge c_0/2$
for large $k$
(\cite[Ch.~II]{Kato1966}).

\textit{(iv)}: Apply
Thm.~\ref{thm:abstract}.
\end{proof}

\begin{remark}[``Block structure''
  is the principle, not a hidden assumption]
\label{rem:block-is-the-point}
A reviewer may observe that $K_k$
acts on $\phi^\perp$ and ask: is this
not a hidden block structure?
The answer is: yes, and this is exactly
the point.
The localization principle says:
\emph{what matters is whether $\phi$
sees the perturbation, not whether the
perturbation is globally large.}
A perturbation that acts entirely on
$\phi^\perp$ is invisible to the
spectral projection $P^\infty
= \phi\otimes\phi$, regardless of its
norm.
This is not a special property of the
construction; it is the \emph{content}
of Lem.~\ref{lem:mechanism-chain}:
rank-one structure means the projection
has only one degree of freedom, and that
degree of freedom is controlled by
$\|(A_k-A)\phi\|$.
The off-diagonal component $S_k$
shows that the perturbation genuinely
touches the spectral mechanism
(it mixes $e_1$ and $e_2$),
yet is controlled because its effect
on $\phi$ is small relative to the gap.
\end{remark}

\begin{theorem}[Main Principle;
  sufficient condition]
\label{thm:abstract}
Under Assumption~S, if
\begin{equation}\label{eq:abstract-ratio}
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  +\frac{|\lambda_k-\lambda|}{\gamma_k}
  \to 0,
\end{equation}
then $\|P_k - P^\infty\|\to 0$.
\end{theorem}
\begin{proof}
Direct from
Lem.~\ref{lem:mechanism-chain}:
steps (M3), (M2), (M1).
\end{proof}

\begin{remark}[Non-circularity]
\label{rem:non-circularity}
$\phi$ is a fixed eigenvector of $A$,
not of $A_k$.
$(A_k-A)\phi$ measures operator deviation
on a fixed vector; no eigenvector of
$A_k$ is used in its definition.
\end{remark}

\begin{remark}[Structural independence]
\label{rem:structural-independence}
No spectral data of $A_k$ enters the
construction of $\phi$.
All $A_k$-quantities appear only as
$(A_k-A)\phi$ or $(A_k-\lambda_k)\phi$.
\end{remark}

\begin{lemma}[Rank-one projection
  convergence implies eigenvector
  convergence]
\label{lem:rank-one-proj}
If $P_k = v_k\otimes v_k$,
$P^\infty = \phi\otimes\phi$,
$\|v_k\|=\|\phi\|=1$,
and $\|P_k - P^\infty\|\to 0$,
then there exist phases $\theta_k$ with
$\|v_k - e^{i\theta_k}\phi\|\to 0$.
\end{lemma}
\begin{proof}
$\|v_k\ip{v_k}{\phi}-\phi\|\to 0$;
$|\alpha_k|:=|\ip{v_k}{\phi}|\to 1$;
choose $\theta_k$ with
$e^{i\theta_k}\alpha_k>0$;
$\|v_k-e^{i\theta_k}\phi\|\to 0$
by standard estimate.
\end{proof}

\begin{theorem}[Necessary condition;
  bounded-gap case]
\label{thm:necessary-bounded-gap}
Under Assumption~S, if
$\|P_k-P^\infty\|\to 0$ and
$\gamma_k\ge c_0>0$,
then \eqref{eq:abstract-ratio} holds.
\end{theorem}
\begin{proof}
\textbf{Eigenvalue term:}
$|\lambda_k-\lambda|/\gamma_k
\le|\lambda_k-\lambda|/c_0\to 0$.

\textbf{Residual term:}
Decompose
$(A_k-A)\phi
= (A_k-A)P_k\phi+(A_k-A)(I-P_k)\phi$.
Term~(II):
$\|(A_k-A)(I-P_k)\phi\|
\le 2M\|(I-P_k)\phi\|\to 0$
($M:=\sup_k\|A_k\|<\infty$).
Term~(I):
$\alpha_k(\lambda_k v_k-Av_k)$;
$\|(\lambda_k-\lambda)v_k\|
=|\lambda_k-\lambda|\to 0$;
$(A-\lambda)v_k
=(A-\lambda)(v_k-e^{i\theta_k}\phi)\to 0$
by Lem.~\ref{lem:rank-one-proj}.
Combining and dividing by $c_0$:
\eqref{eq:abstract-ratio} holds.
\end{proof}

\begin{remark}[Necessity requires
  uniform gap]
\label{rem:necessity-uniform-gap}
Without $\gamma_k\ge c_0>0$,
the ratio $\|(A_k-A)\phi\|/\gamma_k$
may diverge even when
$\|P_k-P^\infty\|\to 0$.
Open Problem~\ref{prob:full-iff}.
\end{remark}

\begin{remark}[Rank-one structure
  is essential]
\label{rem:rank-one-essential}
The necessity direction uses rank-one
structure to extract
$\|v_k-e^{i\theta_k}\phi\|\to 0$
from $\|P_k-P^\infty\|\to 0$.
For higher-multiplicity eigenspaces,
additional assumptions are required.
\end{remark}

\begin{remark}[Novelty: one-vector
  perturbation principle in SOT regime]
\label{rem:abstract-novelty}
Classical perturbation theory
(Davis--Kahan~\cite{DavisKahan1970},
Kato~\cite{Kato1966},
Chatelin~\cite{Chatelin1983},
Osborn~\cite{Osborn1975})
uses global norm control
$\|A_k-A\|\to 0$ or norm-resolvent
convergence to estimate
$\|P_k - P^\infty\|$.
The present result operates in a
strictly weaker regime: no global norm
control is available or assumed,
and projection stability is characterized
by the single vector-local quantity
$\|(A_k-A)\phi\|/\gamma_k$.

The key distinction is not a
re-labelling: classical theory requires
$\|A_k-A\|$ as input to bound the
residual; here the residual
$\|(A_k-A)\phi\|$ is the direct
observable and no global operator norm
enters the sufficient or necessary
conditions.
This goes beyond global-norm-based
perturbation theory in the SOT-only
regime.
The reason it works is structural:
rank-one projections have a single
degree of freedom, and the mechanism
chain (Lem.~\ref{lem:mechanism-chain})
is exact.
\end{remark}

\begin{proposition}[Counterexample:
  ratio blow-up implies projection failure,
  quantitatively]
\label{prop:counterexample}
There exist bounded self-adjoint
$\{A_k\}$, $A$ on $\ell^2$ with
$A_k\xrightarrow{\mathrm{SOT}} A$,
$\lambda_k\to\lambda$, such that
\begin{equation}\label{eq:ratio-blowup}
  r_k :=
  \frac{\|(A_k-A)\phi\|}{\gamma_k}
  \equiv \frac{1}{\sqrt{2}},
\end{equation}
and for all large $k$:
\begin{equation}\label{eq:proj-blowup}
  \|P_k - P^\infty\|
  \ge \tfrac{1}{2}
  = \tfrac{1}{\sqrt{2}}\cdot r_k.
\end{equation}
\end{proposition}
\begin{proof}
Set $\phi=e_1$, $\lambda=0$,
$\gamma_k:=k^{-1/2}$,
$v_k:=\tfrac{1}{\sqrt{2}}(e_1+e_2)$,
\[
  A:=\sum_{n\ge 2}
  n^{-1}(e_n\otimes e_n),
  \quad
  A_k:=\gamma_k(v_k\otimes v_k)
  +\sum_{n\ge 2}
  n^{-1}(e_n\otimes e_n).
\]
$(A_k-A)e_1
= \gamma_k\ip{v_k}{e_1}v_k
= \tfrac{\gamma_k}{\sqrt{2}}\cdot
\tfrac{1}{\sqrt{2}}(e_1+e_2)
= \tfrac{\gamma_k}{2}(e_1+e_2)$,
so... more precisely:
$(A_k-A)e_1
= \gamma_k(v_k\otimes v_k)e_1
= \gamma_k \ip{v_k}{e_1}v_k
= \tfrac{\gamma_k}{\sqrt{2}}\cdot
\tfrac{1}{\sqrt{2}}(e_1+e_2)$,
hence $\|(A_k-A)\phi\|
= \tfrac{\gamma_k}{\sqrt{2}}\cdot
\sqrt{2}\cdot\tfrac{1}{\sqrt{2}}
= \tfrac{\gamma_k}{\sqrt{2}}
\cdot\sqrt{(\tfrac{1}{\sqrt{2}})^2
+(\tfrac{1}{\sqrt{2}})^2}
= \gamma_k/\sqrt{2}$
... let us compute directly:
$\ip{v_k}{e_1} = \tfrac{1}{\sqrt{2}}$,
so $(A_k-A)e_1
= \tfrac{\gamma_k}{\sqrt{2}}v_k
= \tfrac{\gamma_k}{\sqrt{2}}\cdot
\tfrac{1}{\sqrt{2}}(e_1+e_2)
= \tfrac{\gamma_k}{2}(e_1+e_2)$;
$\|(A_k-A)\phi\|
= \tfrac{\gamma_k}{2}\sqrt{2}
= \tfrac{\gamma_k}{\sqrt{2}}$.
So $r_k = 1/\sqrt{2}$ for all $k$.

The eigenvector of $A_k$ for
$\lambda_k=\gamma_k$ is $v_k$;
$P_k = v_k\otimes v_k$.
$\|P_k-P^\infty\|
\ge|\ip{(P_k-P^\infty)e_1}{e_1}|
= |\tfrac{1}{2}-1|
= \tfrac{1}{2}
= \tfrac{1}{\sqrt{2}}\cdot r_k$.
\end{proof}

\begin{remark}[Structural necessity of
  the ratio]
\label{rem:counterexample-meaning}
The bound \eqref{eq:proj-blowup}
is not coincidental: it follows directly
from step (M1) in
Lem.~\ref{lem:mechanism-chain}.
If the ratio $r_k$ stays bounded
away from zero, then
$\|(I-P_k)\phi\|$ stays bounded away
from zero, and by (M1) so does
$\|P_k-P^\infty\|$.
This makes the ratio condition
\eqref{eq:abstract-ratio} not just
sufficient but \emph{structurally
necessary} in the mechanism chain:
the bound is tight.
\end{remark}

%%══ 5. PSWF APPLICATION
\section{PSWF Application: Main Theorems}
\label{sec:main}

This section reduces PSWF projection
stability entirely to Gap-S and
SOT-Faster.
No further PSWF-specific input is needed
beyond Lem.~\ref{lem:inherited}.
Assumption~S is verified by
Lem.~\ref{lem:inherited}(A3)
and Lem.~\ref{lem:eigenvalue-matching}.
Thms.~\ref{thm:convergence},
\ref{thm:rate} apply
Thm.~\ref{thm:abstract} with
$A_k=\mathcal{H}_{c_k}$,
$A=\Hlim$,
$\phi=\Phi_n^{(\infty)}$,
$\lambda=\lambda_n^{(\infty)}$,
$\lambda_k=\lambda_n^{(c_k)}$.

\begin{lemma}[Gap-S supplies gap bound]
\label{lem:gap-bridge}
Under Hyp.~\ref{hyp:gap-structural},
$\gamma_k \ge g(c_k)$ for each
$n \in W_{c_k}$.
\end{lemma}
\begin{proof}
Hyp.~\ref{hyp:gap-structural} gives
$|\lambda_n^{(c_k)}
-\lambda_{n\pm1}^{(c_k)}|\ge g(c_k)$;
eigenvalues are simple and ordered.
\end{proof}

\begin{theorem}[Projection convergence]
\label{thm:convergence}
Under Lem.~\ref{lem:inherited},
Hyps.~\ref{hyp:gap-structural},
\ref{hyp:sot-faster}:
for each fixed $n\in W_{c_k}$,
$\|P_n^{(c_k)} - P_n^{(\infty)}\|\to 0$.
In particular,
$\Phi_n^{(c_k)}\to\Phi_n^{(\infty)}$
strongly (up to phase).
\end{theorem}

\begin{theorem}[Rate]
\label{thm:rate}
Under Hyp.~\ref{hyp:sot-rate}
additionally, for $n\le\kappa c_k$,
$n\in W_{c_k}$, uniformly in $n$:
\[
  \|P_n^{(c_k)} - P_n^{(\infty)}\|
  \le
  \frac{C\,C_\kappa\,c_k^{-1/4}}{g(c_k)}.
\]
If $g(c)\sim(\log c)^{-1}$:
rate $O(c_k^{-1/4}\log c_k)$.
\end{theorem}

%%══ 6. PROOFS
\section{Proofs}
\label{sec:proj}

\begin{lemma}[Approx.\ eigenvector
  $\Rightarrow$ projection bound]
\label{lem:approx-eig}
Let $A$ bounded self-adjoint,
$\lambda$ isolated and simple,
$\gamma>0$, $P$ rank-one.
For $\|\phi\|=1$ and
$\|(A-\lambda)\phi\|\le\varepsilon$:
$\|(I-P)\phi\|\le\varepsilon/\gamma$.
\end{lemma}
\begin{proof}
$(A-\lambda)\phi
=(A-\lambda)(I-P)\phi$;
spectral theorem gives
$\|(I-P)\phi\|\le\varepsilon/\gamma$.
\end{proof}

\begin{proof}[Proof of
  Thm.~\ref{thm:convergence}]
$\tilde\varepsilon_n(k)
:=\|(\mathcal{H}_{c_k}-\lambda_n^{(c_k)})
\Phi_n^{(\infty)}\|
\le\varepsilon_n(k)+C_\kappa c_k^{-1/4}$.
\begin{equation}\label{eq:ratio-to-zero}
  \frac{\tilde\varepsilon_n(k)}{g(c_k)}
  \le
  \underbrace{\frac{\varepsilon_n(k)}
    {g(c_k)}}_{\to 0
    \,(\text{SOT-Faster})}
  +\underbrace{\frac{C_\kappa c_k^{-1/4}}
    {g(c_k)}}_{\to 0
    \,(\text{Rem.}\,\ref{rem:gap-implies-o})}
  \to 0.
\end{equation}
Apply Lem.~\ref{lem:approx-eig}
with $\gamma_k\ge g(c_k)$;
rank-one identity completes.
\end{proof}

\begin{proof}[Proof of Thm.~\ref{thm:rate}]
SOT-Rate gives
$\tilde\varepsilon_n(k)
\le(C'+C_\kappa)c_k^{-1/4}$;
divide by $g(c_k)$.
\end{proof}

\begin{remark}[No accumulation]
\label{rem:nacc}
Gap-S implies finitely many
$\lambda_n^{(\infty)}$ in any
subinterval of $(0,1)$.
\end{remark}

\begin{problem}[Sharp gap scale]
\label{prob:sharp-gc}
Prove Gap-S with
$g(c)\sim(\log c)^{-1}$
via sine-kernel repulsion.

\label{prob:sot-faster}
Prove $\varepsilon_n(k)=o(g(c_k))$.
\end{problem}

\label{prob:full-iff}
Characterize when
$\|P_k-P^\infty\|\to 0$
$\Rightarrow$
$\|(A_k-A)\phi\|/\gamma_k\to 0$
for $\gamma_k\to 0$.
\end{problem}

\begin{problem}[SOT-Rate]
\label{prob:sot-rate}
$\varepsilon_n(k)=O(c_k^{-1/4})$.
\end{problem}

\label{prob:off-diagonal}
Bound
$|\ip{\Phi_n^{(\infty)}}
{\Phi_m^{(c_k)}}|$
for $m\ne n$.
\end{problem}

  convergence]

\end{problem}
