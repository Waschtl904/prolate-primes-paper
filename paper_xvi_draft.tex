\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases, A$_2$-Fold Singularities,\\
  and the Airy Normal Form}

\author{[Author]}

\begin{abstract}
This paper works in the \emph{one-dimensional setting} throughout
($\beta,u\in\mathbb{R}$).
A cubic WKB phase $c_3(c)\beta^3$
produces a degenerate stationary point
of order~2, which is an A$_2$-fold
singularity of the canonical relation
$\mathcal{C}$ in the sense of
Arnol'd's classification \cite{Arnold1972}.
The Airy distribution is the universal
local normal form of this singularity,
via the CFU reduction \cite{CFU1957}
treated as a Fourier integral operator
construction (Remark~\ref{rem:FIO-language}).
Three results are established,
each explicitly conditional on
Assumption~\ref{ass:microlocal}
(uniform A$_2$-stability,
conditions (i)--(vi)):
\begin{enumerate}[\rm(I)]
\item \textit{(Local Airy normal form.)}
  Theorem~\ref{thm:airy-normalform}:
  $\Phi$ admits a microlocal representation
  modulo smoothing operators in Airy
  normal form microlocally near the
  fold point, with transported principal
  symbol $a_{\mathrm{CFU}}\in S^{-1/2}_{1,0}(c)$
  and rapidly decaying remainder.
\item \textit{(Pointwise decay.)}
  Corollary~\ref{cor:pointwise}:
  $|\Phi(u;c)|\le C(1+|u|)^{-3/4}$
  uniformly.
  The exponent $-3/4$ is the intrinsic
  Airy microlocal exponent of an A$_2$-fold
  with conormal amplitude of order $-1/2$
  (Remark~\ref{rem:exponent-origin}).
\item \textit{(Ces\`aro improvement.)}
  Corollary~\ref{cor:global-bound}:
  $|\int_0^U\Phi|=O(U^{1/4})$.
\end{enumerate}
The forward direction is rigorous
under (i)--(vi);
the converse requires the non-standard
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\maketitle

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
We work in one dimension ($\beta,u\in\mathbb{R}$).
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be Lagrangian,
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the projection to the right factor.
$\mathcal{C}$ has an \emph{A$_2$-fold
singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$;
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$.
\end{enumerate}
For $\Psi(\beta,u)=\Theta(\beta;c)+u\beta$
this reduces to $\Theta''(0;c)=0$,
$\Theta'''(0;c)\ne 0$;
the canonical relation is locally
$\mathcal{C}=\{(\beta,\Theta'(\beta;c)+u;u,\beta)\}$.
By Arnol'd's classification \cite{Arnold1972},
an oscillatory integral with an A$_2$-fold
admits a microlocal representation
modulo smoothing in Airy normal form
near $p_0$ (Golubitsky--Guillemin \cite{Golub1973}).
\end{definition}

\begin{proposition}[Category clarification:
  $I^m$ vs.\ $I^{m,l}$]
\label{prop:category}
Two amplitude classes appear in this paper:
\begin{enumerate}[\rm(1)]
\item \textit{H\"ormander FIO class $I^m$}
  \cite{Hormander1983}, Ch.~25:
  smooth amplitudes; standard
  stationary-phase theory applies.
\item \textit{Melrose--Uhlmann class
  $I^{m,l}(\Lambda_0,\Lambda_1)$}
  \cite{MelroseU1979}:
  conormal singularities at the critical
  point; required when the amplitude is
  singular at the stationary point.
\end{enumerate}
The CFU map preserves the class
$I^{m,l}(\Lambda_0,\Lambda_1)$
under the canonical transformations
induced by CFU---not the decomposition
into individual classes.
All amplitude stability claims here
are in the $I^{m,l}$ sense.
\end{proposition}

\begin{remark}[Symbol class chain under CFU]
\label{rem:symbol-chain}
The transported amplitude passes through
three levels:
\begin{enumerate}[\rm(a)]
\item $|\beta|^{-1/2}\chi
  \in I^{-1/4,0}(\Lambda_0,\Lambda_1)$
  (conormal class before CFU);
\item after the CFU canonical transformation
  $\beta\mapsto t(\beta,u;c)$,
  the image lies in
  $I^{m,l}(\Lambda_0,\Lambda_1)$
  (Proposition~\ref{prop:category},
  Lemma~\ref{lem:conormal-stability});
\item restriction to the stationary set
  $\tilde{t}=0$ produces the
  \emph{transported principal symbol}
  $a_{\mathrm{CFU}}(u;c)$,
  which lies effectively in
  $S^{-1/2}_{1,0}(c)$
  (Definition~\ref{def:symbol-class})
  due to the $|\beta|^{-1/2}$-order
  of the original conormal distribution.
\end{enumerate}
This is not a silent class jump:
the $S^{-1/2}$ decay of $a_{\mathrm{CFU}}$
is the direct image of the
$I^{-1/4,0}$-order under CFU restriction
to the principal symbol.
All occurrences of $a_{\mathrm{CFU}}$ below
refer to this transported principal symbol.
\end{remark}

\begin{remark}[FIO equivalence language]
\label{rem:FIO-language}
All statements of the form
``admits a microlocal representation
in Airy normal form''
mean:
\emph{$\Phi$ is microlocally equivalent,
modulo smoothing, to an Airy model
microlocally near the fold point,
via a local canonical transformation
$\beta\mapsto t(\beta,u;c)$.}
This is never a pointwise or global
phase identity.
The Maslov factor is absorbed into
$R(u;c)$ in~\eqref{eq:airy-nf}
(H\"ormander \cite{Hormander1983},
Ch.~25.1).
\end{remark}

\subsection{Fold amplitude, partition,
  and conormal singularity}

The fold amplitude is:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u;c)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta;c)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta.
\end{equation}
Fix a smooth partition of unity
$1 = \phi_{\mathrm{fold}} + \phi_{\mathrm{far}}$
where $\phi_{\mathrm{fold}}$ is supported
near $\beta=0$ (the fold point)
and $\phi_{\mathrm{far}}$ is supported
away from $\beta=0$.
The two regimes are:
\begin{itemize}
\item \textit{Fold zone:}
  $\Phi_{\mathrm{fold}}$ with
  $\phi_{\mathrm{fold}}$---degenerate
  stationary phase, treated by CFU
  + Airy normal form
  (Section~\ref{sec:reduction});
\item \textit{Far zone:}
  $\Phi_{\mathrm{far}}$ with
  $\phi_{\mathrm{far}}$---non-degenerate
  stationary phase,
  $|\partial_\beta^2\Psi|\ge c_0'>0$,
  treated by standard FIO theory.
\end{itemize}
The transition between regimes is
sharp at the microlocal level via
the cutoff $\phi_{\mathrm{fold}}$;
no overlap ambiguity arises modulo
smoothing operators.
\medskip

\noindent\textbf{Conormal singularity.}
$|\beta|^{-1/2}\chi
\in I^{-1/4,0}(\Lambda_0,\Lambda_1)$
(Remark~\ref{rem:symbol-chain}),
where $\Lambda_0$ is the zero-section
and $\Lambda_1$ is the conormal to
$\{\beta=0\}$.
This class is required because the
amplitude singularity occurs exactly
at the critical point.

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
Under Assumption~\ref{ass:fold-local},
$|\Phi_{\mathrm{far}}(u;c)|=O(|u|^{-1/2})$
uniformly in $c\ge c_0$.
\end{theorem}

\begin{proof}
On the far zone $\phi_{\mathrm{far}}$,
the phase is non-degenerate;
van~der~Corput in the $I^{m,l}$ setting
(H\"ormander \cite{Hormander1983},
Ch.~7.7 + 18.1)
gives $O(|u|^{-1/2})$.
\end{proof}

\section{CFU Reduction and Airy Structure}
\label{sec:reduction}

\begin{assumption}[Local stable A$_2$-fold]
\label{ass:fold-local}
$\Theta(\beta;c)=c_3(c)\beta^3+O(\beta^4)$,
$c_3(c)\ne 0$, and:
\begin{enumerate}[\rm(F1)]
\item $\Theta''(\beta;c)\ne 0$
  for $\beta\ne 0$ near $0$.
\item $\int|\beta|^{-1/2}
  e^{i\Theta(\cdot;c)}\chi\,d\beta
  \not\equiv 0$ near $0$.
\item Exactly one critical point
  ($\beta=0$) on $\mathrm{supp}\,\chi$.
\end{enumerate}
\end{assumption}

\begin{lemma}[Degenerate stationary phase
  with conormal weight \Rig]
\label{lem:degenerate-phase}
Define
\begin{equation}\label{eq:model-integral}
  I(\mu) :=
  \int_{\mathbb{R}}
  |t|^{-1/2}\,
  e^{i(\frac{1}{3}t^3 - \mu t)}
  \,dt,
  \qquad |\mu|\ge 1.
\end{equation}
Then $|I(\mu)| \le C\,|\mu|^{-3/4}$
(see Remark~\ref{rem:exponent-origin}).
\end{lemma}

\begin{proof}
By the $I^{m,l}$ calculus
(Proposition~\ref{prop:category};
Melrose--Uhlmann \cite{MelroseU1979}),
$|t|^{-1/2}$ belongs to the
intersecting Lagrangian class,
and $I(\mu)$ is microlocally equivalent,
modulo smoothing, near the fold point to
\[
  I(\mu)
  = a_{\mathrm{CFU}}(\mu)\,
  \mathrm{Ai}(\mu) + r(\mu),
\]
where $a_{\mathrm{CFU}}\in S^{-1/2}_{1,0}$
is the transported principal symbol
(Remark~\ref{rem:symbol-chain})
and $|r(\mu)|\le C_N|\mu|^{-N}$.
Airy asymptotics (Olver \cite{Olver1974},
Ch.~11) give
$|\mathrm{Ai}(\mu)|\le C|\mu|^{-1/4}$.
Hence
$|I(\mu)| \le C'|\mu|^{-1/2}\cdot
C|\mu|^{-1/4} + C_N|\mu|^{-N}
= C''|\mu|^{-3/4}$.
\end{proof}

\begin{remark}[Origin of the $-3/4$ exponent]
\label{rem:exponent-origin}
The exponent $-3/4$ has a single
geometric source in the A$_2$-fold
structure, manifesting via two
analytically distinct---but
geometrically unified---factors:
\[
  \underbrace{
    |a_{\mathrm{CFU}}(\mu)|
    \lesssim|\mu|^{-1/2}
  }_{\text{conormal amplitude, order }(-1/2)}
  \;\times\;
  \underbrace{
    |\mathrm{Ai}(\mu)|
    \lesssim|\mu|^{-1/4}
  }_{\text{A}_2\text{-fold / Airy decay}}
  \;=\;
  |\mu|^{-3/4}.
\]
The two factors are not independent:
both are determined by the A$_2$-fold
geometry via CFU---the fold forces the
Airy normal form, and the $I^{m,l}$-order
of the amplitude is set by the conormal
structure of the fold.
The $-3/4$ exponent is therefore their
\emph{intrinsic combination} within a
single geometric framework.
Comparison: fold alone
($a_{\mathrm{CFU}}\in S^0$)
gives $|\mu|^{-1/4}$;
conormal alone (non-degenerate phase)
gives $|\mu|^{-1/2}$
(Theorem~\ref{thm:endpoint}).
\end{remark}

\begin{lemma}[Conormal amplitude
  stability under CFU \Rig]
\label{lem:conormal-stability}
Let $a_0=|\beta|^{-1/2}\chi
\in I^{-1/4,0}(\Lambda_0,\Lambda_1)$.
Under the CFU canonical transformation
$\beta\mapsto t(\beta,u;c)$,
the transported amplitude preserves
the class $I^{m,l}(\Lambda_0,\Lambda_1)$,
uniformly in $(u,c)$ under
condition~(vi) of
Assumption~\ref{ass:microlocal}.
\end{lemma}

\begin{proof}
The CFU map has
$|\det(dt/d\beta)|\ge\delta_5>0$
and uniform symbol bounds.
The class $I^{m,l}$ is preserved
under such canonical transformations
(Melrose--Uhlmann \cite{MelroseU1979};
Guillemin--Uhlmann \cite{GuillemU1981},
Lemma~2.1).
\end{proof}

\begin{proposition}[CFU: Airy normal form
  modulo smoothing \Rig]
\label{prop:CFU}
Under Assumption~\ref{ass:fold-local},
on the fold zone $\phi_{\mathrm{fold}}$:
\begin{enumerate}[\rm(1)]
\item $\Theta''(0;c)=0$,
  $\Theta'''(0;c)=6c_3(c)\ne 0$.
\item \textit{(Airy normal form.)}
  CFU \cite{CFU1957} provides
  $\beta\mapsto t(\beta,u;c)$,
  $\mu(u;c)$, $R(u;c)$ such that:
  \begin{equation}\label{eq:CFU-integral}
    \Phi_{\mathrm{fold}}(u;c)
    = e^{i(\alpha_*u+R(u;c))}
    \bigl[
    a_{\mathrm{CFU}}(u;c)\,
    \mathrm{Ai}(\mu(u;c))
    + \mathcal{R}(u;c)
    \bigr],
  \end{equation}
  with $a_{\mathrm{CFU}}\in S^{-1/2}_{1,0}(c)$
  the transported principal symbol
  (Remark~\ref{rem:symbol-chain};
  Lemma~\ref{lem:conormal-stability}),
  $|\mathcal{R}|\le C_N(1+|u|)^{-N}$.
\item \textit{(Far zone.)}
  On $\phi_{\mathrm{far}}$:
  standard stationary phase gives
  $|\Phi_{\mathrm{far}}|\le C|u|^{-1/2}$
  (Theorem~\ref{thm:endpoint}).
\end{enumerate}
\end{proposition}

\begin{definition}[Parameter-dependent
  symbol class]
\label{def:symbol-class}
\begin{equation}
  S^m_{1,0}(c)
  :=
  \Bigl\{ a(u;c)
  \;\Big|\;
  |\partial_u^\alpha a|
  \le C_\alpha (1+|u|)^{m-|\alpha|},
  \text{ uniformly in } c\ge c_0
  \Bigr\},
\end{equation}
in the sense of semiclassical analysis
(Zworski \cite{Zworski2012}, Ch.~4).
$c$ is not a phase-space variable.
\end{definition}

\section{Main Results}
\label{sec:results}

\begin{assumption}[Uniform A$_2$-stability
  \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$, with:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta;c)
  =c_3(c)\beta^3+O(\beta^4)$
  in a $c$-independent neighbourhood;
\item $c_3(c)\ge\delta_0>0$;
\item $|\Theta''(\beta^*(u);c)|
  \ge\delta_1|u|^{1/2}$
  (rescaling hypothesis, motivated by
  PSWF phase asymptotics via Olver
  \cite{Olver1974}: encodes the growth
  rate of the second derivative away
  from the fold);
\item $|\partial_u^k J(t,u;c)|
  \le C_k(1+|u|)^{-1/2-k}$;
\item no secondary stationary points;
\item $|\det(dt/d\beta)|\ge\delta_5>0$;
  $|\partial_\beta^j\partial_u^k t|
  \le C_{jk}$ uniformly.
\end{enumerate}
\end{assumption}

\begin{remark}[Scope of the results]
\label{rem:scope}
The main results are microlocal Airy
normal forms near the fold point,
under the strong uniform non-degeneracy
of Assumption~\ref{ass:microlocal}.
The non-trivial content is the
consistent $I^{m,l}$ treatment of
the conormal singularity;
condition~(vi) is the key open
hypothesis for PSWF
(Open Problem~\ref{prob:PSWF-microlocal}).
\end{remark}

\begin{theorem}[Conditional local Airy
  normal form \Rig]
\label{thm:airy-normalform}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
For $|u|\ge\delta_2$,
$\Phi(u;c)$ admits a microlocal
representation modulo smoothing
in Airy normal form microlocally
near the fold point:
\begin{equation}\label{eq:airy-nf}
  e^{i(\alpha_*u+R(u;c))}\bigl[
  a_{\mathrm{CFU}}(u;c)\,
  \mathrm{Ai}(\mu(u;c))
  + r(u;c)\bigr],
\end{equation}
where $a_{\mathrm{CFU}}\in S^{-1/2}_{1,0}(c)$
is the transported principal symbol,
$|r|\le C_N(1+|u|)^{-N}$,
and $R$ absorbs the Maslov factor.
\end{theorem}

\begin{proof}[Proof sketch]
Fold zone: Proposition~\ref{prop:CFU}(2)
+ Lemma~\ref{lem:conormal-stability}.
Far zone: Proposition~\ref{prop:CFU}(3).
\end{proof}

\begin{corollary}[Pointwise decay \Cond]
\label{cor:pointwise}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
\begin{equation}
  |\Phi(u;c)|\le C(1+|u|)^{-3/4},
  \quad\forall u\in\mathbb{R},
\end{equation}
uniformly in $c\ge c_0$.
The exponent $-3/4$ is the intrinsic
Airy microlocal exponent
(Remark~\ref{rem:exponent-origin}).
\end{corollary}

\begin{proof}
Fold zone:
Theorem~\ref{thm:airy-normalform}
+ Lemma~\ref{lem:degenerate-phase}
(Remark~\ref{rem:exponent-origin}).
Compact zone $|u|\le 1$: continuity.
Far zone: Theorem~\ref{thm:endpoint}.
\end{proof}

\begin{remark}[Non-oscillatory mechanism]
\label{rem:oscillation}
The Ces\`aro bound uses only
$|\Phi|\le C(1+|u|)^{-3/4}$
and $L^1$-integration ---
pure integrability, no cancellation.
\end{remark}

\begin{corollary}[Ces\`aro improvement
  \Cond]
\label{cor:global-bound}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
\[
  \left|\int_0^U\Phi(u;c)\,du\right|
  = O(U^{1/4}),
\]
improving upon $O(U^{1/2})$
by Remark~\ref{rem:oscillation}.
\end{corollary}

\subsection{Converse and FIO rigidity}

\begin{assumption}[Microlocal
  reconstruction (non-standard,
  conjectural) \Cond]
\label{hyp:FIO-rigidity}
\textit{Forward CFU (rigorous):}
Proposition~\ref{prop:CFU} gives
A$_2$-fold $\Rightarrow$ Airy class.
\textit{Inverse reconstruction
(conjectural):}
Suppose $\Phi$ is in the Airy class and:
(a) $\sigma(g\mapsto\Phi)|_\mathcal{C}\ne 0$;
(b) $g\mapsto\Phi$ injective mod smoothing;
(c) $\mathrm{WF}(\Phi)
  \subseteq\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
Then $\mathcal{C}$ has an A$_2$-fold
uniformly in $c$.
This is an inverse microlocal conjecture,
not a consequence of standard FIO theory
\cite{Zworski2012,Hormander1983};
verification for PSWF: Paper~XVII.
\end{assumption}

\begin{corollary}[Conditional equivalence]
\label{cor:conditional-equiv}
Under Assumption~\ref{hyp:FIO-rigidity}:
Airy normal form $\Rightarrow$
Assumption~\ref{ass:microlocal}
(modulo the inverse conjecture).
The forward direction
(Proposition~\ref{prop:CFU})
holds rigorously without
Assumption~\ref{hyp:FIO-rigidity}.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item $c_3(c)\ge\delta_0>0$:
  expected from Olver \cite{Olver1974};
  uniform remainder bounds required.
\item $\theta''\ne 0$ for
  $\alpha\ne\alpha_*$, uniformly:
  plausible but not established.
\item Condition~(vi) for the
  PSWF CFU Jacobian:
  technically hardest;
  no uniform result currently available.
\end{enumerate}
These conditions are partially supported
by WKB asymptotics, but no uniform
microlocal analysis is currently
available in the literature.
\end{problem}
