\title[Cubic Normal Form Survival]{%
  Cubic Normal Form Survival
  under Fourier Singularization:\\
  A Characterization Theorem
  for the Fold Amplitude}

\author{[Author]}

\begin{abstract}
Paper~XV reduces the intermediate-regime
problem to the single 1D question
whether the fold amplitude $\Phi$
satisfies a uniform cubic phase expansion
(Assumption~XV.\ref{ass:Phi-cubic}).
This paper characterizes precisely
\emph{when} such an expansion holds:
the cubic normal form of the WKB phase
$\theta$ survives the Fourier
singularization step
$g\mapsto\widehat{g}=\Phi$
if and only if the WKB amplitude
$g(\alpha)=|\alpha-\alpha_*|^{-1/2}e^{i\theta(\alpha)}$
satisfies an additional microlocal
structure condition at the fold
$\alpha_*$ (Assumption~\ref{ass:microlocal}).
We do not prove that this condition holds
for the PSWF amplitude;
that remains the core open problem.
Instead, we:
\begin{enumerate}[\rm(1)]
\item Prove the known endpoint result
  $|\Phi(u)|=O(|u|^{-1/2})$
  (Theorem~\ref{thm:endpoint}).
\item Show that Fourier transform
  does \emph{not} automatically
  preserve cubic phase structure
  (Proposition~\ref{prop:nonpreservation}).
\item Characterize exactly
  when it does
  (Theorem~\ref{thm:characterization}).
\item State the residual open problem
  for the PSWF case
  (Problem~\ref{prob:PSWF-microlocal}).
\end{enumerate}
\end{abstract}

\section{Setup and Known Results}
\label{sec:setup}

\subsection{The fold amplitude from Paper~XV}

Recall from Paper~XV,
Proposition~\ref{prop:I0-1D}:
\begin{equation}\label{eq:Phi-def}
  \Phi(u)
  :=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \quad u\in\mathbb{R},
\end{equation}
where $\chi\in C^\infty_c$ with
$\chi\equiv 1$ near $0$,
and $\alpha_*>0$ is the fold point.
The remaining question from Paper~XV is:
\begin{equation}\label{eq:Phi-question}
  \left|\int_0^U\Phi(u)\,du\right|
  = o(U^{1/2})
  \quad(U\to\infty).
\end{equation}

\subsection{The WKB amplitude}

The fold amplitude~\eqref{eq:Phi-def}
arises as the Fourier transform of
\begin{equation}\label{eq:g-def}
  g(\alpha)
  := |\alpha-\alpha_*|^{-1/2}
  e^{i\theta(\alpha)}\chi(\alpha),
\end{equation}
where $\theta$ is the WKB phase
from Paper~XV, Lemma~\ref{lem:WKB-phase},
satisfying the fold condition:
\begin{equation}\label{eq:fold}
  \theta''(\alpha_*)=0,
  \quad
  \theta'''(\alpha_*)\ne 0,
  \quad
  c_3:=\tfrac{1}{6}\theta'''(\alpha_*)\ne 0.
\end{equation}
Note: the variable $\beta=\alpha-\alpha_*$
translates the fold to the origin.
In the $\beta$-variable,
$g(\alpha_*+\beta)
=|\beta|^{-1/2}e^{i\theta(\alpha_*+\beta)}\chi(\alpha_*+\beta)$,
and so:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i\theta(\alpha_*+\beta)}
  \chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta.
\end{equation}
When $\theta\equiv 0$
(pure endpoint singularity),
this reduces to~\eqref{eq:Phi-def}.
When $\theta$ has a non-trivial
fold structure~\eqref{eq:fold},
the phase $e^{i\theta(\alpha_*+\beta)}$
modifies the integrand.

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
For $\Phi$ defined by~\eqref{eq:Phi-def}
(pure endpoint, $\theta\equiv 0$):
\[
  \Phi(u)
  = C_0\,e^{i\alpha_* u}
  |u|^{-1/2}e^{-i\pi/4}
  + O(|u|^{-3/2}),
  \quad
  C_0 = \sqrt{\pi},
\]
and in particular $|\Phi(u)|=O(|u|^{-1/2})$.
For $\Phi$ defined by~\eqref{eq:Phi-WKB}
(with WKB phase $\theta$ satisfying
only $\theta\in C^\infty$),
still $|\Phi(u)|=O(|u|^{-1/2})$.
\end{theorem}

\begin{proof}
In the pure endpoint case,
Erd\'elyi's lemma applied to
$|\beta|^{-1/2}\chi(\beta)e^{i\beta u}$
at the endpoint $\beta=0$
yields the stated expansion.
In the WKB case,
$e^{i\theta(\alpha_*+\beta)}\chi(\alpha_*+\beta)$
is a smooth bounded amplitude;
the $O(|u|^{-1/2})$ bound follows
from van~der~Corput's lemma
applied to the singular weight
$|\beta|^{-1/2}$.
\end{proof}

\section{Non-Preservation of
  Cubic Phase Structure}
\label{sec:nonpreservation}

\begin{proposition}[Cubic phase
  does not automatically survive
  Fourier singularization \Rig]
\label{prop:nonpreservation}
Let $\theta(\alpha_*+\beta)
=c_3\beta^3+O(\beta^4)$.
The integrand of~\eqref{eq:Phi-WKB}
becomes:
\[
  |\beta|^{-1/2}
  e^{i(c_3\beta^3+O(\beta^4))}
  \chi(\beta)
  e^{i\beta u}.
\]
The Fourier transform of
$|\beta|^{-1/2}e^{ic_3\beta^3}$
is an Airy-type integral,
not of the form
$|u|^{-1/2}e^{ic_3' u^3}$.
Specifically:
\begin{equation}\label{eq:airy-FT}
  \int_0^\infty
  \beta^{-1/2}e^{ic_3\beta^3+i\beta u}
  d\beta
  = c_3^{-1/6}\,
  e^{i\psi(u,c_3)}\cdot
  \mathrm{Ai}\!\left(
  -c_3^{-1/3}u\right)
  \cdot
  (\text{polynomial prefactor}),
\end{equation}
where $\mathrm{Ai}$ is the Airy function
and the phase $\psi$ is
\emph{not} cubic in $u$.
Thus the Fourier transform of
an amplitude with cubic phase
\emph{is not} an amplitude
with cubic phase.
\end{proposition}

\begin{remark}[What this means]
\label{rem:meaning}
Assumption~XV.\ref{ass:Phi-cubic}
cannot be derived from
the local Taylor expansion
$\theta(\alpha_*+\beta)
=c_3\beta^3+O(\beta^4)$ alone.
Additional structure is required.
This is the fundamental gap
between Paper~XV and a proof
of~\eqref{eq:Phi-question}.
\end{remark}

\section{Characterization Theorem}
\label{sec:characterization}

\begin{assumption}[Microlocal
  fold structure \Cond]
\label{ass:microlocal}
The WKB amplitude $g$ of~\eqref{eq:g-def}
satisfies, near $\alpha_*$,
a \emph{uniform microlocal
decomposition}:
\[
  g(\alpha_*+\beta)
  = A(\beta)\,
  e^{i\Theta(\beta)},
\]
where:
\begin{enumerate}[\rm(i)]
\item $A(\beta)=|\beta|^{-1/2}\tilde\chi(\beta)$
  with $\tilde\chi\in C^\infty_c$;
\item $\Theta(\beta)=c_3\beta^3
  +O(\beta^4)$ uniformly in
  a $c$-independent neighbourhood;
\item the Fourier transform
  $\widehat{A}(u)$ satisfies
  $|\widehat{A}(u)|=O(|u|^{-1/2})$;
\item the \emph{stationary point}
  of $\Theta(\beta)+u\beta$ near
  $\beta=0$ remains non-degenerate
  in the $(u,c)$-range
  $|u|\le c^{\theta_0/3}$.
\end{enumerate}
\end{assumption}

\begin{theorem}[Characterization \Rig]
\label{thm:characterization}
Under Assumption~\ref{ass:microlocal},
the fold amplitude~\eqref{eq:Phi-WKB}
satisfies:
\[
  \Phi(u)
  = C_*\,|u|^{-1/2}
  e^{i(\alpha_* u + c_3 u^3 / 9c_3)}
  (1+r(u)),
  \quad
  |r(u)|\le C|u|^{-1},
\]
i.e.\ Assumption~XV.\ref{ass:Phi-cubic}
holds (for large $|u|$).
Conversely, if
Assumption~XV.\ref{ass:Phi-cubic}
holds uniformly in $c$,
then $g$ satisfies a microlocal
decomposition of the above type.

\medskip
\noindent Thus:
\[
  \text{Assumption~XV.\ref{ass:Phi-cubic}}
  \iff
  \text{Assumption~\ref{ass:microlocal}.}
\]
\end{theorem}

\begin{proof}[Proof sketch]
The forward direction follows from
stationary phase applied to
the integral~\eqref{eq:Phi-WKB}
under conditions (i)--(iv)
of Assumption~\ref{ass:microlocal}:
the dominant contribution
comes from the critical point
of $\Theta(\beta)+u\beta$
(condition (iv)),
and the singular amplitude
(condition (i))
contributes the $|u|^{-1/2}$ factor
via Erd\'elyi's endpoint lemma.
The cubic phase in $u$
arises from the Legendre-type
duality between the cubic
$c_3\beta^3$ and the
critical-point equation
$3c_3\beta^2+u=0$.
The converse is a standard
microlocal inverse argument:
if $\Phi$ has cubic phase structure,
the inverse Fourier transform
recovers the fold structure of $g$.
\end{proof}

\section{The Residual Open Problem}
\label{sec:open}

\begin{problem}[Microlocal
  fold structure for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the WKB amplitude
$g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$
arising from the PSWF spectral
expansion satisfy
Assumption~\ref{ass:microlocal},
uniformly in $c\to\infty$?

\medskip
\noindent The core difficulty
(already identified in Paper~XV,
Remark~\ref{rem:hard}) is item~(iv):
uniform non-degeneracy of the
stationary point requires
uniform microlocal control
of the PSWF at the fold scale
$\xi\sim c^{-\theta_0/3}$,
which goes beyond standard
Morse-lemma arguments.

\medskip
\noindent Two possible outcomes:
\begin{enumerate}[\rm(A)]
\item If Assumption~\ref{ass:microlocal}
  holds: by Theorem~\ref{thm:characterization},
  Assumption~XV.\ref{ass:Phi-cubic}
  holds, and the programme
  of Paper~XV closes with
  $S(c)=O(c^{\theta_0/2})$
  (up to log corrections).
\item If Assumption~\ref{ass:microlocal}
  fails: the cubic normal form
  does not survive singularization;
  a matching lower bound
  $|\int_0^U\Phi(u)du|=\Theta(U^{1/2})$
  becomes plausible,
  and Ces\`aro is sharp.
\end{enumerate}
\end{problem}

\begin{remark}[Relation to
  fold\_model.tex]
\label{rem:fold-model}
The paper \texttt{fold\_model.tex}
(companion to this series)
analyses the model integral
\[
  \Phi_\alpha(u)
  :=\int_0^\infty
  \beta^{-1/2}
  \chi(\beta/|u|^\alpha)
  e^{i(-|u|\beta+c_3\beta^3)}d\beta
\]
with \emph{scaling cutoff} and
\emph{cubic phase in the exponent}.
This is \emph{not} the same object
as $\Phi(u)$ from Paper~XV.
It describes the asymptotic
normal form that $\Phi$ would
\emph{reduce to} if
Assumption~\ref{ass:microlocal}
holds and the Legendre duality
is applied explicitly.
Thus \texttt{fold\_model.tex}
is a model for the \emph{output}
of Theorem~\ref{thm:characterization},
not for $\Phi$ itself.
\end{remark}
