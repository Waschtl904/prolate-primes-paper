\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases Induce A$_2$-Fold Singularities\\
  Whose Normal Form Is the Airy Distribution}

\author{[Author]}

\begin{abstract}
Paper~XV reduces the intermediate-regime
problem to whether the fold amplitude~$\Phi$
satisfies an estimate improving upon
Ces\`aro.
This paper establishes the microlocal
mechanism.
The central observation is that a cubic
WKB phase $c_3\beta^3$ produces a
\emph{degenerate stationary point
of order~2} in the Fourier integral,
which is an A$_2$-fold singularity of
the canonical relation;
the Airy distribution is the universal
normal form of this singularity
(Chester--Friedman--Ursell \cite{CFU1957},
Arnold \cite{Arnold1972}).
Under the uniform stability conditions
of Assumption~\ref{ass:microlocal},
\[
  \Phi(u)
  = e^{i\alpha_* u}\,a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
  \qquad a(u,c)\in S^{-1/2}_{1,0},
\]
and $|\int_0^U\Phi(u)\,du|=O(U^{1/4})$,
strictly improving upon Ces\`aro.
The forward direction
(Assumption~\ref{ass:microlocal}
$\Rightarrow$ Airy-class bound)
is rigorous;
the converse requires the additional
global hypotheses of
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
Let $I(u)=\int e^{i\Psi(\beta,u)}A(\beta)\,d\beta$
be a one-dimensional FIO with smooth
non-degenerate phase $\Psi$.
The \emph{canonical relation} is the
Lagrangian submanifold
\[
  \mathcal{C}
  := \bigl\{(\beta,\partial_\beta\Psi;\,
  u,{-\partial_u\Psi})\bigr\}
  \subset T^*\mathbb{R}_\beta
  \times T^*\mathbb{R}_u.
\]
Denote by
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the natural projection.
We say $I$ has an \emph{A$_2$-fold
singularity at $(\beta_0,u_0)$} if:
\begin{enumerate}[\rm(i)]
\item \textit{(Rank-drop of $\pi_u$.)} 
  $\det d\pi_u\big|_{(\beta_0,u_0)}=0$;
\item \textit{(Simple fold / non-degeneracy.)}
  $d\!\left(\det d\pi_u\right)\big|_{(\beta_0,u_0)}\ne 0$.
\end{enumerate}
Conditions~(i)--(ii) define a
\emph{fold} (simple rank-drop singularity)
of $\pi_u$ at $(\beta_0,u_0)$;
this is the A$_2$-catastrophe type
in Arnold's classification \cite{Arnold1972}.
For a one-dimensional FIO with phase
$\Psi(\beta,u)=\Theta(\beta)+u\beta$:
\begin{align}\label{eq:fold-explicit}
  \det d\pi_u=0
  \;&\iff\;
  \Theta''(\beta_0)=0,
  \\
  d\!\left(\det d\pi_u\right)\ne 0
  \;&\iff\;
  \Theta'''(\beta_0)\ne 0.
\end{align}
The Airy function is the universal normal
form of this singularity class:
any FIO with an A$_2$-fold singularity
is, microlocally near $(\beta_0,u_0)$,
equivalent to a scalar multiple of
$\mathrm{Ai}$ under a local symplectic
coordinate change
(Thom--Boardman classification,
Golubitsky--Guillemin \cite{Golub1973}).
\end{definition}

\begin{remark}[Polynomial phases
  are coordinate representations]
\label{rem:phases-coordinates}
The phase formula $e^{i\Theta(\beta)}$
is coordinate-dependent:
it changes under symplectic transformations
of $T^*\mathbb{R}$.
The canonical invariant is the
Lagrangian submanifold
$\Lambda=\{(\beta,\Theta'(\beta))\}
\subset T^*\mathbb{R}$
and its Thom--Boardman stratum.
The cubic representation $e^{ic_3\beta^3}$
and the Airy distribution
$\mathrm{Ai}(c_3^{-1/3}u)$ are
\emph{coordinate representations
of the same A$_2$ Lagrangian geometry}
in the $\beta$- and $u$-variables
respectively.
\end{remark}

\begin{remark}[Microlocal transport,
  not global category invariance]
\label{rem:microlocal-transport}
It is \emph{not} claimed that a
global category $\mathcal{FIO}(A_2)$
is closed under Fourier transform.
What holds is a local statement:
\emph{near a stable A$_2$-fold point,
and under the three conditions of
Assumption~\ref{ass:microlocal}
(stable projection, no higher degeneracy,
controlled amplitude singularity),
the Fourier transform transports
the fold singularity microlocally.}
Without these conditions,
higher strata (A$_3$, cusp)
can appear after transformation.
\end{remark}

\subsection{The fold amplitude}

\begin{equation}\label{eq:Phi-def}
  \Phi(u)
  :=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \quad u\in\mathbb{R},
\end{equation}
$\chi\in C^\infty_c$, $\chi\equiv 1$
near $0$, $\alpha_*>0$ fold point.
The WKB amplitude is
\begin{equation}\label{eq:g-def}
  g(\alpha)
  := |\alpha-\alpha_*|^{-1/2}
  e^{i\theta(\alpha)}\chi(\alpha),
\end{equation}
with WKB phase satisfying
\begin{equation}\label{eq:fold-condition}
  \theta''(\alpha_*)=0,
  \quad
  c_3:=\tfrac{1}{6}\theta'''(\alpha_*)\ne 0.
\end{equation}
$\beta=\alpha-\alpha_*$,
$\Theta(\beta):=\theta(\alpha_*+\beta)$:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta.
\end{equation}

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
$|\Phi(u)|=O(|u|^{-1/2})$
for all $\theta\in C^\infty$.
For $\theta\equiv 0$:
$\Phi(u)=\sqrt{\pi}\,e^{i\alpha_*u}
|u|^{-1/2}e^{-i\pi/4}+O(|u|^{-3/2})$.
\end{theorem}

\begin{proof}
Erd\'elyi's lemma ($\theta\equiv 0$);
van~der~Corput with weight $|\beta|^{-1/2}$
(general $\theta$).
\end{proof}

\section{Cubic Phase Induces A$_2$-Fold}
\label{sec:nonpreservation}

\begin{assumption}[Stable A$_2$-fold:
  local conditions \Cond]
\label{ass:fold-local}
In addition to~\eqref{eq:fold-condition},
the phase $\Theta$ satisfies:
\begin{enumerate}[\rm(F1)]
\item \textit{(No higher degeneracy.)}
  $\Theta^{(4)}(0)$ is bounded;
  no A$_3$- or higher stratum
  degeneracy occurs:
  $\Theta''(\beta)\ne 0$
  for $\beta\ne 0$ in a neighbourhood
  of $0$.
\item \textit{(Non-cancellation of
  amplitude weight.)}
  The weight $|\beta|^{-1/2}$
  does not cancel the fold:
  the stationary-phase integral
  $\int|\beta|^{-1/2}
  e^{i\Theta(\beta)}\chi(\beta)\,d\beta$
  is not identically zero in any
  neighbourhood of $0$.
\item \textit{(Elliptic projection.)}
  The Legendre map
  $\beta\mapsto\Theta'(\beta)$
  has a locally monotone image:
  $\Theta':\mathrm{supp}\,\chi\to\mathbb{R}$
  has at most one critical point,
  at $\beta=0$.
\end{enumerate}
Under (F1)--(F3), the canonical relation
of the FIO in~\eqref{eq:Phi-WKB} has a
precisely A$_2$-fold singularity at
$(\beta,u)=(0,0)$,
and no higher-order degeneration.
\end{assumption}

\begin{proposition}[Cubic WKB phase
  $\Rightarrow$ A$_2$-fold
  $\Rightarrow$ Airy normal form \Rig]
\label{prop:nonpreservation}
Let $\Theta(\beta)=c_3\beta^3+O(\beta^4)$,
$c_3\ne 0$,
and assume~\ref{ass:fold-local}.
Let
$F(u):=\int_{\mathbb{R}}
|\beta|^{-1/2}
e^{i(\Theta(\beta)+\beta u)}
\chi(\beta)\,d\beta$.
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold structure.)} 
  The phase $\Phi_u(\beta)=\Theta(\beta)+u\beta$
  has a degenerate stationary point
  of order~2 at $(\beta_0,u_0)=(0,0)$:
  \begin{gather*}
    \Theta'(0)+0=0,
    \qquad
    \Theta''(0)=0,
    \qquad
    \Theta'''(0)=6c_3\ne 0.
  \end{gather*}
  By Definition~\ref{def:A2-class}
  and~\eqref{eq:fold-explicit}:
  $\det d\pi_u=0$ (from $\Theta''(0)=0$)
  and $d(\det d\pi_u)\ne 0$
  (from $\Theta'''(0)\ne 0$).
  Assumption~\ref{ass:fold-local}(F1)
  ensures this is a simple fold,
  not a cusp or higher stratum.
\item \textit{(Legendre degeneration
  and CFU normal form.)}
  The stationary-phase approximation
  of $F(u)$ is governed by the
  critical point $\beta^*(u)$ satisfying
  $\Theta'(\beta^*(u))=-u$.
  At $u=0$, $\beta^*=0$ and
  the Hessian $\Theta''(\beta^*)=0$
  vanishes: this is a \emph{Legendre
  degeneration of order~2},
  also called a \emph{Whitney fold}
  of the Legendre map.
  By the Thom--Boardman classification
  \cite{Golub1973},
  any smooth family of functions
  with this singularity type
  is locally equivalent
  (via a smooth coordinate change
  $\beta\mapsto t(\beta,u)$)
  to the Airy normal form:
  \begin{equation}\label{eq:CFU-nf}
    \Theta(\beta)+u\beta
    = \tfrac{1}{3}t^3 - c_3^{-1/3}u\,t
    + R(u),
  \end{equation}
  where $R(u)$ is a smooth remainder
  independent of $t$
  (Chester--Friedman--Ursell \cite{CFU1957}).
  Under this change of variables,
  Assumption~\ref{ass:fold-local}(F3)
  ensures the map is a diffeomorphism
  (locally monotone Legendre image)
  and (F2) ensures the amplitude
  $|\beta|^{-1/2}d\beta/dt$
  is non-vanishing.
  The result is:
  \begin{equation}\label{eq:airy-FT}
    F(u)
    \;\sim\;
    c_3^{-1/6}\,P(u)\,
    \mathrm{Ai}\!\bigl(-c_3^{-1/3}u\bigr),
    \quad|u|\to\infty.
  \end{equation}
\item \textit{(Stationary phase value.)}
  The large-argument
  oscillatory phase of
  $\mathrm{Ai}(-c_3^{-1/3}u)$
  for $u\to-\infty$ is,
  by the standard asymptotic expansion
  $\mathrm{Ai}(-t)\sim
  \pi^{-1/2}t^{-1/4}
  \sin(\frac{2}{3}t^{3/2}+\frac{\pi}{4})$
  for $t\to+\infty$ \cite{Olver1974},
  \begin{equation}\label{eq:stat-phase-value}
    S(u)
    := \tfrac{2}{3}c_3^{-1/2}|u|^{3/2}.
  \end{equation}
  This is the \emph{stationary phase
  value} evaluated at $\beta^*(u)$:
  $S(u)=\Theta(\beta^*(u))+u\beta^*(u)$.
  It is an asymptotic quantity,
  not a global phase function defined
  for all $u$:
  the Airy function has no global
  phase representation,
  and $S(u)$ is meaningful only
  in the large-$|u|$ oscillatory regime.
\end{enumerate}
\end{proposition}

\begin{proof}
(1): Direct computation using
$\Theta(\beta)=c_3\beta^3+O(\beta^4)$.
(2): The coordinate change
$\beta\mapsto t(\beta,u)$
solving~\eqref{eq:CFU-nf} exists
as a smooth diffeomorphism by:
(a)~at $(\beta,u)=(0,0)$,
the map $\beta\mapsto t$ is the identity
to leading order;
(b)~condition~(F1) (no higher stratum)
ensures no secondary critical points
of $t\mapsto\beta(t,u)$;
(c)~condition~(F3) (elliptic projection)
ensures the Legendre map is locally
invertible away from $\beta=0$;
(d)~the standard Thom--Boardman
stability theorem for A$_2$-type functions
\cite{Golub1973} guarantees the
coordinate change is smooth.
Condition~(F2) ensures the Jacobian
$|\beta(t,u)|^{-1/2}|d\beta/dt|\ne 0$.
(3): Standard \cite{Olver1974} Ch.~11.
\end{proof}

\section{Airy Normal Form Theorem}
\label{sec:normalform}

\begin{assumption}[Uniform A$_2$-stability
  in the semiclassical regime \Cond]
\label{ass:microlocal}
The WKB amplitude $g$ satisfies
$g(\alpha_*+\beta)=|\beta|^{-1/2}
\tilde\chi(\beta)\,e^{i\Theta(\beta)}$,
$\tilde\chi\in C^\infty_c$.
Assumption~\ref{ass:fold-local}
holds locally.
In addition, uniformly in $c\ge c_0$:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta)=c_3\beta^3+O(\beta^4)$
  uniformly in a $c$-independent neighbourhood;
\item \textit{(A$_2$-stability)}
  $c_3(c)\ge\delta_0>0$;
\item \textit{(Uniform fold projection)}
  $|\Theta''(\beta^*(u))|
  =6c_3|\beta^*(u)|
  \ge\delta_1|u|^{1/2}$
  for $\delta_2\le|u|\le c^{\theta_0/3}$;
\item \textit{(Uniform symbol bound)}
  the Jacobian $|d\beta/dt|$
  from the CFU coordinate change
  is bounded above and below,
  uniformly in $(u,c)$:
  $0<\delta_3\le|d\beta/dt|
  \le\delta_3^{-1}$.
\end{enumerate}
\end{assumption}

\begin{theorem}[Airy normal form \Rig]
\label{thm:airy-normalform}
Under Assumption~\ref{ass:microlocal},
the fold amplitude~\eqref{eq:Phi-WKB}
satisfies:
\begin{equation}\label{eq:airy-normalform}
  \Phi(u)
  = e^{i\alpha_* u}\,
  a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
\end{equation}
where $a(u,c)\in S^{-1/2}_{1,0}$:
$|\partial_u^j a(u,c)|
\le C_j(1+|u|)^{-1/2-j}$,
uniformly in $c$ and $\delta_2\le|u|
\le c^{\theta_0/3}$.
The factor $e^{i\alpha_*u}$ is a global
Fourier shift carrying no microlocal
fold information.
The Airy factor is the A$_2$ normal form;
its large-$|u|$ oscillatory phase is
the stationary phase value $S(u)$
of~\eqref{eq:stat-phase-value},
an asymptotic quantity,
not a global phase function.
\end{theorem}

\begin{proof}[Proof sketch]
Apply CFU \cite{CFU1957}
to~\eqref{eq:Phi-WKB}.
The coordinate change
$\beta\mapsto t(\beta,u)$
from Proposition~\ref{prop:nonpreservation}(2)
exists uniformly by
Assumption~\ref{ass:microlocal}(iii)--(iv):
condition~(iii) is the non-degeneracy
of the fold projection uniformly in
$(u,c)$; condition~(iv) is the uniform
boundedness of the Jacobian.
Under the map:
$|\beta|^{-1/2}\,d\beta
= |t(\beta,u)|^{-1/2}
\cdot|d\beta/dt|^{1/2}\,dt$
(smooth, bounded by (iv)).
The resulting integral is the standard
Airy representation;
$a(u,c)\in S^{-1/2}_{1,0}$
follows from Erd\'elyi's lemma
and condition~(i).
\end{proof}

\begin{lemma}[Airy class
  $\Rightarrow$ improved Ces\`aro \Rig]
\label{lem:airy-integral}
Under the conclusion of
Theorem~\ref{thm:airy-normalform}:
$\bigl|\int_0^U\Phi(u)\,du\bigr|
= O(U^{1/4})$
as $U\to\infty$.
\end{lemma}

\begin{proof}
Write $\Phi(u)=e^{i\alpha_*u}a(u,c)
\mathrm{Ai}(c_3^{-1/3}u)$
and split at $u=1$.
\medskip

\noindent\textit{Region $1\le u\le U$
  (positive argument).}
For $t=c_3^{-1/3}u>0$:
$|\mathrm{Ai}(t)|
\le C t^{-1/4}e^{-\frac{2}{3}t^{3/2}}$
\cite{Olver1974}: exponential decay.
Thus $|\int_1^U a\cdot
\mathrm{Ai}(c_3^{-1/3}u)\,du|
\le C\int_1^\infty
u^{-3/4}e^{-c'u^{3/2}}\,du=O(1)$.
\medskip

\noindent\textit{Region $-U\le u\le -1$
  (negative argument).}
For $t=-c_3^{-1/3}u>0$:
$|\mathrm{Ai}(-t)|=O(t^{-1/4})$
(oscillatory, no exponential decay)
\cite{Olver1974}.
Using $a\in S^{-1/2}_{1,0}$:
\[
  \left|\int_1^U
  a(-u,c)\mathrm{Ai}(-c_3^{-1/3}u)\,du\right|
  \le C\int_1^U u^{-3/4}\,du
  = O(U^{1/4}).
\]
The $e^{i\alpha_*u}$ factor yields
additional cancellation
(van~der~Corput with $\partial_u
e^{i\alpha_*u}=i\alpha_*e^{i\alpha_*u}$)
but cannot worsen the $O(U^{1/4})$ bound.
\medskip

\noindent\textit{Conclusion.}
The negative-argument region
is dominant: $O(U^{1/4})$,
strictly better than Ces\`aro $O(U^{1/2})$.
All three mechanisms
--- $a\in S^{-1/2}_{1,0}$,
Airy polynomial decay in $|u|^{-1/4}$,
and global-in-$u$ symbol uniformity ---
are necessary.
\end{proof}

\subsection{Converse and FIO rigidity}
\label{subsec:converse}

\begin{hypothesis}[FIO inversion
  rigidity \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ lies in the Airy class
($a\in S^{-1/2}_{1,0}$,
stationary phase value $S(u)\sim C|u|^{3/2}$
uniformly in $c$)
and additionally:
\begin{enumerate}[\rm(a)]
\item \textit{(Global ellipticity)}
  $g\mapsto\Phi$ is globally elliptic
  in the stated semiclassical regime;
\item \textit{(No amplitude cancellation)}
  no cancellation between amplitude
  modes of $g$ produces Airy-type
  oscillations without a geometric fold;
\item \textit{(Phase class stability)}
  the phase class of $g$ is stable
  under small perturbations in $c$.
\end{enumerate}
Then the canonical relation of $g$
contains an A$_2$-fold singularity
uniformly in $c$.
\end{hypothesis}

\begin{remark}[Status]
\label{rem:rigidity-status}
Hypothesis~\ref{hyp:FIO-rigidity}
is a global injectivity statement,
not microlocal analysis alone.
Conditions (a)--(c) are necessary
because: without (a),
amplitude interference unrelated
to fold geometry can produce
Airy-type oscillations;
without (b), coincidental cancellations
can mimic an A$_2$ stationary phase value;
without (c), $c$-dependent migration
through Arnold strata invalidates
the uniform fold structure.
\end{remark}

\begin{corollary}[Conditional
  equivalence \Cond]
\label{cor:conditional-equiv}
Under Hypothesis~\ref{hyp:FIO-rigidity}:
$\text{Airy-class bound on }\Phi
\iff
\text{Assumption~\ref{ass:microlocal}}$.
Without it:
$\text{Assumption~\ref{ass:microlocal}}
\Rightarrow
\text{Airy-class bound}$
only.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$
satisfy Assumption~\ref{ass:microlocal}
(and Assumption~\ref{ass:fold-local})
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item \textit{(A$_2$-stability)}
  Does $c_3(c)\ge\delta_0>0$?
  Approach:
  Olver turning-point WKB \cite{Olver1974}
  applied to the PSWF ODE.
\item \textit{(No higher degeneracy)}
  Does Assumption~\ref{ass:fold-local}(F1)
  hold, i.e.\ $\theta''(\alpha)\ne 0$
  for $\alpha\ne\alpha_*$ near $\alpha_*$,
  uniformly?
\item \textit{(Uniform symbol bound)}
  Does Assumption~\ref{ass:microlocal}(iv)
  hold uniformly for the PSWF Jacobian?
\end{enumerate}
Two outcomes:
\begin{enumerate}[\rm(A)]
\item Assumption~\ref{ass:microlocal} holds:
  $|\int_0^U\Phi|=O(U^{1/4})$,
  $S(c)=O(c^{\theta_0/2})$ (up to logs).
\item $c_3(c)\to 0$
  or (F1) fails:
  $\mathcal{C}_c$ exits A$_2$ stratum;
  no stable Airy normal form;
  Ces\`aro sharp.
\end{enumerate}
\end{problem}
