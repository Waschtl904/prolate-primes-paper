\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases, A$_2$-Fold Singularities,\\
  and the Airy Normal Form}

\author{[Author]}

\begin{abstract}
This paper works in the \emph{one-dimensional setting}
($\beta,u\in\mathbb{R}$).
A cubic WKB phase $c_3 \beta^3$ produces an A$_2$-fold
singularity of the canonical relation in the sense of
Arnol'd \cite{Arnold1972}.
Setting $h := c^{-1} \to 0$, we treat the analysis
semiclassically throughout.
The central result is a single CFU reduction theorem
(Theorem~\ref{thm:CFU}): an oscillatory integral with
A$_2$-fold and conormal amplitude
$|\beta|^{-1/2} \in I^{-1/4,0}(\Lambda_0,\Lambda_1)$
is microlocally equivalent, modulo smoothing operators
in a fixed conic neighbourhood of the fold,
to $a_{\mathrm{CFU}}(u;h)\,\mathrm{Ai}(\mu(u;h))$
with transported principal symbol
$a_{\mathrm{CFU}} \in S^{-1/2}_{1,0}(h)$
and remainder $\in S^{-\infty}$, uniformly in $h$.
Three consequences follow in linear order:
\begin{enumerate}[\rm(I)]
\item \textit{(Pointwise decay, \Cond)}
  $|\Phi(u;h)| \le C(1+|u|)^{-3/4}$ uniformly in $h$,
  with exponent $-3/4$ arising from the
  combined A$_2$ + conormal scaling
  (Lemma~\ref{lem:degenerate-phase}).
\item \textit{(Ces\`aro improvement, \Cond)}
  $|\int_0^U \Phi| = O(U^{1/4})$.
\item \textit{(Open)} A$_2$-stability for PSWF amplitudes
  uniformly as $h \to 0$
  (Open Problem~\ref{prob:PSWF}).
\end{enumerate}
The forward direction is rigorous under
Assumption~\ref{ass:A2uniform};
the converse requires the non-standard
Hypothesis~\ref{hyp:rigidity}.
\end{abstract}

\maketitle

\section{Definitions}
\label{sec:defs}

\begin{definition}[A$_2$-fold singularity]\label{def:A2}
Let $\mathcal{C} \subset T^*\mathbb{R}_\beta \times T^*\mathbb{R}_u$
be a Lagrangian submanifold,
$\pi_u : \mathcal{C} \to T^*\mathbb{R}_u$ the projection.
$\mathcal{C}$ has an \emph{A$_2$-fold at $p_0$} if
$\det d\pi_u|_{p_0} = 0$ and
$d(\det d\pi_u)|_{p_0} \ne 0$
(stable Lagrange singularity of A$_2$-type,
Arnol'd \cite{Arnold1972}).
For phase $\Psi(\beta,u) = \Theta(\beta;h^{-1}) + u\beta$
this reduces to $\Theta''(0) = 0$, $\Theta'''(0) \ne 0$.
\end{definition}

\begin{definition}[Conormal amplitude class]\label{def:Iml}
The \emph{Melrose--Uhlmann class}
$I^{m,l}(\Lambda_0,\Lambda_1)$ \cite{MelroseU1979}
consists of distributions with conormal singularities
at intersecting Lagrangians $\Lambda_0,\Lambda_1$.
The amplitude $|\beta|^{-1/2}\chi$
belongs to $I^{-1/4,0}(\Lambda_0,\Lambda_1)$
where $\Lambda_0$ is the zero-section and
$\Lambda_1$ is the conormal to $\{\beta = 0\}$.
This class is required (not the standard FIO class $I^m$)
because the amplitude singularity coincides with the
critical point.
\end{definition}

\begin{definition}[Semiclassical parameter and symbol class]
\label{def:semiclass}
Set $h := c^{-1}$, so $h \to 0$ as $c \to \infty$.
The parameter-dependent symbol class is
\[
  S^m_{1,0}(h) := \bigl\{ a(u;h) \in C^\infty(\mathbb{R}_u)
  \;\big|\;
  |\partial_u^\alpha a| \le C_\alpha (1+|u|)^{m-|\alpha|},
  \text{ uniformly in } h \in (0,h_0]
  \bigr\},
\]
in the sense of semiclassical analysis
(Zworski \cite{Zworski2012}, Ch.~4).
The \emph{transported principal symbol}
$a_{\mathrm{CFU}}(u;h)$ is the image of
$|\beta|^{-1/2}\chi \in I^{-1/4,0}(\Lambda_0,\Lambda_1)$
under CFU restriction to the stationary set
(Theorem~\ref{thm:CFU}); it lies in $S^{-1/2}_{1,0}(h)$.
\end{definition}

\section{The CFU Reduction Theorem}
\label{sec:CFU}

\begin{assumption}[Uniform A$_2$-stability, \Cond]
\label{ass:A2uniform}
The phase satisfies
$\Theta(\beta;h^{-1}) = c_3(h)\beta^3 + O(\beta^4)$
with $c_3(h) \ge \delta_0 > 0$ uniformly,
$\Theta''(\beta) \ne 0$ for $\beta \ne 0$
(unique fold),
and the CFU Jacobian satisfies
$|\det(dt/d\beta)| \ge \delta_1 > 0$
with uniform symbol bounds in $h$
(semiclassical non-degeneracy).
\end{assumption}

\begin{theorem}[CFU reduction with conormal
  amplitude, \Rig]
\label{thm:CFU}
Under Assumption~\ref{ass:A2uniform},
the fold-zone contribution
\[
  \Phi_{\mathrm{fold}}(u;h)
  = e^{i\alpha_* u}
  \int_\mathbb{R}
  |\beta|^{-1/2}\,
  e^{i(\Theta(\beta;h^{-1})+\beta u)}
  \phi_{\mathrm{fold}}(\beta)\,d\beta
\]
satisfies, modulo smoothing operators
in a fixed conic neighbourhood of the fold point
and uniformly in $h \in (0,h_0]$:
\begin{equation}\label{eq:CFU}
  \Phi_{\mathrm{fold}}(u;h)
  = e^{i(\alpha_* u + R(u;h))}
  \bigl[
    a_{\mathrm{CFU}}(u;h)\,\mathrm{Ai}(\mu(u;h))
    + r(u;h)
  \bigr],
\end{equation}
where:
\begin{enumerate}[\rm(i)]
\item $a_{\mathrm{CFU}} \in S^{-1/2}_{1,0}(h)$
  is the transported principal symbol
  (Definition~\ref{def:semiclass});
  it satisfies $|a_{\mathrm{CFU}}(u;h)| \le C|u|^{-1/2}$;
\item $\mu(u;h)$ is the CFU phase parameter
  with $|\mu| \sim |u|^{1/2}$ uniformly;
\item $r(u;h) \in S^{-\infty}$:
  $|r(u;h)| \le C_N(1+|u|)^{-N}$ for all $N$;
\item $R(u;h)$ absorbs the Maslov phase factor
  (H\"ormander \cite{Hormander1983}, Ch.~25.1).
\end{enumerate}
The far-zone contribution
(on $\phi_{\mathrm{far}} = 1 - \phi_{\mathrm{fold}}$,
where $|\partial_\beta^2 \Psi| \ge c_0' > 0$)
satisfies $|\Phi_{\mathrm{far}}(u;h)| \le C|u|^{-1/2}$
by standard non-degenerate stationary phase.
\end{theorem}

\begin{proof}
\textit{Fold zone.}
The amplitude $|\beta|^{-1/2}\chi$
lies in $I^{-1/4,0}(\Lambda_0,\Lambda_1)$
(Definition~\ref{def:Iml};
Melrose--Uhlmann \cite{MelroseU1979}).
The CFU theorem \cite{CFU1957} provides a local
canonical transformation
$\beta \mapsto t(\beta,u;h)$
reducing $\Psi$ to the Airy normal form
$\frac{1}{3}t^3 - \mu(u;h)\,t$.
Under Assumption~\ref{ass:A2uniform},
the Jacobian bound and symbol estimates hold uniformly
in $h$; by Melrose--Uhlmann
\cite{MelroseU1979} and
Guillemin--Uhlmann \cite{GuillemU1981} (Lemma~2.1),
the $I^{m,l}$-class is preserved,
and restriction to the stationary set $t=0$
yields $a_{\mathrm{CFU}} \in S^{-1/2}_{1,0}(h)$.
The Airy function Olver \cite{Olver1974} (Ch.~11)
satisfies $|\mathrm{Ai}(\mu)| \le C|\mu|^{-1/4}$
for $|\mu| \ge 1$; this is a standard reference fact,
not a separate lemma.
\textit{Far zone.}
Non-degenerate stationary phase
(H\"ormander \cite{Hormander1983}, Ch.~7.7)
gives $O(|u|^{-1/2})$.
\end{proof}

\begin{remark}[Microlocal equivalence scope]
\label{rem:scope}
Equation~\eqref{eq:CFU} is a microlocal equivalence
modulo smoothing, not a pointwise or global identity.
It holds in a fixed conic neighbourhood of the fold
in phase space; outside this neighbourhood the
far-zone bound applies.
\end{remark}

\section{Main Results}
\label{sec:results}

\begin{lemma}[Degenerate stationary phase
  with conormal weight, \Rig]
\label{lem:degenerate-phase}
For the model integral
$I(\mu) := \int_\mathbb{R} |t|^{-1/2}
e^{i(\frac{1}{3}t^3 - \mu t)}\,dt$,
$|\mu| \ge 1$:
\[
  |I(\mu)| \le C\,|\mu|^{-3/4}.
\]
\end{lemma}

\begin{proof}
By Theorem~\ref{thm:CFU} applied to the model integral,
$I(\mu) = a_{\mathrm{CFU}}(\mu)\,\mathrm{Ai}(\mu) + r(\mu)$
with $a_{\mathrm{CFU}} \in S^{-1/2}_{1,0}$
and $|r(\mu)| \le C_N|\mu|^{-N}$.
Since $|a_{\mathrm{CFU}}(\mu)| \le C'|\mu|^{-1/2}$
and $|\mathrm{Ai}(\mu)| \le C|\mu|^{-1/4}$
(Olver \cite{Olver1974}, Ch.~11),
\[
  |I(\mu)|
  \le C'|\mu|^{-1/2} \cdot C|\mu|^{-1/4}
     + C_N|\mu|^{-N}
  = C''|\mu|^{-3/4}. \qedhere
\]
The exponent $-3/4 = -1/2 + (-1/4)$
is the intrinsic combination of the
conormal amplitude order $(-1/2)$
and the A$_2$-fold Airy decay $(-1/4)$;
both factors are determined by the same
fold geometry via CFU, not independently.
\end{proof}

\begin{corollary}[Pointwise decay, \Cond]
\label{cor:pointwise}
Under Assumption~\ref{ass:A2uniform},
\[
  |\Phi(u;h)| \le C(1+|u|)^{-3/4}
  \quad \forall\, u \in \mathbb{R},
\]
uniformly in $h \in (0,h_0]$.
\end{corollary}

\begin{proof}
Fold zone: Theorem~\ref{thm:CFU}
+ Lemma~\ref{lem:degenerate-phase}.
Far zone: Theorem~\ref{thm:CFU} (far-zone bound).
Compact zone $|u| \le 1$: continuity.
\end{proof}

\begin{corollary}[Ces\`aro improvement, \Cond]
\label{cor:cesaro}
Under Assumption~\ref{ass:A2uniform},
\[
  \left|\int_0^U \Phi(u;h)\,du\right|
  = O(U^{1/4}),
\]
improving upon $O(U^{1/2})$
by pure $L^1$-integrability of $(1+|u|)^{-3/4}$
(no oscillation cancellation used).
\end{corollary}

\subsection{Converse and rigidity}

\begin{assumption}[Microlocal reconstruction,
  non-standard/conjectural, \Cond]
\label{hyp:rigidity}
\textit{Forward (rigorous):}
Theorem~\ref{thm:CFU} gives
A$_2$-fold $\Rightarrow$ Airy normal form.
\textit{Inverse (conjectural):}
If $\Phi$ is in the Airy class,
$\sigma(g \mapsto \Phi)|_\mathcal{C} \ne 0$,
$g \mapsto \Phi$ is injective mod smoothing,
and $\mathrm{WF}(\Phi) \subseteq \pi_u(\mathcal{C})$
on the A$_2$ stratum,
then $\mathcal{C}$ has an A$_2$-fold uniformly in $h$.
This is an inverse microlocal conjecture,
not a consequence of standard FIO theory;
verification for PSWF: Paper~XVII.
\end{assumption}

\begin{corollary}[Conditional equivalence]
\label{cor:equiv}
Under Assumption~\ref{hyp:rigidity}:
Airy normal form $\Rightarrow$
Assumption~\ref{ass:A2uniform}.
The forward direction holds without
Assumption~\ref{hyp:rigidity}.
\end{corollary}

\section{Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability for PSWF, \Open]
\label{prob:PSWF}
Does the PSWF amplitude satisfy
Assumption~\ref{ass:A2uniform} uniformly as $h \to 0$?
Specifically:
(a) $c_3(h) \ge \delta_0 > 0$:
expected from Olver \cite{Olver1974};
uniform remainder bounds required.
(b) Unique fold uniformly in $h$:
plausible but not established.
(c) CFU Jacobian bound uniformly in $h$:
technically hardest; no result currently available.
These are partially supported by WKB asymptotics
but no uniform microlocal analysis exists
in the literature.
\end{problem}
