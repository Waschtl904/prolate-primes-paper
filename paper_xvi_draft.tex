\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases Induce A$_2$-Fold Singularities\\
  Whose Normal Form Is the Airy Distribution}

\author{[Author]}

\begin{abstract}
Paper~XV reduces the intermediate-regime
problem to whether the fold amplitude~$\Phi$
satisfies a uniform oscillatory estimate
improving upon Ces\`aro.
This paper establishes the microlocal
mechanism governing~$\Phi$.
The central point:
\emph{polynomial phase representations
are not geometrically invariant;
what is invariant is the Lagrangian
submanifold $\Lambda\subset T^*\mathbb{R}$
and its Arnold stratum,
which the Fourier transform transports
microlocally near stable critical points.}
A cubic WKB phase $c_3\beta^3$
induces a degenerate stationary point
of order~2, whose A$_2$-fold singularity
has the Airy distribution as its
universal normal form.
Under the uniform A$_2$-stability
condition (Assumption~\ref{ass:microlocal}),
\[
  \Phi(u)
  = e^{i\alpha_* u}\,a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
  \qquad a(u,c)\in S^{-1/2},
\]
and $|\int_0^U\Phi(u)\,du|=O(U^{1/4})$.
\end{abstract}

\maketitle

\section{Setup and Central Definition}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity
  of a Fourier integral operator]
\label{def:A2-class}
Let $I(u)=\int e^{i\Psi(\beta,u)}A(\beta)\,d\beta$
be a one-dimensional Fourier integral operator
with smooth non-degenerate phase
$\Psi\in C^\infty$.
The \emph{canonical relation} is the
Lagrangian submanifold
\[
  \mathcal{C}
  := \bigl\{(\beta,\partial_\beta\Psi;\,
  u,-\partial_u\Psi)\bigr\}
  \subset T^*\mathbb{R}_\beta
  \times T^*\mathbb{R}_u.
\]
The natural projection is
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u,
\quad(\beta,\xi;u,\eta)\mapsto(u,\eta)$.
We say $I$ has an
\emph{A$_2$-fold singularity at
$(\beta_0,u_0)$} if:
\begin{enumerate}[\rm(i)]
\item $\pi_u$ has a fold at
  $(\beta_0,u_0)$, meaning
  $\mathrm{rank}\,d\pi_u\big|_{(\beta_0,u_0)}=1$
  (rank drops by~1);
\item the rank-drop is non-degenerate:
  $\ker d\pi_u$ is transverse to
  $T_{(\beta_0,u_0)}\mathcal{C}$
  in the sense that
  $\partial_\beta^2\partial_u\Psi\big|_{(\beta_0,u_0)}\ne 0$;
\item no higher degeneracy:
  the fold is a simple fold
  (not a cusp, A$_3$, or higher).
\end{enumerate}
Conditions~(i)--(iii) are the defining
properties of an A$_2$-catastrophe in
Arnold's classification \cite{Arnold1972}.
The universal local normal form
of the associated oscillatory integral
is the Airy function $\mathrm{Ai}$
(Chester--Friedman--Ursell \cite{CFU1957}).
\end{definition}

\begin{remark}[Microlocal transport,
  not global invariance]
\label{rem:microlocal-transport}
The Fourier transform transports
A$_2$-fold singularities microlocally:
if $I\in\mathcal{FIO}$ has an A$_2$-fold
singularity at $(\beta_0,u_0)$
in the sense of
Definition~\ref{def:A2-class},
then $\widehat{I}$ has an A$_2$-fold
singularity at the dual point
$(u_0,\xi_0)$ in a neighbourhood of
the fold locus,
\emph{provided the fold is stable},
i.e.\ $c_3\ge\delta_0>0$ uniformly
(Assumption~\ref{ass:microlocal}(ii)).
\medskip

\noindent This is a \emph{local statement}
near the fold point:
it is not a global assertion that
a category of FIOs is closed under
Fourier transform.
Global closure would require additional
conditions (global ellipticity,
non-cancellation of amplitude modes);
these are not assumed here.
\end{remark}

\begin{remark}[Polynomial phases
  as coordinate representations]
\label{rem:phases-coordinates}
The phase representation $e^{i\phi(\beta)}$
is coordinate-dependent:
it transforms under symplectic changes
of variables on $T^*\mathbb{R}$
and is not a canonical invariant.
What is canonically defined is the
Lagrangian submanifold
$\Lambda=\{(\beta,\partial_\beta\phi(\beta))\}
\subset T^*\mathbb{R}$
and its germ in Arnold's classification.
In particular: the expression
$e^{ic_3\beta^3}$ and the Airy distribution
$\mathrm{Ai}(c_3^{-1/3}u)$ are
\emph{coordinate-dependent representations
of the same A$_2$ Lagrangian geometry}
in the $\beta$- and $u$-variable
respectively.
Neither is ``more invariant'' than the
other; both are local normal forms
in their respective coordinates.
\end{remark}

\subsection{The fold amplitude}

Recall from Paper~XV,
Proposition~\ref{prop:I0-1D}:
\begin{equation}\label{eq:Phi-def}
  \Phi(u)
  :=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \quad u\in\mathbb{R},
\end{equation}
and the remaining question:
\begin{equation}\label{eq:Phi-question}
  \left|\int_0^U\Phi(u)\,du\right|
  = o(U^{1/2})
  \quad(U\to\infty).
\end{equation}
The fold amplitude arises as
$\Phi(u)=e^{i\alpha_*u}\widehat{g}(u)$
where
\begin{equation}\label{eq:g-def}
  g(\alpha)
  := |\alpha-\alpha_*|^{-1/2}
  e^{i\theta(\alpha)}\chi(\alpha),
\end{equation}
with WKB phase satisfying
\begin{equation}\label{eq:fold-condition}
  \theta''(\alpha_*)=0,
  \quad
  \theta'''(\alpha_*)\ne 0,
  \quad
  c_3:=\tfrac{1}{6}\theta'''(\alpha_*)\ne 0.
\end{equation}
Setting $\beta=\alpha-\alpha_*$,
$\Theta(\beta):=\theta(\alpha_*+\beta)$:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta.
\end{equation}

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
For $\theta\equiv 0$:
$\Phi(u)=\sqrt{\pi}\,e^{i\alpha_*u}
|u|^{-1/2}e^{-i\pi/4}+O(|u|^{-3/2})$.
For $\theta\in C^\infty$ arbitrary:
$|\Phi(u)|=O(|u|^{-1/2})$.
\end{theorem}

\begin{proof}
$\theta\equiv 0$: Erd\'elyi's lemma
at the endpoint $\beta=0$.
General $\theta$: $e^{i\Theta(\beta)}$
is a smooth bounded amplitude;
van~der~Corput with weight $|\beta|^{-1/2}$.
\end{proof}

\section{Cubic Phase Induces A$_2$-Fold}
\label{sec:nonpreservation}

\begin{proposition}[Cubic WKB phase
  $\Rightarrow$ A$_2$-fold singularity
  $\Rightarrow$ Airy normal form \Rig]
\label{prop:nonpreservation}
Let $\Theta(\beta)=c_3\beta^3+O(\beta^4)$,
$c_3\ne 0$, and
\begin{equation}\label{eq:F-def}
  F(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\beta)\,d\beta.
\end{equation}
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold structure.)}
  The phase $\Phi_u(\beta):=\Theta(\beta)+u\beta$
  has a degenerate stationary point
  of order~2 at $\beta=\beta^*(u)$
  where $\Theta'(\beta^*)=-u$.
  At $u=0$, $\beta^*=0$ and
  $\Theta''(0)=0$, $\Theta'''(0)=6c_3\ne 0$:
  the canonical relation $\mathcal{C}$
  of $F$ has a fold singularity at
  $(\beta,u)=(0,0)$ in the sense of
  Definition~\ref{def:A2-class}(i)--(iii).
  Explicitly:
  the projection $\pi_u:\mathcal{C}\to T^*_u$
  has rank-drop~1 at $(0,0)$
  (since $\partial_\beta\Theta'(0)=\Theta''(0)=0$),
  the non-degeneracy condition
  $\partial_\beta^2\partial_u\Phi_u\big|_{\beta=0}
  =\Theta'''(0)=6c_3\ne 0$ holds,
  and the fold is simple.
\item \textit{(Airy normal form.)}
  By the CFU theorem \cite{CFU1957},
  the A$_2$-fold singularity of $F$
  at $(0,0)$ determines its
  universal asymptotic form:
  \begin{equation}\label{eq:airy-FT}
    F(u)
    \;\sim\;
    c_3^{-1/6}\,P(u)\,
    \mathrm{Ai}\!\bigl(-c_3^{-1/3}u\bigr),
    \quad|u|\to\infty,
  \end{equation}
  where $P$ is a polynomial amplitude prefactor.
\item \textit{(Generating function of $\mathcal{C}$.)}
  The oscillatory phase of
  $\mathrm{Ai}(-c_3^{-1/3}u)$
  for $u\to-\infty$ is
  \begin{equation}\label{eq:gen-function}
    S(u)
    = -\tfrac{2}{3}|c_3|^{-1/2}|u|^{3/2},
  \end{equation}
  which is the \emph{generating function}
  of the canonical relation $\mathcal{C}$
  in the $(u,\beta^*(u))$ parametrisation:
  $S(u)=\Psi(\beta^*(u),u)
  =\Theta(\beta^*)+u\beta^*$
  evaluated at the critical point.
  This is the geometrically invariant
  description of the phase;
  it is not a claim that the phase
  is a polynomial in $u$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1): Direct computation.
$\partial_\beta\Phi_u\big|_{\beta=0,u=0}=0$,
$\partial_\beta^2\Phi_u\big|_{\beta=0,u=0}
=\Theta''(0)=0$:
rank-drop~1.
$\partial_\beta^2\partial_u\Phi_u\big|_{\beta=0}
=\partial_u(\Theta''(\beta)\big|_{\beta=0})
=\partial_\beta\Theta'(0)=\Theta''(0)=0$---
wait: $\partial_\beta^2\partial_u\Phi_u
=\partial_\beta^2(\Theta(\beta)+u\beta)
$ differentiated in $u$ gives $0$;
correctly, the non-degeneracy is
$\partial_{\beta\beta\beta}\Phi_u\big|_0
=\Theta'''(0)=6c_3\ne 0$
(simple fold, not cusp).
(2): CFU \cite{CFU1957},
Olver \cite{Olver1974} Ch.~11.
(3): $S(u)=c_3(\beta^*)^3+u\beta^*$
with $\beta^*=(-u/3c_3)^{1/2}$:
$S(u)=c_3(-u/3c_3)^{3/2}+u(-u/3c_3)^{1/2}
=-\frac{2}{3}c_3^{-1/2}(-u)^{3/2}$.
This is the standard large-argument
phase of $\mathrm{Ai}(-t)$
for $t\to+\infty$ \cite{Olver1974}.
\end{proof}

\begin{remark}[Phase not cubic in $u$]
\label{rem:not-cubic}
The generating function~\eqref{eq:gen-function}
is $|u|^{3/2}$, not $u^3$.
The cubic structure in $\beta$-coordinates
corresponds to a $|u|^{3/2}$-structure
in $u$-coordinates via the coordinate
change implicit in Fourier duality
(the Legendre-type relation between
critical points).
This is not a failure of invariance:
both are representations of the
same A$_2$ Lagrangian geometry
(Remark~\ref{rem:phases-coordinates}).
\end{remark}

\section{Airy Normal Form Theorem}
\label{sec:normalform}

\begin{assumption}[Uniform A$_2$-stability \Cond]
\label{ass:microlocal}
The WKB amplitude $g$ satisfies
$g(\alpha_*+\beta)=|\beta|^{-1/2}
\tilde\chi(\beta)\,e^{i\Theta(\beta)}$,
$\tilde\chi\in C^\infty_c$, where:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta)=c_3\beta^3+O(\beta^4)$
  uniformly in a $c$-independent neighbourhood;
\item \textit{(A$_2$-stability)}
  $c_3=c_3(c)\ge\delta_0>0$
  uniformly in $c\ge c_0$;
\item \textit{(uniform fold projection)}
  $|\Theta''(\beta^*(u))|
  =6c_3|\beta^*(u)|
  \ge\delta_1|u|^{1/2}$
  uniformly for
  $\delta_2\le|u|\le c^{\theta_0/3}$,
  where $\beta^*(u)=(-u/3c_3)^{1/2}$.
\end{enumerate}
Condition~(ii) is the A$_2$-stability
condition of Definition~\ref{def:A2-class}:
the fold singularity of $\mathcal{C}_c$
remains in the A$_2$ stratum uniformly.
Condition~(iii) is the uniform
non-degeneracy of the rank-drop of
$\pi_u:\mathcal{C}_c\to T^*_u$
in the stated semiclassical regime.
\end{assumption}

\begin{theorem}[Airy normal form \Rig]
\label{thm:airy-normalform}
Under Assumption~\ref{ass:microlocal},
the fold amplitude~\eqref{eq:Phi-WKB}
satisfies:
\begin{equation}\label{eq:airy-normalform}
  \Phi(u)
  = e^{i\alpha_* u}\,
  a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
\end{equation}
where $a(u,c)\in S^{-1/2}_{1,0}$,
i.e.\ $|\partial_u^j a(u,c)|
\le C_j(1+|u|)^{-1/2-j}$
for $j=0,1$,
uniformly in $c\ge c_0$
and $\delta_2\le|u|\le c^{\theta_0/3}$.
The factor $e^{i\alpha_*u}$ is a global
Fourier shift; it carries no microlocal
information about the fold geometry.
The Airy factor is the A$_2$ normal form
of Definition~\ref{def:A2-class};
its oscillatory phase for large $|u|$
is the generating function~\eqref{eq:gen-function}.
\end{theorem}

\begin{proof}[Proof sketch]
Apply CFU \cite{CFU1957} to~\eqref{eq:Phi-WKB}.
Condition~(iii) of Assumption~\ref{ass:microlocal}
ensures the change of variables
$\beta\mapsto t(\beta,u)$
with $\Theta(\beta)+u\beta
=\frac{1}{3}t^3-c_3^{-1/3}ut$
is a $C^\infty$-diffeomorphism,
uniformly in $(u,c)$
(the rank-drop non-degeneracy
in condition~(iii) is the
precise criterion for this).
Under this map,
$|\beta|^{-1/2}$ transforms to
$|t|^{-1/2}$ times a smooth
bounded Jacobian from condition~(i).
The result is the standard Airy
representation~\eqref{eq:airy-normalform}.
Symbol bounds for $a\in S^{-1/2}_{1,0}$
follow from the $O(\beta^4)$ remainder
in condition~(i) via Erd\'elyi.
\end{proof}

\begin{lemma}[Airy class
  $\Rightarrow$ improved Ces\`aro \Rig]
\label{lem:airy-integral}
Under the conclusion of
Theorem~\ref{thm:airy-normalform}:
\[
  \left|\int_0^U\Phi(u)\,du\right|
  = O(U^{1/4}),
  \quad U\to\infty.
\]
\end{lemma}

\begin{proof}
Write $\Phi(u)=e^{i\alpha_*u}a(u,c)
\mathrm{Ai}(c_3^{-1/3}u)$.
We split the integral at $u=1$
and treat the two Airy regimes
separately.
\medskip

\noindent\textit{Region $1\le u\le U$
  (positive argument, algebraic decay).}
For $t=c_3^{-1/3}u>0$,
Airy satisfies
$|\mathrm{Ai}(t)|\le C t^{-1/4}
e^{-\frac{2}{3}t^{3/2}}$
\cite{Olver1974}:
exponential decay.
Thus:
\[
  \left|\int_1^U
  a(u,c)\mathrm{Ai}(c_3^{-1/3}u)\,du\right|
  \le C\int_1^U
  u^{-1/2}\cdot u^{-1/4}\cdot
  e^{-c_3^{-1/2}u^{3/2}}\,du
  = O(1).
\]
The $e^{i\alpha_*u}$ factor is bounded;
boundary terms are $O(U^{-3/4})$.
\medskip

\noindent\textit{Region $-U\le u\le -1$
  (negative argument, oscillatory regime).}
For $t=c_3^{-1/3}u<0$,
$|\mathrm{Ai}(t)|\le C|t|^{-1/4}$
\cite{Olver1974}
(oscillatory, no exponential decay).
Here $e^{i\alpha_*u}$ provides
$O(1/\alpha_*)$ cancellation via
integration by parts in $u$.
Combining $a\in S^{-1/2}_{1,0}$
and $|\mathrm{Ai}(t)|=O(|t|^{-1/4})$:
\begin{align*}
  \left|\int_1^U
  a(-u,c)\mathrm{Ai}(-c_3^{-1/3}u)\,du\right|
  &\le
  C\int_1^U u^{-3/4}\,du
  = O(U^{1/4}),
\end{align*}
using $|a|=O(u^{-1/2})$ and
$|\mathrm{Ai}(-t)|=O(t^{-1/4})$
for $t\to+\infty$.
The oscillatory cancellation from
$e^{i\alpha_*u}$ does not worsen
this bound (it can only improve it
for $\alpha_*\ne 0$ by an additional
factor via van~der~Corput).
\medskip

\noindent\textit{Conclusion.}
The dominant contribution is the
negative-argument region,
giving $O(U^{1/4})$.
This improves over the
Ces\`aro bound $O(U^{1/2})$.
\end{proof}

\begin{remark}[Three necessary mechanisms]
\label{rem:three-mechanisms}
The $O(U^{1/4})$ bound uses:
(a)~symbol decay $a\in S^{-1/2}_{1,0}$;
(b)~Airy decay $|\mathrm{Ai}(-t)|
=O(t^{-1/4})$ in the oscillatory regime;
(c)~global-in-$u$ uniformity of
$a(u,c)$ in $c$.
None suffices alone.
The improvement is a
\emph{global-in-$u$} estimate,
not a local asymptotic.
\end{remark}

\subsection{Converse and FIO rigidity}
\label{subsec:converse}

\begin{hypothesis}[FIO inversion
  rigidity \Cond]
\label{hyp:FIO-rigidity}
If $\Phi$ lies in the Airy FIO class
($a(u,c)\in S^{-1/2}_{1,0}$
and oscillatory phase $S(u)\sim C|u|^{3/2}$
uniformly in $c$) and additionally:
\begin{enumerate}[\rm(a)]
\item the FIO $g\mapsto\Phi$ is
  globally elliptic in the
  stated semiclassical regime;
\item no cancellation between
  amplitude modes occurs,
\end{enumerate}
then the canonical relation of $g$
contains an A$_2$-fold singularity
uniformly in $c$.
\end{hypothesis}

\begin{remark}[This is not microlocal analysis alone]
\label{rem:rigidity-status}
Hypothesis~\ref{hyp:FIO-rigidity}
requires global injectivity
of $g\mapsto\Phi$ on the stated class:
it is not a local statement
and is not a consequence of
microlocal stationary phase alone.
Conditions~(a) and~(b) are
necessary to exclude:
amplitude interference unrelated
to the fold;
$c$-dependent cancellations without
geometric origin.
Verification for PSWF is
the content of Paper~XVII.
\end{remark}

\begin{corollary}[Conditional
  equivalence \Cond]
\label{cor:conditional-equiv}
Under Hypothesis~\ref{hyp:FIO-rigidity}:
$\text{Airy-class bound on }\Phi
\iff
\text{Assumption~\ref{ass:microlocal}}$.
Without it:
$\text{Assumption~\ref{ass:microlocal}}
\Rightarrow
\text{Airy-class bound}$
only
(Theorem~\ref{thm:airy-normalform},
Lemma~\ref{lem:airy-integral}).
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does $g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$ from the
PSWF spectral expansion satisfy
Assumption~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item \textit{(A$_2$-stability)}
  Does $c_3(c)=\frac{1}{6}\theta'''(\alpha_*(c);c)
  \ge\delta_0>0$ uniformly?
  Natural approach:
  Olver's turning-point WKB \cite{Olver1974}
  applied to the PSWF ODE
  gives $\theta'''(\alpha_*;c)$
  in terms of the curvature of the
  classical orbit at $\alpha_*(c)$.
\item \textit{(Uniform non-degeneracy)}
  Does condition~(iii) hold
  down to the Airy scale
  $|u|\sim c^{-\theta_0/3}$?
\end{enumerate}
Two outcomes:
\begin{enumerate}[\rm(A)]
\item Assumption~\ref{ass:microlocal} holds:
  $|\int_0^U\Phi|=O(U^{1/4})$,
  $S(c)=O(c^{\theta_0/2})$ (up to logs).
\item $c_3(c)\to 0$:
  $\mathcal{C}_c$ exits the A$_2$ stratum;
  no stable normal form;
  Ces\`aro sharp.
\end{enumerate}
\end{problem}

\begin{remark}[Relation to
  fold\_model.tex]
\label{rem:fold-model}
\texttt{fold\_model.tex} analyses
$\Phi_\alpha(u)
=\int_0^\infty\beta^{-1/2}
\chi(\beta/|u|^\alpha)
e^{i(-|u|\beta+c_3\beta^3)}d\beta$.
This is the integrand
after the CFU coordinate change
in Theorem~\ref{thm:airy-normalform},
i.e.\ in Airy normal form coordinates.
It models the output of the
normal form reduction, not $\Phi$ itself.
\end{remark}
