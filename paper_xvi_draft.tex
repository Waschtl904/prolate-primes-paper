\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases Induce A$_2$-Fold Singularities\\
  Whose Normal Form Is the Airy Distribution}

\author{[Author]}

\begin{abstract}
Paper~XV reduces the intermediate-regime
problem to whether the fold amplitude~$\Phi$
satisfies an estimate improving upon
Ces\`aro.
This paper establishes the microlocal
mechanism via two separate results.
\begin{enumerate}[\rm(I)]
\item \textit{(Microlocal normal form.)\ }
  Theorem~\ref{thm:airy-normalform}:
  Under the uniform A$_2$-stability
  conditions of Assumption~\ref{ass:microlocal},
  the fold amplitude satisfies
  $\Phi(u)=e^{i\alpha_*u}a(u,c)
  \mathrm{Ai}(c_3^{-1/3}u)$,
  $a\in S^{-1/2}_{1,0}$,
  locally near the fold point.
  This is a microlocal statement;
  it relies on the A$_2$-fold structure
  of the canonical relation and
  the stable equivalence of
  Lagrangian map germs (Arnold
  \cite{Arnold1972},
  Golubitsky--Guillemin \cite{Golub1973}).
\item \textit{(Global integrability corollary.)\ }
  Corollary~\ref{cor:global-bound}:
  The Airy normal form structure,
  combined with
  (a)~uniformity of the
  stationary region in $u$,
  (b)~absence of secondary
  stationary points,
  and (c)~a partition of unity
  in $u$-space,
  yields a global $L^1$-growth bound
  $|\int_0^U\Phi\,du|=O(U^{1/4})$.
  This bound reflects a
  \emph{better integrability exponent
  due to the structural Airy decay};
  it does not use oscillatory
  cancellation.
\end{enumerate}
The forward direction
(Assumption~\ref{ass:microlocal}
$\Rightarrow$ Airy bound)
is rigorous;
the converse requires
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold.
Let $\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
denote the projection to the right factor.
\medskip

\noindent\textit{Geometric
(coordinate-free) definition.}
We say $\mathcal{C}$ has an
\emph{A$_2$-fold singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$
  (rank-drop of $\pi_u$ by~1);
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$
  (simple fold; no higher degeneracy).
\end{enumerate}
This is the A$_2$-type in Arnold's
classification \cite{Arnold1972}.
\medskip

\noindent\textit{Reduction to local
canonical coordinates.}
For a one-dimensional FIO
with phase $\Psi(\beta,u)
=\Theta(\beta)+u\beta$,
the canonical relation is
$\mathcal{C}=\{(\beta,\Theta'(\beta);\,
u,-\beta)\}$.
In these coordinates:
\begin{align}
  \label{eq:fold-i}
  \det\,d\pi_u=0
  \;&\iff\;
  \Theta''(\beta_0)=0,
  \\
  \label{eq:fold-ii}
  d(\det\,d\pi_u)\ne 0
  \;&\iff\;
  \Theta'''(\beta_0)\ne 0.
\end{align}
Equations~\eqref{eq:fold-i}--\eqref{eq:fold-ii}
are the coordinate expression of~(i)--(ii)
for this phase;
the geometric conditions~(i)--(ii)
are invariant under canonical
transformations of $\mathcal{C}$.
\medskip

\noindent By the Thom--Boardman
classification \cite{Golub1973},
any one-dimensional FIO with an
A$_2$-fold singularity is,
microlocally near $p_0$,
equivalent to a scalar multiple
of $\mathrm{Ai}$ under a local
symplectic coordinate change.
\end{definition}

\begin{remark}[Polynomial phases
  are coordinate representations]
\label{rem:phases-coordinates}
The canonical invariant is the
Lagrangian submanifold
$\Lambda=\{(\beta,\Theta'(\beta))\}
\subset T^*\mathbb{R}$
and its Thom--Boardman stratum;
not the phase formula $e^{i\Theta(\beta)}$,
which is coordinate-dependent.
The expressions $e^{ic_3\beta^3}$
and $\mathrm{Ai}(c_3^{-1/3}u)$
are coordinate representations
of the same A$_2$ geometry in the
$\beta$- and $u$-variables respectively.
\end{remark}

\begin{remark}[Microlocal transport
  under Fourier duality]
\label{rem:microlocal-transport}
The Fourier transform acts on canonical
relations by the exchange
$(\beta,\xi;\,u,\eta)
\mapsto(u,\eta;\,\beta,\xi)$.
An A$_2$-fold singularity is
\emph{preserved under this exchange}
in the following precise sense:
if the Lagrangian map germ of
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
belongs to the stable A$_2$-equivalence
class (Whitney fold of Lagrangian map germs,
stable under small perturbations
in the sense of Mather stability
\cite{Golub1973}),
then the germ of
$\pi_\beta:\mathcal{C}\to T^*\mathbb{R}_\beta$
also belongs to the stable A$_2$-class.
\medskip

\noindent This is a local statement
about stable equivalence of Lagrangian
map germs, not a global assertion that
a category of FIOs is closed under
Fourier transform.
In particular:
if the fold is not stable
(e.g.\ $c_3(c)\to 0$,
so the A$_2$-germ degenerates toward
A$_3$),
this preservation fails.
\end{remark}

\subsection{The fold amplitude}

\begin{equation}\label{eq:Phi-def}
  \Phi(u)
  :=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \quad u\in\mathbb{R},
\end{equation}
$\chi\in C^\infty_c$,
$\alpha_*>0$ fold point.
The WKB amplitude:
\begin{equation}\label{eq:g-def}
  g(\alpha)
  := |\alpha-\alpha_*|^{-1/2}
  e^{i\theta(\alpha)}\chi(\alpha),
\end{equation}
$\theta''(\alpha_*)=0$,
$c_3:=\frac{1}{6}\theta'''(\alpha_*)\ne 0$.
$\beta=\alpha-\alpha_*$,
$\Theta(\beta):=\theta(\alpha_*+\beta)$:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta.
\end{equation}

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
$|\Phi(u)|=O(|u|^{-1/2})$
for all $\theta\in C^\infty$.
For $\theta\equiv 0$:
$\Phi(u)=\sqrt{\pi}\,e^{i\alpha_*u}
|u|^{-1/2}e^{-i\pi/4}
+O(|u|^{-3/2})$.
\end{theorem}

\begin{proof}
Erd\'elyi ($\theta\equiv 0$);
van~der~Corput with weight
$|\beta|^{-1/2}$ (general $\theta$).
\end{proof}

\section{Cubic Phase Induces A$_2$-Fold}
\label{sec:nonpreservation}

\begin{assumption}[Stable A$_2$-fold:
local conditions \Cond]
\label{ass:fold-local}
In addition to
$\Theta(\beta)=c_3\beta^3+O(\beta^4)$,
$c_3\ne 0$:
\begin{enumerate}[\rm(F1)]
\item \textit{(No higher degeneracy.)}
  $\Theta''(\beta)\ne 0$
  for $\beta\ne 0$ in
  $\mathrm{supp}\,\chi\setminus\{0\}$;
  at most one fold point (at $\beta=0$),
  no cusp (A$_3$) or higher.
\item \textit{(Non-cancellation.)}
  The integral
  $\int|\beta|^{-1/2}
  e^{i\Theta}\chi\,d\beta$
  is not identically zero
  in any neighbourhood of $0$.
\item \textit{(Locally monotone
  Legendre map.)}
  $\Theta'$ has exactly one critical
  point (at $\beta=0$) on
  $\mathrm{supp}\,\chi$;
  in particular, no secondary
  stationary points exist.
\end{enumerate}
\end{assumption}

\begin{proposition}[Cubic phase
  $\Rightarrow$ A$_2$-fold
  $\Rightarrow$ Airy normal form \Rig]
\label{prop:nonpreservation}
Under Assumption~\ref{ass:fold-local},
let $F(u):=\int|\beta|^{-1/2}
e^{i(\Theta(\beta)+\beta u)}
\chi(\beta)\,d\beta$.
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold.)} $\mathcal{C}$
  has an A$_2$-fold singularity at
  $(0,0)$:
  $\Theta''(0)=0$
  (from $\Theta=c_3\beta^3+O(\beta^4)$),
  $\Theta'''(0)=6c_3\ne 0$;
  (F1) ensures no higher stratum.
\item \textit{(CFU normal form
  as generating function equivalence.)}
  The CFU theorem \cite{CFU1957}
  provides a local smooth change
  of coordinates
  $\beta\mapsto t(\beta,u)$
  and a smooth function $R(u)$
  (independent of $\beta$)
  such that:
  \begin{equation}\label{eq:CFU-nf}
    \Theta(\beta)+u\beta
    = \tfrac{1}{3}t^3
    - \mu(u)\,t
    + R(u),
  \end{equation}
  where $\mu(u)=c_3^{-1/3}u\cdot
  (1+O(u))$ is a smooth function
  (the ``spectral parameter'' of the
  Airy function,
  equalling $c_3^{-1/3}u$ to
  leading order).
  This is a \emph{generating function
  equivalence}:
  the coordinate change $\beta\mapsto t$
  equates the phase generating
  function of $\mathcal{C}$ on the
  left with the Airy generating
  function on the right.
  \medskip

  The Jacobi factor from the change
  of variables is:
  \begin{equation}\label{eq:CFU-jacobi}
    J(t,u) := \frac{d\beta}{dt}
    = \frac{\partial t/\partial\beta}{1}
    \bigg|^{-1}
    ,
  \end{equation}
  bounded above and below
  by Assumption~\ref{ass:microlocal}(iv).
  Condition~(F2) ensures $J\ne 0$.
  Condition~(F3) ensures $\beta\mapsto t$
  is a diffeomorphism on
  $\mathrm{supp}\,\chi$.
  \medskip

  Under~\eqref{eq:CFU-nf},
  the integral becomes:
  \begin{equation}\label{eq:airy-FT}
    F(u)
    \;\sim\;
    e^{iR(u)}\cdot
    \int_{\mathbb{R}}
    |\beta(t,u)|^{-1/2}J(t,u)\,
    e^{i(\frac{1}{3}t^3-\mu(u)t)}\,dt.
  \end{equation}
  The amplitude
  $|\beta(t,u)|^{-1/2}J(t,u)\in
  S^{-1/2}_{1,0}$ (in the $u$-variable)
  is a symbol of order~$-1/2$
  \emph{in the $u$-variable}
  (not directly inherited from the
  $\beta$-side);
  see Remark~\ref{rem:symbol-u}.
  The standard Airy representation
  then gives:
  \begin{equation}\label{eq:airy-rep}
    F(u)
    \;\sim\;
    e^{iR(u)}\,a_0(u,c)\,
    \mathrm{Ai}\!\bigl(-\mu(u)\bigr),
    \quad|u|\to\infty,
  \end{equation}
  where $a_0(u,c)\in S^{-1/2}_{1,0}$.
\item \textit{(Stationary phase value.)}
  The large-$|u|$ oscillatory phase
  of $\mathrm{Ai}(-\mu(u))$
  for $u\to-\infty$ is:
  \begin{equation}\label{eq:stat-phase-value}
    S(u):=\tfrac{2}{3}c_3^{-1/2}|u|^{3/2},
  \end{equation}
  the stationary phase value
  $\Theta(\beta^*)+u\beta^*$
  at $\beta^*\colon\Theta'(\beta^*)=-u$.
  This is an asymptotic quantity only;
  the Airy function has no global
  phase representation.
\end{enumerate}
\end{proposition}

\begin{remark}[Symbol $a$ is in the
  $u$-variable]
\label{rem:symbol-u}
The symbol $a_0(u,c)$ in~\eqref{eq:airy-rep}
arises from the amplitude
$|\beta(t,u)|^{-1/2}J(t,u)$
after integration over $t$.
It is a symbol of order~$-1/2$
\emph{in the output variable $u$},
measuring the rate of decay of
the amplitude as $|u|\to\infty$
in the $u$-space.
It is \emph{not} directly inherited
from the $\beta$-side singularity
$|\beta|^{-1/2}$:
the CFU change of variables
mixes $\beta$- and $u$-dependence,
and the resulting $u$-decay
comes from the Jacobian
$J(t,u)\sim c_3^{-1/6}|u|^{-1/2}$
for large $|u|$.
\end{remark}

\section{Airy Normal Form and
  Global Integrability}
\label{sec:normalform}

\begin{assumption}[Uniform A$_2$-stability \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$.
In addition:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta)=c_3\beta^3+O(\beta^4)$
  uniformly in a $c$-independent neighbourhood;
\item \textit{(A$_2$-stability)}
  $c_3(c)\ge\delta_0>0$ uniformly;
\item \textit{(Uniform fold projection)}
  $|\Theta''(\beta^*(u))|
  \ge\delta_1|u|^{1/2}$
  for $\delta_2\le|u|\le c^{\theta_0/3}$;
\item \textit{(Uniform Jacobian)}
  $0<\delta_3\le|J(t,u)|
  \le\delta_3^{-1}$
  uniformly in $(u,c)$;
\item \textit{(No secondary stationary
  points in $u$)}
  Assumption~\ref{ass:fold-local}(F3)
  holds uniformly in $c$:
  $\Theta'$ has exactly one critical
  point on $\mathrm{supp}\,\chi$
  for all $c\ge c_0$.
\end{enumerate}
\end{assumption}

\begin{theorem}[Airy normal form
  (microlocal, near fold point) \Rig]
\label{thm:airy-normalform}
Under Assumption~\ref{ass:microlocal},
microlocally near the fold point
$(\beta_0,u_0)=(0,0)$,
the fold amplitude~\eqref{eq:Phi-WKB}
satisfies:
\begin{equation}\label{eq:airy-normalform}
  \Phi(u)
  = e^{i\alpha_* u}\,
  a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
\end{equation}
where:
\begin{itemize}
\item $a(u,c)\in S^{-1/2}_{1,0}$:
  $|\partial_u^j a(u,c)|
  \le C_j(1+|u|)^{-1/2-j}$,
  uniformly in $c\ge c_0$
  and $\delta_2\le|u|\le c^{\theta_0/3}$;
\item $a$ is a symbol in the
  \emph{$u$-variable},
  arising from the CFU Jacobian
  (Remark~\ref{rem:symbol-u}),
  not directly from the $\beta$-side
  singularity;
\item $e^{i\alpha_*u}$ is a global
  Fourier shift, carrying no
  microlocal fold information;
\item $S(u)$ of~\eqref{eq:stat-phase-value}
  is the asymptotic stationary phase
  value of $\mathrm{Ai}(c_3^{-1/3}u)$;
  it is not a global phase function.
\end{itemize}
\end{theorem}

\begin{proof}[Proof sketch]
Apply the CFU coordinate change
of Proposition~\ref{prop:nonpreservation}(2)
to~\eqref{eq:Phi-WKB},
uniformly by conditions~(iii)--(v).
Condition~(iii) ensures non-degeneracy
of the fold projection;
(iv) the Jacobian bound;
(v) absence of secondary stationary points,
so the CFU change of variables
is a global diffeomorphism on
$\mathrm{supp}\,\chi$.
The resulting integral is the
standard Airy representation;
$a\in S^{-1/2}_{1,0}$
follows from Erd\'elyi and condition~(i).
\end{proof}

\begin{lemma}[Global $L^1$-growth
  estimate \Rig]
\label{lem:airy-integral}
Under Theorem~\ref{thm:airy-normalform},
with additionally:
\begin{enumerate}[\rm(G1)]
\item \textit{(Uniformity of stationary
  region in $u$.)}
  The Airy representation holds
  uniformly for
  $|u|\le c^{\theta_0/3}$;
\item \textit{(Partition of unity.)} A
  partition of unity
  $\{\varphi_k\}$ in $u$-space
  decomposes $[0,U]$ into:
  the fold zone
  $|u|\le c^{\theta_0/3}$
  (where Theorem~\ref{thm:airy-normalform}
  applies) and the far zone;
\item \textit{(No secondary stationary
  points in $u$.)}
  Assumption~\ref{ass:microlocal}(v).
\end{enumerate}
Then:
$\bigl|\int_0^U\Phi(u)\,du\bigr|
= O(U^{1/4})$
as $U\to\infty$.
This bound reflects a
\emph{better integrability exponent
due to the structural Airy decay};
it does not use oscillatory
cancellation.
\end{lemma}

\begin{proof}
The bound is obtained via an
\emph{absolute-value estimate} throughout;
no oscillatory cancellation
is invoked.
\medskip

\noindent\textit{Partition of unity.}
Fix a smooth partition
$1=\varphi_{+}(u)+\varphi_{-}(u)
+\varphi_0(u)$
where $\varphi_+$ is supported on
$[1,+\infty)$,
$\varphi_-$ on $(-\infty,-1]$,
and $\varphi_0$ on $[-1,1]$.
The contribution from
$\varphi_0$ is $O(1)$.
\medskip

\noindent\textit{Far zone
($u > c^{\theta_0/3}$):}
Theorem~\ref{thm:endpoint}
gives $|\Phi(u)|=O(u^{-1/2})$,
so $\int_{c^{\theta_0/3}}^U u^{-1/2}du
= O(U^{1/2})$.
However, this region contributes
negligibly to the fold mechanism;
it is bounded by
$O(c^{-\theta_0/6})$ for $U\le c^{\theta_0/3}$
via the endpoint bound.
\medskip

\noindent\textit{Fold zone,
$\varphi_+$-part ($1\le u\le U$):}
For $t=c_3^{-1/3}u>0$,
$|\mathrm{Ai}(t)|
\le Cu^{-1/4}e^{-c'u^{3/2}}$
\cite{Olver1974}: exponential decay.
$|e^{i\alpha_*u}|=1$,
$|a(u,c)|\le Cu^{-1/2}$:
\[
  \int_1^U|a|\cdot
  |\mathrm{Ai}(c_3^{-1/3}u)|
  \cdot|e^{i\alpha_*u}|\,du
  \le C\int_1^\infty
  u^{-3/4}e^{-c'u^{3/2}}\,du=O(1).
\]
\medskip

\noindent\textit{Fold zone,
$\varphi_-$-part ($-U\le u\le -1$):}
For $t=c_3^{-1/3}u<0$,
$|\mathrm{Ai}(t)|
=|\mathrm{Ai}(-|c_3^{-1/3}u|)|
\le C|u|^{-1/4}$
\cite{Olver1974}
(oscillatory; no exponential decay).
$|e^{i\alpha_*u}|=1$,
$|a(-u,c)|\le Cu^{-1/2}$:
\[
  \int_1^U|a(-u,c)|\cdot
  |\mathrm{Ai}(-c_3^{-1/3}u)|\,du
  \le C\int_1^U u^{-3/4}\,du
  = O(U^{1/4}).
\]
\medskip

\noindent\textit{Conclusion.}
The dominant contribution is the
oscillatory Airy region ($u<0$):
\[
  \left|\int_0^U\Phi\,du\right|
  \le O(1)+O(U^{1/4})=O(U^{1/4}).
\]
This improvement over
Ces\`aro $O(U^{1/2})$
is due entirely to the
\emph{structural decay}
$|\mathrm{Ai}(-t)|=O(t^{-1/4})$
combined with $a\in S^{-1/2}_{1,0}$:
these two factors give
$u^{-1/2}\cdot u^{-1/4}=u^{-3/4}$
whose integral gives $U^{1/4}$.
The $e^{i\alpha_*u}$ factor
plays no role (bounded by~1 throughout).
\end{proof}

\begin{corollary}[Global $L^1$-bound
  from Assumption~\ref{ass:microlocal} \Rig]
\label{cor:global-bound}
Under Assumption~\ref{ass:microlocal}
(which implies
Lemma~\ref{lem:airy-integral}(G1)--(G3)):
\[
  \left|\int_0^U\Phi(u)\,du\right|
  = O(U^{1/4}),
  \quad U\to\infty.
\]
This is a global $L^1$-growth bound
obtained from the local
Airy normal form structure
of Theorem~\ref{thm:airy-normalform}
via the partition of unity and
uniformity conditions.
The improvement is due to
structural Airy decay;
no oscillatory cancellation is used.
\end{corollary}

\subsection{Converse and FIO rigidity}
\label{subsec:converse}

\begin{hypothesis}[FIO inversion
  rigidity \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ lies in the
Airy class
($a\in S^{-1/2}_{1,0}$,
stationary phase value
$S(u)\sim C|u|^{3/2}$ uniformly in $c$)
and additionally:
\begin{enumerate}[\rm(a)]
\item \textit{(Symbol ellipticity
  on $\mathcal{C}$.)}
  $\sigma(g\mapsto\Phi)\big|_\mathcal{C}
  \ne 0$;
\item \textit{(Injectivity
  mod smoothing.)}
  $g\mapsto\Phi$ is injective
  modulo smoothing operators;
\item \textit{(Wavefront stability.)}
  $\mathrm{WF}(\Phi)
  =\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
\end{enumerate}
Then the canonical relation of $g$
contains an A$_2$-fold singularity
uniformly in $c$.
\end{hypothesis}

\begin{remark}[Status]
\label{rem:rigidity-status}
Conditions (a)--(c) are standard
microlocal ellipticity and injectivity
conditions \cite{Zworski2012}.
(a) prevents amplitude cancellation
unrelated to fold geometry;
(b) rules out smoothing perturbations
mimicking Airy oscillations;
(c) ensures the wavefront set
faithfully reflects the A$_2$ stratum.
Verification for PSWF is
Paper~XVII.
\end{remark}

\begin{corollary}[Conditional equivalence
  \Cond]
\label{cor:conditional-equiv}
Under Hypothesis~\ref{hyp:FIO-rigidity}:
$\mathrm{Airy{-}class\ bound}
\iff
\mathrm{Assumption~\ref{ass:microlocal}}$.
Without it:
$\mathrm{Assumption~\ref{ass:microlocal}}
\Rightarrow
\mathrm{Airy{-}class\ bound}$ only.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$
satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item \textit{(A$_2$-stability)}
  $c_3(c)\ge\delta_0>0$?
  Approach:
  Olver turning-point WKB
  \cite{Olver1974}.
\item \textit{(No higher degeneracy)}
  $\theta''(\alpha)\ne 0$
  for $\alpha\ne\alpha_*$ near $\alpha_*$,
  uniformly?
\item \textit{(Uniform Jacobian)}
  Assumption~\ref{ass:microlocal}(iv)
  for the PSWF CFU Jacobian?
\end{enumerate}
Outcomes:
\begin{enumerate}[\rm(A)]
\item Both Assumptions hold:
  $|\int_0^U\Phi|=O(U^{1/4})$,
  $S(c)=O(c^{\theta_0/2})$.
\item $c_3(c)\to 0$
  or (F1) fails:
  $\mathcal{C}_c$ exits A$_2$ stratum;
  Ces\`aro sharp.
\end{enumerate}
\end{problem}
