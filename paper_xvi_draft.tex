\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases, A$_2$-Fold Singularities,\\
  and the Airy Normal Form}

\author{[Author]}

\begin{abstract}
A cubic WKB phase $c_3\beta^3$ produces
a degenerate stationary point of order~2,
which is an A$_2$-fold singularity of
the canonical relation $\mathcal{C}$.
The Airy distribution is the universal
local normal form of this singularity.
Three results are established,
each explicitly conditional on
Assumption~\ref{ass:microlocal}
(uniform A$_2$-stability,
conditions (i)--(vi)):
\begin{enumerate}[\rm(I)]
\item \textit{(Local Airy normal form.)}
  Theorem~\ref{thm:airy-normalform}:
  microlocally near the fold point,
  \[
    \Phi(u) = e^{i\alpha_*u}\bigl[
    a(u,c)\,\mathrm{Ai}(c_3^{-1/3}u)
    + r(u,c)\bigr],
  \]
  $a\in S^{-1/2}_{1,0}$
  (conditional on (i)--(vi)),
  $r$ rapidly decaying
  (conditional on (vi)).
\item \textit{(Pointwise decay.)}
  Corollary~\ref{cor:pointwise}:
  $|\Phi(u)|\le C(1+|u|)^{-3/4}$
  for all $u$,
  with $C$ uniform across zones
  (conditional on (i)--(vi)).
\item \textit{(Ces\`aro improvement.)}
  Corollary~\ref{cor:global-bound}:
  $|\int_0^U\Phi\,du|=O(U^{1/4})$,
  conditional on (i)--(vi).
\end{enumerate}
The forward direction (Assumption
$\Rightarrow$ bound) is rigorous
under (i)--(vi).
The converse requires the non-standard
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\maketitle

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold,
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the projection to the right factor.
\medskip

\noindent\textit{Geometric definition.}
$\mathcal{C}$ has an
\emph{A$_2$-fold singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$;
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$.
\end{enumerate}
This is the A$_2$-type in Arnold's
classification \cite{Arnold1972}.
\medskip

\noindent\textit{Coordinate reduction.}
For phase $\Psi(\beta,u)
=\Theta(\beta)+u\beta$,
conditions (i)--(ii) reduce to
$\Theta''(\beta_0)=0$,
$\Theta'''(\beta_0)\ne 0$.
These are coordinate expressions;
conditions (i)--(ii) are invariant
under canonical transformations.
By the Thom--Boardman theorem
\cite{Golub1973},
any FIO with an A$_2$-fold is,
\emph{microlocally near $p_0$},
equivalent to $\mathrm{Ai}$
under a local symplectic change.
This is a local-in-phase-space statement.
\end{definition}

\begin{remark}[Microlocal transport]
\label{rem:microlocal-transport}
The Fourier transform acts by the
exchange
$(\beta,\xi;u,\eta)\mapsto(u,\eta;\beta,\xi)$.
If the Lagrangian map germ of $\pi_u$
belongs to the stable A$_2$-class
(Mather stability \cite{Golub1973}),
then the germ of $\pi_\beta$ also
belongs to the stable A$_2$-class.
This is a local statement about
stable equivalence of map germs;
it fails if $c_3(c)\to 0$.
\end{remark}

\subsection{The fold amplitude}

\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta,
\end{equation}
$\chi\in C^\infty_c$,
$\alpha_*>0$, $\theta''(\alpha_*)=0$,
$c_3=\frac{1}{6}\theta'''(\alpha_*)\ne 0$,
$\Theta(\beta)=\theta(\alpha_*+\beta)$.

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
$|\Phi(u)|=O(|u|^{-1/2})$
for all $\theta\in C^\infty$.
\end{theorem}

\begin{proof}
Van~der~Corput with weight
$|\beta|^{-1/2}$; Erd\'elyi for
$\theta\equiv 0$.
\end{proof}

\section{CFU Reduction, Jacobian
  Rescaling, and Symbol Stability}
\label{sec:reduction}

\begin{assumption}[Local stable A$_2$-fold]
\label{ass:fold-local}
$\Theta(\beta)=c_3\beta^3+O(\beta^4)$,
$c_3\ne 0$, and:
\begin{enumerate}[\rm(F1)]
\item $\Theta''(\beta)\ne 0$
  for $\beta\ne 0$ near $0$;
  no cusp or higher stratum.
\item $\int|\beta|^{-1/2}
  e^{i\Theta}\chi\,d\beta
  \not\equiv 0$ near $0$.
\item $\Theta'$ has exactly one
  critical point ($\beta=0$)
  on $\mathrm{supp}\,\chi$;
  no secondary stationary points.
\end{enumerate}
\end{assumption}

\begin{proposition}[CFU as FIO
  conjugation \Rig]
\label{prop:CFU}
Under Assumption~\ref{ass:fold-local},
fix $U_\epsilon=\{|\beta|<\epsilon\}$
and let $1=\phi_{\mathrm{fold}}
+\phi_{\mathrm{far}}$ be a smooth
partition with
$\phi_{\mathrm{fold}}\equiv 1$
on $U_{\epsilon/2}$.
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold.)}
  $\Theta''(0)=0$,
  $\Theta'''(0)=6c_3\ne 0$;
  (F1) confirms A$_2$-type.
\item \textit{(FIO conjugation.)}
  In $U_\epsilon$,
  the CFU theorem \cite{CFU1957}
  provides a \emph{uniform local
  parametrization} $\beta\mapsto t(\beta,u)$
  of the fold region
  (not a global diffeomorphism on
  $\mathrm{supp}\,\chi$)
  such that the oscillatory integral
  is conjugated to Airy normal form
  via an FIO transformation:
  \[
    \int_{U_\epsilon}
    |\beta|^{-1/2}
    e^{i(\Theta(\beta)+u\beta)}
    \phi_{\mathrm{fold}}\,d\beta
    \;=\;
    \int_{\mathbb{R}}
    \tilde{a}(t,u)\,
    e^{i(\frac{1}{3}t^3-\mu(u)t+R(u))}
    \,dt,
  \]
  where $\mu(u)=c_3^{-1/3}u+O(u^2)$,
  $R(u)$ smooth,
  and $\tilde{a}(t,u)
  =|\beta(t,u)|^{-1/2}J(t,u)
  \tilde\chi(t,u)$ is the
  transported amplitude
  with Jacobian $J(t,u):=d\beta/dt$.
  This is an exact identity
  for the \emph{oscillatory integral},
  not a pointwise phase identity.
\item \textit{(Far region.)}
  On $\mathrm{supp}\,\phi_{\mathrm{far}}$,
  $\Theta''\ne 0$ by (F1);
  van~der~Corput gives
  $|\Phi_{\mathrm{far}}(u)|
  \le C|u|^{-1/2}$.
\end{enumerate}
\end{proposition}

\begin{remark}[CFU as FIO, not phase identity]
\label{rem:CFU-FIO}
Proposition~\ref{prop:CFU}(2) is an
identity for the \emph{oscillatory
integral} via FIO conjugation,
not a pointwise Taylor identity
$\Theta(\beta)+u\beta
= \frac{1}{3}t^3-\mu t+R$.
The amplitude $\tilde{a}$ absorbs
all Jacobian factors and Taylor
remainders from the change of variables.
This is the standard CFU interpretation
\cite{CFU1957,Olver1974}.
\end{remark}

\begin{lemma}[Jacobian rescaling \Rig]
\label{lem:Jacobian-rescaling}
Under the canonical Airy rescaling
$t = c_3^{-1/6}|u|^{1/2}\tilde{t}$,
the Jacobian $J(t,u)$ transforms as:
\begin{equation}\label{eq:Jacobi-homog}
  J(t,u)
  = c_3^{-1/6}|u|^{-1/2}
  \tilde{J}(\tilde{t},u),
  \quad
  \tilde{J} = 1+O(|u|^{-1})
  \text{ uniformly as }|u|\to\infty.
\end{equation}
The $|u|^{-1/2}$ factor arises from
the homogeneity of the cubic normal form,
not from the original $|\beta|^{-1/2}$.
\end{lemma}

\begin{proof}
Under $t=c_3^{-1/6}|u|^{1/2}\tilde{t}$,
the Airy phase
$\frac{1}{3}t^3-\mu(u)t$
becomes
$c_3^{-1/2}|u|^{3/2}
(\frac{1}{3}\tilde{t}^3
-\mathrm{sgn}(u)\tilde{t})$.
The transformation law
$J\,dt=d\beta$ gives
$J=(d\beta/d\tilde{t})\cdot(d\tilde{t}/dt)
=\tilde{J}\cdot c_3^{1/6}|u|^{-1/2}$;
$\tilde{J}=d\beta/d\tilde{t}
=1+O(|u|^{-1})$ follows from
the CFU expansion \cite{CFU1957}.
\end{proof}

\begin{lemma}[Symbol stability under
  parametric FIO reduction \Rig]
\label{lem:symbol-stability}
Under Assumption~\ref{ass:microlocal}
(in particular condition (vi)),
let $\tilde{a}(\tilde{t},u)
:=|\beta(\tilde{t},u)|^{-1/2}
\tilde{J}(\tilde{t},u)
\cdot\tilde\chi(\tilde{t},u)$.
\begin{enumerate}[\rm(1)]
\item \textit{(Amplitude class in
  $(\tilde{t},u)$.)}
  $|\partial_{\tilde{t}}^j
  \partial_u^k\tilde{a}(\tilde{t},u)|
  \le C_{jk}(1+|u|)^{-1/2-k}$
  uniformly.
  This uses condition (vi)
  (derivative bounds on
  $t(\beta,u)$ in symbol classes).
\item \textit{(Parametric stationary
  phase in $\tilde{t}$.)}
  By H\"ormander \cite{Hormander1983},
  Ch.~7.7:
  \begin{equation}\label{eq:airy-decomp}
    \int_{\mathbb{R}}
    \tilde{a}(\tilde{t},u)\,
    e^{ic_3^{-1/2}|u|^{3/2}
    (\tilde{t}^3/3-\mathrm{sgn}(u)\tilde{t})}
    \,d\tilde{t}
    = a_0(u,c)\,
    \mathrm{Ai}(c_3^{-1/3}u)
    + r_0(u,c),
  \end{equation}
  where $a_0$ is the symbol coefficient
  extracted from the stationary-phase
  expansion at the cubic critical point
  (not by division by $\mathrm{Ai}$),
  and $r_0$ is rapidly decaying
  (see Remark~\ref{rem:remainder}).
\item \textit{(Symbol and remainder.)}
  Under condition (vi):
  \begin{align}
    \label{eq:symbol-u}
    |\partial_u^j a_0(u,c)|
    &\le C_j(1+|u|)^{-1/2-j},\\
    \label{eq:remainder-u}
    |r_0(u,c)|
    &\le C_{N}(1+|u|)^{-N}
    \quad\text{for each fixed }N.
  \end{align}
\end{enumerate}
\end{lemma}

\begin{remark}[Remainder and
  non-degeneracy]
\label{rem:remainder}
Estimate~\eqref{eq:remainder-u}
holds for each fixed $N$,
uniformly in $c\ge c_0$,
\emph{provided} the non-stationary
region satisfies
$|\partial_{\tilde{t}}
(\tilde{t}^3/3
-\mathrm{sgn}(u)\tilde{t})|
\ge\delta_4>0$
uniformly in $(u,c)$
(Assumption~\ref{ass:microlocal}(vi)).
Without (vi), rapid decay holds
only for each fixed $c$.
\end{remark}

\section{Main Results}
\label{sec:results}

\begin{assumption}[Uniform A$_2$-stability
  \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$, with:
\begin{enumerate}[\rm(i)]
\item $\Theta=c_3\beta^3+O(\beta^4)$
  in a $c$-independent neighbourhood;
\item $c_3(c)\ge\delta_0>0$;
\item $|\Theta''(\beta^*(u))|
  \ge\delta_1|u|^{1/2}$
  for $\delta_2\le|u|\le c^{\theta_0/3}$;
\item $|\partial_u^k J(t,u)|
  \le C_k(1+|u|)^{-1/2-k}$
  uniformly for $k=0,1,2,\ldots$;
\item no secondary stationary points
  uniformly in $c$;
\item \textit{(Uniform symbol-class
  regularity of the CFU map.)}
  There exist $\delta_4,\delta_5>0$
  and constants $C_{jk}$ such that,
  uniformly in $(u,c)$:
  \begin{itemize}
  \item $|\det(dt/d\beta)|\ge\delta_5$
    (Jacobian non-vanishing);
  \item $|\partial_\beta^j\partial_u^k
    t(\beta,u)|\le C_{jk}$
    for all $j,k\ge 0$
    (derivative bounds in symbol classes);
  \item $|\partial_{\tilde{t}}
    (\tilde{t}^3/3
    -\mathrm{sgn}(u)\tilde{t})|
    \ge\delta_4$
    outside the Airy critical region
    (uniform non-degeneracy of
    the non-stationary region).
  \end{itemize}
\end{enumerate}
\end{assumption}

\begin{remark}[Role of condition (vi)]
\label{rem:vi-role}
Condition (vi) is what promotes
all symbol estimates from
``for fixed $c$'' to
``uniform in $c\ge c_0$'':
\begin{itemize}
\item $|\det(dt/d\beta)|\ge\delta_5$
  ensures the change of variables
  $\beta\mapsto t$ stays in symbol
  classes (not merely $C^\infty$);
\item the derivative bounds ensure
  transported amplitudes retain
  symbol estimates;
\item the non-degeneracy of the
  non-stationary region ensures
  uniform rapid decay of $r_0$.
\end{itemize}
Verifying (vi) for PSWF
is part of Open Problem~\ref{prob:PSWF-microlocal}.
\end{remark}

\begin{theorem}[Conditional local
  Airy normal form \Rig]
\label{thm:airy-normalform}
\textbf{Conditional on
Assumption~\ref{ass:microlocal}
(i)--(vi).}
Microlocally near the fold point,
for $\delta_2\le|u|\le c^{\theta_0/3}$:
\begin{equation}\label{eq:airy-nf}
  \Phi(u)
  = e^{i\alpha_*u}\bigl[
  a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr)
  + r(u,c)\bigr],
\end{equation}
where:\vspace{-4pt}
\begin{itemize}
\item $a(u,c)\in S^{-1/2}_{1,0}$
  uniform in $c$
  (Lemma~\ref{lem:symbol-stability}(3),
  requires (vi));
\item $r(u,c)$ rapidly decaying:
  $|r|\le C_N(1+|u|)^{-N}$
  for each fixed $N$,
  uniform in $c$ (requires (vi));
\item the oscillatory structure of
  $\Phi$ is entirely encoded in
  $\mathrm{Ai}(c_3^{-1/3}u)$;
  $a$ is smooth amplitude modulation.
\end{itemize}
\end{theorem}

\begin{proof}[Proof sketch]
Fold zone $U_\epsilon$:
Proposition~\ref{prop:CFU}(2)
(FIO conjugation, exact identity
for the oscillatory integral);
Lemma~\ref{lem:Jacobian-rescaling}
(Airy rescaling);
Lemma~\ref{lem:symbol-stability}
(amplitude class $(1)$ $\to$
symbol in $u$ $(3)$
via H\"ormander Ch.~7.7).
Far zone: Proposition~\ref{prop:CFU}(3).
Uniformity: conditions (i)--(vi).
\end{proof}

\begin{lemma}[Airy-symbol product
  estimate in the fold zone \Rig]
\label{lem:product-estimate}
\textbf{Valid in the fold zone
after canonical Airy rescaling;
not a global statement.}
Under Theorem~\ref{thm:airy-normalform},
for $|u|\ge\delta_2$:\vspace{-4pt}
\begin{enumerate}[\rm(a)]
\item $u\ge\delta_2$:
  $|a(u,c)|\le C_2 u^{-1/2}$
  and $|\mathrm{Ai}(c_3^{-1/3}u)|
  \le C_1 u^{-1/4}e^{-c'u^{3/2}}$
  \cite{Olver1974};\quad
  $|a\cdot\mathrm{Ai}|\le C_1C_2\,u^{-3/4}$.
\item $u\le-\delta_2$:
  $|a|\le C_2|u|^{-1/2}$
  and $|\mathrm{Ai}(c_3^{-1/3}u)|
  \le C_1|u|^{-1/4}$
  \cite{Olver1974};\quad
  $|a\cdot\mathrm{Ai}|\le C_1C_2\,|u|^{-3/4}$.
\end{enumerate}
The factors $|u|^{-1/2}$
(from $a\in S^{-1/2}_{1,0}$)
and $|u|^{-1/4}$
(from Airy decay)
are not independent;
they both arise from the single
Airy normal form representation
\eqref{eq:airy-nf}
and the canonical Airy rescaling.
The bound $|u|^{-3/4}$ is valid
in the fold zone after
Lemma~\ref{lem:Jacobian-rescaling};
it is not claimed globally
without the far-zone argument below.
\end{lemma}

\begin{corollary}[Pointwise decay
  \Cond]
\label{cor:pointwise}
\textbf{Conditional on
Assumption~\ref{ass:microlocal}
(i)--(vi).}
There exists a single constant $C>0$,
uniform in $c\ge c_0$ and across
all three zones, such that:
\begin{equation}\label{eq:pointwise}
  |\Phi(u)|\le C(1+|u|)^{-3/4}
  \quad\text{for all }u\in\mathbb{R}.
\end{equation}
\end{corollary}

\begin{proof}
Three zones, uniform constant
$C=\max(C_1C_2,\,C_3,\,C_4)$:
\begin{itemize}
\item \textit{Fold zone $|u|\ge\delta_2$:}
  Lemma~\ref{lem:product-estimate}.
\item \textit{Compact zone $|u|\le 1$:}
  $|\Phi|\le C_3$ by compactness
  and continuity of FIO amplitude
  (Assumption (vi) provides uniform
  $C^\infty$-bounds;
  Erd\'elyi \cite{Erdelyi1956});
  trivially $\le C_3(1+|u|)^{-3/4}$.
\item \textit{Far zone:}
  Theorem~\ref{thm:endpoint}:
  $|\Phi|\le C_4|u|^{-1/2}
  \le C_4|u|^{-3/4}$
  for $|u|\ge 1$.
\end{itemize}
Transition between fold and far zones
($c^{\theta_0/3}\le|u|
\le 2c^{\theta_0/3}$):
both estimates apply with the same
constants, by conditions (iii)--(iv).
$C:=\max(C_1C_2,C_3,C_4)$
is uniform in $c\ge c_0$.
\end{proof}

\begin{remark}[On oscillation]
\label{rem:oscillation}
For $u<0$, the $|u|^{-1/4}$
Airy decay is a consequence of
stationary phase \emph{within}
the Airy representation
(Theorem~\ref{thm:airy-normalform}).
Once the pointwise bound
\eqref{eq:pointwise} is available,
the $u$-integral in
Corollary~\ref{cor:global-bound}
is estimated by absolute value only.
The method yields a
\emph{geometric decay $\to$
integrability} bound;
oscillatory improvement in $u$
would require a second stationary
phase step in $u$ (beyond scope here).
\end{remark}

\begin{corollary}[Ces\`aro improvement
  \Cond]
\label{cor:global-bound}
\textbf{Conditional on
Assumption~\ref{ass:microlocal}
(i)--(vi).}
\[
  \left|\int_0^U\Phi(u)\,du\right|
  \le C\int_0^U(1+u)^{-3/4}\,du
  = O(U^{1/4}),
\]
improving upon Ces\`aro $O(U^{1/2})$.
The single constant $C$ of
Corollary~\ref{cor:pointwise}
ensures global stability.
The improvement has two origins
(within the Airy normal form):
\begin{enumerate}[\rm(a)]
\item \textit{(Geometric.)}
  A$_2$-fold forces the Airy normal
  form with $|u|^{-1/4}$ decay;
\item \textit{(Analytic.)}
  $a\in S^{-1/2}_{1,0}$
  contributes $|u|^{-1/2}$.
\end{enumerate}
\end{corollary}

\subsection{Converse and FIO rigidity}

\begin{assumption}[Microlocal reconstruction
  (non-standard) \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ is in the Airy class and:
\begin{enumerate}[\rm(a)]
\item $\sigma(g\mapsto\Phi)
  \big|_\mathcal{C}\ne 0$;
\item $g\mapsto\Phi$ is injective
  modulo smoothing operators;
\item $\mathrm{WF}(\Phi)
  \subseteq\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
\end{enumerate}
Then $\mathcal{C}$ contains an
A$_2$-fold singularity
uniformly in $c$.
\medskip

\noindent\textit{Status.}
This is a
\emph{non-standard microlocal
reconstruction hypothesis},
not a consequence of standard
FIO theory \cite{Zworski2012,Hormander1983}.
Verification for PSWF: Paper~XVII.
\end{assumption}

\begin{corollary}[Conditional equivalence]
\label{cor:conditional-equiv}
Under Assumption~\ref{hyp:FIO-rigidity}:
Airy-class bound $\Rightarrow$
Assumption~\ref{ass:microlocal}
(modulo the non-standard hypothesis).
The forward direction
(Assumption~\ref{ass:microlocal}
$\Rightarrow$ bound)
is rigorous without
Assumption~\ref{hyp:FIO-rigidity}.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$
satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
In particular:
\begin{enumerate}[\rm(a)]
\item $c_3(c)\ge\delta_0>0$?
  (Olver turning-point WKB
  \cite{Olver1974}.)
\item $\theta''(\alpha)\ne 0$
  for $\alpha\ne\alpha_*$, uniformly?
\item Conditions (iv) and (vi)
  for the PSWF CFU Jacobian?
\end{enumerate}
Outcomes:
(A)~Assumptions hold:
$|\int_0^U\Phi|=O(U^{1/4})$.
(B)~$c_3(c)\to 0$:
Ces\`aro sharp.
\end{problem}
