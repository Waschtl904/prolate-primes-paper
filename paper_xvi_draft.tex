\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases, A$_2$-Fold Singularities,\\
  and the Airy Normal Form}

\author{[Author]}

\begin{abstract}
A cubic WKB phase $c_3\beta^3$ produces
a degenerate stationary point of order~2,
which is an A$_2$-fold singularity of
the canonical relation $\mathcal{C}$.
The Airy distribution is the universal
local normal form of this singularity.
Under the uniform A$_2$-stability
conditions of Assumption~\ref{ass:microlocal},
three results are established:
\begin{enumerate}[\rm(I)]
\item \textit{(Local Airy normal form.)}
  Theorem~\ref{thm:airy-normalform}:
  microlocally near the fold point,
  \[
    \Phi(u) = e^{i\alpha_*u}\bigl[
    a(u,c)\,\mathrm{Ai}(c_3^{-1/3}u)
    + r(u,c)\bigr],
  \]
  $a\in S^{-1/2}_{1,0}$ uniform in $c$,
  $r$ rapidly decaying uniform in $c$
  (under non-degeneracy
  Assumption~\ref{ass:microlocal}(vi)).
\item \textit{(Pointwise decay.)}
  Lemma~\ref{lem:pointwise}:
  $|\Phi(u)|\le C(1+|u|)^{-3/4}$
  with a single constant $C$
  uniform across all three zones.
\item \textit{(Ces\`aro improvement.)}
  Corollary~\ref{cor:global-bound}:
  $|\int_0^U\Phi\,du|=O(U^{1/4})$.
\end{enumerate}
The forward direction is rigorous;
the converse requires
Hypothesis~\ref{hyp:FIO-rigidity}
(non-standard microlocal reconstruction
hypothesis).
\end{abstract}

\maketitle

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold,
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the projection to the right factor.
\medskip

\noindent\textit{Geometric definition.}
$\mathcal{C}$ has an
\emph{A$_2$-fold singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$;
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$.
\end{enumerate}
This is the A$_2$-type in Arnold's
classification \cite{Arnold1972}.
\medskip

\noindent\textit{Coordinate reduction.}
For phase $\Psi(\beta,u)
=\Theta(\beta)+u\beta$,
conditions (i)--(ii) reduce to:
\begin{align}
  \label{eq:fold-i}
  \Theta''(\beta_0)&=0,\\
  \label{eq:fold-ii}
  \Theta'''(\beta_0)&\ne 0.
\end{align}
These are coordinate expressions;
conditions (i)--(ii) are invariant
under canonical transformations.
By the Thom--Boardman theorem
\cite{Golub1973},
any FIO with an A$_2$-fold is,
\emph{microlocally near $p_0$}
(local-in-phase-space),
equivalent to $\mathrm{Ai}$
under a local symplectic change.
\end{definition}

\begin{remark}[Microlocal transport]
\label{rem:microlocal-transport}
The Fourier transform acts by the
canonical relation exchange
$(\beta,\xi;u,\eta)\mapsto(u,\eta;\beta,\xi)$.
If the Lagrangian map germ of $\pi_u$
belongs to the stable A$_2$-class
(Mather stability \cite{Golub1973}),
then the germ of $\pi_\beta$ also
belongs to the stable A$_2$-class.
This is a local statement about
stable equivalence of map germs;
it fails if $c_3(c)\to 0$.
\end{remark}

\subsection{The fold amplitude}

\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta,
\end{equation}
$\chi\in C^\infty_c$,
$\alpha_*>0$, $\theta''(\alpha_*)=0$,
$c_3=\frac{1}{6}\theta'''(\alpha_*)\ne 0$,
$\Theta(\beta)=\theta(\alpha_*+\beta)$.

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
$|\Phi(u)|=O(|u|^{-1/2})$
for all $\theta\in C^\infty$.
\end{theorem}

\begin{proof}
Van~der~Corput with weight
$|\beta|^{-1/2}$; Erd\'elyi for
$\theta\equiv 0$.
\end{proof}

\section{CFU Reduction and Symbol Stability}
\label{sec:reduction}

\begin{assumption}[Local stable A$_2$-fold]
\label{ass:fold-local}
$\Theta(\beta)=c_3\beta^3+O(\beta^4)$,
$c_3\ne 0$, and:
\begin{enumerate}[\rm(F1)]
\item $\Theta''(\beta)\ne 0$
  for $\beta\ne 0$ near $0$;
  no cusp or higher stratum.
\item $\int|\beta|^{-1/2}
  e^{i\Theta}\chi\,d\beta
  \not\equiv 0$ near $0$.
\item $\Theta'$ has exactly one
  critical point ($\beta=0$)
  on $\mathrm{supp}\,\chi$;
  no secondary stationary points.
\end{enumerate}
\end{assumption}

\begin{proposition}[Local CFU reduction \Rig]
\label{prop:CFU}
Under Assumption~\ref{ass:fold-local},
fix $U_\epsilon=\{|\beta|<\epsilon\}$
and let $1=\phi_{\mathrm{fold}}
+\phi_{\mathrm{far}}$ be a smooth
partition with
$\phi_{\mathrm{fold}}\equiv 1$
on $U_{\epsilon/2}$.
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold.)}
  $\Theta''(0)=0$,
  $\Theta'''(0)=6c_3\ne 0$;
  \eqref{eq:fold-i}--\eqref{eq:fold-ii}
  and (F1) confirm A$_2$-type.
\item \textit{(Phase equivalence modulo
  canonical transformation.)}
  In $U_\epsilon$,
  the CFU theorem \cite{CFU1957}
  guarantees the existence of a
  smooth local map
  $\beta\mapsto t(\beta,u)$
  --- a \emph{uniform local
  parametrization} of the fold
  (not a global diffeomorphism
  on $\mathrm{supp}\,\chi$) ---
  such that the phase satisfies:
  \begin{equation}\label{eq:CFU-equiv}
    \Theta(\beta)+u\beta
    \;\equiv\;
    \tfrac{1}{3}t(\beta,u)^3
    - \mu(u)\,t(\beta,u)
    + R(u)
    \quad\text{in }U_\epsilon,
  \end{equation}
  where $\mu(u)=c_3^{-1/3}u+O(u^2)$
  and $R(u)$ is smooth.
  This is a \emph{phase equivalence
  modulo smooth canonical transformation};
  the original remainder from Taylor
  expansion is absorbed entirely into
  the Jacobian $J(t,u):=d\beta/dt$
  and the transported amplitude,
  not retained as a separate phase term.
\item \textit{(Far region.)}
  On $\mathrm{supp}\,\phi_{\mathrm{far}}$,
  $\Theta''\ne 0$ by (F1);
  van~der~Corput gives
  $|\Phi_{\mathrm{far}}(u)|
  \le C|u|^{-1/2}$.
\end{enumerate}
\end{proposition}

\begin{lemma}[Jacobian rescaling \Rig]
\label{lem:Jacobian-rescaling}
Under the local canonical change of
variables $\beta\mapsto t(\beta,u)$
of Proposition~\ref{prop:CFU}(2),
\emph{after rescaling to canonical
Airy coordinates}
$t = c_3^{-1/6}|u|^{1/2}\tilde{t}$,
the Jacobi factor satisfies:
\begin{equation}\label{eq:Jacobi-homog}
  J(t,u)
  = c_3^{-1/6}|u|^{-1/2}
  \tilde{J}(\tilde{t},u),
\end{equation}
where $\tilde{J}$ is smooth and
satisfies $\tilde{J}=1+O(|u|^{-1})$
uniformly as $|u|\to\infty$.
The $|u|^{-1/2}$ factor is a consequence
of the homogeneity of the cubic
normal form under Airy rescaling,
not of the original $|\beta|^{-1/2}$
amplitude.
\end{lemma}

\begin{proof}
Under $t=c_3^{-1/6}|u|^{1/2}\tilde{t}$,
the Airy phase
$\frac{1}{3}t^3-\mu(u)t$
becomes
$c_3^{-1/2}|u|^{3/2}
(\frac{1}{3}\tilde{t}^3-\mathrm{sgn}(u)\tilde{t})$,
which is the standard Airy scaling.
The Jacobian transforms as
$d\beta/dt=(d\beta/d\tilde{t})\cdot
(d\tilde{t}/dt)
=c_3^{-1/6}|u|^{1/2}\cdot
(d\beta/d\tilde{t})\cdot|u|^{-1/2}
\cdot c_3^{1/6}$;
the smooth factor
$\tilde{J}=d\beta/d\tilde{t}$
satisfies $\tilde{J}=1+O(|u|^{-1})$
by the CFU expansion \cite{CFU1957}.
\end{proof}

\begin{lemma}[Symbol stability under
  parametric FIO reduction \Rig]
\label{lem:symbol-stability}
Under Assumption~\ref{ass:microlocal},
let $\tilde{a}(\tilde{t},u)
:=|\beta(\tilde{t},u)|^{-1/2}
\tilde{J}(\tilde{t},u)
\cdot\tilde\chi(\tilde{t},u)$
where $\tilde\chi$ is the transported
cutoff (smooth, compactly supported
in $\tilde{t}$ uniformly in $u$).
\begin{enumerate}[\rm(1)]
\item \textit{(Amplitude class in
  $(\tilde{t},u)$.)}
  Under Assumption~\ref{ass:microlocal}(iv),
  $\tilde{a}\in C^\infty$ with
  \[
    |\partial_{\tilde{t}}^j
    \partial_u^k\tilde{a}(\tilde{t},u)|
    \le C_{jk}(1+|u|)^{-1/2-k}
  \]
  uniformly in $(\tilde{t},u,c)$.
\item \textit{(Parametric stationary
  phase.)}
  By H\"ormander \cite{Hormander1983},
  Ch.~7.7, applied with parameter $u$
  and Airy phase
  $\tilde{t}^3/3-\mathrm{sgn}(u)\tilde{t}$:
  \begin{equation}\label{eq:airy-decomp}
    \int_{\mathbb{R}}
    \tilde{a}(\tilde{t},u)\,
    e^{ic_3^{-1/2}|u|^{3/2}
    (\tilde{t}^3/3-\mathrm{sgn}(u)\tilde{t})}
    \,d\tilde{t}
    = a_0(u,c)\,
    \mathrm{Ai}(c_3^{-1/3}u)
    + r_0(u,c),
  \end{equation}
  where $a_0$ is the symbol coefficient
  extracted from the stationary-phase
  expansion at the cubic critical point
  (not by division by $\mathrm{Ai}$),
  and $r_0$ satisfies:
\item \textit{(Symbol and remainder
  estimates.)}
  Under Assumption~\ref{ass:microlocal}(vi)
  (uniform non-degeneracy of the
  non-stationary region):
  \begin{align}
    \label{eq:symbol-u}
    |\partial_u^j a_0(u,c)|
    &\le C_j(1+|u|)^{-1/2-j},\\
    \label{eq:remainder-u}
    |r_0(u,c)|
    &\le C_{N}(1+|u|)^{-N}
    \quad\text{for each fixed }N,
  \end{align}
  uniformly in $c\ge c_0$.
  In particular $a_0\in S^{-1/2}_{1,0}$
  in the $u$-variable.
\end{enumerate}
\end{lemma}

\begin{remark}[On the remainder estimate]
\label{rem:remainder}
The rapid decay~\eqref{eq:remainder-u}
holds for each fixed $N$ uniformly in $c$
\emph{provided} the non-stationary
region (i.e.\ where
$|\partial_{\tilde{t}}(\tilde{t}^3/3
-\mathrm{sgn}(u)\tilde{t})|\ge\delta>0$)
is uniformly non-degenerate in $c$.
This is guaranteed by
Assumption~\ref{ass:microlocal}(vi).
Without (vi), rapid decay
is only claimed for fixed $c$.
\end{remark}

\section{Main Results}
\label{sec:results}

\begin{assumption}[Uniform A$_2$-stability \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$, with:
\begin{enumerate}[\rm(i)]
\item $\Theta=c_3\beta^3+O(\beta^4)$
  in a $c$-independent neighbourhood;
\item $c_3(c)\ge\delta_0>0$;
\item $|\Theta''(\beta^*(u))|
  \ge\delta_1|u|^{1/2}$
  for $\delta_2\le|u|\le c^{\theta_0/3}$;
\item the Jacobi factor satisfies
  $|\partial_u^k J(t,u)|
  \le C_k(1+|u|)^{-1/2-k}$
  uniformly for $k=0,1,2,\ldots$;
\item no secondary stationary points:
  (F3) holds uniformly in $c$;
\item \textit{(Uniform non-degeneracy
  of non-stationary region.)}
  There exists $\delta_4>0$ such that,
  outside the Airy critical region,
  $|\partial_{\tilde{t}}
  (\tilde{t}^3/3-\mathrm{sgn}(u)\tilde{t})|
  \ge\delta_4$ uniformly in $(u,c)$;
  moreover, the canonical transformation
  $\beta\mapsto t(\beta,u)$ satisfies
  $|\det(dt/d\beta)|\ge\delta_5>0$
  and $|\partial_\beta^j\partial_u^k
  t(\beta,u)|\le C_{jk}$
  uniformly in $(\beta,u,c)$
  for all $j,k\ge 0$.
\end{enumerate}
\end{assumption}

\begin{remark}[Uniformity of
  the symbol class]
\label{rem:uniformity}
Condition~(vi) is the key assumption
that promotes the symbol estimate
$a_0\in S^{-1/2}_{1,0}$ from a
statement for fixed $c$ to one
uniform in $c\ge c_0$.
Specifically:
\begin{itemize}
\item the Jacobian bound $|\det(dt/d\beta)|
  \ge\delta_5>0$ ensures the change of
  variables stays in symbol classes
  (not merely $C^\infty$);
\item the derivative bounds on $t(\beta,u)$
  ensure all transported amplitudes
  retain their symbol estimates after
  the CFU change of variables.
\end{itemize}
Verifying (vi) for PSWF is part of
Open Problem~\ref{prob:PSWF-microlocal}.
\end{remark}

\begin{theorem}[Local Airy normal form \Rig]
\label{thm:airy-normalform}
Under Assumption~\ref{ass:microlocal},
microlocally near the fold point
and for $\delta_2\le|u|\le c^{\theta_0/3}$:
\begin{equation}\label{eq:airy-nf}
  \Phi(u)
  = e^{i\alpha_*u}\bigl[
  a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr)
  + r(u,c)\bigr],
\end{equation}
where:
\begin{itemize}
\item $a(u,c)\in S^{-1/2}_{1,0}$
  uniform in $c$
  (Lemma~\ref{lem:symbol-stability}(3)
  under Assumption (vi));
\item $r(u,c)$ satisfies
  $|r(u,c)|\le C_N(1+|u|)^{-N}$
  for each fixed $N$,
  uniform in $c$
  (Remark~\ref{rem:remainder});
\item the oscillatory structure is
  entirely encoded in
  $\mathrm{Ai}(c_3^{-1/3}u)$;
  $a$ is a smooth amplitude modulation.
\end{itemize}
\end{theorem}

\begin{proof}[Proof sketch]
Fold zone $U_\epsilon$:
Proposition~\ref{prop:CFU}(2)
(phase equivalence modulo canonical
transformation, remainder absorbed
into amplitude);
Lemma~\ref{lem:Jacobian-rescaling}
(Airy rescaling of Jacobian);
Lemma~\ref{lem:symbol-stability}
(amplitude class in $(\tilde{t},u)$
$\to$ symbol in $u$ via
H\"ormander Ch.~7.7).
Far zone: Proposition~\ref{prop:CFU}(3).
Uniformity: Assumption~\ref{ass:microlocal}
conditions (iii)--(vi).
\end{proof}

\begin{lemma}[Pointwise decay \Rig]
\label{lem:pointwise}
Under Assumption~\ref{ass:microlocal},
there exists a single constant $C>0$,
uniform across all three zones
and in $c\ge c_0$, such that:
\begin{equation}\label{eq:pointwise}
  |\Phi(u)|\le C(1+|u|)^{-3/4}
  \quad\text{for all }u\in\mathbb{R}.
\end{equation}
\end{lemma}

\begin{proof}
Write $\Phi=\Phi_{\mathrm{fold}}
+\Phi_{\mathrm{far}}$.\medskip

\noindent\textit{Case $u\ge 1$ (fold zone).}
By Theorem~\ref{thm:airy-normalform}:
$|\mathrm{Ai}(c_3^{-1/3}u)|
\le C_1 u^{-1/4}e^{-c'u^{3/2}}$
\cite{Olver1974},
$|a|\le C_2 u^{-1/2}$,
$|r|\le C_N(1+u)^{-N}$:
the product structure in the WKB
representation after reduction gives
$|\Phi|\le(C_1 C_2)u^{-3/4}
+(C_N)(1+u)^{-N}$.\medskip

\noindent\textit{Case $u\le -1$ (fold zone).}
$|\mathrm{Ai}(c_3^{-1/3}u)|
\le C_1|u|^{-1/4}$ \cite{Olver1974}
(oscillatory; no exponential decay);
$|a|\le C_2|u|^{-1/2}$:
product structure gives
$|\Phi|\le C_1 C_2|u|^{-3/4}$.\medskip

\noindent\textit{Case $|u|\le 1$
(compact transition zone).}
The amplitude $\tilde{a}(\tilde{t},u)$
is smooth in $u$;
by \emph{compactness} of $\{|u|\le 1\}$
and \emph{continuity} of the FIO
amplitude (Assumption (vi) provides
uniform $C^\infty$-bounds),
$|\Phi(u)|\le C_3$
uniformly on this region
(Erd\'elyi \cite{Erdelyi1956}).
Hence $|\Phi|\le C_3\le C_3(1+|u|)^{-3/4}$
here.\medskip

\noindent\textit{Far zone
($|u|\ge c^{\theta_0/3}$).}
Theorem~\ref{thm:endpoint}:
$|\Phi_{\mathrm{far}}|
\le C_4|u|^{-1/2}\le C_4|u|^{-3/4}$
for $|u|\ge 1$.\medskip

\noindent\textit{Uniform coupling
of three zones.}
The constants $C_1,C_2,C_3,C_4$
are all uniform in $c\ge c_0$
by Assumption~\ref{ass:microlocal}
(vi) (Jacobian and derivative bounds),
(ii) ($c_3\ge\delta_0$), and
(iii) (non-degeneracy).
In the transition between fold and
far zones
($c^{\theta_0/3}\le|u|
\le 2c^{\theta_0/3}$),
both estimates apply;
condition (iii) ensures their
constants are compatible.
Setting $C:=\max(C_1C_2,C_3,C_4)$
gives~\eqref{eq:pointwise}.
\end{proof}

\begin{remark}[Product structure vs.\ decay]
\label{rem:product}
In Cases $u\ge 1$ and $u\le -1$,
the exponent $-3/4$ arises from the
\emph{product structure in the WKB
representation after CFU reduction}:
$a\in S^{-1/2}_{1,0}$ contributes
$|u|^{-1/2}$,
and $|\mathrm{Ai}(c_3^{-1/3}u)|$
contributes $|u|^{-1/4}$.
These are not independent decay
mechanisms; they arise together
from the single Airy-normal-form
representation~\eqref{eq:airy-nf}.
The two factors are parts of the
same WKB structure, not separate
phenomena that are multiplied post hoc.
\end{remark}

\begin{corollary}[Ces\`aro improvement \Rig]
\label{cor:global-bound}
Under Assumption~\ref{ass:microlocal}:
\[
  \left|\int_0^U\Phi(u)\,du\right|
  \le C\int_0^U(1+u)^{-3/4}\,du
  = O(U^{1/4}),
\]
improving upon Ces\`aro $O(U^{1/2})$.
The single constant $C$ of
Lemma~\ref{lem:pointwise} ensures
the bound is globally stable,
not merely piecewise sharp.
The improvement originates from:
\begin{enumerate}[\rm(a)]
\item \textit{(Geometric.)}
  A$_2$-fold structure forces the Airy
  normal form with $|u|^{-1/4}$ decay;
\item \textit{(Analytic.)}
  $a\in S^{-1/2}_{1,0}$ contributes
  $|u|^{-1/2}$,
  yielding the integrable singularity
  $|u|^{-3/4}$.
\end{enumerate}
\end{corollary}

\subsection{Converse and FIO rigidity}

\begin{assumption}[Microlocal reconstruction
  (non-standard) \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ is in the Airy class and:
\begin{enumerate}[\rm(a)]
\item $\sigma(g\mapsto\Phi)
  \big|_\mathcal{C}\ne 0$;
\item $g\mapsto\Phi$ is injective
  modulo smoothing operators;
\item $\mathrm{WF}(\Phi)
  \subseteq\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
\end{enumerate}
Then $\mathcal{C}$ contains an
A$_2$-fold singularity
uniformly in $c$.
\medskip

\noindent\textit{Status.}
This is a
\emph{non-standard microlocal
reconstruction hypothesis},
not a consequence of standard
FIO theory \cite{Zworski2012,Hormander1983}.
Conditions (a)--(c) go beyond
standard ellipticity.
Verification for PSWF: Paper~XVII.
\end{assumption}

\begin{corollary}[Conditional equivalence]
\label{cor:conditional-equiv}
Under Assumption~\ref{hyp:FIO-rigidity}:
Airy-class bound
$\Rightarrow$
Assumption~\ref{ass:microlocal}
(modulo the non-standard
reconstruction hypothesis).
The forward direction holds rigorously
without Assumption~\ref{hyp:FIO-rigidity}.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$
satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item $c_3(c)\ge\delta_0>0$?
  (Olver turning-point WKB
  \cite{Olver1974}.)
\item $\theta''(\alpha)\ne 0$
  for $\alpha\ne\alpha_*$, uniformly?
\item Conditions (iv) and (vi)
  for the PSWF CFU Jacobian?
\end{enumerate}
Outcomes:
(A)~Assumptions hold:
$|\int_0^U\Phi|=O(U^{1/4})$.
(B)~$c_3(c)\to 0$:
Ces\`aro sharp.
\end{problem}
