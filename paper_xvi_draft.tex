\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases Induce A$_2$-Fold Singularities\\
  Whose Normal Form Is the Airy Distribution}

\author{[Author]}

\begin{abstract}
Paper~XV reduces the intermediate-regime
problem to whether the fold amplitude~$\Phi$
satisfies a uniform oscillatory estimate
improving upon Ces\`aro
(Assumption~XV.\ref{ass:Phi-cubic}).
This paper establishes the microlocal
mechanism governing~$\Phi$.
The key structural insight is:
\emph{polynomial phase representations
are not invariant objects under
Fourier duality; what is invariant
is the Lagrangian submanifold $\Lambda\subset T^*\mathbb{R}$
and its Arnold stratum.}
Concretely, a cubic WKB phase $c_3\beta^3$
does not produce a cubic phase in $u$;
instead it induces a degenerate
stationary point of order~2 in the
Fourier integral, whose universal
normal form (in the sense of Arnold)
is the Airy distribution.
Under a stable embedding in the
$\mathcal{FIO}(A_2)$ class
(Definition~\ref{def:A2-class},
Assumption~\ref{ass:microlocal}),
the fold amplitude satisfies
\[
  \Phi(u)
  = e^{i\alpha_* u}\,a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
  \qquad a(u,c)\in S^{-1/2},
\]
and $|\int_0^U\Phi(u)\,du|=O(U^{1/4})$,
strictly improving upon Ces\`aro.
The converse requires a global FIO
inversion hypothesis
(Hypothesis~\ref{hyp:FIO-rigidity}).
The core open problem is
Problem~\ref{prob:PSWF-microlocal}.
\end{abstract}

\section{Setup and Central Definition}
\label{sec:setup}

\begin{definition}[A$_2$-Lagrangian fold class]
\label{def:A2-class}
Let $I(u)=\int e^{i\Phi(\beta,u)}A(\beta)\,d\beta$
be a one-dimensional Fourier integral operator.
Its \emph{canonical relation} is
\[
  \mathcal{C}
  := \bigl\{(\beta,\xi)\in T^*\mathbb{R}\times T^*\mathbb{R}
  : \xi = \partial_\beta\Phi(\beta,u)\bigr\}.
\]
We say $I$ belongs to the
\emph{A$_2$-Lagrangian fold class},
written $I\in\mathcal{FIO}(A_2)$,
if the projection
$\pi_u:\mathcal{C}\to T^*_u\mathbb{R}$
has a fold singularity at some point
$\beta_0$:
\begin{equation}\label{eq:fold-projection}
  \partial_\beta(\partial_\beta\Phi)(\beta_0,u_0)=0,
  \qquad
  \partial_{\beta\beta}^2(\partial_\beta\Phi)(\beta_0,u_0)\ne 0.
\end{equation}
This is an A$_2$-catastrophe
in Arnold's classification \cite{Arnold1972}.
The universal normal form of this
singularity class is the Airy function
$\mathrm{Ai}$.
\medskip

\noindent The Fourier transform
transports Arnold strata:
\begin{equation}\label{eq:strata-transport}
  \widehat{\phantom{x}}
  \;:\;
  \mathcal{FIO}(A_k)
  \longrightarrow
  \mathcal{FIO}(A_k),
  \quad k=0,1,2,
\end{equation}
\emph{under stable embeddings
in the respective stratum},
i.e.\ when the phase function
remains in the same Arnold
equivalence class uniformly.
In particular:
$A_0$ (smooth/non-oscillatory) $\to$ $A_0$;
$A_1$ (non-degenerate stationary point) $\to$ $A_1$;
$A_2$ (fold/cubic degeneration) $\to$ $A_2$.
\emph{This is not a universal statement}:
it holds only when the canonical
relation stays uniformly in the given stratum
(cf.\ Assumption~\ref{ass:microlocal}(ii)).
\end{definition}

\begin{remark}[Polynomial phases are
  not invariant objects]
\label{rem:phases-not-invariant}
The phase representation $e^{i\phi(\beta)}$
is coordinate-dependent:
it changes under canonical transformations
and is not stable under Legendre duality.
What is canonically defined is the
Lagrangian submanifold $\Lambda
=\{(\beta,\partial_\beta\phi(\beta))\}
\subset T^*\mathbb{R}$
and its Arnold stratum.
In particular: the cubic class
$\{e^{ic_3\beta^3}\}$ and the
Airy class $\{\mathrm{Ai}(c_3^{-1/3}u)\}$
are \emph{representations} of the
same A$_2$ Lagrangian geometry
in two different coordinate systems
($\beta$-space and $u$-space);
they are not related by preservation
of a phase formula, but by
the coordinate change inherent in
the Fourier transform.
\end{remark}

\subsection{The fold amplitude from Paper~XV}

Recall from Paper~XV,
Proposition~\ref{prop:I0-1D}:
\begin{equation}\label{eq:Phi-def}
  \Phi(u)
  :=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \quad u\in\mathbb{R},
\end{equation}
where $\chi\in C^\infty_c$ with
$\chi\equiv 1$ near $0$,
and $\alpha_*>0$ is the fold point.
The remaining question from Paper~XV is:
\begin{equation}\label{eq:Phi-question}
  \left|\int_0^U\Phi(u)\,du\right|
  = o(U^{1/2})
  \quad(U\to\infty).
\end{equation}

\subsection{The WKB amplitude}

The fold amplitude arises as the
Fourier transform of
\begin{equation}\label{eq:g-def}
  g(\alpha)
  := |\alpha-\alpha_*|^{-1/2}
  e^{i\theta(\alpha)}\chi(\alpha),
\end{equation}
where $\theta$ is the WKB phase
from Paper~XV, Lemma~\ref{lem:WKB-phase},
satisfying the fold condition:
\begin{equation}\label{eq:fold}
  \theta''(\alpha_*)=0,
  \quad
  \theta'''(\alpha_*)\ne 0,
  \quad
  c_3:=\tfrac{1}{6}\theta'''(\alpha_*)\ne 0.
\end{equation}
Setting $\beta=\alpha-\alpha_*$
and $\Theta(\beta):=\theta(\alpha_*+\beta)$:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i\Theta(\beta)}
  \chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta.
\end{equation}

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
For $\Phi$ defined by~\eqref{eq:Phi-def}
($\theta\equiv 0$):
$\Phi(u)=\sqrt{\pi}\,e^{i\alpha_*u}
|u|^{-1/2}e^{-i\pi/4}+O(|u|^{-3/2})$.
For $\Phi$ defined by~\eqref{eq:Phi-WKB}
with $\theta\in C^\infty$ arbitrary,
still $|\Phi(u)|=O(|u|^{-1/2})$.
\end{theorem}

\begin{proof}
Pure endpoint: Erd\'elyi's lemma at $\beta=0$.
WKB case: van~der~Corput
with singular weight $|\beta|^{-1/2}$.
\end{proof}

\section{Cubic Phase Induces
  A$_2$-Fold Singularity}
\label{sec:nonpreservation}

\begin{proposition}[Cubic WKB phase
  $\Rightarrow$ degenerate stationary
  point of order~2 $\Rightarrow$
  Airy normal form \Rig]
\label{prop:nonpreservation}
Let $\Theta(\beta)=c_3\beta^3+O(\beta^4)$
with $c_3\ne 0$.
Consider
\begin{equation}\label{eq:F-def}
  F(u):=\int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\beta)\,d\beta.
\end{equation}
\begin{enumerate}[\rm(1)]
\item The phase $\Phi_u(\beta):=\Theta(\beta)+u\beta$
  has a \emph{degenerate stationary point
  of order~2} at $\beta=0$ when $u=0$:
  \[
    \Phi_u'(\beta)\big|_{\beta=0,u=0}=0,
    \quad
    \Phi_u''(\beta)\big|_{\beta=0,u=0}=0,
    \quad
    \Phi_u'''(\beta)\big|_{\beta=0}=6c_3\ne 0.
  \]
  This is precisely an A$_2$-fold singularity
  in the sense of Definition~\ref{def:A2-class}:
  the projection $\pi_u:\mathcal{C}\to T^*_u$
  satisfies~\eqref{eq:fold-projection}
  at $(\beta_0,u_0)=(0,0)$.
\item By the CFU normal form theorem
  \cite{CFU1957},
  the universal asymptotic form of $F(u)$
  is determined by this A$_2$ singularity:
  \begin{equation}\label{eq:airy-FT}
    F(u)\;
    \sim\;
    c_3^{-1/6}\,P(u)\,
    \mathrm{Ai}\!\bigl(-c_3^{-1/3}u\bigr),
    \quad|u|\to\infty,
  \end{equation}
  where $P$ is a polynomial amplitude prefactor.
\item The oscillatory phase of $\mathrm{Ai}(-c_3^{-1/3}u)$
  for large $|u|$ is, symbolically:
  \begin{equation}\label{eq:symbolic-phase}
    S(u)
    \sim -\tfrac{2}{3}|c_3|^{-1/2}|u|^{3/2}.
  \end{equation}
  This is not a Legendre identity but a
  \emph{symbolic representation}:
  $S(u)$ is the value of the phase at
  the (now coalescing) critical points,
  and equals the generating function
  of the canonical relation~$\mathcal{C}$
  in the dual parametrisation.
\end{enumerate}
Conclusion: the Fourier transform of
an amplitude with cubic WKB phase maps
not to a cubic polynomial phase in $u$,
but to an Airy distribution,
via the chain
\[
  \text{cubic phase in }\beta
  \;\Rightarrow\;
  \text{degenerate stat. point of order~2}
  \;\Rightarrow\;
  \text{A}_2\text{-fold singularity}
  \;\Rightarrow\;
  \text{Airy normal form.}
\]
\end{proposition}

\begin{proof}
Statement~(1): direct computation.
Statement~(2): Chester--Friedman--Ursell
\cite{CFU1957}, Olver \cite{Olver1974} Ch.~11.
Statement~(3): standard large-argument
expansion of the Airy function
$\mathrm{Ai}(-t)\sim\pi^{-1/2}t^{-1/4}
\sin(\frac{2}{3}t^{3/2}+\frac{\pi}{4})$
for $t\to+\infty$ \cite{Olver1974}.
The phase $-\frac{2}{3}c_3^{-1/2}|u|^{3/2}$
is a symbolic expression
for this oscillatory regime;
it is not asserted as a closed-form
phase function defined for all $u$.
\end{proof}

\begin{remark}[No polynomial phase hierarchy]
\label{rem:no-hierarchy}
Polynomial phases $\{e^{ic_k\beta^k}\}_{k\ge 2}$
are not invariant under Fourier duality:
they are coordinate representations
of Lagrangian submanifolds,
not canonical objects.
What the Fourier transform preserves
is the Arnold stratum of the
Lagrangian submanifold $\Lambda$
(under the stable embedding condition).
In particular: $\mathcal{FIO}(A_2)\to\mathcal{FIO}(A_2)$
under the qualification of
Definition~\ref{def:A2-class};
this is not a statement about any
``cubic stability class''.
\end{remark}

\begin{remark}[Consequence for
  Assumption~XV.\ref{ass:Phi-cubic}]
\label{rem:consequence}
Assumption~XV.\ref{ass:Phi-cubic}
postulates a cubic polynomial phase $e^{ic_3u^3}$.
Proposition~\ref{prop:nonpreservation}
shows this cannot arise from the WKB structure:
the correct output is Airy-class,
not cubic polynomial.
Assumption~XV.\ref{ass:Phi-cubic}
should be read as a proxy for
the Airy-class bound on $\Phi$,
not as a literal phase formula.
\end{remark}

\section{Airy Normal Form Theorem}
\label{sec:normalform}

\begin{assumption}[Uniform A$_2$-stability \Cond]
\label{ass:microlocal}
The WKB amplitude $g$ of~\eqref{eq:g-def}
satisfies, near $\alpha_*$, a uniform
microlocal decomposition
$g(\alpha_*+\beta)=A(\beta)\,e^{i\Theta(\beta)}$
where:
\begin{enumerate}[\rm(i)]
\item $A(\beta)=|\beta|^{-1/2}\tilde\chi(\beta)$,
  $\tilde\chi\in C^\infty_c$;
\item $\Theta(\beta)=c_3\beta^3+O(\beta^4)$
  uniformly, with
  $c_3=c_3(c)\ge\delta_0>0$
  uniformly in $c\ge c_0$
  \emph{(A$_2$-stability:
  $\mathcal{C}_c$ stays in the fold stratum)};
\item $|\widehat{A}(u)|=O(|u|^{-1/2})$;
\item $|\Theta''(\beta^*(u))|
  =6c_3|\beta^*(u)|
  \ge\delta_1|u|^{1/2}$
  uniformly for
  $\delta_2\le|u|\le c^{\theta_0/3}$
  \emph{(uniform non-degeneracy of
  the fold projection $\pi_u$)}.
\end{enumerate}
\end{assumption}

\begin{theorem}[Airy normal form \Rig]
\label{thm:airy-normalform}
Under Assumption~\ref{ass:microlocal},
the fold amplitude~\eqref{eq:Phi-WKB}
satisfies:
\begin{equation}\label{eq:airy-normalform}
  \Phi(u)
  = e^{i\alpha_* u}\,
  a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
\end{equation}
where $a(u,c)\in S^{-1/2}$,
i.e.\ $|a(u,c)|\le C(1+|u|)^{-1/2}$
and $|\partial_u a|\le C(1+|u|)^{-3/2}$,
uniformly in $c\ge c_0$
and $\delta_2\le|u|\le c^{\theta_0/3}$.
The factor $e^{i\alpha_*u}$ is a
global Fourier shift (translation
of $\alpha_*$ to the origin);
it carries no microlocal information
about the fold geometry.
The Airy factor $\mathrm{Ai}(c_3^{-1/3}u)$
is the A$_2$ normal form
of Definition~\ref{def:A2-class};
its oscillatory phase for large $|u|$
is described symbolically by
$S(u)\sim -\frac{2}{3}c_3^{-1/2}|u|^{3/2}$
(Proposition~\ref{prop:nonpreservation}(3)).
\end{theorem}

\begin{proof}[Proof sketch]
Apply CFU \cite{CFU1957} to~\eqref{eq:Phi-WKB}.
Condition~(iv) ensures the coordinate
change $\beta\mapsto t(\beta,u)$ satisfying
$\Theta(\beta)+u\beta=\frac{1}{3}t^3-c_3^{-1/3}ut$
is a smooth diffeomorphism uniformly
in $(u,c)$.
Condition~(i) gives
$|\beta|^{-1/2}\mapsto|t|^{-1/2}$
times a smooth Jacobian;
condition~(ii) controls the remainder.
The result is the Airy representation~\eqref{eq:airy-normalform};
symbol bounds on $a$ follow
from Erd\'elyi's lemma.
\end{proof}

\begin{lemma}[Airy class $\Rightarrow$
  improved Ces\`aro \Rig]
\label{lem:airy-integral}
Under the conclusion of
Theorem~\ref{thm:airy-normalform}:
\[
  \left|\int_0^U\Phi(u)\,du\right|
  = O(U^{1/4}),
  \quad U\to\infty.
\]
\end{lemma}

\begin{proof}
Three mechanisms combine:
\begin{enumerate}[\rm(a)]
\item \emph{Symbol decay}:
  $a\in S^{-1/2}$ gives
  $|a(u,c)|=O(u^{-1/2})$.
\item \emph{Airy decay}:
  $|\mathrm{Ai}(t)|\le C|t|^{-1/4}$
  for $t\to+\infty$ \cite{Olver1974};
  for $t<0$, Airy oscillates
  without additional growth.
\item \emph{Global phase shift}:
  $e^{i\alpha_*u}$ contributes only
  boundary terms $O(U^{-3/4})$
  via integration by parts.
\end{enumerate}
Combining (a) and (b):
$\int_1^U|a(u,c)|\,|\mathrm{Ai}(c_3^{-1/3}u)|\,du
\le C\int_1^U u^{-3/4}\,du=O(U^{1/4})$.
All three mechanisms are necessary;
none suffices alone.
In particular, this is a
\emph{global-in-$u$} estimate:
it requires uniform symbol control
over the full range $[1,U]$,
not just local asymptotics.
\end{proof}

\subsection{Converse and FIO rigidity}
\label{subsec:converse}

\begin{hypothesis}[FIO inversion
  rigidity \Cond]
\label{hyp:FIO-rigidity}
If $\Phi$ lies in the Airy FIO class
(i.e.\ $a(u,c)\in S^{-1/2}$
and the oscillatory phase of $\Phi$
is $S(u)\sim C|u|^{3/2}$ uniformly
in $c$), then the wavefront set
and canonical relation of $g$
contain an A$_2$-fold singularity
uniformly in $c$.
\end{hypothesis}

\begin{remark}[Status of
  Hypothesis~\ref{hyp:FIO-rigidity}]
\label{rem:rigidity-status}
This is a global FIO injectivity
statement: the inverse Fourier
transform recovers $g$ only
up to microlocal equivalence.
It is possible in principle for
$\Phi$ to have Airy-type oscillations
for $c$-dependent analytic reasons
unrelated to the fold structure of $g$
(e.g.\ amplitude interference without
global ellipticity or without absence
of cancellations between amplitude modes).
Hypothesis~\ref{hyp:FIO-rigidity}
rules this out by assuming
global ellipticity and
non-degeneracy of the phase class.
Verification for the PSWF case
is the content of Paper~XVII.
\end{remark}

\begin{corollary}[Conditional equivalence \Cond]
\label{cor:conditional-equiv}
Under Hypothesis~\ref{hyp:FIO-rigidity}:
$\text{Airy-class bound on }\Phi
\iff
\text{Assumption~\ref{ass:microlocal}}$.
Without it, only
$\text{Assumption~\ref{ass:microlocal}}
\Rightarrow
\text{Airy-class bound}$
stands (Theorem~\ref{thm:airy-normalform},
Lemma~\ref{lem:airy-integral}).
\end{corollary}

\section{The Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability for PSWF \Open]
\label{prob:PSWF-microlocal}
Does $g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$
satisfy Assumption~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item \emph{(A$_2$-stability)}
  Does $c_3(c)\ge\delta_0>0$ uniformly?
  Approach: Olver turning-point WKB
  \cite{Olver1974} applied to the PSWF ODE
  gives $\theta'''(\alpha_*;c)$ in terms
  of the curvature of the classical
  trajectory at $\alpha_*(c)$.
\item \emph{(Uniform fold projection)}
  Does condition~(iv) hold down to
  the Airy scale $|u|\sim c^{-\theta_0/3}$?
\end{enumerate}
Two outcomes:
\begin{enumerate}[\rm(A)]
\item Assumption~\ref{ass:microlocal} holds:
  $|\int_0^U\Phi\,du|=O(U^{1/4})$,
  $S(c)=O(c^{\theta_0/2})$ (up to logs).
\item $c_3(c)\to 0$:
  $\mathcal{C}_c$ exits A$_2$ stratum;
  no stable normal form;
  Ces\`aro sharp.
\end{enumerate}
\end{problem}

\begin{remark}[Relation to
  fold\_model.tex]
\label{rem:fold-model}
\texttt{fold\_model.tex} analyses
$\Phi_\alpha(u)
:=\int_0^\infty\beta^{-1/2}
\chi(\beta/|u|^\alpha)
e^{i(-|u|\beta+c_3\beta^3)}d\beta$.
This is the integrand
\emph{after} the CFU coordinate change
in Theorem~\ref{thm:airy-normalform},
expressed in Airy normal form coordinates.
It is a model for the output
of the CFU reduction, not for $\Phi$ itself.
\end{remark}
