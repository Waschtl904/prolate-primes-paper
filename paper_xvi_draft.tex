\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases, A$_2$-Fold Singularities,\\
  and the Airy Normal Form}

\author{[Author]}

\begin{abstract}
This paper works in the \emph{one-dimensional setting} throughout
($\beta,u\in\mathbb{R}$).
A cubic WKB phase $c_3(c)\beta^3$
produces a degenerate stationary point
of order~2, which is an A$_2$-fold
singularity of the canonical relation
$\mathcal{C}$ in the sense of
Arnol'd's classification
\cite{Arnold1972}.
The Airy distribution is the universal
local normal form of this singularity,
in the sense of microlocal equivalence
modulo smoothing operators,
via the CFU reduction
\cite{CFU1957} treated as
a Fourier integral operator construction
with canonical relation
$\mathcal{C}\subset
T^*\mathbb{R}_\beta\times T^*\mathbb{R}_u$
equivalent to the Airy Lagrangian
(Remark~\ref{rem:FIO-language}).
Three results are established,
each explicitly conditional on
Assumption~\ref{ass:microlocal}
(uniform A$_2$-stability,
conditions (i)--(vi)):
\begin{enumerate}[\rm(I)]
\item \textit{(Local Airy normal form.)}
  Theorem~\ref{thm:airy-normalform}:
  $\Phi$ is microlocally equivalent,
  modulo smoothing operators,
  to an Airy model integral
  with $a\in S^{-1/2}_{1,0}(c)$
  (parameter-dependent symbol family
  in $u$, uniform in $c\ge c_0$;
  $c$ is an external parameter,
  not a phase-space variable;
  see Definition~\ref{def:symbol-class})
  and rapidly decaying remainder
  (conditional on (i)--(vi)).
\item \textit{(Pointwise decay.)}
  Corollary~\ref{cor:pointwise}:
  $|\Phi(u;c)|\le C(1+|u|)^{-3/4}$
  uniformly (conditional on (i)--(vi)).
\item \textit{(Ces\`aro improvement.)}
  Corollary~\ref{cor:global-bound}:
  integrability improvement
  $|\int_0^U\Phi|=O(U^{1/4})$,
  conditional on (i)--(vi).
\end{enumerate}
The forward direction is rigorous
under (i)--(vi);
the converse requires the non-standard
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\maketitle

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
We work throughout in one spatial
dimension ($\beta,u\in\mathbb{R}$).
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold,
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the projection to the right factor.
\medskip

\noindent\textit{Geometric definition.}
$\mathcal{C}$ has an
\emph{A$_2$-fold singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$;
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$.
\end{enumerate}
These are the conditions for a
\emph{stable Lagrange singularity of
A$_2$-type} in Arnol'd's
classification \cite{Arnold1972}
(rank-1 degeneration of the
Lagrangian projection,
with non-degenerate unfolding).
\medskip

\noindent\textit{Coordinate reduction.}
For the phase
$\Psi(\beta,u)=\Theta(\beta;c)+u\beta$,
conditions (i)--(ii) reduce to
$\Theta''(0;c)=0$,
$\Theta'''(0;c)\ne 0$.
These are coordinate expressions;
conditions (i)--(ii) are invariant
under canonical transformations.
The associated canonical relation is
locally given by
\[
  \mathcal{C} = \bigl\{(\beta,\,\partial_\beta\Psi;\,
  u,\,-\partial_u\Psi)\bigr\}
  = \bigl\{(\beta,\,\Theta'(\beta;c)+u;\,
  u,\,-\beta)\bigr\}.
\]
At the fold point $\beta=0$,
$\partial_\beta(\Theta'(\beta;c)+u)=\Theta''(0;c)=0$,
confirming the degeneration of $\pi_u$.
\medskip

\noindent\textit{Normal form.}
By Arnol'd's classification
\cite{Arnold1972} of stable
Lagrange singularities,
any oscillatory integral with an
A$_2$-fold is
\emph{microlocally equivalent,
modulo smoothing operators,}
to the Airy integral
$\mathrm{Ai}$
near $p_0$.
This is a local-in-phase-space
statement about the germ of the
oscillatory integral; it does not
give a global or pointwise identity.
The Golubitsky--Guillemin framework
\cite{Golub1973} provides the
coordinate-free stability conditions
for such map germs.
\end{definition}

\begin{remark}[FIO equivalence language]
\label{rem:FIO-language}
Throughout this paper, all statements
of the form
``reduced to Airy normal form''
or ``microlocally equivalent to Ai''
mean precisely:
\emph{the oscillatory integral for
$\Phi$ is microlocally equivalent,
modulo smoothing operators,
to an Airy model integral,
in the sense of Fourier integral
operators with canonical relation
equivalent to the Airy Lagrangian,
via a local canonical transformation
$\beta\mapsto t(\beta,u;c)$.}
This is \emph{never} a pointwise
phase identity
$\Theta(\beta;c)+u\beta
=\frac{1}{3}t^3-\mu t+R$,
nor an exact equality of integrals
without remainder control.
Arnol'd's stable A$_2$-classification
\cite{Arnold1972} guarantees this
microlocal equivalence locally;
all Taylor remainders from the
change of variables are absorbed
into the amplitude.
The Maslov phase factor is absorbed
into $R(u;c)$ in~\eqref{eq:airy-nf};
its computation from the
Maslov index of $\mathcal{C}$
follows H\"ormander
\cite{Hormander1983}, Ch.~25.1.
\end{remark}

\begin{remark}[Microlocal transport]
\label{rem:microlocal-transport}
The Fourier transform acts by the
exchange
$(\beta,\xi;u,\eta)\mapsto(u,\eta;\beta,\xi)$.
If the Lagrangian map germ of $\pi_u$
belongs to the stable A$_2$-class
(Arnol'd--Mather stability
\cite{Arnold1972,Golub1973}),
then the germ of $\pi_\beta$ also
belongs to the stable A$_2$-class.
This is a local statement;
it fails if $c_3(c)\to 0$.
\end{remark}

\subsection{The fold amplitude and
  conormal singularity}

The fold amplitude is:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u;c)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta;c)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta,
\end{equation}
where $\chi\in C^\infty_c$,
$\alpha_*>0$,
$\theta''(\alpha_*;c)=0$,
$c_3(c):=\frac{1}{6}\theta'''(\alpha_*;c)\ne 0$,
$\Theta(\beta;c):=\theta(\alpha_*+\beta;c)$.
\medskip

\noindent\textbf{Conormal singularity
and non-standard stationary phase.}
The weight $|\beta|^{-1/2}$ is not
locally integrable at $\beta=0$;
the integral~\eqref{eq:Phi-WKB} is
interpreted as a distributional
oscillatory integral with
\[
  |\beta|^{-1/2}\chi
  \in I^{-1/4}(\mathbb{R},\{0\}),
\]
in the sense of H\"ormander
\cite{Hormander1983}, Ch.~18.2,
and Guillemin--Uhlmann
\cite{GuillemU1981}.
\medskip

\noindent\textit{Critical remark.}
The amplitude singularity occurs
\emph{exactly at the critical point}
$\beta=0$ of the phase.
This is \emph{not} covered by
standard stationary-phase theory
(which assumes smooth or symbol-class
amplitudes away from the critical set).
The correct framework is
\emph{non-standard stationary phase
for oscillatory integrals with
conormal amplitude singularity
at the critical point},
as developed in
Melrose--Uhlmann \cite{MelroseU1979}
and H\"ormander \cite{Hormander1983},
Ch.~18.1--18.2.
All stationary-phase and FIO
operations in this paper are
performed within this framework.

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
Under Assumption~\ref{ass:fold-local},
$|\Phi(u;c)|=O(|u|^{-1/2})$
uniformly in $c\ge c_0$.
\end{theorem}

\begin{proof}
The amplitude
$a_0=|\beta|^{-1/2}\chi
\in I^{-1/4}(\mathbb{R},\{0\})$;
all operations are distributional
(Melrose--Uhlmann \cite{MelroseU1979};
H\"ormander \cite{Hormander1983},
Ch.~18.1--18.2).
For $\theta\equiv 0$,
Erd\'elyi \cite{Erdelyi1956}
gives exact asymptotics.
For general $\theta$,
van~der~Corput in the
non-standard distributional setting
(H\"ormander \cite{Hormander1983},
Ch.~7.7 + 18.1)
gives $O(|u|^{-1/2})$,
provided $\Theta''(\beta^*;c)
\ge\delta_1|u|^{1/2}$
(Assumption~\ref{ass:microlocal}(iii)),
which ensures uniform non-degeneracy
for the van~der~Corput bound
in $c$.
\end{proof}

\section{CFU Reduction, Conormal Stability,
  and Symbol Classes}
\label{sec:reduction}

\begin{assumption}[Local stable A$_2$-fold]
\label{ass:fold-local}
$\Theta(\beta;c)=c_3(c)\beta^3+O(\beta^4)$,
$c_3(c)\ne 0$, and:
\begin{enumerate}[\rm(F1)]
\item $\Theta''(\beta;c)\ne 0$
  for $\beta\ne 0$ near $0$
  (stable A$_2$: no cusp, no
  higher-order degeneracy).
\item The oscillatory integral
  $\int|\beta|^{-1/2}
  e^{i\Theta(\cdot;c)}\chi\,d\beta
  \not\equiv 0$ near $0$.
\item $\partial_\beta\Theta(\beta;c)$
  has exactly one critical point
  ($\beta=0$) on $\mathrm{supp}\,\chi$;
  no secondary stationary points.
\end{enumerate}
\end{assumption}

\begin{lemma}[Conormal amplitude
  stability under CFU \Rig]
\label{lem:conormal-stability}
Let $a_0=|\beta|^{-1/2}\chi
\in I^{-1/4}(\mathbb{R},\{0\})$.
Under the CFU change of variables
$\beta\mapsto t(\beta,u;c)$
of Proposition~\ref{prop:CFU},
the transported amplitude
\[
  \tilde{a}(t,u;c)
  := a_0(\beta(t,u;c))\cdot J(t,u;c)
  \cdot\tilde\chi(t,u;c)
\]
satisfies
$\tilde{a}(\cdot,u;c)
\in I^{-1/4}(\mathbb{R},\{0\})$
in $t$, for each fixed $(u,c)$.
Under condition (vi) of
Assumption~\ref{ass:microlocal},
this holds uniformly in $(u,c)$.
\end{lemma}

\begin{proof}
The map $t\mapsto\beta(t,u;c)$
is a local diffeomorphism
with $|\det(dt/d\beta)|\ge\delta_5>0$
and all derivatives bounded
(condition (vi)).
Pushforward of conormal
distributions under local
diffeomorphisms preserves
the $I^m$ class
(H\"ormander \cite{Hormander1983},
Ch.~18.2;
Guillemin--Uhlmann \cite{GuillemU1981},
Lemma~2.1);
uniformity in $(u,c)$
follows from condition (vi):
the uniform lower bound $\delta_5$
and the uniform derivative bounds
$|\partial_\beta^j\partial_u^k t|\le C_{jk}$
ensure that the pushforward construction
(which depends on derivatives of the
diffeomorphism) produces
$I^{-1/4}$-bounds that are
\emph{independent of $(u,c)$}.
The Jacobian $J\in C^\infty$
with bounded derivatives,
so multiplication preserves $I^m$.
\end{proof}

\begin{proposition}[CFU: microlocal
  equivalence modulo smoothing \Rig]
\label{prop:CFU}
Under Assumption~\ref{ass:fold-local}
(see Remark~\ref{rem:FIO-language}).
Fix $U_\epsilon=\{|\beta|<\epsilon\}$
and a smooth partition
$1=\phi_{\mathrm{fold}}
+\phi_{\mathrm{far}}$.
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold.)}
  $\Theta''(0;c)=0$,
  $\Theta'''(0;c)=6c_3(c)\ne 0$
  (stable A$_2$ by (F1)).
\item \textit{(Microlocal equivalence
  modulo smoothing.)}
  The CFU theorem \cite{CFU1957}
  provides a smooth local map
  $\beta\mapsto t(\beta,u;c)$
  and smooth functions
  $\mu(u;c)=c_3(c)^{-1/3}u+O(u^2)$,
  $R(u;c)$ such that:
  \begin{align}\label{eq:CFU-integral}
    &\int_{U_\epsilon}
    |\beta|^{-1/2}
    e^{i(\Theta(\beta;c)+u\beta)}
    \phi_{\mathrm{fold}}\,d\beta
    \\
    &\quad=
    e^{iR(u;c)}
    \int_{\mathbb{R}}
    \tilde{a}(t,u;c)\,
    e^{i(\frac{1}{3}t^3-\mu(u;c)\,t)}
    \,dt
    + \mathcal{R}(u;c),\notag
  \end{align}
  where $\tilde{a}\in I^{-1/4}
  (\mathbb{R},\{0\})$
  (Lemma~\ref{lem:conormal-stability}).

  \noindent\textit{Smoothing remainder.}
  There exists a family of operators
  $R_c\in\Psi^{-\infty}(\mathbb{R})$
  such that
  $\mathcal{R}(u;c)=(R_c\,1)(u)$,
  and for every $N\in\mathbb{N}$
  and $s\in\mathbb{R}$,
  \[
    R_c: H^s(\mathbb{R})\to H^{s+N}(\mathbb{R})
  \]
  with operator norms bounded
  uniformly in $c\ge c_0$.
  Equivalently, for each fixed $N$,
  \[
    |\mathcal{R}(u;c)|
    \le C_N(1+|u|)^{-N}
    \quad\text{uniformly in }c\ge c_0.
  \]
  \textit{Dependence on (vi):}
  both the $\Psi^{-\infty}$-membership
  and the uniform operator-norm bounds
  require condition~(vi) of
  Assumption~\ref{ass:microlocal};
  without~(vi), $R_c\in\Psi^{-\infty}$
  holds for each fixed $c$ only.
  The remainder arises from the
  far tail of the $\tilde{t}$-integral
  (cut off outside the Airy critical
  region) and from the CFU Taylor
  residual; both are controlled
  uniformly by the $\delta_4$-
  and $C_{jk}$-bounds in (vi).

  This is a
  \emph{microlocal equivalence
  modulo smoothing operators},
  not a pointwise phase identity
  (Remark~\ref{rem:FIO-language}).
\item \textit{(Far region.)}
  $|\Phi_{\mathrm{far}}(u;c)|
  \le C|u|^{-1/2}$ by van~der~Corput.
\end{enumerate}
\end{proposition}

\begin{lemma}[Jacobian rescaling \Rig]
\label{lem:Jacobian-rescaling}
Under the canonical Airy rescaling
$t = c_3(c)^{-1/6}|u|^{1/2}\tilde{t}$:
\begin{equation}\label{eq:Jacobi-homog}
  J(t,u;c)
  = c_3(c)^{-1/6}|u|^{-1/2}
  \tilde{J}(\tilde{t},u;c),
  \quad
  \tilde{J} = 1+O(|u|^{-1}),
\end{equation}
uniformly in $c$ under condition (vi).
The $|u|^{-1/2}$ factor arises from
the homogeneity of the cubic normal
form, not from $|\beta|^{-1/2}$.
\end{lemma}

\begin{proof}
Cubic Airy phase becomes
$c_3(c)^{-1/2}|u|^{3/2}
(\frac{1}{3}\tilde{t}^3
-\mathrm{sgn}(u)\tilde{t})$;
$J = \tilde{J}\cdot c_3(c)^{1/6}
|u|^{-1/2}$
from $J\,dt=d\beta$;
$\tilde{J}=1+O(|u|^{-1})$
by CFU \cite{CFU1957}.
\end{proof}

\begin{definition}[Parameter-dependent
  symbol class]
\label{def:symbol-class}
\emph{Throughout, $c\ge c_0$ is an
external parameter, not a phase-space
variable.}
We define
\begin{equation}\label{eq:symbol-class-c-refined}
  S^m_{1,0}(c)
  :=
  \Bigl\{ a(u;c)\in C^\infty(\mathbb{R}_u)
  \;\Big|\;
  |\partial_u^\alpha a(u;c)|
  \le C_\alpha (1+|u|)^{m-|\alpha|}
  \text{ for all }\alpha,
  \text{ uniformly in } c\ge c_0
  \Bigr\}.
\end{equation}

\noindent\textit{Interpretation.}
This is a \emph{parameter-dependent
symbol class in the variable $u$},
uniform in $c$, and \emph{not} a
H\"ormander symbol class on an
extended phase space.
In particular:
\begin{itemize}[nosep]
\item no derivatives in $c$
  are included;
\item $c$ does not play the role
  of a dual or phase variable;
\item the class does \emph{not}
  correspond to
  $S^m_{1,0}(\mathbb{R}_{u,c})$.
\end{itemize}
All symbol estimates are to be
understood as uniform families of
classical $S^m_{1,0}(\mathbb{R}_u)$
symbols parametrised by $c$.
\end{definition}

\begin{lemma}[Parameter-dependent
  symbol stability \Rig]
\label{lem:symbol-stability}
Under Assumption~\ref{ass:microlocal},
let
$\tilde{a}(\tilde{t},u;c)
:=|\beta(\tilde{t},u;c)|^{-1/2}
\tilde{J}(\tilde{t},u;c)
\cdot\tilde\chi(\tilde{t},u;c)$.
\begin{enumerate}[\rm(1)]
\item \textit{(Amplitude class.)}
  Under condition (vi),
  $|\partial_{\tilde{t}}^j
  \partial_u^k\tilde{a}|
  \le C_{jk}(1+|u|)^{-1/2-k}$
  uniformly in $(\tilde{t},u,c)$;
  in particular
  $\tilde{a}(\tilde{t},\cdot;c)
  \in S^{-1/2}_{1,0}(c)$
  (Definition~\ref{def:symbol-class})
  uniformly in $\tilde{t}$.
\item \textit{(Parametric stationary
  phase.)}
  By H\"ormander \cite{Hormander1983},
  Ch.~7.7,
  applied with uniform parameter
  bounds from condition (vi)
  (explicit uniform non-stationary
  phase bound $\delta_4$ from (vi)):
  \begin{equation}\label{eq:airy-decomp}
    \int\tilde{a}(\tilde{t},u;c)\,
    e^{ic_3(c)^{-1/2}|u|^{3/2}
    (\tilde{t}^3/3
    -\mathrm{sgn}(u)\tilde{t})}
    d\tilde{t}
    = a_0(u;c)\,
    \mathrm{Ai}(c_3(c)^{-1/3}u)
    + r_0(u;c).
  \end{equation}
  Here $a_0$ is extracted from
  the stationary-phase expansion
  at the cubic critical point;
  it is \emph{not} defined by
  dividing by $\mathrm{Ai}$.
\item \textit{(Symbol and remainder.)}
  Under condition (vi):
  \begin{align}
    \label{eq:symbol-u}
    a_0(\cdot;c)
    &\in S^{-1/2}_{1,0}(c)
    \quad\text{(in }u\text{,
    uniform in }c\text{)},\\
    \label{eq:remainder-u}
    |r_0(u;c)|
    &\le C_{N}(1+|u|)^{-N}
    \quad(\text{each fixed }N,
    \text{ uniform in }c\ge c_0).
  \end{align}
  Without condition (vi),
  these hold for each fixed $c$
  only.
\end{enumerate}
\end{lemma}

\section{Main Results}
\label{sec:results}

\begin{assumption}[Uniform A$_2$-stability
  \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$, with:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta;c)
  =c_3(c)\beta^3+O(\beta^4)$
  in a $c$-independent neighbourhood;
\item $c_3(c)\ge\delta_0>0$;
\item $|\Theta''(\beta^*(u);c)|
  \ge\delta_1|u|^{1/2}$
  for $\delta_2\le|u|\le c^{\theta_0/3}$
  (uniform van~der~Corput bound);
\item $|\partial_u^k J(t,u;c)|
  \le C_k(1+|u|)^{-1/2-k}$
  uniformly for all $k$;
\item no secondary stationary points
  uniformly in $c$;
\item \textit{(Uniform symbol-class
  regularity; parameter-dependent
  FIO class.)}
  Uniformly in $(\beta,u,c)$:
  $|\det(dt/d\beta)|\ge\delta_5>0$;
  $|\partial_\beta^j\partial_u^k
  t(\beta,u;c)|\le C_{jk}$
  for all $j,k\ge 0$
  (placing $t(\cdot,u;c)$
  in the parameter-dependent
  class $S^0_{1,0}(c)$
  of Definition~\ref{def:symbol-class},
  as a family of canonical
  transformations, uniform in $c$;
  recall $c$ is an external parameter,
  not a phase-space variable);
  and
  $|\partial_{\tilde{t}}
  (\tilde{t}^3/3
  -\mathrm{sgn}(u)\tilde{t})|
  \ge\delta_4>0$
  outside the Airy critical region
  (uniform non-stationary phase
  bound for H\"ormander Ch.~7.7).
\end{enumerate}
\end{assumption}

\begin{remark}[Role of (vi) and
  parameter-dependent FIO class]
\label{rem:vi-role}
Condition (vi) promotes all estimates
from ``fixed $c$'' to
``uniform in $c\ge c_0$'':
\begin{itemize}
\item $|\det(dt/d\beta)|\ge\delta_5$
  enables
  Lemma~\ref{lem:conormal-stability};
\item $t(\cdot,u;c)\in S^0_{1,0}(c)$
  (as a parameter-dependent family;
  Definition~\ref{def:symbol-class})
  ensures transported amplitudes
  stay in $S^m_{1,0}(c)$;
\item $\delta_4$-bound provides the
  uniform non-stationary phase
  hypothesis required by
  H\"ormander Ch.~7.7
  for uniform operator-norm bounds
  on $R_c\in\Psi^{-\infty}$
  and uniform pointwise remainder
  estimates~\eqref{eq:remainder-u}.
\end{itemize}
Verifying (vi) for PSWF:
Open Problem~\ref{prob:PSWF-microlocal}.
\end{remark}

\begin{theorem}[Conditional local Airy
  normal form \Rig]
\label{thm:airy-normalform}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi);
see Remark~\ref{rem:FIO-language}
for the precise sense of
``microlocal equivalence modulo
smoothing''.}
For $\delta_2\le|u|\le c^{\theta_0/3}$,
$\Phi(u;c)$ is microlocally
equivalent, modulo smoothing operators,
to:
\begin{equation}\label{eq:airy-nf}
  e^{i(\alpha_*u+R(u;c))}\bigl[
  a(u;c)\,
  \mathrm{Ai}\!\bigl(c_3(c)^{-1/3}u\bigr)
  + r(u;c)\bigr],
\end{equation}
where
$a(u;c)\in S^{-1/2}_{1,0}(c)$
(parameter-dependent symbol family
in $u$, uniform in $c\ge c_0$;
Definition~\ref{def:symbol-class};
Lemma~\ref{lem:symbol-stability}(3),
requires (vi)),
$|r(u;c)|\le C_N(1+|u|)^{-N}$
for each fixed $N$ (requires (vi)),
and $R(u;c)$ absorbs the Maslov
phase factor
(Remark~\ref{rem:FIO-language}).
\end{theorem}

\begin{proof}[Proof sketch]
Fold zone:
Proposition~\ref{prop:CFU}(2)
(microlocal equivalence modulo
smoothing; conormal amplitude
stable by
Lemma~\ref{lem:conormal-stability});
Lemma~\ref{lem:Jacobian-rescaling};
Lemma~\ref{lem:symbol-stability}
(amplitude in $S^{-1/2}_{1,0}(c)$
under (i)--(vi)).
Far zone:
Proposition~\ref{prop:CFU}(3).
\end{proof}

\begin{lemma}[Joint scaling in fold zone \Rig]
\label{lem:product-estimate}
\textit{Valid in the fold zone
after canonical Airy rescaling;
not a global statement.}
Under Theorem~\ref{thm:airy-normalform},
for $|u|\ge\delta_2$:
\[
|a(u;c)\cdot\mathrm{Ai}(c_3(c)^{-1/3}u)|
\le C_1 C_2\,|u|^{-3/4}.
\]

\medskip
\noindent\textit{Interpretation of the
exponent $-3/4$.}
The decay exponent $-3/4$ should be
understood as arising from a
\emph{single degenerate stationary
phase expansion} of the oscillatory
integral in the Airy normal form,
and not as a product of two independent
decay mechanisms.

More precisely, after the canonical
Airy rescaling, the integral
\eqref{eq:airy-decomp} is governed
by a cubic phase with a non-degenerate
critical point in the rescaled
variable $\tilde{t}$.
Applying stationary phase in $\tilde{t}$
yields a leading contribution
of size $|u|^{-3/4}$,
where the exponent reflects the
combined scaling of:
\begin{itemize}
\item the cubic degeneracy of
  the phase (A$_2$-fold), and
\item the conormal structure of
  the amplitude transported under CFU.
\end{itemize}

The factorisation
\[
  |a(u;c)| \lesssim |u|^{-1/2},
  \qquad
  |\mathrm{Ai}(c_3(c)^{-1/3}u)|
  \lesssim |u|^{-1/4}
\]
is therefore only an
\emph{a posteriori} bound at the
level of estimates, and does not
reflect the mechanism by which
the exponent is produced.
\end{lemma}

\begin{corollary}[Pointwise decay \Cond]
\label{cor:pointwise}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
There exists $C>0$,
uniform in $c\ge c_0$
and across all three zones, such that:
\begin{equation}\label{eq:pointwise}
  |\Phi(u;c)|\le C(1+|u|)^{-3/4}
  \quad\forall u\in\mathbb{R}.
\end{equation}
\end{corollary}

\begin{proof}
$C:=\max(C_1C_2,\,C_3,\,C_4)$:
\begin{itemize}
\item \textit{Fold zone $|u|\ge\delta_2$:}
  Lemma~\ref{lem:product-estimate}.
\item \textit{Compact zone $|u|\le 1$:}
  compactness + continuity
  (condition (vi))
  $\Rightarrow$ $|\Phi|\le C_3$
  (Erd\'elyi \cite{Erdelyi1956}).
\item \textit{Far zone:}
  $|\Phi|\le C_4|u|^{-1/2}
  \le C_4|u|^{-3/4}$
  (Theorem~\ref{thm:endpoint}).
\end{itemize}
Transition
($c^{\theta_0/3}\le|u|
\le 2c^{\theta_0/3}$):
conditions (iii)--(iv)
make constants compatible;
$C$ is uniform by (ii) and (vi).
\end{proof}

\begin{remark}[Integrability,
  not oscillatory cancellation]
\label{rem:oscillation}
Corollary~\ref{cor:global-bound}
exploits an \emph{integrability
improvement}:
the absolute-value bound
$|\Phi|\le C(1+|u|)^{-3/4}$
yields $O(U^{1/4})$
by integration.
No oscillatory cancellation in $u$
is used.
An oscillatory improvement would
require a second stationary-phase
step in $u$, beyond the scope here.
\end{remark}

\begin{corollary}[Ces\`aro improvement
  \Cond]
\label{cor:global-bound}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
\[
  \left|\int_0^U\Phi(u;c)\,du\right|
  \le C\int_0^U(1+u)^{-3/4}\,du
  = O(U^{1/4}),
\]
improving upon Ces\`aro $O(U^{1/2})$
by integrability gain
(Remark~\ref{rem:oscillation}).
The exponent $-3/4$ is the
joint scaling factor from
Lemma~\ref{lem:product-estimate}.
\end{corollary}

\subsection{Converse and FIO rigidity}

\begin{assumption}[Microlocal
  reconstruction (non-standard) \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ is in the Airy class and:
\begin{enumerate}[\rm(a)]
\item $\sigma(g\mapsto\Phi)
  \big|_\mathcal{C}\ne 0$;
\item $g\mapsto\Phi$ is injective
  modulo smoothing;
\item $\mathrm{WF}(\Phi)
  \subseteq\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
\end{enumerate}
Then $\mathcal{C}$ contains an
A$_2$-fold singularity
uniformly in $c$.
\medskip

\noindent\textit{Status.}
This is an
\emph{inverse problem /
rigidity conjecture},
not a consequence of standard FIO
theory \cite{Zworski2012,Hormander1983}.
Conditions (a)--(c) are stronger
than standard ellipticity;
the implication
``Airy-class bound $\Rightarrow$
A$_2$-fold'' is not available
in current FIO theory
without additional structural
assumptions.
Verification for PSWF: Paper~XVII.
\end{assumption}

\begin{corollary}[Conditional equivalence]
\label{cor:conditional-equiv}
Under Assumption~\ref{hyp:FIO-rigidity}:
Airy-class bound $\Rightarrow$
Assumption~\ref{ass:microlocal}
(modulo the non-standard hypothesis).
The forward direction holds
rigorously without
Assumption~\ref{hyp:FIO-rigidity}.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha;c)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha;c)}$
satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item $c_3(c)\ge\delta_0>0$
  (Olver WKB \cite{Olver1974})?
\item $\theta''(\alpha;c)\ne 0$
  for $\alpha\ne\alpha_*$, uniformly?
\item Conditions (iv) and (vi)
  for the PSWF CFU Jacobian?
\end{enumerate}
Outcomes:
(A)~All hold:
$|\int_0^U\Phi|=O(U^{1/4})$.
(B)~$c_3(c)\to 0$:
Ces\`aro sharp.
\end{problem}
