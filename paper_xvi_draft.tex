\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases Induce A$_2$-Fold Singularities\\
  Whose Normal Form Is the Airy Distribution}

\author{[Author]}

\begin{abstract}
Paper~XV reduces the intermediate-regime
problem to whether the fold amplitude~$\Phi$
satisfies an estimate improving upon
Ces\`aro.
This paper establishes the microlocal
mechanism.
A cubic WKB phase $c_3\beta^3$
produces a degenerate stationary point
of order~2 in the Fourier integral.
Geometrically, this is an A$_2$-fold
singularity of the canonical relation:
the projection $\pi_u:\mathcal{C}\to
T^*\mathbb{R}_u$ has a rank-drop whose
non-degeneracy is expressed, in local
canonical coordinates, by
$\Theta''(\beta_0)=0$, $\Theta'''(\beta_0)\ne 0$.
The Airy distribution is the
universal normal form of this A$_2$
singularity class
(Chester--Friedman--Ursell \cite{CFU1957},
Arnold \cite{Arnold1972},
Golubitsky--Guillemin \cite{Golub1973}).
Under the uniform stability conditions
of Assumption~\ref{ass:microlocal},
\[
  \Phi(u)
  = e^{i\alpha_* u}\,a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
  \qquad a(u,c)\in S^{-1/2}_{1,0},
\]
and
$|\int_0^U\Phi(u)\,du|=O(U^{1/4})$
(Lemma~\ref{lem:airy-integral},
via an absolute-value estimate;
no oscillatory cancellation is used).
The forward implication
(Assumption $\Rightarrow$ Airy bound)
is rigorous;
the converse requires the additional
hypotheses of
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold (canonical relation of
a one-dimensional FIO).
Denote by $\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the projection to the right factor.
\medskip

\noindent\textit{Geometric (coordinate-free)
definition.}
We say $\mathcal{C}$ has an
\emph{A$_2$-fold singularity at
$p_0\in\mathcal{C}$} if:
\begin{enumerate}[\rm(i)]
\item \textit{(Rank-drop of $\pi_u$.)} 
  $\det\,d\pi_u\big|_{p_0}=0$
  (the projection drops rank by~1);
\item \textit{(Simple fold.)}
  $d\!\bigl(\det\,d\pi_u\bigr)\big|_{p_0}\ne 0$
  (the rank-drop locus is a smooth
  hypersurface in $\mathcal{C}$;
  no higher-order degeneracy).
\end{enumerate}
This is the A$_2$-catastrophe type
in Arnold's classification \cite{Arnold1972}.
\medskip

\noindent\textit{Reduction to local
canonical coordinates.}
For a one-dimensional FIO with phase
$\Psi(\beta,u)=\Theta(\beta)+u\beta$,
the canonical relation is
$\mathcal{C}=\{(\beta,\Theta'(\beta);\,
u,-\beta)\}$.
In these coordinates,
\begin{align}
  \label{eq:fold-i}
  \det\,d\pi_u\big|_{(\beta_0,u_0)}=0
  \;&\iff\;
  \Theta''(\beta_0)=0,
  \\
  \label{eq:fold-ii}
  d\!\bigl(\det\,d\pi_u\bigr)\big|_{(\beta_0,u_0)}\ne 0
  \;&\iff\;
  \Theta'''(\beta_0)\ne 0.
\end{align}
Equations~\eqref{eq:fold-i}--\eqref{eq:fold-ii}
are a coordinate expression of
conditions~(i)--(ii)
for this choice of phase function;
the geometric conditions~(i)--(ii)
are invariant under canonical
transformations of $\mathcal{C}$.
\medskip

\noindent By the Thom--Boardman
classification theorem
\cite{Golub1973},
the Airy function is the
universal local normal form of every
one-dimensional FIO with an A$_2$-fold
singularity:
microlocally near $p_0$,
such an FIO is equivalent (under a
local symplectic coordinate change)
to a scalar multiple of $\mathrm{Ai}$.
\end{definition}

\begin{remark}[Polynomial phases
  are coordinate representations]
\label{rem:phases-coordinates}
The phase formula $e^{i\Theta(\beta)}$
is coordinate-dependent:
it changes under canonical transformations
of $T^*\mathbb{R}$.
The canonical invariant is the
Lagrangian submanifold
$\Lambda=\{(\beta,\Theta'(\beta))\}
\subset T^*\mathbb{R}$
and its Thom--Boardman stratum.
The expressions $e^{ic_3\beta^3}$
and $\mathrm{Ai}(c_3^{-1/3}u)$
are coordinate representations
of the same A$_2$ Lagrangian geometry
in the $\beta$- and $u$-variables
respectively.
\end{remark}

\begin{remark}[Microlocal transport:
  canonical relation swap]
\label{rem:microlocal-transport}
The Fourier transform acts on
canonical relations by the
\emph{canonical relation swap}
$(\beta,\xi;\,u,\eta)\mapsto(u,\eta;\,\beta,\xi)$,
i.e.\ it exchanges the left and right
factors of $T^*\mathbb{R}_\beta\times
T^*\mathbb{R}_u$.
Under this swap:
an A$_2$-fold singularity of
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
corresponds, after the swap,
to an A$_2$-fold singularity of
$\pi_\beta:\mathcal{C}\to T^*\mathbb{R}_\beta$
(the projection to the left factor),
\emph{provided both projections
$\pi_u$ and $\pi_\beta$
satisfy the non-degeneracy
condition~(ii) of
Definition~\ref{def:A2-class}
at the corresponding points}.
Without this two-sided non-degeneracy,
the fold can degenerate further
(to A$_3$, cusp, or higher)
after the swap.
\end{remark}

\subsection{The fold amplitude}

\begin{equation}\label{eq:Phi-def}
  \Phi(u)
  :=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \qquad u\in\mathbb{R},
\end{equation}
$\chi\in C^\infty_c$,
$\alpha_*>0$ fold point.
The WKB amplitude:
\begin{equation}\label{eq:g-def}
  g(\alpha)
  := |\alpha-\alpha_*|^{-1/2}
  e^{i\theta(\alpha)}\chi(\alpha),
\end{equation}
with $\theta''(\alpha_*)=0$,
$c_3:=\frac{1}{6}\theta'''(\alpha_*)\ne 0$.
$\beta=\alpha-\alpha_*$,
$\Theta(\beta):=\theta(\alpha_*+\beta)$:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta.
\end{equation}

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
$|\Phi(u)|=O(|u|^{-1/2})$ for all
$\theta\in C^\infty$.
For $\theta\equiv 0$:
$\Phi(u)=\sqrt{\pi}\,e^{i\alpha_*u}
|u|^{-1/2}e^{-i\pi/4}+O(|u|^{-3/2})$.
\end{theorem}

\begin{proof}
Erd\'elyi ($\theta\equiv 0$);
van~der~Corput with weight
$|\beta|^{-1/2}$ (general $\theta$).
\end{proof}

\section{Cubic Phase Induces A$_2$-Fold}
\label{sec:nonpreservation}

\begin{assumption}[Stable A$_2$-fold:
  local conditions \Cond]
\label{ass:fold-local}
In addition to
$\Theta(\beta)=c_3\beta^3+O(\beta^4)$,
$c_3\ne 0$:
\begin{enumerate}[\rm(F1)]
\item \textit{(No higher degeneracy.)}
  $\Theta''(\beta)\ne 0$ for
  $\beta\ne 0$ in
  $\mathrm{supp}\,\tilde\chi\setminus\{0\}$:
  $\mathcal{C}$ has exactly one fold point,
  with no cusp (A$_3$) or higher.
\item \textit{(Non-cancellation.)}
  The oscillatory integral
  $\int|\beta|^{-1/2}
  e^{i\Theta(\beta)}\chi(\beta)\,d\beta$
  is not identically zero in any
  neighbourhood of $0$.
\item \textit{(Locally monotone
  Legendre map.)}
  $\Theta':\mathrm{supp}\,\chi\to\mathbb{R}$
  has exactly one critical point
  (at $\beta=0$):
  the Legendre map is a local
  diffeomorphism away from $\beta=0$,
  ensuring the projection $\pi_u$
  is non-singular away from the fold.
\end{enumerate}
\end{assumption}

\begin{proposition}[Cubic WKB phase
  $\Rightarrow$ A$_2$-fold
  $\Rightarrow$ Airy normal form \Rig]
\label{prop:nonpreservation}
Let $\Theta(\beta)=c_3\beta^3+O(\beta^4)$,
$c_3\ne 0$, satisfying
Assumption~\ref{ass:fold-local}.
Let $F(u):=\int_{\mathbb{R}}
|\beta|^{-1/2}
e^{i(\Theta(\beta)+\beta u)}
\chi(\beta)\,d\beta$.
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold structure.)}
  $\mathcal{C}$ has an A$_2$-fold
  singularity at $(\beta_0,u_0)=(0,0)$.
  In local canonical coordinates
  $\Psi=\Theta(\beta)+u\beta$,
  conditions~\eqref{eq:fold-i}--\eqref{eq:fold-ii}
  read:
  $\Theta''(0)=0$ (from
  $\Theta(\beta)=c_3\beta^3+O(\beta^4)$)
  and $\Theta'''(0)=6c_3\ne 0$.
  Assumption~\ref{ass:fold-local}(F1)
  guarantees no higher stratum.
\item \textit{(Symplectic normal form
  reduction via CFU.)}
  The canonical relation swap
  (Remark~\ref{rem:microlocal-transport})
  transports the A$_2$-fold
  from $\pi_u$ to $\pi_\beta$:
  Remark~\ref{rem:microlocal-transport}
  applies since (F1)--(F3) ensure
  both projections are non-degenerate
  away from the fold point.
  The CFU theorem \cite{CFU1957}
  then provides a local smooth
  coordinate change
  $\beta\mapsto t(\beta,u)$
  which is \emph{canonical}
  (i.e.\ it preserves the symplectic
  form on $\mathcal{C}$)
  and satisfies:
  \begin{equation}\label{eq:CFU-nf}
    \Theta(\beta)+u\beta
    = \tfrac{1}{3}t^3 - c_3^{-1/3}u\,t
    + R(u),
  \end{equation}
  where $R(u)=R(u;c)$ is smooth.
  The map $\beta\mapsto t$ is a
  local diffeomorphism
  by (F3) (away from $\beta=0$)
  and by A$_2$-stability
  (near $\beta=0$).
  Under~\eqref{eq:CFU-nf},
  condition~(F2) ensures
  $|\beta(t,u)|^{-1/2}|d\beta/dt|$
  is non-zero,
  and the resulting integral is
  the standard Airy representation:
  \begin{equation}\label{eq:airy-FT}
    F(u)
    \;\sim\;
    e^{iR(u)}\,c_3^{-1/6}\,P(u)\,
    \mathrm{Ai}\!\bigl(-c_3^{-1/3}u\bigr),
    \quad|u|\to\infty.
  \end{equation}
\item \textit{(Stationary phase value.)}
  The large-$|u|$ oscillatory phase of
  $\mathrm{Ai}(-c_3^{-1/3}u)$
  is, by the standard expansion
  $\mathrm{Ai}(-t)\sim
  \pi^{-1/2}t^{-1/4}
  \sin(\frac{2}{3}t^{3/2}+\frac{\pi}{4})$
  for $t\to+\infty$ \cite{Olver1974}:
  \begin{equation}\label{eq:stat-phase-value}
    S(u):=\tfrac{2}{3}c_3^{-1/2}|u|^{3/2}.
  \end{equation}
  This is the stationary phase value
  $S(u)=\Theta(\beta^*(u))+u\beta^*(u)$
  at the critical point
  $\beta^*(u)\colon\Theta'(\beta^*)=-u$.
  It is an \emph{asymptotic quantity}:
the Airy function has no global
  phase representation,
  and $S(u)$ is meaningful only
  in the oscillatory regime $u\to-\infty$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Direct computation;
Definition~\ref{def:A2-class}
and~\eqref{eq:fold-i}--\eqref{eq:fold-ii}.
(2) The canonical coordinate change:
at $(\beta,u)=(0,0)$,
$\beta\mapsto t$ is the identity
to leading order;
(F1) (no secondary critical points)
ensures $t(\beta,u)$ is a
global diffeomorphism on
$\mathrm{supp}\,\chi$;
(F3) ensures the Legendre map
is a local diffeomorphism away from $\beta=0$;
Thom--Boardman stability
\cite{Golub1973} guarantees
the canonical form~\eqref{eq:CFU-nf}
exists as a smooth family
and the coordinate change
is canonical (symplectic-form-preserving
on $\mathcal{C}$) \cite{CFU1957}.
(F2) ensures non-vanishing Jacobian
amplitude.
(3) Standard \cite{Olver1974} Ch.~11.
\end{proof}

\section{Airy Normal Form Theorem}
\label{sec:normalform}

\begin{assumption}[Uniform A$_2$-stability \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local} holds
uniformly in $c\ge c_0$.
In addition:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta)=c_3\beta^3+O(\beta^4)$
  uniformly in a $c$-independent neighbourhood;
\item \textit{(A$_2$-stability)}
  $c_3(c)\ge\delta_0>0$ uniformly;
\item \textit{(Uniform fold projection)}
  $|\Theta''(\beta^*(u))|
  \ge\delta_1|u|^{1/2}$
  uniformly for $\delta_2\le|u|
  \le c^{\theta_0/3}$;
\item \textit{(Uniform Jacobian)}
  $0<\delta_3\le|d\beta/dt|
  \le\delta_3^{-1}$
  uniformly in $(u,c)$.
\end{enumerate}
\end{assumption}

\begin{theorem}[Airy normal form \Rig]
\label{thm:airy-normalform}
Under Assumption~\ref{ass:microlocal},
the fold amplitude~\eqref{eq:Phi-WKB}
satisfies:
\begin{equation}\label{eq:airy-normalform}
  \Phi(u)
  = e^{i\alpha_* u}\,
  a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
\end{equation}
where $a(u,c)\in S^{-1/2}_{1,0}$:
$|\partial_u^j a(u,c)|
\le C_j(1+|u|)^{-1/2-j}$,
uniformly in $c$ and
$\delta_2\le|u|\le c^{\theta_0/3}$.
The factor $e^{i\alpha_*u}$ is a global
Fourier shift; it carries no microlocal
fold information.
The stationary phase value of
$\mathrm{Ai}(c_3^{-1/3}u)$
for large $|u|$ is $S(u)$
of~\eqref{eq:stat-phase-value},
an asymptotic quantity,
not a global phase function.
\end{theorem}

\begin{proof}[Proof sketch]
Apply the canonical coordinate change
of Proposition~\ref{prop:nonpreservation}(2)
to~\eqref{eq:Phi-WKB},
uniformly in $(u,c)$:
condition~(iii) is the uniform
non-degeneracy of the fold projection;
condition~(iv) is the uniform
Jacobian bound.
The resulting integral is the
standard Airy representation;
$a\in S^{-1/2}_{1,0}$ follows
from Erd\'elyi and condition~(i).
\end{proof}

\begin{lemma}[Airy class
  $\Rightarrow$ improved Ces\`aro \Rig]
\label{lem:airy-integral}
Under the conclusion of
Theorem~\ref{thm:airy-normalform}:
$\bigl|\int_0^U\Phi(u)\,du\bigr|
= O(U^{1/4})$ as $U\to\infty$.
\end{lemma}

\begin{proof}
The bound is obtained via an
\emph{absolute-value estimate};
no oscillatory cancellation
(van der Corput or otherwise)
is invoked.
Write $\Phi(u)=e^{i\alpha_*u}a(u,c)
\mathrm{Ai}(c_3^{-1/3}u)$ and split at $u=1$.
\medskip

\noindent\textit{Region $0\le u\le 1$:}
bounded, contributes $O(1)$.
\medskip

\noindent\textit{Region $1\le u\le U$
  (positive argument of Ai).}
$|\mathrm{Ai}(c_3^{-1/3}u)|
\le C u^{-1/4}e^{-c'u^{3/2}}$
\cite{Olver1974}: exponential decay.
Using $|a(u,c)|\le Cu^{-1/2}$
and $|e^{i\alpha_*u}|=1$:
\[
  \int_1^U|a|\cdot
  |\mathrm{Ai}(c_3^{-1/3}u)|\,du
  \le C\int_1^\infty
  u^{-3/4}e^{-c'u^{3/2}}\,du
  = O(1).
\]
\medskip

\noindent\textit{Region $-U\le u\le -1$
  (negative argument of Ai).}
$|\mathrm{Ai}(c_3^{-1/3}u)|
= |\mathrm{Ai}(-c_3^{-1/3}|u|)|
\le C|u|^{-1/4}$
\cite{Olver1974}
(oscillatory; no exponential decay).
Using $|a(-u,c)|\le Cu^{-1/2}$
and $|e^{-i\alpha_*u}|=1$:
\[
  \int_1^U|a(-u,c)|\cdot
  |\mathrm{Ai}(-c_3^{-1/3}u)|\,du
  \le C\int_1^U u^{-3/4}\,du
  = O(U^{1/4}).
\]
This is the dominant contribution.
The factor $e^{i\alpha_*u}$
is bounded by~1 throughout;
its oscillation is not used
and cannot improve this bound
without additional structure.
\medskip

\noindent\textit{Conclusion.}
$|\int_0^U\Phi\,du|
\le O(1)+O(U^{1/4})=O(U^{1/4})$,
strictly better than
Ces\`aro $O(U^{1/2})$.
The bound uses:
(a)~$a\in S^{-1/2}_{1,0}$ (uniform in $c$),
(b)~$|\mathrm{Ai}(-t)|=O(t^{-1/4})$,
(c)~$|e^{i\alpha_*u}|\le 1$.
None of (a)--(c) alone suffices;
the improvement is a global-in-$u$
estimate, not a local asymptotic.
\end{proof}

\subsection{Converse and FIO rigidity}
\label{subsec:converse}

\begin{hypothesis}[FIO inversion
  rigidity \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ lies in the Airy class
($a\in S^{-1/2}_{1,0}$,
stationary phase value $S(u)\sim C|u|^{3/2}$
uniformly in $c$)
and additionally:
\begin{enumerate}[\rm(a)]
\item \textit{(Symbol non-vanishing
  on $\mathcal{C}$.)}
  The full symbol $\sigma$
  of the FIO $g\mapsto\Phi$
  satisfies $\sigma|_{\mathcal{C}}\ne 0$
  (ellipticity on the canonical relation);
\item \textit{(Injectivity mod smoothing.)}
  $g\mapsto\Phi$ is injective
  modulo smoothing operators;
\item \textit{(Wavefront stability.)}
  $\mathrm{WF}(\Phi)\subset
  \pi_u(\mathcal{C})$
  and this inclusion is an equality
  on the A$_2$ stratum.
\end{enumerate}
Then the canonical relation of $g$
contains an A$_2$-fold singularity
uniformly in $c$.
\end{hypothesis}

\begin{remark}[Status]
\label{rem:rigidity-status}
Conditions (a)--(c) are standard
microlocal ellipticity and injectivity
conditions in FIO theory
\cite{Zworski2012}.
(a) excludes amplitude cancellation
unrelated to fold geometry;
(b) rules out smoothing perturbations
that could mimic Airy oscillations;
(c) ensures the wavefront set
faithfully reflects the A$_2$ stratum.
Without (a)--(c),
$c$-dependent amplitude interference
or smoothing can produce Airy-type
bounds without geometric fold structure.
Verification for the PSWF case
is the content of Paper~XVII.
\end{remark}

\begin{corollary}[Conditional equivalence
  \Cond]
\label{cor:conditional-equiv}
Under Hypothesis~\ref{hyp:FIO-rigidity}:
$\mathrm{Airy{-}class\ bound}
\iff
\mathrm{Assumption~\ref{ass:microlocal}}$.
Without it:
$\mathrm{Assumption~\ref{ass:microlocal}}
\Rightarrow
\mathrm{Airy{-}class\ bound}$ only.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$
satisfy Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item \textit{(A$_2$-stability)}
  $c_3(c)\ge\delta_0>0$?
  Approach:
  Olver turning-point WKB \cite{Olver1974}.
\item \textit{(No higher degeneracy)}
  $\theta''(\alpha)\ne 0$ for
  $\alpha\ne\alpha_*$ near $\alpha_*$,
  uniformly?
\item \textit{(Uniform Jacobian)}
  Assumption~\ref{ass:microlocal}(iv)
  for the PSWF CFU Jacobian?
\end{enumerate}
Two outcomes:
\begin{enumerate}[\rm(A)]
\item Assumptions hold:
  $|\int_0^U\Phi|=O(U^{1/4})$,
  $S(c)=O(c^{\theta_0/2})$.
\item $c_3(c)\to 0$ or (F1) fails:
  $\mathcal{C}_c$ exits A$_2$ stratum;
  Ces\`aro sharp.
\end{enumerate}
\end{problem}
