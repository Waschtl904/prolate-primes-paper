\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases Induce A$_2$-Fold Singularities\\
  Whose Normal Form Is the Airy Distribution}

\author{[Author]}

\begin{abstract}
Paper~XV reduces the intermediate-regime
problem to whether the fold amplitude~$\Phi$
satisfies an estimate improving upon
Ces\`aro.
This paper establishes the microlocal
mechanism via three separate results.
\begin{enumerate}[\rm(I)]
\item \textit{(Microlocal normal form.)}
  Theorem~\ref{thm:airy-normalform}:
  Under the uniform A$_2$-stability
  conditions of Assumption~\ref{ass:microlocal},
  $\Phi(u)=e^{i\alpha_*u}a(u,c)
  \mathrm{Ai}(c_3^{-1/3}u)$,
  $a\in S^{-1/2}_{1,0}$,
  microlocally near the fold point.
\item \textit{(Pointwise decay.)}
  Lemma~\ref{lem:pointwise}:
  Under the same assumptions,
  $|\Phi(u)|\le C(1+|u|)^{-3/4}$
  uniformly for all $u$,
  bridging the local Airy structure
  and the global integral estimate.
\item \textit{(Global $L^1$-growth bound.)}
  Corollary~\ref{cor:global-bound}:
  $|\int_0^U\Phi\,du|=O(U^{1/4})$,
  obtained by integrating the
  pointwise bound;
  no additional oscillatory cancellation
  beyond what is already encoded
  in the Airy normal form is used.
\end{enumerate}
The forward direction is rigorous;
the converse requires
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold.
Let $\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
denote the projection to the right factor.
\medskip

\noindent\textit{Geometric
(coordinate-free) definition.}
We say $\mathcal{C}$ has an
\emph{A$_2$-fold singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$
  (rank-drop of $\pi_u$ by~1);
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$
  (simple fold; no higher degeneracy).
\end{enumerate}
This is the A$_2$-type in Arnold's
classification \cite{Arnold1972}.
\medskip

\noindent\textit{Reduction to local
canonical coordinates.}
For a one-dimensional FIO
with phase $\Psi(\beta,u)
=\Theta(\beta)+u\beta$,
the canonical relation is
$\mathcal{C}=\{(\beta,\Theta'(\beta);\,
u,-\beta)\}$.
In these coordinates:
\begin{align}
  \label{eq:fold-i}
  \det\,d\pi_u=0
  \;&\iff\;
  \Theta''(\beta_0)=0,
  \\
  \label{eq:fold-ii}
  d(\det\,d\pi_u)\ne 0
  \;&\iff\;
  \Theta'''(\beta_0)\ne 0.
\end{align}
Equations~\eqref{eq:fold-i}--\eqref{eq:fold-ii}
are the coordinate expression of~(i)--(ii)
for this phase;
the geometric conditions~(i)--(ii)
are invariant under canonical
transformations of $\mathcal{C}$.
\medskip

\noindent By the Thom--Boardman
classification theorem \cite{Golub1973},
any one-dimensional FIO with an
A$_2$-fold singularity is,
microlocally near $p_0$,
equivalent to a scalar multiple
of $\mathrm{Ai}$ under a local
symplectic coordinate change.
\end{definition}

\begin{remark}[Polynomial phases
  are coordinate representations]
\label{rem:phases-coordinates}
The canonical invariant is the
Lagrangian submanifold
$\Lambda=\{(\beta,\Theta'(\beta))\}
\subset T^*\mathbb{R}$
and its Thom--Boardman stratum.
The expressions $e^{ic_3\beta^3}$
and $\mathrm{Ai}(c_3^{-1/3}u)$
are coordinate representations
of the same A$_2$ geometry.
\end{remark}

\begin{remark}[Microlocal transport
  under Fourier duality]
\label{rem:microlocal-transport}
The Fourier transform acts on canonical
relations by the exchange
$(\beta,\xi;\,u,\eta)
\mapsto(u,\eta;\,\beta,\xi)$.
An A$_2$-fold singularity is
preserved under this exchange in the
following precise sense:
if the Lagrangian map germ of
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
belongs to the stable A$_2$-equivalence
class (Whitney fold of Lagrangian
map germs, stable in the sense of
Mather stability \cite{Golub1973}),
then the germ of
$\pi_\beta:\mathcal{C}
\to T^*\mathbb{R}_\beta$
also belongs to the stable A$_2$-class.
This is a local statement about
stable equivalence of Lagrangian map
germs; it fails if the fold is
not Mather-stable
(e.g.\ if $c_3(c)\to 0$).
\end{remark}

\subsection{The fold amplitude}

\begin{equation}\label{eq:Phi-def}
  \Phi(u)
  :=\int_{\mathbb{R}}
  |\beta|^{-1/2}\chi(\alpha_*+\beta)
  e^{i\beta u}\,d\beta,
  \quad u\in\mathbb{R},
\end{equation}
$\chi\in C^\infty_c$,
$\alpha_*>0$ fold point.
The WKB amplitude:
\begin{equation}\label{eq:g-def}
  g(\alpha)
  := |\alpha-\alpha_*|^{-1/2}
  e^{i\theta(\alpha)}\chi(\alpha),
\end{equation}
$\theta''(\alpha_*)=0$,
$c_3:=\frac{1}{6}\theta'''(\alpha_*)\ne 0$.
$\beta=\alpha-\alpha_*$,
$\Theta(\beta):=\theta(\alpha_*+\beta)$:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta.
\end{equation}

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
$|\Phi(u)|=O(|u|^{-1/2})$
for all $\theta\in C^\infty$.
For $\theta\equiv 0$:
$\Phi(u)=\sqrt{\pi}\,e^{i\alpha_*u}
|u|^{-1/2}e^{-i\pi/4}
+O(|u|^{-3/2})$.
\end{theorem}

\begin{proof}
Erd\'elyi ($\theta\equiv 0$);
van~der~Corput with weight
$|\beta|^{-1/2}$ (general $\theta$).
\end{proof}

\section{Cubic Phase Induces A$_2$-Fold}
\label{sec:nonpreservation}

\begin{assumption}[Stable A$_2$-fold:
local conditions \Cond]
\label{ass:fold-local}
In addition to
$\Theta(\beta)=c_3\beta^3+O(\beta^4)$,
$c_3\ne 0$:
\begin{enumerate}[\rm(F1)]
\item \textit{(No higher degeneracy.)}
  $\Theta''(\beta)\ne 0$
  for $\beta\ne 0$ in
  $\mathrm{supp}\,\chi\setminus\{0\}$;
  at most one fold point (at $\beta=0$),
  no cusp or higher stratum.
\item \textit{(Non-cancellation.)}
  $\int|\beta|^{-1/2}
  e^{i\Theta}\chi\,d\beta$
  is not identically zero
  near $0$.
\item \textit{(Locally monotone
  Legendre map.)}
  $\Theta'$ has exactly one critical
  point (at $\beta=0$)
  on $\mathrm{supp}\,\chi$;
  no secondary stationary points.
\end{enumerate}
\end{assumption}

\begin{proposition}[Cubic phase
  $\Rightarrow$ A$_2$-fold
  $\Rightarrow$ local Airy normal form \Rig]
\label{prop:nonpreservation}
Under Assumption~\ref{ass:fold-local},
let $F(u):=\int|\beta|^{-1/2}
e^{i(\Theta(\beta)+\beta u)}
\chi(\beta)\,d\beta$.
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold.)}
  $\mathcal{C}$ has an A$_2$-fold
  singularity at $(0,0)$:
  $\Theta''(0)=0$,
  $\Theta'''(0)=6c_3\ne 0$;
  (F1) ensures no higher stratum.
\item \textit{(Local CFU normal form.)}
  In a neighbourhood $U_\epsilon$
  of $\beta=0$ (the fold region),
  the CFU theorem \cite{CFU1957}
  provides a local smooth coordinate
  change $\beta\mapsto t(\beta,u)$
  --- a \emph{local diffeomorphism
  on $U_\epsilon$, not on all of
  $\mathrm{supp}\,\chi$} ---
  and smooth functions $R(u)$,
  $\mu(u)=c_3^{-1/3}u(1+O(u))$ such that:
  \begin{equation}\label{eq:CFU-nf}
    \Theta(\beta)+u\beta
    = \tfrac{1}{3}t^3
    - \mu(u)\,t
    + R(u),
    \quad \beta\in U_\epsilon.
  \end{equation}
  This is a \emph{generating function
  equivalence}: it equates the phase
  generating function of $\mathcal{C}$
  with the Airy generating function
  on $U_\epsilon$.
  It does not extend to a global
  diffeomorphism on $\mathrm{supp}\,\chi$.
  \medskip

  The Jacobi factor
  $J(t,u):=d\beta/dt$
  is smooth and satisfies
  $J(t,u)\sim c_3^{-1/6}|u|^{-1/2}$
  for large $|u|$
  (by standard CFU expansion \cite{CFU1957}).
  \medskip

  Outside $U_\epsilon$
  (the non-fold region),
  $\Theta''(\beta)\ne 0$ by (F1),
  so $\Phi$ is a non-degenerate
  oscillatory integral
  with symbol in $S^{-1/2}_{1,0}$;
  standard van~der~Corput gives
  $|\Phi(u)|\le C|u|^{-1/2}$
  in this region.
\item \textit{(Symbol class\
  $S^{-1/2}_{1,0}$ in $u$.)}
  In the fold region,
  the amplitude
  $|\beta(t,u)|^{-1/2}J(t,u)$
  is a symbol of order $-1/2$
  in $u$.
  This follows from the
  \emph{parametric stationary phase
  method} (H\"ormander
  \cite{Hormander1983}, Ch.~7.7):
  the $t$-integral over the
  Airy kernel with smooth
  amplitude $a(t,u)$ is a
  classical symbol in $u$
  of order $-1/2$,
  \[
    \int_{\mathbb{R}}
    a(t,u)\,
    e^{i(\frac{1}{3}t^3-\mu(u)t)}
    \,dt
    = a_0(u)\,\mathrm{Ai}(-\mu(u)),
    \quad a_0\in S^{-1/2}_{1,0},
  \]
  where $a_0$ is not directly
  inherited from $|\beta|^{-1/2}$
  on the $\beta$-side but arises
  from the Jacobian
  $J(t,u)\sim c_3^{-1/6}|u|^{-1/2}$.
\item \textit{(Stationary phase value.)}
  The asymptotic phase of
  $\mathrm{Ai}(-\mu(u))$
  for $u\to-\infty$ is
  $S(u)=\frac{2}{3}c_3^{-1/2}|u|^{3/2}$,
  the stationary phase value at
  $\beta^*:\Theta'(\beta^*)=-u$.
  This is an asymptotic quantity;
  the Airy function has no global
  phase representation.
\end{enumerate}
\end{proposition}

\section{Airy Normal Form,
  Pointwise Decay,
  and Global Integrability}
\label{sec:normalform}

\begin{assumption}[Uniform A$_2$-stability \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$.
In addition:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta)=c_3\beta^3+O(\beta^4)$
  uniformly in a $c$-independent
  neighbourhood;
\item \textit{(A$_2$-stability)}
  $c_3(c)\ge\delta_0>0$ uniformly;
\item \textit{(Uniform fold projection)}
  $|\Theta''(\beta^*(u))|
  \ge\delta_1|u|^{1/2}$
  for $\delta_2\le|u|\le c^{\theta_0/3}$;
\item \textit{(Uniform Jacobian)}
  $0<\delta_3\le|J(t,u)|
  \le\delta_3^{-1}$
  uniformly in $(u,c)$;
\item \textit{(No secondary stationary
  points)}
  (F3) holds uniformly in $c$:
  $\Theta'$ has exactly one critical
  point on $\mathrm{supp}\,\chi$
  for all $c\ge c_0$.
\end{enumerate}
\end{assumption}

\begin{theorem}[Local Airy normal form \Rig]
\label{thm:airy-normalform}
Under Assumption~\ref{ass:microlocal},
microlocally near the fold point
$(\beta_0,u_0)=(0,0)$:
\begin{equation}\label{eq:airy-normalform}
  \Phi(u)
  = e^{i\alpha_* u}\,
  a(u,c)\,
  \mathrm{Ai}\!\bigl(c_3^{-1/3}u\bigr),
\end{equation}
where $a(u,c)\in S^{-1/2}_{1,0}$
in the $u$-variable
(H\"ormander parametric stationary
phase \cite{Hormander1983}):
$|\partial_u^j a|\le C_j(1+|u|)^{-1/2-j}$,
uniformly in $c$ and
$\delta_2\le|u|\le c^{\theta_0/3}$.
The factor $e^{i\alpha_*u}$ carries
no microlocal fold information.
\end{theorem}

\begin{proof}[Proof sketch]
Apply the local CFU coordinate
change of
Proposition~\ref{prop:nonpreservation}(2)
to the fold region $U_\epsilon$;
condition~(iii) ensures non-degeneracy
of the fold projection,
(iv) bounds the Jacobian,
(v) ensures no secondary stationary
points.
Outside $U_\epsilon$,
the amplitude is in $S^{-1/2}_{1,0}$
by van~der~Corput.
The symbol class follows from
H\"ormander's parametric stationary
phase (Proposition~\ref{prop:nonpreservation}(3)).
\end{proof}

\begin{lemma}[Pointwise decay \Rig]
\label{lem:pointwise}
Under Assumption~\ref{ass:microlocal},
\begin{equation}\label{eq:pointwise}
  |\Phi(u)|
  \le C(1+|u|)^{-3/4}
  \quad\text{for all }u\in\mathbb{R},
\end{equation}
uniformly in $c\ge c_0$.
\end{lemma}

\begin{proof}
Decompose via a smooth partition
of unity $1=\varphi_{\mathrm{fold}}
+\varphi_{\mathrm{far}}$,
where $\varphi_{\mathrm{fold}}$ is
supported in $|u|\le 2c^{\theta_0/3}$
(fold zone) and
$\varphi_{\mathrm{far}}$ is supported
in $|u|\ge c^{\theta_0/3}$ (far zone).
\medskip

\noindent\textit{Fold zone.}
By Theorem~\ref{thm:airy-normalform}:
$|\Phi(u)|
\le|a(u,c)|\cdot
|\mathrm{Ai}(c_3^{-1/3}u)|$.
\begin{itemize}
\item For $u\ge 0$:
  $|\mathrm{Ai}(c_3^{-1/3}u)|
  \le C|u|^{-1/4}e^{-c'u^{3/2}}
  \le C|u|^{-1/4}$
  and $|a|\le C|u|^{-1/2}$,
  so $|\Phi|\le C|u|^{-3/4}$.
\item For $u<0$:
  $|\mathrm{Ai}(c_3^{-1/3}u)|
  =|\mathrm{Ai}(-c_3^{-1/3}|u|)|
  \le C|u|^{-1/4}$
  \cite{Olver1974}
  and $|a|\le C|u|^{-1/2}$,
  so $|\Phi|\le C|u|^{-3/4}$.
\end{itemize}
Both cases give
$|\Phi(u)|\le C(1+|u|)^{-3/4}$
in the fold zone.
\medskip

\noindent\textit{Far zone.}
Theorem~\ref{thm:endpoint} gives
$|\Phi(u)|\le C|u|^{-1/2}$.
Since $|u|\ge c^{\theta_0/3}$,
we have $|u|^{-1/2}\le|u|^{-3/4}$
for $|u|\ge 1$.
Hence $|\Phi(u)|\le C|u|^{-3/4}$
in the far zone as well.
\medskip

\noindent\textit{Transition.}
In the overlap region
$c^{\theta_0/3}\le|u|
\le 2c^{\theta_0/3}$,
both bounds hold and give
$|\Phi|\le C|u|^{-3/4}$
with uniform constant
(by conditions (iii)--(iv)
of Assumption~\ref{ass:microlocal}).
Combining all zones:
$|\Phi(u)|\le C(1+|u|)^{-3/4}$
uniformly for all $u$.
\end{proof}

\begin{lemma}[Global $L^1$-growth
  estimate \Rig]
\label{lem:airy-integral}
Under Lemma~\ref{lem:pointwise}:
\[
  \left|\int_0^U\Phi(u)\,du\right|
  \le \int_0^U|\Phi(u)|\,du
  \le C\int_0^U(1+u)^{-3/4}\,du
  = O(U^{1/4}).
\]
This bound uses the pointwise
estimate~\eqref{eq:pointwise}
and no oscillatory cancellation
beyond what is already encoded
in the Airy normal form
(i.e.\ the $|u|^{-1/4}$ decay
of $\mathrm{Ai}(-t)$ for $t\to+\infty$
is itself a consequence of
stationary phase within the
Airy representation;
no further cancellation is invoked here).
\end{lemma}

\begin{corollary}[Global $L^1$-bound \Rig]
\label{cor:global-bound}
Under Assumption~\ref{ass:microlocal}:
\[
  \left|\int_0^U\Phi(u)\,du\right|
  = O(U^{1/4}),
  \quad U\to\infty.
\]
The improvement over Ces\`aro
$O(U^{1/2})$ is due to the
structural decay
$(1+|u|)^{-3/4}$ established
in Lemma~\ref{lem:pointwise},
which combines:
(a)~$a\in S^{-1/2}_{1,0}$
(parametric stationary phase),
(b)~$|\mathrm{Ai}(-t)|=O(t^{-1/4})$
(Airy asymptotics),
and (c)~the uniform transition
between fold and far zones.
\end{corollary}

\subsection{Converse and FIO rigidity}
\label{subsec:converse}

\begin{hypothesis}[FIO inversion
  rigidity \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ lies in the Airy class
and additionally:
\begin{enumerate}[\rm(a)]
\item \textit{(Symbol ellipticity
  on $\mathcal{C}$.)}
  $\sigma(g\mapsto\Phi)
  \big|_\mathcal{C}\ne 0$;
\item \textit{(Injectivity
  mod smoothing.)}
  $g\mapsto\Phi$ is injective
  modulo smoothing operators;
\item \textit{(Wavefront stability.)}
  $\mathrm{WF}(\Phi)
  =\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
\end{enumerate}
Then the canonical relation of $g$
contains an A$_2$-fold singularity
uniformly in $c$.
\end{hypothesis}

\begin{remark}[Status]
\label{rem:rigidity-status}
Conditions (a)--(c) are standard
microlocal conditions
\cite{Zworski2012}.
Verification for PSWF is Paper~XVII.
\end{remark}

\begin{corollary}[Conditional equivalence
  \Cond]
\label{cor:conditional-equiv}
Under Hypothesis~\ref{hyp:FIO-rigidity}:
$\mathrm{Airy\text{-}class\ bound}
\iff
\mathrm{Assumption~\ref{ass:microlocal}}$.
Without it: $\Rightarrow$ only.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha)}$
satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item \textit{(A$_2$-stability)}
  $c_3(c)\ge\delta_0>0$?
  Approach: Olver turning-point WKB
  \cite{Olver1974}.
\item \textit{(No higher degeneracy)}
  $\theta''(\alpha)\ne 0$
  for $\alpha\ne\alpha_*$,
  uniformly?
\item \textit{(Uniform Jacobian)}
  Assumption~\ref{ass:microlocal}(iv)
  for the PSWF CFU Jacobian?
\end{enumerate}
Outcomes:
\begin{enumerate}[\rm(A)]
\item Assumptions hold:
  $|\int_0^U\Phi|=O(U^{1/4})$,
  $S(c)=O(c^{\theta_0/2})$.
\item $c_3(c)\to 0$ or (F1) fails:
  $\mathcal{C}_c$ exits A$_2$ stratum;
  Ces\`aro sharp.
\end{enumerate}
\end{problem}
