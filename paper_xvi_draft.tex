\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases, A$_2$-Fold Singularities,\\
  and the Airy Normal Form}

\author{[Author]}

\begin{abstract}
This paper works in the \emph{one-dimensional setting} throughout
($\beta,u\in\mathbb{R}$).
A cubic WKB phase $c_3(c)\beta^3$
produces a degenerate stationary point
of order~2, which is an A$_2$-fold
singularity of the canonical relation
$\mathcal{C}$ in the sense of
Arnol'd's classification
\cite{Arnold1972}.
The Airy distribution is the universal
local normal form of this singularity,
via the CFU reduction
\cite{CFU1957} treated as
a Fourier integral operator construction
(Remark~\ref{rem:FIO-language}).
Three results are established,
each explicitly conditional on
Assumption~\ref{ass:microlocal}
(uniform A$_2$-stability,
conditions (i)--(vi)):
\begin{enumerate}[\rm(I)]
\item \textit{(Local Airy normal form.)}
  Theorem~\ref{thm:airy-normalform}:
  $\Phi$ admits a microlocal
  representation modulo smoothing
  operators in Airy normal form
  microlocally near the fold point,
  with $a\in S^{-1/2}_{1,0}(c)$
  and rapidly decaying remainder.
\item \textit{(Pointwise decay.)}
  Corollary~\ref{cor:pointwise}:
  $|\Phi(u;c)|\le C(1+|u|)^{-3/4}$
  uniformly.
  The exponent $-3/4$ is the intrinsic
  Airy microlocal exponent of an
  A$_2$-fold with conormal amplitude
  of order $-1/2$; it arises from two
  analytically distinct mechanisms
  unified by a single geometric source
  (Remark~\ref{rem:two-mechanisms}).
\item \textit{(Ces\`aro improvement.)}
  Corollary~\ref{cor:global-bound}:
  $|\int_0^U\Phi|=O(U^{1/4})$.
\end{enumerate}
The forward direction is rigorous
under (i)--(vi);
the converse requires the non-standard
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\maketitle

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
We work throughout in one spatial
dimension ($\beta,u\in\mathbb{R}$).
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold,
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the projection to the right factor.
\medskip

\noindent\textit{Geometric definition.}
$\mathcal{C}$ has an
\emph{A$_2$-fold singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$;
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$.
\end{enumerate}
These are the conditions for a
\emph{stable Lagrange singularity of
A$_2$-type} in Arnol'd's
classification \cite{Arnold1972}.
\medskip

\noindent\textit{Coordinate reduction.}
For the phase
$\Psi(\beta,u)=\Theta(\beta;c)+u\beta$,
conditions (i)--(ii) reduce to
$\Theta''(0;c)=0$,
$\Theta'''(0;c)\ne 0$.
The canonical relation is locally
\[
  \mathcal{C}
  = \bigl\{(\beta,\,\Theta'(\beta;c)+u;\,
  u,\,\beta)\bigr\},
\]
and $\partial_\beta\xi=\Theta''(0;c)=0$
at the fold.
\medskip

\noindent\textit{Normal form.}
By Arnol'd's classification
\cite{Arnold1972},
any oscillatory integral with an A$_2$-fold
admits a microlocal representation
modulo smoothing operators
in Airy normal form near $p_0$
(Golubitsky--Guillemin \cite{Golub1973}).
\end{definition}

\begin{proposition}[Category clarification:
  $I^m$ vs.\ $I^{m,l}$]
\label{prop:category}
Two distinct amplitude classes appear
in this paper:
\begin{enumerate}[\rm(1)]
\item \textit{H\"ormander FIO class $I^m$}
  (H\"ormander \cite{Hormander1983},
  Ch.~25):
  smooth amplitudes only;
  standard stationary-phase theory applies.
\item \textit{Melrose--Uhlmann class
  $I^{m,l}(\Lambda_0,\Lambda_1)$}
  (\cite{MelroseU1979}):
  conormal singularities at the critical
  point; required whenever the amplitude
  is singular at the stationary point.
\end{enumerate}
The CFU map preserves the class
$I^{m,l}(\Lambda_0,\Lambda_1)$
under the canonical transformations
induced by CFU---not the decomposition
into individual classes.
All amplitude stability claims in this
paper are in the $I^{m,l}$ sense.
\end{proposition}

\begin{remark}[FIO equivalence language
  and sign convention]
\label{rem:FIO-language}
\textit{Sign convention.}
We use $\eta=\partial_u\Psi$ throughout.
\medskip

\noindent\textit{FIO equivalence.}
All statements of the form
``admits a microlocal representation
in Airy normal form''
mean:
\emph{the oscillatory integral for
$\Phi$ is microlocally equivalent,
modulo smoothing operators,
to an Airy model integral
microlocally near the fold point,
via a local canonical transformation
$\beta\mapsto t(\beta,u;c)$.}
This is \emph{never} a pointwise
or global phase identity.
The Maslov phase factor is absorbed
into $R(u;c)$ in~\eqref{eq:airy-nf}
(H\"ormander \cite{Hormander1983},
Ch.~25.1).
\end{remark}

\subsection{The fold amplitude and
  conormal singularity}

The fold amplitude is:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u;c)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta;c)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta,
\end{equation}
where $\chi\in C^\infty_c$,
$\alpha_*>0$,
$\theta''(\alpha_*;c)=0$,
$c_3(c):=\frac{1}{6}\theta'''(\alpha_*;c)\ne 0$.
\medskip

\noindent\textbf{Conormal singularity.}
The amplitude $|\beta|^{-1/2}\chi$
belongs to
$I^{-1/4,0}(\Lambda_0,\Lambda_1)$
in the Melrose--Uhlmann sense
(Proposition~\ref{prop:category}),
where $\Lambda_0$ is the zero-section
and $\Lambda_1$ is the conormal to
$\{\beta=0\}$.
The amplitude singularity occurs exactly
at the critical point $\beta=0$;
this is outside the scope of standard
FIO theory and requires the full
$I^{m,l}$ calculus.

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
Under Assumption~\ref{ass:fold-local},
$|\Phi(u;c)|=O(|u|^{-1/2})$
uniformly in $c\ge c_0$.
\end{theorem}

\begin{proof}
$a_0=|\beta|^{-1/2}\chi
\in I^{-1/4,0}(\Lambda_0,\Lambda_1)$
(Melrose--Uhlmann \cite{MelroseU1979};
H\"ormander \cite{Hormander1983},
Ch.~18.1--18.2).
Van~der~Corput in the $I^{m,l}$ setting
gives $O(|u|^{-1/2})$.
\end{proof}

\section{CFU Reduction and Airy Structure}
\label{sec:reduction}

\begin{assumption}[Local stable A$_2$-fold]
\label{ass:fold-local}
$\Theta(\beta;c)=c_3(c)\beta^3+O(\beta^4)$,
$c_3(c)\ne 0$, and:
\begin{enumerate}[\rm(F1)]
\item $\Theta''(\beta;c)\ne 0$
  for $\beta\ne 0$ near $0$.
\item $\int|\beta|^{-1/2}
  e^{i\Theta(\cdot;c)}\chi\,d\beta
  \not\equiv 0$ near $0$.
\item Exactly one critical point
  ($\beta=0$) on $\mathrm{supp}\,\chi$.
\end{enumerate}
\end{assumption}

\begin{lemma}[Degenerate stationary phase
  with conormal weight \Rig]
\label{lem:degenerate-phase}
Define
\begin{equation}\label{eq:model-integral}
  I(\mu) :=
  \int_{\mathbb{R}}
  |t|^{-1/2}\,
  e^{i(\frac{1}{3}t^3 - \mu t)}
  \,dt,
  \qquad \mu\in\mathbb{R},\;|\mu|\ge 1.
\end{equation}
Then $|I(\mu)| \le C\,|\mu|^{-3/4}$.
The exponent $-3/4$ is the
\emph{intrinsic Airy microlocal exponent}
of an A$_2$-fold singularity with
conormal amplitude of order~$-1/2$;
it arises from two analytically distinct
mechanisms unified by a single geometric
source
(Remark~\ref{rem:two-mechanisms}).
\end{lemma}

\begin{proof}
By the $I^{m,l}(\Lambda_0,\Lambda_1)$
calculus of Melrose--Uhlmann
\cite{MelroseU1979}
(Proposition~\ref{prop:category}),
the amplitude $|t|^{-1/2}$
belongs to the intersecting Lagrangian
class, and $I(\mu)$ is microlocally
equivalent, modulo smoothing,
microlocally near the fold point, to
\[
  I(\mu)
  = a(\mu)\,\mathrm{Ai}(\mu) + r(\mu).
\]
Here $a(\mu)$ denotes the CFU-pulled back
amplitude evaluated at the stationary
point in the Airy chart
(i.e.\ the transported $I^{m,l}$-symbol
after the CFU canonical transformation
$t\mapsto\tilde{t}(t,\mu)$,
restricted to $\tilde{t}=0$);
it lies in the pushforward class
under CFU and satisfies
$a\in S^{-1/2}_{1,0}$,
giving $|a(\mu)|\le C'|\mu|^{-1/2}$.
The remainder satisfies
$|r(\mu)|\le C_N|\mu|^{-N}$
for each $N$.

The Airy function satisfies
(Olver \cite{Olver1974}, Ch.~11):
\[
  |\mathrm{Ai}(\mu)|
  \le C\,|\mu|^{-1/4}
  \quad \text{for } |\mu|\ge 1.
\]
Hence:
\[
  |I(\mu)|
  \le |a(\mu)|\cdot|\mathrm{Ai}(\mu)|
  + |r(\mu)|
  \le C'|\mu|^{-1/2}\cdot C|\mu|^{-1/4}
  + C_N|\mu|^{-N}
  = C''|\mu|^{-3/4}.
  \qedhere
\]
\end{proof}

\begin{remark}[Two mechanisms, one geometric source]
\label{rem:two-mechanisms}
The estimate
$|I(\mu)|\lesssim|\mu|^{-3/4}$
arises from two analytically distinct
mechanisms:
\begin{enumerate}[\rm(1)]
\item \textit{Degenerate phase (Airy):}
  the A$_2$-fold contributes
  $|\mathrm{Ai}(\mu)|\lesssim|\mu|^{-1/4}$;
\item \textit{Conormal amplitude transport:}
  $a\in S^{-1/2}$ contributes
  $|a(\mu)|\lesssim|\mu|^{-1/2}$.
\end{enumerate}
These are not independent: both arise
from the same A$_2$-fold geometry
via the CFU reduction.
The fold forces the Airy normal form,
and the $I^{m,l}$-class of the
amplitude is determined by the
conormal structure of the fold.
The $-3/4$ exponent is thus their
\emph{intrinsic combination}
within a single geometric framework,
not a coincidental product of two
separate effects.
\end{remark}

\begin{remark}[Origin of the $-3/4$ exponent]
\label{rem:exponent-origin}
\[
  \underbrace{|a(\mu)|\lesssim|\mu|^{-1/2}}_{%
    \text{conormal amplitude, order }(-1/2)}
  \;\times\;
  \underbrace{|\mathrm{Ai}(\mu)|\lesssim|\mu|^{-1/4}}_{%
    \text{A}_2\text{-fold / Airy decay}}
  \;=\;
  |\mu|^{-3/4}.
\]
Neither factor alone produces $|\mu|^{-3/4}$:
the fold alone (smooth amplitude,
$a\in S^0$) gives $|\mu|^{-1/4}$;
the conormal weight alone
(non-degenerate phase) gives $|\mu|^{-1/2}$
(Theorem~\ref{thm:endpoint}).
\end{remark}

\begin{lemma}[Conormal amplitude
  stability under CFU \Rig]
\label{lem:conormal-stability}
Let $a_0=|\beta|^{-1/2}\chi
\in I^{-1/4,0}(\Lambda_0,\Lambda_1)$.
Under the CFU change of variables
$\beta\mapsto t(\beta,u;c)$,
the transported amplitude
$\tilde{a}(t,u;c)$
preserves the class
$I^{m,l}(\Lambda_0,\Lambda_1)$
under the canonical transformations
induced by CFU,
uniformly in $(u,c)$ under
condition~(vi) of
Assumption~\ref{ass:microlocal}.
\end{lemma}

\begin{proof}
The CFU map is a local canonical
transformation with
$|\det(dt/d\beta)|\ge\delta_5>0$
and uniform symbol bounds (condition (vi)).
The class $I^{m,l}(\Lambda_0,\Lambda_1)$
is preserved under such transformations
(Melrose--Uhlmann \cite{MelroseU1979};
Guillemin--Uhlmann \cite{GuillemU1981},
Lemma~2.1).
\end{proof}

\begin{proposition}[CFU: Airy normal form
  modulo smoothing \Rig]
\label{prop:CFU}
Under Assumption~\ref{ass:fold-local}:
\begin{enumerate}[\rm(1)]
\item $\Theta''(0;c)=0$,
  $\Theta'''(0;c)=6c_3(c)\ne 0$.
\item \textit{(Airy normal form
  modulo smoothing.)}
  The CFU theorem \cite{CFU1957}
  provides $\beta\mapsto t(\beta,u;c)$,
  $\mu(u;c)$, $R(u;c)$ such that:
  \begin{align}\label{eq:CFU-integral}
    \int_{U_\epsilon}
    |\beta|^{-1/2}
    e^{i(\Theta(\beta;c)+u\beta)}
    \phi_{\mathrm{fold}}\,d\beta
    =
    e^{iR(u;c)}
    \bigl[
    a(u;c)\,\mathrm{Ai}(\mu(u;c))
    + \mathcal{R}(u;c)
    \bigr],
  \end{align}
  where $a(u;c)\in S^{-1/2}_{1,0}(c)$
  lies in the pushforward class under CFU
  (i.e.\ the CFU-transported
  $I^{m,l}$-symbol evaluated at the
  fold critical set,
  Lemma~\ref{lem:conormal-stability}),
  and $|\mathcal{R}|\le C_N(1+|u|)^{-N}$.
\item \textit{(Far region.)}
  $|\partial_\beta^2\Psi|\ge c_0'>0$;
  standard stationary phase gives
  $|\Phi_{\mathrm{far}}|\le C|u|^{-1/2}$.
\end{enumerate}
\end{proposition}

\begin{definition}[Parameter-dependent
  symbol class]
\label{def:symbol-class}
\begin{equation}
  S^m_{1,0}(c)
  :=
  \Bigl\{ a(u;c)\in C^\infty(\mathbb{R}_u)
  \;\Big|\;
  |\partial_u^\alpha a(u;c)|
  \le C_\alpha (1+|u|)^{m-|\alpha|},
  \text{ uniformly in } c\ge c_0
  \Bigr\},
\end{equation}
in the sense of semiclassical analysis
(Zworski \cite{Zworski2012}, Ch.~4).
$c$ is not a phase-space variable.
\end{definition}

\section{Main Results}
\label{sec:results}

\begin{assumption}[Uniform A$_2$-stability
  \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$, with:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta;c)
  =c_3(c)\beta^3+O(\beta^4)$
  in a $c$-independent neighbourhood;
\item $c_3(c)\ge\delta_0>0$;
\item $|\Theta''(\beta^*(u);c)|
  \ge\delta_1|u|^{1/2}$;
\item $|\partial_u^k J(t,u;c)|
  \le C_k(1+|u|)^{-1/2-k}$;
\item no secondary stationary points;
\item $|\det(dt/d\beta)|\ge\delta_5>0$;
  $|\partial_\beta^j\partial_u^k
  t|\le C_{jk}$ uniformly.
\end{enumerate}
\end{assumption}

\begin{remark}[Scope of the results]
\label{rem:scope}
Under Assumption~\ref{ass:microlocal},
the main results are local Airy normal
forms microlocally near the fold point,
under strong uniform non-degeneracy
conditions.
The non-trivial content is the
consistent $I^{m,l}$ treatment
of the conormal singularity;
condition~(vi) is the key hypothesis
for PSWF
(Open Problem~\ref{prob:PSWF-microlocal}).
\end{remark}

\begin{theorem}[Conditional local Airy
  normal form \Rig]
\label{thm:airy-normalform}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
For $|u|\ge\delta_2$,
$\Phi(u;c)$ admits a microlocal
representation modulo smoothing
operators in Airy normal form
microlocally near the fold point:
\begin{equation}\label{eq:airy-nf}
  e^{i(\alpha_*u+R(u;c))}\bigl[
  a(u;c)\,
  \mathrm{Ai}\!\bigl(\mu(u;c)\bigr)
  + r(u;c)\bigr],
\end{equation}
where $a\in S^{-1/2}_{1,0}(c)$
lies in the pushforward class under CFU,
$|r|\le C_N(1+|u|)^{-N}$,
and $R$ absorbs the Maslov factor.
\end{theorem}

\begin{proof}[Proof sketch]
Fold zone:
Proposition~\ref{prop:CFU}(2)
+ Lemma~\ref{lem:conormal-stability}.
Far zone:
Proposition~\ref{prop:CFU}(3).
\end{proof}

\begin{remark}[Decay mechanism]
\label{rem:mechanism}
The pointwise bound
$|\Phi(u;c)|\lesssim|u|^{-3/4}$
has a single geometric source
(Remarks~\ref{rem:two-mechanisms},
\ref{rem:exponent-origin}):
\[
  \text{A}_2\text{-fold}
  +
  \text{conormal amplitude }(-1/2)
  \;\Longrightarrow\;
  |\Phi|\lesssim|u|^{-3/4}.
\]
\end{remark}

\begin{corollary}[Pointwise decay \Cond]
\label{cor:pointwise}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
\begin{equation}
  |\Phi(u;c)|\le C(1+|u|)^{-3/4}
  \quad\forall u\in\mathbb{R},
\end{equation}
uniformly in $c\ge c_0$.
\end{corollary}

\begin{proof}
Fold zone: Theorem~\ref{thm:airy-normalform}
+ Lemma~\ref{lem:degenerate-phase}
(Remark~\ref{rem:exponent-origin}).
Compact zone $|u|\le 1$: continuity.
Far zone: Theorem~\ref{thm:endpoint}.
\end{proof}

\begin{remark}[Non-oscillatory mechanism]
\label{rem:oscillation}
The Ces\`aro bound uses only
$|\Phi|\le C(1+|u|)^{-3/4}$
and $L^1$-integration ---
a non-oscillatory mechanism
(pure integrability, no cancellation).
\end{remark}

\begin{corollary}[Ces\`aro improvement
  \Cond]
\label{cor:global-bound}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
\[
  \left|\int_0^U\Phi(u;c)\,du\right|
  = O(U^{1/4}),
\]
improving upon $O(U^{1/2})$
by Remark~\ref{rem:oscillation}.
\end{corollary}

\subsection{Converse and FIO rigidity}

\begin{assumption}[Microlocal
  reconstruction (non-standard) \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ is in the Airy class and:
(a) $\sigma(g\mapsto\Phi)|_\mathcal{C}\ne 0$;
(b) $g\mapsto\Phi$ injective mod smoothing;
(c) $\mathrm{WF}(\Phi)
  \subseteq\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
Then $\mathcal{C}$ has an A$_2$-fold
uniformly in $c$.
This is an inverse microlocal conjecture
(not a consequence of standard FIO theory
\cite{Zworski2012,Hormander1983});
verification for PSWF: Paper~XVII.
\end{assumption}

\begin{corollary}[Conditional equivalence]
\label{cor:conditional-equiv}
Under Assumption~\ref{hyp:FIO-rigidity}:
Airy normal form $\Rightarrow$
Assumption~\ref{ass:microlocal}
(modulo the non-standard hypothesis).
The forward direction holds rigorously
without Assumption~\ref{hyp:FIO-rigidity}.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item $c_3(c)\ge\delta_0>0$:
  expected from Olver \cite{Olver1974};
  uniform remainder bounds required.
\item $\theta''\ne 0$ for
  $\alpha\ne\alpha_*$, uniformly:
  plausible but not established.
\item Condition~(vi) for the
  PSWF CFU Jacobian:
  technically hardest;
  no uniform result currently available.
\end{enumerate}
These conditions are partially
supported by WKB asymptotics,
but no uniform microlocal analysis
is currently available in the literature.
\end{problem}
