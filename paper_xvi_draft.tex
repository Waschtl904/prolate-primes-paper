\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases, A$_2$-Fold Singularities,\\
  and the Airy Normal Form}

\author{[Author]}

\begin{abstract}
A cubic WKB phase $c_3(c)\beta^3$
produces a degenerate stationary point
of order~2, which is an A$_2$-fold
singularity of the canonical relation
$\mathcal{C}$.
The Airy distribution is the universal
local normal form of this singularity.
Three results are established,
each explicitly conditional on
Assumption~\ref{ass:microlocal}
(uniform A$_2$-stability,
conditions (i)--(vi)):
\begin{enumerate}[\rm(I)]
\item \textit{(Local Airy normal form.)}
  Theorem~\ref{thm:airy-normalform}:
  $\Phi$ is microlocally equivalent,
  after a local canonical transformation,
  to
  $e^{i\alpha_*u}[a(u,c)\,
  \mathrm{Ai}(c_3^{-1/3}u)+r(u,c)]$,
  with $a\in S^{-1/2}_{1,0}$ and
  $r$ rapidly decaying
  (both conditional on (i)--(vi)).
\item \textit{(Pointwise decay.)}
  Corollary~\ref{cor:pointwise}:
  $|\Phi(u)|\le C(1+|u|)^{-3/4}$
  uniformly for all $u$
  (conditional on (i)--(vi)).
\item \textit{(Ces\`aro improvement.)}
  Corollary~\ref{cor:global-bound}:
  $|\int_0^U\Phi\,du|=O(U^{1/4})$,
  conditional on (i)--(vi).
\end{enumerate}
The forward direction is rigorous
under (i)--(vi);
the converse requires the non-standard
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\maketitle

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold,
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the projection to the right factor.
\medskip

\noindent\textit{Geometric definition.}
$\mathcal{C}$ has an
\emph{A$_2$-fold singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$;
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$.
\end{enumerate}
This is the A$_2$-type in Arnold's
classification \cite{Arnold1972}.
\medskip

\noindent\textit{Coordinate reduction.}
For phase $\Psi(\beta,u)
=\Theta(\beta;c)+u\beta$
(where $c$ is the parameter
and $\Theta(\beta;c)
=\theta(\alpha_*+\beta;c)$
carries the $c$-dependence
explicitly),
conditions (i)--(ii) reduce to
$\Theta''(0;c)=0$,
$\Theta'''(0;c)\ne 0$.
These are coordinate expressions;
conditions (i)--(ii) are invariant
under canonical transformations.
By the Thom--Boardman theorem
\cite{Golub1973},
any FIO with an A$_2$-fold is,
\emph{microlocally near $p_0$},
microlocally equivalent to
$\mathrm{Ai}$
after a local symplectic change;
this is a local-in-phase-space statement.
\end{definition}

\begin{remark}[Microlocal transport]
\label{rem:microlocal-transport}
The Fourier transform acts by the
exchange
$(\beta,\xi;u,\eta)\mapsto(u,\eta;\beta,\xi)$.
If the Lagrangian map germ of $\pi_u$
belongs to the stable A$_2$-class
(Mather stability \cite{Golub1973}),
then the germ of $\pi_\beta$ also
belongs to the stable A$_2$-class.
This is a local statement;
it fails if $c_3(c)\to 0$.
\end{remark}

\subsection{The fold amplitude}

The fold amplitude is defined as:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u;c)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta;c)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta,
\end{equation}
where $\chi\in C^\infty_c$,
$\alpha_*>0$,
$\theta''(\alpha_*;c)=0$,
$c_3(c):=\frac{1}{6}\theta'''(\alpha_*;c)\ne 0$,
and $\Theta(\beta;c):=\theta(\alpha_*+\beta;c)$
(the $c$-dependence of $\Theta$ is
carried throughout).
\medskip

\noindent\textbf{Interpretation of the
singular weight.}
The weight $|\beta|^{-1/2}$ is not
locally integrable at $\beta=0$.
The integral~\eqref{eq:Phi-WKB}
is therefore interpreted as a
\emph{distributional oscillatory
integral} whose amplitude
$|\beta|^{-1/2}\chi(\alpha_*+\beta)$
lies in the class of
\emph{conormal distributions}
with singularity at $\beta=0$
(cf.\ H\"ormander \cite{Hormander1983},
Ch.~18.2):
$|\beta|^{-1/2}\in I^{m}(\mathbb{R},
\{0\})$ with $m=-1/4$.
All subsequent stationary-phase
and FIO operations are
performed in this distributional
sense.

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
$|\Phi(u;c)|=O(|u|^{-1/2})$
for all $\theta\in C^\infty$,
uniformly in $c$.
\end{theorem}

\begin{proof}
The amplitude $|\beta|^{-1/2}\chi$
has a conormal singularity at $\beta=0$.
For $\theta\equiv 0$,
Erd\'elyi \cite{Erdelyi1956} gives
the exact asymptotics.
For general $\theta$,
van~der~Corput's lemma
(with degenerate amplitude
$|\beta|^{-1/2}$, cf.\
H\"ormander \cite{Hormander1983},
Ch.~7.7 applied distributionally)
gives $O(|u|^{-1/2})$.
\end{proof}

\section{CFU Reduction and Symbol Stability}
\label{sec:reduction}

\begin{assumption}[Local stable A$_2$-fold]
\label{ass:fold-local}
$\Theta(\beta;c)=c_3(c)\beta^3+O(\beta^4)$,
$c_3(c)\ne 0$, and:
\begin{enumerate}[\rm(F1)]
\item $\Theta''(\beta;c)\ne 0$
  for $\beta\ne 0$ near $0$;
  no cusp or higher stratum.
\item $\int|\beta|^{-1/2}
  e^{i\Theta(\cdot;c)}\chi\,d\beta
  \not\equiv 0$ near $0$.
\item $\partial_\beta\Theta(\beta;c)$
  has exactly one critical point
  ($\beta=0$) on $\mathrm{supp}\,\chi$;
  no secondary stationary points.
\end{enumerate}
\end{assumption}

\begin{proposition}[CFU as FIO
  conjugation \Rig]
\label{prop:CFU}
Under Assumption~\ref{ass:fold-local},
fix $U_\epsilon=\{|\beta|<\epsilon\}$
and a smooth partition
$1=\phi_{\mathrm{fold}}
+\phi_{\mathrm{far}}$,
$\phi_{\mathrm{fold}}\equiv 1$
on $U_{\epsilon/2}$.
\begin{enumerate}[\rm(1)]
\item \textit{(A$_2$-fold.)}
  $\Theta''(0;c)=0$,
  $\Theta'''(0;c)=6c_3(c)\ne 0$;
  (F1) confirms A$_2$-type.
\item \textit{(FIO conjugation to
  Airy normal form.)}
  In $U_\epsilon$, the CFU theorem
  \cite{CFU1957} guarantees a
  smooth local map
  $\beta\mapsto t(\beta,u;c)$
  --- a \emph{uniform local
  parametrization} of the fold
  (not global on $\mathrm{supp}\,\chi$)
  --- and smooth functions
  $\mu(u;c)=c_3(c)^{-1/3}u+O(u^2)$
  and $R(u;c)$, such that
  the phase is
  \emph{microlocally equivalent,
  after the local canonical
  transformation $\beta\mapsto t$,
  to the Airy normal form}
  $\frac{1}{3}t^3-\mu(u;c)\,t$;
  the remainder is a smooth
  symbol class function absorbed
  entirely into the transported
  amplitude $\tilde{a}(t,u;c)$.
  Concretely, for the oscillatory
  integral (not pointwise in phase):
  \begin{align}\label{eq:CFU-integral}
    &\int_{U_\epsilon}
    |\beta|^{-1/2}
    e^{i(\Theta(\beta;c)+u\beta)}
    \phi_{\mathrm{fold}}\,d\beta
    \\
    &\quad=
    e^{iR(u;c)}
    \int_{\mathbb{R}}
    \tilde{a}(t,u;c)\,
    e^{i(\frac{1}{3}t^3-\mu(u;c)\,t)}
    \,dt,\notag
  \end{align}
  where $\tilde{a}(t,u;c)
  =|\beta(t,u;c)|^{-1/2}
  J(t,u;c)\tilde\chi(t,u;c)$
  and $J(t,u;c):=d\beta/dt$.
  This is an \emph{exact identity
  for the oscillatory integral};
  it is not a pointwise phase identity.
\item \textit{(Far region.)}
  On $\mathrm{supp}\,\phi_{\mathrm{far}}$,
  $\Theta''(\beta;c)\ne 0$;
  van~der~Corput gives
  $|\Phi_{\mathrm{far}}(u;c)|
  \le C|u|^{-1/2}$ uniformly.
\end{enumerate}
\end{proposition}

\begin{remark}[Phase equivalence,
  not identity]
\label{rem:CFU-FIO}
Equation~\eqref{eq:CFU-integral}
is an exact identity for the
\emph{oscillatory integral}
via FIO conjugation,
not the pointwise relation
$\Theta(\beta;c)+u\beta
\equiv \frac{1}{3}t^3-\mu t+R$.
The phase $\Theta(\beta;c)+u\beta$
is \emph{microlocally equivalent}
to $\frac{1}{3}t^3-\mu(u;c)t$
\emph{after the canonical
transformation} $\beta\mapsto t$;
all Taylor remainders from
the substitution are absorbed
into $\tilde{a}$, not kept
as separate phase terms.
This is the standard FIO
interpretation of CFU
\cite{CFU1957,Olver1974}.
\end{remark}

\begin{lemma}[Jacobian rescaling \Rig]
\label{lem:Jacobian-rescaling}
Under the canonical Airy rescaling
$t = c_3(c)^{-1/6}|u|^{1/2}\tilde{t}$,
the Jacobian transforms as:
\begin{equation}\label{eq:Jacobi-homog}
  J(t,u;c)
  = c_3(c)^{-1/6}|u|^{-1/2}
  \tilde{J}(\tilde{t},u;c),
  \quad
  \tilde{J} = 1+O(|u|^{-1})
  \text{ uniformly.}
\end{equation}
The $|u|^{-1/2}$ factor arises from
the homogeneity of the cubic
normal form under Airy rescaling
(not from $|\beta|^{-1/2}$).
\end{lemma}

\begin{proof}
Under $t=c_3(c)^{-1/6}|u|^{1/2}\tilde{t}$,
the Airy phase
$\frac{1}{3}t^3-\mu(u;c)t$
becomes
$c_3(c)^{-1/2}|u|^{3/2}
(\frac{1}{3}\tilde{t}^3
-\mathrm{sgn}(u)\tilde{t})$.
From $J\,dt=d\beta$:
$J = \tilde{J}\cdot c_3(c)^{1/6}
|u|^{-1/2}$;
$\tilde{J}=d\beta/d\tilde{t}
=1+O(|u|^{-1})$
by CFU expansion \cite{CFU1957}.
\end{proof}

\begin{lemma}[Symbol stability \Rig]
\label{lem:symbol-stability}
Under Assumption~\ref{ass:microlocal},
let
$\tilde{a}(\tilde{t},u;c)
:=|\beta(\tilde{t},u;c)|^{-1/2}
\tilde{J}(\tilde{t},u;c)
\cdot\tilde\chi(\tilde{t},u;c)$.
\begin{enumerate}[\rm(1)]
\item \textit{(Amplitude class.)}
  Under condition (vi),
  $|\partial_{\tilde{t}}^j
  \partial_u^k\tilde{a}|
  \le C_{jk}(1+|u|)^{-1/2-k}$
  uniformly in $(\tilde{t},u,c)$.
\item \textit{(Parametric stationary
  phase.)}
  By H\"ormander \cite{Hormander1983},
  Ch.~7.7:
  \begin{equation}\label{eq:airy-decomp}
    \int\tilde{a}(\tilde{t},u;c)\,
    e^{ic_3(c)^{-1/2}|u|^{3/2}
    (\tilde{t}^3/3
    -\mathrm{sgn}(u)\tilde{t})}
    \,d\tilde{t}
    = a_0(u;c)\,
    \mathrm{Ai}(c_3(c)^{-1/3}u)
    + r_0(u;c).
  \end{equation}
  The coefficient $a_0$ is extracted
  from the stationary-phase expansion
  at the cubic critical point;
  it is \emph{not} defined by
  dividing by $\mathrm{Ai}$.
\item \textit{(Estimates.)}
  Under condition (vi):
  \begin{align}
    \label{eq:symbol-u}
    |\partial_u^j a_0(u;c)|
    &\le C_j(1+|u|)^{-1/2-j},\\
    \label{eq:remainder-u}
    |r_0(u;c)|
    &\le C_{N}(1+|u|)^{-N}
    \quad\text{(each fixed }N).
  \end{align}
  Hence $a_0\in S^{-1/2}_{1,0}$
  in $u$, uniform in $c$.
\end{enumerate}
\end{lemma}

\begin{remark}[Remainder uniformity]
\label{rem:remainder}
Estimate~\eqref{eq:remainder-u}
holds uniform in $c\ge c_0$
\emph{only under} condition (vi)
(uniform non-degeneracy of the
non-stationary region).
Without (vi), rapid decay holds
only for each fixed $c$.
\end{remark}

\section{Main Results}
\label{sec:results}

\begin{assumption}[Uniform A$_2$-stability
  \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$, with:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta;c)
  =c_3(c)\beta^3+O(\beta^4)$
  in a $c$-independent neighbourhood;
\item $c_3(c)\ge\delta_0>0$;
\item $|\Theta''(\beta^*(u);c)|
  \ge\delta_1|u|^{1/2}$
  for $\delta_2\le|u|\le c^{\theta_0/3}$;
\item $|\partial_u^k J(t,u;c)|
  \le C_k(1+|u|)^{-1/2-k}$
  uniformly for all $k$;
\item no secondary stationary points
  uniformly in $c$;
\item \textit{(Uniform symbol-class
  regularity of the CFU map.)}
  Uniformly in $(\beta,u,c)$:
  $|\det(dt/d\beta)|\ge\delta_5>0$;
  $|\partial_\beta^j\partial_u^k
  t(\beta,u;c)|\le C_{jk}$
  for all $j,k\ge 0$;
  and $|\partial_{\tilde{t}}
  (\tilde{t}^3/3
  -\mathrm{sgn}(u)\tilde{t})|
  \ge\delta_4>0$
  outside the Airy critical region.
\end{enumerate}
\end{assumption}

\begin{remark}[Role of (vi)]
Condition (vi) is what promotes
symbol estimates from
``fixed $c$'' to
``uniform in $c\ge c_0$''.
Without it, all claims marked \Rig\ hold
for each fixed $c$ only.
Verifying (vi) for PSWF:
Open Problem~\ref{prob:PSWF-microlocal}.
\end{remark}

\begin{theorem}[Conditional local
  Airy normal form \Rig]
\label{thm:airy-normalform}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
For $\delta_2\le|u|\le c^{\theta_0/3}$,
the amplitude $\Phi(u;c)$
is microlocally equivalent,
after a local canonical transformation,
to:
\begin{equation}\label{eq:airy-nf}
  \Phi(u;c)
  = e^{i\alpha_*u}\bigl[
  a(u;c)\,
  \mathrm{Ai}\!\bigl(c_3(c)^{-1/3}u\bigr)
  + r(u;c)\bigr],
\end{equation}
where $a(u;c)\in S^{-1/2}_{1,0}$
(uniform in $c$, requires (vi))
and $|r(u;c)|\le C_N(1+|u|)^{-N}$
for each fixed $N$
(requires (vi)).
The oscillatory structure of $\Phi$
is entirely encoded in
$\mathrm{Ai}(c_3(c)^{-1/3}u)$;
$a$ is a smooth amplitude modulation.
\end{theorem}

\begin{proof}[Proof sketch]
Fold zone $U_\epsilon$:
Proposition~\ref{prop:CFU}(2)
(exact identity for the
oscillatory integral via FIO
conjugation; not a pointwise
phase identity);
Lemma~\ref{lem:Jacobian-rescaling}
(Airy rescaling of $J$);
Lemma~\ref{lem:symbol-stability}
(amplitude class in $(\tilde{t},u)$
$\to$ symbol in $u$
via H\"ormander Ch.~7.7,
under conditions (i)--(vi)).
Far zone:
Proposition~\ref{prop:CFU}(3).
\end{proof}

\begin{lemma}[Airy-symbol estimate
  in fold zone \Rig]
\label{lem:product-estimate}
\textit{Valid in the fold zone
after canonical Airy rescaling;
not a global statement.}
Under Theorem~\ref{thm:airy-normalform},
for $|u|\ge\delta_2$:
\begin{enumerate}[\rm(a)]
\item $u\ge\delta_2$:
  $|a\cdot\mathrm{Ai}(c_3^{-1/3}u)|
  \le C_1C_2\,u^{-3/4}$
  (Airy: $C_1 u^{-1/4}e^{-c'u^{3/2}}$;
  symbol: $C_2 u^{-1/2}$).
\item $u\le-\delta_2$:
  $|a\cdot\mathrm{Ai}(c_3^{-1/3}u)|
  \le C_1C_2\,|u|^{-3/4}$
  (Airy oscillatory:
  $C_1|u|^{-1/4}$;
  symbol: $C_2|u|^{-1/2}$).
\end{enumerate}
The exponent $-3/4$ is the
\emph{joint scaling factor
of the full asymptotic expansion}
after CFU reduction and Airy rescaling:
the factors $|u|^{-1/4}$ and $|u|^{-1/2}$
arise from a single stationary-phase
expansion in $\tilde{t}$,
not from two independent decay
mechanisms multiplied post hoc.
\end{lemma}

\begin{corollary}[Pointwise decay
  \Cond]
\label{cor:pointwise}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
There exists $C>0$,
uniform in $c\ge c_0$
and across all three zones, such that:
\begin{equation}\label{eq:pointwise}
  |\Phi(u;c)|\le C(1+|u|)^{-3/4}
  \quad\text{for all }u\in\mathbb{R}.
\end{equation}
\end{corollary}

\begin{proof}
$C:=\max(C_1C_2,\,C_3,\,C_4)$
from three zones:
\begin{itemize}
\item \textit{Fold zone $|u|\ge\delta_2$:}
  Lemma~\ref{lem:product-estimate}.
\item \textit{Compact zone $|u|\le 1$:}
  by compactness and continuity
  of the FIO amplitude
  (Assumption (vi)),
  $|\Phi|\le C_3$ uniformly
  (Erd\'elyi \cite{Erdelyi1956});
  trivially $\le C_3(1+|u|)^{-3/4}$.
\item \textit{Far zone:}
  $|\Phi|\le C_4|u|^{-1/2}
  \le C_4|u|^{-3/4}$
  for $|u|\ge 1$
  (Theorem~\ref{thm:endpoint}).
\end{itemize}
Transition region: conditions
(iii)--(iv) make the constants
compatible.
\end{proof}

\begin{remark}[Oscillation]
\label{rem:oscillation}
For $u<0$,
the $|u|^{-1/4}$ Airy decay
is a consequence of stationary phase
within the Airy representation
(Theorem~\ref{thm:airy-normalform}).
Once~\eqref{eq:pointwise} holds,
the $u$-integral in
Corollary~\ref{cor:global-bound}
is estimated by absolute value only.
Obtaining oscillatory improvement
would require a second
stationary-phase step in $u$;
that is beyond the scope here.
\end{remark}

\begin{corollary}[Ces\`aro improvement
  \Cond]
\label{cor:global-bound}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
\[
  \left|\int_0^U\Phi(u;c)\,du\right|
  \le C\int_0^U(1+u)^{-3/4}\,du
  = O(U^{1/4}),
\]
improving upon Ces\`aro $O(U^{1/2})$.
The exponent $-3/4$ arises as the
joint scaling factor of the
full asymptotic expansion
(Lemma~\ref{lem:product-estimate});
the single constant $C$ of
Corollary~\ref{cor:pointwise}
ensures global stability.
\end{corollary}

\subsection{Converse and FIO rigidity}

\begin{assumption}[Microlocal
  reconstruction (non-standard)
  \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ is in the Airy class and:
\begin{enumerate}[\rm(a)]
\item $\sigma(g\mapsto\Phi)
  \big|_\mathcal{C}\ne 0$;
\item $g\mapsto\Phi$ is injective
  modulo smoothing;
\item $\mathrm{WF}(\Phi)
  \subseteq\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
\end{enumerate}
Then $\mathcal{C}$ contains an
A$_2$-fold singularity
uniformly in $c$.
\medskip

\noindent\textit{Status.}
This is a
\emph{non-standard microlocal
reconstruction hypothesis},
not a consequence of standard FIO
theory \cite{Zworski2012,Hormander1983}.
Verification for PSWF: Paper~XVII.
\end{assumption}

\begin{corollary}[Conditional
  equivalence]
\label{cor:conditional-equiv}
Under Assumption~\ref{hyp:FIO-rigidity}:
Airy-class bound $\Rightarrow$
Assumption~\ref{ass:microlocal}
(modulo the non-standard hypothesis).
The forward direction holds
rigorously without
Assumption~\ref{hyp:FIO-rigidity}.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha;c)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha;c)}$
satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item $c_3(c)\ge\delta_0>0$?
  (Olver turning-point WKB
  \cite{Olver1974}.)
\item $\theta''(\alpha;c)\ne 0$
  for $\alpha\ne\alpha_*$, uniformly?
\item Conditions (iv) and (vi)
  for the PSWF CFU Jacobian?
\end{enumerate}
Outcomes: (A)~Assumptions hold:
$|\int_0^U\Phi|=O(U^{1/4})$.
(B)~$c_3(c)\to 0$:
Ces\`aro sharp.
\end{problem}
