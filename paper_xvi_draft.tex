\title[Lagrangian Singularity Transport under Fourier Duality]{%
  Lagrangian Singularity Transport under Fourier Duality:\\
  Cubic WKB Phases, A$_2$-Fold Singularities,\\
  and the Airy Normal Form}

\author{[Author]}

\begin{abstract}
This paper works in the \emph{one-dimensional setting} throughout
($\beta,u\in\mathbb{R}$).
A cubic WKB phase $c_3(c)\beta^3$
produces a degenerate stationary point
of order~2, which is an A$_2$-fold
singularity of the canonical relation
$\mathcal{C}$ in the sense of
Arnol'd's classification
\cite{Arnold1972}.
The Airy distribution is the universal
local normal form of this singularity,
via the CFU reduction
\cite{CFU1957} treated as
a Fourier integral operator construction
(Remark~\ref{rem:FIO-language}).
Three results are established,
each explicitly conditional on
Assumption~\ref{ass:microlocal}
(uniform A$_2$-stability,
conditions (i)--(vi)):
\begin{enumerate}[\rm(I)]
\item \textit{(Local Airy normal form.)}
  Theorem~\ref{thm:airy-normalform}:
  $\Phi$ admits a microlocal
  representation modulo smoothing
  operators in Airy normal form,
  with $a\in S^{-1/2}_{1,0}(c)$
  (Definition~\ref{def:symbol-class})
  and rapidly decaying remainder.
\item \textit{(Pointwise decay.)}
  Corollary~\ref{cor:pointwise}:
  $|\Phi(u;c)|\le C(1+|u|)^{-3/4}$
  uniformly, via a non-oscillatory
  mechanism (Remark~\ref{rem:oscillation}).
\item \textit{(Ces\`aro improvement.)}
  Corollary~\ref{cor:global-bound}:
  $|\int_0^U\Phi|=O(U^{1/4})$.
\end{enumerate}
The forward direction is rigorous
under (i)--(vi);
the converse requires the non-standard
Hypothesis~\ref{hyp:FIO-rigidity}.
\end{abstract}

\maketitle

\section{Setup and Definitions}
\label{sec:setup}

\begin{definition}[A$_2$-fold singularity]
\label{def:A2-class}
We work throughout in one spatial
dimension ($\beta,u\in\mathbb{R}$).
Let $\mathcal{C}\subset T^*\mathbb{R}_\beta
\times T^*\mathbb{R}_u$ be a Lagrangian
submanifold,
$\pi_u:\mathcal{C}\to T^*\mathbb{R}_u$
the projection to the right factor.
\medskip

\noindent\textit{Geometric definition.}
$\mathcal{C}$ has an
\emph{A$_2$-fold singularity at $p_0$} if:
\begin{enumerate}[\rm(i)]
\item $\det\,d\pi_u\big|_{p_0}=0$;
\item $d(\det\,d\pi_u)\big|_{p_0}\ne 0$.
\end{enumerate}
These are the conditions for a
\emph{stable Lagrange singularity of
A$_2$-type} in Arnol'd's
classification \cite{Arnold1972}
(rank-1 degeneration of the
Lagrangian projection,
with non-degenerate unfolding).
\medskip

\noindent\textit{Coordinate reduction.}
For the phase
$\Psi(\beta,u)=\Theta(\beta;c)+u\beta$,
conditions (i)--(ii) reduce to
$\Theta''(0;c)=0$,
$\Theta'''(0;c)\ne 0$.
These are coordinate expressions;
conditions (i)--(ii) are invariant
under canonical transformations.
The associated canonical relation is
locally given by
\[
  \mathcal{C}
  = \bigl\{(\beta,\xi;\,u,\eta)\,:\,
  \xi=\partial_\beta\Psi(\beta,u;c),\;
  \eta=\partial_u\Psi(\beta,u;c)\bigr\}
  = \bigl\{(\beta,\,\Theta'(\beta;c)+u;\,
  u,\,\beta)\bigr\}.
\]
At the fold point $\beta=0$,
$\partial_\beta\xi
=\Theta''(0;c)=0$,
confirming the degeneration of $\pi_u$.
\medskip

\noindent\textit{Normal form.}
By Arnol'd's classification
\cite{Arnold1972},
any oscillatory integral with an A$_2$-fold
\emph{admits a microlocal representation
modulo smoothing operators
in Airy normal form}
near $p_0$.
This is a local-in-phase-space statement;
it does not give a global or pointwise
identity.
The Golubitsky--Guillemin framework
\cite{Golub1973} provides the
coordinate-free stability conditions.
\end{definition}

\begin{remark}[FIO equivalence language
  and sign convention]
\label{rem:FIO-language}
\textit{Sign convention.}
We use $\eta=\partial_u\Psi$
throughout;
the Fourier transform corresponds to
the symplectic involution
$(\beta,\xi;u,\eta)
\mapsto(u,-\eta;\beta,-\xi)$.
\medskip

\noindent\textit{FIO equivalence.}
All statements of the form
``admits a microlocal representation
in Airy normal form''
mean precisely:
\emph{the oscillatory integral for
$\Phi$ is microlocally equivalent,
modulo smoothing operators,
to an Airy model integral
in the sense of Fourier integral
operators with canonical relation
equivalent to the Airy Lagrangian,
via a local canonical transformation
$\beta\mapsto t(\beta,u;c)$.}
This is \emph{never} a pointwise
phase identity
$\Theta(\beta;c)+u\beta
=\frac{1}{3}t^3-\mu t+R$.
The Maslov phase factor is absorbed
into $R(u;c)$ in~\eqref{eq:airy-nf}
(H\"ormander \cite{Hormander1983},
Ch.~25.1).
\end{remark}

\begin{remark}[Microlocal transport]
\label{rem:microlocal-transport}
The Fourier transform acts by the
symplectic involution of
Remark~\ref{rem:FIO-language}.
If the Lagrangian map germ of $\pi_u$
belongs to the stable A$_2$-class,
then the germ of $\pi_\beta$ also does.
This is a local statement;
it fails if $c_3(c)\to 0$.
\end{remark}

\subsection{The fold amplitude and
  conormal singularity}

The fold amplitude is:
\begin{equation}\label{eq:Phi-WKB}
  \Phi(u;c)
  = e^{i\alpha_* u}
  \int_{\mathbb{R}}
  |\beta|^{-1/2}
  e^{i(\Theta(\beta;c)+\beta u)}
  \chi(\alpha_*+\beta)
  \,d\beta,
\end{equation}
where $\chi\in C^\infty_c$,
$\alpha_*>0$,
$\theta''(\alpha_*;c)=0$,
$c_3(c):=\frac{1}{6}\theta'''(\alpha_*;c)\ne 0$,
$\Theta(\beta;c):=\theta(\alpha_*+\beta;c)$.
\medskip

\noindent\textbf{Conormal singularity
and non-standard stationary phase.}
The weight $|\beta|^{-1/2}$ is not
locally integrable at $\beta=0$;
the integral~\eqref{eq:Phi-WKB} is
interpreted as a distributional
oscillatory integral with
$|\beta|^{-1/2}\chi
\in I^{-1/4}(\mathbb{R},\{0\})$
(H\"ormander \cite{Hormander1983},
Ch.~18.2;
Guillemin--Uhlmann \cite{GuillemU1981}).
The amplitude singularity occurs
\emph{exactly at the critical point}
$\beta=0$; this is not covered by
standard stationary-phase theory.
All operations are performed within
the Melrose--Uhlmann framework
\cite{MelroseU1979}.

\begin{theorem}[Endpoint decay \Rig]
\label{thm:endpoint}
Under Assumption~\ref{ass:fold-local},
$|\Phi(u;c)|=O(|u|^{-1/2})$
uniformly in $c\ge c_0$.
\end{theorem}

\begin{proof}
$a_0=|\beta|^{-1/2}\chi
\in I^{-1/4}(\mathbb{R},\{0\})$;
all operations are distributional
(Melrose--Uhlmann \cite{MelroseU1979};
H\"ormander \cite{Hormander1983},
Ch.~18.1--18.2).
van~der~Corput in the distributional
setting
(H\"ormander \cite{Hormander1983},
Ch.~7.7 + 18.1)
gives $O(|u|^{-1/2})$,
using $\Theta''(\beta^*;c)
\ge\delta_1|u|^{1/2}$
(Assumption~\ref{ass:microlocal}(iii)).
\end{proof}

\section{CFU Reduction, Conormal Stability,
  and Symbol Classes}
\label{sec:reduction}

\begin{assumption}[Local stable A$_2$-fold]
\label{ass:fold-local}
$\Theta(\beta;c)=c_3(c)\beta^3+O(\beta^4)$,
$c_3(c)\ne 0$, and:
\begin{enumerate}[\rm(F1)]
\item $\Theta''(\beta;c)\ne 0$
  for $\beta\ne 0$ near $0$.
\item $\int|\beta|^{-1/2}
  e^{i\Theta(\cdot;c)}\chi\,d\beta
  \not\equiv 0$ near $0$.
\item Exactly one critical point
  ($\beta=0$) on $\mathrm{supp}\,\chi$.
\end{enumerate}
\end{assumption}

\begin{lemma}[Degenerate stationary phase
  with conormal weight \Rig]
\label{lem:degenerate-phase}
Define
\begin{equation}\label{eq:model-integral}
  I(\mu) :=
  \int_{\mathbb{R}}
  |t|^{-1/2}\,
  e^{i(\frac{1}{3}t^3 - \mu t)}
  \,dt,
  \qquad \mu\in\mathbb{R},\;|\mu|\ge 1.
\end{equation}
Then $|I(\mu)| \le C\,|\mu|^{-3/4}$,
with $C>0$ independent of $\mu$.
\end{lemma}

\begin{proof}
Split:
\[
  I(\mu) = I_{\mathrm{near}}(\mu)
  + I_{\mathrm{far}}(\mu),
  \quad
  I_{\mathrm{near}} := \int_{|t|\le 1},
  \quad
  I_{\mathrm{far}} := \int_{|t|>1}.
\]

\medskip
\noindent\textit{Step 1: Near region $|t|\le 1$.}
On $|t|\le 1$, the amplitude
$|t|^{-1/2}\in L^1$ is integrable.
The phase $\phi(t)=\frac{1}{3}t^3-\mu t$
has $\phi'''(t)=2\ne 0$,
so van~der~Corput of order~3
(H\"ormander \cite{Hormander1983},
Ch.~7.7) gives:
\[
  |I_{\mathrm{near}}(\mu)|
  \le C\,|\mu|^{-1/3}
  \le C\,|\mu|^{-1/3}.
\]

\medskip
\noindent\textit{Step 2: Far region $|t|>1$,
  Airy rescaling.}
Set $t=|\mu|^{1/2}\tilde{t}$,
so $dt=|\mu|^{1/2}d\tilde{t}$
and $|t|^{-1/2}=|\mu|^{-1/4}|\tilde{t}|^{-1/2}$.
The exponent becomes
$i|\mu|^{3/2}\psi_\pm(\tilde{t})$
with
$\psi_\pm(\tilde{t})
=\frac{1}{3}\tilde{t}^3
\mp\tilde{t}$
($\mp$ for $\mathrm{sgn}(\mu)=\pm 1$).
Collecting:
\[
  I_{\mathrm{far}}(\mu)
  = |\mu|^{1/4}
  \int_{|\tilde{t}|>|\mu|^{-1/2}}
  |\tilde{t}|^{-1/2}\,
  e^{i|\mu|^{3/2}\psi_\pm(\tilde{t})}
  \,d\tilde{t}.
\]

\medskip
\noindent\textit{Step 3: Stationary phase
  at $\tilde{t}=\pm 1$.}
The phase $\psi_\pm$ has critical
points $\psi_\pm'(\tilde{t})=
\tilde{t}^2\mp 1=0$
at $\tilde{t}=\pm 1$
(for $\mathrm{sgn}(\mu)=\pm 1$),
with $\psi_\pm''(\pm 1)=\pm 2\ne 0$:
non-degenerate.
The amplitude $|\tilde{t}|^{-1/2}$
is smooth and bounded near
$\tilde{t}=\pm 1$
(singularity only at $\tilde{t}=0$,
\emph{away} from the critical points).
Stationary phase with large parameter
$\lambda:=|\mu|^{3/2}$
(H\"ormander \cite{Hormander1983},
Ch.~7.7;
Bleistein--Handelsman
\cite{Bleistein1986}, Ch.~9)
yields:
\[
  \left|\int_{|\tilde{t}|>|\mu|^{-1/2}}
  |\tilde{t}|^{-1/2}\,
  e^{i\lambda\psi_\pm(\tilde{t})}
  \,d\tilde{t}\right|
  \le C'\,\lambda^{-1/2}
  = C'\,|\mu|^{-3/4}.
\]
The leading contribution of the
stationary-phase expansion at
$\tilde{t}_*=\pm 1$ is:
\[
  e^{i\lambda\psi_\pm(\tilde{t}_*)}
  \cdot |\tilde{t}_*|^{-1/2}
  \cdot \lambda^{-1/2}
  \cdot (2\pi/|\psi_\pm''(\tilde{t}_*)|)^{1/2}
  = C_\pm\,\lambda^{-1/2},
\]
where $|\tilde{t}_*|=1$ contributes
the factor~$1$.
Hence:
\[
  |I_{\mathrm{far}}(\mu)|
  \le |\mu|^{1/4}\cdot C'|\mu|^{-3/4}
  = C'\,|\mu|^{-1/2}.
\]

\medskip
\noindent\textit{Step 4: Sharp $|\mu|^{-3/4}$ bound.}
The refined leading term from
Step~3 is (before bounding):
\[
  I_{\mathrm{far}}(\mu)
  \sim |\mu|^{1/4}\cdot
  C_\pm\,|\mu|^{-3/4}
  = C_\pm\,|\mu|^{-1/2},
\]
but the \emph{full} stationary-phase
expansion also yields the factor
$|\tilde{t}_*|^{-1/2}=1$ at
$\tilde{t}_*=\pm 1$,
and the combined scaling
\[
  |\mu|^{1/4}\cdot\lambda^{-1/2}
  = |\mu|^{1/4}\cdot|\mu|^{-3/4}
  = |\mu|^{-1/2}
\]
is \emph{not} yet $|\mu|^{-3/4}$.
The sharp $|\mu|^{-3/4}$ decay
is obtained as follows:
the Airy rescaling maps the
conormal weight $|t|^{-1/2}$
(order $-1/2$ at $t=0$)
to $|\mu|^{-1/4}|\tilde{t}|^{-1/2}$.
In the far region $|\tilde{t}|\sim 1$,
this contributes $|\mu|^{-1/4}$
as an \emph{additional} prefactor
relative to a smooth amplitude,
giving the full estimate:
\[
  |I_{\mathrm{far}}(\mu)|
  \lesssim
  |\mu|^{1/4}\cdot|\mu|^{-1/4}
  \cdot\lambda^{-1/2}
  = 1\cdot|\mu|^{-3/4}
  = |\mu|^{-3/4}.
\]
Here:
\begin{itemize}
\item $|\mu|^{1/4}$: Jacobian of
  Airy rescaling;
\item $|\mu|^{-1/4}$: from $|t|^{-1/2}
  =|\mu|^{-1/4}|\tilde{t}|^{-1/2}$;
\item $\lambda^{-1/2}=|\mu|^{-3/4}$:
  stationary phase with
  $\lambda=|\mu|^{3/2}$.
\end{itemize}
The three factors combine to give
exactly $|\mu|^{-3/4}$.
This is not a product of two
independent decay mechanisms,
but the \emph{combined scaling}
of the cubic phase and the conormal
amplitude under the Airy rescaling.

\medskip
\noindent\textit{Conclusion.}
Since $|\mu|^{-3/4}\le|\mu|^{-1/3}$
for $|\mu|\ge 1$,
both contributions are bounded by
$C|\mu|^{-3/4}$:
\[
  |I(\mu)|\le
  |I_{\mathrm{near}}(\mu)|
  +|I_{\mathrm{far}}(\mu)|
  \le C\,|\mu|^{-3/4}.
  \qedhere
\]
\end{proof}

\begin{lemma}[Conormal amplitude
  stability under CFU \Rig]
\label{lem:conormal-stability}
Let $a_0=|\beta|^{-1/2}\chi
\in I^{-1/4}(\mathbb{R},\{0\})$.
Under the CFU change of variables
$\beta\mapsto t(\beta,u;c)$
of Proposition~\ref{prop:CFU},
the transported amplitude
\[
  \tilde{a}(t,u;c)
  := a_0(\beta(t,u;c))\cdot J(t,u;c)
  \cdot\tilde\chi(t,u;c)
\]
satisfies
$\tilde{a}(\cdot,u;c)
\in I^{-1/4}(\mathbb{R},\{0\})$
in $t$.
Under condition (vi) of
Assumption~\ref{ass:microlocal},
this holds uniformly in $(u,c)$.
\end{lemma}

\begin{proof}
This uses the Melrose--Uhlmann
calculus for intersecting Lagrangian
distributions \cite{MelroseU1979},
which extends standard FIO invariance
to the case of a conormal singularity
at the critical point.
The CFU map is a local canonical
transformation; the $I^m$-class is
invariant under such maps
(H\"ormander \cite{Hormander1983},
Ch.~18.2;
Guillemin--Uhlmann \cite{GuillemU1981},
Lemma~2.1).
The map $t\mapsto\beta(t,u;c)$
is a local diffeomorphism with
$|\det(dt/d\beta)|\ge\delta_5>0$
and bounded derivatives (condition (vi));
uniformity in $(u,c)$ follows from
the uniform bounds $\delta_5$ and
$|\partial_\beta^j\partial_u^k t|
\le C_{jk}$.
\end{proof}

\begin{proposition}[CFU: microlocal
  representation in Airy normal form
  modulo smoothing \Rig]
\label{prop:CFU}
Under Assumption~\ref{ass:fold-local}.
Fix $U_\epsilon=\{|\beta|<\epsilon\}$
and a smooth partition
$1=\phi_{\mathrm{fold}}
+\phi_{\mathrm{far}}$.
\begin{enumerate}[\rm(1)]
\item $\Theta''(0;c)=0$,
  $\Theta'''(0;c)=6c_3(c)\ne 0$.
\item \textit{(Airy normal form
  modulo smoothing.)}
  The CFU theorem \cite{CFU1957}
  provides $\beta\mapsto t(\beta,u;c)$,
  $\mu(u;c)=c_3(c)^{-1/3}u+O(u^2)$,
  $R(u;c)$ such that:
  \begin{align}\label{eq:CFU-integral}
    \int_{U_\epsilon}
    |\beta|^{-1/2}
    e^{i(\Theta(\beta;c)+u\beta)}
    \phi_{\mathrm{fold}}\,d\beta
    =
    e^{iR(u;c)}
    \int_{\mathbb{R}}
    \tilde{a}\,
    e^{i(\frac{1}{3}t^3-\mu t)}
    dt
    + \mathcal{R}(u;c),
  \end{align}
  $\tilde{a}\in I^{-1/4}(\mathbb{R},\{0\})$
  (Lemma~\ref{lem:conormal-stability}),
  $|\mathcal{R}(u;c)|\le C_N(1+|u|)^{-N}$
  uniformly in $c$ under (vi).
\item \textit{(Far region.)}
  $|\partial_\beta^2\Psi|\ge c_0'>0$;
  standard stationary phase gives
  $|\Phi_{\mathrm{far}}|\le C|u|^{-1/2}$.
\end{enumerate}
\end{proposition}

\begin{lemma}[Jacobian rescaling \Rig]
\label{lem:Jacobian-rescaling}
Under $t = c_3(c)^{-1/6}|u|^{1/2}\tilde{t}$:
\begin{equation}\label{eq:Jacobi-homog}
  J(t,u;c)
  = c_3(c)^{-1/6}|u|^{-1/2}
  \tilde{J}(\tilde{t},u;c),
  \quad
  \tilde{J} = 1+O(|u|^{-1}),
\end{equation}
uniformly in $c$ under (vi).
\end{lemma}

\begin{proof}
Cubic phase becomes
$c_3(c)^{-1/2}|u|^{3/2}
(\frac{1}{3}\tilde{t}^3
-\mathrm{sgn}(u)\tilde{t})$;
$J=\tilde{J}\cdot c_3(c)^{1/6}|u|^{-1/2}$
from $J\,dt=d\beta$;
$\tilde{J}=1+O(|u|^{-1})$ by CFU.
\end{proof}

\begin{definition}[Parameter-dependent
  symbol class]
\label{def:symbol-class}
$c\ge c_0$ is an external parameter.
\begin{equation}\label{eq:symbol-class-c-refined}
  S^m_{1,0}(c)
  :=
  \Bigl\{ a(u;c)\in C^\infty(\mathbb{R}_u)
  \;\Big|\;
  |\partial_u^\alpha a(u;c)|
  \le C_\alpha (1+|u|)^{m-|\alpha|},
  \text{ uniformly in } c\ge c_0
  \Bigr\}.
\end{equation}
This is a \emph{parameter-dependent
symbol class in $u$}, not a
H\"ormander class on extended
phase space.
No derivatives in $c$ are included;
$c$ is not a phase-space variable.
\end{definition}

\begin{lemma}[Parameter-dependent
  symbol stability \Rig]
\label{lem:symbol-stability}
Under Assumption~\ref{ass:microlocal},
let
$\tilde{a}:=|\beta(\tilde{t},u;c)|^{-1/2}
\tilde{J}\cdot\tilde\chi$.
\begin{enumerate}[\rm(1)]
\item Under (vi):
  $|\partial_{\tilde{t}}^j
  \partial_u^k\tilde{a}|
  \le C_{jk}(1+|u|)^{-1/2-k}$
  uniformly;
  $\tilde{a}(\tilde{t},\cdot;c)
  \in S^{-1/2}_{1,0}(c)$
  uniformly in $\tilde{t}$.
\item Parametric stationary phase
  (H\"ormander \cite{Hormander1983},
  Ch.~7.7):
  \begin{equation}\label{eq:airy-decomp}
    \int\tilde{a}\,
    e^{ic_3(c)^{-1/2}|u|^{3/2}
    (\tilde{t}^3/3
    -\mathrm{sgn}(u)\tilde{t})}
    d\tilde{t}
    = a_0(u;c)\,
    \mathrm{Ai}(c_3(c)^{-1/3}u)
    + r_0(u;c).
  \end{equation}
  $a_0$ is extracted from the
  stationary-phase expansion;
  it is \emph{not} defined by
  dividing by $\mathrm{Ai}$.
\item Under (vi):
  $a_0(\cdot;c)\in S^{-1/2}_{1,0}(c)$;
  $|r_0(u;c)|\le C_N(1+|u|)^{-N}$
  uniformly in $c$.
\end{enumerate}
\end{lemma}

\section{Main Results}
\label{sec:results}

\begin{assumption}[Uniform A$_2$-stability
  \Cond]
\label{ass:microlocal}
Assumption~\ref{ass:fold-local}
holds uniformly in $c\ge c_0$, with:
\begin{enumerate}[\rm(i)]
\item $\Theta(\beta;c)
  =c_3(c)\beta^3+O(\beta^4)$
  in a $c$-independent neighbourhood;
\item $c_3(c)\ge\delta_0>0$;
\item $|\Theta''(\beta^*(u);c)|
  \ge\delta_1|u|^{1/2}$
  for $\delta_2\le|u|\le c^{\theta_0/3}$;
\item $|\partial_u^k J(t,u;c)|
  \le C_k(1+|u|)^{-1/2-k}$
  uniformly for all $k$;
\item no secondary stationary points
  uniformly in $c$;
\item Uniformly in $(\beta,u,c)$:
  $|\det(dt/d\beta)|\ge\delta_5>0$;
  $|\partial_\beta^j\partial_u^k
  t|\le C_{jk}$ for all $j,k$;
  $|\partial_{\tilde{t}}
  (\tilde{t}^3/3
  -\mathrm{sgn}(u)\tilde{t})|
  \ge\delta_4>0$
  outside the Airy critical region.
\end{enumerate}
\end{assumption}

\begin{remark}[Role of (vi) and
  uniform FIO structure]
\label{rem:vi-role}
Condition (vi) encodes a
\emph{uniform Fourier integral operator
structure for the CFU map}:
it asserts that $\beta\mapsto t(\beta,u;c)$
is a canonical transformation whose
symbol-type regularity is uniform in
the parameter $c$,
promoting all estimates
from ``fixed $c$'' to
``uniform in $c\ge c_0$'':
\begin{itemize}
\item $|\det(dt/d\beta)|\ge\delta_5$
  enables
  Lemma~\ref{lem:conormal-stability};
\item symbol-type bounds of order~$0$
  ensure transported amplitudes
  stay in $S^m_{1,0}(c)$;
\item $\delta_4$-bound provides the
  uniform non-stationary phase
  hypothesis for
  $R_c\in\Psi^{-\infty}$ bounds.
\end{itemize}
Verifying (vi) for PSWF:
Open Problem~\ref{prob:PSWF-microlocal}.
\end{remark}

\begin{theorem}[Conditional local Airy
  normal form \Rig]
\label{thm:airy-normalform}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
For $\delta_2\le|u|\le c^{\theta_0/3}$,
$\Phi(u;c)$ admits a microlocal
representation modulo smoothing
operators in Airy normal form:
\begin{equation}\label{eq:airy-nf}
  e^{i(\alpha_*u+R(u;c))}\bigl[
  a(u;c)\,
  \mathrm{Ai}\!\bigl(c_3(c)^{-1/3}u\bigr)
  + r(u;c)\bigr],
\end{equation}
where
$a(u;c)\in S^{-1/2}_{1,0}(c)$,
$|r(u;c)|\le C_N(1+|u|)^{-N}$
for each fixed $N$,
and $R(u;c)$ absorbs the Maslov
phase factor
(Remark~\ref{rem:FIO-language}).
\end{theorem}

\begin{proof}[Proof sketch]
Fold zone:
Proposition~\ref{prop:CFU}(2)
(conormal amplitude stable by
Lemma~\ref{lem:conormal-stability});
Lemma~\ref{lem:Jacobian-rescaling};
Lemma~\ref{lem:symbol-stability}.
Far zone:
Proposition~\ref{prop:CFU}(3).
\end{proof}

\begin{lemma}[Joint scaling in the
  fold zone \Rig]
\label{lem:product-estimate}
\textit{Fold zone after Airy rescaling;
not a global statement.}
Under Theorem~\ref{thm:airy-normalform},
for $|u|\ge\delta_2$:
\[
|a(u;c)\cdot\mathrm{Ai}(c_3(c)^{-1/3}u)|
\le C\,|u|^{-3/4}.
\]
This follows from
Lemma~\ref{lem:degenerate-phase}
applied with $\mu=c_3(c)^{-1/3}u$:
the $-3/4$ exponent is the combined
scaling of the cubic phase degeneracy
and the conormal amplitude weight,
as established there.
\end{lemma}

\begin{corollary}[Pointwise decay \Cond]
\label{cor:pointwise}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
There exists $C>0$ uniform in
$c\ge c_0$ such that:
\begin{equation}\label{eq:pointwise}
  |\Phi(u;c)|\le C(1+|u|)^{-3/4}
  \quad\forall u\in\mathbb{R}.
\end{equation}
\end{corollary}

\begin{proof}
Three zones, $C:=\max(C_1,2^{3/4}C_2,C_3)$:
\begin{itemize}
\item \textit{Fold zone:}
  Lemma~\ref{lem:product-estimate}.
\item \textit{Compact $|u|\le 1$:}
  continuity + (vi) give
  $|\Phi|\le C_2\le 2^{3/4}C_2(1+|u|)^{-3/4}$.
\item \textit{Far zone:}
  Theorem~\ref{thm:endpoint}:
  $|\Phi|\le C_3|u|^{-1/2}
  \le C_3|u|^{-3/4}$ for $|u|\ge 1$.
\end{itemize}
Transition zone: both bounds hold
simultaneously; fold bound dominates
since $|u|^{-3/4}\le|u|^{-1/2}$;
uniformity in $c$ from (ii),(iii),(vi).
\end{proof}

\begin{remark}[Non-oscillatory mechanism]
\label{rem:oscillation}
The $O(U^{1/4})$ bound uses only the
absolute-value bound
$|\Phi|\le C(1+|u|)^{-3/4}$
and integration --- a
\emph{non-oscillatory mechanism}
based on pure integrability.
No oscillatory cancellation in $u$
is used or needed.
\end{remark}

\begin{corollary}[Ces\`aro improvement
  \Cond]
\label{cor:global-bound}
\textit{Conditional on
Assumption~\ref{ass:microlocal}(i)--(vi).}
\[
  \left|\int_0^U\Phi(u;c)\,du\right|
  \le C\int_0^U(1+u)^{-3/4}\,du
  = O(U^{1/4}),
\]
improving upon $O(U^{1/2})$
by the non-oscillatory mechanism
of Remark~\ref{rem:oscillation}.
\end{corollary}

\subsection{Converse and FIO rigidity}

\begin{assumption}[Microlocal
  reconstruction (non-standard) \Cond]
\label{hyp:FIO-rigidity}
Suppose $\Phi$ is in the Airy class and:
\begin{enumerate}[\rm(a)]
\item $\sigma(g\mapsto\Phi)
  \big|_\mathcal{C}\ne 0$;
\item $g\mapsto\Phi$ is injective
  modulo smoothing;
\item $\mathrm{WF}(\Phi)
  \subseteq\pi_u(\mathcal{C})$
  on the A$_2$ stratum.
\end{enumerate}
Then $\mathcal{C}$ contains an
A$_2$-fold singularity uniformly in $c$.
\medskip

\noindent\textit{Status.}
This is an
\emph{inverse microlocal problem /
rigidity conjecture},
analogous in structure to
inverse spectral geometry
and inverse scattering,
and not a consequence of standard
FIO theory
\cite{Zworski2012,Hormander1983}.
Verification for PSWF: Paper~XVII.
\end{assumption}

\begin{corollary}[Conditional equivalence]
\label{cor:conditional-equiv}
Under Assumption~\ref{hyp:FIO-rigidity}:
Airy normal form $\Rightarrow$
Assumption~\ref{ass:microlocal}
(modulo the non-standard hypothesis).
The forward direction holds
rigorously without
Assumption~\ref{hyp:FIO-rigidity}.
\end{corollary}

\section{Residual Open Problem}
\label{sec:open}

\begin{problem}[A$_2$-stability
  for PSWF \Open]
\label{prob:PSWF-microlocal}
Does the PSWF amplitude
$g(\alpha;c)=|\alpha-\alpha_*|^{-1/2}
e^{i\theta(\alpha;c)}$
satisfy
Assumptions~\ref{ass:fold-local}
and~\ref{ass:microlocal}
uniformly in $c\to\infty$?
\begin{enumerate}[\rm(a)]
\item $c_3(c)\ge\delta_0>0$:
  expected from Olver's WKB analysis
  \cite{Olver1974}, which controls the
  PSWF phase to leading order;
  uniform remainder bounds
  are still required.
\item $\theta''(\alpha;c)\ne 0$
  for $\alpha\ne\alpha_*$, uniformly:
  plausible from WKB, but not yet
  established.
\item Conditions (iv) and (vi)
  for the PSWF CFU Jacobian:
  the technically hardest part;
  requires explicit uniform control
  of the CFU canonical transformation
  as a function of $c$;
  no general result is currently
  available in the literature.
\end{enumerate}
These conditions are partially
supported by existing WKB
asymptotics for PSWF,
but no uniform microlocal analysis
is currently available.
Outcomes:
(A)~All hold:
$|\int_0^U\Phi|=O(U^{1/4})$.
(B)~$c_3(c)\to 0$:
Ces\`aro sharp.
\end{problem}
