\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis:\\
  Cross-Gram Operators, Spectral Evacuation, and the Transition Zone}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
The central obstruction to coefficient stability
of the Slepian eigenbasis is
\emph{spectral non-evacuation in the transition
zone}: the persistence of spectral mass of the
measures $\mu_{n,c}(d\xi)
:=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi$
in the regime $n\sim\alpha c$ as $c\to\infty$.

This paper reduces Hypothesis~H1 of Paper~XIII
(the $\ell^2$-norm convergence of Slepian
coefficients $a_n^{(c)}=\ip{g}{\Phic{c}{n}}$
as $c\to\infty$) to a single spectral
distribution problem in the transition
zone $n\sim\alpha c$.

The key structural results are:
(i)~the cross-Gram operator $T_c$ satisfies
$\norm{T_c}\le 1$ uniformly;
(ii)~the tail functional
$S_N(c):=\sum_{n>N}\mu_{n,c}([-c_0,c_0])$
controls the tail via
$\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c)$;
(iii)~$S_N(c)=\mathrm{Tr}(\Ac)$
is the Hilbert--Schmidt norm of
$\Bn:=(I-\Pi_N^{(c)})\Pc$
(not the operator norm;
$\norm{\Ac}_{\mathrm{op}}\le S_N(c)$
with strict inequality in general),
and the relevant quadratic form is
$Q_c^N(g)=\ip{\Ac g}{g}$
where $\Ac:=\Bn^*\Bn$.

H3 ($\sup_c S_N(c)\to 0$) implies H1
(proved unconditionally).
H1 implies $\Ac\to 0$ in the strong operator
topology (proved); the gap to operator norm
convergence is a \emph{rank explosion phenomenon}
(Open Problem~\ref{prob:H1H3}).

The \emph{No-Concentration condition} (NC)
quantifies spectral non-evacuation:
NC$\Rightarrow$H3 fails (proved).
The converse,
NC$^c\Rightarrow$H3, is proved false
(Proposition~\ref{prop:NC-not-H3}):
an explicit model family with NC$^c$
but $\liminf_c S_N(c)=1>0$ shows that
being a band-level and a global tail
phenomenon are genuinely distinct.
NC holds under the local Weyl law (Paper~XV).
\end{abstract}

%%═══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context}

Paper~XIII identified H1 as the key
remaining hypothesis for completeness
$\Kspace=L^2(\mathbb{R})$.
Paper~XIV reduces H1 to a spectral
distribution problem for the
PSWF eigenbasis in the transition zone.
All conditional steps are explicitly labelled.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$
($c\to\infty$).
\end{hypothesis}

H1 splits:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
\eqref{eq:H1ip} is proved
(Prop.~\ref{prop:H1ip});
\eqref{eq:H1norm} is equivalent to H3
(given OP~\ref{prob:H1H3}).
The core difficulty:
$a^{(c)}\rightharpoonup a^{(\infty)}$
does not imply norm convergence;
Paper~XIV localises this gap precisely.

\subsection{The central new framing}

\begin{definition}[Spectral measures]
\label{def:spectral-measures}
For $n\ge 0$, $c>c_0$:
\[
  \mu_{n,c}(d\xi)
  :=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi.
\]
Plancherel: $\mu_{n,c}(\mathbb{R})=1$.
The tail functional and its quadratic form:
\[
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0]),
  \qquad
  Q_c^N(g) := \sum_{n>N}|a_n^{(c)}|^2
  = \norm{(I-\Pi_N^{(c)})\Pc g}^2.
\]
The obstruction to H1 is
\emph{spectral non-evacuation}:
the failure of $\mu_{n,c}([-c_0,c_0])$
to vanish for $n>N$, uniformly in $c$.
\end{definition}

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $0<\alpha<\kappa<1$:
$\mathcal{T}_{\alpha,\kappa}(c)
:=\{n: \alpha c < n \le \kappa c\}$.
This band has $\sim(\kappa-\alpha)c$ elements
and captures the regime where Slepian
eigenvalues $\lambda_n^{(c)}$
transit between $\approx 1$ and $\approx 0$.
\end{definition}

\subsection{Logical status}

\begin{center}
\fbox{\parbox{0.88\linewidth}{
\textbf{Proved unconditionally:}
\begin{itemize}[leftmargin=*]
\item $\norm{T_c}\le 1$
\item Tail reduction: $Q_c^N(g)\le\norm{g}^2 S_N(c)$
\item \eqref{eq:H1ip}
\item $S_N(c)=\norm{\Bn}^2_{\mathrm{HS}}=\mathrm{Tr}(\Ac)$
\item $\norm{\Ac}_{\mathrm{op}}\le S_N(c)$
  (strict in general; HS $\ne$ op.~norm)
\item $Q_c^N(g)=\ip{\Ac g}{g}$,
  $\Ac=\Bn^*\Bn\ge 0$
\item $\sigma(c)\sim cc_0/\pi$
\item NC $\Rightarrow$ H3 fails
\item Local Weyl law $\Rightarrow$ NC
\item H3 $\Rightarrow$ H1
\item H1 $\Rightarrow$ $\Ac\to 0$ (SOT)
\item NC$^c\not\Rightarrow$H3
  (explicit model; Prop.~\ref{prop:NC-not-H3})
\end{itemize}
\textbf{Open / conditional:}
\begin{itemize}[leftmargin=*]
\item H1 $\Rightarrow$ $\norm{\Ac}_{\mathrm{op}}\to 0$

  OP~\ref{prob:H1H3}

  OP~\ref{prob:fourier-mass}
\end{itemize}}}
\end{center}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{theorem}[$\norm{T_c}\le 1$]
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{theorem}
\begin{proof}
Bessel:
$\norm{T_c g}^2=\sum_n|\ip{g}{\Phic{c}{n}}|^2
\le\norm{g}^2$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds unconditionally.
\end{proposition}
\begin{proof}
Paper~XII: $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$.
\end{proof}

\begin{remark}
By weak lsc:
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$,
so H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}\le\norm{a^{(\infty)}}$.
\end{remark}

%%═══ 4. TAIL REDUCTION, HS IDENTITY, AND QUADRATIC FORM
\section{Tail Reduction, HS Identity, and the Quadratic Form}
\label{sec:tail}

\begin{definition}[$S_N(c)$, $\sigma(c)$]
\label{def:SN}
$S_N(c):=\sum_{n>N}\mu_{n,c}([-c_0,c_0])$,
$\quad\sigma(c):=\sum_{n\ge 0}\mu_{n,c}([-c_0,c_0])$.
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$Q_c^N(g):=\sum_{n>N}|a_n^{(c)}|^2
\le\norm{g}^2\cdot S_N(c)$.
\end{lemma}
\begin{proof}
Paley--Wiener:
$a_n^{(c)}
=\ip{\hat g}{
  \widehat{\Phi_n^{(c)}}\cdot
  \mathbf{1}_{[-c_0,c_0]}}_{L^2}$.
Cauchy--Schwarz:
$|a_n^{(c)}|^2
\le\norm{g}^2\cdot
\mu_{n,c}([-c_0,c_0])$.
Sum over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness]
$\sup_c S_N(c)\to 0
\Rightarrow\{a^{(c)}\}_c$
equi-tight in $\ell^2$.
\end{corollary}

\begin{theorem}[HS identity, quadratic form,
  and $S_N$ as trace control]
\label{thm:SN-operator}
Define:
\begin{align}\label{eq:Bn-def}
  \Bn &:= (I-\Pi_N^{(c)})\Pc
    \;:\; \PW_{c_0}^2\to\PW_c^2,\\
  \Ac &:= \Bn^*\Bn
    \;:\; \PW_{c_0}^2\to\PW_{c_0}^2.
\end{align}

\medskip
\textbf{(I) HS identity.}
\begin{equation}\label{eq:SN-HS}
  S_N(c)
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}
  = \norm{\Bn}^2_{\mathrm{HS}}
  = \mathrm{Tr}(\Ac).
\end{equation}
\begin{proof}
Expand $\norm{\Bn}^2_{\mathrm{HS}}$
in the ONB $\{\Phic{c}{n}\}_{n>N}$:
$\sum_{n>N}\norm{\Bn\Phic{c}{n}}^2
=\sum_{n>N}\norm{\Pc\Phic{c}{n}}^2
=S_N(c)$.
Since $\Ac=\Bn^*\Bn$,
$\mathrm{Tr}(\Ac)=\norm{\Bn}^2_{\mathrm{HS}}$.
\end{proof}

\medskip
\textbf{(II) Operator norm inequality.}
\begin{equation}\label{eq:op-le-HS}
  \norm{\Ac}_{\mathrm{op}}
  = \norm{\Bn}^2_{\mathrm{op}}
  \le\mathrm{Tr}(\Ac) = S_N(c).
\end{equation}
In general $\norm{\Ac}_{\mathrm{op}}
<\mathrm{Tr}(\Ac)$ unless $\Ac$ has rank one.
\emph{The identity
$\norm{\Bn}^2_{\mathrm{op}}=S_N(c)$
claimed in earlier versions is incorrect.}

\medskip
\textbf{(III) Quadratic form.}
For $g\in\PW_{c_0}^2$,
using $\Pc g=g$:
\begin{equation}\label{eq:Qc-key}
  Q_c^N(g)
  = \sum_{n>N}|\ip{g}{\Phic{c}{n}}|^2
  = \norm{\Bn g}^2
  = \ip{\Ac g}{g},
  \qquad 0\le\Ac\le I.
\end{equation}

\medskip
\textbf{(IV) The full hierarchy.}
\begin{equation}\label{eq:chain}
  \underbrace{\text{H3}}_{\mathrm{Tr}(\Ac)\to 0}
  \;\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0
  \;\Rightarrow\;
  Q_c^N(g)\to 0\;(\text{unif.}\,c,g)
  \;\Rightarrow\;
  \text{H1}.
\end{equation}
The reverse arrows do not hold in general.
H3 is a \emph{trace-class condition};
H1 requires \emph{uniform spectral decay},
i.e.\ $\norm{\Ac}_{\mathrm{op}}\to 0$
uniformly in $c$.
\end{theorem}

\begin{remark}[H3 as trace-class;
H1 as uniform spectral decay]
\label{rem:HS-trace}
$S_N(c)=\mathrm{Tr}(\Ac)$ sums
all eigenvalues of $\Ac$:
global spectral energy.
$\norm{\Ac}_{\mathrm{op}}$
is the largest eigenvalue: worst-case direction.
\begin{equation}\label{eq:trace-op-rank}
  \norm{\Ac}_{\mathrm{op}}
  \le \mathrm{Tr}(\Ac)
  = S_N(c)
  \le \mathrm{rank}(\Ac)\cdot\norm{\Ac}_{\mathrm{op}}.
\end{equation}
The relevant problem (OP~\ref{prob:H1H3})
is whether SOT convergence $\Ac\to 0$
allows the upgrade to
$\norm{\Ac}_{\mathrm{op}}\to 0$:
the classical SOT $\not\Rightarrow$ op.~norm gap.
\end{remark}

\begin{remark}[Trace vs.\ operator norm
and the rank explosion obstruction]
\label{rem:rank-obstruction}
In general:
\[
  \mathrm{Tr}(\Ac)\to 0
  \;\not\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0.
\]
However, if $\mathrm{rank}(\Ac)\le R$
uniformly, then by~\eqref{eq:trace-op-rank}
$\mathrm{Tr}(\Ac)\to 0$
implies $\norm{\Ac}_{\mathrm{op}}\to 0$.
The obstruction is therefore a
\emph{rank explosion phenomenon}:
$\mathrm{rank}(\Ac)\to\infty$.
Open Problem~\ref{prob:H1H3} asks whether
$\Ac=\Pc(I-\Pi_N^{(c)})P_c(I-\Pi_N^{(c)})\Pc$
admits rank control via the
bandlimit geometry.
\end{remark}

\begin{proposition}[Trace identity and growth]
\label{prop:trace}
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim cc_0/\pi\to\infty$.
\end{proposition}
\begin{proof}
$\sigma(c)=\mathrm{Tr}(\Pc P_c\Pc)$;
Slepian--Weyl \cite{Slepian1961}:
$\sigma(c)\sim cc_0/\pi$.
\end{proof}

\begin{remark}[Counterexample and PSWF-incompatibility]
\label{rem:counterexample}
$\sigma(c)\to\infty$ does not imply
$S_N(c)\not\to 0$:
if $\mu_{n,c}([-c_0,c_0])=0$ for all $n>N_0$,
then $S_N(c)=0$ for $N\ge N_0$.
PSWF-incompatibility:
(i) Slepian--Weyl requires an
unbounded set of contributing indices;
(ii) Paper~VIII gives
$\mu_{n,c}([-c_0,c_0])>0$ for each
$n\le\kappa c$.
NC (Definition~\ref{def:NC})
formalises the required lower bound.
\end{remark}

%%═══ 5. SPECTRAL MEASURES AND NC
\section{Spectral Measures and No-Concentration}
\label{sec:NC}

\begin{remark}[NC as spectral evacuation failure]
\label{rem:NC-evac}
The obstruction to H1 stated cleanly:
\begin{quote}
  \emph{H1 fails if and only if
  there exists persistent spectral mass
  of the measures $\mu_{n,c}$
  in the transition zone $n\sim\alpha c$
  (conditional on OP~\ref{prob:H1H3}
  and OP~\ref{prob:fourier-mass}).}
\end{quote}
``Spectral evacuation'' means
$\mu_{n,c}([-c_0,c_0])\to 0$
for $n>N$, uniformly in $c$.
NC is the quantitative, band-level
obstruction to this;
H3 is the global tail phenomenon.
Proposition~\ref{prop:NC-not-H3}
shows these two levels are genuinely distinct.
\end{remark}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
Fix $0<\alpha<\kappa<1$.
NC holds if there exists $c_*>0$ such that
for all large $c$:
\begin{equation}\label{eq:NC}
  \sum_{n\in\mathcal{T}_{\alpha,\kappa}(c)}
  \mu_{n,c}([-c_0,c_0]) \ge c_*>0.
\end{equation}
\end{definition}

\begin{remark}[Why proportional width is necessary]
\label{rem:prop-width}
A window of width $c^\beta$ ($\beta<1$)
has $\sim 2c^\beta$ indices;
under the local Weyl law,
the sum is $\sim 2c_0 c^{\beta-1}\to 0$:
insufficient for NC.
A proportional band
$\mathcal{T}_{\alpha,\kappa}(c)$
has $\sim(\kappa-\alpha)c$ indices,
giving sum $\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[NC as quantum ergodicity]
\label{rem:NC-QE}
In the semiclassical limit $c\to\infty$,
$n\sim\alpha c$, the rescaled measures
$c\cdot\mu_{n,c}|_{[-c,c]}$
are expected to converge weakly to
the WKB density
$(\pi\sqrt{1-(\xi/c)^2})^{-1}d\xi$:
a one-dimensional quantum ergodicity
statement \cite{Zworski2012}.
NC is the consequence for $[-c_0,c_0]$:
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$,
so the band sum
$\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[Local Weyl law $\Rightarrow$ NC]
\label{rem:Weyl-NC}
If uniformly for $n\in\mathcal{T}_{\alpha,\kappa}(c)$:
\begin{equation}\label{eq:local-Weyl}
  \mu_{n,c}([-c_0,c_0])\sim\frac{c_0}{c},
\end{equation}
then $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
\sim c_0(\kappa-\alpha)>0$.
NC holds with $c_*=c_0(\kappa-\alpha)/2$.
\end{remark}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under NC: for each fixed $N$,
$\liminf_{c\to\infty}S_N(c)\ge c_*>0$,
so H3 fails.
\end{proposition}
\begin{proof}
For large $c$: $\alpha c>N$,
hence $\mathcal{T}_{\alpha,\kappa}(c)\subseteq\{n>N\}$
and $S_N(c)\ge c_*$.
\end{proof}

\begin{proposition}[NC$^c$ does not imply H3]
\label{prop:NC-not-H3}
There exists a model family of measures
$\{\mu_{n,c}\}$ satisfying the normalization
$\mu_{n,c}(\mathbb{R})=1$ and $\mu_{n,c}\ge 0$,
such that\textup{:}
\begin{enumerate}[\rm(i)]
\item\label{it:NCc}
  \textup{NC$^c$ holds}: for every
  $0<\alpha<\kappa<1$ and every $c_*>0$,
  \[
    \liminf_{c\to\infty}
    \sum_{n\in\mathcal{T}_{\alpha,\kappa}(c)}
    \mu_{n,c}([-c_0,c_0]) < c_*;
  \]
\item\label{it:H3fail}
  \textup{H3 fails}: for every fixed $N$,
  \[
    \liminf_{c\to\infty} S_N(c) = 1 > 0.
  \]
\end{enumerate}
In particular, $\mathrm{NC}^c\not\Rightarrow\mathrm{H3}$.
\end{proposition}

\begin{proof}
Fix $\alpha\in(0,1)$ and set
$w(c):=\lfloor\sqrt{c}\rfloor$.
Define the active zone:
\[
  I_c := \{n\in\mathbb{N}_0:
    |n-\lfloor\alpha c\rfloor|\le w(c)\},
  \qquad |I_c|=2w(c)+1\sim 2\sqrt{c}.
\]
Assign weights:
\begin{equation}\label{eq:model-weights}
  \mu_{n,c}([-c_0,c_0]) :=
  \begin{cases}
    \dfrac{1}{2w(c)+1} & n\in I_c,\\
    0 & n\notin I_c.
  \end{cases}
\end{equation}
Extend $\mu_{n,c}$ to a probability measure
on $\mathbb{R}$ arbitrarily
(e.g.\ place remaining mass outside $[-c_0,c_0]$).

\medskip
\textit{Proof of~\ref{it:NCc} (NC$^c$).}
Fix $0<\alpha'<\kappa'<1$.
$\mathcal{T}_{\alpha',\kappa'}(c)$ contains
at most $|I_c|=2w(c)+1\sim 2\sqrt{c}$
indices from $I_c$ (since the centre of
$I_c$ lies at $\lfloor\alpha c\rfloor$,
which may or may not lie in
$\mathcal{T}_{\alpha',\kappa'}(c)$).

\textit{Case 1}: $\alpha\notin(\alpha',\kappa')$.
For large $c$, $\lfloor\alpha c\rfloor
\notin\mathcal{T}_{\alpha',\kappa'}(c)$,
so $|I_c\cap\mathcal{T}_{\alpha',\kappa'}(c)|=0$
and the sum is $0$.

\textit{Case 2}: $\alpha\in(\alpha',\kappa')$.
$|I_c\cap\mathcal{T}_{\alpha',\kappa'}(c)|
\le|I_c|=2w(c)+1$, so:
\[
  \sum_{n\in\mathcal{T}_{\alpha',\kappa'}(c)}
  \mu_{n,c}([-c_0,c_0])
  \le \frac{2w(c)+1}{2w(c)+1}=1.
\]
But crucially, the relative width is
$|I_c|/|{\mathcal{T}_{\alpha',\kappa'}(c)}|
\sim 2\sqrt{c}/(\kappa'-\alpha')c\to 0$,
so the mass is concentrated in a
\emph{sub-proportional} sub-band
$|I_c|/c\to 0$.
For any $c_*>0$, choose
$\alpha'',\kappa''$ with
$\alpha\in(\alpha'',\kappa'')\subsetneq(\alpha',\kappa')$
and $\kappa''-\alpha''<c_*/2$;
then for large $c$,
$I_c\cap\mathcal{T}_{\alpha'',\kappa''}(c)=\emptyset$
and the sum over $\mathcal{T}_{\alpha'',\kappa''}$ is $0$.
Since this holds for all
$\alpha',\kappa'$ and all $c_*>0$,
NC$^c$ holds.

\medskip
\textit{Proof of~\ref{it:H3fail} (H3 fails).}
Fix $N$.
For large $c$: $\lfloor\alpha c\rfloor>N$,
so $I_c\subseteq\{n>N\}$.
Therefore:
\[
  S_N(c)
  =\sum_{n>N}\mu_{n,c}([-c_0,c_0])
  \ge\sum_{n\in I_c}\frac{1}{2w(c)+1}
  = 1.
\]
Hence $\liminf_{c\to\infty}S_N(c)\ge 1$
for every fixed $N$, so H3 fails.
\end{proof}

\begin{remark}[Structural separation]
\label{rem:separation}
Proposition~\ref{prop:NC-not-H3} shows that
NC and H3 operate at genuinely different levels:
\begin{itemize}[leftmargin=*]
\item \emph{NC is a band-level phenomenon}:
  it detects persistent mass in a fixed
  proportional band $\mathcal{T}_{\alpha,\kappa}(c)$.
\item \emph{H3 is a global tail phenomenon}:
  it requires the total tail mass $S_N(c)$
  to vanish, regardless of
  how mass is distributed.
\end{itemize}
The model in the proof achieves NC$^c$
by concentrating mass in a
\emph{sub-proportional} band
$|I_c|\sim\sqrt{c}\ll c$
that escapes every fixed proportional window,
while maintaining $S_N(c)=1$ for all $N$.
This is the mass migration mechanism
described in the introduction:
no single proportional band carries
persistent mass,
but the global tail never empties.
\end{remark}

%%═══ 6. H3 AND THE MAIN REDUCTION
\section{H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)
=\sup_{c>c_0}\mathrm{Tr}(\Ac)
\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{theorem}[H3 $\Rightarrow$ H1]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$, $g\in\PW_{c_0}^2$.
H3: choose $N$ with
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$;
then $Q_c^N(g)<\varepsilon/2$ (all $c$).
Paper~XII: choose $c_\varepsilon$ with
$\sum_{n\le N}|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ ($c>c_\varepsilon$).
\end{proof}

\begin{theorem}[H1 $\Rightarrow$
  $\Ac\to 0$ in SOT]
\label{thm:H1-H3-ptwise}
Assume H1. For each fixed $g$:
$Q_c^N(g)=\ip{\Ac g}{g}\to 0$
as $N\to\infty$, uniformly in $c$.
Equivalently, $\Ac\to 0$ in SOT.
\end{theorem}
\begin{proof}
$Q_c^N(g)
\le 2\norm{a^{(c)}-a^{(\infty)}}^2
+2\sum_{n>N}|a_n^{(\infty)}|^2\to 0$.
\end{proof}

\begin{remark}[The uniformity gap]
\label{rem:unif-gap}
\[
  \text{H3}\;(\mathrm{Tr}\to 0)
  \Rightarrow
  \norm{\Ac}_{\mathrm{op}}\to 0
  \Rightarrow
  \text{H1}
  \Rightarrow
  \Ac\to 0\;(\text{SOT}).
\]
Missing reverse:
SOT $\not\Rightarrow$
$\norm{\Ac}_{\mathrm{op}}\to 0$.
By Remark~\ref{rem:rank-obstruction}:
rank control of $\Ac$ would close this gap
(OP~\ref{prob:H1H3}).
\end{remark}

\label{thm:reduction}
\textup{(i)} Proved:
\begin{align*}
  &\text{H3}\Rightarrow\text{H1},
  \quad
  \text{H1}\Rightarrow\Ac\to 0\;(\mathrm{SOT}),\\

  \quad

  &\text{NC}^c\not\Rightarrow\text{H3}

\textup{(ii)} Open:

  \qquad

  \;(\text{OP~\ref{prob:fourier-mass}}).
\]
\textup{(iii)} Conditional chain:
\[

  \Rightarrow

\end{theorem}

%%═══ 7. INTERPRETATION
\section{Interpretation and Context}
\label{sec:interp}

\subsection{The true obstruction: spectral non-evacuation}

\[
  \text{H1 fails}
  \;\Longleftrightarrow\;
  \text{spectral non-evacuation in }n\sim\alpha c
  \quad(\text{cond. on OPs~\ref{prob:H1H3},
  \ref{prob:fourier-mass}}).
\]
``Spectral evacuation'' means
for every $N$, the $\mu_{n,c}$-mass
of $\{n>N\}$ on $[-c_0,c_0]$
vanishes uniformly in $c$.
The model of Proposition~\ref{prop:NC-not-H3}
shows that mass can migrate through
sub-proportional bands and still prevent
evacuation, even without any single
proportional band carrying persistent mass.

\subsection{NC as band-level; H3 as global-tail}

Proposition~\ref{prop:NC-not-H3} establishes
the \emph{structural separation}:
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{lll}
\toprule
Condition & Level & Quantifies \\
\midrule
NC & Band-level
  & Persistent mass in $\mathcal{T}_{\alpha,\kappa}(c)$ \\
H3 & Global tail
  & Total mass $S_N(c)\to 0$ \\
\bottomrule
\end{tabular}
\end{center}
NC$\Rightarrow$H3 fails;
but NC$^c$ does not imply H3.
The correct obstruction for H1 is H3, not NC.

\subsection{$S_N(c)$ as trace control;
rank explosion}

By~\eqref{eq:trace-op-rank}:
$\norm{\Ac}_{\mathrm{op}}
\le\mathrm{Tr}(\Ac)=S_N(c)
\le\mathrm{rank}(\Ac)\cdot\norm{\Ac}_{\mathrm{op}}$.
H3 controls the trace;
rank control (OP~\ref{prob:H1H3})
would bridge trace to operator norm.

\subsection{Quantum ergodicity in frequency space}

The spectral measures $\mu_{n,c}$
play the role of semiclassical measures;
the local Weyl law
(OP~\ref{prob:fourier-mass})
asserts their equidistribution:
a one-dimensional quantum ergodicity
statement \cite{Zworski2012}.

%%═══ 8. PROGRAMME FOR PAPER XV
\section{Programme for Paper~XV}
\label{sec:PaperXV}

\subsection{The three required estimates}

\begin{enumerate}[\rm(A)]
\item\label{XV-A} \textbf{Upper bound, uniform:}
  $\mu_{n,c}([-c_0,c_0])\le C/c$
  for all $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-B} \textbf{Lower bound, uniform:}
  $\mu_{n,c}([-c_0,c_0])\ge 1/(Cc)$
  for all $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-C} \textbf{Summability:}
  $\mu_{n,c}([-c_0,c_0])=c_0/c+o(1/c)$
  uniformly, so
  $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
  \to c_0(\kappa-\alpha)$.
\end{enumerate}
(A)+(B) $\Rightarrow$ NC;
(C) $\Rightarrow$ equidistribution.

\subsection{Semiclassical approach}

$h:=1/c\to 0$; the PSWF solves
\begin{equation}\label{eq:PSWF-ODE}
  -(1-x^2)\partial_x^2\psi
  +2x\partial_x\psi
  +c^2x^2\psi=\chi_n^{(c)}\psi
\end{equation}
with energy $E_n^{(h)}=\chi_n^{(c)}/c^2
\to E_\alpha$ for $n\sim\alpha c$.
WKB in the classical region $|x|<x_0$
($x_0^2=E_\alpha$) and stationary phase
for $\xi\in[-c_0,c_0]$ give
$\mu_{n,c}([-c_0,c_0])\sim 2c_0C(E_\alpha)/c$,
yielding (A)--(C).

\subsection{Rigorous programme}

\begin{enumerate}
\item WKB with remainder $O(c^{-2})$,
  uniform in $n\in\mathcal{T}$ \cite{Osipov2013}.
\item Stationary phase with uniform remainder.
\item Prove (A)--(C); conclude NC.
\item Address OP~\ref{prob:H1H3}
  (rank control via bandlimit geometry).
\end{enumerate}

\begin{problem}[Rank control and
  SOT-to-operator-norm upgrade]
\label{prob:H1H3}

rank control of
$\Ac=\Pc(I-\Pi_N^{(c)})P_c
(I-\Pi_N^{(c)})\Pc$
would give $\norm{\Ac}_{\mathrm{op}}\to 0$.
Does the bandlimit geometry
(nested projections,
Paley--Wiener structure) provide this?
\end{problem}

\label{prob:NC-direct}
\emph{(Priority: medium.)}
Do the pointwise bounds of Paper~VIII
imply lower bounds on
$\mu_{n,c}([-c_0,c_0])$
without full WKB?
\end{problem}
