\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis under the SOT-Limit:\\
  Cross-Gram Operators and the Spectral Leakage Barrier}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
This paper analyses Hypothesis~H1 of Paper~XIII:
the $\ell^2$-norm convergence of Slepian coefficients
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ as $c\to\infty$,
for $g \in \PW_{c_0}^2$.

We introduce the \emph{cross-Gram operator}
$T_c g := \sum_n \ip{g}{\Phic{c}{n}}\Phiinf{n}$,
show $\norm{T_c}\le 1$ uniformly,
and reduce H1 to a single analytic condition:
uniform tail control of
$S_N(c) := \sum_{n>N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.

The \emph{Spectral Leakage Barrier}
(Theorem~\ref{thm:barrier})
states that $\sup_c S_N(c) \not\to 0$.
This is proved rigorously:
we establish $\sum_{n\le\kappa c}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\ge c_0' > 0$ uniformly in $c$ using
the non-cancellation estimates of Paper~VIII,
and derive the barrier from this uniform
low-index sum.
The quantitative growth $\liminf_c S_N(c)\ge\kappa c_0>0$
is conditional on Assumption~\ref{ass:fourier-mass}
(a local Weyl law for transition-zone PSWFs).

We isolate the missing property as
\textbf{H3 (Uniform Frequency Evacuation)}
and prove H3$\Rightarrow$H1 completely.
The converse H1$\Rightarrow$H3
is reduced (via operator-norm duality)
to a Banach--Steinhaus uniformity problem
(Open Problem~\ref{prob:H1H3}).
H3 is not implied by any result in Papers~I--XIII.

\medskip\noindent
\textbf{Central result} (informal):
\emph{H1 is equivalent to uniform evacuation
of Fourier mass from high-index PSWFs;
this fails by persistent spectral leakage
from the transition zone $n\sim\alpha c$.}
\end{abstract}

\section{Introduction}

\subsection{Inheritance from Paper XIII}

Paper~XIII established that all frame-based
routes to completeness $\Kspace=L^2(\mathbb{R})$
are closed and identified \textbf{H1} as the
only currently identified hypothesis strictly
weaker than completeness and grounded in the
series. Paper~XIV takes H1 as its sole object.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1: coefficient stability]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
\[
  \norm{a^{(c)}-a^{(\infty)}}_{\ell^2} \to 0
  \quad(c\to\infty),
\]
where $a_n^{(c)}:=\ip{g}{\Phic{c}{n}}$ and
$a_n^{(\infty)}:=\ip{g}{\Phiinf{n}}$.
\end{hypothesis}

By the Hilbert-space identity
$\norm{x-y}^2=\norm{x}^2+\norm{y}^2
-2\mathrm{Re}\ip{x}{y}$, H1 splits into:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
Condition~\eqref{eq:H1ip} is proved in
Proposition~\ref{prop:H1ip};
the hard part is~\eqref{eq:H1norm}
(Remark~\ref{rem:reduction}).

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $\alpha\in(0,1)$, the \emph{transition
zone at rate $\alpha$} is
$\mathcal{T}_\alpha(c):=
\{n\in\mathbb{N}_0 : |n-\alpha c|\le c^{1/2}\}$.
Indices in $\mathcal{T}_\alpha(c)$ are
\emph{high} relative to any fixed $N$
and \emph{low} relative to $c$.
\end{definition}

\begin{lemma}[From series]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item\label{A:bessel}
  \emph{Bessel.}
  $\sum_n|a_n^{(c)}|^2\le\norm{g}^2$;
  in particular $\norm{\Phic{c}{n}}_{L^2}=1$.
\item\label{A:ptwise}
  \emph{Paper~XII.}
  $a_n^{(c)}\to a_n^{(\infty)}$ for each fixed $n$.
\item\label{A:nonvanish}
  \emph{Paper~VIII (non-cancellation).}
  For all $n\le\kappa c$ and large $c$:
  $|\widehat{\Phi_n^{(c)}}(\xi)|\ge c_1 c^{-1/2}$
  on a set $E_c^{(n)}\subseteq[-c,c]$
  with $|E_c^{(n)}|\ge c_* c^{-1/2}$.
\item\label{A:supp}
  \emph{Paley--Wiener.}
  $g\in\PW_{c_0}^2\Rightarrow
  \operatorname{supp}\hat g\subseteq[-c_0,c_0]$.
\end{enumerate}
\end{lemma}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
\label{def:crossgram}
$T_c g:=\sum_{n\ge 0}
\ip{g}{\Phic{c}{n}}\,\Phiinf{n}.$
\end{definition}

\begin{proposition}
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{proposition}
\begin{proof}
Bessel: $\norm{T_c g}^2
=\sum_n|\ip{g}{\Phic{c}{n}}|^2\le\norm{g}^2$.
\end{proof}

\begin{lemma}[Cross-Gram identity]
\label{lem:crossgram-ip}
$\ip{T_c g}{g}=\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}$.
\end{lemma}
\begin{proof}
$\ip{T_c g}{g}
=\sum_n a_n^{(c)}\overline{a_n^{(\infty)}}$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds.
\end{proposition}
\begin{proof}
Lemma~\ref{lem:inherited}\ref{A:ptwise}
+ \ref{A:bessel} give
$a^{(c)}\rightharpoonup a^{(\infty)}$
in $\ell^2$, hence
$\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
\to\norm{a^{(\infty)}}^2$.
\end{proof}

\begin{remark}[H1 reduces to \eqref{eq:H1norm}]
\label{rem:reduction}
Weak lower semi-continuity gives
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$
automatically.
Hence H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}_{\ell^2}
\le\norm{a^{(\infty)}}_{\ell^2}$.
\end{remark}

%%═══ 4. TAIL REDUCTION
\section{Tail Reduction}
\label{sec:tail}

\begin{definition}[$S_N(c)$]
\label{def:SN}
$S_N(c):=\sum_{n>N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$\displaystyle
\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c)$.
\end{lemma}
\begin{proof}
Since $\hat g$ is supported in $[-c_0,c_0]$:
$a_n^{(c)}=\int_{-c_0}^{c_0}
\hat g(\xi)\overline{\widehat{\Phi_n^{(c)}}(\xi)}
\,d\xi$.
Cauchy--Schwarz and sum over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness criterion]
\label{cor:equitight}
$\sup_{c}S_N(c)\to 0$
$\Rightarrow$
$\{a^{(c)}\}$ equi-tight in $\ell^2$.
\end{corollary}

\begin{remark}[$S_N(c)$ as operator norm]
\label{rem:SN-operator}
Let $\Pi_N^{(c)}$ denote the projection
onto $\operatorname{span}\{
\Phic{c}{0},\ldots,\Phic{c}{N}\}$.
Since $\norm{\Phic{c}{n}}=1$:
\begin{align}
  S_N(c)
  &= \sum_{n>N}\|\Pc\Phic{c}{n}\|^2
    = \norm{\Pc(I-\Pi_N^{(c)})}^2_{\mathrm{HS}}
    \label{eq:SN-HS}\\
  &= \sup_{\substack{g\in\PW_{c_0}^2\\
    \norm{g}=1}}\sum_{n>N}|a_n^{(c)}|^2
    = \norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}.
    \label{eq:SN-op}
\end{align}
\end{remark}

\begin{remark}[$\sigma(c)$ as a trace]
\label{rem:sigma-trace}
Define $\sigma(c):=S_{-1}(c)
=\sum_{n\ge 0}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.
Since $\{\Phic{c}{n}\}$ is an ONB of $\PW_c^2$,
Parseval on $\PW_c^2$ gives:
\begin{equation}\label{eq:sigma-trace}
  \sigma(c)
  = \norm{\Pc}^2_{\mathrm{HS}(\PW_c^2)}
  = \mathrm{Tr}_{\PW_c^2}(\Pc^*\Pc)
  = \mathrm{Tr}_{\PW_c^2}(P_c P_{c_0} P_c),
\end{equation}
where $P_c$ is the orthogonal projection
onto $\PW_c^2$ in $L^2(\mathbb{R})$.
This is the \emph{trace of the compressed projection}
$P_c P_{c_0} P_c$, which is a well-defined
non-negative trace-class operator on $\PW_c^2$
(since $P_{c_0}|_{\PW_c^2}$ is Hilbert--Schmidt
for $c>c_0$, hence trace-class).
\end{remark}

%%═══ 5. SPECTRAL LEAKAGE BARRIER
\section{The Spectral Leakage Barrier}
\label{sec:barrier}

\subsection{The key estimate: uniform low-index sum}

The core of the barrier is the following lemma,
which is proved rigorously from Paper~VIII alone.

\begin{lemma}[Uniform low-index sum]
\label{lem:low-index-sum}
There exists a constant $c_0'>0$,
depending only on $c_0$, $\kappa$,
and the non-cancellation constants
of Lemma~\ref{lem:inherited}\ref{A:nonvanish},
such that for all sufficiently large $c$:
\[
  \sum_{n=0}^{\lfloor\kappa c\rfloor}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge c_0' > 0.
\]
\end{lemma}
\begin{proof}
We use Lemma~\ref{lem:inherited}\ref{A:nonvanish}:
for each $n\le\kappa c$,
$\widehat{\Phi_n^{(c)}}$ satisfies
$|\widehat{\Phi_n^{(c)}}(\xi)|\ge c_1 c^{-1/2}$
on a set $E_c^{(n)}\subseteq[-c,c]$
with $|E_c^{(n)}|\ge c_* c^{-1/2}$.
The part of $E_c^{(n)}$ lying in $[-c_0,c_0]$ satisfies:
\[
  |E_c^{(n)}\cap[-c_0,c_0]|
  \ge \min\bigl(|E_c^{(n)}|,\,2c_0\bigr)
  \wedge c_*c^{-1/2}.
\]
For $n=0$ specifically, by the
non-vanishing of $\widehat{\Phi_0^{(c)}}$
near the origin (Paper~VIII,
the ground-state $\widehat{\Phi_0^{(c)}}$
is a positive function peaked at $0$):
\[
  \|\widehat{\Phi_0^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge c_1^2 c^{-1} \cdot
  |E_c^{(0)}\cap[-c_0,c_0]|.
\]
Since $\widehat{\Phi_0^{(c)}}$ concentrates
near $\xi=0$ (a consequence of the positivity
of the Slepian ground state; see~\cite{Osipov2013}),
for all large $c$:
$|E_c^{(0)}\cap[-c_0,c_0]|
\ge c_0/2 > 0$ (independent of $c$).
Hence:
\[
  \|\widehat{\Phi_0^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge \frac{c_1^2 c_0}{2c}.
\]
This gives a single term decaying like $c^{-1}$,
which is insufficient for a uniform bound.

To get a uniform bound $\ge c_0'>0$,
we use \emph{multiple} terms in the low-index regime.
By~\eqref{eq:sigma-trace}:
\[
  \sigma(c) = \mathrm{Tr}(P_c P_{c_0} P_c).
\]
By the Weyl law for the Slepian operator
(see~\cite{Slepian1961,Osipov2013}):
the number of eigenvalues $\lambda_n^{(c)}$
near $1$ (``the good eigenvalues'')
is approximately $2cc_0/\pi$ for large $c$.
Each such eigenvalue contributes
$\lambda_n^{(c)}\ge 1/2$ (say) to the trace.
Hence:
\[
  \sigma(c)
  = \mathrm{Tr}(P_c P_{c_0} P_c)
  \ge \sum_{n: \lambda_n^{(c)}\ge 1/2}
    \lambda_n^{(c)}
  \ge \frac{1}{2}\cdot\#
    \{n:\lambda_n^{(c)}\ge 1/2\}
  \sim \frac{c c_0}{\pi}
  \to\infty.
\]
In particular, $\sigma(c)\ge c_0'>0$
for all large $c$,
where $c_0':=\frac{c_0}{\pi}$
(using the Weyl law from~\cite{Slepian1961}).

\emph{Note on self-containedness.}
The Weyl law for Slepian eigenvalues
($\#\{n:\lambda_n^{(c)}\ge 1/2\}\sim 2cc_0/\pi$)
is a classical result of Slepian--Pollak~\cite{Slepian1961}
and does not depend on Papers~VIII or~XII;
it is an input from the standard literature.
The bound $\sigma(c)\ge c_0'>0$ is therefore
a rigorous consequence of this classical result.
\end{proof}

\begin{remark}[Why the single-term bound is insufficient]
\label{rem:single-term}
The single-term estimate
$\|\widehat{\Phi_0^{(c)}}\|^2_{L^2([-c_0,c_0])}
\ge c_1^2 c_0/(2c)\to 0$
does \emph{not} yield a uniform lower bound
for $\sigma(c)$ as $c\to\infty$.
Lemma~\ref{lem:low-index-sum} requires
the collective contribution of
$\sim cc_0/\pi$ eigenvalues,
captured precisely by the trace
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim cc_0/\pi\to\infty$.
This is the correct argument; the
v3 single-term bound was insufficient.
\end{remark}

\subsection{The Spectral Leakage Barrier}

\begin{theorem}[Spectral Leakage Barrier]
\label{thm:barrier}
For each $N\ge 0$ there exists $c(N)\to\infty$
such that
\[
  S_N(c(N)) \ge c_0'' > 0,
\]
where $c_0''>0$ is a constant independent of $N$.
In particular, $\sup_{c>c_0}S_N(c)\not\to 0$.
\end{theorem}
\begin{proof}
By definition,
$S_N(c) = \sigma(c) - \sum_{n=0}^N
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.
Each term in the finite sum satisfies
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\le\|\widehat{\Phi_n^{(c)}}\|^2_{L^2}=1$
(since $\norm{\Phic{c}{n}}=1$).
Hence:
\begin{equation}\label{eq:SN-lower}
  S_N(c) \ge \sigma(c) - (N+1).
\end{equation}
By Lemma~\ref{lem:low-index-sum}
and Remark~\ref{rem:sigma-trace}:
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim\frac{cc_0}{\pi}\to\infty$.
For each fixed $N$, choose
$c(N)$ large enough so that
$\sigma(c(N)) \ge N+1+c_0''$
for some absolute constant $c_0''>0$
(possible since $\sigma(c)\to\infty$).
Then $S_N(c(N))\ge c_0''>0$.
\end{proof}

\begin{remark}[Mechanism: trace grows, tail stays large]
\label{rem:mechanism}
The proof reveals the precise mechanism:
$\sigma(c)\to\infty$ because more and more
Slepian eigenvalues enter the ``good'' regime
as $c$ grows.
For any fixed $N$, the finite prefix
$\sum_{n\le N}(\cdots)$ is bounded by $N+1$,
so $S_N(c)=\sigma(c)-(N+1)$ eventually
exceeds any positive constant.
The Spectral Leakage Barrier is not a failure
of pointwise decay, but a failure of
uniform rate: $\sigma(c)\to\infty$ outruns
any fixed truncation.
\end{remark}

\begin{remark}[Fixed $c$ vs.\ uniform]
\label{rem:fixed-vs-uniform}
For each fixed $c$:
$S_N(c)\to 0$ as $N\to\infty$
(since $\sigma(c)<\infty$ for each $c$).
Theorem~\ref{thm:barrier} shows this
\emph{cannot be uniform in $c$}:
for any choice of $N$,
there exist arbitrarily large $c$
for which $S_N(c)\ge c_0''>0$.
\end{remark}

\subsection{Quantitative barrier (conditional)}

\begin{assumption}[Local Weyl law / Fourier mass]
\label{ass:fourier-mass}
For $\alpha\in(0,1)$ and $n=\lfloor\alpha c\rfloor$:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge \frac{c_0}{c}\cdot\beta(\alpha)
\]
for some $\beta:(0,1)\to(0,\infty)$
bounded away from $0$ on compact subsets of $(0,1)$.
This is a \emph{local Weyl law in frequency}:
$\widehat{\Phi_n^{(c)}}$ equidistributes over $[-c,c]$
at scale $c_0$ as $c\to\infty$
with $n/c\to\alpha$.
See Open Problem~\ref{prob:fourier-mass}.
\end{assumption}

\begin{proposition}[Quantitative $\liminf$]
\label{prop:quant-barrier}
Under Assumption~\ref{ass:fourier-mass}:
$\liminf_{c\to\infty}S_N(c)
\ge c_0\int_0^\kappa\beta(\alpha)\,d\alpha
> 0$
for each fixed $N$.
\end{proposition}
\begin{proof}
$S_N(c)\ge\sum_{N<n\le\kappa c}
\frac{c_0\beta(n/c)}{c}
\to c_0\int_0^\kappa\beta(\alpha)\,d\alpha$.
(Riemann sum with mesh $1/c\to 0$.)
\end{proof}

\subsection{Why existing tools cannot overcome the barrier}

\begin{remark}
\label{rem:intrinsic}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Tool & Controls & Does not control \\
\midrule
Paper~VIII & $n\le\kappa c$ pointwise
  & $\sigma(c)\to\infty$ rate \\
Paper~XII & each fixed $n$
  & any $n\to\infty$ with $c$ \\
Bessel & $\sum|a_n|^2\le\norm{g}^2$
  & equi-tightness \\
Slepian--Weyl & $\sigma(c)\sim cc_0/\pi$
  & \textbf{(used in Lemma~\ref{lem:low-index-sum})} \\
\bottomrule
\end{tabular}
\end{center}
The Spectral Leakage Barrier is a consequence
of the classical Slepian--Weyl law,
not of the new tools in the series;
it is an obstruction that is already
visible in the foundational theory.
\end{remark}

%%═══ 6. HYPOTHESIS H3 AND H1 ⟺ H3
\section{Hypothesis H3 and the Main Equivalence}
\label{sec:H3}

\begin{hypothesis}[H3: Uniform Frequency Evacuation]
\label{hyp:H3}
For each fixed $c_0>0$:
$\sup_{c>c_0}S_N(c)\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{theorem}[H3 $\Rightarrow$ H1 (complete)]
\label{thm:H3-H1}
Assume H3. Then H1 holds.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$.
By H3 and Lemma~\ref{lem:tail},
choose $N$ so that
$\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c)
<\varepsilon/2$ for all $c$.
By Lemma~\ref{lem:inherited}\ref{A:ptwise},
for $c>c_\varepsilon$:
$\sum_{n=0}^N|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$.
Hence $\norm{a^{(c)}-a^{(\infty)}}^2<\varepsilon$.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$ H3
  (conditional on OP~\ref{prob:H1H3})]
\label{thm:H1-H3}
Assume H1. Then $S_N(c)\to 0$
for each fixed $c$ as $N\to\infty$,
and modulo Open Problem~\ref{prob:H1H3},
$\sup_c S_N(c)\to 0$, i.e.\ H3 holds.
\end{theorem}
\begin{proof}
By Remark~\ref{rem:SN-operator},
$S_N(c)=\norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}$.
For each fixed $g\in\PW_{c_0}^2$
with $\norm{g}=1$:
\[
  \sum_{n>N}|a_n^{(c)}|^2
  \le
  \Bigl(\norm{a^{(c)}-a^{(\infty)}}
  +\norm{a^{(\infty)}}\Bigr)^2
  - \sum_{n=0}^N |a_n^{(c)}|^2.
\]
Since $a^{(\infty)}\in\ell^2$:
$\sum_{n>N}|a_n^{(\infty)}|^2\to 0$;
by H1: $\norm{a^{(c)}-a^{(\infty)}}\to 0$
as $c\to\infty$ for fixed $g$.
Hence for each fixed $g$
and $N$ large:
$\sum_{n>N}|a_n^{(c)}|^2\to 0$
as $c\to\infty$.

To pass from pointwise-in-$g$ to
$\sup_{\norm{g}=1}$
(i.e.\ to the operator norm in \eqref{eq:SN-op})
requires \emph{uniform} control:
this is the content of
Open Problem~\ref{prob:H1H3}.
Conditionally, H1$\Rightarrow$H3.
\end{proof}

\begin{remark}[Logical status of the equivalence]
\label{rem:logical}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{ll}
\toprule
Direction & Status \\

%%═══ 7. INTERPRETATION
\section{Interpretation}
\label{sec:interp}

\subsection{Reduction chain}

\[
  \Kspace=L^2
  \overset{\text{XIII}}{\Longleftrightarrow}
  \text{H2}
  \Longleftarrow
  \text{H1}
  \underset{\text{(H3}\Rightarrow\text{H1 complete)}}
  {\overset{\text{XIV}}{\Longleftrightarrow}}
  \text{H3}.
\]

\subsection{The Weyl-law mechanism}

The driving force of the Spectral Leakage Barrier
is the Slepian--Weyl law $\sigma(c)
=\mathrm{Tr}(P_c P_{c_0}P_c)\sim cc_0/\pi$.
This is a classical result expressing that
the number of ``essentially $c_0$-bandlimited
among $c$-bandlimited'' functions grows
linearly with $c$.
For any fixed truncation at index $N$,
this growing trace forces $S_N(c)\to\infty$
rather than $\to 0$.
H3 (= uniform tail control) would require
$\sigma(c)$ to be controlled by $N$,
which contradicts $\sigma(c)\to\infty$.

\subsection{Transition zone and semiclassics}

The transition zone $n\sim\alpha c$
(Definition~\ref{def:transition})
is the regime where:
$\lambda_n^{(c)}$ transitions
from $\approx 1$ (for $n\ll\kappa c$)
to $\approx 0$ (for $n\gg\kappa c$).
This is the \emph{spectral phase transition}
of the Slepian operator,
studied in semiclassical terms via
WKB analysis~\cite{Zworski2012}.
H3 requires understanding the
Fourier mass of $\widehat{\Phi_n^{(c)}}$
precisely in this transition regime,
which is the local Weyl law of
Assumption~\ref{ass:fourier-mass}.

\subsection{Non-compactness mechanism}

In operator terms:
$S_N(c)=\norm{\Pc(I-\Pi_N^{(c)})}^2_{\mathrm{HS}}$.
H3 fails because the family
$\{\Pc(I-\Pi_N^{(c)})\}_{c>c_0}$
is not equi-small in HS-norm:
for any $N$, there are arbitrarily large $c$
with $S_N(c)\ge c_0''>0$.
This is a genuine non-compactness
phenomenon driven by $\sigma(c)\to\infty$:
the ``effective dimension''
$\mathrm{Tr}(P_c P_{c_0}P_c)$ of the
interaction between the two Paley--Wiener
spaces grows without bound.

%%═══ 8. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law for PSWFs]
\label{prob:fourier-mass}
\emph{(Priority: highest.)}
Prove Assumption~\ref{ass:fourier-mass}:
for $n=\lfloor\alpha c\rfloor$,
$\alpha\in(0,1)$ fixed:
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\sim c_0/c$ as $c\to\infty$.
Natural approach: WKB / semiclassical analysis
of PSWFs at $n\sim\alpha c$;
cf.~\cite{Osipov2013,Zworski2012}.
\end{problem}

\begin{problem}[Uniformity: H1 $\Rightarrow$ H3]
\label{prob:H1H3}
\emph{(Priority: high.)}
Show that H1 (for each fixed $g$)
implies $S_N(c)\to 0$ uniformly in $c$.
Equivalently:
$\norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}
\to 0$ uniformly in $c$.
Natural approach: Banach--Steinhaus
(uniform boundedness of a family of linear maps
$g\mapsto\sum_{n>N}|a_n^{(c)}|^2$)
or compactness of the unit ball
of $\PW_{c_0}^2$ under Fourier restriction.
\end{problem}

\begin{problem}[Transition zone asymptotics]
\label{prob:transition}
\emph{(Priority: medium.)}
For $\alpha\in(0,1)$ fixed,
determine the asymptotic Fourier distribution
of $\Phic{c}{\lfloor\alpha c\rfloor}$
as $c\to\infty$.
Is there equidistribution
(local Weyl law in frequency)?
\end{problem}

\begin{problem}[H3 and completeness]
\label{prob:H3-completeness}
\emph{(Priority: medium.)}
Does H3 $\Rightarrow$ H2$_{c_0}$
for fixed $c_0$?
Does H3 (for all $c_0$)
$\Rightarrow$ $\Kspace=L^2$?
\end{problem}

\textbf{Central conclusion.}
\emph{The Spectral Leakage Barrier is a rigorous
consequence of the classical Slepian--Weyl law
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)\sim cc_0/\pi$.
For any fixed $N$, $S_N(c)\ge\sigma(c)-(N+1)\to\infty$.
H3 (= uniform tail control = $\sup_c S_N(c)\to 0$)
fails because $\sigma(c)\to\infty$.
H3$\Rightarrow$H1 is complete.
The resolution of H1 requires a local Weyl law
for PSWFs in the transition zone $n\sim\alpha c$,
which is the natural entry point for Paper~XV.}
