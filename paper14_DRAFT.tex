\begin{abstract}
This paper analyses Hypothesis~H1 of Paper~XIII:
the $\ell^2$-norm convergence of Slepian coefficients
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ as $c\to\infty$.
We do not prove or disprove H1,
but reduce it precisely to a spectral
distribution problem in the transition zone
$n\sim\alpha c$.

The \emph{cross-Gram operator}
$T_c g := \sum_n \ip{g}{\Phic{c}{n}}\Phiinf{n}$
satisfies $\norm{T_c}\le 1$ uniformly.
The \emph{tail functional}
$S_N(c):=\sum_{n>N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$
equals the operator norm
$\norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}$,
so H3 ($\sup_c S_N(c)\to 0$)
is equivalent to SOT-convergence
$\Pi_N^{(c)}\Pc\to\Pc$, uniformly in $c$.
H3$\Rightarrow$H1 is proved unconditionally;
H1$\Rightarrow$H3 is Open Problem~\ref{prob:H1H3}.

The \emph{No-Concentration condition} (NC)
asserts that Fourier mass in a proportional
band $n\in(\alpha c,\kappa c)$ of transition-zone
indices is bounded below by $c_*>0$, uniformly in $c$.
We prove: NC$\Rightarrow$H3 fails.
The converse (NC fails$\Rightarrow$H3) is \emph{not proved};
it is an additional logical gap, noted explicitly
in Theorem~\ref{thm:reduction}.

The full conditional chain is therefore:
\begin{equation*}
  \underbrace{\text{local Weyl law}}_{
    \text{OP}\,\ref{prob:fourier-mass}}
  \Rightarrow \text{NC}
  \Rightarrow \text{H3 fails}
  \overset{\text{OP}\,\ref{prob:H1H3}}
  \Rightarrow \text{H1 fails.}
\end{equation*}

\medskip\noindent
\textbf{Central result} (honest summary):
\emph{H1 is connected to NC via a chain with
two conditional steps (OPs~\ref{prob:H1H3},
\ref{prob:fourier-mass}) and one missing
implication (NC$^c\Rightarrow$H3;
OP~\ref{prob:NC-H3}).
Paper~XIV provides all structural tools
and isolates all logical gaps precisely.
Paper~XV addresses the local Weyl law.}
\end{abstract}

\subsection{Context}

Paper~XIII identified H1 as the key remaining

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$
($c\to\infty$).
\end{hypothesis}

H1 splits:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
\eqref{eq:H1ip} is proved in
Proposition~\ref{prop:H1ip};
\eqref{eq:H1norm} is equivalent to H3
(given OP~\ref{prob:H1H3}).

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $\alpha\in(0,1)$ and $\omega:(0,\infty)
\to(0,\infty)$ with
$\omega(c)\to\infty$, $\omega(c)=o(c)$:
$\mathcal{T}_{\alpha,\omega}(c):=
\{n:|n-\alpha c|\le\omega(c)\}$.
For NC (Definition~\ref{def:NC}),
the relevant band is proportional:
$n\in(\alpha c,\kappa c)$.
\end{definition}

\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
\label{def:crossgram}
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{theorem}[$\norm{T_c}\le 1$]
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{theorem}
\begin{proof}
Bessel: $\norm{T_c g}^2
=\sum_n|\ip{g}{\Phic{c}{n}}|^2\le\norm{g}^2$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds.
\end{proposition}
\begin{proof}
Bessel + Paper~XII:
$a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$.
\end{proof}

\begin{remark}[H1 reduces to \eqref{eq:H1norm}]
\label{rem:H1-reduces}
Weak lsc:
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$,
so H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}\le\norm{a^{(\infty)}}$.
\end{remark}

%%═══ 4. TAIL REDUCTION AND OPERATOR IDENTITY
\section{Tail Reduction and Operator Identity}
\label{sec:tail}

\begin{definition}[$S_N(c)$, $\sigma(c)$]
\label{def:SN}
\[
  S_N(c):=\sum_{n>N}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])},
  \quad
  \sigma(c):=\sum_{n\ge 0}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}.
\]
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$\displaystyle\sum_{n>N}|a_n^{(c)}|^2
\le\norm{g}^2\cdot S_N(c).$
\end{lemma}
\begin{proof}
Paley--Wiener: $\hat g$ supported on $[-c_0,c_0]$,
so
$a_n^{(c)}=\ip{\hat g}{
  \widehat{\Phi_n^{(c)}}\cdot\mathbf{1}_{[-c_0,c_0]}}_{L^2}$.
Cauchy--Schwarz:
$|a_n^{(c)}|^2
\le\norm{\hat g}^2_{L^2}
\cdot\norm{\widehat{\Phi_n^{(c)}}}^2_{L^2([-c_0,c_0])}$.
Summing over $n>N$ and using
$\norm{\hat g}_{L^2}=\norm{g}_{L^2}$:
$\sum_{n>N}|a_n^{(c)}|^2
\le\norm{g}^2 S_N(c)$.
\end{proof}

\begin{corollary}[Equi-tightness]
\label{cor:equitight}
$\sup_c S_N(c)\to 0\Rightarrow
\{a^{(c)}\}$ equi-tight in $\ell^2$.
\end{corollary}

\begin{theorem}[$S_N(c)$ as operator norm;
H3 as SOT-convergence]
\label{thm:SN-operator}
Let $\Pi_N^{(c)}$ be the orthogonal
projection onto
$\operatorname{span}\{\Phic{c}{0},\ldots,\Phic{c}{N}\}$.
Define the operator on $\PW_{c_0}^2$:
\begin{equation}\label{eq:op-explicit}
  A_N^{(c)} := \Pc(I-\Pi_N^{(c)})
  = \sum_{n>N}\ip{\cdot}{\Phic{c}{n}}_{\PW_c^2}
  \Pc\Phic{c}{n}.
\end{equation}
Then:
\begin{equation}\label{eq:SN-HS}
  S_N(c)
  = \norm{A_N^{(c)}}^2_{\mathrm{HS}}
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2},
\end{equation}
and (since $A_N^{(c)}$ has orthogonal range components):
\begin{equation}\label{eq:SN-op}
  \norm{A_N^{(c)}}^2_{\mathrm{op}}
  = \sup_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}
  \le S_N(c).
\end{equation}
Moreover:
\begin{equation}\label{eq:SN-sup-g}
  \sup_{\substack{g\in\PW_{c_0}^2\\
    \norm{g}=1}}
  \sum_{n>N}|\ip{g}{\Phic{c}{n}}|^2
  = \norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}
  = S_N(c).
\end{equation}
\eqref{eq:SN-sup-g} follows because:
\begin{equation}\label{eq:SN-proof-step}
  \sum_{n>N}|\ip{g}{\Phic{c}{n}}|^2
  = \sum_{n>N}|\ip{\Pc g}{\Phic{c}{n}}|^2
  = \norm{(I-\Pi_N^{(c)})\Pc g}^2,
\end{equation}
so the supremum over $\norm{g}=1$ is
$\norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}$;
and expanding the HS-norm of
$(I-\Pi_N^{(c)})\Pc$
in the ONB $\{\Phic{c}{n}\}$
gives $\sum_{n>N}\norm{\Pc\Phic{c}{n}}^2=S_N(c)$.

\medskip
Hence H3 ($\sup_c S_N(c)\to 0$) is equivalent to:
\begin{equation}\label{eq:H3-SOT}
  \norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}
  \to 0
  \quad\text{uniformly in }c,
  \quad\text{i.e.}\quad
  \Pi_N^{(c)}\Pc\xrightarrow{\text{SOT,unif.}\,c}\Pc.
\end{equation}
\end{theorem}

\begin{remark}[HS vs.\ operator norm]
\label{rem:HS-vs-op}
Note that $\norm{A}^2_{\mathrm{op}}
\le\norm{A}^2_{\mathrm{HS}}$ in general.
Equation~\eqref{eq:SN-sup-g} shows
that the supremum-over-$g$ formula
recovers the HS-norm $S_N(c)$,
not the operator norm~\eqref{eq:SN-op}.
This is because the components
$\Pc\Phic{c}{n}$ are \emph{not}
orthogonal in $L^2$:
the Gram matrix
$G_{mn}=\ip{\Pc\Phic{c}{m}}{\Pc\Phic{c}{n}}$
is not diagonal.
The identification $S_N(c)=
\norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}$
holds via~\eqref{eq:SN-proof-step}:
the sup-over-$g$ is the operator norm
of $(I-\Pi_N^{(c)})\Pc$,
which equals the HS-norm by~\eqref{eq:SN-HS}
\emph{only because}
the vectors $\{\Phic{c}{n}\}_{n>N}$
are orthonormal in $\PW_c^2$
(not in $L^2$).
\end{remark}

\begin{proposition}[Trace identity and growth]
\label{prop:trace}
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim cc_0/\pi\to\infty$.
\end{proposition}
\begin{proof}
$P_{c_0}|_{\PW_c^2}$ is HS for $c>c_0$,
so $P_c P_{c_0}P_c$ is trace-class;
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)$.
Slepian--Weyl \cite{Slepian1961}:
$\sigma(c)\sim cc_0/\pi$.
\end{proof}

\begin{remark}[Counterexample and PSWF-incompatibility]
\label{rem:counterexample}
\emph{Abstract.}
$\sigma(c)\to\infty$ does not imply
$S_N(c)\not\to 0$:
set $b_n^{(c)}=0$ for $n>N_0$,
$\sum_{n\le N_0}b_n^{(c)}=\sigma(c)\to\infty$.
Then $S_N(c)=0$ for $N\ge N_0$.

\emph{Why PSWFs avoid this.}
(i) The Slepian--Weyl law shows
$\sim cc_0/\pi$ eigenvalues
$\lambda_n^{(c)}$ near $1$,
indexed by a \emph{growing} set
(not a fixed $N_0$).
(ii) Paper~VIII gives
$|\widehat{\Phi_n^{(c)}}|\ge c_1c^{-1/2}$
on a set of measure $\ge c_*c^{-1/2}$
for each $n\le\kappa c$;
this prevents mass from vacating
the transition zone entirely.
The abstract scenario is therefore
PSWF-incompatible, but NC is still
required as a formal condition;
it holds under the local Weyl law
(OP~\ref{prob:fourier-mass}).
\end{remark}

%%═══ 5. THE NO-CONCENTRATION CONDITION
\section{The No-Concentration Condition}
\label{sec:NC}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
Fix $0<\alpha<\kappa<1$.
NC holds if there exists $c_*>0$ such that
for all sufficiently large $c$:
\begin{equation}\label{eq:NC}
  \sum_{\alpha c < n\le\kappa c}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge c_* > 0.
\end{equation}
The proportional band $(\alpha c,\kappa c)$
has $\sim(\kappa-\alpha)c$ indices;
$c_*$ is independent of $c$.
\end{definition}

\begin{remark}[Why proportional width is necessary]
\label{rem:prop-width}
A narrow window $|n-\alpha c|\le c^\beta$
($\beta<1$) has only $\sim 2c^\beta$ indices.
Under the local Weyl law~\eqref{eq:local-Weyl},
each contributes $\sim c_0/c$,
so the window sum is $\sim 2c_0 c^{\beta-1}\to 0$.
This is \emph{insufficient} for NC.
The correct band requires proportional width:
$(\kappa-\alpha)c$ indices,
each contributing $\sim c_0/c$,
yielding total $\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[Local Weyl law $\Rightarrow$ NC]
\label{rem:Weyl-NC}
If for $n=\lfloor\alpha c\rfloor$:
\begin{equation}\label{eq:local-Weyl}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim \frac{c_0}{c}
  \quad(c\to\infty),
\end{equation}
then by a Riemann sum argument:
$\sum_{\alpha c<n\le\kappa c}\|\cdots\|^2
\sim c_0\int_\alpha^\kappa d\alpha
=c_0(\kappa-\alpha)>0$.
So NC holds with
$c_*=c_0(\kappa-\alpha)/2$.
\end{remark}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under NC: for each fixed $N$,
$\liminf_{c\to\infty}S_N(c)\ge c_*>0$,
so $\sup_c S_N(c)\not\to 0$,
i.e.\ H3 fails.
\end{proposition}
\begin{proof}
For large $c$: $\alpha c>N$,
so $(\alpha c,\kappa c)\subseteq\{n>N\}$.
Hence $S_N(c)\ge
\sum_{\alpha c<n\le\kappa c}\|\cdots\|^2\ge c_*$.
\end{proof}

\begin{remark}[NC$^c\Rightarrow$H3: not proved]
\label{rem:NC-converse}
Proposition~\ref{prop:barrier} gives
NC$\Rightarrow$H3 fails.
The converse
(NC fails $\Rightarrow$ H3 holds)
is \emph{not proved here}.
NC is sufficient for the barrier
but may not be necessary:
H3 might fail for other reasons,
or H3 might hold even if NC is
nearly violated.
The precise relationship between
$\neg$NC and H3
is Open Problem~\ref{prob:NC-H3}.
\end{remark}

%%═══ 6. H3 AND THE MAIN REDUCTION
\section{H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)\xrightarrow{N\to\infty}0$;
equivalently:
$\Pi_N^{(c)}\Pc\to\Pc$
SOT, uniformly in $c$.
\end{hypothesis}

\begin{theorem}[H3 $\Rightarrow$ H1]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$, $g\in\PW_{c_0}^2$.
H3: choose $N$ with
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$,
so $\sum_{n>N}|a_n^{(c)}|^2<\varepsilon/2$ (all $c$).
Paper~XII: choose $c_\varepsilon$ with
$\sum_{n\le N}|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ ($c>c_\varepsilon$).
Combine.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$
  tail control for fixed $g$]
\label{thm:H1-H3-ptwise}
Assume H1. For each fixed $g$:
$\sum_{n>N}|a_n^{(c)}|^2\to 0$
as $N\to\infty$, uniformly in $c$
(for that $g$).
\end{theorem}
\begin{proof}
$\sum_{n>N}|a_n^{(c)}|^2
\le 2\norm{a^{(c)}-a^{(\infty)}}^2
+2\sum_{n>N}|a_n^{(\infty)}|^2\to 0$.
\end{proof}

\begin{remark}[The two gaps in H1$\Rightarrow$H3]
\label{rem:two-gaps}
Theorem~\ref{thm:H1-H3-ptwise} gives
$\sum_{n>N}|a_n^{(c)}(g)|^2\to 0$
for each fixed $g$.
Passing to H3 requires:
\[
  \sup_{\norm{g}=1}\sum_{n>N}|a_n^{(c)}|^2
  = S_N(c)\to 0\quad\text{uniformly in }c.
\]
This requires interchanging
$\sup_g$ and $\lim_N$,
which is a quadratic-form uniform
boundedness problem (not directly
linear Banach--Steinhaus).
This is Open Problem~\ref{prob:H1H3}.
\end{remark}

\begin{theorem}[Main reduction
  (conditional equivalences)]
\label{thm:reduction}
The following implications are established:
\begin{align}
  \text{H3}&\Rightarrow\text{H1}
    \tag{proved; Thm.~\ref{thm:H3-H1}}
    \label{tag:H3H1}\\
  \text{H1}&\Rightarrow
    S_N(c)\to 0\text{ (fixed }g\text{)}
    \tag{proved; Thm.~\ref{thm:H1-H3-ptwise}}
    \label{tag:H1ptwise}\\
  \text{H1}&\Rightarrow\text{H3}
    \tag{OP~\ref{prob:H1H3}}
    \label{tag:H1H3}\\
  \text{NC}&\Rightarrow\text{H3 fails}
    \tag{proved; Prop.~\ref{prop:barrier}}
    \label{tag:NCH3}\\
  \text{local Weyl law}
    &\Rightarrow\text{NC}
    \tag{proved; Remark~\ref{rem:Weyl-NC}}
    \label{tag:WeylNC}\\
  \text{NC}^c
    &\Rightarrow\text{H3}
    \tag{\textbf{not proved};
      OP~\ref{prob:NC-H3}}
    \label{tag:NCDH3}
\end{align}
The conditional chain in one direction:
\[
  \underbrace{\text{local Weyl law}}_{
    \text{OP}\,\ref{prob:fourier-mass}}
  \xRightarrow{(\ref{tag:WeylNC})}
  \text{NC}
  \xRightarrow{(\ref{tag:NCH3})}
  \text{H3 fails}
  \xRightarrow{\text{OP}\,\ref{prob:H1H3}}
  \text{H1 fails.}
\]
The other direction:
\[
  \text{H3 holds}
  \xRightarrow{(\ref{tag:H3H1})}
  \text{H1}.
\]
\emph{The full equivalence}
``H1 $\Leftrightarrow$ H3 $\Leftrightarrow$ NC$^c$''
\emph{requires additionally:
(i)~OP~\ref{prob:H1H3} (H1$\Rightarrow$H3)
and (ii)~OP~\ref{prob:NC-H3}
(NC$^c\Rightarrow$H3).
Neither is proved here.}
\end{theorem}

%%═══ 7. INTERPRETATION
\section{Interpretation and Context}
\label{sec:interp}

\subsection{The reduction hierarchy}

\[
  \underbrace{\text{H1}}_{
    \ell^2\text{-convergence}}
  \xleftrightarrow{\text{(conditional)}}
  \underbrace{\text{H3}}_{
    \text{operator SOT}}
  \xleftrightarrow{\text{(conditional)}}
  \underbrace{\text{NC}^c}_{
    \text{spectral distribution}}
  \xleftrightarrow{\text{(OP XV)}}
  \underbrace{\text{local Weyl law}}_{
    \text{semiclassics}}.
\]
Each arrow is either proved
or reduced to a single, precisely stated
open problem.

\subsection{From functional analysis to semiclassics}

H1 is a functional-analytic statement;
H3 is operator-theoretic
(SOT-convergence, uniform in $c$);
NC is a spectral-distribution statement;
the local Weyl law is semiclassical.
This hierarchy mirrors the standard
progression in the theory of
pseudodifferential operators
\cite{Zworski2012}.

\subsection{Relation to quantum ergodicity}

NC is formally analogous to
\emph{quantum ergodicity}:
$\widehat{\Phi_n^{(c)}}$ should
equidistribute over $[-c,c]$ at scale $c_0$
as $n/c\to\alpha$.
In the standard QE setting,
one studies position-space equidistribution
of Laplacian eigenfunctions.
Here the relevant phase space is
one-dimensional (the frequency interval $[-c,c]$)
and the equidistribution is in frequency.
The relevant object is a \emph{semiclassical measure}

%%═══ 8. PROGRAMME FOR PAPER XV
\section{Programme for Paper~XV:
The Local Weyl Law in the Transition Zone}
\label{sec:PaperXV}

Paper~XV must establish (or refute)
the local Weyl law~\eqref{eq:local-Weyl}.
We describe the natural approach.

\subsection{Semiclassical scaling}

Set $h:=1/c\to 0$.
The index $n\sim\alpha c$ corresponds to
energy level $\alpha$ in the
semiclassical limit $h\to 0$.
The PSWF $\Phi_n^{(c)}$ solves
(in $x\in[-1,1]$):
\begin{equation}\label{eq:PSWF-ODE}
  -(1-x^2)\psi'' + 2x\psi'
  + c^2 x^2\psi = \chi_n^{(c)}\psi,
\end{equation}
which in the semiclassical variable $h=1/c$ reads:
\[
  -h^2(1-x^2)\partial_x^2\psi
  +2hx\partial_x\psi
  + x^2\psi = E_n^{(h)}\psi,
\]
where $E_n^{(h)}:=\chi_n^{(c)}/c^2$.
This is a Schr\"odinger-type operator
with effective potential $V(x)=x^2$
and semiclassical parameter $h$.

\subsection{WKB in the transition zone}

For $n\sim\alpha c$,
the energy $E_n^{(h)}$ is near
a specific value $E_\alpha$ determined by
the semiclassical quantisation condition.
The \emph{classical turning points}
in position space are at $x_0$ where
$V(x_0)=E_\alpha$,
i.e.\ $x_0^2=E_\alpha$.

The WKB ansatz for $\psi(x)$:
\begin{itemize}
\item Classical region $|x|<x_0$:
  $\psi(x)\sim A(x)\cos(S(x)/h+\varphi)$
  with amplitude $A(x)\sim(E_\alpha-x^2)^{-1/4}$.
\item Evanescent region $|x|>x_0$:
  $\psi(x)\sim B(x)e^{-|T(x)|/h}$,
  exponentially small.
\end{itemize}

\subsection{Fourier mass via stationary phase}

The Fourier transform
$\widehat{\Phi_n^{(c)}}(\xi)
=\int_{-1}^1 \Phi_n^{(c)}(x)e^{-i\xi x}dx$
(restricted to $[-c_0,c_0]$)
can be analysed by stationary phase
as $c\to\infty$ (i.e.\ $h\to 0$).
The stationary points are at
$x=\partial_\xi S(\xi)=\xi/c$ (roughly).
For $\xi\in[-c_0,c_0]$ with $c_0<c$:
the stationary point lies in the
classical region $|x|<x_0$,
so the Fourier mass is controlled
by the WKB amplitude:
\[
  |\widehat{\Phi_n^{(c)}}(\xi)|^2
  \sim \frac{C}{c}\cdot
  \frac{1}{\sqrt{E_\alpha-(\xi/c)^2}}
  \sim \frac{C}{c}
  \quad(\xi\in[-c_0,c_0],\; c_0\ll c).
\]
Integrating over $\xi\in[-c_0,c_0]$:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim \frac{C\cdot 2c_0}{c}
  = \frac{2Cc_0}{c},
\]
which is exactly the local Weyl law~\eqref{eq:local-Weyl}
(with constant $C$ depending on $E_\alpha$).

\subsection{Mathematical programme}

Paper~XV should:
\begin{enumerate}
\item Make the WKB analysis rigorous:
  control error terms in the
  semiclassical limit $h\to 0$
  at the energy level $E_\alpha$;
  cf.\ Osipov--Rokhlin--Xiao
  \cite{Osipov2013}.
\item Prove the Fourier-mass bound:
  $\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim c_0/c$ for $n=\lfloor\alpha c\rfloor$.
\item Conclude NC
  (by Remark~\ref{rem:Weyl-NC})
  and Spectral Leakage
  (by Proposition~\ref{prop:barrier}).
\item Address OP~\ref{prob:H1H3}
  (H1$\Rightarrow$H3) and
  OP~\ref{prob:NC-H3}
  (NC$^c\Rightarrow$H3)
  to close the equivalence.
\end{enumerate}

%%═══ 9. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law;
  Paper~XV, Step~1]
\label{prob:fourier-mass}
\emph{(Priority: critical.)}
Prove~\eqref{eq:local-Weyl}:
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\sim c_0/c$
for $n=\lfloor\alpha c\rfloor$.
Approach: WKB + stationary phase
(\S\ref{sec:PaperXV});
cf.\ \cite{Osipov2013,Zworski2012}.
\end{problem}

\begin{problem}[H1$\Rightarrow$H3;
  quadratic uniformity]
\label{prob:H1H3}
\emph{(Priority: high.)}
Show:
$\norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}\to 0$
uniformly in $c$, given H1.
Note: the map $g\mapsto\sum_{n>N}|a_n^{(c)}|^2$
is a quadratic form;
standard linear Banach--Steinhaus
does not directly apply.
\end{problem}

\begin{problem}[NC$^c\Rightarrow$H3;
  missing converse]
\label{prob:NC-H3}
\emph{(Priority: high.)}
If NC fails (no persistent mass
in any proportional transition band),
does H3 hold?
This closes the logical equivalence
H3 $\Leftrightarrow$ NC$^c$
and together with OP~\ref{prob:H1H3}
would give H1 $\Leftrightarrow$ NC$^c$.
\end{problem}

\begin{problem}[NC from Papers~VIII, XII]
\label{prob:NC-direct}
\emph{(Priority: medium.)}
Is NC provable from the pointwise
bounds of Paper~VIII alone,
without full WKB asymptotics?
\end{problem}
