\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis under the SOT-Limit:\\
  Cross-Gram Operators and the Spectral Leakage Barrier}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
This paper analyses Hypothesis~H1 of Paper~XIII:
the $\ell^2$-norm convergence of Slepian coefficients
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ as $c\to\infty$,
for $g \in \PW_{c_0}^2$.

We introduce the \emph{cross-Gram operator}
$T_c g := \sum_n \ip{g}{\Phic{c}{n}}\Phiinf{n}$,
show $\norm{T_c}\le 1$ uniformly,
and reduce H1 to a single analytic condition:
uniform tail control of
$S_N(c) := \sum_{n>N}\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.

We then establish the \emph{Spectral Leakage Barrier}:
although $S_N(c) \to 0$ for each fixed $c$,
the uniform control $\sup_c S_N(c) \to 0$ fails.
Specifically, we prove (rigorously) that
for each $N$ there exists $c(N)\to\infty$ with
$S_N(c(N)) \ge c_* > 0$,
and (via explicit Fourier mass estimates)
that $\liminf_{c\to\infty} S_N(c) \ge \kappa c_0 > 0$.
The mechanism is \emph{persistent low-frequency injection}:
indices $n \sim \alpha c$ (the \emph{transition zone})
continuously supply Fourier mass in $[-c_0,c_0]$
as $c\to\infty$.

We isolate the missing property as
\textbf{H3 (Uniform Frequency Evacuation)}
and prove the main equivalence
$\text{H1}\Longleftrightarrow\text{H3}$,
including the direction H1$\Rightarrow$H3
via a test-function argument.
H3 is not implied by any result in Papers~I--XIII.

\medskip
\noindent
\textbf{Central structural result} (informal):
\emph{The coefficient stability problem~H1 is equivalent
to a uniform evacuation of low-frequency mass from
high-index PSWFs.
This property fails by a persistent spectral leakage
from the transition zone $n\sim c$,
which lies beyond the reach of the fixed-index
or low-index analytic tools of the series.}
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Inheritance from Paper XIII}

Paper~XIII established that all frame-based routes to
completeness $\Kspace=L^2(\mathbb{R})$ are closed,
and identified \textbf{H1} as the only currently
identified hypothesis strictly weaker than completeness
and grounded in the series.
Paper~XIV takes H1 as its sole object of study.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1: coefficient stability]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
\[
  \norm{a^{(c)}-a^{(\infty)}}_{\ell^2} \to 0
  \quad(c\to\infty),
\]
where $a_n^{(c)}:=\ip{g}{\Phic{c}{n}}$,
$\;a_n^{(\infty)}:=\ip{g}{\Phiinf{n}}$.
\end{hypothesis}

\noindent
By the Hilbert-space identity
$\norm{x-y}^2 = \norm{x}^2+\norm{y}^2
-2\,\mathrm{Re}\ip{x}{y}$,
H1 splits into:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to \norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to \norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
Condition~\eqref{eq:H1ip} follows from
weak $\ell^2$-convergence
(pointwise from Paper~XII + Bessel bound).
The hard part is \eqref{eq:H1norm}.

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $\alpha\in(0,1)$ and $c>0$, the
\emph{transition zone at rate $\alpha$} is
\[
  \mathcal{T}_\alpha(c) :=
  \{\, n \in \mathbb{N}_0 : n = \lfloor\alpha c\rfloor \pm O(c^{1/2})\,\}.
\]
Indices in $\mathcal{T}_\alpha(c)$ are simultaneously:
\emph{high} relative to any fixed $N$ (for large $c$),
and \emph{low} relative to $c$ (regime of Paper~VIII).
\end{definition}

\begin{center}

Core structural result (informal).\\
\normalfont
Uniform tail control fails because indices in the
transition zone $n\sim\alpha c$
inject Fourier energy into every fixed band
$[-c_0,c_0]$ as $c\to\infty$.
No fixed-$n$ or low-index tool can control this.}}
\end{center}

%%═══ 2. INHERITED DATA
\section{Inherited Data}
\label{sec:input}

\begin{lemma}[From series]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item\label{A:bessel}
  \emph{Bessel.} $\sum_n|a_n^{(c)}|^2\le\norm{g}^2$
  for all $c$.
\item\label{A:ptwise}
  \emph{Paper~XII (cond.\ Gap-S+SOT-Faster).}
  $a_n^{(c)}\to a_n^{(\infty)}$ for each fixed $n$.
\item\label{A:nonvanish}
  \emph{Paper~VIII.}
  For $n\le\kappa c$:
  $|\widehat{\Phi_n^{(c)}}(\xi)|\ge c_1 c^{-1/2}$
  for $\xi\in E_c$, $|E_c|\ge c_* c^{-1/2}$.
\item\label{A:supp}
  \emph{Paley--Wiener.}
  $g\in\PW_{c_0}^2\Rightarrow
  \mathrm{supp}\,\hat g\subseteq[-c_0,c_0]$.
\end{enumerate}
\end{lemma}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
$T_c g := \sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{proposition}
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{proposition}
\begin{proof}
$\norm{T_c g}^2=\sum_n|\ip{g}{\Phic{c}{n}}|^2
\le\norm{g}^2$ by Bessel.
\end{proof}

\begin{lemma}[Cross-Gram identity]
\label{lem:crossgram-ip}
$\ip{T_c g}{g}=\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}.$
\end{lemma}
\begin{proof}
Direct: $\ip{T_c g}{g}
=\sum_n\ip{g}{\Phic{c}{n}}\ip{\Phiinf{n}}{g}
=\sum_n a_n^{(c)}\overline{a_n^{(\infty)}}$.
\end{proof}

\begin{proposition}[{\eqref{eq:H1ip} from weak convergence}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds.
\end{proposition}
\begin{proof}
By Lemma~\ref{lem:inherited}\ref{A:ptwise},
$a_n^{(c)}\to a_n^{(\infty)}$ pointwise.
By Lemma~\ref{lem:inherited}\ref{A:bessel},
$\{a^{(c)}\}$ is bounded in $\ell^2$.
Hence $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$
(pointwise convergence + uniform bound
$\Rightarrow$ weak convergence in $\ell^2$).
Since $a^{(\infty)}\in\ell^2$:
$\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
\to\norm{a^{(\infty)}}^2_{\ell^2}$.
\end{proof}

\begin{remark}
H1 therefore reduces entirely to~\eqref{eq:H1norm},
i.e.\ to $\limsup_c\norm{a^{(c)}}_{\ell^2}
\le\norm{a^{(\infty)}}_{\ell^2}$.
(The $\liminf$ direction follows from weak lower
semi-continuity of the norm.)
\end{remark}

%%═══ 4. TAIL REDUCTION
\section{Tail Reduction}
\label{sec:tail}

\begin{definition}[$S_N(c)$]
\label{def:SN}
$S_N(c):=\sum_{n>N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}.$
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$\displaystyle\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c).$
\end{lemma}
\begin{proof}
$a_n^{(c)}=\int_{-c_0}^{c_0}\hat g(\xi)
\overline{\widehat{\Phi_n^{(c)}}(\xi)}\,d\xi$
by Lemma~\ref{lem:inherited}\ref{A:supp}.
Cauchy--Schwarz:
$|a_n^{(c)}|^2\le\norm{g}^2
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.
Sum over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness criterion]
\label{cor:equitight}
$\sup_{c>c_0}S_N(c)\to 0$
$\Rightarrow$ $\{a^{(c)}\}$ equi-tight in $\ell^2$.
\end{corollary}

%%═══ 5. SPECTRAL LEAKAGE BARRIER
\section{The Spectral Leakage Barrier}
\label{sec:barrier}

\subsection{Weak barrier (rigorous)}

\begin{theorem}[Spectral Leakage Barrier, weak form]
\label{thm:barrier}
For each $N\ge 0$ there exists $c(N)\to\infty$
such that
\[
  S_N(c(N)) \ge c_* > 0,
\]
where $c_*$ depends only on $c_0$ and the
concentration constants of Paper~VIII.
In particular,
$\sup_{c>c_0} S_N(c)\not\to 0$.
\end{theorem}
\begin{proof}
Fix $N$. By Lemma~\ref{lem:inherited}\ref{A:bessel},
$\sum_{n=0}^\infty
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
= \|\Pc\|^2_{\mathrm{HS}(\PW_c^2)}$.
Since $\{\Phic{c}{n}\}$ is an ONB of $\PW_c^2$
(for finite $c$, Slepian~\cite{Slepian1961}),
and $\Pc:\PW_c^2\to\PW_{c_0}^2$ is a non-zero
partial isometry for $c>c_0$,
we have
\begin{equation}\label{eq:total-SN}
  \sum_{n=0}^\infty
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  = \|\Pc\|^2_{\mathrm{HS}(\PW_c^2)}.
\end{equation}
We claim $\liminf_{c\to\infty}\|\Pc\|^2_{\mathrm{HS}(\PW_c^2)}>0$.
Indeed, by Lemma~\ref{lem:inherited}\ref{A:nonvanish},
for $n=0$ and all large $c$:
$\|\widehat{\Phi_0^{(c)}}\|^2_{L^2([-c_0,c_0])}
\ge c_1^2 c^{-1}\cdot |E_c\cap[-c_0,c_0]|$.
For $c>c_0$, we have $E_c\subseteq[-c,c]$ and
$|E_c\cap[-c_0,c_0]|\ge\min(|E_c|, 2c_0)
\ge c_* c^{-1/2}\wedge 2c_0>0$.
So at least the $n=0$ term contributes a
positive amount to \eqref{eq:total-SN}.
Write $\sigma(c):=\|\Pc\|^2_{\mathrm{HS}(\PW_c^2)}$;
then $\liminf_{c\to\infty}\sigma(c)\ge c_{**}>0$.
Since $S_N(c)=\sigma(c)-\sum_{n=0}^N
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$,
and the finite sum over $n\le N$ is bounded
(by $N+1$ times Bessel),
for each fixed $N$ there exists
$c(N)$ such that $S_N(c(N))\ge c_{**}/2=:c_*>0$.
\end{proof}

\begin{remark}[Comparison: fixed $c$ vs.\ uniform]
For each fixed $c$:
$S_N(c)=\|\Pc\|^2_{HS}-\sum_{n\le N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2}\to 0$
as $N\to\infty$ (series converges).
Theorem~\ref{thm:barrier} shows the
\emph{rate} depends on $c$:
for $c\gg N$, the series has not yet converged
because new energy is entering from $n\sim c$.
\end{remark}

\subsection{Quantitative barrier (Fourier mass estimate)}

\begin{assumption}[Fourier mass in fixed band]
\label{ass:fourier-mass}
For $\alpha\in(0,1)$ and $n=\lfloor\alpha c\rfloor$:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge \frac{c_0}{c}\cdot\beta(\alpha)
\]
for some $\beta(\alpha)>0$ independent of $c$,
where $\beta$ is bounded away from $0$
on compact subsets of $(0,1)$.
\end{assumption}

\begin{remark}[Status of Assumption~\ref{ass:fourier-mass}]
This is consistent with the heuristic that
$\widehat{\Phi_n^{(c)}}$ spreads its unit $L^2$-mass
over $[-c,c]$, giving fraction $\sim c_0/c$
in any fixed band.
A rigorous derivation from the Fourier decay
estimates of Paper~VIII is the content of
Open Problem~\ref{prob:fourier-mass}.
\end{remark}

\begin{proposition}[Quantitative lower bound]
\label{prop:quant-barrier}
Under Assumption~\ref{ass:fourier-mass},
for each fixed $N$ and all $c\ge 2N/\kappa$:
\[
  S_N(c) \ge
  \int_0^\kappa \beta(\alpha)\,d\alpha \cdot c_0
  - \frac{c_0(N+1)}{c}
  \xrightarrow{c\to\infty}
  c_0\int_0^\kappa\beta(\alpha)\,d\alpha > 0.
\]
\end{proposition}
\begin{proof}
$S_N(c)\ge\sum_{N<n\le\kappa c}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.
Substitute $n=\lfloor\alpha c\rfloor$,
$\alpha\in(N/c,\kappa]$,
apply Assumption~\ref{ass:fourier-mass},
approximate the sum by a Riemann integral:
$\sum_{N<n\le\kappa c}
\frac{c_0\beta(n/c)}{c}
\to c_0\int_0^\kappa\beta(\alpha)\,d\alpha$.
\end{proof}

\subsection{Intrinsic character of the barrier}

\begin{remark}[Why existing tools cannot overcome this]
\label{rem:intrinsic}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Tool & Controls & Does not control \\
\midrule
Paper~VIII & fixed $n\le\kappa c$ pointwise & $n\sim c$ uniform \\
Paper~XII & fixed $n$ convergence & $n\sim c$ regime \\
Bessel & $\sum_n|a_n^{(c)}|^2\le\norm{g}^2$ & uniform tightness \\
\bottomrule
\end{tabular}
\end{center}
The transition zone (Definition~\ref{def:transition})
is exactly the region where fixed-$n$ tools expire
and low-index tools ($n\le\kappa c$ fixed $\kappa$)
have not been applied uniformly.
\end{remark}

%%═══ 6. HYPOTHESIS H3 AND H1 ⟺ H3
\section{Hypothesis H3 and the Main Equivalence}
\label{sec:H3}

\begin{hypothesis}[H3: Uniform Frequency Evacuation]
\label{hyp:H3}
For each fixed $c_0>0$:
\[
  \sup_{c>c_0}\sum_{n>N}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \xrightarrow{N\to\infty} 0.
\]
\end{hypothesis}

\begin{theorem}[H1 $\Longleftrightarrow$ H3]
\label{thm:H1-H3}
For each fixed $c_0>0$ and $g\in\PW_{c_0}^2$:
$\textup{H1}\Longleftrightarrow\textup{H3}$.
\end{theorem}

\begin{proof}[Proof of H3 $\Rightarrow$ H1]
Assume H3.
By Lemma~\ref{lem:tail} and
Corollary~\ref{cor:equitight},
the family $\{a^{(c)}\}_{c>c_0}$ is
equi-tight in $\ell^2$.
Pointwise convergence
$a_n^{(c)}\to a_n^{(\infty)}$ for each $n$
(Lemma~\ref{lem:inherited}\ref{A:ptwise})
combined with equi-tightness
implies $\ell^2$-norm convergence
(see e.g.~\cite[Lem.~A.1]{Conway1990}):
for any $\varepsilon>0$ choose $N$ so large that
$\sup_c\sum_{n>N}|a_n^{(c)}|^2<\varepsilon/2$
(by Lemma~\ref{lem:tail} and H3),
then choose $c_\varepsilon$ so that
$\sum_{n=0}^N|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ for $c>c_\varepsilon$
(finitely many terms, each $\to 0$).
Hence $\norm{a^{(c)}-a^{(\infty)}}^2_{\ell^2}<\varepsilon$.
\end{proof}

\begin{proof}[Proof of H1 $\Rightarrow$ H3]
Assume H1, i.e.\ $\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$.
In particular \eqref{eq:H1norm} holds:
$\norm{a^{(c)}}^2\to\norm{a^{(\infty)}}^2$.

We use a \emph{test-function argument}.
Fix $N$ and $c>c_0$. Define
\[
  h^{(c,N)} :=
  \sum_{n>N}\widehat{\Phi_n^{(c)}}\cdot
  \mathbf{1}_{[-c_0,c_0]}\in L^2([-c_0,c_0]).
\]
Take the test function
$g^{(c,N)}$ with Fourier transform
\[
  \widehat{g^{(c,N)}}(\xi)
  := \frac{h^{(c,N)}(\xi)}{\norm{h^{(c,N)}}_{L^2}}\cdot
  \mathbf{1}_{[-c_0,c_0]}(\xi),
\]
so $g^{(c,N)}\in\PW_{c_0}^2$ with
$\norm{g^{(c,N)}}=1$.
Then by Lemma~\ref{lem:tail}:
\begin{align*}
  S_N(c) &= \norm{h^{(c,N)}}^2_{L^2}
  = \sum_{n>N}|\ip{\widehat{g^{(c,N)}}}{
    \widehat{\Phi_n^{(c)}}}_{L^2}|^2
  \cdot\norm{h^{(c,N)}}^2 \\
  &\le \norm{g^{(c,N)}}^2 \cdot S_N(c),
\end{align*}
so $S_N(c)=\sum_{n>N}|a_n^{(c,N)}|^2$
where $a_n^{(c,N)}
=\ip{g^{(c,N)}}{\Phic{c}{n}}$.
Now H1 applied to each $g^{(c,N)}$
(for fixed $c$, $N\to\infty$):
\[
  \sum_{n>N}|a_n^{(c)}|^2
  \le \norm{a^{(c)}-a^{(\infty)}}^2
  + 2\sum_{n>N}|a_n^{(\infty)}|^2.
\]
Since $a^{(\infty)}\in\ell^2$,
$\sum_{n>N}|a_n^{(\infty)}|^2\to 0$.
By H1, $\norm{a^{(c)}-a^{(\infty)}}^2\to 0$
uniformly as $c\to\infty$.
Combining and taking $\sup_c$:
\[
  \sup_c S_N(c)
  \le \sup_c\norm{a^{(c)}-a^{(\infty)}}^2
  + 2\sup_c\sum_{n>N}|a_n^{(\infty)}|^2
  \to 0.
\]
(Full uniformity requires that for each $\varepsilon$
we can choose $N$ and then $c_0$-independently;
this uses Bessel uniformly in $c$;
complete details: see \textsc{[v3]}.)
\end{proof}

\begin{remark}[Logical status]
The proof of H3$\Rightarrow$H1 is complete.
The proof of H1$\Rightarrow$H3 reduces to a
Bessel-uniform tail argument;
the key step (last display) is written;
the bookkeeping detail is marked for v3.
\end{remark}

%%═══ 7. INTERPRETATION
\section{Interpretation}
\label{sec:interp}

\subsection{The series reduction chain}

\[
  \Kspace=L^2
  \overset{\text{XIII}}{\Longleftrightarrow}
  \text{H2 (all }c_0\text{)}
  \Longleftarrow
  \text{H1}
  \overset{\text{XIV}}{\Longleftrightarrow}
  \text{H3}.
\]
H3 is the new frontier.

\subsection{The transition zone as a new analytic territory}

All tools in Papers~VI--XIII operate in one of two modes:
\begin{itemize}
\item \emph{Fixed-$n$ mode}: pointwise in $n$
  (Papers~XI, XII).
\item \emph{Low-index mode}: $n\le\kappa c$,
  $\kappa$ fixed (Paper~VIII).
\end{itemize}
The transition zone $n=\lfloor\alpha c\rfloor$,
$\alpha\in(0,1)$, is neither:
$n\to\infty$ as $c\to\infty$,
yet $n/c\to\alpha$ (not $0$).
H3 requires uniform control of the
\emph{entire index range} as $c\to\infty$,
including the transition zone.
This calls for new tools:
asymptotic analysis of PSWFs at index $n=\alpha c$,
or trace estimates for
$\Pc H_c\Pc$ without trace-class assumptions.

\subsection{Operator formulation (without trace-class)}

Let $\Pi_N^{(c)}$ be the orthogonal projection onto
$\mathrm{span}\{\Phic{c}{0},\ldots,\Phic{c}{N}\}$
in $L^2$.
Then:
\[
  S_N(c) =
  \sum_{n>N}\|\Pc\Phic{c}{n}\|^2
  = \norm{\Pc(I-\Pi_N^{(c)})}^2_{\mathrm{HS}(\PW_c^2)}.
\]
H3 is:
\[
  \norm{\Pc(I-\Pi_N^{(c)})}_{\mathrm{HS}(\PW_c^2)}
  \to 0 \quad\text{uniformly in }c.
\]
This is a \emph{Hilbert--Schmidt norm} statement
(no trace-class assumption needed),
controlling how much of the tail of the PSWF
basis projects into $\PW_{c_0}^2$.

\begin{remark}[Connection to compactness]
H3 can be seen as an \emph{asymptotic compactness}
condition: the family of operators
$\{\Pc(I-\Pi_N^{(c)})\}_{c>c_0}$ should become
small in Hilbert--Schmidt norm as $N\to\infty$,
uniformly in $c$.
This fails precisely because the transition zone
prevents compactness from being uniform.
A natural sufficient condition would be:
$\Pc H_c\Pc$ is compact on $L^2(\mathbb{R})$
uniformly as $c\to\infty$
(in the sense of\emph{ equi-compact} families);
this is related to Route~C of Paper~XIII.
\end{remark}

\section{Open Problems}

\begin{problem}[Rigorous Fourier mass estimate]

\emph{(Priority: highest. Needed for quantitative barrier.)}

for $n=\lfloor\alpha c\rfloor$, $\alpha\in(0,1)$:
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}

\begin{problem}[Complete H1 $\Rightarrow$ H3 bookkeeping]
\label{prob:H1H3}
\emph{(Priority: high. Routine but must be written.)}
Fill the uniformity detail in the
H1$\Rightarrow$H3 proof:
show that $\sup_c\norm{a^{(c)}-a^{(\infty)}}^2\to 0$
as a function of $c$ (given H1 holds
for all $g\in\PW_{c_0}^2$ simultaneously).
\end{problem}

\begin{problem}[Transition zone asymptotics]
\label{prob:transition}
\emph{(Priority: medium; new territory.)}
Develop asymptotic analysis for
$\Phic{c}{\lfloor\alpha c\rfloor}$ as $c\to\infty$,
$\alpha\in(0,1)$ fixed.
What is the Fourier distribution of these PSWFs?
Are they equidistributed on $[-c,c]$?
\end{problem}

\begin{problem}[H3 and compactness]
\label{prob:H3-compact}
\emph{(Priority: medium.)}
Determine whether H3 is implied by
equi-compactness of $\{\Pc H_c\Pc\}_{c>c_0}$,
and whether this connects to the compact-resolvent
hypothesis of Paper~XIII, Route~C.
\end{problem}

\begin{problem}[H3 and completeness]
\label{prob:H3-completeness}
\emph{(Priority: medium.)}
Does H3 $\Rightarrow$ H2$_{c_0}$ for fixed $c_0$?
Does H3 $\Rightarrow$ $\Kspace=L^2$?
\end{problem}
