  Cross-Gram Operators, Spectral Evacuation, and the Transition Zone}

\author{[Author]}
\address{[Address]}

\begin{abstract}
The central obstruction to coefficient stability
of the Slepian eigenbasis is
\emph{spectral non-evacuation in the transition
zone}: the persistence of spectral mass of the
measures $\mu_{n,c}(d\xi)
:=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi$
in the regime $n\sim\alpha c$ as $c\to\infty$.

This paper reduces Hypothesis~H1 of Paper~XIII
(the $\ell^2$-norm convergence of Slepian
coefficients $a_n^{(c)}=\ip{g}{\Phic{c}{n}}$
as $c\to\infty$) to a single spectral
distribution problem in the transition
zone $n\sim\alpha c$.

The key structural results are:
(i)~the cross-Gram operator $T_c$ satisfies
$\norm{T_c}\le 1$ uniformly;
(ii)~the tail functional
\[
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0])
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}
\]
controls the tail via
$\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c)$;
(iii)~the operator $\Bn:=(I-\Pi_N^{(c)})\Pc$
satisfies $\norm{\Bn}_{\mathrm{op}}\le 1$;
the inequality $\norm{\Ac}_{\mathrm{op}}\le S_N(c)$
is established via finite-dimensional
truncation $\Ac^{(M)}$ and monotone limit
$M\to\infty$, without any global
trace-class or Hilbert--Schmidt assumption;
(iv)~the quadratic form
$Q_c^N(g)=\ip{\Ac g}{g}$
where $\Ac:=\Bn^*\Bn\ge 0$, $0\le\Ac\le I$.

H3 ($\sup_c S_N(c)\to 0$) implies H1.
The gap between quadratic-form convergence
and operator norm convergence is identified
as a \emph{double-scaling Szeg\H{o} problem}
(Definition~\ref{def:double-scaling})
for the truncated Gram operator
$\Ac^{(M)}=P_{V_M^{(c)}}P_{c_0}P_{V_M^{(c)}}$
(\S\ref{sec:szego}).
The resolution requires not merely
pointwise quasi-orthogonality~(D-weak)
but \emph{collective quasi-orthogonality}~(D-strong):
$\|\Ac^{(M)}-D^{(M)}\|_{\mathrm{HS}}\to 0$
for $M(c)\sim\beta c$.
The primary analytic obstruction is the
\emph{near-diagonal regime} $|n-m|=O(1)$,
where oscillatory methods fail and the
problem reduces to amplitude orthogonality
of adjacent PSWFs
(Open Problem~\ref{prob:near-diagonal},
Remark~\ref{rem:WKB-reduction}).
\end{abstract}

%%═══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context}

Paper~XIII identified H1 as the key
remaining hypothesis for completeness
$\Kspace=L^2(\mathbb{R})$.
Paper~XIV reduces H1 to a spectral
distribution problem for the
PSWF eigenbasis in the transition zone.
All conditional steps are explicitly labelled.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$
($c\to\infty$).
\end{hypothesis}

H1 splits into two components:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
\eqref{eq:H1ip} is proved unconditionally
(Prop.~\ref{prop:H1ip}).
The obstruction is entirely in~\eqref{eq:H1norm}.
The core difficulty:
$a^{(c)}\rightharpoonup a^{(\infty)}$
does not imply norm convergence;
Paper~XIV localises this gap precisely
as a spectral distribution problem.

\subsection{The central new framing}

\begin{definition}[Spectral measures and tail functional]
\label{def:spectral-measures}
For $n\ge 0$, $c>c_0$, define the
spectral probability measures
\[
  \mu_{n,c}(d\xi)
  :=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi,
  \qquad
  \mu_{n,c}(\mathbb{R})=1\;(\text{Plancherel}).
\]
The tail functional is defined directly
via the PSWF-ONB expansion:
\begin{equation}\label{eq:SN-def}
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0])
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}.
\end{equation}
The obstruction to H1 is
\emph{spectral non-evacuation}:
the failure of $S_N(c)$ to vanish
uniformly in $c$ as $N\to\infty$.
\end{definition}

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $0<\alpha<\kappa<1$:
$\mathcal{T}_{\alpha,\kappa}(c)
:=\{n: \alpha c < n \le \kappa c\}$.
This band has $\sim(\kappa-\alpha)c$ elements
and captures the regime where Slepian
eigenvalues $\lambda_n^{(c)}$
transit between $\approx 1$ and $\approx 0$.
\end{definition}

\subsection{Logical status}

\begin{center}
\fbox{\parbox{0.88\linewidth}{
\textbf{Proved unconditionally:}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{theorem}[$\norm{T_c}\le 1$]
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{theorem}
\begin{proof}
Bessel:
$\norm{T_c g}^2=\sum_n|\ip{g}{\Phic{c}{n}}|^2
\le\norm{g}^2$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds unconditionally.
\end{proposition}
\begin{proof}
Paper~XII: $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$.
\end{proof}

\begin{remark}
By weak lower semicontinuity:
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$,
so H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}\le\norm{a^{(\infty)}}$.
The entire obstruction is in~\eqref{eq:H1norm}.
\end{remark}

%%═══ 4. TAIL REDUCTION AND THE QUADRATIC FORM
\section{Tail Reduction, Operator Structure, and the Quadratic Form}
\label{sec:tail}

\begin{remark}[Foundational conventions]
\label{rem:foundation}
Throughout this section:
$S_N(c)$ is defined by the convergent
series~\eqref{eq:SN-def}, not via a global
operator trace.
$P_{c_0}$ is not Hilbert--Schmidt on
$L^2(\mathbb{R})$; no such property is used.
All operator norm inequalities are derived
via finite-dimensional truncations $\Ac^{(M)}$
followed by a monotone limit $M\to\infty$.
Eigenvalue-counting arguments are carried
out only at finite truncation level and
are never passed to the limit.
\end{remark}

\begin{definition}[$S_N(c)$, $\sigma(c)$, finite truncations]
\label{def:SN}
\begin{align*}
  S_N(c) &:= \sum_{n>N}\mu_{n,c}([-c_0,c_0]),\\
  S_N^{(M)}(c) &:= \sum_{n=N+1}^{M}
            \mu_{n,c}([-c_0,c_0]),\\
  \sigma(c) &:= \sum_{n\ge 0}\mu_{n,c}([-c_0,c_0])
            \sim \frac{cc_0}{\pi}
            \quad\text{(Slepian--Weyl
            \cite{Slepian1961})}.
\end{align*}
$S_N^{(M)}(c)\nearrow S_N(c)$ monotonely.
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$Q_c^N(g):=\sum_{n>N}|a_n^{(c)}|^2
\le\norm{g}^2\cdot S_N(c).$
\end{lemma}
\begin{proof}
Cauchy--Schwarz gives
$|a_n^{(c)}|^2
\le\norm{g}^2\cdot\mu_{n,c}([-c_0,c_0])$.
Sum over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness]
$\sup_c S_N(c)\to 0
\Rightarrow\{a^{(c)}\}_c$
equi-tight in $\ell^2$.
\end{corollary}

\begin{theorem}[Operator structure and spectral sum identity]
\label{thm:SN-operator}
Define:
\begin{align*}
  \Bn &:= (I-\Pi_N^{(c)})\Pc,\quad
  \Ac := \Bn^*\Bn,\\
  V_M^{(c)} &:= \mathrm{span}
    \{\Phic{c}{n} : N < n \le M\},\\
  \Ac^{(M)} &:= P_{V_M^{(c)}}P_{c_0}P_{V_M^{(c)}}
    \;:\; V_M^{(c)}\to V_M^{(c)}.
\end{align*}

\textbf{(0) Contraction.}
$\norm{\Bn}_{\mathrm{op}}\le 1$,
$0\le\Ac\le I$, $0\le\Ac^{(M)}\le I$.

\medskip
\textbf{(I) Spectral sum identity.}
\begin{equation}\label{eq:TrPSWF}
  \mathrm{Tr}_{\mathrm{PSWF}}(\Ac)
  := \lim_{M\to\infty}\mathrm{Tr}(\Ac^{(M)})
  = S_N(c).
\end{equation}

\medskip
\textbf{(II) Operator norm inequality.}
\begin{equation}\label{eq:op-le-spectral}
  \norm{\Ac}_{\mathrm{op}} \le S_N(c).
\end{equation}

\medskip
\textbf{(III) Quadratic form.}
\begin{equation}\label{eq:Qc-key}
  Q_c^N(g) = \norm{\Bn g}^2 = \ip{\Ac g}{g}.
\end{equation}

\medskip
\textbf{(IV) The full hierarchy.}
\begin{equation}\label{eq:chain}
  \underbrace{\text{H3}}_{S_N(c)\to 0}
  \;\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0
  \;\Rightarrow\;
  \text{H1}
  \;\Rightarrow\;
  \ip{\Ac g}{g}\to 0.
\end{equation}
\end{theorem}

\begin{remark}[Non-interchangeability of limit and supremum]
\label{rem:lim-sup}
The failure to upgrade
$\ip{\Ac g}{g}\to 0$ to $\norm{\Ac}_{\mathrm{op}}\to 0$
is the non-interchangeability of
$\lim_{c\to\infty}$ and $\sup_{\norm{g}=1}$.
\end{remark}

\begin{remark}[Convergence hierarchy]
\label{rem:convergence-hierarchy}
(1)~H3 controls all spectral mass simultaneously.
(2)~$\norm{\Ac}_{\mathrm{op}}\to 0$ controls
the largest eigenvalue uniformly.
(3)~$\ip{\Ac g}{g}\to 0$ each fixed $g$:
weakest; proved from H1.
The gap between (3) and (2) is
OP~\ref{prob:H1H3}.
\end{remark}

\begin{lemma}[Finite-dimensional effective rank bound]
\label{lem:rank-lower}
For $\varepsilon\in(0,1)$ and finite $M>N$:
\[
  \#\{j:\lambda_j^{(M)}\ge\varepsilon\}
  \ge
  \frac{S_N^{(M)}(c)-\varepsilon(M-N)}{1-\varepsilon}.
\]
\end{lemma}
\begin{proof}
Split trace at $\varepsilon$; rearrange.
\end{proof}

%%═══ 5. SPECTRAL MEASURES AND NC
\section{Spectral Measures and No-Concentration}
\label{sec:NC}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
NC holds if $\exists\, c_*>0$ such that
for all large $c$:
$\sum_{n\in\mathcal{T}_{\alpha,\kappa}(c)}
\mu_{n,c}([-c_0,c_0]) \ge c_*>0$.
\end{definition}

\begin{remark}[NC and semiclassical equidistribution]
\label{rem:NC-QE}
Under the local Weyl law,
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$
for $n\in\mathcal{T}_{\alpha,\kappa}(c)$,
consistent with uniform phase-space spreading
\cite{Zworski2012}.
NC holds with $c_*=c_0(\kappa-\alpha)/2$.
\end{remark}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under NC: $\liminf_{c\to\infty}S_N(c)\ge c_*>0$.
\end{proposition}

\begin{proposition}[Logical separation]
\label{prop:NC-not-H3}
NC is sufficient but not necessary for H3 to fail.
\end{proposition}
\begin{proof}
Set $I_c:=\{n:|n-\lfloor\alpha c\rfloor|
\le\lfloor\sqrt{c}\rfloor\}$,
$\mu_{n,c}([-c_0,c_0]):=1/|I_c|$
for $n\in I_c$. Then $S_N(c)\ge 1$
but NC fails.
\end{proof}

%%═══ 6. H3 AND THE MAIN REDUCTION
\section{H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{theorem}[H3 $\Rightarrow$ H1]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$. H3 gives $N$ with
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$.
Lemma~\ref{lem:tail}: $Q_c^N(g)<\varepsilon/2$.
Paper~XII: $\sum_{n\le N}|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ for large $c$. Adding gives H1.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$ quadratic-form convergence]
\label{thm:H1-H3-ptwise}
Assume H1. Then $Q_c^N(g)\to 0$ each fixed $g,N$.
\end{theorem}
\begin{proof}
$Q_c^N(g)
\le 2\norm{a^{(c)}-a^{(\infty)}}^2_{\ell^2}
+2\sum_{n>N}|a_n^{(\infty)}|^2\to 0$.
\end{proof}

\begin{theorem}[Main reduction]
\label{thm:reduction}
\textup{(i)} Proved:
H3$\Rightarrow$H1,
H1$\Rightarrow\ip{\Ac g}{g}\to 0$,
NC$\Rightarrow$H3 fails,
local Weyl$\Rightarrow$NC.

\textup{(ii)} Open:
OP~\ref{prob:H1H3}, OP~\ref{prob:near-diagonal},
OP~\ref{prob:fourier-mass}.

\textup{(iii)} Conditional:
local Weyl$\Rightarrow$NC$\Rightarrow$H3
fails$\Rightarrow$H1 fails.
\end{theorem}

\subsection{Connection to semiclassical Szeg\H{o} theory}
\label{sec:szego}

\subsubsection*{Concrete operator form}

\begin{equation}\label{eq:AcM-concrete}
  \Ac^{(M)} = P_{V_M^{(c)}}P_{c_0}P_{V_M^{(c)}}
  \;:\; V_M^{(c)}\to V_M^{(c)},
\end{equation}
with entries
\begin{equation}\label{eq:gram}
  (\Ac^{(M)})_{nm}
  = \int_{-c_0}^{c_0}
    \widehat{\Phi_n^{(c)}}(\xi)\,
    \overline{\widehat{\Phi_m^{(c)}}(\xi)}\,d\xi,
  \quad N < n,m \le M.
\end{equation}

\subsubsection*{Double-scaling Szeg\H{o} regime}

\begin{definition}[Double-scaling Szeg\H{o} regime]
\label{def:double-scaling}
The family $\Ac^{(M(c))}$ constitutes a
\emph{double-scaling Szeg\H{o} problem} if
$c\to\infty$, $M(c)\sim\beta c$, and both:
(i)~the symbol has a semiclassical limit
($h=1/c\to 0$, symbol $\mathbf{1}_{|\xi|\le c_0}$
on the energy surface $\{n/c\to\alpha\}$), and
(ii)~$\dim V_{M(c)}^{(c)}\sim\beta c$.
\end{definition}

\begin{remark}[Why double-scaling differs from classical Szeg\H{o}]
\label{rem:double-scaling}
Three features interact simultaneously:
(1)~$h=1/c\to 0$ changes the eigenbasis;
(2)~$\dim V_{M(c)}^{(c)}\sim\beta c\to\infty$
at the same rate;
(3)~$\lim_{c}\lim_{M}\ne
\lim_{c,\,M(c)\sim\beta c}$.
With $O(c^2)$ off-diagonal pairs each
satisfying $|(\Ac^{(M)})_{nm}|\le\varepsilon(c)/c$,
\[
  \|\Ac^{(M)}-D^{(M)}\|_{\mathrm{HS}}^2
  \le c^2\cdot\frac{\varepsilon(c)^2}{c^2}
  = \varepsilon(c)^2 \to 0,
\]
so uniform (D-weak) forces (D-strong).
A merely pointwise $o(1/c)$ bound
without uniformity is insufficient.
\end{remark}

\begin{remark}[WKB reduction of the near-diagonal regime]
\label{rem:WKB-reduction}
In the transition zone $n\sim\alpha c$,
the Fourier transforms
$\widehat{\Phi_n^{(c)}}(\xi)$ admit a
semiclassical WKB-type representation
\begin{equation}\label{eq:WKB-heuristic}
  \widehat{\Phi_n^{(c)}}(\xi)
  \approx A_n(\xi)\,e^{i n \theta(\xi)/c},
  \qquad \xi\in[-c_0,c_0],
\end{equation}
where $\theta(\xi)$ is a smooth phase
independent of $n$, and $A_n(\xi)$ is a
slowly varying amplitude.
Substituting into~\eqref{eq:gram}:
\begin{equation}\label{eq:WKB-gram}
  (\Ac^{(M)})_{nm}
  \approx \int_{-c_0}^{c_0}
  A_n(\xi)\,\overline{A_m(\xi)}\,
  e^{i(n-m)\theta(\xi)/c}\,d\xi.
\end{equation}
For $|n-m|\gg 1$: oscillatory phase
yields decay via stationary phase.
For $|n-m|=O(1)$: phase is $O(1/c)$,
no oscillatory cancellation.
The near-diagonal regime thus reduces to:
\begin{equation}\label{eq:amplitude-reduction}
  (\Ac^{(M)})_{n,n+k}
  \approx \int_{-c_0}^{c_0}
  A_n(\xi)\,\overline{A_{n+k}(\xi)}\,d\xi,
  \qquad k=O(1).
\end{equation}
The key obstruction is the lack of
orthogonality of the amplitudes $\{A_n\}$
on the restricted window $[-c_0,c_0]$.
Establishing uniform $o(1/c)$ bounds here
requires quantitative control of the
$n$-dependence of $A_n(\xi)$ via the
PSWF differential equation.
\end{remark}

\begin{remark}[Regime decomposition and the near-diagonal obstruction]
\label{rem:regime-decomposition}
\begin{equation}\label{eq:HS-regimes}
  \|\Ac^{(M)} - D^{(M)}\|_{\mathrm{HS}}^2
  = \Sigma_{\mathrm{near}}
  + \Sigma_{\mathrm{int}}
  + \Sigma_{\mathrm{far}},
\end{equation}
where the sums run over
$|n-m|\le C$,
$C<|n-m|\le c^\theta$,
$|n-m|>c^\theta$ respectively.
\begin{itemize}[leftmargin=*]
\item \textbf{Far ($|n-m|>c^\theta$):}
  strong oscillation; standard stationary
  phase gives decay.
\item \textbf{Intermediate:}
  weaker oscillation; uniform transition
  estimates required.
\item \textbf{Near-diagonal ($|n-m|\le C$):}
  phase $O(1/c)$; stationary phase fails.
  Requires WKB amplitude analysis
  (Remark~\ref{rem:WKB-reduction}).
  \emph{Primary technical obstruction.}
\end{itemize}
\end{remark}

\subsubsection*{Correct scaling}

Under the local Weyl law,
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$ and
$\bar\lambda^{(M)}(c)\sim c_0/c\to 0$.
The correct open condition is:
\begin{equation}\label{eq:open-condition}
  S_N^{(M(c))}(c) \gtrsim 1
  \quad\text{for }M(c)\sim\beta c.
\end{equation}

\subsubsection*{No single-mode dominance}

\begin{lemma}[No single-mode dominance]
\label{lem:no-single-mode}
Under the local Weyl law and $M(c)\sim\beta c$:
$\max_{N<n\le M(c)} (\Ac^{(M)})_{nn}
= O(1/c)$.
Any total mass $S_N^{(M(c))}(c)\gtrsim 1$
must arise from $\sim c$ modes collectively.
\end{lemma}
\begin{proof}
Each diagonal entry $\sim c_0/c = O(1/c)$.
\end{proof}

\subsubsection*{Quasi-orthogonality}

\textbf{(D-weak) Uniform pointwise.}
\begin{equation}\label{eq:quasi-orth}
  \sup_{\substack{n\ne m \\
  n,m\in\mathcal{T}_{\alpha,\kappa}(c)}}
  \left|(\Ac^{(M)})_{nm}\right|
  \le \frac{\varepsilon(c)}{c},
  \quad \varepsilon(c)\to 0,
  \quad M\sim\beta c.
\end{equation}
Uniform (D-weak) implies (D-strong)
via Remark~\ref{rem:double-scaling}.
A merely pointwise $o(1/c)$ bound
without uniformity is insufficient.

\medskip
\textbf{(D-strong) Collective.}
\begin{equation}\label{eq:HS-diag}
  \|\Ac^{(M)}-D^{(M)}\|_{\mathrm{HS}} \to 0
  \quad(M\sim\beta c).
\end{equation}

\subsubsection*{Diagonal reduction}

\begin{proposition}[Diagonal reduction under HS control]
\label{prop:HS-reduction}
Assume~(C) and~(D-strong). Then:
\[
  S_N^{(M(c))}(c)
  \sim \sum_{n=N+1}^{M(c)}
  \mu_{n,c}([-c_0,c_0])
  \sim c_0\beta > 0.
\]
\end{proposition}

\begin{conjecture}[Double-scaling Szeg\H{o} behaviour]
\label{conj:szego}
Assume~(C) and~(D-strong),
$M(c)\sim\beta c$. Then:
(i)~$\bar\lambda^{(M(c))}(c)\sim c_0/c\to 0$;
(ii)~$S_N^{(M(c))}(c)\sim c_0\beta>0$;
(iii)~$\#\{j:\lambda_j^{(M(c))}\ge\varepsilon\}
\to\infty$ for fixed small $\varepsilon>0$.
\end{conjecture}

%%═══ 8. PROGRAMME FOR PAPER XV
\section{Programme for Paper~XV}
\label{sec:PaperXV}

\subsection{Required estimates}

\begin{enumerate}[\rm(A)]
\item $\mu_{n,c}([-c_0,c_0])\le C/c$,
  uniform in $\mathcal{T}_{\alpha,\kappa}(c)$.
\item $\mu_{n,c}([-c_0,c_0])\ge 1/(Cc)$,
  uniform in $\mathcal{T}_{\alpha,\kappa}(c)$.
\item $\mu_{n,c}([-c_0,c_0])=c_0/c+o(1/c)$
  uniformly.
\item[\rm(D-weak)] Uniform pointwise
  quasi-orthogonality~\eqref{eq:quasi-orth}
  with $\varepsilon(c)\to 0$ uniform over
  all pairs, including near-diagonal
  $|n-m|=O(1)$ (OP~\ref{prob:near-diagonal}).
\item[\rm(D-strong)] $\|\Ac^{(M)}-D^{(M)}\|_{\mathrm{HS}}
  \to 0$; implied by uniform (D-weak).
\end{enumerate}

\subsection{Semiclassical approach and regime strategy}

$h:=1/c\to 0$.
WKB gives (A)--(C) \cite{Osipov2013}.
The three regimes of~\eqref{eq:HS-regimes}
require separate treatment:
\begin{itemize}[leftmargin=*]
\item \textbf{Far ($|n-m|>c^\theta$):}
  Standard stationary phase;
  $\Sigma_{\mathrm{far}}=o(1)$.
\item \textbf{Intermediate:}
  Uniform transition estimates;
  $\Sigma_{\mathrm{int}}=o(1)$.
\item \textbf{Near-diagonal ($|n-m|\le C$):}
  WKB amplitude orthogonality
  (OP~\ref{prob:near-diagonal});
  primary technical target.
\end{itemize}
Establishing $\Sigma_{\mathrm{near}}
+\Sigma_{\mathrm{int}}+\Sigma_{\mathrm{far}}
=o(1)$ yields (D-strong) and closes
OP~\ref{prob:H1H3}.

%%═══ 9. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law; Paper~XV]
\label{prob:fourier-mass}
\emph{(Priority: critical.)}
Prove (A)--(C):
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$
uniformly in $\mathcal{T}_{\alpha,\kappa}(c)$.
\end{problem}

\begin{problem}[Total spectral mass along linear scales]
\label{prob:H1H3}
\emph{(Priority: high.)}
Show $S_N^{(M(c))}(c)\gtrsim 1$
for $M(c)\sim\beta c$.
Follows from (C)+(D-strong)
via Proposition~\ref{prop:HS-reduction}.
The key step is OP~\ref{prob:near-diagonal}.
\end{problem}

\begin{problem}[Near-diagonal cancellation]
\label{prob:near-diagonal}
\emph{(Core analytic target; Paper~XV.)}

Establish the uniform bound, for fixed $C>0$:
\begin{equation}\label{eq:near-diagonal}
  \int_{-c_0}^{c_0}
  \widehat{\Phi_n^{(c)}}(\xi)\,
  \overline{\widehat{\Phi_{n+k}^{(c)}}(\xi)}\,d\xi
  = o\!\left(\frac{1}{c}\right),
\end{equation}
uniformly in $n\in\mathcal{T}_{\alpha,\kappa}(c)$
and $|k|\le C$.

\medskip\noindent
\textbf{Interpretation.}
This is precisely $\Sigma_{\mathrm{near}}$
in~\eqref{eq:HS-regimes}, and the primary
obstruction to uniform (D-weak).

\medskip\noindent
\textbf{Heuristic reduction.}
Under~\eqref{eq:WKB-heuristic},
\eqref{eq:near-diagonal} reduces to:
\begin{equation}\label{eq:amplitude-target}
  \int_{-c_0}^{c_0}
  A_n(\xi)\,\overline{A_{n+k}(\xi)}\,d\xi
  = o\!\left(\frac{1}{c}\right),
  \qquad k=O(1),
\end{equation}
uniformly in $n\sim\alpha c$.
The obstruction is not insufficient
oscillation but the lack of orthogonality
of the amplitudes $\{A_n\}$ on $[-c_0,c_0]$.

\medskip\noindent
\textbf{Expected approach.}
\begin{itemize}[leftmargin=*]
\item WKB asymptotics for
  $\widehat{\Phi_n^{(c)}}$,
\item quantitative $n$-derivative control
  of $A_n(\xi)$ via the PSWF ODE,
\item $L^2([-c_0,c_0])$ estimates for
  differences of adjacent modes,
\item structural relations between
  $\Phi_n^{(c)}$ and $\Phi_{n+1}^{(c)}$
  induced by the differential equation.
\end{itemize}

\medskip\noindent
\textbf{Role in the programme.}
A positive resolution of~\eqref{eq:near-diagonal},
combined with far and intermediate regime
estimates, yields uniform (D-weak),
hence (D-strong), and closes
OP~\ref{prob:H1H3} via
Proposition~\ref{prop:HS-reduction}.
\end{problem}

\begin{problem}[NC from Papers~VIII, XII]
\label{prob:NC-direct}
\emph{(Priority: medium.)}
Do the pointwise bounds of Paper~VIII
imply lower bounds on
$\mu_{n,c}([-c_0,c_0])$
without full WKB?
\end{problem}

\medskip\noindent
\textbf{Central conclusion.}
\emph{The obstruction to coefficient stability
is spectral non-evacuation.
All operator-theoretic structure is
established without global trace-class
or Hilbert--Schmidt assumptions.
The gap between quadratic-form convergence
and operator norm convergence is a
\emph{double-scaling Szeg\H{o} problem}.
Its resolution requires uniform (D-weak)
across all pairs $(n,m)$, which forces
(D-strong) and closes OP~\ref{prob:H1H3}.
The primary analytic obstruction is the
near-diagonal regime $|n-m|=O(1)$
(OP~\ref{prob:near-diagonal}), where
oscillatory methods fail and the problem
reduces to amplitude orthogonality of
adjacent PSWFs on $[-c_0,c_0]$
(Remark~\ref{rem:WKB-reduction}).
No single mode can carry macroscopic mass;
mass accumulates collectively across $\sim c$
modes, making (D-strong) both the natural
target and the operative sufficient condition.
Paper~XIV is logically complete up to
three open problems.}
