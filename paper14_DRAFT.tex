\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis:\\
  Cross-Gram Operators, Spectral Evacuation, and the Transition Zone}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
The central obstruction to coefficient stability
of the Slepian eigenbasis is
\emph{spectral non-evacuation in the transition
zone}: the persistence of spectral mass of the
measures $\mu_{n,c}(d\xi)
:=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi$
in the regime $n\sim\alpha c$ as $c\to\infty$.

This paper reduces Hypothesis~H1 of Paper~XIII
(the $\ell^2$-norm convergence of Slepian
coefficients $a_n^{(c)}=\ip{g}{\Phic{c}{n}}$
as $c\to\infty$) to a single spectral
distribution problem in the transition
zone $n\sim\alpha c$.

The key structural results are:
(i)~the cross-Gram operator $T_c$ satisfies
$\norm{T_c}\le 1$ uniformly;
(ii)~the tail functional
\[
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0])
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}
\]
controls the tail via
$\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c)$;
(iii)~the operator $\Bn:=(I-\Pi_N^{(c)})\Pc$
satisfies $\norm{\Bn}_{\mathrm{op}}\le 1$
(composition of orthogonal projections);
the inequality $\norm{\Ac}_{\mathrm{op}}\le S_N(c)$
is established via finite-dimensional
truncation $\Ac^{(M)}$ and monotone limit
$M\to\infty$, without any global
trace-class or Hilbert--Schmidt assumption;
(iv)~the quadratic form
$Q_c^N(g)=\ip{\Ac g}{g}$
where $\Ac:=\Bn^*\Bn\ge 0$, $0\le\Ac\le I$.

H3 ($\sup_c S_N(c)\to 0$) implies H1
(proved unconditionally).
H1 implies $\ip{\Ac g}{g}\to 0$ for each
fixed $g$ (pointwise quadratic form
convergence, diagonal WOT-type);
the gap to operator norm convergence
involves a \emph{finite-dimensional effective
rank conjecture}: for each $M$ and
$\varepsilon\in(0,1)$, the lower bound
\[
  \#\{j:\lambda_j^{(M)}\ge\varepsilon\}
  \;\ge\;
  \frac{S_N^{(M)}(c)-\varepsilon(M-N)}{1-\varepsilon}
\]
holds at every finite truncation level
(Lemma~\ref{lem:rank-lower}), but whether
the count $\#\{j:\lambda_j^{(M(c))}\ge\varepsilon\}
\to\infty$ along some scale $M(c)\to\infty$
remains open (Open Problem~\ref{prob:H1H3}).
This open problem is identified as a
\emph{semiclassical Szeg\H{o}-type problem}
for the eigenvalue distribution of the
truncated Gram operator $\Ac^{(M)}$
(\S\ref{sec:szego}).

The \emph{No-Concentration condition} (NC)
quantifies spectral non-evacuation:
NC$\Rightarrow$H3 fails (proved).
The converse is false even at the abstract
level: NC is sufficient but not necessary
for H3 to fail, as shown by
Proposition~\ref{prop:NC-not-H3}
(a logical separation result for abstract
measure families; PSWF compatibility
is not claimed).
NC holds under the local Weyl law (Paper~XV).
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Context}

Paper~XIII identified H1 as the key
remaining hypothesis for completeness
$\Kspace=L^2(\mathbb{R})$.
Paper~XIV reduces H1 to a spectral
distribution problem for the
PSWF eigenbasis in the transition zone.
All conditional steps are explicitly labelled.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$
($c\to\infty$).
\end{hypothesis}

H1 splits:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
\eqref{eq:H1ip} is proved
(Prop.~\ref{prop:H1ip});
\eqref{eq:H1norm} is equivalent to H3
(given OP~\ref{prob:H1H3}).
The core difficulty:
$a^{(c)}\rightharpoonup a^{(\infty)}$
does not imply norm convergence;
Paper~XIV localises this gap precisely.

\subsection{The central new framing}

\begin{definition}[Spectral measures and tail functional]
\label{def:spectral-measures}
For $n\ge 0$, $c>c_0$, define the
spectral probability measures
\[
  \mu_{n,c}(d\xi)
  :=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi,
  \qquad
  \mu_{n,c}(\mathbb{R})=1\;(\text{Plancherel}).
\]
The tail functional is defined directly
via the PSWF-ONB expansion:
\begin{equation}\label{eq:SN-def}
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0])
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}.
\end{equation}
This is a convergent series of non-negative
terms (each bounded by 1 by Plancherel);
no global operator-theoretic interpretation
is required for this definition.
The quadratic form in $g$:
\[
  Q_c^N(g) := \sum_{n>N}|a_n^{(c)}|^2
  = \norm{(I-\Pi_N^{(c)})\Pc g}^2.
\]
The obstruction to H1 is
\emph{spectral non-evacuation}:
the failure of $\mu_{n,c}([-c_0,c_0])$
to vanish for $n>N$, uniformly in $c$.
\end{definition}

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $0<\alpha<\kappa<1$:
$\mathcal{T}_{\alpha,\kappa}(c)
:=\{n: \alpha c < n \le \kappa c\}$.
This band has $\sim(\kappa-\alpha)c$ elements
and captures the regime where Slepian
eigenvalues $\lambda_n^{(c)}$
transit between $\approx 1$ and $\approx 0$.
\end{definition}

\begin{lemma}[From series]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item $\sum_n|a_n^{(c)}|^2\le\norm{g}^2$;
  $\norm{\Phic{c}{n}}_{L^2}=1$.
\item $a_n^{(c)}\to a_n^{(\infty)}$ (each fixed $n$).
\item For $n\le\kappa c$:
  $|\widehat{\Phi_n^{(c)}}(\xi)|\ge c_1 c^{-1/2}$
  on $E_c^{(n)}\subseteq[-c,c]$,
  $|E_c^{(n)}|\ge c_* c^{-1/2}$.
\item $\operatorname{supp}\hat g\subseteq[-c_0,c_0]$.
\end{enumerate}
\end{lemma}

%%‚ïê‚ïê‚ïê 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{theorem}[$\norm{T_c}\le 1$]
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{theorem}
\begin{proof}
Bessel:
$\norm{T_c g}^2=\sum_n|\ip{g}{\Phic{c}{n}}|^2
\le\norm{g}^2$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds unconditionally.
\end{proposition}
\begin{proof}
Paper~XII: $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$.
\end{proof}

\begin{remark}
By weak lower semicontinuity:
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$,
so H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}\le\norm{a^{(\infty)}}$.
\end{remark}

%%‚ïê‚ïê‚ïê 4. TAIL REDUCTION AND THE QUADRATIC FORM
\section{Tail Reduction, Operator Structure, and the Quadratic Form}
\label{sec:tail}

\begin{remark}[Foundational conventions]
\label{rem:foundation}
Throughout this section the following
conventions apply:
\begin{itemize}[leftmargin=*]
\item $S_N(c)$ is \emph{defined} by the
  convergent series~\eqref{eq:SN-def};
  it is not defined via a global operator
  trace on $L^2(\mathbb{R})$.
\item $P_{c_0}:L^2(\mathbb{R})\to\PW_{c_0}^2$
  is the orthogonal projection (Fourier
  multiplier by $\mathbf{1}_{[-c_0,c_0]}$).
  Its integral kernel is the translation-
  invariant sinc kernel
  $K(x,y)=\sin(c_0(x-y))/(\pi(x-y))$,
  which satisfies
  $\|P_{c_0}\|_{\mathrm{HS}}^2
  =\int_{\mathbb{R}}\int_{\mathbb{R}}|K(x-y)|^2\,dx\,dy
  =\infty$.
  Hence $P_{c_0}$ is \emph{not}
  Hilbert--Schmidt on $L^2(\mathbb{R})$,
  and no such property is used anywhere below.
\item All operator norm and spectral sum
  inequalities are derived via
  finite-dimensional truncations $\Ac^{(M)}$
  (defined below) followed by a monotone
  limit $M\to\infty$.
  Eigenvalue-counting arguments are carried
  out \emph{only} at finite truncation level
  and are \emph{never} passed to the limit.
\end{itemize}
\end{remark}

\begin{definition}[$S_N(c)$, $\sigma(c)$, and finite truncations]
\label{def:SN}
\begin{align*}
  S_N(c) &:= \sum_{n>N}\mu_{n,c}([-c_0,c_0])
            = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2},\\
  S_N^{(M)}(c) &:= \sum_{n=N+1}^{M}
            \mu_{n,c}([-c_0,c_0]),
            \qquad M>N,\\
  \sigma(c) &:= \sum_{n\ge 0}\mu_{n,c}([-c_0,c_0]).
\end{align*}
Each term $\mu_{n,c}([-c_0,c_0])\in[0,1]$;
the series $S_N(c)$ converges by
Lemma~\ref{lem:inherited}(A),
and $S_N^{(M)}(c)\nearrow S_N(c)$ monotonely.

\medskip\noindent
\emph{Divergence of $\sigma(c)$ vs.\
vanishing of $S_N(c)$.}
By the classical Slepian--Weyl result
(Remark~\ref{rem:sigma-slepian}),
$\sigma(c)\sim cc_0/\pi\to\infty$.
This does \emph{not} preclude $S_N(c)\to 0$:
the total spectral mass can grow while
the tail $\{n>N\}$ evacuates.
H3 asks precisely whether this tail evacuation
occurs uniformly in $c$.
\end{definition}

\begin{remark}[Classical status of $\sigma(c)$]
\label{rem:sigma-slepian}
The asymptotics $\sigma(c)\sim cc_0/\pi$
are classical Slepian--Weyl \cite{Slepian1961},
established in the time-limited / band-limited
operator framework on $L^2[-1,1]$.
We use this only as an asymptotic reference;
no global trace-class claim on
$L^2(\mathbb{R})$ is made or needed.
\end{remark}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
\[
  Q_c^N(g):=\sum_{n>N}|a_n^{(c)}|^2
  \le\norm{g}^2\cdot S_N(c).
\]
\end{lemma}
\begin{proof}
Since $\operatorname{supp}\hat g\subseteq[-c_0,c_0]$:
\[
  a_n^{(c)}
  =\ip{\hat g}{
    \widehat{\Phi_n^{(c)}}\cdot
    \mathbf{1}_{[-c_0,c_0]}}_{L^2}.
\]
Cauchy--Schwarz:
$|a_n^{(c)}|^2
\le\norm{g}^2\cdot\mu_{n,c}([-c_0,c_0])$.
Sum over $n>N$ and use~\eqref{eq:SN-def}.
\end{proof}

\begin{corollary}[Equi-tightness]
$\sup_c S_N(c)\to 0
\Rightarrow\{a^{(c)}\}_c$
equi-tight in $\ell^2$.
\end{corollary}

\begin{theorem}[Operator structure and spectral sum identity]
\label{thm:SN-operator}
Define the global operators:
\begin{align}\label{eq:Bn-def}
  \Bn &:= (I-\Pi_N^{(c)})\Pc
    \;:\; \PW_{c_0}^2\to\PW_c^2,\\
  \Ac &:= \Bn^*\Bn
    \;:\; \PW_{c_0}^2\to\PW_{c_0}^2,
\end{align}
and for each finite $M>N$ the
finite-dimensional compression:
\begin{align}\label{eq:BnM-def}
  V_M^{(c)} &:= \mathrm{span}
    \{\Phic{c}{n} : N < n \le M\},\\
  \Bn^{(M)} &:=
    (I-\Pi_N^{(c)})\Pc\big|_{V_M^{(c)}},\\
  \Ac^{(M)} &:= (\Bn^{(M)})^*\Bn^{(M)}
    \;:\; V_M^{(c)}\to V_M^{(c)}.
\end{align}

\medskip
\textbf{(0) Contraction.}
Both $\Pc$ and $(I-\Pi_N^{(c)})$
are orthogonal projections,
hence contractions of norm $\le 1$.
Their composition satisfies:
\begin{equation}\label{eq:Bn-contraction}
  \norm{\Bn}_{\mathrm{op}}
  \le\norm{(I-\Pi_N^{(c)})}_{\mathrm{op}}
     \cdot\norm{\Pc}_{\mathrm{op}}
  \le 1.
\end{equation}
Consequently:
\begin{equation}\label{eq:Ac-bounded}
  0\le\Ac=\Bn^*\Bn\le I,
  \qquad
  \text{all eigenvalues of }\Ac\text{ in }[0,1].
\end{equation}
The same bound applies to each $\Ac^{(M)}$:
$0\le\Ac^{(M)}\le I$ with
eigenvalues $\lambda_j^{(M)}\in[0,1]$.
No Hilbert--Schmidt property of $\Pc$
on $L^2(\mathbb{R})$ is used.

\medskip
\textbf{(I) Spectral sum identity.}
For each finite $M$, $\Ac^{(M)}$ is a
positive semidefinite $(M-N)\times(M-N)$
matrix on the finite-dimensional space
$V_M^{(c)}$, so its trace is
basis-independent and equals the sum
of its eigenvalues:
\begin{equation}\label{eq:trace-M}
  \mathrm{Tr}(\Ac^{(M)})
  = \sum_{j}\lambda_j^{(M)}
  = \sum_{n=N+1}^{M}
    \ip{\Ac^{(M)}\Phic{c}{n}}{\Phic{c}{n}}
  = \sum_{n=N+1}^{M}
    \norm{\Bn^{(M)}\Phic{c}{n}}^2
  = S_N^{(M)}(c).
\end{equation}
Taking $M\to\infty$:
$S_N^{(M)}(c)\nearrow S_N(c)$,
and we define
\begin{equation}\label{eq:TrPSWF}
  \mathrm{Tr}_{\mathrm{PSWF}}(\Ac)
  := \lim_{M\to\infty}\mathrm{Tr}(\Ac^{(M)})
  = S_N(c).
\end{equation}
This is the limit of genuine finite-dimensional
traces; it is \emph{not} asserted to be a
trace in $\mathfrak{S}_1(L^2(\mathbb{R}))$,
nor is it interpreted as a sum of eigenvalues
of the global operator $\Ac$.
At each finite stage~\eqref{eq:trace-M}
is exact linear algebra.

\medskip
\textbf{(II) Operator norm inequality via finite truncation.}
At each finite stage, since $\Ac^{(M)}$
is a positive semidefinite matrix with
eigenvalues $\lambda_j^{(M)}\in[0,1]$:
\begin{equation}\label{eq:op-le-trace-M}
  \norm{\Ac^{(M)}}_{\mathrm{op}}
  = \max_j\lambda_j^{(M)}
  \le \sum_j\lambda_j^{(M)}
  = \mathrm{Tr}(\Ac^{(M)})
  = S_N^{(M)}(c).
\end{equation}
This is exact finite-dimensional linear algebra:
$\lambda_{\max}\le\sum_j\lambda_j$
holds for any positive semidefinite matrix
with eigenvalues in $[0,1]$.
As $M$ increases, $\norm{\Ac^{(M)}}_{\mathrm{op}}$
is non-decreasing
(compressions to larger subspaces
have no smaller operator norm), so:
\begin{equation}\label{eq:op-le-spectral}
  \norm{\Ac}_{\mathrm{op}}
  = \lim_{M\to\infty}\norm{\Ac^{(M)}}_{\mathrm{op}}
  \le \lim_{M\to\infty}S_N^{(M)}(c)
  = S_N(c).
\end{equation}
The inequality is strict in general.
This argument uses \emph{no} global
trace-class or Hilbert--Schmidt property.

\medskip
\textbf{(III) Quadratic form.}
For $g\in\PW_{c_0}^2$,
using $\Pc g=g$:
\begin{equation}\label{eq:Qc-key}
  Q_c^N(g)
  = \sum_{n>N}|\ip{g}{\Phic{c}{n}}|^2
  = \norm{\Bn g}^2
  = \ip{\Ac g}{g},
  \qquad 0\le\Ac\le I.
\end{equation}

\medskip
\textbf{(IV) The full hierarchy.}
\begin{equation}\label{eq:chain}
  \underbrace{\text{H3}}_{S_N(c)\to 0}
  \;\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0
  \;\Rightarrow\;
  Q_c^N(g)\to 0\;(\text{all }g)
  \;\Rightarrow\;
  \text{H1}
  \;\Rightarrow\;
  \ip{\Ac g}{g}\to 0\;(\text{each fixed }g).
\end{equation}
The reverse arrows do not hold in general.
H3 is a spectral-sum condition on the
PSWF-ONB; H1 yields only pointwise
quadratic form convergence (diagonal
WOT-type, not full SOT).

\medskip
\begin{center}
\fbox{\parbox{0.84\linewidth}{
\textbf{Key distinction.}
The obstruction is not the size of the
spectral mass $S_N(c)$, but the
\emph{distribution of the spectral mass
across eigenvalues} of $\Ac^{(M)}$.
A large trace is compatible with all mass
concentrated on a single eigenvalue;
controlling the number of large eigenvalues
requires spectral distribution input
beyond the trace alone
(see \S\ref{sec:szego} and
OP~\ref{prob:H1H3}).}}
\end{center}
\end{theorem}

\begin{remark}[Convergence hierarchy]
\label{rem:convergence-hierarchy}
The chain~\eqref{eq:chain} distinguishes
four levels:
\begin{enumerate}[\rm(1)]
\item \emph{H3} ($S_N(c)\to 0$):
  strongest; controls all terms of
  $S_N(c)$ simultaneously.
\item \emph{Op-norm $\to 0$}:
  controls the largest eigenvalue
  uniformly in direction.
\item \emph{$Q_c^N(g)\to 0$ uniformly in $g$}:
  equivalent to op-norm $\to 0$.
\item \emph{$\ip{\Ac g}{g}\to 0$ each fixed $g$}
  (diagonal WOT-type): weakest;
  proved from H1.
  Not equivalent to SOT.
\end{enumerate}
The gap between (4) and (2) is the
finite-dimensional effective rank
growth conjecture (OP~\ref{prob:H1H3}),
identified in \S\ref{sec:szego} as a
semiclassical Szeg\H{o}-type problem.
\end{remark}

\begin{remark}[Spectral sum vs.\ operator norm;
  separation of trace and spectral counting]
\label{rem:HS-trace}
The inequality~\eqref{eq:op-le-spectral}
follows from finite truncation and
monotone limit, not from any global
trace-class argument.

\medskip\noindent
\emph{Warning on spectral counting.}
The quantity $S_N(c)=\mathrm{Tr}_{\mathrm{PSWF}}(\Ac)$
is defined as a limit of finite-dimensional
traces~\eqref{eq:TrPSWF}; it controls
the spectral \emph{sum} of $\Ac$ in the
PSWF-ONB sense.
It does \emph{not} directly control the
number of large eigenvalues of the global
operator $\Ac$: eigenvalue counting is not
stable under monotone operator limits
(only upper semicontinuity of individual
eigenvalues holds in general).
All eigenvalue-counting results in this paper
(Lemma~\ref{lem:rank-lower}) are therefore
stated exclusively at finite truncation
level $M$ and are not passed to the limit.

\medskip\noindent
The structural relation available is:
\begin{equation}\label{eq:op-SN-relation}
  \norm{\Ac}_{\mathrm{op}} \le S_N(c),
  \qquad
  \norm{\Ac^{(M)}}_{\mathrm{op}}
  \le S_N^{(M)}(c)
  \le \norm{\Ac^{(M)}}_{\mathrm{op}}\cdot(M-N).
\end{equation}
The right-hand inequality reflects that
the trace of a positive semidefinite
$(M-N)\times(M-N)$ matrix is at most
its operator norm times its dimension.
No global analogue of this relation is
asserted for $\Ac$.
\end{remark}

\begin{lemma}[Finite-dimensional effective rank bound]
\label{lem:rank-lower}
Let $c>c_0$, $N\ge 0$, $M>N$, and let
$\Ac^{(M)}$ be the finite-dimensional
compression on
$V_M^{(c)}=\mathrm{span}\{\Phic{c}{n}:N<n\le M\}$
as in Theorem~\ref{thm:SN-operator}.
Denote its eigenvalues by
$\lambda_1^{(M)},\dots,\lambda_{M-N}^{(M)}\in[0,1]$
and set
$S_N^{(M)}(c)=\sum_{j=1}^{M-N}\lambda_j^{(M)}
=\mathrm{Tr}(\Ac^{(M)})$.

Then for every $\varepsilon\in(0,1)$:
\begin{equation}\label{eq:eff-rank-M}
  \#\{j:\lambda_j^{(M)}\ge\varepsilon\}
  \;\ge\;
  \frac{S_N^{(M)}(c)-\varepsilon(M-N)}{1-\varepsilon}.
\end{equation}
\end{lemma}

\begin{proof}
Split the trace at the threshold $\varepsilon$:
\[
  S_N^{(M)}(c)
  =\sum_{\lambda_j^{(M)}\ge\varepsilon}\lambda_j^{(M)}
  +\sum_{\lambda_j^{(M)}<\varepsilon}\lambda_j^{(M)}.
\]
Estimate each part. For the first sum,
since each $\lambda_j^{(M)}\le 1$:
\[
  \sum_{\lambda_j^{(M)}\ge\varepsilon}\lambda_j^{(M)}
  \le\#\{j:\lambda_j^{(M)}\ge\varepsilon\}.
\]
For the second sum, since there are at most
$M-N$ eigenvalues in total and each
$\lambda_j^{(M)}<\varepsilon$:
\[
  \sum_{\lambda_j^{(M)}<\varepsilon}\lambda_j^{(M)}
  \le\varepsilon(M-N).
\]
Combining:
\[
  S_N^{(M)}(c)
  \le\#\{j:\lambda_j^{(M)}\ge\varepsilon\}
  +\varepsilon(M-N).
\]
Rearranging yields~\eqref{eq:eff-rank-M}.
\end{proof}

\begin{remark}[Scope and limitations of
  Lemma~\ref{lem:rank-lower}]
\label{rem:rank-scope}
Lemma~\ref{lem:rank-lower} is a statement
about the finite matrix $\Ac^{(M)}$ only.
The right-hand side of~\eqref{eq:eff-rank-M}
contains the dimension-dependent term
$\varepsilon(M-N)$, which grows with $M$.
Consequently:
\begin{enumerate}[\rm(i)]
\item The bound~\eqref{eq:eff-rank-M}
  is non-trivial only when
  $S_N^{(M)}(c)>\varepsilon(M-N)$,
  i.e.\ when the \emph{mean spectral density}
  \[
    \bar\lambda^{(M)}(c)
    := \frac{S_N^{(M)}(c)}{M-N}
  \]
  satisfies $\bar\lambda^{(M)}(c)>\varepsilon$.
\item No passage $M\to\infty$ is performed
  in~\eqref{eq:eff-rank-M}.
  In particular, no claim is made about
  eigenvalue counts for the global
  operator $\Ac$.
\item The open question (OP~\ref{prob:H1H3})
  is whether there exists a scale
  $M=M(c)\to\infty$ such that
  $\bar\lambda^{(M(c))}(c)\not\to 0$
  as $c\to\infty$:
  a \emph{non-vanishing mean spectral density}
  condition along the truncation scale.
\end{enumerate}
\end{remark}

\begin{remark}[Spectral sum vs.\ op-norm:
  independence warning]
\label{rem:rank-obstruction}
In general:
$S_N(c)\to 0
\not\Rightarrow
\norm{\Ac}_{\mathrm{op}}\to 0$.
The implication holds if there exists
$M_0<\infty$ such that
$\norm{\Ac}_{\mathrm{op}}
=\norm{\Ac^{(M_0)}}_{\mathrm{op}}$
(i.e.\ $\Ac$ is effectively finite-rank
at scale $M_0$ uniformly in $c$):
then $\norm{\Ac}_{\mathrm{op}}
\le S_N^{(M_0)}(c)\le S_N(c)\to 0$.
But this uniformity is precisely what
OP~\ref{prob:H1H3} asks for,
and it is not available from the spectral
sum $S_N(c)$ alone.
\end{remark}

%%‚ïê‚ïê‚ïê 5. SPECTRAL MEASURES AND NC
\section{Spectral Measures and No-Concentration}
\label{sec:NC}

\begin{remark}[NC as spectral evacuation failure]
\label{rem:NC-evac}
The obstruction to H1 stated cleanly:
\begin{quote}
  \emph{H1 fails if and only if
  there exists persistent spectral mass
  of the measures $\mu_{n,c}$
  in the transition zone $n\sim\alpha c$
  (conditional on OP~\ref{prob:H1H3}
  and OP~\ref{prob:fourier-mass}).}
\end{quote}
``Spectral evacuation'' means
$S_N(c)\to 0$ uniformly in $c$.
NC is the quantitative band-level sufficient
condition for non-evacuation.
Proposition~\ref{prop:NC-not-H3}
shows NC is sufficient but not necessary
for H3 to fail.
\end{remark}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
Fix $0<\alpha<\kappa<1$.
NC holds if there exists $c_*>0$ such that
for all large $c$:
\begin{equation}\label{eq:NC}
  \sum_{n\in\mathcal{T}_{\alpha,\kappa}(c)}
  \mu_{n,c}([-c_0,c_0]) \ge c_*>0.
\end{equation}
\end{definition}

\begin{remark}[Why proportional width is necessary]
\label{rem:prop-width}
A window of width $c^\beta$ ($\beta<1$)
has $\sim 2c^\beta$ indices;
under the local Weyl law,
the sum is $\sim 2c_0 c^{\beta-1}\to 0$:
insufficient for NC.
A proportional band
$\mathcal{T}_{\alpha,\kappa}(c)$
has $\sim(\kappa-\alpha)c$ indices,
giving sum $\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[NC as quantum ergodicity]
\label{rem:NC-QE}
In the semiclassical limit $c\to\infty$,
$n\sim\alpha c$, the rescaled measures
$c\cdot\mu_{n,c}|_{[-c,c]}$
are expected to converge weakly to
the WKB density
$(\pi\sqrt{1-(\xi/c)^2})^{-1}d\xi$:
a one-dimensional quantum ergodicity
statement \cite{Zworski2012}.
NC is the consequence for $[-c_0,c_0]$:
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$,
so the band sum
$\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[Local Weyl law $\Rightarrow$ NC]
\label{rem:Weyl-NC}
If uniformly for $n\in\mathcal{T}_{\alpha,\kappa}(c)$:
\begin{equation}\label{eq:local-Weyl}
  \mu_{n,c}([-c_0,c_0])\sim\frac{c_0}{c},
\end{equation}
then $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
\sim c_0(\kappa-\alpha)>0$.
NC holds with $c_*=c_0(\kappa-\alpha)/2$.
\end{remark}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under NC: for each fixed $N$,
$\liminf_{c\to\infty}S_N(c)\ge c_*>0$,
so H3 fails.
\end{proposition}
\begin{proof}
For large $c$: $\alpha c>N$,
hence $\mathcal{T}_{\alpha,\kappa}(c)\subseteq\{n>N\}$
and $S_N(c)\ge c_*$.
\end{proof}

\begin{proposition}[NC$^c$ does not imply H3;
  logical separation]
\label{prop:NC-not-H3}
\emph{Scope and categorical disclaimer.
This is a logical separation result
in the abstract category of families
of probability measures
$\{\mu_{n,c}\}$ satisfying only
normalisation and positivity.
It does not constrain PSWF spectral geometry:
the model family does not arise from
$|\widehat{\Phi_n^{(c)}}|^2\,d\xi$,
and PSWF realizability is neither
asserted nor needed.
In particular, the model below
concentrates mass in a sub-proportional
window of width $\sim\sqrt{c}$, which is
incompatible with semiclassical
equidistribution (local Weyl law);
this confirms the abstract separation
is not PSWF-realistic, while remaining
logically relevant for the implication
structure.}

\medskip\noindent
There exists a model family
$\{\mu_{n,c}\}$ with $\mu_{n,c}\ge 0$,
$\mu_{n,c}(\mathbb{R})=1$, such that:
\begin{enumerate}[\rm(i)]
\item\label{it:NCc}
  \textup{NC fails}:
  for every $0<\alpha'<\kappa'<1$
  and every $c_*>0$,
  $\liminf_{c\to\infty}
  \sum_{n\in\mathcal{T}_{\alpha',\kappa'}(c)}
  \mu_{n,c}([-c_0,c_0]) < c_*$;
\item\label{it:H3fail}
  \textup{H3 fails}:
  for every fixed $N\ge 0$,
  $\liminf_{c\to\infty} S_N(c) \ge 1$.
\end{enumerate}
In particular, NC is not necessary for H3
to fail: $\mathrm{NC}^c\not\Rightarrow\mathrm{H3}$
at the abstract level.
\end{proposition}

\begin{proof}
Fix $\alpha\in(0,1)$.
Set $w(c):=\lfloor\sqrt{c}\rfloor$ and
\[
  I_c := \{n\in\mathbb{N}_0 :
    |n-\lfloor\alpha c\rfloor|\le w(c)\},
  \qquad |I_c|=2w(c)+1.
\]
Set $\mu_{n,c}([-c_0,c_0])
:=1/|I_c|$ for $n\in I_c$,
$0$ otherwise; extend to a probability
measure by placing remaining mass
outside $[-c_0,c_0]$.

\textit{Proof of~\ref{it:H3fail}.}
For $c>((N+1)/\alpha)^2$:
$\lfloor\alpha c\rfloor-w(c)>N$,
so $I_c\subseteq\{n>N\}$,
giving $S_N(c)\ge 1$.

\textit{Proof of~\ref{it:NCc}.}
\textit{Case~1}: $\alpha\notin[\alpha',\kappa']$.
For large $c$, $I_c\cap
\mathcal{T}_{\alpha',\kappa'}(c)=\emptyset$,
so the band sum is $0$.

\textit{Case~2}: $\alpha\in(\alpha',\kappa')$.
Set $\varepsilon
:=\tfrac{1}{2}\min(\alpha-\alpha',\kappa'-\alpha)$.
For large $c$, all $n\in I_c$ satisfy
$|n/c-\alpha|<\varepsilon/2$;
choosing $\beta$ with
$\alpha<\beta<\kappa'$, $\beta-\alpha>\varepsilon$,
gives $I_c\cap\mathcal{T}_{\beta,\kappa'}(c)
=\emptyset$, so NC fails for
$\mathcal{T}_{\beta,\kappa'}$.
\end{proof}

\begin{remark}[Structural separation]
\label{rem:separation}
The model concentrates mass in a
sub-proportional zone $|I_c|\sim 2\sqrt{c}\ll c$,
escaping every fixed proportional window,
while maintaining $S_N(c)\ge 1$.
This sub-proportional localisation is
incompatible with semiclassical
equidistribution ($\mu_{n,c}\sim c_0/c$
requires a proportional count of modes),
confirming the abstract character of the
separation.
NC is sufficient but not necessary
for H3 to fail.
PSWF realizability is addressed by
OP~\ref{prob:fourier-mass}.
\end{remark}

%%‚ïê‚ïê‚ïê 6. H3 AND THE MAIN REDUCTION
\section{H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{remark}[H3 vs.\ classical Slepian asymptotics]
\label{rem:H3-vs-Slepian}
Classical Slepian theory fixes $c$ and studies
$\lambda_n^{(c)}$ near the transition count
$n\sim 2c/\pi$.
H3 fixes $N$ and requires
$\sup_{c>c_0}S_N(c)\to 0$ as $N\to\infty$:
a uniform-in-$c$, large-$N$ condition,
strictly stronger than any fixed-$c$ result.
\end{remark}

\begin{theorem}[H3 $\Rightarrow$ H1]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$, $g\in\PW_{c_0}^2$.
H3: choose $N$ with
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$;
Lemma~\ref{lem:tail}: $Q_c^N(g)<\varepsilon/2$
for all $c$.
Paper~XII: choose $c_\varepsilon$ with
$\sum_{n\le N}|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ for $c>c_\varepsilon$.
Adding gives $\norm{a^{(c)}-a^{(\infty)}}^2<\varepsilon$.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$
  pointwise quadratic form convergence]
\label{thm:H1-H3-ptwise}
Assume H1. For each fixed $g\in\PW_{c_0}^2$
and each fixed $N$:
$\ip{\Ac g}{g}=Q_c^N(g)\to 0$
as $c\to\infty$, then $N\to\infty$.
This is diagonal WOT-type convergence;
it is not uniform in $g$ and does not
imply $\norm{\Ac}_{\mathrm{op}}\to 0$.
\end{theorem}
\begin{proof}
$Q_c^N(g)
\le 2\norm{a^{(c)}-a^{(\infty)}}^2_{\ell^2}
+2\sum_{n>N}|a_n^{(\infty)}|^2$.
H1: first term $\to 0$.
$a^{(\infty)}\in\ell^2$: second $\to 0$.
\end{proof}

\begin{remark}[The uniformity gap]
\label{rem:unif-gap}
\[
  \text{H3}\Rightarrow
  \norm{\Ac}_{\mathrm{op}}\to 0\Rightarrow
  \text{H1}\Rightarrow
  \ip{\Ac g}{g}\to 0
  \;(\text{diagonal WOT}).
\]
The missing reverse:
diagonal WOT $\not\Rightarrow$
$\norm{\Ac}_{\mathrm{op}}\to 0$.
Closing this gap requires a non-vanishing
mean spectral density along some truncation
scale $M(c)$ (OP~\ref{prob:H1H3}),
equivalently a semiclassical Szeg\H{o}-type
law for the truncated Gram operator
(\S\ref{sec:szego}).
\end{remark}

\begin{theorem}[Main reduction]
\label{thm:reduction}
\textup{(i)} Proved unconditionally:
\begin{align*}
  &\text{H3}\Rightarrow\text{H1},\quad
  \text{H1}\Rightarrow\ip{\Ac g}{g}\to 0
  \;(\text{diagonal WOT}),\\
  &\text{NC}\Rightarrow\text{H3 fails},\quad
  \text{local Weyl}\Rightarrow\text{NC},\\
  &\text{NC sufficient, not necessary for H3 to fail}
  \;(\text{Prop.}~\ref{prop:NC-not-H3}).
\end{align*}
\textup{(ii)} Open:
whether the mean spectral density
$\bar\lambda^{(M(c))}(c)
:=S_N^{(M(c))}(c)/M(c)\not\to 0$
(equivalently, $S_N^{(M(c))}(c)\gtrsim M(c)$)
along some scale $M(c)\to\infty$
(OP~\ref{prob:H1H3});
NC holds
(OP~\ref{prob:fourier-mass}).

\textup{(iii)} Conditional chain:
$\text{local Weyl}\Rightarrow
\text{NC}\Rightarrow
\text{H3 fails}\Rightarrow
\text{H1 fails}.$
\end{theorem}

\subsection{The true obstruction: spectral non-evacuation}

  \quad(\text{cond. on OPs~\ref{prob:H1H3},

\]
``Spectral evacuation'' means $S_N(c)\to 0$
uniformly in $c$.
The model of Proposition~\ref{prop:NC-not-H3}
shows mass can persist in $S_N(c)$ without
any single proportional band carrying it.

\subsection{NC as band-level; H3 as global-tail}

\subsection{$S_N(c)$ controls spectral sum, not spectral distribution}

The fundamental separation established in this paper:
\[
  \norm{\Ac}_{\mathrm{op}} \le S_N(c)
  \qquad\text{(proved via finite truncation)},
\]
\[
  S_N^{(M)}(c) \le \norm{\Ac^{(M)}}_{\mathrm{op}}\cdot(M-N)
  \qquad\text{(finite-dimensional; not passed to limit)}.
\]
$S_N(c)$ is a spectral sum in the PSWF-ONB sense.
It controls the operator norm from above.
It does \emph{not} control how many eigenvalues
are large, nor the mean spectral density
$\bar\lambda^{(M)}(c)=S_N^{(M)}(c)/(M-N)$
at growing scales: both require spectral
distribution input (OP~\ref{prob:H1H3}).

\subsection{Quantum ergodicity in frequency space}

The spectral measures $\mu_{n,c}$
are semiclassical measures;
the local Weyl law (OP~\ref{prob:fourier-mass})
asserts equidistribution on $[-c_0,c_0]$
\cite{Zworski2012}.

\subsection{Connection to semiclassical Szeg\H{o} theory}
\label{sec:szego}

\subsubsection*{Identification of $\Ac^{(M)}$ as a truncated Gram operator}

The matrix $\Ac^{(M)}$ is the Gram matrix
of the restricted Fourier transforms:
\begin{equation}\label{eq:gram}
  (\Ac^{(M)})_{nm}
  = \ip{\Pc\Phic{c}{n}}{\Pc\Phic{c}{m}}_{L^2}
  = \int_{-c_0}^{c_0}
    \widehat{\Phi_n^{(c)}}(\xi)\,
    \overline{\widehat{\Phi_m^{(c)}}(\xi)}\,d\xi,
  \quad N < n,m \le M.
\end{equation}
Its diagonal entries are
$(\Ac^{(M)})_{nn}=\mu_{n,c}([-c_0,c_0])$;
its trace is $S_N^{(M)}(c)$.
The eigenvalues of $\Ac^{(M)}$ encode
the distribution of Fourier energy in
$[-c_0,c_0]$ across the PSWF modes
$N<n\le M$, with off-diagonal entries
measuring correlations between modes.

\subsubsection*{Toeplitz structure and phase space symbol}

In the semiclassical limit $c\to\infty$,
$n\sim\alpha c$, the operator $\Ac$
is heuristically a \emph{semiclassical
Toeplitz operator} with phase space symbol
\[
  a(x,\xi) = \mathbf{1}_{|\xi|\le c_0}
  \cdot \mathbf{1}_{\text{transition zone}},
\]
acting on the PSWF eigenbasis.
The Toeplitz structure arises because
$\Ac = \Pc(I-\Pi_N^{(c)})\Pc$ is a
compression of the frequency-projection
operator $\Pc$ to the complement of the
first $N$ PSWF modes.
In this framework, the classical Szeg\H{o}
theorem predicts that the eigenvalue
distribution of $\Ac^{(M)}$, as
$M\to\infty$ and $c\to\infty$ simultaneously,
converges to a distribution determined
by the phase space overlap of the
Paley--Wiener band $|\xi|\le c_0$
and the transition zone $n\sim\alpha c$
\cite{BottcherSilbermann1990,Simon2011}.

\subsubsection*{The open problem as a Szeg\H{o}-type question}

Under the local Weyl law
(Remark~\ref{rem:Weyl-NC}),
each diagonal entry satisfies
$(\Ac^{(M)})_{nn}=\mu_{n,c}([-c_0,c_0])
\sim c_0/c$,
giving $S_N^{(M)}(c)\sim c_0(M-N)/c$.
The mean spectral density is then
\[
  \bar\lambda^{(M)}(c)
  = \frac{S_N^{(M)}(c)}{M-N}
  \sim \frac{c_0}{c}.
\]
For this to remain bounded away from zero,
one needs $M(c)\lesssim c$,
i.e.\ the truncation scale must grow
at most linearly in $c$.
The key additional ingredient is the
\emph{quasi-orthogonality} of the
restricted Fourier transforms:
\begin{equation}\label{eq:quasi-orth}
  \left|\int_{-c_0}^{c_0}
  \widehat{\Phi_n^{(c)}}(\xi)\,
  \overline{\widehat{\Phi_m^{(c)}}(\xi)}\,d\xi
  \right| = o\!\left(\frac{1}{c}\right)
  \quad(n\ne m,\; n,m\in\mathcal{T}_{\alpha,\kappa}(c)),
\end{equation}
which would imply that the off-diagonal
entries of $\Ac^{(M)}$ are negligible
relative to the diagonal, so the
eigenvalues of $\Ac^{(M)}$ are well
approximated by the diagonal entries
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$,
giving a non-degenerate mean spectral
density $\bar\lambda^{(M(c))}(c)\sim c_0/c>0$
for $M(c)=O(c)$.

\medskip\noindent
Open Problem~\ref{prob:H1H3} can thus be
precisely reformulated:

\begin{center}
\fbox{\parbox{0.84\linewidth}{
\textbf{Semiclassical Szeg\H{o} reformulation
of OP~\ref{prob:H1H3}.}
Does the eigenvalue distribution of the
truncated Gram operator $\Ac^{(M(c))}$
converge to a non-degenerate density as
$c\to\infty$ and $M(c)\to\infty$?
Equivalently: does~\eqref{eq:quasi-orth}
hold, so that $\Ac^{(M)}$ is
asymptotically diagonal in the PSWF basis,
with non-vanishing mean spectral density
$\bar\lambda^{(M(c))}(c)\not\to 0$?}}
\end{center}

\begin{conjecture}[Semiclassical Szeg\H{o} behaviour]
\label{conj:szego}
The eigenvalue distribution of $\Ac^{(M(c))}$
converges, as $c\to\infty$ with
$M(c)\sim\beta c$ for fixed $\beta\in(0,1)$,
to a non-degenerate density concentrated
near $c_0/c$, determined by the phase
space overlap of the Paley--Wiener band
$|\xi|\le c_0$ and the transition zone
$n\in\mathcal{T}_{\alpha,\kappa}(c)$.
In particular:
\[
  \bar\lambda^{(M(c))}(c)
  = \frac{S_N^{(M(c))}(c)}{M(c)}
  \;\xrightarrow{c\to\infty}\;
  \frac{c_0}{c}\cdot(1+o(1)) > 0,
\]
so that
$S_N^{(M(c))}(c)\gtrsim M(c)\cdot c_0/c\to\infty$
and $\#\{j:\lambda_j^{(M(c))}\ge\varepsilon\}\to\infty$
for all sufficiently small $\varepsilon>0$.
\end{conjecture}

\begin{remark}[Programme status]
Conjecture~\ref{conj:szego} is a programme,
not a proved statement.
Its verification requires:
(1)~the local Weyl law~\eqref{eq:local-Weyl}
(target of Paper~XV);
(2)~the quasi-orthogonality
estimate~\eqref{eq:quasi-orth}
(off-diagonal decay of $\Ac^{(M)}$;
not yet established);
(3)~a semiclassical Szeg\H{o}-type theorem
for the resulting approximately-diagonal
positive semidefinite matrix family.
Steps (1)--(3) together constitute the
full programme of Paper~XV and beyond.
\end{remark}

\section{Programme for Paper~XV}
\label{sec:PaperXV}

\subsection{The three required estimates}

\begin{enumerate}[\rm(A)]
\item\label{XV-A} \textbf{Upper bound:}
  $\mu_{n,c}([-c_0,c_0])\le C/c$,
  uniform in $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-B} \textbf{Lower bound:}
  $\mu_{n,c}([-c_0,c_0])\ge 1/(Cc)$,
  uniform in $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-C} \textbf{Summability:}
  $\mu_{n,c}([-c_0,c_0])=c_0/c+o(1/c)$
  uniformly, giving
  $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
  \to c_0(\kappa-\alpha)$.
\end{enumerate}
(A)+(B) $\Rightarrow$ NC;
(C) $\Rightarrow$ equidistribution.

\subsection{Semiclassical approach}

$h:=1/c\to 0$; the PSWF solves
\begin{equation}\label{eq:PSWF-ODE}
  -(1-x^2)\partial_x^2\psi
  +2x\partial_x\psi
  +c^2x^2\psi=
    \chi_n^{(c)}\psi
\end{equation}
with $E_n^{(h)}=\chi_n^{(c)}/c^2\to E_\alpha$
for $n\sim\alpha c$.
WKB in $|x|<x_0$ ($x_0^2=E_\alpha$)
and stationary phase for $\xi\in[-c_0,c_0]$
give $\mu_{n,c}([-c_0,c_0])
\sim 2c_0C(E_\alpha)/c$,
yielding (A)--(C).

\subsection{Rigorous programme}

\begin{enumerate}
\item WKB with remainder $O(c^{-2})$,
  uniform in $n\in\mathcal{T}$
  \cite{Osipov2013}.
\item Stationary phase with uniform remainder.
\item Prove (A)--(C); conclude NC.
\item Establish quasi-orthogonality~\eqref{eq:quasi-orth}
  via off-diagonal stationary phase:
  for $n\ne m$ in $\mathcal{T}_{\alpha,\kappa}(c)$,
  the integral
  $\int_{-c_0}^{c_0}
  \widehat{\Phi_n^{(c)}}(\xi)\,
  \overline{\widehat{\Phi_m^{(c)}}(\xi)}\,d\xi$
  is controlled by the phase difference
  $(n-m)/c$; rapid oscillation for
  $|n-m|\gg 1$ yields the $o(1/c)$ bound.
\item Combine (A)--(C) and~\eqref{eq:quasi-orth}
  to verify Conjecture~\ref{conj:szego}:
  the eigenvalue distribution of $\Ac^{(M(c))}$
  is asymptotically determined by its diagonal,
  giving non-vanishing mean spectral density
  $\bar\lambda^{(M(c))}(c)\sim c_0/c$
  and closing OP~\ref{prob:H1H3}.
\end{enumerate}

\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law; Paper~XV]
\label{prob:fourier-mass}
\emph{(Priority: critical.)}
Prove (A)--(C) of \S\ref{sec:PaperXV}:
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$
uniformly in $n\in\mathcal{T}_{\alpha,\kappa}(c)$.

\medskip\noindent
\emph{Additional difficulty:}
The $\Phi_n^{(c)}$ are orthogonal in
$L^2[-1,1]$ but not in
$L^2[-c_0/c,c_0/c]$ after rescaling,
introducing off-diagonal correlations.
Paper~XV must control both individual
$\mu_{n,c}$ asymptotics via WKB
and correlation decay in the transition zone
via quasi-orthogonality~\eqref{eq:quasi-orth}.
\end{problem}

\begin{problem}[Non-vanishing mean spectral density]
\label{prob:H1H3}
\emph{(Priority: high.)}

\medskip\noindent
\textbf{Setup.}
For each finite $M>N$, define the
\emph{mean spectral density} of $\Ac^{(M)}$:
\[
  \bar\lambda^{(M)}(c)
  := \frac{S_N^{(M)}(c)}{M-N}
  = \frac{\mathrm{Tr}(\Ac^{(M)})}{M-N}
  \in [0,1].
\]
By Lemma~\ref{lem:rank-lower}, the
finite-dimensional effective rank satisfies:
\[
  \#\{j:\lambda_j^{(M)}\ge\varepsilon\}
  \ge
  \frac{(\bar\lambda^{(M)}(c)-\varepsilon)(M-N)}{1-\varepsilon},
\]
which is non-trivial exactly when
$\bar\lambda^{(M)}(c)>\varepsilon$.

\medskip\noindent
\textbf{The open condition.}
Does there exist a scale
$M=M(c)\to\infty$ (as $c\to\infty$,
fixed $N$, $\varepsilon$) such that
\begin{equation}\label{eq:mean-density}
  \bar\lambda^{(M(c))}(c)
  = \frac{S_N^{(M(c))}(c)}{M(c)}
  \not\to 0\;?
\end{equation}
Equivalently:
\begin{equation}\label{eq:linear-growth}
  S_N^{(M(c))}(c) \gtrsim M(c)\;?
\end{equation}

\medskip\noindent
\textbf{Why this is a density problem.}
Condition~\eqref{eq:mean-density} is precisely
a \emph{non-vanishing mean spectral density}
along the truncation scale $M(c)$:
the spectral mass is not asymptotically
concentrated on a negligible fraction of
the $M(c)$ eigenvalues, but spread with
positive average density.
This is the finite-dimensional analogue of
spectral equidistribution at the operator
truncation level.

\medskip\noindent
\textbf{Szeg\H{o} reformulation.}
As identified in \S\ref{sec:szego},
this is equivalent to asking whether
$\Ac^{(M(c))}$ exhibits semiclassical
Szeg\H{o} behaviour: asymptotic diagonality
in the PSWF basis (quasi-orthogonality~%
\eqref{eq:quasi-orth}) combined with the
local Weyl law (OP~\ref{prob:fourier-mass})
would yield $\bar\lambda^{(M(c))}(c)
\sim c_0/c > 0$ for $M(c)=O(c)$.

\medskip\noindent
\textbf{Conjecture.}
Paley--Wiener / bandlimit geometry forces
$\bar\lambda^{(M(c))}(c)\ge c_*/c>0$
uniformly along the scale $M(c)\sim\beta c$
for some $\beta\in(0,1)$ and $c_*>0$.
If confirmed:
\begin{enumerate}[\rm(i)]
\item $\#\{j:\lambda_j^{(M(c))}\ge\varepsilon\}
  \ge c_*(1-\varepsilon)^{-1}M(c)/(2c)\to\infty$,
\item combined with
  $\norm{\Ac}_{\mathrm{op}}\le S_N(c)$,
  this closes the diagonal-WOT-to-op-norm gap.
\end{enumerate}
\end{problem}

\begin{problem}[NC from Papers~VIII, XII]
\label{prob:NC-direct}
\emph{(Priority: medium.)}
Do the pointwise bounds of Paper~VIII
imply lower bounds on
$\mu_{n,c}([-c_0,c_0])$
without full WKB?
\end{problem}

\medskip\noindent
\textbf{Central conclusion.}
\emph{The obstruction to coefficient stability
is spectral non-evacuation:
persistence of $S_N(c)>0$ uniformly in $c$
as $N\to\infty$.
All operator-theoretic structure rests on:
(a)~$\Bn=(I-\Pi_N^{(c)})\Pc$ being a
composition of orthogonal projections
($\norm{\Bn}_{\mathrm{op}}\le 1$,
$0\le\Ac\le I$);
(b)~$S_N(c)$ being the convergent
PSWF-ONB spectral sum~\eqref{eq:SN-def},
interpreted as a limit of finite-dimensional
traces and \emph{not} as a sum of eigenvalues
of the global operator $\Ac$;
(c)~the inequality
$\norm{\Ac}_{\mathrm{op}}\le S_N(c)$
established via finite-dimensional
truncation $\Ac^{(M)}$ and monotone
limit $M\to\infty$, without any global
Hilbert--Schmidt or trace-class assumption;
(d)~the finite-dimensional effective rank bound
(Lemma~\ref{lem:rank-lower}), which is a
statement about the matrix $\Ac^{(M)}$ only
and is never passed to the limit.
The gap between diagonal WOT-convergence
and operator norm convergence is now
precisely identified as a
\emph{non-vanishing mean spectral density}
problem at a growing truncation scale
(OP~\ref{prob:H1H3}),
and further as a \emph{semiclassical
Szeg\H{o}-type problem} for the truncated
Gram operator $\Ac^{(M)}$ (\S\ref{sec:szego}):
whether quasi-orthogonality~\eqref{eq:quasi-orth}
holds and the eigenvalue distribution of
$\Ac^{(M(c))}$ converges to a non-degenerate
density determined by the phase space overlap
of the Paley--Wiener band and the transition zone.
No HS property of $P_{c_0}$ on
$L^2(\mathbb{R})$ is used or claimed.
Paper~XIV is logically complete up to
two open problems.}
