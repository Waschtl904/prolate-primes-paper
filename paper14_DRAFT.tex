\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis under the SOT-Limit:\\
  Cross-Gram Operators, Tail Reduction, and the Spectral Distribution Problem}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
This paper analyses Hypothesis~H1 of Paper~XIII:
the $\ell^2$-norm convergence of Slepian coefficients
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ as $c\to\infty$.
We do not prove H1, but reduce it precisely to
a single open problem in semiclassical spectral theory.

The \emph{cross-Gram operator}
$T_c g := \sum_n \ip{g}{\Phic{c}{n}}\Phiinf{n}$
satisfies $\norm{T_c}\le 1$ uniformly.
The \emph{tail functional}
$S_N(c):=\sum_{n>N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$
characterises H1 completely:
$\text{H1}\Leftrightarrow\text{H3}$
(Hypothesis H3: $\sup_c S_N(c)\to 0$),
where H3$\Rightarrow$H1 is proved unconditionally
and H1$\Rightarrow$H3 is conditional on
Open Problem~\ref{prob:H1H3}.
Moreover, $S_N(c)=\norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}$,
so H3 is equivalent to SOT-convergence
$\Pi_N^{(c)}\Pc\to\Pc$ as $N\to\infty$,
uniformly in $c$.

We identify the central obstruction:
the Slepian--Weyl law gives
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim cc_0/\pi$,
but large total Fourier mass does not
by itself prevent H3:
the mass could concentrate into a
finite set of indices.
The \emph{No-Concentration condition} (NC)
rules this out: NC asserts that indices
in the transition zone $n\sim\alpha c$
carry at least $c_*>0$ Fourier mass,
uniformly in $c$.
Under NC, H3 fails (Spectral Leakage Barrier);
NC is precisely a local Weyl law for PSWFs
in the transition zone.

\medskip\noindent
\textbf{Central result} (informal):
\emph{H1 $\Leftrightarrow$ H3 $\Leftrightarrow$ absence of NC.
NC is a local Weyl law for the PSWF eigenbasis
in the transition zone $n\sim\alpha c$:
this is the natural entry point for Paper~XV.}
\end{abstract}

%%═══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context and scope}

Paper~XIII identified H1 as the key remaining
hypothesis for completeness $\Kspace=L^2(\mathbb{R})$.
This paper reduces H1 to a single analytic condition
(NC) that is accessible by semiclassical methods.
All unconditional results are clearly labelled.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$
($c\to\infty$).
\end{hypothesis}

H1 splits:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
\eqref{eq:H1ip} is proved in
Proposition~\ref{prop:H1ip};
\eqref{eq:H1norm} is equivalent to H3.

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $\alpha\in(0,1)$ and $\beta\in(0,1]$:
$\mathcal{T}_{\alpha,\beta}(c):=
\{n\in\mathbb{N}_0 : |n-\alpha c|\le c^\beta\}$.
The width $c^\beta$ may be taken as any function
$\omega(c)=o(c)$ with $\omega(c)\to\infty$;
the key regime is $n/c\to\alpha$.
\end{definition}

\begin{remark}[Role of the transition zone]
\label{rem:TZ-role}
For $n\ll c$: $\lambda_n^{(c)}\approx 1$
(PSWF concentrates in $[-c_0,c_0]$,
``good'' regime).
For $n\gg c$: $\lambda_n^{(c)}\approx 0$
(mass negligible in $[-c_0,c_0]$).
The transition zone is the regime where
eigenvalues transit between these extremes:
Fourier mass is neither negligible
nor controlled by fixed-$n$ tools,
and the Slepian--Weyl law shows
$\sim cc_0/\pi$ such indices contribute.
\end{remark}

\subsection{Summary of results}

\begin{center}
\fbox{\parbox{0.88\linewidth}{
\textbf{Proved unconditionally:}
\begin{itemize}[leftmargin=*]
\item $\norm{T_c}\le 1$;
H3$\Rightarrow$H1
(Theorems~\ref{prop:Tc-bounded},
\ref{thm:H3-H1})
\item Tail reduction
(Lemma~\ref{lem:tail})
\item $S_N(c)$ as operator norm;

(Theorem~\ref{thm:SN-operator})
\item $\sigma(c)\sim cc_0/\pi$
(Proposition~\ref{prop:trace})
\item Counterexample:
$\sigma(c)\to\infty\not\Rightarrow S_N(c)\not\to 0$

(Remark~\ref{rem:counterexample})

\end{itemize}}}
\end{center}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
\label{def:crossgram}
$T_c g:=\sum_{n}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{theorem}[$\norm{T_c}\le 1$]
\label{prop:Tc-bounded}
$\norm{T_c}_{\mathcal{B}(L^2)}\le 1$
uniformly in $c$.
\end{theorem}
\begin{proof}
Bessel: $\norm{T_c g}^2=
\sum_n|\ip{g}{\Phic{c}{n}}|^2\le\norm{g}^2$.
\end{proof}

\begin{remark}[Why $\norm{T_c}\le 1$ matters]
\label{rem:Tc-meaning}
The bound $\norm{T_c}\le 1$ gives uniform
(in $c$) weak compactness of the family
$\{T_c g : c>c_0\}$ for each fixed $g$.
It also ensures that all coefficient
estimates scale correctly with $\norm{g}$,
which is essential for the tail reduction.
\end{remark}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds unconditionally.
\end{proposition}
\begin{proof}
Bessel + Paper~XII: $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$,
so $\ip{a^{(c)}}{a^{(\infty)}}\to\norm{a^{(\infty)}}^2$.
\end{proof}

\begin{remark}[H1 reduces to \eqref{eq:H1norm}]
\label{rem:H1-reduces}
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$
(weak lsc), so H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}\le\norm{a^{(\infty)}}$.
\end{remark}

%%═══ 4. TAIL REDUCTION
\section{Tail Reduction and Operator Identity}
\label{sec:tail}

\begin{definition}[$S_N(c)$, $\sigma(c)$]
\label{def:SN}
\[
  S_N(c):=\sum_{n>N}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])},
  \quad
  \sigma(c):=\sum_{n\ge 0}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}.
\]
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$ and all $c,N$:
\[
  \sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2\cdot S_N(c).
\]
\end{lemma}
\begin{proof}
$a_n^{(c)}=\ip{\hat g}{
\widehat{\Phi_n^{(c)}}\cdot\mathbf{1}_{[-c_0,c_0]}}$
(Paley--Wiener). Cauchy--Schwarz, sum $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness]
\label{cor:equitight}
$\sup_c S_N(c)\to 0\Rightarrow
\{a^{(c)}\}$ equi-tight in $\ell^2$.
\end{corollary}

\begin{theorem}[$S_N(c)$ as operator norm;
H3 as SOT-convergence]
\label{thm:SN-operator}
Let $\Pi_N^{(c)}$ project onto
$\operatorname{span}\{\Phic{c}{0},\ldots,\Phic{c}{N}\}$.
Then:
\begin{equation}\label{eq:SN-op}
  S_N(c)
  = \norm{\Pc(I-\Pi_N^{(c)})}^2_{\mathrm{HS}}
  = \norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}
  = \sup_{\substack{g\in\PW_{c_0}^2\\
    \norm{g}=1}}
    \sum_{n>N}|a_n^{(c)}|^2.
\end{equation}
Hence:
\begin{equation}\label{eq:H3-SOT}
  \text{H3}\;\Longleftrightarrow\;
  \Pi_N^{(c)}\Pc \xrightarrow{\;
  \text{SOT},\,\text{unif.}\;c\;} \Pc
  \quad(N\to\infty),
\end{equation}
where ``SOT, unif.\ $c$'' means
$\norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}\to 0$
uniformly in $c$.
\end{theorem}
\begin{proof}
Since $\norm{\Phic{c}{n}}=1$, writing
$f_n:=\Pc\Phic{c}{n}$:
$S_N(c)=\sum_{n>N}\norm{f_n}^2
=\norm{\Pc(I-\Pi_N^{(c)})}^2_{\mathrm{HS}}$.
The operator norm equality follows from:
$\sup_{\norm{g}=1}\sum_{n>N}|a_n^{(c)}|^2
=\norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}$
(by definition of operator norm via the
Hilbert--Schmidt expansion).
H3 states $\sup_c
\norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}\to 0$,
which is exactly SOT-convergence
$\Pi_N^{(c)}\Pc\to\Pc$,
uniform in $c$.
\end{proof}

\begin{remark}[Why the SOT formulation is strong]
\label{rem:SOT-strong}
Formula~\eqref{eq:H3-SOT} reframes H3
as a convergence problem in operator theory:
H3 asks whether the finite-rank projections
$\Pi_N^{(c)}$, composed with $\Pc$,
converge to $\Pc$ in operator norm,
uniformly over the one-parameter family $c$.
This is strictly stronger than
pointwise-in-$g$ (= strong operator)
convergence, and it is precisely
the uniformity in $c$ that is hard.
This connects H3 directly to
questions of uniform approximation
of operators in the SOT topology,
an active area in operator algebras.
\end{remark}

\begin{proposition}[Trace identity and growth]
\label{prop:trace}
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim cc_0/\pi\to\infty$.
\end{proposition}
\begin{proof}
$P_{c_0}|_{\PW_c^2}$ is Hilbert--Schmidt
for $c>c_0$ (hence $P_c P_{c_0}P_c$
is trace-class on $\PW_c^2$
with trace $\sigma(c)$).
Slepian--Weyl \cite{Slepian1961}:
$\#\{n:\lambda_n^{(c)}\ge 1/2\}\sim 2cc_0/\pi$.
\end{proof}

\begin{remark}[Counterexample:
  $\sigma(c)\to\infty\not\Rightarrow
  S_N(c)\not\to 0$; and why PSWFs avoid this]
\label{rem:counterexample}
\emph{Abstract counterexample.}
Let $b_n^{(c)}\ge 0$
with $\sigma(c)=\sum_n b_n^{(c)}\to\infty$
but $b_n^{(c)}=0$ for all $n>N_0$
(some fixed $N_0$, all $c$).
Then $S_N(c)=0$ for $N\ge N_0$
despite $\sigma(c)\to\infty$.
The Bessel bound ($b_n^{(c)}\le 1$)
cannot exclude this:
it only shows $\sum_{n\le N}b_n^{(c)}\le N+1$,
but the mass could migrate into a different
finite window for each $c$.

\emph{Why this is incompatible with PSWF structure.}
For PSWFs, this abstract scenario would require:
(i) the Slepian eigenvalues
$\lambda_n^{(c)}
=\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c,c])}$
to concentrate entirely in $\{n\le N_0\}$
as $c\to\infty$;
but the Slepian--Weyl law states that
$\sim cc_0/\pi$ eigenvalues are near~$1$
and they are \emph{indexed by a growing set};
(ii) the PSWFs with $n\sim\alpha c$
would need their entire Fourier mass
to escape $[-c_0,c_0]$,
contradicting the non-vanishing
estimates of Paper~VIII.
The abstract counterexample is therefore
logically valid but PSWF-incompatible:
it serves to show that $\sigma(c)\to\infty$
is insufficient \emph{by itself},
and that NC is a genuine additional condition
that nevertheless holds for PSWFs
(pending the local Weyl law;
OP~\ref{prob:fourier-mass}).
\end{remark}

%%═══ 5. THE NO-CONCENTRATION CONDITION
\section{The No-Concentration Condition}
\label{sec:NC}

\subsection{Definition and equivalent forms}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
Fix $\alpha\in(0,1)$,
$\kappa\in(\alpha,1)$,
and $\omega(c)=o(c)$ with $\omega(c)\to\infty$
(e.g.\ $\omega(c)=c^{1/2}$).
We say \emph{NC holds at rate $\alpha$} if
there exists $c_*>0$ such that:
\begin{equation}\label{eq:NC}
  \sum_{|n-\alpha c|\le\omega(c)}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge c_* > 0
  \quad\text{for all large } c.
\end{equation}
That is: transition-zone indices
persistently carry at least $c_*>0$
total Fourier mass, uniformly in $c$.
\end{definition}

\begin{remark}[NC is uniform in $c$]
\label{rem:NC-uniform}
Crucially, the lower bound $c_*>0$
in~\eqref{eq:NC} is \emph{independent of $c$}.
This is what distinguishes NC from the
weaker statement
``each index $n\sim\alpha c$ has positive mass
for each fixed $c$'':
NC requires the mass to remain
bounded away from $0$ as $c$ varies.
This uniformity is exactly what is needed
to obtain $\liminf_c S_N(c)>0$
in Proposition~\ref{prop:barrier}.
\end{remark}

\begin{remark}[NC is a local Weyl law]
\label{rem:NC-Weyl}
The quantitative local Weyl law:
\begin{equation}\label{eq:local-Weyl}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim \frac{c_0}{c}
  \quad(n=\lfloor\alpha c\rfloor,
  \; c\to\infty)
\end{equation}
would imply NC immediately:
$\sum_{|n-\alpha c|\le\omega(c)}
\|\cdots\|^2
\sim c_0\cdot 2\omega(c)/c
\to 0$ if $\omega(c)/c\to 0$,
but for $\omega(c)=c^{1/2}$:
$\sim 2c_0 c^{-1/2}\to 0$.

\emph{Caution}: the window sum
$\sum_{|n-\alpha c|\le c^{1/2}}$
has $\sim 2c^{1/2}$ terms, each $\sim c_0/c$,
giving total $\sim 2c_0 c^{-1/2}\to 0$.
This is \emph{insufficient} for NC!
\textbf{The correct window} for NC
requires $\omega(c)/c$ bounded away from $0$,
i.e.\ $\omega(c)\sim\delta c$ for some
$\delta>0$: a window of proportional width.
\end{remark}

\begin{definition}[NC (corrected, proportional window)]
\label{def:NC2}
Fix $0<\alpha<\kappa<1$.
NC holds if there exists $c_*>0$ such that:
\begin{equation}\label{eq:NC2}
  \sum_{\alpha c < n\le\kappa c}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge c_* > 0
  \quad\text{for all large }c.
\end{equation}
Under the local Weyl law~\eqref{eq:local-Weyl},
this sum is $\sim c_0(\kappa-\alpha)>0$:
NC holds with $c_*=c_0(\kappa-\alpha)/2$.
\end{definition}

\begin{remark}[Local Weyl law $\Rightarrow$ NC
  with proportional window]
\label{rem:Weyl-NC}
Under~\eqref{eq:local-Weyl}:
\[
  \sum_{\alpha c<n\le\kappa c}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim \frac{c_0}{c}\cdot
  (\kappa-\alpha)c
  = c_0(\kappa-\alpha) > 0.
\]
This is a Riemann sum for
$c_0\int_\alpha^\kappa d\alpha=c_0(\kappa-\alpha)$.
So the local Weyl law (OP~\ref{prob:fourier-mass})
immediately implies NC
(Definition~\ref{def:NC2})
with $c_*=c_0(\kappa-\alpha)$.
\end{remark}

\subsection{The Spectral Leakage Barrier}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under NC (Definition~\ref{def:NC2}):
$\liminf_{c\to\infty}\, S_N(c)\ge c_*>0$
for each fixed $N$,
and thus $\sup_c S_N(c)\not\to 0$,
i.e.\ H3 fails.
\end{proposition}
\begin{proof}
For $c$ large, $\alpha c > N$,
so $\{n:\alpha c<n\le\kappa c\}\subseteq\{n:n>N\}$.
Hence $S_N(c)\ge
\sum_{\alpha c<n\le\kappa c}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\ge c_*$.
\end{proof}

\begin{remark}[Two roles of $\sigma(c)$]
\label{rem:two-roles}
Proposition~\ref{prop:trace} ($\sigma(c)\to\infty$)
shows total mass grows;
it is necessary for the obstruction
but not sufficient.
NC (Definition~\ref{def:NC2})
shows mass in a \emph{proportional band}
$n\in(\alpha c,\kappa c)$ stays
bounded below by $c_*$;
this is sufficient.
The distinction:
Slepian--Weyl = growth;
NC = localisation of growth.
\end{remark}

%%═══ 6. H3 AND THE MAIN REDUCTION
\section{H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)\xrightarrow{N\to\infty}0$.
(Equivalently: $\Pi_N^{(c)}\Pc\to\Pc$ SOT
uniformly in $c$.)
\end{hypothesis}

\begin{theorem}[H3 $\Rightarrow$ H1]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$, $g\in\PW_{c_0}^2$.
H3: choose $N$ so that
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$;
then $\sum_{n>N}|a_n^{(c)}|^2<\varepsilon/2$
for all $c$ (Lemma~\ref{lem:tail}).
Paper~XII: choose $c_\varepsilon$ so
$\sum_{n=0}^N|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ for $c>c_\varepsilon$.
Combine.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$ H3 for fixed $g$]
\label{thm:H1-H3-ptwise}
Assume H1. For each fixed $g\in\PW_{c_0}^2$:
$\sum_{n>N}|a_n^{(c)}|^2\to 0$
as $N\to\infty$, uniformly in $c$
(for that fixed $g$).
\end{theorem}
\begin{proof}
$\sum_{n>N}|a_n^{(c)}|^2
\le 2\norm{a^{(c)}-a^{(\infty)}}^2
+2\sum_{n>N}|a_n^{(\infty)}|^2\to 0$.
\end{proof}

\begin{remark}[The uniformity gap]
\label{rem:uniformity-gap}
Theorem~\ref{thm:H1-H3-ptwise} gives:
for each fixed $g$,
$\sum_{n>N}|a_n^{(c)}(g)|^2\to 0$
(uniformly in $c$).
To obtain H3 one needs:
\[
  \sup_{\norm{g}=1}
  \sum_{n>N}|a_n^{(c)}|^2
  = S_N(c) \to 0
  \quad\text{uniformly in }c,
\]
i.e.\ the supremum over $g$ and the
limit $N\to\infty$ must interchange.
This is a \emph{uniform boundedness}
problem: the family of functionals
$g\mapsto\sum_{n>N}|a_n^{(c)}(g)|^2$
(indexed by $c$) is bounded pointwise;
Banach--Steinhaus would give uniform
boundedness if the family were linear,
but these are quadratic forms.
The precise statement reduces to
$\norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}\to 0$
uniformly in $c$,
which is Open Problem~\ref{prob:H1H3}.
\end{remark}

\begin{theorem}[Main reduction]
\label{thm:reduction}
The following implications hold:
\begin{align}
  \text{NC}
  &\overset{\text{Prop.}\,\ref{prop:barrier}}
    {\Longrightarrow}
    \text{H3 fails}
    \label{eq:NC-H3}\\
  \text{H3}
  &\overset{\text{Thm.}\,\ref{thm:H3-H1}}
    {\Longrightarrow}
    \text{H1}
    \label{eq:H3-H1}\\
  \text{H1}
  &\overset{\text{Thm.}\,\ref{thm:H1-H3-ptwise}}
    {\Longrightarrow}
    S_N(c)\to 0
    \text{ (fixed }g\text{)}
    \label{eq:H1-ptwise}\\
  \text{H1}
  &\overset{\text{OP}\,\ref{prob:H1H3}}
    {\Longrightarrow}
    \text{H3}
    \label{eq:H1-H3}
\end{align}
Combining:
\[
  \text{NC}
  \;\Rightarrow\;
  \text{H3 fails}
  \;\overset{(\ref{eq:H1-H3})}\Rightarrow\;
  \text{H1 fails}
  \qquad\text{(given OP~\ref{prob:H1H3})}
\]
\[
  \text{NC}^c\;(\text{NC fails})
  \;\Longrightarrow\;
  \text{H3 may hold}
  \;\Rightarrow\;
  \text{H1.}
\]
Thus: \emph{H1 is equivalent to absence of NC,
modulo OP~\ref{prob:H1H3} and
OP~\ref{prob:fourier-mass}.}
\end{theorem}

%%═══ 7. INTERPRETATION
\section{Interpretation}
\label{sec:interp}

\subsection{Structure of the reduction}

The central achievement is the chain:
\[
  \underbrace{\text{H1}}_{
    \ell^2\text{-convergence}}
  \;\xleftrightarrow{\text{XIV}\,
    +\,\text{OP}\,\ref{prob:H1H3}}\;
  \underbrace{\text{H3}}_{
    \text{uniform tail control}}
  \;\xleftrightarrow{\text{NC}}\;
  \underbrace{\text{local Weyl law}}_{
    n\sim\alpha c,\;
    \text{Paper~XV}}
\]
Each arrow is either proved
or reduced to a single,
precisely stated open problem.

\subsection{From functional analysis to semiclassics}

H1 is a functional-analytic statement
($\ell^2$-convergence of coefficients).
H3 is an operator-analytic statement
(SOT-convergence of finite-rank projections).
NC is a semiclassical statement
(equidistribution of PSWF mass
in the transition zone).
The reduction H1$\to$H3$\to$NC
moves from pure functional analysis
to operator theory to semiclassics:
this is the natural hierarchy
for problems involving eigenfunction
asymptotics of pseudodifferential operators
\cite{Zworski2012}.

\subsection{Relation to quantum ergodicity}

NC (Definition~\ref{def:NC2}) is formally
similar to \emph{quantum ergodicity}:
Fourier mass of eigenfunctions
($\widehat{\Phi_n^{(c)}}$)
equidistributes over frequency bands
as $n/c\to\alpha$.
Unlike quantum ergodicity on Riemannian
manifolds, here the relevant phase space
is one-dimensional (the interval $[-c,c]$)
and the equidistribution is in frequency,
not position.
The connection to the theory of
semiclassical measures
\cite{Zworski2012} provides
a natural toolkit for Paper~XV.

\subsection{Why existing tools fail}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Tool & Controls & Does not control \\
\midrule
Bessel
  & Prefix $\le N+1$
  & Tail; NC \\
Paper~XII
  & $a_n^{(c)}$ (fixed $n$)
  & $n\sim\alpha c$ \\
Paper~VIII
  & Pointwise $|\widehat{\Phi_n^{(c)}}|$
  & Integrated mass over bands \\
Slepian--Weyl
  & $\sigma(c)\sim cc_0/\pi$
  & Index distribution of mass \\
\textbf{NC/Weyl law}
  & Transition-zone mass
  & \textbf{(= Paper~XV)} \\
\bottomrule
\end{tabular}
\end{center}

%%═══ 8. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law for PSWFs;
  Paper~XV entry point]
\label{prob:fourier-mass}
\emph{(Priority: critical.)}
For $n=\lfloor\alpha c\rfloor$, $\alpha\in(0,1)$:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim \frac{c_0}{c}
  \quad(c\to\infty).
\]
This implies NC (Remark~\ref{rem:Weyl-NC})
and hence H3 fails and (given
OP~\ref{prob:H1H3}) H1 fails.
Approach: WKB / semiclassical analysis
for PSWFs at $n\sim\alpha c$
\cite{Osipov2013,Zworski2012}.
\end{problem}

\begin{problem}[H1$\Rightarrow$H3: uniformity]
\label{prob:H1H3}
\emph{(Priority: high.}
\emph{Determines whether H1$\Leftrightarrow$H3.})
Show:
$\norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}
\to 0$ uniformly in $c$,
following from H1.
Note: the obstacle is that
$g\mapsto\sum_{n>N}|a_n^{(c)}(g)|^2$
is a quadratic form, not linear;
the Banach--Steinhaus argument
must be adapted or replaced.
\end{problem}

\begin{problem}[NC without the full local Weyl law]
\label{prob:NC-direct}
\emph{(Priority: medium.)}
Is NC provable directly from
Papers~VIII and~XII,
without the full local Weyl law~\eqref{eq:local-Weyl}?
If so, the Spectral Leakage Barrier
becomes unconditional.
\end{problem}

\textbf{Central conclusion.}
\emph{H1 reduces precisely to NC:
the no-concentration of PSWF Fourier mass
in the transition zone $n\sim\alpha c$,
where the mass from any proportional band
$(\alpha c,\kappa c)$ is uniformly bounded
below by $c_0(\kappa-\alpha)$
(under the local Weyl law).

the single missing ingredient is
the local Weyl law for PSWFs
in the transition zone,
which is precisely the entry point
for Paper~XV.}
