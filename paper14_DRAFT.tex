  Coefficient Stability of the Slepian Eigenbasis:\\

\author{[Author]}

\begin{abstract}
This paper reduces Hypothesis~H1 of Paper~XIII
(the $\ell^2$-norm convergence of Slepian
coefficients $a_n^{(c)}=\ip{g}{\Phic{c}{n}}$
as $c\to\infty$) to a single spectral
distribution problem in the transition
zone $n\sim\alpha c$.

The key structural results are:
(i)~the cross-Gram operator $T_c$ satisfies
$\norm{T_c}\le 1$ uniformly;
(ii)~the tail functional
$S_N(c):=\sum_{n>N}\mu_{n,c}([-c_0,c_0])$,
where $\mu_{n,c}(d\xi)
:=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi$,
controls the tail via
$\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c)$;
(iii)~$S_N(c)$ is the
\emph{Hilbert--Schmidt norm} of the
truncation operator
$\Bn:=(I-\Pi_N^{(c)})\Pc$
(not the operator norm),
and the relevant quadratic form is
$Q_c^N(g):=\norm{\Bn g}^2
=\ip{\Ac g}{g}$
where $\Ac:=\Bn^*\Bn$.

H3 ($\sup_c S_N(c)\to 0$) implies H1
(proved unconditionally).
The obstruction to H1 is
\emph{spectral non-evacuation}:
``H1 fails if and only if
there exists persistent spectral mass
of the measures $\mu_{n,c}$
in the transition zone $n\sim\alpha c$''
(conditional on two open problems).

The \emph{No-Concentration condition} (NC)
quantifies this: NC$\Rightarrow$H3 fails
(proved); NC holds under the local Weyl
law (Paper~XV).
NC$^c\Rightarrow$H3 is a logically independent
gap, likely false without additional structure.

H1$\Rightarrow$H3 requires uniform convergence
of the quadratic forms $Q_c^N$,
equivalent to strong-to-operator convergence
of $\Ac\to 0$: this is
Open Problem~\ref{prob:H1H3}.
\end{abstract}

%%═══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context}

Paper~XIII identified H1 as the key
remaining hypothesis for completeness
$\Kspace=L^2(\mathbb{R})$.
Paper~XIV reduces H1 to a spectral
distribution problem for the
PSWF eigenbasis in the transition zone.
All conditional steps are explicitly labelled.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$
($c\to\infty$).
\end{hypothesis}

H1 splits:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
\eqref{eq:H1ip} is proved (Prop.~\ref{prop:H1ip});
\eqref{eq:H1norm} is equivalent to H3
(given OP~\ref{prob:H1H3}).
The core difficulty: $a^{(c)}
\rightharpoonup a^{(\infty)}$ (weak convergence)
does not in general imply norm convergence;
Paper~XIV localises this gap precisely.

\subsection{The central new framing}

\begin{definition}[Spectral measures]
\label{def:spectral-measures}
For $n\ge 0$, $c>c_0$:
\[
  \mu_{n,c}(d\xi)
  :=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi.
\]
Plancherel: $\mu_{n,c}(\mathbb{R})=1$.
The tail functional and its quadratic form:
\[
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0]),
  \qquad
  Q_c^N(g) := \sum_{n>N}|a_n^{(c)}|^2
  = \norm{(I-\Pi_N^{(c)})\Pc g}^2.
\]
The obstruction to H1 is
\emph{spectral non-evacuation}:
the failure of $\mu_{n,c}([-c_0,c_0])$
to vanish for $n>N$, uniformly in $c$.
\end{definition}

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $0<\alpha<\kappa<1$:
$\mathcal{T}_{\alpha,\kappa}(c)
:=\{n: \alpha c < n \le \kappa c\}$.
This band has $\sim(\kappa-\alpha)c$ elements
and captures the regime where Slepian
eigenvalues $\lambda_n^{(c)}$
transit between $\approx 1$ and $\approx 0$.
\end{definition}

\subsection{Logical status}

\begin{center}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{theorem}[$\norm{T_c}\le 1$]
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{theorem}
\begin{proof}
Bessel:
$\norm{T_c g}^2=\sum_n|\ip{g}{\Phic{c}{n}}|^2
\le\norm{g}^2$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds unconditionally.
\end{proposition}
\begin{proof}
Paper~XII: $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$.
\end{proof}

\begin{remark}
By weak lsc:
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$,
so H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}\le\norm{a^{(\infty)}}$.
\end{remark}

%%═══ 4. TAIL REDUCTION, HS IDENTITY, AND QUADRATIC FORM
\section{Tail Reduction, HS Identity, and the Quadratic Form}
\label{sec:tail}

\begin{definition}[$S_N(c)$, $\sigma(c)$]
\label{def:SN}
$S_N(c):=\sum_{n>N}\mu_{n,c}([-c_0,c_0])$,
$\quad\sigma(c):=\sum_{n\ge 0}\mu_{n,c}([-c_0,c_0])$.
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$Q_c^N(g):=\sum_{n>N}|a_n^{(c)}|^2
\le\norm{g}^2\cdot S_N(c)$.
\end{lemma}
\begin{proof}
Paley--Wiener:
$a_n^{(c)}
=\ip{\hat g}{
  \widehat{\Phi_n^{(c)}}\cdot
  \mathbf{1}_{[-c_0,c_0]}}_{L^2}$.
Cauchy--Schwarz:
$|a_n^{(c)}|^2
\le\norm{g}^2\cdot
\mu_{n,c}([-c_0,c_0])$.
Sum over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness]
$\sup_c S_N(c)\to 0
\Rightarrow\{a^{(c)}\}_c$ equi-tight in $\ell^2$.
\end{corollary}

\begin{theorem}[HS identity, quadratic form,
  and $S_N$ as trace control]
\label{thm:SN-operator}
Define:
\begin{align}\label{eq:Bn-def}
  \Bn &:= (I-\Pi_N^{(c)})\Pc
    \;:\; \PW_{c_0}^2\to\PW_c^2,\\
  \Ac &:= \Bn^*\Bn
    \;:\; \PW_{c_0}^2\to\PW_{c_0}^2.
\end{align}

\medskip
\textbf{(I) HS identity.}
\begin{equation}\label{eq:SN-HS}
  S_N(c)
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}
  = \norm{\Bn}^2_{\mathrm{HS}}.
\end{equation}
\begin{proof}
Expand $\norm{\Bn}^2_{\mathrm{HS}}$
in the ONB $\{\Phic{c}{n}\}_{n>N}$:
$\sum_{n>N}\norm{\Bn\Phic{c}{n}}^2
=\sum_{n>N}\norm{\Pc\Phic{c}{n}}^2
=S_N(c)$.
\end{proof}

\medskip
\textbf{(II) Operator norm inequality.}
\begin{equation}\label{eq:op-le-HS}
  \norm{\Bn}^2_{\mathrm{op}}
  \le\norm{\Bn}^2_{\mathrm{HS}} = S_N(c).
\end{equation}
In general, $\norm{\Bn}_{\mathrm{op}}
<\norm{\Bn}_{\mathrm{HS}}$ unless
$\Bn$ has rank one.
\emph{The identity
$\norm{\Bn}^2_{\mathrm{op}}=S_N(c)$
claimed in earlier versions is incorrect.}

\medskip
\textbf{(III) Quadratic form and key step.}
For $g\in\PW_{c_0}^2$,
using $\Pc g=g$:
\begin{equation}\label{eq:Qc-key}
  Q_c^N(g)
  = \sum_{n>N}|\ip{g}{\Phic{c}{n}}|^2
  = \norm{(I-\Pi_N^{(c)})g}^2_{\PW_c^2}
  = \norm{\Bn g}^2
  = \ip{\Ac g}{g}.
\end{equation}
So the tail sum is a quadratic form
$Q_c^N(g)=\ip{\Ac g}{g}$
with $0\le\Ac\le I$
(positive semidefinite, bounded by $1$).

\medskip
\textbf{(IV) H3 = uniform operator convergence of $\Ac$.}
H3 states:
\begin{equation}\label{eq:H3-op}
  \sup_c S_N(c) = \sup_c\norm{\Bn}^2_{\mathrm{HS}}
  \to 0 \quad (N\to\infty).
\end{equation}
By~\eqref{eq:op-le-HS}:
$H3\Rightarrow
\norm{\Bn}_{\mathrm{op}}\to 0$
uniformly in $c$,
which is equivalent to
$\norm{\Ac}_{\mathrm{op}}\to 0$
uniformly in $c$.
Conversely,
$\norm{\Ac}_{\mathrm{op}}\to 0$ uniformly in $c$
implies $Q_c^N(g)\le\norm{\Ac}_{\mathrm{op}}\norm{g}^2
\to 0$ uniformly in $(g,c)$,
so $\sup_{\norm{g}=1}Q_c^N(g)\to 0$ uniformly in $c$;
but $\sup_{\norm{g}=1}Q_c^N(g)
=\norm{\Ac}_{\mathrm{op}}
\le S_N(c)$,
so this is weaker than H3.
Explicitly:
\begin{equation}\label{eq:chain}
  \text{H3}
  \;\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0
  \;(\text{unif.}\,c)
  \;\Rightarrow\;
  Q_c^N(g)\to 0
  \;(\text{unif.}\,c,g),
\end{equation}
but the second arrow does not reverse
in general.
H3 is the strongest condition
(via HS norm / trace control),
while $\norm{\Ac}_{\mathrm{op}}\to 0$
is the operationally relevant one.
\end{theorem}

\begin{remark}[HS norm as trace control;
operational role of $\norm{\Ac}_{\mathrm{op}}$]
\label{rem:HS-trace}
$S_N(c)=\norm{\Bn}^2_{\mathrm{HS}}$
=\mathrm{Tr}(\Ac)$
is a \emph{trace-type} quantity:
it sums contributions from all
$n>N$ uniformly,
giving ``global spectral energy''.
The operator norm $\norm{\Ac}_{\mathrm{op}}$
is the ``worst-case direction'';
it satisfies
$\norm{\Ac}_{\mathrm{op}}
=\sup_{n>N}\lambda_n(\Ac)$
(largest eigenvalue of $\Ac$),
which is smaller than $\mathrm{Tr}(\Ac)=S_N(c)$
(sum of all eigenvalues).

The relevant problem (H1$\Leftrightarrow$H3
via OP~\ref{prob:H1H3})
is whether \emph{pointwise}
$Q_c^N(g)\to 0$ (strong convergence
$\Ac\to 0$ in the SOT sense)
implies \emph{uniform}
$\norm{\Ac}_{\mathrm{op}}\to 0$.
This is the classical
strong $\not\Rightarrow$ operator convergence gap.
\end{remark}

\begin{proposition}[Trace identity and growth]
\label{prop:trace}
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim cc_0/\pi\to\infty$.
\end{proposition}
\begin{proof}
$\sigma(c)=\mathrm{Tr}(\Pc P_c\Pc)$;
Slepian--Weyl \cite{Slepian1961}:
$\sigma(c)\sim cc_0/\pi$.
\end{proof}

\begin{remark}[Counterexample and PSWF-incompatibility]
\label{rem:counterexample}
$\sigma(c)\to\infty$ does not imply
$S_N(c)\not\to 0$:
if $\mu_{n,c}([-c_0,c_0])=0$ for all $n>N_0$,
then $S_N(c)=0$ for $N\ge N_0$.
PSWF-incompatibility:
(i) Slepian--Weyl requires an
\emph{unbounded} set of contributing indices;
(ii) Paper~VIII gives
$\mu_{n,c}([-c_0,c_0])>0$ for each
$n\le\kappa c$.
NC (Definition~\ref{def:NC})
formalises the required lower bound.
\end{remark}

%%═══ 5. SPECTRAL MEASURES AND NC
\section{Spectral Measures and No-Concentration}
\label{sec:NC}

\begin{remark}[NC as spectral evacuation failure]
\label{rem:NC-evac}
The obstruction to H1 can now be stated cleanly:
\begin{quote}
  \emph{H1 fails if and only if
  there exists persistent spectral mass
  of the measures $\mu_{n,c}$
  in the transition zone $n\sim\alpha c$
  (conditional on OPs~\ref{prob:H1H3}
  and~\ref{prob:fourier-mass}).}
\end{quote}
``Spectral evacuation'' means
$\mu_{n,c}([-c_0,c_0])\to 0$
for $n>N$, uniformly in $c$,
as $N\to\infty$.
NC is the quantitative obstruction to this.
\end{remark}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
Fix $0<\alpha<\kappa<1$.
NC holds if there exists $c_*>0$ such that
for all large $c$:
\begin{equation}\label{eq:NC}
  \sum_{n\in\mathcal{T}_{\alpha,\kappa}(c)}
  \mu_{n,c}([-c_0,c_0]) \ge c_*>0.
\end{equation}
$c_*$ is independent of $c$.
\end{definition}

\begin{remark}[Why proportional width is necessary]
\label{rem:prop-width}
A window of width $c^\beta$ ($\beta<1$)
has $\sim 2c^\beta$ indices;
under the local Weyl law,
the sum is $\sim 2c_0 c^{\beta-1}\to 0$:
insufficient for NC.
A proportional band
$\mathcal{T}_{\alpha,\kappa}(c)$
has $\sim(\kappa-\alpha)c$ indices,
giving sum $\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[NC as quantum ergodicity]
\label{rem:NC-QE}
In the semiclassical limit $c\to\infty$,
$n\sim\alpha c$,
the rescaled measures
$c\cdot\mu_{n,c}|_{[-c,c]}$
are expected to converge weakly to
the WKB density
$(\pi\sqrt{1-(\xi/c)^2})^{-1}d\xi$.
This is a one-dimensional quantum ergodicity
statement in frequency space.
NC is the consequence for $[-c_0,c_0]$:
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$,
so the band sum is
$\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[Local Weyl law $\Rightarrow$ NC]
\label{rem:Weyl-NC}
If uniformly for $n\in\mathcal{T}_{\alpha,\kappa}(c)$:
\begin{equation}\label{eq:local-Weyl}
  \mu_{n,c}([-c_0,c_0])\sim\frac{c_0}{c},
\end{equation}
then the Riemann sum
$\sum_{n\in\mathcal{T}_{\alpha,\kappa}(c)}
\mu_{n,c}([-c_0,c_0])
\sim c_0(\kappa-\alpha)>0$.
NC holds with $c_*=c_0(\kappa-\alpha)/2$.
\end{remark}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under NC: for each fixed $N$,
$\liminf_{c\to\infty}S_N(c)\ge c_*>0$,
so H3 fails.
\end{proposition}
\begin{proof}
For large $c$: $\alpha c>N$,
so $\mathcal{T}_{\alpha,\kappa}(c)\subseteq\{n>N\}$.
Hence $S_N(c)\ge c_*$.
\end{proof}

\begin{remark}[NC$^c\Rightarrow$H3: likely false]
\label{rem:NC-converse}
NC$^c$ means: no proportional band
carries persistent mass $\ge c_*$.
But this permits:
\begin{itemize}[leftmargin=*]
\item Mass spreading over many thin windows,
  each with $\mu_{n,c}([-c_0,c_0])\to 0$
  but with $S_N(c)$ bounded below;
\item Mass migrating through a sequence of
  thin bands (not any fixed proportional band),
  yet preventing H3.
\end{itemize}
Formally, NC$^c$ controls
$\liminf_c\sum_{n\in\mathcal{T}}\mu_{n,c}$
for each fixed band $\mathcal{T}$,
but not $\liminf_c S_N(c)
=\liminf_c\sum_{n>N}\mu_{n,c}([-c_0,c_0])$.
This is Open Problem~\ref{prob:NC-H3}.
\end{remark}

%%═══ 6. H3 AND THE MAIN REDUCTION
\section{H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)
=\sup_{c>c_0}\norm{\Bn}^2_{\mathrm{HS}}
\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{theorem}[H3 $\Rightarrow$ H1]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$, $g\in\PW_{c_0}^2$.
H3: choose $N$ with
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$;
then $Q_c^N(g)<\varepsilon/2$ (all $c$).
Paper~XII: choose $c_\varepsilon$ with
$\sum_{n\le N}|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ ($c>c_\varepsilon$).
\end{proof}

\begin{theorem}[H1 $\Rightarrow$
  $Q_c^N(g)\to 0$, fixed $g$]
\label{thm:H1-H3-ptwise}
Assume H1. For each fixed $g$:
$\sup_c Q_c^N(g)\to 0$ as $N\to\infty$.
Equivalently: $\Ac\to 0$ in the SOT sense
(strong operator topology), for fixed $g$.
\end{theorem}
\begin{proof}
$Q_c^N(g)
\le 2\norm{a^{(c)}-a^{(\infty)}}^2
+2\sum_{n>N}|a_n^{(\infty)}|^2\to 0$.
\end{proof}

\begin{remark}[The uniformity gap]
\label{rem:unif-gap}
The gap between
Theorems~\ref{thm:H3-H1}
and~\ref{thm:H1-H3-ptwise}:
\[
  \text{SOT: }\Ac\to 0\;(\forall g)
  \quad\not\Rightarrow\quad
  \text{operator: }\norm{\Ac}_{\mathrm{op}}\to 0.
\]
This is the classical
strong operator convergence gap.
H3 ($\sup_c\mathrm{Tr}(\Ac)\to 0$)
is stronger than both;
OP~\ref{prob:H1H3} asks whether H1
(which gives SOT convergence)
allows the upgrade to
$\norm{\Ac}_{\mathrm{op}}\to 0$,
uniformly in $c$.
\end{remark}

\begin{theorem}[Main reduction
  (all implications honest)]
\label{thm:reduction}
\textup{(i)} Proved:
\[
  \text{H3}\Rightarrow\text{H1},
  \quad
  \text{H1}\Rightarrow\Ac\to 0\;(\mathrm{SOT}),
  \quad
  \text{NC}\Rightarrow\text{H3 fails},
  \quad
  \text{local Weyl}\Rightarrow\text{NC}.
\]
\textup{(ii)} Open:
\[
  \text{H1}\Rightarrow\norm{\Ac}_{\mathrm{op}}\to 0
  \;(\text{OP~\ref{prob:H1H3}}),
  \quad
  \text{NC holds}
  \;(\text{OP~\ref{prob:fourier-mass}}),
  \quad
  \text{NC}^c\Rightarrow\text{H3}
  \;(\text{OP~\ref{prob:NC-H3}}).
\]
\textup{(iii)} Conditional chain:
\[
  \underbrace{\text{local Weyl law}}_{
    \text{OP~\ref{prob:fourier-mass}}}
  \Rightarrow
  \text{NC}
  \Rightarrow
  \text{H3 fails}
  \underset{\text{OP~\ref{prob:H1H3}}}
  {\Rightarrow}
  \text{H1 fails.}
\]
\textup{(iv)} The full equivalence
``H1 $\Leftrightarrow$ H3
$\Leftrightarrow$ NC$^c$''
requires all three open problems.
\end{theorem}

\section{Interpretation and Context}
\label{sec:interp}

\subsection{The true obstruction: spectral non-evacuation}

The central insight:
\[
  \text{H1 fails}
  \;\Longleftrightarrow\;
  \text{spectral non-evacuation in }n\sim\alpha c
  \quad(\text{cond. on OPs~\ref{prob:H1H3},
  \ref{prob:fourier-mass}}).
\]
``Spectral evacuation'' would mean:
for every $N$, the total $\mu_{n,c}$-mass
of $\{n>N\}$ on $[-c_0,c_0]$
vanishes uniformly in $c$.
This is H3.
The measures $\mu_{n,c}$ in the transition zone
prevent this evacuation.

\subsection{$S_N(c)$ as trace control}

$S_N(c)=\mathrm{Tr}(\Ac)$ sums
all eigenvalues of $\Ac=\Bn^*\Bn$;
it is a global spectral energy.
$\norm{\Ac}_{\mathrm{op}}$ is the
largest eigenvalue only.
The distinction:
\[
  \norm{\Ac}_{\mathrm{op}}
  \le\mathrm{Tr}(\Ac)
  =S_N(c)
  \le\mathrm{rank}(\Ac)\cdot\norm{\Ac}_{\mathrm{op}}.
\]
H3 controls the trace;
the operator norm of $\Ac$ is what
H1 would need to control
(via OP~\ref{prob:H1H3}).

\subsection{Quantum ergodicity in frequency space}

The spectral measures $\mu_{n,c}$
play the role of semiclassical measures.
NC asserts their non-vacuation of $[-c_0,c_0]$;
the local Weyl law
(OP~\ref{prob:fourier-mass})
asserts their equidistribution in $[-c,c]$:
$c\cdot\mu_{n,c}|_{[-c,c]}\rightharpoonup
(\pi\sqrt{1-\xi^2/c^2})^{-1}d\xi$,
a one-dimensional quantum ergodicity
statement \cite{Zworski2012}.

%%═══ 8. PROGRAMME FOR PAPER XV
\section{Programme for Paper~XV}
\label{sec:PaperXV}

\subsection{The three required estimates}

Paper~XV must establish:
\begin{enumerate}[\rm(A)]
\item\label{XV-A} \textbf{Upper bound, uniform:}
  $\mu_{n,c}([-c_0,c_0])\le C/c$
  for all $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-B} \textbf{Lower bound, uniform:}
  $\mu_{n,c}([-c_0,c_0])\ge 1/(Cc)$
  for all $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-C} \textbf{Summability:}
  $\mu_{n,c}([-c_0,c_0])=c_0/c+o(1/c)$
  uniformly, so the Riemann sum gives
  $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
  \to c_0(\kappa-\alpha)$.
\end{enumerate}
(A)+(B) $\Rightarrow$ NC;
(C) $\Rightarrow$ equidistribution.

\subsection{Semiclassical approach}

Set $h:=1/c\to 0$.
The PSWF $\Phi_n^{(c)}$ solves
\begin{equation}\label{eq:PSWF-ODE}
  -(1-x^2)\partial_x^2\psi
  +2x\partial_x\psi
  +c^2x^2\psi=\chi_n^{(c)}\psi
\end{equation}
with semiclassical parameter $h=1/c$
and energy level
$E_n^{(h)}:=\chi_n^{(c)}/c^2\to E_\alpha$
for $n\sim\alpha c$.

\subsection{WKB and stationary phase}

Classical turning points of~\eqref{eq:PSWF-ODE}
at $x_0$ with $x_0^2=E_\alpha$.
WKB in the classical region $|x|<x_0$:
$\psi(x)\sim A(x)\cos(S(x)/h+\varphi)$.
Stationary phase for
$\widehat{\Phi_n^{(c)}}(\xi)
=\int_{-1}^1\Phi_n^{(c)}(x)e^{-i\xi x}dx$
with $\xi\in[-c_0,c_0]$ gives:
\[
  \mu_{n,c}([-c_0,c_0])
  \sim \frac{2c_0 C(E_\alpha)}{c},
\]
yielding estimates~(\ref{XV-A})--(\ref{XV-C}).

\subsection{Rigorous programme}

\begin{enumerate}
\item WKB with remainder $O(h^2)=O(c^{-2})$,
  uniform in $n\in\mathcal{T}_{\alpha,\kappa}(c)$;
  cf.\ \cite{Osipov2013}.
\item Stationary phase with uniform remainder.
\item Prove (A)--(C);
  conclude NC and Spectral Leakage.
\item Address OP~\ref{prob:H1H3}
  (whether SOT$\Rightarrow$op.~norm
  for $\Ac$ via compactness of $\Pc$).
\end{enumerate}

%%═══ 9. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law; Paper~XV]
\label{prob:fourier-mass}
\emph{(Priority: critical.)}
Prove estimates~(\ref{XV-A})--(\ref{XV-C}):
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$
uniformly in $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\end{problem}

\begin{problem}[H1$\Rightarrow$op.~norm;
SOT to operator upgrade]
\label{prob:H1H3}
\emph{(Priority: high.)}
H1 gives $\Ac\to 0$ in SOT (Thm.~\ref{thm:H1-H3-ptwise}).
Does this upgrade to
$\norm{\Ac}_{\mathrm{op}}\to 0$
uniformly in $c$?
This would follow if
$\{\Ac:c>c_0\}$ is equicontinuous
or if $\Pc$ is compact
(which it is not on $L^2$,
but the question is whether
the interaction with $\Pi_N^{(c)}$
gives effective compactness).
Alternative: reformulate H3
as $\mathrm{Tr}(\Ac)\to 0$
and ask whether H1 controls the trace directly.
\end{problem}

\begin{problem}[NC$^c\Rightarrow$H3;
logically independent]
\label{prob:NC-H3}
\emph{(Priority: clarification; likely false.)}
Remark~\ref{rem:NC-converse}
gives two mechanisms
by which NC$^c$ can coexist with
H3 failing.
Construct an explicit example
(real or model) to confirm.
\end{problem}

\begin{problem}[NC from Papers~VIII, XII]
\label{prob:NC-direct}
\emph{(Priority: medium.)}
Do the pointwise bounds of Paper~VIII
imply lower bounds on
$\mu_{n,c}([-c_0,c_0])$
without full WKB?
\end{problem}
