  Coefficient Stability of the Slepian Eigenbasis:\\
  Cross-Gram Operators, Spectral Evacuation, and the Transition Zone}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
The central obstruction to coefficient stability
of the Slepian eigenbasis is
\emph{spectral non-evacuation in the transition
zone}: the persistence of spectral mass of the
measures $\mu_{n,c}(d\xi)
:=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi$
in the regime $n\sim\alpha c$ as $c\to\infty$.

This paper reduces Hypothesis~H1 of Paper~XIII
(the $\ell^2$-norm convergence of Slepian
coefficients $a_n^{(c)}=\ip{g}{\Phic{c}{n}}$
as $c\to\infty$) to a single spectral
distribution problem in the transition
zone $n\sim\alpha c$.

The key structural results are:
(i)~the cross-Gram operator $T_c$ satisfies
$\norm{T_c}\le 1$ uniformly;
(ii)~the tail functional
\[
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0])
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}
\]
controls the tail via
$\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c)$;
(iii)~the operator $\Bn:=(I-\Pi_N^{(c)})\Pc$
satisfies $\norm{\Bn}_{\mathrm{op}}\le 1$
(composition of orthogonal projections),
and $S_N(c)$ equals the basis-relative
spectral sum $\sum_{n>N}\norm{\Bn\Phic{c}{n}}^2$,
which we write formally as $\mathrm{Tr}_{\mathrm{PSWF}}(\Ac)$
(trace in the sense of the PSWF-ONB
spectral decomposition, not a global
operator trace on $L^2(\mathbb{R})$);
(iv)~$\norm{\Ac}_{\mathrm{op}}\le S_N(c)$
with strict inequality in general,
and the relevant quadratic form is
$Q_c^N(g)=\ip{\Ac g}{g}$
where $\Ac:=\Bn^*\Bn\ge 0$.

H3 ($\sup_c S_N(c)\to 0$) implies H1
(proved unconditionally).
H1 implies $\ip{\Ac g}{g}\to 0$ for each
fixed $g$ (pointwise quadratic form
convergence, diagonal WOT-type);
the gap to operator norm convergence
involves a \emph{rank growth conjecture}:
the current lower bound
$\mathrm{rank}(\Ac)\ge S_N(c)$
is proved, but $\mathrm{rank}(\Ac)\to\infty$
remains open (Open Problem~\ref{prob:H1H3}).

The \emph{No-Concentration condition} (NC)
quantifies spectral non-evacuation:
NC$\Rightarrow$H3 fails (proved).
The converse is false even at the abstract
level: NC is sufficient but not necessary
for H3 to fail, as shown by
Proposition~\ref{prop:NC-not-H3}
(a logical separation result for abstract
measure families; PSWF compatibility
is not claimed).
NC holds under the local Weyl law (Paper~XV).
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Context}

Paper~XIII identified H1 as the key
remaining hypothesis for completeness
$\Kspace=L^2(\mathbb{R})$.
Paper~XIV reduces H1 to a spectral
distribution problem for the
PSWF eigenbasis in the transition zone.
All conditional steps are explicitly labelled.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$
($c\to\infty$).
\end{hypothesis}

H1 splits:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
\eqref{eq:H1ip} is proved
(Prop.~\ref{prop:H1ip});
\eqref{eq:H1norm} is equivalent to H3
(given OP~\ref{prob:H1H3}).
The core difficulty:
$a^{(c)}\rightharpoonup a^{(\infty)}$
does not imply norm convergence;
Paper~XIV localises this gap precisely.

\subsection{The central new framing}

\begin{definition}[Spectral measures and tail functional]
\label{def:spectral-measures}
For $n\ge 0$, $c>c_0$, define the
spectral probability measures
\[
  \mu_{n,c}(d\xi)
  :=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi,
  \qquad
  \mu_{n,c}(\mathbb{R})=1\;(\text{Plancherel}).
\]
The tail functional is defined directly
via the PSWF-ONB expansion:
\begin{equation}\label{eq:SN-def}
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0])
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}.
\end{equation}
This is a convergent series of non-negative
terms (each bounded by 1 by Plancherel);
no global operator-theoretic interpretation
is required for this definition.
The quadratic form in $g$:
\[
  Q_c^N(g) := \sum_{n>N}|a_n^{(c)}|^2
  = \norm{(I-\Pi_N^{(c)})\Pc g}^2.
\]
The obstruction to H1 is
\emph{spectral non-evacuation}:
the failure of $\mu_{n,c}([-c_0,c_0])$
to vanish for $n>N$, uniformly in $c$.
\end{definition}

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $0<\alpha<\kappa<1$:
$\mathcal{T}_{\alpha,\kappa}(c)
:=\{n: \alpha c < n \le \kappa c\}$.
This band has $\sim(\kappa-\alpha)c$ elements
and captures the regime where Slepian
eigenvalues $\lambda_n^{(c)}$
transit between $\approx 1$ and $\approx 0$.
\end{definition}

\begin{lemma}[From series]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item $\sum_n|a_n^{(c)}|^2\le\norm{g}^2$;

\item $a_n^{(c)}\to a_n^{(\infty)}$ (each fixed $n$).
\item For $n\le\kappa c$:
  $|\widehat{\Phi_n^{(c)}}(\xi)|\ge c_1 c^{-1/2}$
  on $E_c^{(n)}\subseteq[-c,c]$,
  $|E_c^{(n)}|\ge c_* c^{-1/2}$.
\item $\operatorname{supp}\hat g\subseteq[-c_0,c_0]$.
\end{enumerate}
\end{lemma}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{theorem}[$\norm{T_c}\le 1$]
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{theorem}
\begin{proof}
Bessel:
$\norm{T_c g}^2=\sum_n|\ip{g}{\Phic{c}{n}}|^2
\le\norm{g}^2$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds unconditionally.
\end{proposition}
\begin{proof}
Paper~XII: $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$.
\end{proof}

\begin{remark}
By weak lower semicontinuity:
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$,
so H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}\le\norm{a^{(\infty)}}$.
\end{remark}

%%═══ 4. TAIL REDUCTION AND THE QUADRATIC FORM
\section{Tail Reduction, Operator Structure, and the Quadratic Form}
\label{sec:tail}

\begin{remark}[Foundational convention]
\label{rem:foundation}
Throughout this section, all operator-theoretic
quantities ($\mathrm{Tr}$, HS norm, rank)
are understood relative to the PSWF-ONB
$\{\Phic{c}{n}\}_{n>N}$.
Specifically:
\begin{itemize}[leftmargin=*]
\item $S_N(c)$ is \emph{defined} by the
  convergent series~\eqref{eq:SN-def};
  it is not defined via a global operator trace
  on $L^2(\mathbb{R})$.
\item The notation $\mathrm{Tr}_{\mathrm{PSWF}}(\Ac)$
  means $\sum_{n>N}\ip{\Ac\Phic{c}{n}}{\Phic{c}{n}}$,
  a basis-relative spectral sum.
\item $P_{c_0}:L^2(\mathbb{R})\to\PW_{c_0}^2$
  is the orthogonal projection (Fourier
  multiplier by $\mathbf{1}_{[-c_0,c_0]}$);
  it is \emph{not} Hilbert--Schmidt on
  $L^2(\mathbb{R})$ (its kernel is translation-
  invariant, hence $\|P_{c_0}\|_{\mathrm{HS}}
  =\infty$). No HS property of $P_{c_0}$
  on $L^2(\mathbb{R})$ is claimed or used.
\end{itemize}
\end{remark}

\begin{definition}[$S_N(c)$ and $\sigma(c)$]
\label{def:SN}
\begin{align*}
  S_N(c) &:= \sum_{n>N}\mu_{n,c}([-c_0,c_0])
            = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2},\\
  \sigma(c) &:= \sum_{n\ge 0}\mu_{n,c}([-c_0,c_0]).
\end{align*}
Both series converge: each term
$\mu_{n,c}([-c_0,c_0])\in[0,1]$,
and for $S_N(c)$ the sum is finite
for each fixed $N,c$ by
Lemma~\ref{lem:inherited}(A).

\medskip\noindent
\emph{Divergence of $\sigma(c)$ vs.\ vanishing of $S_N(c)$.}
The classical Slepian--Weyl result
(in the $L^2[-1,1]$ setting, see
Remark~\ref{rem:sigma-slepian})
gives $\sigma(c)\sim cc_0/\pi\to\infty$.
This does \emph{not} preclude $S_N(c)\to 0$:
the total spectral mass grows while
the high-index tail $\{n>N\}$ may evacuate.
H3 asks whether this tail evacuation
occurs uniformly in $c$.
\end{definition}

\begin{remark}[Classical status of $\sigma(c)$]
\label{rem:sigma-slepian}
The identity $\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)$
and the asymptotics $\sigma(c)\sim cc_0/\pi$
are classical results of Slepian--Weyl
\cite{Slepian1961}, established in the
time-limited / band-limited operator framework
on $L^2[-1,1]$ (or equivalently on
finite-dimensional eigenspace truncations).
We use $\sigma(c)\sim cc_0/\pi$ only as an
asymptotic reference; no global trace-class
claim on $L^2(\mathbb{R})$ is made.
\end{remark}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
\[
  Q_c^N(g):=\sum_{n>N}|a_n^{(c)}|^2
  \le\norm{g}^2\cdot S_N(c).
\]
\end{lemma}
\begin{proof}
Since $\operatorname{supp}\hat g\subseteq[-c_0,c_0]$:
\[
  a_n^{(c)}
  =\ip{\hat g}{
    \widehat{\Phi_n^{(c)}}\cdot
    \mathbf{1}_{[-c_0,c_0]}}_{L^2}.
\]
Cauchy--Schwarz:
$|a_n^{(c)}|^2
\le\norm{g}^2\cdot
\mu_{n,c}([-c_0,c_0])$.
Sum over $n>N$ and use~\eqref{eq:SN-def}.
\end{proof}

\begin{corollary}[Equi-tightness]
$\sup_c S_N(c)\to 0
\Rightarrow\{a^{(c)}\}_c$
equi-tight in $\ell^2$.
\end{corollary}

\begin{theorem}[Operator structure and spectral sum identity]
\label{thm:SN-operator}
Define:
\begin{align}\label{eq:Bn-def}
  \Bn &:= (I-\Pi_N^{(c)})\Pc
    \;:\; \PW_{c_0}^2\to\PW_c^2,\\
  \Ac &:= \Bn^*\Bn
    \;:\; \PW_{c_0}^2\to\PW_{c_0}^2.
\end{align}

\medskip
\textbf{(0) Contraction.}
Both $\Pc$ and $(I-\Pi_N^{(c)})$
are orthogonal projections on $L^2(\mathbb{R})$,
hence contractions of operator norm $\le 1$.
Their composition satisfies:
\begin{equation}\label{eq:Bn-contraction}
  \norm{\Bn}_{\mathrm{op}}
  \le\norm{(I-\Pi_N^{(c)})}_{\mathrm{op}}
     \cdot\norm{\Pc}_{\mathrm{op}}
  \le 1\cdot 1 = 1.
\end{equation}
Consequently:
\begin{equation}\label{eq:Ac-bounded}
  0\le\Ac=\Bn^*\Bn
  \le\norm{\Bn}^2_{\mathrm{op}}\cdot I\le I.
\end{equation}
No Hilbert--Schmidt property of $\Pc$
on $L^2(\mathbb{R})$ is used here.

\medskip
\textbf{(I) Basis-relative spectral sum identity.}
\begin{equation}\label{eq:SN-spectral}
  S_N(c)
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}
  = \sum_{n>N}\norm{\Bn\Phic{c}{n}}^2_{L^2}
  =: \mathrm{Tr}_{\mathrm{PSWF}}(\Ac).
\end{equation}
\begin{proof}
The first equality is Definition~\ref{def:SN}.
For the second: since $\Phic{c}{n}\in\PW_c^2$
and $n>N$, we have
$(I-\Pi_N^{(c)})\Phic{c}{n}=\Phic{c}{n}$,
so
\[
  \norm{\Bn\Phic{c}{n}}^2
  =\norm{(I-\Pi_N^{(c)})\Pc\Phic{c}{n}}^2
  =\norm{\Pc\Phic{c}{n}}^2.
\]
The notation $\mathrm{Tr}_{\mathrm{PSWF}}(\Ac)$
stands for the basis-relative sum
$\sum_{n>N}\ip{\Ac\Phic{c}{n}}{\Phic{c}{n}}
=\sum_{n>N}\norm{\Bn\Phic{c}{n}}^2$,
which equals $S_N(c)$.
This is a convergent series of non-negative
reals; it is \emph{not} asserted to be a
trace in the functional-analytic sense of
$\mathfrak{S}_1(L^2(\mathbb{R}))$.
\end{proof}

\medskip
\textbf{(II) Operator norm inequality.}
Since $0\le\Ac\le I$, all eigenvalues
$\lambda_j(\Ac)\in[0,1]$, so:
\begin{equation}\label{eq:op-le-spectral}
  \norm{\Ac}_{\mathrm{op}}
  = \sup_j\lambda_j(\Ac)
  \le \sum_j\lambda_j(\Ac)
  = \mathrm{Tr}_{\mathrm{PSWF}}(\Ac)
  = S_N(c).
\end{equation}
The inequality is strict unless $\Ac$
has rank one.

\medskip
\textbf{(III) Quadratic form.}
For $g\in\PW_{c_0}^2$,
using $\Pc g=g$:
\begin{equation}\label{eq:Qc-key}
  Q_c^N(g)
  = \sum_{n>N}|\ip{g}{\Phic{c}{n}}|^2
  = \norm{\Bn g}^2
  = \ip{\Ac g}{g},
  \qquad 0\le\Ac\le I.
\end{equation}

\medskip
\textbf{(IV) The full hierarchy.}
\begin{equation}\label{eq:chain}
  \underbrace{\text{H3}}_{S_N(c)\to 0}
  \;\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0
  \;\Rightarrow\;
  Q_c^N(g)\to 0\;(\text{all }g)
  \;\Rightarrow\;
  \text{H1}
  \;\Rightarrow\;
  \ip{\Ac g}{g}\to 0\;(\text{each fixed }g).
\end{equation}
The reverse arrows do not hold in general.
H3 is a \emph{spectral-sum condition}
on the PSWF-ONB;
H1 yields only \emph{pointwise quadratic
form convergence} on each fixed $g$
(diagonal WOT-type, not full SOT).
\end{theorem}

\begin{remark}[Precise convergence hierarchy]
\label{rem:convergence-hierarchy}
The chain~\eqref{eq:chain} distinguishes
four levels:
\begin{enumerate}[\rm(1)]
\item \emph{H3} ($S_N(c)\to 0$):
  strongest; controls all terms
  $\mu_{n,c}([-c_0,c_0])$ simultaneously.
\item \emph{Op-norm $\to 0$}:
  controls the largest eigenvalue of $\Ac$
  uniformly in direction.
\item \emph{$Q_c^N(g)\to 0$ uniformly in $g$}:
  equivalent to op-norm $\to 0$.
\item \emph{$\ip{\Ac g}{g}\to 0$ each fixed $g$}
  (diagonal WOT-type): weakest;
  proved from H1 via $\ell^2$ convergence.
  Not equivalent to SOT
  ($\norm{\Ac g}\to 0$ for each $g$).
\end{enumerate}
The gap between (4) and (2) is the
rank growth conjecture (OP~\ref{prob:H1H3}).
\end{remark}

\begin{remark}[Spectral sum vs.\ operator norm]
\label{rem:HS-trace}
\begin{equation}\label{eq:trace-op-rank}
  \norm{\Ac}_{\mathrm{op}}
  \le \mathrm{Tr}_{\mathrm{PSWF}}(\Ac)
  = S_N(c)
  \le \mathrm{rank}(\Ac)\cdot\norm{\Ac}_{\mathrm{op}}.
\end{equation}
Neither direction (large $S_N(c) \Rightarrow$
large rank, or large rank $\Rightarrow$ large
$S_N(c)$) is automatic.
Closing the gap from diagonal WOT to
op-norm convergence requires eigenvalue
delocalisation beyond~\eqref{eq:trace-op-rank}.
\end{remark}

\begin{lemma}[Rank lower bound]
\label{lem:rank-lower}
For all $c>c_0$ and $N\ge 0$:
\begin{equation}\label{eq:rank-lower}
  \mathrm{rank}(\Ac)
  \ge \mathrm{Tr}_{\mathrm{PSWF}}(\Ac)
  = S_N(c).
\end{equation}
\end{lemma}
\begin{proof}
By~\eqref{eq:Ac-bounded},
$\lambda_j(\Ac)\in[0,1]$ for all $j$.
Therefore:
\[
  S_N(c)
  = \sum_j \lambda_j(\Ac)
  \le \mathrm{rank}(\Ac)\cdot
  \max_j\lambda_j(\Ac)
  \le \mathrm{rank}(\Ac).
\]
\end{proof}

\begin{remark}[Rank growth conjecture]
\label{rem:rank-growth}
Lemma~\ref{lem:rank-lower} gives
$\mathrm{rank}(\Ac)\ge S_N(c)$.
Under NC, $S_N(c)\ge c_*>0$ uniformly,
so the rank is bounded away from zero.
The current bounds do \emph{not} imply
$\mathrm{rank}(\Ac)\to\infty$:
that would require a lower bound on the
number of eigenvalues of $\Ac$ that remain
bounded away from zero (eigenvalue
delocalisation), which is not available
from~\eqref{eq:trace-op-rank} alone.
Whether $\mathrm{rank}(\Ac)\to\infty$
as $c\to\infty$ (for fixed $N$) is
an open conjecture (OP~\ref{prob:H1H3}).
\end{remark}

\begin{remark}[Spectral-sum vs.\ op-norm:
  independence warning]
\label{rem:rank-obstruction}
In general:
\[
  S_N(c)\to 0
  \;\not\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0.
\]
However, if $\mathrm{rank}(\Ac)\le R$
uniformly, then by~\eqref{eq:trace-op-rank},
$S_N(c)\to 0$ implies
$\norm{\Ac}_{\mathrm{op}}\to 0$.
Neither direction is automatic without
spectral distribution input.
\end{remark}

%%═══ 5. SPECTRAL MEASURES AND NC
\section{Spectral Measures and No-Concentration}
\label{sec:NC}

\begin{remark}[NC as spectral evacuation failure]
\label{rem:NC-evac}
The obstruction to H1 stated cleanly:
\begin{quote}
  \emph{H1 fails if and only if
  there exists persistent spectral mass
  of the measures $\mu_{n,c}$
  in the transition zone $n\sim\alpha c$
  (conditional on OP~\ref{prob:H1H3}
  and OP~\ref{prob:fourier-mass}).}
\end{quote}
``Spectral evacuation'' means
$\mu_{n,c}([-c_0,c_0])\to 0$
for $n>N$, uniformly in $c$,
i.e.\ $S_N(c)\to 0$.
NC is the quantitative, band-level
sufficient condition for non-evacuation;
H3 is the global tail condition.
Proposition~\ref{prop:NC-not-H3}
shows NC is \emph{sufficient}
but \emph{not necessary} for H3 to fail.
\end{remark}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
Fix $0<\alpha<\kappa<1$.
NC holds if there exists $c_*>0$ such that
for all large $c$:
\begin{equation}\label{eq:NC}
  \sum_{n\in\mathcal{T}_{\alpha,\kappa}(c)}
  \mu_{n,c}([-c_0,c_0]) \ge c_*>0.
\end{equation}
\end{definition}

\begin{remark}[Why proportional width is necessary]
\label{rem:prop-width}
A window of width $c^\beta$ ($\beta<1$)
has $\sim 2c^\beta$ indices;
under the local Weyl law,
the sum is $\sim 2c_0 c^{\beta-1}\to 0$:
insufficient for NC.
A proportional band
$\mathcal{T}_{\alpha,\kappa}(c)$
has $\sim(\kappa-\alpha)c$ indices,
giving sum $\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[NC as quantum ergodicity]
\label{rem:NC-QE}
In the semiclassical limit $c\to\infty$,
$n\sim\alpha c$, the rescaled measures
$c\cdot\mu_{n,c}|_{[-c,c]}$
are expected to converge weakly to
the WKB density
$(\pi\sqrt{1-(\xi/c)^2})^{-1}d\xi$:
a one-dimensional quantum ergodicity
statement \cite{Zworski2012}.
NC is the consequence for $[-c_0,c_0]$:
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$,
so the band sum
$\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[Local Weyl law $\Rightarrow$ NC]
\label{rem:Weyl-NC}
If uniformly for $n\in\mathcal{T}_{\alpha,\kappa}(c)$:
\begin{equation}\label{eq:local-Weyl}
  \mu_{n,c}([-c_0,c_0])\sim\frac{c_0}{c},
\end{equation}
then $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
\sim c_0(\kappa-\alpha)>0$.
NC holds with $c_*=c_0(\kappa-\alpha)/2$.
\end{remark}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under NC: for each fixed $N$,
$\liminf_{c\to\infty}S_N(c)\ge c_*>0$,
so H3 fails.
\end{proposition}
\begin{proof}
For large $c$: $\alpha c>N$,
hence $\mathcal{T}_{\alpha,\kappa}(c)\subseteq\{n>N\}$
and $S_N(c)\ge c_*$.
\end{proof}

\begin{proposition}[NC$^c$ does not imply H3;
  logical separation]
\label{prop:NC-not-H3}
\emph{Scope and categorical disclaimer.
This is a logical separation result
operating in the abstract category of
families of probability measures
$\{\mu_{n,c}\}$ satisfying only
normalisation ($\mu_{n,c}(\mathbb{R})=1$)
and positivity ($\mu_{n,c}\ge 0$).
It classifies the implication structure
between NC and H3 as abstract conditions
on such families.
It does not constrain PSWF spectral geometry:
the model family below does not arise from
$|\widehat{\Phi_n^{(c)}}|^2\,d\xi$,
and realizability within the PSWF
operator class is neither asserted nor needed.
The purpose is solely to show that NC and H3
are logically independent at the abstract level.}

\medskip\noindent
There exists a model family
$\{\mu_{n,c}\}_{n\ge 0,\,c>0}$
with $\mu_{n,c}\ge 0$,
$\mu_{n,c}(\mathbb{R})=1$, such that:
\begin{enumerate}[\rm(i)]
\item\label{it:NCc}
  \textup{NC fails}:
  for every $0<\alpha'<\kappa'<1$
  and every $c_*>0$,
  \[
    \liminf_{c\to\infty}
    \sum_{n\in\mathcal{T}_{\alpha',\kappa'}(c)}
    \mu_{n,c}([-c_0,c_0]) < c_*;
  \]
\item\label{it:H3fail}
  \textup{H3 fails}: for every fixed $N\ge 0$,
  \[
    \liminf_{c\to\infty} S_N(c) \ge 1.
  \]
\end{enumerate}
In particular, NC is not necessary for H3
to fail: $\mathrm{NC}^c\not\Rightarrow\mathrm{H3}$
at the abstract level.
\end{proposition}

\begin{proof}
Fix $\alpha\in(0,1)$.
Set $w(c):=\lfloor\sqrt{c}\rfloor\ge 1$
and define
\[
  I_c := \bigl\{n\in\mathbb{N}_0 :
    |n-\lfloor\alpha c\rfloor|\le w(c)\bigr\},
  \qquad |I_c|=2w(c)+1.
\]
Assign weights
\[
  \mu_{n,c}([-c_0,c_0]) :=
  \begin{cases}
    \dfrac{1}{|I_c|} & n\in I_c,\\
    0 & n\notin I_c,
  \end{cases}
\]
and extend $\mu_{n,c}$ to a probability
measure by placing remaining mass outside
$[-c_0,c_0]$.

\medskip
\textit{Proof of~\ref{it:H3fail}.}
Fix $N\ge 0$.
For $c>((N+1)/\alpha)^2$:
$\lfloor\alpha c\rfloor-w(c)
\ge \alpha c - 1 - \sqrt{c} > N$,
so $I_c\subseteq\{n>N\}$.
Therefore $S_N(c)\ge\sum_{n\in I_c}1/|I_c|=1$.

\medskip
\textit{Proof of~\ref{it:NCc}.}
Fix $0<\alpha'<\kappa'<1$ and $c_*>0$.

\textit{Case~1}: $\alpha\notin[\alpha',\kappa']$.
Then $\delta:=\mathrm{dist}(\alpha,[\alpha',\kappa'])>0$.
For large $c$ and $n\in I_c$:
$|n/c-\alpha|\le 2/\sqrt{c}<\delta/2$,
so $I_c\cap\mathcal{T}_{\alpha',\kappa'}(c)=\emptyset$
and the band sum is $0<c_*$.

\textit{Case~2}: $\alpha\in(\alpha',\kappa')$.
Set $\varepsilon
:=\tfrac{1}{2}\min(\alpha-\alpha',\,\kappa'-\alpha)>0$.
For large $c$, $|n/c-\alpha|<\varepsilon/2$
for all $n\in I_c$.
Choose $\beta$ with $\alpha<\beta<\kappa'$
and $\beta-\alpha>\varepsilon$;
then $I_c\cap\mathcal{T}_{\beta,\kappa'}(c)=\emptyset$
for large $c$, so NC fails for
$\mathcal{T}_{\beta,\kappa'}$.
\end{proof}

\begin{remark}[Structural separation and realizability]
\label{rem:separation}
The model concentrates mass in a
sub-proportional zone $|I_c|\sim 2\sqrt{c}\ll c$
that escapes every fixed proportional window,
while maintaining $S_N(c)\ge 1$.
NC is sufficient but not necessary
for H3 to fail.
Realizability of such behaviour within
the actual PSWF spectral measures
is addressed by OP~\ref{prob:fourier-mass}.
\end{remark}

%%═══ 6. H3 AND THE MAIN REDUCTION
\section{H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{remark}[H3 vs.\ classical Slepian asymptotics]
\label{rem:H3-vs-Slepian}
Classical Slepian theory fixes $c$ and studies
the distribution of $\lambda_n^{(c)}$ near
$n\sim 2c/\pi$ (Landau--Pollak--Slepian
transition count).
H3 fixes $N$ and requires
$\sup_{c>c_0} S_N(c)\to 0$ as $N\to\infty$:
a \emph{uniform-in-$c$, large-$N$} condition,
strictly stronger than any fixed-$c$ result.
H3 serves as a sufficient condition for H1.
\end{remark}

\begin{theorem}[H3 $\Rightarrow$ H1]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$, $g\in\PW_{c_0}^2$.
H3: choose $N$ with
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$;
then $Q_c^N(g)<\varepsilon/2$ for all $c$
(Lemma~\ref{lem:tail}).
Paper~XII: choose $c_\varepsilon$ with
$\sum_{n\le N}|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ for $c>c_\varepsilon$.
Adding gives $\norm{a^{(c)}-a^{(\infty)}}^2<\varepsilon$.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$
  pointwise quadratic form convergence]
\label{thm:H1-H3-ptwise}
Assume H1. For each fixed $g\in\PW_{c_0}^2$
and each fixed $N$:
\[
  \ip{\Ac g}{g} = Q_c^N(g) \to 0
  \quad\text{as }c\to\infty,
  \text{ then }N\to\infty.
\]
This is pointwise quadratic form convergence
(diagonal WOT-type);
it is \emph{not} uniform in $g$
and does not imply $\norm{\Ac}_{\mathrm{op}}\to 0$.
\end{theorem}
\begin{proof}
$Q_c^N(g)
\le 2\norm{a^{(c)}-a^{(\infty)}}^2_{\ell^2}
+2\sum_{n>N}|a_n^{(\infty)}|^2$.
H1: first term $\to 0$ as $c\to\infty$.
$a^{(\infty)}\in\ell^2$: second term $\to 0$
as $N\to\infty$.
\end{proof}

\begin{remark}[The uniformity gap]
\label{rem:unif-gap}
\[
  \text{H3}\;(S_N(c)\to 0)
  \Rightarrow
  \norm{\Ac}_{\mathrm{op}}\to 0
  \Rightarrow
  \text{H1}
  \Rightarrow
  \ip{\Ac g}{g}\to 0
  \;(\text{diagonal WOT, each fixed }g).
\]
The missing reverse:
diagonal WOT $\not\Rightarrow$
$\norm{\Ac}_{\mathrm{op}}\to 0$.
Rank control of $\Ac$ would close this gap
(Remark~\ref{rem:rank-obstruction});
the current lower bound
$\mathrm{rank}(\Ac)\ge S_N(c)$
(Lemma~\ref{lem:rank-lower})
does not imply rank growth.
\end{remark}

\begin{theorem}[Main reduction]
\label{thm:reduction}
\textup{(i)} Proved unconditionally:
\begin{align*}
  &\text{H3}\Rightarrow\text{H1},
  \quad
  \text{H1}\Rightarrow\ip{\Ac g}{g}\to 0
  \;(\text{diagonal WOT, each fixed }g),\\
  &\text{NC}\Rightarrow\text{H3 fails},
  \quad
  \text{local Weyl}\Rightarrow\text{NC},\\
  &\text{NC sufficient but not necessary
  for H3 to fail}
  \;(\text{Prop.}~\ref{prop:NC-not-H3},
  \text{ abstract level only}).
\end{align*}
\textup{(ii)} Open:
\[
  \mathrm{rank}(\Ac)\to\infty
  \;(\text{OP~\ref{prob:H1H3}}),
  \qquad
  \text{NC holds}
  \;(\text{OP~\ref{prob:fourier-mass}}).
\]
\textup{(iii)} Conditional chain:
\[
  \underbrace{\text{local Weyl law}}_{
    \text{OP~\ref{prob:fourier-mass}}}
  \Rightarrow
  \text{NC}
  \Rightarrow
  \text{H3 fails}
  \underset{\text{OP~\ref{prob:H1H3}}}
  {\Rightarrow}
  \text{H1 fails.}
\]
\end{theorem}

%%═══ 7. INTERPRETATION
\section{Interpretation and Context}
\label{sec:interp}

\subsection{The true obstruction: spectral non-evacuation}

\[
  \text{H1 fails}
  \;\Longleftrightarrow\;
  \text{spectral non-evacuation in }n\sim\alpha c
  \quad(\text{cond. on OPs~\ref{prob:H1H3},
  \ref{prob:fourier-mass}}).
\]
``Spectral evacuation'' means
$S_N(c)\to 0$ uniformly in $c$.
The model of Proposition~\ref{prop:NC-not-H3}
shows that mass can persist in $S_N(c)$
without any single proportional band
carrying it (at the abstract level).
NC is sufficient but not necessary
for this failure mode.

\subsection{NC as band-level; H3 as global-tail}

\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{lll}
\toprule
Condition & Level & Role \\
\midrule
NC & Band-level (sufficient)
  & Persistent mass in $\mathcal{T}_{\alpha,\kappa}(c)$ \\
H3 & Global tail (sufficient for H1)
  & $S_N(c)\to 0$ uniformly in $c$ \\
\bottomrule
\end{tabular}
\end{center}
NC$\Rightarrow$H3 fails (proved);
NC is not necessary for H3 to fail
(Prop.~\ref{prop:NC-not-H3}, abstract level).
The correct obstruction for H1 is H3.

\subsection{$S_N(c)$ as spectral sum; rank lower bound}

\[
  S_N(c)
  \le \mathrm{rank}(\Ac),
  \qquad
  \norm{\Ac}_{\mathrm{op}}
  \le S_N(c)
  \le \mathrm{rank}(\Ac)\cdot\norm{\Ac}_{\mathrm{op}}.
\]
H3 controls $S_N(c)$ from above;
the rank lower bound controls it from below.
Closing the gap requires eigenvalue
delocalisation (OP~\ref{prob:H1H3}).

\subsection{Quantum ergodicity in frequency space}

The spectral measures $\mu_{n,c}$
play the role of semiclassical measures;
the local Weyl law (OP~\ref{prob:fourier-mass})
asserts equidistribution on $[-c_0,c_0]$:
a one-dimensional quantum ergodicity
statement \cite{Zworski2012}.

%%═══ 8. PROGRAMME FOR PAPER XV
\section{Programme for Paper~XV}
\label{sec:PaperXV}

\subsection{The three required estimates}

\begin{enumerate}[\rm(A)]
\item\label{XV-A} \textbf{Upper bound, uniform:}
  $\mu_{n,c}([-c_0,c_0])\le C/c$
  for all $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-B} \textbf{Lower bound, uniform:}
  $\mu_{n,c}([-c_0,c_0])\ge 1/(Cc)$
  for all $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-C} \textbf{Summability:}
  $\mu_{n,c}([-c_0,c_0])=c_0/c+o(1/c)$
  uniformly, so
  $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
  \to c_0(\kappa-\alpha)$.
\end{enumerate}
(A)+(B) $\Rightarrow$ NC;
(C) $\Rightarrow$ equidistribution.

\subsection{Semiclassical approach}

$h:=1/c\to 0$; the PSWF solves
\begin{equation}\label{eq:PSWF-ODE}
  -(1-x^2)\partial_x^2\psi
  +2x\partial_x\psi
  +c^2x^2\psi=\chi_n^{(c)}\psi
\end{equation}
with energy $E_n^{(h)}=\chi_n^{(c)}/c^2
\to E_\alpha$ for $n\sim\alpha c$.
WKB in the classical region $|x|<x_0$
($x_0^2=E_\alpha$) and stationary phase
for $\xi\in[-c_0,c_0]$ give
$\mu_{n,c}([-c_0,c_0])\sim 2c_0C(E_\alpha)/c$,
yielding (A)--(C).

\subsection{Rigorous programme}

\begin{enumerate}
\item WKB with remainder $O(c^{-2})$,
  uniform in $n\in\mathcal{T}$ \cite{Osipov2013}.
\item Stationary phase with uniform remainder.
\item Prove (A)--(C); conclude NC.
\item Address OP~\ref{prob:H1H3}
  (rank growth via bandlimit geometry).
\end{enumerate}

%%═══ 9. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law; Paper~XV]
\label{prob:fourier-mass}
\emph{(Priority: critical.)}
Prove (A)--(C) of \S\ref{sec:PaperXV}:
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$
uniformly in $n\in\mathcal{T}_{\alpha,\kappa}(c)$.

\medskip\noindent
\emph{Additional difficulty
(orthogonality and band-sum correlations):}
The $\Phi_n^{(c)}$ are globally orthogonal
in $L^2[-1,1]$ but not in
$L^2[-c_0/c,\,c_0/c]$ after rescaling,
introducing off-diagonal correlations in
$\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])$.
Paper~XV must control both individual
$\mu_{n,c}$ asymptotics via WKB
and the correlation decay of PSWF overlaps
in the transition zone.
\end{problem}

\begin{problem}[Rank growth conjecture]
\label{prob:H1H3}
\emph{(Priority: high.)}
Lemma~\ref{lem:rank-lower} gives
$\mathrm{rank}(\Ac)\ge S_N(c)$;
under NC, this is uniformly positive.
The current bounds do \emph{not} imply
$\mathrm{rank}(\Ac)\to\infty$:
that requires a growing number of eigenvalues
of $\Ac$ to be bounded away from zero
(eigenvalue delocalisation).
Conjecture: the Paley--Wiener / bandlimit
geometry forces $\mathrm{rank}(\Ac)\to\infty$
as $c\to\infty$ (fixed $N$).
If true, combined with
$S_N(c)=\mathrm{Tr}_{\mathrm{PSWF}}(\Ac)$
and~\eqref{eq:trace-op-rank},
this forces $\norm{\Ac}_{\mathrm{op}}\to 0$,
closing the diagonal-WOT-to-op-norm gap.
\end{problem}

\begin{problem}[NC from Papers~VIII, XII]
\label{prob:NC-direct}
\emph{(Priority: medium.)}
Do the pointwise bounds of Paper~VIII
imply lower bounds on
$\mu_{n,c}([-c_0,c_0])$
without full WKB?
\end{problem}
