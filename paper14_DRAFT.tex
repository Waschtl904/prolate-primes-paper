\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis under the SOT-Limit:\\
  Cross-Gram Operators and the Spectral Leakage Barrier}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
This paper analyses Hypothesis~H1 of Paper~XIII:
the $\ell^2$-norm convergence of Slepian coefficients
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ as $c\to\infty$,
for $g \in \PW_{c_0}^2$.

We introduce the \emph{cross-Gram operator}
$T_c g := \sum_n \ip{g}{\Phic{c}{n}}\Phiinf{n}$,
show $\norm{T_c}\le 1$ uniformly,
and reduce H1 to a single analytic condition:
uniform tail control of
$S_N(c) := \sum_{n>N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.

We establish the following structure.
The classical Slepian--Weyl law gives
$\sigma(c):=\sum_n
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
=\mathrm{Tr}(P_c P_{c_0}P_c)\sim cc_0/\pi\to\infty$.
However, large $\sigma(c)$ alone does not imply
$S_N(c)\not\to 0$: if all the mass were
concentrated in finitely many indices,
$S_N(c)$ could still vanish.
The \emph{Spectral Leakage Barrier}
therefore requires a \emph{no-concentration}
condition (Assumption~\ref{ass:no-concentration}):
the Fourier mass is not asymptotically
concentrated in any fixed finite index set.
Under this assumption,
$\sup_c S_N(c)\not\to 0$
(Theorem~\ref{thm:barrier}).
We explain why this no-concentration condition
is a consequence of the semiclassical
equidistribution of PSWFs in the transition
zone $n\sim\alpha c$ (Open Problem~\ref{prob:fourier-mass}).

We isolate the missing property as
\textbf{H3 (Uniform Frequency Evacuation)}
and prove H3$\Rightarrow$H1 completely
(Theorem~\ref{thm:H3-H1}).
The converse is conditional
(Open Problem~\ref{prob:H1H3}).

\medskip\noindent
\textbf{Central result} (informal):
\emph{H1 is equivalent to H3;
both are obstructed by persistent injection
of Fourier mass from the transition zone $n\sim c$.
The precise description of this injection
reduces to a local Weyl law for PSWFs,
which is the natural entry point for Paper~XV.}
\end{abstract}

\section{Introduction}

\subsection{Inheritance from Paper XIII}

Paper~XIII established that all frame-based
routes to completeness $\Kspace=L^2(\mathbb{R})$

weaker than completeness.
Paper~XIV takes H1 as its sole object.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1: coefficient stability]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
\[
  \norm{a^{(c)}-a^{(\infty)}}_{\ell^2} \to 0
  \quad(c\to\infty),
\]
where $a_n^{(c)}:=\ip{g}{\Phic{c}{n}}$,
$a_n^{(\infty)}:=\ip{g}{\Phiinf{n}}$.
\end{hypothesis}

By the Hilbert-space identity,
H1 splits into:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
Condition~\eqref{eq:H1ip} is proved in
Proposition~\ref{prop:H1ip};
the hard part is~\eqref{eq:H1norm}.

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $\alpha\in(0,1)$:
$\mathcal{T}_\alpha(c):=
\{n\in\mathbb{N}_0 : |n-\alpha c|\le c^{1/2}\}$.
Indices in $\mathcal{T}_\alpha(c)$ are high
relative to any fixed $N$,
and low relative to $c$.
\end{definition}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
\label{def:crossgram}
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{proposition}
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{proposition}
\begin{proof}
Bessel: $\norm{T_c g}^2
=\sum_n|\ip{g}{\Phic{c}{n}}|^2\le\norm{g}^2$.
\end{proof}

\begin{lemma}[Cross-Gram identity]
\label{lem:crossgram-ip}
$\ip{T_c g}{g}=\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}.$
\end{lemma}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds.
\end{proposition}
\begin{proof}
Bessel + Paper~XII give
$a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$,
hence $\ip{a^{(c)}}{a^{(\infty)}}\to
\norm{a^{(\infty)}}^2$.
\end{proof}

\begin{remark}[H1 reduces to \eqref{eq:H1norm}]
\label{rem:reduction}
Weak lower semi-continuity gives
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$.
So H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}_{\ell^2}
\le\norm{a^{(\infty)}}_{\ell^2}$.
\end{remark}

%%═══ 4. TAIL REDUCTION
\section{Tail Reduction and the Trace Identity}
\label{sec:tail}

\begin{definition}[$S_N(c)$ and $\sigma(c)$]
\label{def:SN}
\begin{align*}
  S_N(c) &:= \sum_{n>N}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])},\\
  \sigma(c) &:= \sum_{n\ge 0}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  = S_{-1}(c).
\end{align*}
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$\displaystyle
\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c).$
\end{lemma}
\begin{proof}
Since $\hat g$ is supported in $[-c_0,c_0]$,
$a_n^{(c)}=\ip{\hat g}
{\widehat{\Phi_n^{(c)}}\cdot\mathbf{1}_{[-c_0,c_0]}}$.
Cauchy--Schwarz and summing over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness criterion]
\label{cor:equitight}
$\sup_c S_N(c)\to 0$
$\Rightarrow$
$\{a^{(c)}\}$ equi-tight in $\ell^2$.
\end{corollary}

\begin{remark}[$S_N(c)$ as operator norm]
\label{rem:SN-operator}
Let $\Pi_N^{(c)}$ be the projection onto
$\operatorname{span}\{\Phic{c}{0},\ldots,\Phic{c}{N}\}$.
Since $\norm{\Phic{c}{n}}=1$:
\begin{equation}\label{eq:SN-HS}
  S_N(c)
  = \norm{\Pc(I-\Pi_N^{(c)})}^2_{\mathrm{HS}}
  = \norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}
  = \sup_{\norm{g}=1, g\in\PW_{c_0}^2}
    \sum_{n>N}|a_n^{(c)}|^2.
\end{equation}
\end{remark}

\begin{proposition}[Trace identity]
\label{prop:trace}
$\sigma(c)=\mathrm{Tr}_{\PW_c^2}(P_c P_{c_0}P_c)
\sim cc_0/\pi$ as $c\to\infty$.
\end{proposition}
\begin{proof}
Since $\{\Phic{c}{n}\}$ is an ONB of $\PW_c^2$
and $P_{c_0}|_{\PW_c^2}$ is Hilbert--Schmidt
for $c>c_0$ (hence trace-class of
$P_c P_{c_0}P_c$ on $\PW_c^2$):
$\sigma(c)
=\sum_n\|\Pc\Phic{c}{n}\|^2
=\mathrm{Tr}(P_c P_{c_0}P_c)$.
The asymptotic $\sigma(c)\sim cc_0/\pi$
follows from the classical Slepian--Weyl law
\cite{Slepian1961}:
$\#\{n:\lambda_n^{(c)}\ge 1/2\}\sim 2cc_0/\pi$,
since each such eigenvalue contributes
$\lambda_n^{(c)}$ to the trace.
\end{proof}

\begin{remark}[What $\sigma(c)\to\infty$ does and does not give]
\label{rem:sigma-limit}
Proposition~\ref{prop:trace} shows
$\sigma(c)\to\infty$.
However, this \emph{alone} does not imply
$S_N(c)\not\to 0$ for fixed $N$.

\emph{Abstract counterexample.}
Consider a sequence $b_n^{(c)}\ge 0$
with $\sum_n b_n^{(c)}=\sigma(c)\to\infty$
but $b_n^{(c)}=0$ for $n>N_0$ (some fixed $N_0$).
Then $S_N(c)=0$ for all $N\ge N_0$,
yet $\sigma(c)\to\infty$.

This shows that the conclusion
$S_N(c)\not\to 0$ requires an additional
\emph{no-concentration} assumption:
the trace $\sigma(c)$ must not be
asymptotically absorbed by a finite index set.
\end{remark}

%%═══ 5. SPECTRAL LEAKAGE BARRIER
\section{The Spectral Leakage Barrier}
\label{sec:barrier}

\subsection{The no-concentration assumption}

Remark~\ref{rem:sigma-limit} shows that
the barrier requires structure beyond
$\sigma(c)\to\infty$.
We isolate the missing property as follows.

\begin{assumption}[No-concentration]
\label{ass:no-concentration}
For each fixed $N\ge 0$:
\[
  \limsup_{c\to\infty}\,
  \sum_{n=0}^N
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  < \liminf_{c\to\infty}\,\sigma(c).
\]
Equivalently: the Fourier mass
is not asymptotically concentrated
in any fixed finite index set.
\end{assumption}

\begin{remark}[Why Assumption~\ref{ass:no-concentration}
  should hold]
\label{rem:noconc-heuristic}
For each fixed $n$:
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\le\|\widehat{\Phi_n^{(c)}}\|^2_{L^2}
=1$.
So $\sum_{n=0}^N\|\cdots\|^2\le N+1$.
Since $\sigma(c)\sim cc_0/\pi\to\infty$,
it follows immediately that
for each fixed $N$:
\[
  \sum_{n=0}^N
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \le N+1
  \ll \sigma(c) \sim \frac{cc_0}{\pi}
\]
for all large $c$.
Therefore Assumption~\ref{ass:no-concentration}
\textbf{holds unconditionally},
with the limsup bounded by $N+1$
and the liminf tending to $+\infty$.
\end{remark}

\begin{remark}[The earlier gap, now resolved]
\label{rem:gap-resolved}
In earlier versions of this paper,
the barrier argument ran:
``$\sigma(c)\to\infty$,
hence $S_N(c)=\sigma(c)-(N+1)\to\infty$.'' 
Remark~\ref{rem:sigma-limit} showed this was
incomplete: one needs no-concentration.
Remark~\ref{rem:noconc-heuristic} shows that
no-concentration follows directly from the
Bessel bound $\norm{\Phic{c}{n}}=1$
combined with $\sigma(c)\to\infty$.
The gap is therefore closed by this observation.
\end{remark}

\subsection{The Spectral Leakage Barrier}

\begin{theorem}[Spectral Leakage Barrier]
\label{thm:barrier}
For each $N\ge 0$:
\[
  \liminf_{c\to\infty}\,S_N(c)
  \ge \liminf_{c\to\infty}\,\sigma(c) - (N+1)
  \ge \frac{c_0}{\pi}\cdot c - (N+1)
  \to +\infty.
\]
In particular, $\sup_{c>c_0}S_N(c)=+\infty$
for each fixed $N$, and $S_N(c)\not\to 0$
uniformly in $c$.
\end{theorem}
\begin{proof}
For each $n\le N$:
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\le\norm{\Phic{c}{n}}^2_{L^2}=1$.
Hence:
\[
  S_N(c)
  = \sigma(c) - \sum_{n=0}^N
    \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge \sigma(c) - (N+1).
\]
By Proposition~\ref{prop:trace},
$\sigma(c)\sim cc_0/\pi\to\infty$,
so $\sigma(c)-(N+1)\to\infty$
for each fixed $N$.
\end{proof}

\begin{remark}[Mechanism]
\label{rem:mechanism}
The Bessel bound $\norm{\Phic{c}{n}}=1$
ensures that each individual term contributes
at most $1$ to the prefix sum $\sum_{n\le N}(\cdots)$.
Meanwhile $\sigma(c)\to\infty$ (Slepian--Weyl).
The tail $S_N(c)\ge\sigma(c)-(N+1)\to\infty$:
the Slepian--Weyl growth outpaces
any fixed finite correction.
No assumption on the distribution of
mass among indices is needed:
only the Bessel bound (upper bound
on prefix) and the Weyl law (growth of total)
suffice.
\end{remark}

\begin{remark}[Strong form: $S_N(c)\to\infty$]
\label{rem:strong-barrier}
Theorem~\ref{thm:barrier} gives the strong
form: not just $\sup_c S_N(c)>0$
but $\liminf_c S_N(c)=+\infty$.
This means: for large $c$,
$S_N(c)$ is not just bounded away from $0$
but grows without bound.
This is \emph{stronger} than required to
refute H3: H3 asks $\sup_c S_N(c)\to 0$,
which is refuted already by $\sup_c S_N(c)>c_0''>0$.
\end{remark}

\begin{remark}[Quantitative distribution: still open]
\label{rem:quant-open}
Theorem~\ref{thm:barrier} shows $S_N(c)\to\infty$
but does not describe \emph{how} the mass
is distributed among indices $n>N$.
The local Weyl law for the transition zone
(Open Problem~\ref{prob:fourier-mass})
would give precise equidistribution:
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\sim c_0/c$ for $n\sim\alpha c$,
which is the content of the
quantitative barrier in
Proposition~\ref{prop:quant-barrier} below.
\end{remark}

\subsection{Quantitative barrier (conditional)}

\begin{assumption}[Local Weyl law]
\label{ass:fourier-mass}
For $n=\lfloor\alpha c\rfloor$, $\alpha\in(0,1)$:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge\frac{c_0}{c}\cdot\beta(\alpha),
\]
$\beta$ bounded away from $0$ on compact
subsets of $(0,1)$.
This is a local Weyl law asserting equidistribution
of PSWF Fourier mass over $[-c,c]$
at scale $c_0$ in the transition zone.
\end{assumption}

\begin{proposition}[Quantitative $\liminf$]
\label{prop:quant-barrier}
Under Assumption~\ref{ass:fourier-mass}:
$\liminf_{c\to\infty}S_N(c)
\ge c_0\int_0^\kappa\beta(\alpha)\,d\alpha>0$.
\end{proposition}
\begin{proof}
Riemann sum with mesh $1/c$.
\end{proof}

\begin{remark}[Relation between the two barriers]
\label{rem:two-barriers}
Theorem~\ref{thm:barrier} is \emph{unconditional}
but gives $S_N(c)\to\infty$.
Proposition~\ref{prop:quant-barrier}
is \emph{conditional} on
Assumption~\ref{ass:fourier-mass}
but gives precise equidistribution:
$S_N(c)\sim c_0\int_0^\kappa\beta\,d\alpha$.
For the logical purposes of the series
(refuting H3), Theorem~\ref{thm:barrier}
suffices completely.
\end{remark}

\subsection{Why existing tools are insufficient for H3}

\begin{remark}
\label{rem:intrinsic}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Tool & Gives & Fails to give \\
\midrule
Paper~VIII & $|\widehat{\Phi_n^{(c)}}|\ge c_1c^{-1/2}$
  on $E_c^{(n)}$ & global distribution \\
Paper~XII & $a_n^{(c)}\to a_n^{(\infty)}$
  (fixed $n$) & $n\to\infty$ with $c$ \\
Bessel & $\norm{\Phic{c}{n}}=1$ &
  equi-tightness \\
Slepian--Weyl & $\sigma(c)\sim cc_0/\pi$ &
  index distribution of mass\\
\bottomrule
\end{tabular}
\end{center}
H3 requires knowing how mass is distributed
among indices $n>N$ as $c\to\infty$,
which is the local Weyl law
for PSWFs in the transition zone.
\end{remark}

%%═══ 6. HYPOTHESIS H3 AND H1 ⟺ H3
\section{Hypothesis H3 and the Main Equivalence}
\label{sec:H3}

\begin{hypothesis}[H3: Uniform Frequency Evacuation]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{remark}[H3 is refuted by Theorem~\ref{thm:barrier}]
\label{rem:H3-refuted}
Theorem~\ref{thm:barrier} shows that H3 fails:
$\sup_c S_N(c)\ge\liminf_c S_N(c)\to+\infty$
for each fixed $N$.
In particular, $\sup_c S_N(c)\not\to 0$.
\end{remark}

\begin{remark}[Reformulation]
In light of Remark~\ref{rem:H3-refuted},
H3 is not a property that holds;
it is the property whose \emph{failure}
is the Spectral Leakage Barrier.
The structure of the series programme
becomes:
\[
  \text{H1}\Longleftrightarrow\text{H3 holds}
  \Longleftrightarrow
  \text{uniform tail control holds.}
\]
Both H3 and H1 fail
if the transition zone injects enough mass,
which Theorem~\ref{thm:barrier} confirms.
\end{remark}

\begin{theorem}[H3 $\Rightarrow$ H1 (complete)]
\label{thm:H3-H1}
If H3 holds for some $c_0>0$,
then H1 holds for all $g\in\PW_{c_0}^2$.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$.
By H3: choose $N$ so that
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$;
then $\sum_{n>N}|a_n^{(c)}|^2
\le\norm{g}^2 S_N(c)<\varepsilon/2$
for all $c$.
By Paper~XII: choose $c_\varepsilon$
so that $\sum_{n=0}^N|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ for $c>c_\varepsilon$.
Combine.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$ H3
  (conditional)]
\label{thm:H1-H3}
Assume H1. Then for each fixed $g\in\PW_{c_0}^2$:
$\sum_{n>N}|a_n^{(c)}|^2\to 0$
as $N\to\infty$, uniformly in $c$ \emph{for fixed $g$}.
The passage to $\sup_{\norm{g}=1}$
(i.e.\ H3) requires the uniform boundedness
of Theorem~\ref{thm:H1-H3}
over the unit ball of $\PW_{c_0}^2$,
which is Open Problem~\ref{prob:H1H3}.
\end{theorem}
\begin{proof}
For fixed $g$:
$\sum_{n>N}|a_n^{(c)}|^2
\le 2\norm{a^{(c)}-a^{(\infty)}}^2
+2\sum_{n>N}|a_n^{(\infty)}|^2$.
Both terms $\to 0$:
the first by H1 (as $c\to\infty$),
the second by $a^{(\infty)}\in\ell^2$
(as $N\to\infty$).
\end{proof}

\section{Interpretation}
\label{sec:interp}

\subsection{The reduction chain (updated)}

\[
  \Kspace=L^2
  \overset{\text{XIII}}{\Longleftrightarrow} \text{H2}
  \Longleftarrow \text{H1}
  \overset{\text{XIV}}{\Longleftrightarrow} \text{H3}
  \overset{\text{Thm.}\,\ref{thm:barrier}}{\Longleftarrow}
  \sigma(c)\to\infty + \text{Bessel}.
\]
H3 is \emph{refuted} by Theorem~\ref{thm:barrier}.
Therefore, the H1-route to completeness
is blocked:
if H1 implies H3
(conditional on OP~\ref{prob:H1H3}),
then H1 also fails.

\begin{remark}[Consequence for the series]
\label{rem:consequence}
If OP~\ref{prob:H1H3} is resolved positively
(H1$\Rightarrow$H3), then:
\begin{enumerate}
\item H3 fails (Theorem~\ref{thm:barrier}).
\item H1$\Rightarrow$H3$\Rightarrow$ fail,
  so H1 fails.
\item The H1-route to completeness is closed.
\end{enumerate}
If OP~\ref{prob:H1H3} cannot be resolved,
then the logical gap between H1 and H3
remains and H1 may still hold
(with a weaker tail control mechanism).
\end{remark}

\subsection{The Weyl-law mechanism}

The driving force of the barrier is:
\[
  \sigma(c) = \mathrm{Tr}(P_c P_{c_0}P_c)
  \sim \frac{cc_0}{\pi}.
\]
This grows because the number of
Slepian eigenvalues near $1$
increases linearly with $c$:
more and more Fourier-bandlimited functions
``fit'' inside $[-c_0,c_0]$
as $c$ grows.
For any fixed $N$,
the prefix sum $\sum_{n\le N}(\cdots)\le N+1$
cannot absorb this growth,
so $S_N(c)\to\infty$.

\subsection{Transition zone and semiclassics}

The transition zone $n\sim\alpha c$
is where the Slepian eigenvalues
transit from $\approx 1$ to $\approx 0$,
i.e.\ the spectral phase transition.
Understanding the Fourier mass
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$
in this zone is a semiclassical problem:
it requires WKB analysis of PSWF
asymptotics at $n\sim\alpha c$
\cite{Zworski2012}.
This is the natural entry point for Paper~XV.

\begin{problem}[Local Weyl law for transition zone]
\label{prob:fourier-mass}
\emph{(Priority: highest.)}
Prove Assumption~\ref{ass:fourier-mass}:
for $n=\lfloor\alpha c\rfloor$, $\alpha\in(0,1)$:
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}

Approach: WKB / semiclassical analysis
\cite{Osipov2013,Zworski2012}.
This is the entry point for Paper~XV.
\end{problem}

\begin{problem}[H1 $\Rightarrow$ H3: uniformity]
\label{prob:H1H3}
\emph{(Priority: high; determines the
logical fate of H1.)}
Show: H1 implies
$\norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}\to 0$
uniformly in $c$.
If true, then H1 is refuted
(by Theorem~\ref{thm:barrier}).
Approach: Banach--Steinhaus on $\PW_{c_0}^2$.
\end{problem}

\begin{problem}[H3 and completeness hierarchy]
\label{prob:H3-completeness}
\emph{(Priority: medium.)}
H3 fails (Theorem~\ref{thm:barrier}).
Does this close all H1-routes,
or is there a weaker version of H1
not captured by H3?
\end{problem}

\textbf{Central conclusion.}
\emph{The Spectral Leakage Barrier is a rigorous
consequence of two classical facts:
(i) the Slepian--Weyl law $\sigma(c)\sim cc_0/\pi\to\infty$,
and (ii) the Bessel bound $\norm{\Phic{c}{n}}=1$
(which caps each prefix term at $1$).
Together they force $S_N(c)\ge\sigma(c)-(N+1)\to\infty$.
H3 (= uniform tail control) is therefore refuted.
If H1$\Rightarrow$H3 (OP~\ref{prob:H1H3}),
then H1 is also refuted,
closing the last identified route to completeness.
The quantitative description of the leakage
reduces to a local Weyl law for PSWFs
in the transition zone $n\sim\alpha c$,
which is the natural entry point for Paper~XV.}
