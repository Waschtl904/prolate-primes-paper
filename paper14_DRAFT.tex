\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis:\\
  Cross-Gram Operators, Spectral Evacuation, and the Transition Zone}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
The central obstruction to coefficient stability
of the Slepian eigenbasis is
\emph{spectral non-evacuation in the transition
zone}: the persistence of spectral mass of the
measures $\mu_{n,c}(d\xi)
:=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi$
in the regime $n\sim\alpha c$ as $c\to\infty$.

This paper reduces Hypothesis~H1 of Paper~XIII
(the $\ell^2$-norm convergence of Slepian
coefficients $a_n^{(c)}=\ip{g}{\Phic{c}{n}}$
as $c\to\infty$) to a single spectral
distribution problem in the transition
zone $n\sim\alpha c$.

The key structural results are:
(i)~the cross-Gram operator $T_c$ satisfies
$\norm{T_c}\le 1$ uniformly;
(ii)~the tail functional
$S_N(c):=\sum_{n>N}\mu_{n,c}([-c_0,c_0])$
controls the tail via
$\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c)$;
(iii)~$S_N(c)=\mathrm{Tr}(\Ac)
=\mathrm{Tr}((I-\Pi_N^{(c)})\Pc(I-\Pi_N^{(c)}))$
is the Hilbert--Schmidt norm of
$\Bn:=(I-\Pi_N^{(c)})\Pc$,
which is Hilbert--Schmidt since $\Pc$ has a
Sinc integral kernel;
(iv)~$\norm{\Ac}_{\mathrm{op}}\le S_N(c)$
with strict inequality in general,
and the relevant quadratic form is
$Q_c^N(g)=\ip{\Ac g}{g}$
where $\Ac:=\Bn^*\Bn\ge 0$.

H3 ($\sup_c S_N(c)\to 0$) implies H1
(proved unconditionally).
H1 implies $\Ac\to 0$ in the strong operator
topology (proved in the sense that
$\ip{\Ac g}{g}\to 0$ for each fixed $g$);
the gap to operator norm convergence
involves a \emph{rank growth conjecture}:
the current lower bound
$\mathrm{rank}(\Ac)\ge S_N(c)$
is proved, but $\mathrm{rank}(\Ac)\to\infty$
remains open (Open Problem~\ref{prob:H1H3}).

The \emph{No-Concentration condition} (NC)
quantifies spectral non-evacuation:
NC$\Rightarrow$H3 fails (proved).
The converse is false even at the abstract
level: NC is sufficient but not necessary
for H3 to fail, as shown by
Proposition~\ref{prop:NC-not-H3}
(a logical separation result for abstract
measure families; PSWF compatibility
is not claimed).
NC holds under the local Weyl law (Paper~XV).
\end{abstract}

\section{Introduction}
\label{sec:intro}

\subsection{Context}

Paper~XIII identified H1 as the key
remaining hypothesis for completeness
$\Kspace=L^2(\mathbb{R})$.
Paper~XIV reduces H1 to a spectral
distribution problem for the
PSWF eigenbasis in the transition zone.
All conditional steps are explicitly labelled.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}\to 0$
($c\to\infty$).
\end{hypothesis}

H1 splits:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
\eqref{eq:H1ip} is proved
(Prop.~\ref{prop:H1ip});
\eqref{eq:H1norm} is equivalent to H3
(given OP~\ref{prob:H1H3}).
The core difficulty:
$a^{(c)}\rightharpoonup a^{(\infty)}$
does not imply norm convergence;
Paper~XIV localises this gap precisely.

\subsection{The central new framing}

\begin{definition}[Spectral measures]
\label{def:spectral-measures}
For $n\ge 0$, $c>c_0$:
\[
  \mu_{n,c}(d\xi)
  :=|\widehat{\Phi_n^{(c)}}(\xi)|^2\,d\xi.
\]
Plancherel: $\mu_{n,c}(\mathbb{R})=1$.
The tail functional and its quadratic form:
\[
  S_N(c) := \sum_{n>N}\mu_{n,c}([-c_0,c_0]),
  \qquad
  Q_c^N(g) := \sum_{n>N}|a_n^{(c)}|^2
  = \norm{(I-\Pi_N^{(c)})\Pc g}^2.
\]
The obstruction to H1 is
\emph{spectral non-evacuation}:
the failure of $\mu_{n,c}([-c_0,c_0])$
to vanish for $n>N$, uniformly in $c$.
\end{definition}

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $0<\alpha<\kappa<1$:
$\mathcal{T}_{\alpha,\kappa}(c)
:=\{n: \alpha c < n \le \kappa c\}$.
This band has $\sim(\kappa-\alpha)c$ elements
and captures the regime where Slepian
eigenvalues $\lambda_n^{(c)}$
transit between $\approx 1$ and $\approx 0$.
\end{definition}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{theorem}[$\norm{T_c}\le 1$]
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{theorem}
\begin{proof}
Bessel:
$\norm{T_c g}^2=\sum_n|\ip{g}{\Phic{c}{n}}|^2
\le\norm{g}^2$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds unconditionally.
\end{proposition}
\begin{proof}
Paper~XII: $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$.
\end{proof}

\begin{remark}
By weak lower semicontinuity:
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$,
so H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}\le\norm{a^{(\infty)}}$.
\end{remark}

%%═══ 4. TAIL REDUCTION, HS IDENTITY, AND QUADRATIC FORM
\section{Tail Reduction, HS Identity, and the Quadratic Form}
\label{sec:tail}

\begin{definition}[$S_N(c)$, $\sigma(c)$]
\label{def:SN}
$S_N(c):=\sum_{n>N}\mu_{n,c}([-c_0,c_0])$,
$\quad\sigma(c):=\sum_{n\ge 0}\mu_{n,c}([-c_0,c_0])$.

\medskip\noindent
Note that $S_N(c)$ admits an equivalent
operator-theoretic expression:
\begin{equation}\label{eq:SN-trace-explicit}
  S_N(c)
  = \mathrm{Tr}\bigl((I-\Pi_N^{(c)})\Pc(I-\Pi_N^{(c)})\bigr),
\end{equation}
as shown in Theorem~\ref{thm:SN-operator}(I) below.
This identity is the bridge between
the measure-theoretic and operator-theoretic
formulations; it is proved, not assumed.
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$Q_c^N(g):=\sum_{n>N}|a_n^{(c)}|^2
\le\norm{g}^2\cdot S_N(c)$.
\end{lemma}
\begin{proof}
Paley--Wiener:
$a_n^{(c)}
=\ip{\hat g}{
  \widehat{\Phi_n^{(c)}}\cdot
  \mathbf{1}_{[-c_0,c_0]}}_{L^2}$.
Cauchy--Schwarz:
$|a_n^{(c)}|^2
\le\norm{g}^2\cdot
\mu_{n,c}([-c_0,c_0])$.
Sum over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness]
$\sup_c S_N(c)\to 0
\Rightarrow\{a^{(c)}\}_c$
equi-tight in $\ell^2$.
\end{corollary}

\begin{theorem}[HS identity, quadratic form,
  and $S_N$ as trace control]
\label{thm:SN-operator}
Define:
\begin{align}\label{eq:Bn-def}
  \Bn &:= (I-\Pi_N^{(c)})\Pc
    \;:\; \PW_{c_0}^2\to\PW_c^2,\\
  \Ac &:= \Bn^*\Bn
    \;:\; \PW_{c_0}^2\to\PW_{c_0}^2.
\end{align}

\medskip
\textbf{(0) Hilbert--Schmidt and contraction.}
The operator $\Pc:\PW_{c_0}^2\to L^2(\mathbb{R})$
has integral kernel
$K(x,y)=\frac{\sin c_0(x-y)}{\pi(x-y)}$
(the Sinc kernel), which lies in
$L^2(\mathbb{R}\times\mathbb{R})$.
Hence $\Pc$ is Hilbert--Schmidt.
Since $(I-\Pi_N^{(c)})$ is an orthogonal
projection (hence a contraction),
$\Bn=(I-\Pi_N^{(c)})\Pc$ is also
Hilbert--Schmidt and satisfies
\begin{equation}\label{eq:Bn-contraction}
  \norm{\Bn}_{\mathrm{op}}
  \le\norm{(I-\Pi_N^{(c)})}_{\mathrm{op}}
     \cdot\norm{\Pc}_{\mathrm{op}}
  \le 1\cdot 1 = 1,
\end{equation}
where $\norm{\Pc}_{\mathrm{op}}\le 1$
since $\Pc$ is itself an orthogonal projection.
Consequently $\Ac=\Bn^*\Bn$ satisfies
$0\le\Ac\le\norm{\Bn}^2_{\mathrm{op}}\cdot I\le I$.

\medskip
\textbf{(I) HS identity.}
\begin{equation}\label{eq:SN-HS}
  S_N(c)
  = \sum_{n>N}\norm{\Pc\Phic{c}{n}}^2_{L^2}
  = \norm{\Bn}^2_{\mathrm{HS}}
  = \mathrm{Tr}(\Ac)
  = \mathrm{Tr}\bigl((I-\Pi_N^{(c)})\Pc(I-\Pi_N^{(c)})\bigr).
\end{equation}
\begin{proof}
Expand $\norm{\Bn}^2_{\mathrm{HS}}$
in the ONB $\{\Phic{c}{n}\}_{n>N}$:
\[
  \norm{\Bn}^2_{\mathrm{HS}}
  =\sum_{n>N}\norm{\Bn\Phic{c}{n}}^2
  =\sum_{n>N}\norm{(I-\Pi_N^{(c)})\Pc\Phic{c}{n}}^2
  =\sum_{n>N}\norm{\Pc\Phic{c}{n}}^2
  =S_N(c),
\]
where the third equality uses
$(I-\Pi_N^{(c)})\Phic{c}{n}=\Phic{c}{n}$
for $n>N$.
Since $\Ac=\Bn^*\Bn$ and $\Bn$ is HS,
$\mathrm{Tr}(\Ac)=\norm{\Bn}^2_{\mathrm{HS}}$.
The last equality in~\eqref{eq:SN-HS}
follows from cyclicity of trace:
$\mathrm{Tr}(\Bn^*\Bn)
=\mathrm{Tr}(\Bn\Bn^*)
=\mathrm{Tr}((I-\Pi_N^{(c)})\Pc(I-\Pi_N^{(c)}))$.
\end{proof}

\medskip
\textbf{(II) Operator norm inequality.}
\begin{equation}\label{eq:op-le-HS}
  \norm{\Ac}_{\mathrm{op}}
  = \norm{\Bn}^2_{\mathrm{op}}
  \le\norm{\Bn}^2_{\mathrm{HS}}
  = \mathrm{Tr}(\Ac) = S_N(c).
\end{equation}
In general $\norm{\Ac}_{\mathrm{op}}
<\mathrm{Tr}(\Ac)$ unless $\Ac$ has rank one.
\emph{The identity
$\norm{\Bn}^2_{\mathrm{op}}=S_N(c)$
claimed in earlier versions is incorrect.}

\medskip
\textbf{(III) Quadratic form.}
For $g\in\PW_{c_0}^2$,
using $\Pc g=g$:
\begin{equation}\label{eq:Qc-key}
  Q_c^N(g)
  = \sum_{n>N}|\ip{g}{\Phic{c}{n}}|^2
  = \norm{\Bn g}^2
  = \ip{\Ac g}{g},
  \qquad 0\le\Ac\le I.
\end{equation}

\medskip
\textbf{(IV) The full hierarchy.}
\begin{equation}\label{eq:chain}
  \underbrace{\text{H3}}_{\mathrm{Tr}(\Ac)\to 0}
  \;\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0
  \;\Rightarrow\;
  Q_c^N(g)\to 0\;(\text{unif.}\,g,\,\text{each fixed }c)
  \;\Rightarrow\;
  \text{H1}.
\end{equation}
The reverse arrows do not hold in general.
H3 is a \emph{trace-class condition};
H1 requires only pointwise convergence
of $\ip{\Ac g}{g}\to 0$ for each fixed $g$
(SOT in the sense of quadratic forms).
\end{theorem}

\begin{remark}[H3 as trace-class;
H1 as SOT convergence]
\label{rem:HS-trace}
$S_N(c)=\mathrm{Tr}(\Ac)$ sums
all eigenvalues of $\Ac$:
global spectral energy.
$\norm{\Ac}_{\mathrm{op}}$
is the largest eigenvalue: worst-case direction.
\begin{equation}\label{eq:trace-op-rank}
  \norm{\Ac}_{\mathrm{op}}
  \le \mathrm{Tr}(\Ac)
  = S_N(c)
  \le \mathrm{rank}(\Ac)\cdot\norm{\Ac}_{\mathrm{op}}.
\end{equation}
The relevant open problem (OP~\ref{prob:H1H3})
is whether SOT convergence $\ip{\Ac g}{g}\to 0$
(for each fixed $g$) implies
$\norm{\Ac}_{\mathrm{op}}\to 0$:
the classical SOT $\not\Rightarrow$ op.~norm gap.
\end{remark}

\begin{lemma}[Rank lower bound]
\label{lem:rank-lower}
For all $c>c_0$ and $N\ge 0$:
\begin{equation}\label{eq:rank-lower}
  \mathrm{rank}(\Ac) \ge \mathrm{Tr}(\Ac) = S_N(c).
\end{equation}
\end{lemma}
\begin{proof}
By Theorem~\ref{thm:SN-operator}(0),
$0\le\Ac\le I$, so all eigenvalues
$\lambda_j(\Ac)\in[0,1]$.
Therefore:
\[
  \mathrm{Tr}(\Ac)
  = \sum_j \lambda_j(\Ac)
  \le \mathrm{rank}(\Ac)\cdot
  \max_j\lambda_j(\Ac)
  \le \mathrm{rank}(\Ac)\cdot 1
  = \mathrm{rank}(\Ac).
\]
\end{proof}

\begin{remark}[Rank growth conjecture]
\label{rem:rank-growth}
Lemma~\ref{lem:rank-lower} gives
$\mathrm{rank}(\Ac)\ge S_N(c)$.
Under NC, $S_N(c)\ge c_*>0$ uniformly,
so the rank is bounded away from zero.
The current bounds do \emph{not} imply
$\mathrm{rank}(\Ac)\to\infty$:
that would require additional input
on the eigenvalue distribution of $\Ac$
beyond the trace bound.
Whether $\mathrm{rank}(\Ac)\to\infty$
as $c\to\infty$ (for fixed $N$) is
an open conjecture (OP~\ref{prob:H1H3});
if true, combined with
$\mathrm{Tr}(\Ac)=S_N(c)$
and~\eqref{eq:trace-op-rank},
it would force $\norm{\Ac}_{\mathrm{op}}\to 0$,
closing the gap to H1.
\end{remark}

\begin{remark}[Trace vs.\ operator norm
and the rank obstruction]
\label{rem:rank-obstruction}
In general:
\[
  \mathrm{Tr}(\Ac)\to 0
  \;\not\Rightarrow\;
  \norm{\Ac}_{\mathrm{op}}\to 0.
\]
However, if $\mathrm{rank}(\Ac)\le R$
uniformly, then by~\eqref{eq:trace-op-rank}
$\mathrm{Tr}(\Ac)\to 0$
implies $\norm{\Ac}_{\mathrm{op}}\to 0$.
Open Problem~\ref{prob:H1H3} asks whether
$\Ac=\Pc(I-\Pi_N^{(c)})P_c(I-\Pi_N^{(c)})\Pc$
admits such rank control via the
bandlimit geometry.
\end{remark}

\begin{proposition}[Trace identity and growth]
\label{prop:trace}
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim cc_0/\pi\to\infty$.
\end{proposition}
\begin{proof}
$\sigma(c)=\mathrm{Tr}(\Pc P_c\Pc)$;
Slepian--Weyl \cite{Slepian1961}:
$\sigma(c)\sim cc_0/\pi$.
\end{proof}

\begin{remark}[Counterexample and PSWF-incompatibility]
\label{rem:counterexample}
$\sigma(c)\to\infty$ does not imply
$S_N(c)\not\to 0$:
if $\mu_{n,c}([-c_0,c_0])=0$ for all $n>N_0$,
then $S_N(c)=0$ for $N\ge N_0$.
PSWF-incompatibility:
(i)~Slepian--Weyl requires an
unbounded set of contributing indices;
(ii)~Paper~VIII gives
$\mu_{n,c}([-c_0,c_0])>0$ for each
$n\le\kappa c$.
NC (Definition~\ref{def:NC})
formalises the required lower bound.
\end{remark}

%%═══ 5. SPECTRAL MEASURES AND NC
\section{Spectral Measures and No-Concentration}
\label{sec:NC}

\begin{remark}[NC as spectral evacuation failure]
\label{rem:NC-evac}
The obstruction to H1 stated cleanly:
\begin{quote}
  \emph{H1 fails if and only if
  there exists persistent spectral mass
  of the measures $\mu_{n,c}$
  in the transition zone $n\sim\alpha c$
  (conditional on OP~\ref{prob:H1H3}
  and OP~\ref{prob:fourier-mass}).}
\end{quote}
``Spectral evacuation'' means
$\mu_{n,c}([-c_0,c_0])\to 0$
for $n>N$, uniformly in $c$.
NC is the quantitative, band-level
obstruction to this;
H3 is the global tail condition.
Proposition~\ref{prop:NC-not-H3}
shows that NC is \emph{sufficient}
but \emph{not necessary} for H3 to fail,
even at the abstract measure-theoretic level.
\end{remark}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
Fix $0<\alpha<\kappa<1$.
NC holds if there exists $c_*>0$ such that
for all large $c$:
\begin{equation}\label{eq:NC}
  \sum_{n\in\mathcal{T}_{\alpha,\kappa}(c)}
  \mu_{n,c}([-c_0,c_0]) \ge c_*>0.
\end{equation}
\end{definition}

\begin{remark}[Why proportional width is necessary]
\label{rem:prop-width}
A window of width $c^\beta$ ($\beta<1$)
has $\sim 2c^\beta$ indices;
under the local Weyl law,
the sum is $\sim 2c_0 c^{\beta-1}\to 0$:
insufficient for NC.
A proportional band
$\mathcal{T}_{\alpha,\kappa}(c)$
has $\sim(\kappa-\alpha)c$ indices,
giving sum $\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[NC as quantum ergodicity]
\label{rem:NC-QE}
In the semiclassical limit $c\to\infty$,
$n\sim\alpha c$, the rescaled measures
$c\cdot\mu_{n,c}|_{[-c,c]}$
are expected to converge weakly to
the WKB density
$(\pi\sqrt{1-(\xi/c)^2})^{-1}d\xi$:
a one-dimensional quantum ergodicity
statement \cite{Zworski2012}.
NC is the consequence for $[-c_0,c_0]$:
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$,
so the band sum
$\sim c_0(\kappa-\alpha)>0$.
\end{remark}

\begin{remark}[Local Weyl law $\Rightarrow$ NC]
\label{rem:Weyl-NC}
If uniformly for $n\in\mathcal{T}_{\alpha,\kappa}(c)$:
\begin{equation}\label{eq:local-Weyl}
  \mu_{n,c}([-c_0,c_0])\sim\frac{c_0}{c},
\end{equation}
then $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
\sim c_0(\kappa-\alpha)>0$.
NC holds with $c_*=c_0(\kappa-\alpha)/2$.
\end{remark}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under NC: for each fixed $N$,
$\liminf_{c\to\infty}S_N(c)\ge c_*>0$,
so H3 fails.
\end{proposition}
\begin{proof}
For large $c$: $\alpha c>N$,
hence $\mathcal{T}_{\alpha,\kappa}(c)\subseteq\{n>N\}$
and $S_N(c)\ge c_*$.
\end{proof}

\begin{proposition}[NC$^c$ does not imply H3;
  logical separation]
\label{prop:NC-not-H3}
\emph{Scope: this is a logical separation result
for abstract probability measures satisfying
only normalisation $\mu_{n,c}(\mathbb{R})=1$
and positivity $\mu_{n,c}\ge 0$.
No claim is made that the model family
below arises from the PSWF spectral measures
$|\widehat{\Phi_n^{(c)}}|^2\,d\xi$.
The result establishes that NC and H3 are
logically independent at the abstract level:
NC is sufficient for H3 to fail
(Proposition~\ref{prop:barrier}),
but not necessary.}

\medskip\noindent
There exists a model family
$\{\mu_{n,c}\}_{n\ge 0,\,c>0}$
satisfying $\mu_{n,c}\ge 0$ and
$\mu_{n,c}(\mathbb{R})=1$,
such that\textup{:}
\begin{enumerate}[\rm(i)]
\item\label{it:NCc}
  \textup{NC fails (NC$^c$ holds)}:
  for every $0<\alpha'<\kappa'<1$
  and every $c_*>0$,
  \[
    \liminf_{c\to\infty}
    \sum_{n\in\mathcal{T}_{\alpha',\kappa'}(c)}
    \mu_{n,c}([-c_0,c_0]) < c_*;
  \]
\item\label{it:H3fail}
  \textup{H3 fails}: for every fixed $N\ge 0$,
  \[
    \liminf_{c\to\infty} S_N(c) \ge 1.
  \]
\end{enumerate}
In particular, NC is not necessary for H3
to fail: $\mathrm{NC}^c\not\Rightarrow\mathrm{H3}$,
even at the abstract level.
\end{proposition}

\begin{proof}
Fix $\alpha\in(0,1)$.
Set $w(c):=\lfloor\sqrt{c}\rfloor\ge 1$
and define the active zone
\[
  I_c := \bigl\{n\in\mathbb{N}_0 :
    |n-\lfloor\alpha c\rfloor|\le w(c)\bigr\},
  \qquad |I_c|=2w(c)+1.
\]
Assign weights
\begin{equation}\label{eq:model-weights}
  \mu_{n,c}([-c_0,c_0]) :=
  \begin{cases}
    \dfrac{1}{|I_c|} & n\in I_c,\\
    0 & n\notin I_c,
  \end{cases}
\end{equation}
and extend $\mu_{n,c}$ to a probability
measure on $\mathbb{R}$ by placing the
remaining mass at an arbitrary point
outside $[-c_0,c_0]$.

\medskip
\textit{Proof of~\ref{it:H3fail}.}
Fix $N\ge 0$.
For all $c>((N+1)/\alpha)^2$ we have
$\lfloor\alpha c\rfloor-w(c)
\ge \alpha c - 1 - \sqrt{c} > N$
(since $\alpha c - 1 - \sqrt{c}\to+\infty$),
so $I_c\subseteq\{n>N\}$.
Therefore
\[
  S_N(c)
  \ge\sum_{n\in I_c}\frac{1}{|I_c|}
  = 1.
\]

\medskip
\textit{Proof of~\ref{it:NCc}.}
Fix $0<\alpha'<\kappa'<1$ and $c_*>0$.

\textit{Case~1}: $\alpha\notin[\alpha',\kappa']$.
Then $\delta:=\mathrm{dist}(\alpha,[\alpha',\kappa'])>0$.
For $n\in I_c$:
$|n/c-\alpha|\le(w(c)+1)/c\le 2/\sqrt{c}$,
so for large $c$, $|n/c-\alpha|<\delta/2$,
giving $n/c\notin[\alpha',\kappa']$,
i.e.\ $I_c\cap\mathcal{T}_{\alpha',\kappa'}(c)=\emptyset$.
The band sum is $0<c_*$.

\textit{Case~2}: $\alpha\in(\alpha',\kappa')$.
Set $\varepsilon
:=\tfrac{1}{2}\min(\alpha-\alpha',\,\kappa'-\alpha)>0$.
For $n\in I_c$:
\[
  \left|\frac{n}{c}-\alpha\right|
  \le \frac{w(c)+1}{c}
  \le \frac{1}{\sqrt{c}}+\frac{2}{c}
  \to 0,
\]
so for large $c$, $|n/c-\alpha|<\varepsilon/2$
for all $n\in I_c$.
Choose $\beta$ with
$\alpha<\beta<\kappa'$ and $\beta-\alpha>\varepsilon$.
Then $n/c<\alpha+\varepsilon/2<\beta$
for all $n\in I_c$ and large $c$,
so $I_c\cap\mathcal{T}_{\beta,\kappa'}(c)=\emptyset$.
Hence
\[
  \sum_{n\in\mathcal{T}_{\beta,\kappa'}(c)}
  \mu_{n,c}([-c_0,c_0]) = 0 < c_*,
\]
and NC fails for the proportional band
$\mathcal{T}_{\beta,\kappa'}$.
\end{proof}

\begin{remark}[Structural separation
  and necessity]
\label{rem:separation}
Proposition~\ref{prop:NC-not-H3} shows that:
\begin{itemize}[leftmargin=*]
\item \emph{NC is a band-level condition}:
  persistent mass in a fixed proportional
  band $\mathcal{T}_{\alpha,\kappa}(c)$.
\item \emph{H3 is a global tail condition}:
  total tail mass $S_N(c)\to 0$.
\item \emph{NC is not necessary} for H3 to fail:
  the model achieves NC$^c$ by concentrating
  mass in a sub-proportional zone
  $|I_c|\sim 2\sqrt{c}\ll c$ that escapes
  every fixed proportional window,
  while maintaining $S_N(c)=1$ for all $N$.
\end{itemize}
The correct obstruction to H1 is H3,
not NC.
Whether the actual PSWF measures
$\mu_{n,c}=|\widehat{\Phi_n^{(c)}}|^2\,d\xi$
can exhibit NC$^c$-with-H3-fail behaviour
is addressed by OP~\ref{prob:fourier-mass}.
\end{remark}

%%═══ 6. H3 AND THE MAIN REDUCTION
\section{H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)
=\sup_{c>c_0}\mathrm{Tr}(\Ac)
\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{remark}[H3 vs.\ classical Slepian asymptotics]
\label{rem:H3-vs-Slepian}
H3 operates in a different asymptotic regime
from classical Slepian theory.
In the classical setting one fixes $c$ and
studies the eigenvalue distribution of
$\lambda_n^{(c)}$ for $n\sim 2c/\pi$
(the Landau--Pollak--Slepian transition count).
H3, by contrast, fixes $N$ and requires
$\sup_{c>c_0} S_N(c)\to 0$ as $N\to\infty$:
a \emph{uniform-in-$c$, large-$N$} condition.
This is strictly stronger than any
fixed-$c$ eigenvalue distribution result,
and is not implied by classical asymptotics.
H3 is a hypothesis whose role is
to serve as a sufficient condition for H1
(Theorem~\ref{thm:H3-H1}).
\end{remark}

\begin{theorem}[H3 $\Rightarrow$ H1]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$, $g\in\PW_{c_0}^2$.
H3: choose $N$ with
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$;
then $Q_c^N(g)<\varepsilon/2$ for all $c$.
Paper~XII: choose $c_\varepsilon$ with
$\sum_{n\le N}|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ for $c>c_\varepsilon$.
Adding gives $\norm{a^{(c)}-a^{(\infty)}}^2<\varepsilon$.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$
  $\Ac\to 0$ in SOT]
\label{thm:H1-H3-ptwise}
Assume H1. For each fixed $g\in\PW_{c_0}^2$:
\[
  \ip{\Ac g}{g} = Q_c^N(g) \to 0
  \quad\text{as }N\to\infty,
\]
i.e.\ $\Ac\to 0$ in the strong operator
topology (convergence of quadratic forms
on each fixed element).
\end{theorem}
\begin{proof}
$Q_c^N(g)
= \norm{a^{(c)}-a^{(\infty)}}^2
  - \sum_{n\le N}|a_n^{(c)}-a_n^{(\infty)}|^2
  + \sum_{n>N}|a_n^{(\infty)}|^2
\le 2\norm{a^{(c)}-a^{(\infty)}}^2
+2\sum_{n>N}|a_n^{(\infty)}|^2$.
H1 gives the first term $\to 0$ over $c$;
$a^{(\infty)}\in\ell^2$ gives the second $\to 0$
over $N$.
\end{proof}

\begin{remark}[The uniformity gap]
\label{rem:unif-gap}
\[
  \text{H3}\;(\mathrm{Tr}\to 0)
  \Rightarrow
  \norm{\Ac}_{\mathrm{op}}\to 0
  \Rightarrow
  \text{H1}
  \Rightarrow
  \ip{\Ac g}{g}\to 0\;\text{(SOT, each fixed }g\text{)}.
\]
The missing reverse is:
SOT $\not\Rightarrow$
$\norm{\Ac}_{\mathrm{op}}\to 0$
in general.
By Remark~\ref{rem:rank-obstruction},
rank control of $\Ac$ would close this gap;
by Lemma~\ref{lem:rank-lower},
the current lower bound is
$\mathrm{rank}(\Ac)\ge S_N(c)$.
\end{remark}

\begin{theorem}[Main reduction]
\label{thm:reduction}
\textup{(i)} Proved unconditionally:
\begin{align*}
  &\text{H3}\Rightarrow\text{H1},
  \quad
  \text{H1}\Rightarrow\Ac\to 0\;(\mathrm{SOT}),\\
  &\text{NC}\Rightarrow\text{H3 fails},
  \quad
  \text{local Weyl}\Rightarrow\text{NC},\\
  &\text{NC sufficient but not necessary
  for H3 to fail}
  \;(\text{Prop.}~\ref{prop:NC-not-H3}).
\end{align*}
\textup{(ii)} Open:
\[
  \mathrm{rank}(\Ac)\to\infty
  \;(\text{OP~\ref{prob:H1H3}}),
  \qquad
  \text{NC holds}
  \;(\text{OP~\ref{prob:fourier-mass}}).
\]
\textup{(iii)} Conditional chain:
\[
  \underbrace{\text{local Weyl law}}_{
    \text{OP~\ref{prob:fourier-mass}}}
  \Rightarrow
  \text{NC}
  \Rightarrow
  \text{H3 fails}
  \underset{\text{OP~\ref{prob:H1H3}}}
  {\Rightarrow}
  \text{H1 fails.}
\]
\end{theorem}

%%═══ 7. INTERPRETATION
\section{Interpretation and Context}
\label{sec:interp}

\subsection{The true obstruction: spectral non-evacuation}

\[
  \text{H1 fails}
  \;\Longleftrightarrow\;
  \text{spectral non-evacuation in }n\sim\alpha c
  \quad(\text{cond. on OPs~\ref{prob:H1H3},
  \ref{prob:fourier-mass}}).
\]
``Spectral evacuation'' means
for every $N$, the $\mu_{n,c}$-mass
of $\{n>N\}$ on $[-c_0,c_0]$
vanishes uniformly in $c$.
The model of Proposition~\ref{prop:NC-not-H3}
shows (at the abstract level) that mass can
migrate through sub-proportional bands
and prevent evacuation without any single
proportional band carrying persistent mass.
NC is sufficient but not necessary
for this failure mode.

\subsection{NC as band-level; H3 as global-tail}

\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{lll}
\toprule
Condition & Level & Role \\
\midrule
NC & Band-level (sufficient)
  & Persistent mass in $\mathcal{T}_{\alpha,\kappa}(c)$ \\
H3 & Global tail (necessary/sufficient)
  & Total mass $S_N(c)\to 0$ \\
\bottomrule
\end{tabular}
\end{center}
NC$\Rightarrow$H3 fails (proved);
NC is not necessary for H3 to fail
(Prop.~\ref{prop:NC-not-H3}).
The correct obstruction for H1 is H3.

\subsection{$S_N(c)$ as trace;
rank lower bound}

By Lemma~\ref{lem:rank-lower}
and~\eqref{eq:trace-op-rank}:
\[
  S_N(c)
  \le \mathrm{rank}(\Ac),
  \qquad
  \norm{\Ac}_{\mathrm{op}}
  \le S_N(c)
  \le \mathrm{rank}(\Ac)\cdot\norm{\Ac}_{\mathrm{op}}.
\]
H3 controls the trace from above;
the rank lower bound controls it from below.
Whether $\mathrm{rank}(\Ac)\to\infty$
is the open conjecture (OP~\ref{prob:H1H3}).

\subsection{Quantum ergodicity in frequency space}

The spectral measures $\mu_{n,c}$
play the role of semiclassical measures;
the local Weyl law (OP~\ref{prob:fourier-mass})
asserts their equidistribution on $[-c_0,c_0]$:
a one-dimensional quantum ergodicity
statement \cite{Zworski2012}.

\subsection{Why existing tools are insufficient}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Tool & Controls & Does not control \\
\midrule

  & $\sum_{n\le N}\mu_{n,c}\le N+1$
  & Tail; NC \\
Paper~XII
  & SOT: $\ip{\Ac g}{g}\to 0$ (fixed $g$)
  & $\norm{\Ac}_{\mathrm{op}}$; rank \\
Paper~VIII
  & $|\widehat{\Phi}|$ pointwise
  & $\mu_{n,c}([-c_0,c_0])$ integrated \\
Slepian--Weyl
  & $\sigma(c)=\mathrm{Tr}(\Ac)$ growth
  & Band distribution; rank \\

  & NC; spectral evacuation
  & (target of Paper~XV) \\
\bottomrule
\end{tabular}
\end{center}

%%═══ 8. PROGRAMME FOR PAPER XV
\section{Programme for Paper~XV}
\label{sec:PaperXV}

\subsection{The three required estimates}

\begin{enumerate}[\rm(A)]
\item\label{XV-A} \textbf{Upper bound, uniform:}
  $\mu_{n,c}([-c_0,c_0])\le C/c$
  for all $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-B} \textbf{Lower bound, uniform:}
  $\mu_{n,c}([-c_0,c_0])\ge 1/(Cc)$
  for all $n\in\mathcal{T}_{\alpha,\kappa}(c)$.
\item\label{XV-C} \textbf{Summability:}
  $\mu_{n,c}([-c_0,c_0])=c_0/c+o(1/c)$
  uniformly, so
  $\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])
  \to c_0(\kappa-\alpha)$.
\end{enumerate}
(A)+(B) $\Rightarrow$ NC;
(C) $\Rightarrow$ equidistribution.

\subsection{Semiclassical approach}

$h:=1/c\to 0$; the PSWF solves
\begin{equation}\label{eq:PSWF-ODE}
  -(1-x^2)\partial_x^2\psi
  +2x\partial_x\psi
  +c^2x^2\psi=\chi_n^{(c)}\psi
\end{equation}
with energy $E_n^{(h)}=\chi_n^{(c)}/c^2
\to E_\alpha$ for $n\sim\alpha c$.
WKB in the classical region $|x|<x_0$
($x_0^2=E_\alpha$) and stationary phase
for $\xi\in[-c_0,c_0]$ give
$\mu_{n,c}([-c_0,c_0])\sim 2c_0C(E_\alpha)/c$,
yielding (A)--(C).

\subsection{Rigorous programme}

\begin{enumerate}
\item WKB with remainder $O(c^{-2})$,
  uniform in $n\in\mathcal{T}$ \cite{Osipov2013}.
\item Stationary phase with uniform remainder.
\item Prove (A)--(C); conclude NC.
\item Address OP~\ref{prob:H1H3}
  (rank growth via bandlimit geometry).
\end{enumerate}

%%═══ 9. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law; Paper~XV]
\label{prob:fourier-mass}
\emph{(Priority: critical.)}
Prove (A)--(C) of \S\ref{sec:PaperXV}:
$\mu_{n,c}([-c_0,c_0])\sim c_0/c$
uniformly in $n\in\mathcal{T}_{\alpha,\kappa}(c)$.

\medskip\noindent
\emph{Additional difficulty
(orthogonality and band-sum correlations):}
The $\Phi_n^{(c)}$ are globally orthogonal
in $L^2[-1,1]$ but not in
$L^2[-c_0/c,\,c_0/c]$ after rescaling,
introducing off-diagonal correlations in
$\sum_{n\in\mathcal{T}}\mu_{n,c}([-c_0,c_0])$.
Paper~XV must control both individual
$\mu_{n,c}$ asymptotics via WKB
and the correlation decay of PSWF overlaps
in the transition zone.
\end{problem}

\begin{problem}[Rank growth conjecture]
\label{prob:H1H3}
\emph{(Priority: high.)}
Lemma~\ref{lem:rank-lower} gives
$\mathrm{rank}(\Ac)\ge S_N(c)$;
under NC, this is uniformly positive.
The current bounds do not imply
$\mathrm{rank}(\Ac)\to\infty$.
Conjecture: the bandlimit geometry
(nested projections,
Paley--Wiener structure) forces
$\mathrm{rank}(\Ac)\to\infty$ as $c\to\infty$
(for fixed $N$).
If true, combined with
$\mathrm{Tr}(\Ac)=S_N(c)$
and~\eqref{eq:trace-op-rank},
this would force $\norm{\Ac}_{\mathrm{op}}\to 0$,
closing the SOT-to-op-norm gap
and completing H1.
\end{problem}

\begin{problem}[NC from Papers~VIII, XII]
\label{prob:NC-direct}
\emph{(Priority: medium.)}
Do the pointwise bounds of Paper~VIII
imply lower bounds on
$\mu_{n,c}([-c_0,c_0])$
without full WKB?
\end{problem}

\emph{The obstruction to coefficient stability
is spectral non-evacuation:
the persistence of $\mu_{n,c}$-mass
in the transition zone $n\sim\alpha c$.
The operator $\Ac=\Bn^*\Bn$ satisfies
$0\le\Ac\le I$ (proved),
