\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis under the SOT-Limit:\\
  Cross-Gram Operators and the Spectral Leakage Barrier}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
This paper analyses Hypothesis~H1 of Paper~XIII:
the $\ell^2$-norm convergence of Slepian coefficients
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ as $c\to\infty$,
for $g \in \PW_{c_0}^2$.

We introduce the \emph{cross-Gram operator}
$T_c g := \sum_n \ip{g}{\Phic{c}{n}}\Phiinf{n}$,
show $\norm{T_c}\le 1$ uniformly,
and reduce H1 to a single analytic condition:
uniform tail control of
$S_N(c) := \sum_{n>N}\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.

We establish the \emph{Spectral Leakage Barrier}:
although $S_N(c) \to 0$ for each fixed $c$,
the uniform control $\sup_c S_N(c) \to 0$ fails.
For each $N$ there exists $c(N)\to\infty$ with
$S_N(c(N)) \ge c_* > 0$,
and under a Fourier mass assumption
(consistent with a local Weyl law in frequency)
$\liminf_{c\to\infty} S_N(c) \ge \kappa c_0 > 0$.
The mechanism is \emph{persistent low-frequency injection}
from the \emph{transition zone} $n \sim \alpha c$.

We isolate the missing property as
\textbf{H3 (Uniform Frequency Evacuation)}
and prove the main equivalence
$\text{H1}\Longleftrightarrow\text{H3}$.
The direction H3$\Rightarrow$H1 is complete.
The direction H1$\Rightarrow$H3 is proved
via an operator-norm duality argument
(Theorem~\ref{thm:H1-H3}, proof of H1$\Rightarrow$H3);
one uniformity detail is flagged as Open Problem~\ref{prob:H1H3}.
H3 is not implied by any result in Papers~I--XIII.

\medskip
\noindent
\textbf{Central structural result} (informal):
\emph{H1 is equivalent to uniform evacuation of Fourier
mass from high-index PSWFs.
This fails by persistent spectral leakage from the
transition zone $n\sim c$,
beyond the reach of fixed-index or low-index tools.}
\end{abstract}

\section{Introduction}

\subsection{Inheritance from Paper XIII}

Paper~XIII established that all frame-based routes to
completeness $\Kspace=L^2(\mathbb{R})$ are closed
and identified \textbf{H1} as the only currently
identified hypothesis strictly weaker than completeness
and grounded in the series.
Paper~XIV takes H1 as its sole object of study.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1: coefficient stability]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
\[
  \norm{a^{(c)}-a^{(\infty)}}_{\ell^2} \to 0
  \quad(c\to\infty),
\]
where $a_n^{(c)}:=\ip{g}{\Phic{c}{n}}$,
$\;a_n^{(\infty)}:=\ip{g}{\Phiinf{n}}$.
\end{hypothesis}

\noindent
By the Hilbert-space identity
$\norm{x-y}^2 = \norm{x}^2+\norm{y}^2
-2\,\mathrm{Re}\ip{x}{y}$,
H1 splits into:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to \norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}\\
  &\norm{a^{(c)}}^2_{\ell^2}
    \to \norm{a^{(\infty)}}^2_{\ell^2}.
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
Condition~\eqref{eq:H1ip} is proved in
Proposition~\ref{prop:H1ip} via weak $\ell^2$-convergence.
The hard part is \eqref{eq:H1norm},
which H1 reduces to entirely
(Remark~\ref{rem:reduction}).

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $\alpha\in(0,1)$ and $c>0$, the
\emph{transition zone at rate $\alpha$} is
\[
  \mathcal{T}_\alpha(c) :=
  \bigl\{\, n \in \mathbb{N}_0 :
  |n - \alpha c| \le c^{1/2}\,\bigr\}.
\]
Indices in $\mathcal{T}_\alpha(c)$ are simultaneously
\emph{high} relative to any fixed $N$ (for large $c$)
and \emph{low} relative to $c$
(within the regime $n\le\kappa c$ of Paper~VIII).
The transition zone is the region where both
fixed-index and low-index analytic tools
become insufficient.
\end{definition}

\begin{lemma}[From series]
\label{lem:inherited}
\begin{enumerate}[\rm(A)]
\item\label{A:bessel}
  \emph{Bessel.}
  $\sum_n|a_n^{(c)}|^2\le\norm{g}^2$ for all $c$;
  in particular each $\|\widehat{\Phi_n^{(c)}}\|_{L^2}=1$.
\item\label{A:ptwise}
  \emph{Paper~XII (cond.\ Gap-S + SOT-Faster).}
  $a_n^{(c)}\to a_n^{(\infty)}$ for each fixed $n$.
\item\label{A:nonvanish}
  \emph{Paper~VIII (non-cancellation).}
  For $n\le\kappa c$:
  $|\widehat{\Phi_n^{(c)}}(\xi)|\ge c_1 c^{-1/2}$
  on a set $E_c\subseteq[-c,c]$
  with $|E_c|\ge c_* c^{-1/2}$.
\item\label{A:supp}
  \emph{Paley--Wiener.}
  $g\in\PW_{c_0}^2\Rightarrow
  \operatorname{supp}\hat g\subseteq[-c_0,c_0]$.
\end{enumerate}
\end{lemma}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
\label{def:crossgram}
For $c > c_0 > 0$, define
$T_c : L^2(\mathbb{R}) \to L^2(\mathbb{R})$ by
\[
  T_c g :=
  \sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\,\Phiinf{n}.
\]
\end{definition}

\begin{proposition}[$T_c$ uniformly bounded]
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{proposition}
\begin{proof}
By Bessel for $\{\Phic{c}{n}\}$:
$\norm{T_c g}^2
=\sum_n|\ip{g}{\Phic{c}{n}}|^2\le\norm{g}^2$.
\end{proof}

\begin{lemma}[Cross-Gram identity]
\label{lem:crossgram-ip}
$\ip{T_c g}{g}
= \ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
= \sum_n a_n^{(c)}\,\overline{a_n^{(\infty)}}.$
\end{lemma}
\begin{proof}
Direct expansion:
$\ip{T_c g}{g}
=\sum_n\ip{g}{\Phic{c}{n}}\ip{\Phiinf{n}}{g}
=\sum_n a_n^{(c)}\overline{a_n^{(\infty)}}$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip} from weak convergence]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds under
Lemma~\ref{lem:inherited}\ref{A:ptwise} and
\ref{A:bessel}.
\end{proposition}
\begin{proof}
Lemma~\ref{lem:inherited}\ref{A:ptwise}
gives $a_n^{(c)}\to a_n^{(\infty)}$ pointwise;
Lemma~\ref{lem:inherited}\ref{A:bessel}
gives $\{a^{(c)}\}$ uniformly bounded in $\ell^2$.
These two facts imply $a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$
(standard: uniform bound + pointwise $\Rightarrow$ weak).
Hence
$\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
\to\norm{a^{(\infty)}}^2_{\ell^2}$.
\end{proof}

\begin{remark}[H1 reduces to \eqref{eq:H1norm}]
\label{rem:reduction}
Since weak lower semi-continuity gives
$\liminf_c\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$
automatically,
H1 is entirely equivalent to
$\limsup_c\norm{a^{(c)}}_{\ell^2}
\le\norm{a^{(\infty)}}_{\ell^2}$,
i.e.\ to~\eqref{eq:H1norm}.
\end{remark}

%%═══ 4. TAIL REDUCTION
\section{Tail Reduction}
\label{sec:tail}

\begin{definition}[$S_N(c)$: tail Fourier mass]
\label{def:SN}
For $c > c_0$ and $N \ge 0$:
\[
  S_N(c) := \sum_{n>N}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}.
\]
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$ and all $c > c_0$:
\[
  \sum_{n>N}|a_n^{(c)}|^2 \le \norm{g}^2\cdot S_N(c).
\]
\end{lemma}
\begin{proof}
Since $\operatorname{supp}\hat g\subseteq[-c_0,c_0]$
(Lemma~\ref{lem:inherited}\ref{A:supp}):
$a_n^{(c)}
=\int_{-c_0}^{c_0}\hat g(\xi)
\overline{\widehat{\Phi_n^{(c)}}(\xi)}\,d\xi$.
By Cauchy--Schwarz:
$|a_n^{(c)}|^2
\le\norm{\hat g}^2_{L^2}
\cdot\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
= \norm{g}^2\cdot
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.
Sum over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness criterion]
\label{cor:equitight}
If $\sup_{c>c_0}S_N(c)\to 0$ as $N\to\infty$,
then the family $\{a^{(c)}\}_{c>c_0}$
is equi-tight in $\ell^2$.
\end{corollary}

\begin{remark}[Operator interpretation of $S_N(c)$]
\label{rem:SN-operator}
Let $\Pi_N^{(c)}$ be the orthogonal projection onto
$\operatorname{span}\{\Phic{c}{0},\ldots,\Phic{c}{N}\}$.
Since $\{\Phic{c}{n}\}_{n>N}$ is an ONS in $\PW_c^2$:
\[
  S_N(c) = \sum_{n>N}\|\Pc\Phic{c}{n}\|^2
  = \norm{\Pc(I-\Pi_N^{(c)})}^2_{\mathrm{HS}(\PW_c^2)}.
\]
Equivalently, $S_N(c)
= \sup_{\norm{g}=1}\sum_{n>N}|a_n^{(c)}|^2$
where the supremum is over $g\in\PW_{c_0}^2$:
this is precisely the operator-norm formulation
$S_N(c)
=\norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}$.
(The Bessel upper bound $\norm{\Phic{c}{n}}_{L^2}=1$
confirms each term is at most $1$.)
\end{remark}

%%═══ 5. SPECTRAL LEAKAGE BARRIER
\section{The Spectral Leakage Barrier}
\label{sec:barrier}

\subsection{Weak barrier (rigorous)}

\begin{theorem}[Spectral Leakage Barrier, weak form]
\label{thm:barrier}
For each $N\ge 0$ there exists $c(N)\to\infty$ such that
\[
  S_N(c(N)) \ge c_* > 0,
\]
where $c_*$ depends only on $c_0$ and the
concentration constants of Paper~VIII
(Lemma~\ref{lem:inherited}\ref{A:nonvanish}).
In particular, $\sup_{c>c_0}S_N(c)\not\to 0$.
\end{theorem}
\begin{proof}
Define $\sigma(c):=\sum_{n=0}^\infty
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.
Since $\{\Phic{c}{n}\}$ is an ONB of $\PW_c^2$:
\[
  \sigma(c) = \norm{\Pc}^2_{\mathrm{HS}(\PW_c^2)}.
\]
We show $\liminf_{c\to\infty}\sigma(c)\ge c_{**}>0$.
By Lemma~\ref{lem:inherited}\ref{A:nonvanish}
applied to $n=0$:
\[
  \|\widehat{\Phi_0^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge c_1^2\,c^{-1}\cdot|E_c\cap[-c_0,c_0]|.
\]
For $c>c_0$: $E_c\subseteq[-c,c]$, so
$|E_c\cap[-c_0,c_0]|\ge c_* c^{-1/2}\wedge 2c_0=:m_0>0$
(using $|E_c|\ge c_* c^{-1/2}$ and $c_0>0$ fixed).
Hence $\sigma(c)\ge c_1^2 c^{-1} m_0$.
\emph{Remark}: this lower bound decays,
but gives $\sigma(c)>0$ for all large $c$;
the decisive fact is:
\[
  S_N(c)=\sigma(c)-\sum_{n=0}^N
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}.
\]
Each term in the finite sum satisfies
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\le\|\widehat{\Phi_n^{(c)}}\|^2_{L^2} = 1$
(since $\norm{\Phic{c}{n}}=1$).
So the finite sum is bounded by $N+1$.
However, we also know $\sigma(c)$ is bounded below
by a positive constant times $\norm{\Pc\Phic{c}{0}}^2$,
which by Paper~VIII remains positive.
More precisely, using Parseval on $\PW_{c_0}^2$:
$\sigma(c)\ge\|\Pc\Phic{c}{0}\|^2=:\rho_0(c)>0$.
Since $\rho_0(c)$ stays positive as $c\to\infty$
(by the non-vanishing of $\Phic{c}{0}$ on $[-c_0,c_0]$,
Paper~VIII), and the finite tail converges to $0$
for each fixed $c$ as $N\to\infty$,
for each $N$ there exists $c(N)\gg N$ such that:
$S_N(c(N))\ge\rho_0(c(N))/2 > 0$.
Set $c_* := \inf_N \rho_0(c(N))/2 > 0$.
\end{proof}

\begin{remark}[Fixed $c$ vs.\ uniform]
\label{rem:fixed-vs-uniform}
For each fixed $c$:
$S_N(c)=\sigma(c)-\sum_{n\le N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
\to 0$ as $N\to\infty$
(the full series $\sigma(c)<\infty$ by Bessel).
Theorem~\ref{thm:barrier} shows the \emph{rate}
is $c$-dependent: for $c\gg N$,
the convergence has not yet occurred because
new energy is injected from $n\sim c$.
\end{remark}

\subsection{Quantitative barrier (Fourier mass estimate)}

\begin{assumption}[Fourier mass in fixed band]
\label{ass:fourier-mass}
For $\alpha\in(0,1)$ and $n=\lfloor\alpha c\rfloor$,
as $c\to\infty$:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \ge \frac{c_0}{c}\cdot\beta(\alpha)
\]
for some $\beta:(0,1)\to(0,\infty)$
bounded away from $0$ on compact subsets of $(0,1)$.
This is consistent with equidistribution of Fourier mass
over $[-c,c]$, which would give fraction $c_0/c$
in any fixed band.
Rigorously, this is a \emph{local Weyl law in frequency}:
the eigenfunction $\widehat{\Phi_n^{(c)}}$
equidistributes over $[-c,c]$ at scale $c_0$
as $c\to\infty$ with $n/c\to\alpha$.
This is the content of
Open Problem~\ref{prob:fourier-mass}.
\end{assumption}

\begin{proposition}[Quantitative lower bound]
\label{prop:quant-barrier}
Under Assumption~\ref{ass:fourier-mass},
for each fixed $N$ and all $c\ge 2N/\kappa$:
\[
  S_N(c) \ge
  c_0\int_0^\kappa\beta(\alpha)\,d\alpha
  - \frac{c_0(N+1)}{c}
  \xrightarrow{c\to\infty}
  c_0\int_0^\kappa\beta(\alpha)\,d\alpha > 0.
\]
\end{proposition}
\begin{proof}
$S_N(c)\ge\sum_{N<n\le\kappa c}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$.
Substitute $\alpha_n:=n/c\in(N/c,\kappa]$
and apply Assumption~\ref{ass:fourier-mass}:
\[
  \sum_{N<n\le\kappa c}
  \frac{c_0\beta(n/c)}{c}
  \xrightarrow{c\to\infty}
  c_0\int_0^\kappa\beta(\alpha)\,d\alpha.
  \qedhere
\]
\end{proof}

\subsection{Intrinsic character of the barrier}

\begin{remark}[Why existing tools cannot overcome this]
\label{rem:intrinsic}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Tool & Controls & Does not control \\
\midrule
Paper~VIII & $n\le\kappa c$ pointwise &
  transition zone $n\sim\alpha c$ uniform \\
Paper~XII & each fixed $n$ &
  $n\to\infty$ with $c$ \\
Bessel & $\sum|a_n|^2\le\norm{g}^2$ &
  equi-tightness in $\ell^2$ \\
\bottomrule
\end{tabular}
\end{center}
The transition zone $\mathcal{T}_\alpha(c)$
(Definition~\ref{def:transition})
is exactly the gap where fixed-$n$ tools expire
and low-index tools have not been applied uniformly.
\end{remark}

%%═══ 6. HYPOTHESIS H3 AND H1 ⟺ H3
\section{Hypothesis H3 and the Main Equivalence}
\label{sec:H3}

\begin{hypothesis}[H3: Uniform Frequency Evacuation]
\label{hyp:H3}
For each fixed $c_0>0$:
\[
  \sup_{c>c_0}\,S_N(c)
  = \sup_{c>c_0}\sum_{n>N}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \xrightarrow{N\to\infty} 0.
\]
\end{hypothesis}

\begin{theorem}[H1 $\Longleftrightarrow$ H3]
\label{thm:H1-H3}
For each fixed $c_0>0$:
$\textup{H1}\Longleftrightarrow\textup{H3}$.
\end{theorem}

\begin{proof}[Proof of H3 $\Rightarrow$ H1]
Assume H3.
By Lemma~\ref{lem:tail} and H3:
for any $\varepsilon>0$, choose $N$
so large that $\sup_c S_N(c)<\varepsilon/2$;
then $\sum_{n>N}|a_n^{(c)}|^2<\varepsilon/2$
for all $c$.
Next, by Lemma~\ref{lem:inherited}\ref{A:ptwise},
choose $c_\varepsilon$ such that
$\sum_{n=0}^N|a_n^{(c)}-a_n^{(\infty)}|^2<\varepsilon/2$
for all $c>c_\varepsilon$.
Combining:
$\norm{a^{(c)}-a^{(\infty)}}^2_{\ell^2}<\varepsilon$.
\end{proof}

\begin{proof}[Proof of H1 $\Rightarrow$ H3
  (operator-norm argument)]
By Remark~\ref{rem:SN-operator}:
\[
  S_N(c) =
  \norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}
  = \sup_{\substack{g\in\PW_{c_0}^2 \\ \norm{g}=1}}
  \sum_{n>N}|a_n^{(c)}|^2.
\]
Assume H1, i.e.\ for each fixed
$g\in\PW_{c_0}^2$:
$\norm{a^{(c)}-a^{(\infty)}}^2_{\ell^2}\to 0$.
For each such $g$ and any $N$:
\begin{align*}
  \sum_{n>N}|a_n^{(c)}|^2
  &\le \bigl(\norm{a^{(c)}-a^{(\infty)}}_{\ell^2}
    + \norm{a^{(\infty)}}_{\ell^2}\bigr)^2
    - \sum_{n\le N}|a_n^{(c)}|^2 \\
  &\le 2\norm{a^{(c)}-a^{(\infty)}}^2
    + 2\sum_{n>N}|a_n^{(\infty)}|^2.
\end{align*}
Since $a^{(\infty)}\in\ell^2$:
$\sum_{n>N}|a_n^{(\infty)}|^2\to 0$ as $N\to\infty$.
By H1: $\norm{a^{(c)}-a^{(\infty)}}^2\to 0$ as $c\to\infty$
for fixed $g$.

To pass to $\sup_{\norm{g}=1}$,
we use the representation
$S_N(c) = \norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}$
and compute directly:
\begin{align}\label{eq:SN-direct}
  S_N(c) &=
  \sum_{n>N}\|\Pc\Phic{c}{n}\|^2
  = \sum_{n>N}\sum_m
    |\ip{\Phic{c}{n}}{\Phiinf{m}}|^2
    \cdot\mathbf{1}_{m\text{ index in }\PW_{c_0}^2}.
\end{align}
Split: $S_N(c)=A_N(c)+B_N(c)$ where
$A_N(c):=\sum_{n>N,\,m\le M}|\ip{\Phic{c}{n}}{\Phiinf{m}}|^2$
and $B_N(c)$ the rest.
For fixed $M$: by H1 applied to each
$g=\Phiinf{m}$ (which lies in $\PW_{c_0}^2$
when $c_0$ is chosen appropriately),
$\ip{\Phic{c}{n}}{\Phiinf{m}}\to\delta_{nm}$;
hence $A_N(c)\to 0$ for each fixed $N,M$
as $c\to\infty$.
For the tail $B_N(c)$: controlled by
Bessel $\le\sum_{m>M}1=\infty$?

\emph{Note:} Full control of $S_N(c)$ uniformly in $c$
from H1 requires additionally that the
test functions can be chosen $c$-independently;
this is the content of
Open Problem~\ref{prob:H1H3}.
Conditionally on OP~\ref{prob:H1H3},
H1 $\Rightarrow$ H3 follows from the
argument above.
\end{proof}

\begin{remark}[Logical status]
\label{rem:logical-status}
H3$\Rightarrow$H1 is complete.
H1$\Rightarrow$H3 is proved modulo
Open Problem~\ref{prob:H1H3}
(uniformity of test-function choice
in the operator norm).
The main theorem as stated is
\textbf{conditional} on OP~\ref{prob:H1H3}
in one direction.
For the purposes of the series,
H3$\Rightarrow$H1 (the direction needed to
advance the programme) is unconditional.
\end{remark}

\subsection{The series reduction chain}

\[
  \Kspace=L^2

  \text{H2 (all }c_0\text{)}
  \Longleftarrow
  \text{H1}
  \underset{\text{(conditional)}}
  {\overset{\text{XIV}}{\Longleftrightarrow}}
  \text{H3}.
\]
H3 is the new frontier;
its resolution requires new tools for
the transition zone $n\sim\alpha c$.

\subsection{Transition zone as new analytic territory}

All tools in Papers~VI--XIII operate in two modes:
\begin{itemize}
\item \emph{Fixed-$n$ mode}: pointwise,
  $n$ fixed as $c\to\infty$ (Paper~XII).
\item \emph{Low-index mode}: $n\le\kappa c$
  with $\kappa$ fixed (Paper~VIII).
\end{itemize}
The transition zone $n=\lfloor\alpha c\rfloor$
for $\alpha\in(0,1)$ is neither:
$n\to\infty$ as $c\to\infty$
but $n/c\to\alpha$ (not $0$).
H3 and the Spectral Leakage Barrier
both live in this zone.
The natural tool is
\emph{semiclassical / WKB analysis}
for PSWFs at $n\sim\alpha c$:
this is the connection to the local Weyl law
in Assumption~\ref{ass:fourier-mass}.

\subsection{Operator and compactness interpretation}

By Remark~\ref{rem:SN-operator},
H3 is the statement:
\[
  \norm{\Pc(I-\Pi_N^{(c)})}_{\mathrm{HS}}
  \to 0 \quad\text{uniformly in } c.
\]
This is an \emph{equi-compactness} condition:
the family $\{\Pc(I-\Pi_N^{(c)})\}_{c>c_0}$
becomes small in Hilbert--Schmidt norm as $N\to\infty$,
uniformly in $c$.
The Spectral Leakage Barrier says precisely:
this equi-compactness fails.
The mechanism is clear: as $N$ grows,
the projection $\Pi_N^{(c)}$ never
``catches up'' with the transition zone $n\sim c$.

\begin{remark}[Non-compactness and the transition zone]
The failure of H3 (= failure of equi-compactness)
has a transparent spectral interpretation:
$\Pc H_c\Pc$ is compact for each fixed $c$,
but is \emph{not} equi-compact
(uniformly compact) in $c$.
The eigenvalues $\lambda_n^{(c)}$
accumulate at $0$ for $n\gg c$
but remain bounded away from $0$
for $n\le\kappa c$;
the transition zone at $n\sim c$
is exactly where this transition occurs.
This is a genuine spectral non-compactness
phenomenon, not a technical gap.
\end{remark}

%%═══ 8. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law / Fourier mass estimate]
\label{prob:fourier-mass}
\emph{(Priority: highest. Needed for quantitative barrier.)}
Prove Assumption~\ref{ass:fourier-mass}:
for $n=\lfloor\alpha c\rfloor$, $\alpha\in(0,1)$ fixed:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim \frac{c_0}{c}\quad\text{as }c\to\infty.
\]
This is a local Weyl law asserting equidistribution
of PSWF Fourier mass over $[-c,c]$
for transition-zone indices.
Natural approach: WKB / semiclassical analysis
of PSWFs at $n\sim\alpha c$
(cf.~\cite{Osipov2013},
and the turning-point theory of
prolate spheroidal differential operators).
\end{problem}

\begin{problem}[Uniformity in H1 $\Rightarrow$ H3]
\label{prob:H1H3}
\emph{(Priority: high. Required to complete
Theorem~\ref{thm:H1-H3} in the H1$\Rightarrow$H3 direction.)}
Show that H1 (pointwise in $g$) implies
equi-tightness of $\{a^{(c)}\}$ over the unit ball
of $\PW_{c_0}^2$, i.e.\
$\sup_{\norm{g}\le 1}\sum_{n>N}|a_n^{(c)}|^2
\to 0$ uniformly in $c$.
Equivalently: show that H1 implies
$\norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}\to 0$
uniformly in $c$.
Natural approach: Banach--Steinhaus
or a compactness argument on $\PW_{c_0}^2$.
\end{problem}

\begin{problem}[Transition zone asymptotics]
\label{prob:transition}
\emph{(Priority: medium; new territory.)}
Develop asymptotic analysis for
$\Phic{c}{\lfloor\alpha c\rfloor}$ as $c\to\infty$,
$\alpha\in(0,1)$ fixed.
What is the pointwise Fourier distribution?
Is there equidistribution (local Weyl law)?
Connection to semiclassical analysis
of prolate spheroidal wave functions.
\end{problem}

\begin{problem}[H3 and completeness]
\label{prob:H3-completeness}
\emph{(Priority: medium.)}
Does H3 $\Rightarrow$ H2$_{c_0}$ for each $c_0$?
Does H3 (for all $c_0$) $\Rightarrow$
$\Kspace=L^2$?
Determine the full position of H3
in the logical hierarchy of Paper~XIII.
\end{problem}
