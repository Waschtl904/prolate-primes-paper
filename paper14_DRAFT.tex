\title[Coefficient Stability of the Slepian Eigenbasis]{%
  Coefficient Stability of the Slepian Eigenbasis under the SOT-Limit:\\
  Cross-Gram Operators, Tail Reduction, and the Spectral Distribution Problem}

\author{[Author]}
\address{[Address]}
\email{[Email]}

\begin{abstract}
This paper analyses Hypothesis~H1 of Paper~XIII:
the $\ell^2$-norm convergence of Slepian coefficients
$a_n^{(c)} = \ip{g}{\Phic{c}{n}}$ as $c\to\infty$,
for $g \in \PW_{c_0}^2$.
We do not prove H1, but reduce it precisely
to a single spectral distribution problem.

We introduce the \emph{cross-Gram operator}
$T_c g := \sum_n \ip{g}{\Phic{c}{n}}\Phiinf{n}$,
show $\norm{T_c}\le 1$ uniformly in $c$,
and prove:
\[
  \text{H1} \iff
  \sup_{c>c_0} S_N(c) \xrightarrow{N\to\infty} 0,
\]
where $S_N(c):=\sum_{n>N}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$
(the \emph{tail functional}).
The right-hand condition is Hypothesis~H3
(Uniform Frequency Evacuation).
H3$\Rightarrow$H1 is proved unconditionally;
H1$\Rightarrow$H3 is conditional
on Open Problem~\ref{prob:H1H3}.

We identify the \emph{Spectral Leakage Barrier}:
the classical Slepian--Weyl law gives
$\sigma(c):=\sum_n
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
=\mathrm{Tr}(P_c P_{c_0}P_c)\sim cc_0/\pi\to\infty$,
so the total Fourier mass grows without bound.
However, $\sigma(c)\to\infty$ alone does
\emph{not} imply H3 fails:
the mass could, in principle, concentrate
into a fixed finite set of indices,
leaving $S_N(c)=0$ for all large $c$.
We prove this concentration is prevented if
and only if a \emph{no-concentration condition}
(NC, Assumption~\ref{ass:NC}) holds,
which we show is equivalent to
a \emph{local Weyl law} for PSWFs
in the transition zone $n\sim\alpha c$
(Open Problem~\ref{prob:fourier-mass}).

\medskip\noindent
\textbf{Central result} (informal):
\emph{H1 reduces to NC.
NC reduces to a local Weyl law for PSWFs
in the transition zone.
This is the natural and precise entry point
for Paper~XV.}
\end{abstract}

%%═══ 1. INTRODUCTION
\section{Introduction}
\label{sec:intro}

\subsection{Context and scope}

Paper~XIII identified \textbf{H1}
(coefficient stability of the Slepian eigenbasis)
as the key remaining hypothesis for
completeness $\Kspace=L^2(\mathbb{R})$.
Paper~XIV does not resolve H1,
but reduces it precisely to a new problem
in semiclassical spectral theory.
All unconditionally proved results are
clearly labelled; all conditional results
state their assumptions explicitly.

\subsection{H1: the norm-convergence problem}

\begin{hypothesis}[H1: coefficient stability]
\label{hyp:H1}
For each $c_0>0$ and $g\in\PW_{c_0}^2$:
\[
  \norm{a^{(c)}-a^{(\infty)}}_{\ell^2} \to 0
  \quad(c\to\infty),
\]
where $a_n^{(c)}:=\ip{g}{\Phic{c}{n}}$,
$a_n^{(\infty)}:=\ip{g}{\Phiinf{n}}$.
\end{hypothesis}

H1 splits into the inner-product condition:
\begin{align}
  &\mathrm{Re}\ip{a^{(c)}}{a^{(\infty)}}_{\ell^2}
    \to\norm{a^{(\infty)}}^2
    \tag{H1-ip}\label{eq:H1ip}
\end{align}
(proved in Proposition~\ref{prop:H1ip})
and the norm condition:
\begin{align}
  &\norm{a^{(c)}}^2_{\ell^2}
    \to\norm{a^{(\infty)}}^2_{\ell^2}
    \tag{H1-norm}\label{eq:H1norm}
\end{align}
(the hard part; equivalent to H3 below).

\subsection{The transition zone}

\begin{definition}[Transition zone]
\label{def:transition}
For $\alpha\in(0,1)$:
$\mathcal{T}_\alpha(c):=
\{n\in\mathbb{N}_0 : |n-\alpha c|\le c^{1/2}\}$.
These are the indices that are high relative
to any fixed $N$, yet low relative to $c$.
\end{definition}

%%═══ 3. CROSS-GRAM OPERATOR
\section{The Cross-Gram Operator}
\label{sec:crossgram}

\begin{definition}[Cross-Gram operator]
\label{def:crossgram}
$T_c g:=\sum_{n\ge 0}\ip{g}{\Phic{c}{n}}\Phiinf{n}.$
\end{definition}

\begin{proposition}
\label{prop:Tc-bounded}
$\norm{T_c}\le 1$ uniformly in $c$.
\end{proposition}
\begin{proof}
Bessel: $\norm{T_c g}^2
=\sum_n|\ip{g}{\Phic{c}{n}}|^2\le\norm{g}^2$.
\end{proof}

\begin{proposition}[\eqref{eq:H1ip}]
\label{prop:H1ip}
Condition~\eqref{eq:H1ip} holds.
\end{proposition}
\begin{proof}
Bessel + Paper~XII give
$a^{(c)}\rightharpoonup a^{(\infty)}$
weakly in $\ell^2$,
hence
$\ip{a^{(c)}}{a^{(\infty)}}\to\norm{a^{(\infty)}}^2$.
\end{proof}

\begin{remark}[H1 reduces to \eqref{eq:H1norm}]
\label{rem:reduction}
Weak lower semi-continuity gives
$\liminf\norm{a^{(c)}}\ge\norm{a^{(\infty)}}$
automatically.
So H1 $\Leftrightarrow$
$\limsup_c\norm{a^{(c)}}_{\ell^2}
\le\norm{a^{(\infty)}}_{\ell^2}$.
\end{remark}

%%═══ 4. TAIL REDUCTION
\section{Tail Reduction and the Trace Identity}
\label{sec:tail}

\begin{definition}[$S_N(c)$, $\sigma(c)$]
\label{def:SN}
\[
  S_N(c):=\sum_{n>N}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])},
  \qquad
  \sigma(c):=\sum_{n\ge 0}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}.
\]
\end{definition}

\begin{lemma}[Tail reduction]
\label{lem:tail}
For $g\in\PW_{c_0}^2$:
$\sum_{n>N}|a_n^{(c)}|^2\le\norm{g}^2 S_N(c).$
\end{lemma}
\begin{proof}
$a_n^{(c)}=\ip{\hat g}{
\widehat{\Phi_n^{(c)}}\cdot\mathbf{1}_{[-c_0,c_0]}}$
(Paley--Wiener).
Cauchy--Schwarz and sum over $n>N$.
\end{proof}

\begin{corollary}[Equi-tightness criterion]
\label{cor:equitight}
$\sup_c S_N(c)\to 0$
$\Rightarrow$
$\{a^{(c)}\}$ is equi-tight in $\ell^2$.
\end{corollary}

\begin{remark}[$S_N(c)$ as operator norm]
\label{rem:SN-operator}
Let $\Pi_N^{(c)}$ project onto
$\operatorname{span}\{\Phic{c}{0},\ldots,\Phic{c}{N}\}$.
Since $\norm{\Phic{c}{n}}=1$:
\begin{equation}\label{eq:SN-HS}
  S_N(c)
  = \norm{\Pc(I-\Pi_N^{(c)})}^2_{\mathrm{HS}}
  = \sup_{\substack{g\in\PW_{c_0}^2\\
    \norm{g}=1}}
    \sum_{n>N}|a_n^{(c)}|^2
  = \norm{(I-\Pi_N^{(c)})\Pc}^2_{\mathrm{op}}.
\end{equation}
\end{remark}

\begin{proposition}[Trace identity and growth]
\label{prop:trace}
$\sigma(c)=\mathrm{Tr}(P_c P_{c_0}P_c)
\sim cc_0/\pi\to\infty$.
\end{proposition}
\begin{proof}
$P_{c_0}|_{\PW_c^2}$ is Hilbert--Schmidt
for $c>c_0$, so $P_c P_{c_0}P_c$ is trace-class
on $\PW_c^2$ with trace
$\sum_n\|\Pc\Phic{c}{n}\|^2=\sigma(c)$.
The Slepian--Weyl law \cite{Slepian1961}
gives $\#\{n:\lambda_n^{(c)}\ge 1/2\}
\sim 2cc_0/\pi$,
hence $\sigma(c)\sim cc_0/\pi$.
\end{proof}

\begin{remark}[Critical observation:
  $\sigma(c)\to\infty$ does not imply
  $S_N(c)\not\to 0$]
\label{rem:counterexample}
This is the key subtlety of the paper.

\emph{Abstract counterexample.}
Let $b_n^{(c)}\ge 0$ with
$\sigma(c)=\sum_n b_n^{(c)}\to\infty$,
but for each $c$ there exists $N_0(c)\le N_0$
(fixed) such that $b_n^{(c)}=0$ for $n>N_0$.
Then $S_N(c)=0$ for all $N\ge N_0$,
yet $\sigma(c)\to\infty$.

This shows: the decomposition
$S_N(c)=\sigma(c)-\sum_{n\le N}b_n^{(c)}$
cannot yield $S_N(c)\not\to 0$
from $\sigma(c)\to\infty$ alone,
because the prefix $\sum_{n\le N}b_n^{(c)}$
might absorb all the growing mass.
Note that $\sum_{n\le N}b_n^{(c)}\le N+1$
(Bessel bound) shows the prefix is
\emph{bounded}, but does not show the
tail is large: the mass could migrate
into a different finite set of indices
for each $c$, remaining bounded
while $\sigma(c)\to\infty$ via
coordinates that happen to all lie in
$\{0,\ldots,N\}$.

\emph{The missing ingredient is therefore:}
the Fourier mass must not be
asymptotically absorbable by
any fixed finite index set.
This is the No-Concentration condition below.
\end{remark}

%%═══ 5. THE SPECTRAL DISTRIBUTION PROBLEM
\section{The Spectral Distribution Problem}
\label{sec:barrier}

\subsection{The No-Concentration condition}

\begin{definition}[No-Concentration (NC)]
\label{def:NC}
We say \emph{No-Concentration holds at rate $\alpha$}
if for some $\alpha\in(0,1)$
and some $\kappa\in(\alpha,1)$:
\[
  \liminf_{c\to\infty}\,
  \sum_{\lfloor\alpha c\rfloor < n
    \le\lfloor\kappa c\rfloor}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  > 0.
\]
That is: indices in a growing band
$n\in(\alpha c, \kappa c)$
persistently carry positive Fourier mass.
\end{definition}

\begin{remark}[NC vs.\ the Bessel bound]
\label{rem:NC-vs-Bessel}
The Bessel bound gives:
$\sum_{n=0}^N b_n^{(c)}\le N+1$.
This shows the prefix sum is bounded,
but says nothing about whether the
\emph{tail} $\sum_{n>N}$ is positive.
NC is an additional assumption
exactly at the level needed to conclude
$S_N(c)\ge c_*>0$ for all large $c$.
Bessel does not imply NC.
\end{remark}

\begin{assumption}[NC]
\label{ass:NC}
No-Concentration holds at some rate $\alpha\in(0,1)$:
\[
  \liminf_{c\to\infty}\,
  \sum_{n\sim\alpha c}
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  > 0.
\]
\end{assumption}

\begin{proposition}[Spectral Leakage Barrier]
\label{prop:barrier}
Under Assumption~\ref{ass:NC}:
$\liminf_{c\to\infty}\,S_N(c)>0$
for each fixed $N$,
and therefore $\sup_c S_N(c)\not\to 0$.
In particular, H3 fails under NC.
\end{proposition}
\begin{proof}
For each $N$, take $c$ large enough so that
$\lfloor\alpha c\rfloor > N$.
Then the band $\{n:\alpha c < n\le\kappa c\}
\subseteq\{n:n>N\}$.
Hence $S_N(c)\ge
\sum_{n\sim\alpha c}
\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$,
which is bounded below by a positive constant
by NC.
\end{proof}

\begin{remark}[The role of $\sigma(c)\to\infty$]
\label{rem:sigma-role}
Proposition~\ref{prop:trace}
($\sigma(c)\to\infty$) is \emph{necessary}
but not \emph{sufficient}:
it shows that total Fourier mass grows,
but does not localise where.
NC is the localisation assertion:
mass persists in the transition zone
$n\sim\alpha c$.
The Slepian--Weyl law gives growth;
NC gives distribution.
Both are required.
\end{remark}

\subsection{NC is equivalent to a local Weyl law}

\begin{remark}[NC $\Leftrightarrow$
  local Weyl law]
\label{rem:NC-Weyl}
Assumption~\ref{ass:NC} is
\emph{precisely} a local version of the
global Weyl law of Proposition~\ref{prop:trace}.
The global law says $\sum_n b_n^{(c)}\to\infty$;
NC says some \emph{band} $n\sim\alpha c$
carries positive mass.
The quantitative local form:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim \frac{c_0}{c}
  \quad \text{for } n=\lfloor\alpha c\rfloor
\]
(Open Problem~\ref{prob:fourier-mass})
would imply NC immediately:
$\sum_{n\sim\alpha c}
\|\cdots\|^2
\sim c_0\cdot\kappa > 0$.
\end{remark}

\subsection{What is and is not proved}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Statement & Status & Requires \\
\midrule
$\sigma(c)\to\infty$ & Proved & Slepian--Weyl \\
Prefix $\le N+1$ & Proved & Bessel \\
$S_N(c)\not\to 0$ & Conditional & NC \\
NC holds & Open & OP~\ref{prob:fourier-mass} \\
\bottomrule
\end{tabular}
\end{center}

\begin{remark}[Why NC is not a "minor" assumption]
\label{rem:NC-not-minor}
NC is exactly the assertion that
PSWF Fourier mass does not disappear
from the transition zone as $c\to\infty$.
This is a non-trivial statement:
the PSWFs $\widehat{\Phi_n^{(c)}}$
for $n\sim\alpha c$ concentrate
in frequency near the transition
between the ``good'' regime ($n\ll c$)
and the ``small'' regime ($n\gg c$).
Their behaviour is governed by
semiclassical WKB asymptotics
\cite{Zworski2012},
not by the pointwise results of
Papers~VIII or XII.
\end{remark}

%%═══ 6. H3 AND THE MAIN REDUCTION
\section{Hypothesis H3 and the Main Reduction}
\label{sec:H3}

\begin{hypothesis}[H3: Uniform Frequency Evacuation]
\label{hyp:H3}
$\sup_{c>c_0}S_N(c)\xrightarrow{N\to\infty}0$.
\end{hypothesis}

\begin{theorem}[H3 $\Rightarrow$ H1 (complete)]
\label{thm:H3-H1}
H3 implies H1.
\end{theorem}
\begin{proof}
Fix $\varepsilon>0$.
By H3, choose $N$ so that
$\sup_c S_N(c)<\varepsilon/(2\norm{g}^2)$.
By Lemma~\ref{lem:tail}:
$\sum_{n>N}|a_n^{(c)}|^2<\varepsilon/2$
for all $c$.
By Paper~XII, choose $c_\varepsilon$:
$\sum_{n=0}^N|a_n^{(c)}-a_n^{(\infty)}|^2
<\varepsilon/2$ for $c>c_\varepsilon$.
Hence $\norm{a^{(c)}-a^{(\infty)}}^2<\varepsilon$.
\end{proof}

\begin{theorem}[H1 $\Rightarrow$ H3
  (conditional)]
\label{thm:H1-H3}
Assume H1. Then for each fixed
$g\in\PW_{c_0}^2$:
$\sum_{n>N}|a_n^{(c)}|^2\to 0$
as $N\to\infty$, uniformly in $c$
for that fixed $g$.
The passage to
$\sup_{\norm{g}=1}$ (i.e.\ H3)
requires Open Problem~\ref{prob:H1H3}.
\end{theorem}
\begin{proof}
For fixed $g$:
$\sum_{n>N}|a_n^{(c)}|^2
\le 2\norm{a^{(c)}-a^{(\infty)}}^2
+2\sum_{n>N}|a_n^{(\infty)}|^2
\to 0$
(by H1 and $a^{(\infty)}\in\ell^2$).
\end{proof}

\begin{theorem}[Main reduction]
\label{thm:reduction}
The following chain holds:
\[
  \text{NC}
  \overset{\text{Prop.}\,\ref{prop:barrier}}
  {\Longrightarrow}
  \text{H3 fails}
  \overset{\text{Thm.}\,\ref{thm:H3-H1}}
  {\Longleftrightarrow}
  \text{H1 fails (if OP~\ref{prob:H1H3})}
\]
\[
  \text{H3}
  \overset{\text{Thm.}\,\ref{thm:H3-H1}}
  {\Longrightarrow}
  \text{H1},
  \qquad
  \text{H1}
  \overset{\text{Thm.}\,\ref{thm:H1-H3}
    +\,\text{OP}\,\ref{prob:H1H3}}
  {\Longrightarrow}
  \text{H3}.
\]
The resolution of H1 is therefore equivalent to:
\begin{itemize}
\item Either: show NC fails
  (PSWF mass evacuates the transition zone).
\item Or: resolve OP~\ref{prob:H1H3}
  and OP~\ref{prob:fourier-mass}
  (local Weyl law $\Rightarrow$ NC $\Rightarrow$
  H3 fails $\Rightarrow$ H1 fails).
\end{itemize}
\end{theorem}

\begin{remark}[Logical summary]
\label{rem:logical}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lll}
\toprule
Statement & Status & Depends on \\
\midrule
H3$\Rightarrow$H1 & \textbf{Proved} & --- \\
H1$\Rightarrow$H3 (fixed $g$)
  & \textbf{Proved} & --- \\
H1$\Rightarrow$H3 ($\sup_g$)
  & Open & OP~\ref{prob:H1H3} \\
NC$\Rightarrow$H3 fails
  & \textbf{Proved} & NC \\
NC holds
  & Open & OP~\ref{prob:fourier-mass} \\
H1 fails
  & Open
  & OP~\ref{prob:H1H3} + NC \\
\bottomrule
\end{tabular}
\end{center}
\end{remark}

%%═══ 7. INTERPRETATION
\section{Interpretation}
\label{sec:interp}

\subsection{What this paper achieves}

The main contribution is a precise reduction:
\[
  \text{H1}
  \underset{\text{tail reduction}}
  {\xrightarrow{\quad\sim\quad}}
  S_N(c)\to 0
  \underset{\text{NC + Weyl}}
  {\xleftarrow{\quad?\quad}}
  \text{transition zone asymptotics.}
\]
This is the structure of a
\emph{reduction paper}:
it does not close H1,
but replaces an opaque hypothesis
(H1: $\ell^2$-convergence)
by a transparent analytic condition
(NC: Fourier mass in transition zone).
The latter is accessible by known
semiclassical tools \cite{Zworski2012}.

\subsection{Why the transition zone is the obstacle}

For indices $n\ll c$ (low regime):
$\lambda_n^{(c)}\approx 1$,
and $\widehat{\Phi_n^{(c)}}$ concentrates
strongly in $[-c_0,c_0]$
(near-complete localisation).
For $n\gg c$ (high regime):
$\lambda_n^{(c)}\approx 0$
and mass in $[-c_0,c_0]$ is negligible.
The transition zone $n\sim\alpha c$
is exactly where the mass
$\|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}$
is neither negligible nor controllable
by low-index tools,
and the Slepian--Weyl law counts
exactly these indices.

\subsection{Comparison with existing tools}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
Tool & Controls & Does not control \\
\midrule
Bessel & Prefix $\le N+1$ & Tail \\
Paper~XII & $a_n^{(c)}$ (fixed $n$)
  & $n\to\infty$ with $c$ \\
Paper~VIII & $|\widehat{\Phi_n^{(c)}}|$
  pointwise & mass distribution \\
Slepian--Weyl & $\sigma(c)\sim cc_0/\pi$
  & index distribution of mass \\
\textbf{NC / Weyl law}
  & transition zone mass
  & \textbf{(= missing tool)} \\
\bottomrule
\end{tabular}
\end{center}

%%═══ 8. OPEN PROBLEMS
\section{Open Problems}
\label{sec:open}

\begin{problem}[Local Weyl law for PSWFs;
  entry point for Paper~XV]
\label{prob:fourier-mass}
\emph{(Priority: critical.)}
For $n=\lfloor\alpha c\rfloor$,
$\alpha\in(0,1)$ fixed:
\[
  \|\widehat{\Phi_n^{(c)}}\|^2_{L^2([-c_0,c_0])}
  \sim \frac{c_0}{c}
  \quad(c\to\infty).
\]
This would imply NC and thus
$S_N(c)\not\to 0$ rigorously.
Natural approach: WKB / semiclassical
asymptotics for PSWFs at $n\sim\alpha c$
\cite{Osipov2013,Zworski2012}.
This is the natural entry point for Paper~XV,
and the single most important open question
raised by Paper~XIV.
\end{problem}

\begin{problem}[H1 $\Rightarrow$ H3: uniformity]
\label{prob:H1H3}
\emph{(Priority: high.)}
Show: H1 implies
$\norm{(I-\Pi_N^{(c)})\Pc}_{\mathrm{op}}\to 0$
uniformly in $c$.
This is the Banach--Steinhaus step that
closes H1$\Leftrightarrow$H3.
If resolved positively together with
OP~\ref{prob:fourier-mass},
then H1 is refuted.
\end{problem}

\begin{problem}[NC without local Weyl law]
\label{prob:NC-direct}
\emph{(Priority: medium.)}
Is there a direct proof of NC
