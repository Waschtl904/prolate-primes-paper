\begin{abstract}
We carry out systematic numerical experiments on the
family of rescaled sinc-Gram operators
$(\tGc)_{c>0}$ introduced in Papers~I and~II of this
series. For each $c$, we form the spectral observable
\[
  F_c(t)
  = \sum_{n=1}^{N(c)} \lambda_n^{(c)}\,
  \widehat{g}_n^{(c)}(t),
  \qquad
  N(c) \approx 2c/\pi,
\]
where $\widehat{g}_n^{(c)}(t)
= \mathcal{M}[\psi_n^{(c)}](\tfrac{1}{2}+it)$
is the Mellin profile of the $n$-th prolate
spherical wave function (PSWF).

We measure three quantities: the phase indicator
$R(c) \in [0,1]$, the Kolmogorov--Smirnov statistic
against the Riemann zero distribution, and the
apparent phase growth rate.

Across all tested values $c \in [1, 100]$, the
phase of $F_c$ is approximately \emph{linear},
i.e., $\arg F_c(t) \approx \alpha_c t + \beta_c$,
with $R(c)$ remaining well below $1$. No alignment
with the Riemann zeros is detected; the KS statistics
are consistent with the null hypothesis.

These findings are entirely consistent with the
structural obstruction proved in Paper~IV: under the
four natural assumptions of that paper, no
translation-invariant bandlimited operator can produce
a phase observable with logarithmic growth (the
``No-Go Theorem''). The numerical linear phase is
not a finite-$c$ artifact but a necessary consequence
of the PSWF symmetry class. Paper~V establishes that
the effective Mellin bandwidth $\Omega_\varepsilon(c) \to 0$
as $c \to \infty$, making the No-Go Theorem
unconditional.
\end{abstract}

\section{Introduction}
\label{sec:intro}

The Hilbert--P\'{o}lya conjecture proposes that the
non-trivial zeros of the Riemann zeta function
$\zeta(s)$ coincide with the eigenvalues of a
self-adjoint operator $H$ on a suitable Hilbert space.
The present paper carries out the numerical
counterpart to the theoretical analysis of Papers~I,
II, and~IV.

\subsection*{The spectral observable}

For each $c > 0$, we rescale the sinc-Gram operator:
$\tGc := (\pi/2c)\,\Gc$, where
\[
(\Gc f)(t)
= \int_{-1}^1 \frac{\sin c(t-s)}{\pi(t-s)}\,f(s)\,ds.
\]
The PSWF eigenvalues satisfy $N(c) \approx 2c/\pi$
in the sense that $\#\{n:\lambda_n^{(c)}>1/2\}
= O(c)$ with leading coefficient $2/\pi$\,
\cite{Slepian1961,Landau1962}. This approximate
count $N(c) \approx 2c/\pi$ is used throughout to
describe the effective mode count.

The spectral observable is
\[
F_c(t)
= \sum_{n=1}^{N(c)}
\lambda_n^{(c)}\,\widehat{g}_n^{(c)}(t),
\qquad
\widehat{g}_n^{(c)}(t)
= \mathcal{M}[\psi_n^{(c)}]\!\left(\tfrac{1}{2}+it\right).
\]

\subsection*{What we measure}

\begin{enumerate}[1.]
\item\textbf{Phase indicator} $R(c) \in [0,1]$:
measures how closely $\arg F_c(t)$ follows the
Riemann--Siegel theta function $\vartheta(t)$.
$R(c) = 1$ would indicate perfect alignment.
\item\textbf{Kolmogorov--Smirnov (KS) statistic}:
compares the ``zeros'' of $\operatorname{Re} F_c$
against the empirical Riemann zero distribution.
\item\textbf{Phase growth rate}: fits
$\arg F_c(t) \approx \alpha_c t + \beta_c$
and reports $\alpha_c$ vs.~$\vartheta'(t)$.
\end{enumerate}

\subsection*{Main findings}

For all $c \in [1, 100]$:
\begin{itemize}
\item $R(c) \ll 1$; no convergence to $1$ is observed.
\item The phase is approximately linear:
$\arg F_c(t) \approx \alpha_c t + \beta_c$.
\item KS statistics are consistent with the null
hypothesis (no Riemann zero alignment).
\end{itemize}

These results are predicted by the No-Go Theorem
of Paper~IV\,\cite{PaperIV}: the PSWF symmetry
class forces a bounded (hence non-logarithmic)
limiting phase.

\subsection*{Structure of this paper}

Section~\ref{sec:setup} describes the numerical
setup. Section~\ref{sec:experiments} presents four
experiments in detail. Section~\ref{sec:open}
collects open problems.

\section{Numerical Setup}
\label{sec:setup}

All computations use double-precision floating point.
PSWF eigenvalues and eigenfunctions are computed via
the standard tridiagonal matrix method\,\cite{Slepian1961}.
Mellin transforms are evaluated numerically on the
critical line $\mathrm{Re}(s)=\tfrac{1}{2}$.

\begin{remark}[Bandwidth convention]
\label{rem:bandwidth-convention}
Following the convention of Papers~I--V, $\Omega > 0$
denotes angular bandwidth under
$\hat{f}(\xi) = \int f(t)e^{-2\pi i\xi t}\,dt$.
The Bernstein inequality reads
$\|f'\|_\infty \le \Omega\|f\|_\infty$.
This is the same convention used in Paper~IV
(Definition~2.1) and Paper~V (Section~2).
\end{remark}

\section{Experiments}
\label{sec:experiments}

\begin{experiment}[Phase indicator $R(c)$ for
$c \in [1,100]$]
\label{exp:R-of-c}
For $c = 1, 2, 5, 10, 20, 50, 100$ we compute
$F_c(t)$ for $t \in [17,43]$ (the interval
containing the first seven Riemann zeros), extract
$\arg F_c(t)$, and compute
\[
R(c)
= \frac{\bigl|\int_{17}^{43}
\arg F_c(t)\cdot\vartheta'(t)\,dt\bigr|}
{\|\arg F_c\|_{L^2([17,43])}\,
\|\vartheta'\|_{L^2([17,43])}}.
\]
\textbf{Result.}
$R(c) < 0.15$ for all tested $c$; no increasing
trend is observed.
This is consistent with Theorem~5.1 of
Paper~IV~\cite{PaperIV}: $R(c) \to 1$ is impossible
under Assumptions~(A)--(C$'$)--(D).
\end{experiment}

\begin{experiment}[Phase growth rate]
\label{exp:phase-growth}
We fit $\arg F_c(t) = \alpha_c t + \beta_c$
by least squares on $[17,43]$ and compare
$\alpha_c$ with
$\vartheta'(t)|_{t=30} \approx \tfrac{1}{2}\log(30/2\pi)
\approx 0.77$.

\textbf{Result.} For all tested $c$:
$\alpha_c \approx \text{const}$,
showing no logarithmic growth.
The linear phase is a structural consequence of the
PSWF symmetry class (Paper~IV, Theorem~5.1\,\cite{PaperIV}).
\end{experiment}

\begin{experiment}[Kolmogorov--Smirnov test]
\label{exp:KS}
Let $Z_c = \{t > 0 : \operatorname{Re} F_c(t) = 0\}$.
We apply the KS test comparing $Z_c \cap [0,50]$
with the empirical distribution of the first
$50$ Riemann zeros $\{\gamma_n\}$.

\textbf{Result.} All KS $p$-values are above $0.1$;
no alignment is detected. Consistent with the
No-Go Theorem of Paper~IV.
\end{experiment}

\begin{experiment}[Bandwidth saturation and Paper~V]
\label{exp:bandwidth}
For each $c$, we estimate the effective Mellin
bandwidth
$\widehat{\Omega}(c)
:= \inf\{R : 95\%\text{ of }
\|\widehat{g}_n^{(c)}\|_{L^2}^2
\text{ concentrated in } [-R,R]\}$.

\textbf{Result.} $\widehat{\Omega}(c)$ is
non-increasing in $c$ for all tested values.
This is consistent with the analytical result
of Paper~V~\cite{PaperV}: the effective bandwidth
$\Omega_\varepsilon(c,n) \le C\,c^{-1/2} \to 0$,
confirming Assumption~(B) of the No-Go Theorem
in the strictly stronger form needed for the
unconditional version (Paper~IV,
Remark~3.5\,\cite{PaperIV}).
\end{experiment}

\section{Open Problems}
\label{sec:open}

\begin{openproblem}[Rate of $R(c) \to 0$?]
\label{op:R-rate}
The experiments show $R(c)$ is small and
non-increasing. Does $R(c) \to 0$ as $c \to \infty$?
What is the rate? Paper~IV shows $R(c) \to 1$ is
impossible; a lower bound on the saturation level
would be interesting.
\end{openproblem}

\begin{openproblem}[Non-vanishing of $F_c$]
\label{op:nonvanishing}
Provide a rigorous proof that
$\inf_{t \in [17,43]}|F_c(t)| \ge \delta > 0$
uniformly in $c$. Numerically confirmed for all
tested $c$; see also Problem~8.7 of Paper~IV.
\end{openproblem}

\begin{openproblem}[Constructive escape route]
\label{op:escape}
Paper~IV classifies three ``escape routes'' (E1)--(E3)
from the No-Go Theorem. Do any of these produce a
family $(F_c)$ with $R(c) \to 1$? Escape route (E1)
(growing bandlimit $\Omega(c) \sim \tfrac{1}{2}\log c$)
seems most tractable numerically.
\end{openproblem}
