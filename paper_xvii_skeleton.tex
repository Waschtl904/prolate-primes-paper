\title[A$_2$-Stability for PSWF Amplitudes]{%
  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
(conditions (A1)--(A3)) uniformly as $h := c^{-1} \to 0$.

\medskip
\noindent\textbf{Main technical contribution.}
Lemma~\ref{lem:diff-olver} (Differentiated Olver Estimate)
establishes uniform $C^1$-control of the Olver phase $\phi(x;h)$
via the integral representation
\[
  \phi(x;h)
  = \mathrm{sgn}(x-x_0(h))\cdot
    \Bigl(\tfrac{3}{2}\int_{x_0(h)}^x p(s;h)\,ds\Bigr)^{2/3},
  \quad
  p(x;h) := \sqrt{E_n(h)-V(x)},
\]
reducing all $C^1$-control to uniform control of $p(x;h)$
and its integral on $K_\varepsilon$ (away from the turning point).
No ODE, no Gr\"onwall inequality, no Wasow expansion needed.

\medskip
\noindent\textbf{Meta-remark.}
CFU stability depends only on:
energy convergence $E_n(h)\to\lambda_n$;
$L^\infty$-control of $V$ on $\mathrm{supp}\,\chi$;
spectral isolation of the eigenbranch;
and distance from the turning point.
This is general well beyond the PSWF setting.
\end{abstract}

The following are used as black boxes:
\begin{itemize}
\item \textbf{thm:CFU-geom} (XVI):
  A$_2$-fold $\Rightarrow$ Airy normal form.
\item \textbf{thm:CFU-symbol} (XVI):
  $I^{-1/4,0}(\Lambda_0,\Lambda_1) \to S^{-1/2}_{1,0}(h)$, uniform in $h$.
\item \textbf{Assumption A2} (XVI):
  (A1) $c_3(h)\ge\delta_0>0$;
  (A2) $|\det(dt/d\beta)|\ge\delta_1>0$;
  (A3) unique fold on $\mathrm{supp}\,\chi$.
\item \textbf{hyp:rigidity} (XVI): Airy class $\Rightarrow$ A$_2$-fold uniform;
  verified here for PSWF.
\end{itemize}

\begin{remark}[Airy $\not\Rightarrow$ CFU]
\label{rem:airy-not-cfu}
Airy normal form is local. CFU admissibility is global and uniform in $h$.
The gap is closed by
Proposition~\ref{prop:branch-selection},
Assumption~\ref{ass:boundary-away},
and Lemma~\ref{lem:diff-olver}.
\end{remark}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF as semiclassical eigenfamily]\label{def:pswf}
$\psi_n(x;c)$ satisfy
$\mathcal{L}_c\psi_n = \chi_n\psi_n$,
$\mathcal{L}_c = -\tfrac{d}{dx}[(1-x^2)\tfrac{d}{dx}] + c^2x^2$.
With $h = c^{-1}$ and $E_n(h) := \chi_n h^2$,
this is the semiclassical problem
$(h^2\mathcal{L}_{h^{-1}})\psi_n = E_n(h)\psi_n$
with potential $V(x) = x^2/(1-x^2)$.
Turning points $x_\pm(h)$: $V(x_\pm) = E_n(h)$.
\end{definition}

\begin{remark}[Eigenvalue branch, momentum limit, turning-point smoothness]
\label{rem:eigenbranch}
$E_n(h) \to \lambda_n \in (0,1)$ smoothly as $h\to 0$;
$p_0(x) := \sqrt{\lambda_n - V(x)}$.

\medskip
\noindent\textit{Uniform positivity on $K_\varepsilon$.}
For every $\varepsilon>0$,
on $K_\varepsilon := \mathrm{supp}\,\chi\setminus U_\varepsilon$
(where $U_\varepsilon$ is an open $\varepsilon$-neighborhood
of $x_0^{(0)}$, the limiting turning point):
\[
  \inf_{x\in K_\varepsilon} p_0(x) \;\ge\; c_\varepsilon > 0.
\]
This holds because $K_\varepsilon$ avoids both the turning point
(where $p_0 = 0$) and the endpoints $\pm 1$
(Assumption~\ref{ass:boundary-away}).

\medskip
\noindent\textit{Smooth turning-point shift.}
$V(x_0(h)) = E_n(h)$ and $V'(x_0^{(0)}) \ne 0$
(strict monotonicity of $V$ near $x_0^{(0)}$);
the implicit function theorem gives $h\mapsto x_0(h)$ smooth
with $x_0(h)\to x_0^{(0)}$ as $h\to 0$.
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi \subset [-1+\delta_{\rm bdy},\,1-\delta_{\rm bdy}]$
for fixed $\delta_{\rm bdy}>0$, independently of $h$.
Consequences: $V,V',V'' \in L^\infty$ uniformly on $\mathrm{supp}\,\chi$;
boundary-layer effects and $V\to\infty$ are excluded.
\end{assumption}

%% ============================================================
\section{Olver's Uniform Asymptotic Scheme}
\label{sec:olver}
%% ============================================================

Olver \cite{Olver1974} (Ch.~10--11) gives
\[
  \psi_n(x;h)
  \sim h^{-1/4}\,a(x;h)\,
  \mathrm{Ai}(h^{-2/3}\phi(x;h))
  + \mathcal{O}(h^\infty),
\]
uniformly on $\mathrm{supp}\,\chi$ (Assumption~\ref{ass:boundary-away}).

\begin{remark}[Olver--CFU gap]
\label{rem:olver-gap}
Olver does not provide:
Lagrangian/FIO structure;
global geometry of $\Lambda_h$;
uniform transversality of $\pi_h$;
or uniform $C^1$-control of $\phi(x;h)$.
\emph{Classical warning:}
$f_h(x) = x+h\sin(x/h) \to x$ uniformly
but $f_h' = 1+\cos(x/h)$ does not converge.
\end{remark}

%% ============================================================
\section{Branch Selection}
\label{sec:branch}
%% ============================================================

\begin{remark}[WKB two-branch structure]
$\Lambda_h = \Lambda_h^+\cup\Lambda_h^-$,
$\Lambda_h^\pm = \{\xi=\pm p(x;h)\}$.
Condition~(P2) fails for $\Lambda_h$ as a whole.
\end{remark}

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
Fix $\varepsilon>0$. On $K_\varepsilon$:
\[
  \delta_\xi(\varepsilon)
  := \tfrac{1}{2}\inf_{h\in(0,h_0],\,x\in K_\varepsilon} p(x;h)
  > 0.
\]
Positivity holds because $K_\varepsilon$ excludes the turning point
(where $p=0$) as well as $x\to\pm 1$
(Assumption~\ref{ass:boundary-away}).
Note: $\delta_\xi$ depends on $\varepsilon$; branch separation
holds \emph{microlocally away from the turning point},
not on all of $\mathrm{supp}\,\chi$.

\medskip
There exists $\Pi^+ = \mathrm{Op}_h(\sigma^+)$ with
$\sigma^+\in C_c^\infty(T^*(-1,1))$
supported on $\{\xi\ge\delta_\xi/2\}$, such that:
\begin{enumerate}[{\rm(i)}]
\item $\chi\psi_n(\cdot;h) = \Pi^+(\chi\psi_n) + \mathcal{O}(h^\infty)$
  in $L^2$, uniform in $h$.
\item $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\end{enumerate}
\end{proposition}

\begin{remark}[Proof: ellipticity and no frequency overlap]
\label{rem:branch-proof}
On $K_\varepsilon$:
$\Lambda_h^+\subset\{\xi\ge\delta_\xi\}$ and
$\Lambda_h^-\subset\{\xi\le -\delta_\xi\}$
are microlocally separated with gap $2\delta_\xi(\varepsilon)>0$.
$\sigma^+$ is elliptic on $\{\xi\ge\delta_\xi\}$ and zero
on $\{\xi\le 0\}$: no frequency overlap between
$\Lambda_h^+$ and $\Lambda_h^-$ on $\mathrm{supp}\,\chi\cap K_\varepsilon$.
The $\mathcal{O}(h^\infty)$ remainder follows from the
$h$-pseudodifferential Egorov theorem \cite{Zworski2012}.
\emph{Caveat:} branch separation breaks down in an
$\varepsilon$-neighborhood of the turning point;
the construction is only valid on $K_\varepsilon$.
\end{remark}

%% ============================================================
\section{Global Geometry: (P1) and (P3)}
\label{sec:phi-conditions}
%% ============================================================

\begin{remark}[(P1): global generating function]
On $\Lambda_h^+\cap\pi^{-1}(K_\varepsilon)$,
$\xi=p(x;h)>0$ is single-valued,
so $\phi(x;h)$ (via the integral representation below)
is a global generating function.
Status: \Cond.
\end{remark}

\begin{remark}[(P3): no secondary caustics, spectral gap condition]
\label{rem:p3}
\emph{Statement.}
On $\mathrm{supp}\,\chi$, there are no secondary caustics;
the only fold of $\Lambda_h^+$ on $K_\varepsilon$ is the
A$_2$-fold at $x_0(h)$.

\medskip
\noindent\emph{Required input.}
This holds provided the eigenbranch $E_n(h)$
is spectrally isolated:
\begin{equation}\label{eq:specgap}
  \exists\, C>0,\, h_0>0:\quad
  \mathrm{dist}\bigl(E_n(h),\,\sigma(\mathcal{L}_h)\setminus\{E_n(h)\}\bigr)
  \;\ge\; C h
  \quad \forall\, h\in(0,h_0].
\end{equation}
This quantitative gap prevents:
avoided crossings and cluster behavior;
microlocal frequency overlap between different eigenbranches;
mode mixing at the symbol level of $\Pi^+$.

\medskip
\noindent\emph{Source.}
Estimate~\eqref{eq:specgap} is \Cond;
it follows from Sturm--Liouville theory
(eigenvalue asymptotics for $\mathcal{L}_h$ give
$E_{n+1}(h)-E_n(h) \sim C h$ for fixed $n$;
see, e.g., \cite{Olver1974} Ch.~12).
Status: \Cond.
\end{remark}

\section{The Olver--FIO Lift}
\label{sec:lift}
%% ============================================================

\begin{lemma}[Microlocal Structure, \Cond]
\label{lem:olver-fio-lift-micro}
Under Assumption~\ref{ass:boundary-away}
and the smooth-eigenbranch hypothesis:
\begin{enumerate}[{\rm(i)}]
\item
  $\Pi^+(\chi\psi_n(\cdot;h))\in I^{-1/4,0}(\Lambda_h^+)$
  with phase $\phi(x;h)$.
\item
  $\Lambda_h^+$ has a single A$_2$-fold at $(x_0,0)$
  with $c_3(h)\ge\delta_0>0$ uniformly.
\end{enumerate}
\end{lemma}

\begin{lemma}[Uniform Projection, \Cond]
\label{lem:uniform-projection}
Under Lemma~\ref{lem:olver-fio-lift-micro},
Proposition~\ref{prop:branch-selection},
(P1), (P3), and Lemma~\ref{lem:diff-olver}:
\[
  \inf_{h\in(0,h_0],\,x\in K_\varepsilon}|\phi'(x;h)| \ge \delta > 0,
  \qquad
  |\det\,d\pi_h^+|\ge\delta_2>0 \text{ on } K_\varepsilon.
\]
\end{lemma}

\begin{remark}
On $K_\varepsilon$: $\phi'=p/\phi$;
numerator $p\ge\delta_\xi>0$ (Proposition~\ref{prop:branch-selection});
denominator $|\phi|\ge c_\varepsilon''>0$ (Lemma~\ref{lem:diff-olver}(C));
uniform convergence from Lemma~\ref{lem:diff-olver}(D). \Cond.
\end{remark}

%% ============================================================
\section{The Differentiated Olver Estimate}
\label{sec:diff-olver}
%% ============================================================

\begin{lemma}[Monotonicity of $I$]
\label{lem:I-monotone}
On $K_\varepsilon$, $p(x;h)>0$ uniformly, so the map
$x\mapsto I(x;h) = \int_{x_0(h)}^x p(s;h)\,ds$
is strictly increasing for $x > x_0(h)$
and strictly decreasing for $x < x_0(h)$.
Consequently $\mathrm{sgn}(I(x;h)) = \mathrm{sgn}(x-x_0(h))$
and $|I(x;h)| \ge \delta_\xi(\varepsilon)\cdot|x-x_0(h)|$.
\end{lemma}

\begin{lemma}[Differentiated Olver Estimate, \Cond]
\label{lem:diff-olver}
\emph{(Main technical lemma.)}\;
Fix $\varepsilon>0$.
Define $p$, $I$, and $\phi$ as in the abstract,
with positive branch consistent with
Proposition~\ref{prop:branch-selection}.
On $K_\varepsilon$:

\begin{enumerate}[{\rm(A)}]
\item \textit{(Uniform convergence of $p$, via rationalization.)}
  Since $p(x;h)\ge\delta_\xi(\varepsilon)>0$
  and $p_0(x)\ge c_\varepsilon>0$ on $K_\varepsilon$:
  \[
    |p(x;h)-p_0(x)|
    = \frac{|E_n(h)-\lambda_n|}{p(x;h)+p_0(x)}
    \;\le\; \frac{|E_n(h)-\lambda_n|}{\delta_\xi+c_\varepsilon}
    \;=:\; A_\varepsilon(h) = \mathcal{O}(h).
  \]
  \emph{The Lipschitz route via $\sqrt{\,\cdot\,}$
  fails at $p_0=0$; the rationalization is the correct argument.}

\item \textit{(Uniform convergence of $I$, with turning-point shift.)}
  \begin{align*}
    |I(x;h)-I_0(x)|
    &\le \underbrace{\|p(\cdot;h)\|_{C^0}\cdot|x_0(h)-x_0^{(0)}|}_{\text{shift term}}
       + \underbrace{|K_\varepsilon|\cdot A_\varepsilon(h)}_{\text{integrand term}}
    = \mathcal{O}(h),
  \end{align*}
  using $x_0(h)\to x_0^{(0)}$ smoothly
  (IFT, $V'(x_0^{(0)})\ne 0$; Remark~\ref{rem:eigenbranch})
  and (A).

\item \textit{(Denominator bounded away from zero.)}
  By Lemma~\ref{lem:I-monotone} and $|x-x_0(h)|\ge\varepsilon/2$
  on $K_\varepsilon$ for $h\le h_0$:
  \[
    |I(x;h)| \ge \delta_\xi(\varepsilon)\cdot\tfrac{\varepsilon}{2}
    \;=:\; c_\varepsilon' > 0,
  \]
  hence
  $|\phi(x;h)| = (\tfrac{3}{2}|I(x;h)|)^{2/3}
  \ge (\tfrac{3}{4}\delta_\xi\varepsilon)^{2/3} =: c_\varepsilon'' > 0$
  uniformly.

\item \textit{(Uniform $C^1$-control via quotient.)}
  $\phi'(x;h)=p(x;h)/\phi(x;h)$ and $\phi_0'=p_0/\phi_0$.
  \[
    |\phi'-\phi_0'|
    \le \frac{|p-p_0|}{|\phi|} + p_0\cdot\frac{|\phi-\phi_0|}{|\phi|\cdot|\phi_0|}
    \;\le\; \frac{A_\varepsilon(h)}{c_\varepsilon''}
           + \frac{p_0^{\max}\cdot B_\varepsilon(h)}{(c_\varepsilon'')^2}
    \to 0
  \]
  uniformly, where $B_\varepsilon(h):=\|\phi(\cdot;h)-\phi_0\|_{C^0}=\mathcal{O}(h)$
  follows from (B).
  Since $\phi_0'\ge c_\varepsilon/c_\varepsilon''>0$:
  \[
    \inf_{h\le h_0,\,x\in K_\varepsilon}|\phi'(x;h)|\;\ge\;
    \tfrac{1}{2}\inf_{K_\varepsilon}|\phi_0'|>0.
  \]
\end{enumerate}
\end{lemma}

\begin{remark}[Robustness]
\label{rem:robustness}
The proof uses only:
(i) $E_n(h)\to\lambda_n$ smoothly;
(ii) $V\in L^\infty$ on $\mathrm{supp}\,\chi$;
(iii) rationalization of $|p-p_0|$ (key step);
(iv) monotonicity of $I$ from Lemma~\ref{lem:I-monotone};
(v) IFT for $x_0(h)$.
The argument is general: it applies to any semiclassical
Schr\"odinger operator with a simple turning point,
$V\in L^\infty$, and smooth energy convergence.
\end{remark}

\section{Derivation of Assumption~A2}
\label{sec:a2-derivation}
%% ============================================================

\begin{proposition}[A2, \Cond]
\label{prop:a2-from-lemmas}
Under all preceding lemmas and conditions:
$|\det(dt/d\beta)|\ge\delta_1>0$ uniformly in $h$.
\end{proposition}

\begin{proof}[Proof sketch]
Restrict to $\Lambda_h^+$ (Proposition~\ref{prop:branch-selection}).
By (P1): $\phi$ is a global generating function on $K_\varepsilon$.
By (P3): no secondary caustics.
The CFU Jacobian is controlled by $|\phi'(x;h)|$;
Lemma~\ref{lem:diff-olver}(D) gives $|\phi'|\ge\delta>0$ uniformly.
\end{proof}

\section{Target Theorem}
\label{sec:target}
%% ============================================================

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
Under Assumption~\ref{ass:boundary-away},
the smooth-eigenbranch hypothesis,
and the spectral gap condition~\eqref{eq:specgap},
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly as $h\to 0$,
and \textup{hyp:rigidity} (XVI) holds.
\end{theorem}

\begin{remark}[Dependency chain]
\label{rem:chain}
\[
  \bigl[E_n\to\lambda_n,\;V\in L^\infty,\;
  V'(x_0^{(0)})\ne 0,\;
  \mathrm{dist}\ge Ch\bigr]
  \;\Rightarrow\;
  \bigl[\text{Lem.}~\ref{lem:diff-olver},\;
  \text{Prop.}~\ref{prop:branch-selection}\bigr]
  \;\Rightarrow\;
  \text{Lem.}~\ref{lem:uniform-projection}
  \;\Rightarrow\;
  \text{Prop.}~\ref{prop:a2-from-lemmas}
  \;\Rightarrow\;
  \text{Thm.}~\ref{thm:xvii-main}.
\]
\end{remark}

\begin{remark}[Scope]
Paper~XVII does not prove the full inverse microlocal conjecture,
global bounds, or concentration-of-energy results for PSWF.
\end{remark}
