  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 uniformly as $h:=c^{-1}\to 0$.

\medskip
\noindent\textbf{Assumption~A2.}
(A1) $c_3(h)\ge\delta_0>0$;
(A2) $|\det(dt/d\beta)|\ge\delta_1>0$;
(A3) unique A$_2$-fold ($V'(x_0)\ne 0$, $V''(x_0)\ne 0$).

\medskip
\noindent\textbf{Three hard cores.}
\begin{enumerate}[\rm(C1)]
\item Spectral gap: $E_{n+1}(h)-E_n(h)\ge Ch$ (Theorem~\ref{thm:specgap}).
\item Smooth eigenbranch: $h\mapsto E_n(h)$ real-analytic (Theorem~\ref{thm:eigenbranch}).
\item Olver $C^1$-stability: $\inf_{K_\varepsilon}|\phi'(\cdot;h)|\ge\delta>0$
  (Lemmas~\ref{lem:C0-airy}--\ref{lem:tp-perturbation}).
\end{enumerate}
\end{abstract}

\label{rem:cfu-geom-nontrivial}

\end{remark}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF, classical and semiclassical momentum]
\label{def:pswf-p0}
$\psi_n$ satisfy $(h^2\mathcal{L}_{h^{-1}})\psi_n=E_n(h)\psi_n$,
$V(x)=x^2/(1-x^2)$, $E_n(h):=\chi_n h^2$.

\medskip
\noindent\textbf{Limiting energy:}
$\lambda_n:=\lim_{h\to 0}E_n(h)$
(well-defined by Theorem~\ref{thm:eigenbranch}).

\medskip
\noindent\textbf{Classical momentum:}
\begin{equation}\label{eq:p0-def}
  p_0(x) := \sqrt{\lambda_n - V(x)}, \quad x\in\{\lambda_n > V(x)\}.
\end{equation}

\noindent\textbf{Semiclassical momentum:}
$p(x;h):=\sqrt{E_n(h)-V(x)}$ for $x\in K_\varepsilon$, $h\le h_0$
(positive by Proposition~\ref{prop:branch-selection}(i)).
\end{definition}

\begin{hypothesis}[A$_2$-fold at the turning point]
\label{hyp:a2-fold}
The limiting turning point $x_0^{(0)}\in(0,1-\delta_{\rm bdy})$ satisfies:
\begin{enumerate}[{\rm(F1)}]
\item $V'(x_0^{(0)})\ne 0$.
  \emph{Verification:}
  $V'(x)=2x/(1-x^2)^2>0$ on $(0,1-\delta_{\rm bdy})$. \Rig.
\item $V''(x_0^{(0)})\ne 0$.
  \emph{Verification:}
  $V''(x)=2(1+3x^2)/(1-x^2)^3>0$ on $(0,1)$. \Rig.
\end{enumerate}
(F1) ensures IFT applicability;
(F2) ensures A$_2$-stability of the normal form
(no higher-order fold degeneracy).
\end{hypothesis}

\begin{remark}[Shape of $V$ and unique turning points]
\label{rem:V-shape}
\begin{enumerate}[{\rm(i)}]
\item $V(0)=0$, $V\to+\infty$ as $x\to\pm 1$, $V(-x)=V(x)$.
\item $V'>0$ on $(0,1)$; $V'<0$ on $(-1,0)$. \Rig.
\item Unique turning point per half-interval for $\lambda_n<V(1-\delta_{\rm bdy})$. \Rig.
\item Multiple turning points: each is a separate A$_2$-fold
  (Hypothesis~\ref{hyp:a2-fold} applies to each). \Rig.
\end{enumerate}
\end{remark}

\begin{remark}[Geometric positivity]
\label{rem:geo-positivity}
$K_\varepsilon:=\mathrm{supp}\,\chi\setminus U_\varepsilon$.
By (F1) and strict monotonicity:
$c_\varepsilon^{\rm geo}:=\inf_{K_\varepsilon}(\lambda_n-V)>0$. \Rig.
\textbf{All constants are $h$-independent.}
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$,
independently of $h$.
Implies: $V,V',V''\in L^\infty$;
$\lambda_n<V(1-\delta_{\rm bdy})$ (turning point exists).
\end{assumption}

\section{Hard Core (C2): Smooth Eigenbranch}
\label{sec:eigenbranch}
%% ============================================================

\begin{theorem}[Smooth eigenbranch, \Cond]
\label{thm:eigenbranch}
Assume:
\begin{enumerate}[{\rm(E1)}]
\item \textit{(Fixed domain.)} $h^2\mathcal{L}_{h^{-1}}$
  is defined on $\mathcal{D}=H^2(-1,1)\cap H^1_0(-1,1)$,
  independent of $h\in(0,h_0)$. \Rig\ (SL operator; $V\in L^\infty$).
\item \textit{(Norm-resolvent analyticity.)}
  $h\mapsto (h^2\mathcal{L}_{h^{-1}}-z)^{-1}$
  is analytic in $\mathcal{B}(L^2)$ for $z\notin\sigma$.
  \emph{Verification:}
  coefficients of $\mathcal{L}_{h^{-1}}$ are polynomial in $c=h^{-1}$;
  multiplied by $h^2$ they are polynomial in $h$,
  hence analytic; the resolvent inherits analyticity
  by standard perturbation theory \cite{Kato1966} Ch.~VII.
\item \textit{(Simple eigenvalue.)} $E_n(h_0)$ is simple:
  SL oscillation theory. \Rig.
\end{enumerate}
By Kato \cite{Kato1966} Ch.~VII.\S2:
$h^2\mathcal{L}_{h^{-1}}$ is an analytic family of type~(A);
hence $h\mapsto E_n(h)$ is real-analytic with
$E_n(h)\to\lambda_n$ as $h\to 0$;
$x_0(h)$ is real-analytic by IFT + (F1).
\end{theorem}

\begin{remark}[Why domain constancy matters for Kato type~(A)]
Kato type~(A) requires a fixed domain independent of the parameter.
For $h^2\mathcal{L}_{h^{-1}}$ on $(-1,1)$ with $V\in L^\infty$
(Assumption~\ref{ass:boundary-away}),
the domain $H^2\cap H^1_0$ is $h$-independent (E1).
The parameter $h$ enters only through the operator coefficients,
not through boundary conditions or function space---
this is what separates PSWF from operators with $h$-dependent domains.
\end{remark}

\section{Hard Core (C1): Spectral Gap}
\label{sec:specgap}
%% ============================================================

\begin{lemma}[Bohr--Sommerfeld inversion, \Cond]
\label{lem:bs-inversion}
Let $S(E):=\int_{x_-(E)}^{x_+(E)}\sqrt{E-V}\,dx$.
Under (H1)--(H3) of Theorem~\ref{thm:specgap}:
\begin{enumerate}[{\rm(i)}]
\item $S'(E)=T(E)/2>0$: $S$ is strictly increasing near $\lambda_n$.
\item $S$ is a local $C^\infty$ diffeomorphism;
  $(S^{-1})'=2/T(\lambda_n)$.
\item \textit{(Uniform inversion in $h$.)} By MVT with
  $\xi\in(E_n,E_{n+1})\to\lambda_n$:
\[
  E_{n+1}-E_n
  =\frac{S(E_{n+1})-S(E_n)}{S'(\xi)}
  =\frac{\pi h+\mathcal{O}(h^2)}{T(\lambda_n)/2+\mathcal{O}(h)}
  =\frac{2\pi h}{T(\lambda_n)}+\mathcal{O}(h^2).
\]
The denominator bound $S'(\xi)\ge T(\lambda_n)/4>0$ holds
for $h\le h_0$ small, ensuring the inversion is uniform in $h$.
\end{enumerate}
\end{lemma}

\begin{theorem}[Spectral gap, \Cond]
\label{thm:specgap}
Assume:
\begin{enumerate}[{\rm(H1)}]
\item 1D integrability: automatic. \Rig.
\item $0<T(\lambda_n)<\infty$: $V-\lambda_n$ has simple zeros at $x_\pm$
  (Hypothesis~\ref{hyp:a2-fold}(F1)); square-root singularity integrable.
\item $\partial_E T(\lambda_n)\ne 0$:
  $\partial_E T(\lambda_n)
  =-\tfrac12\oint(\lambda_n-V)^{-3/2}\,dx<0$,
  finite and nonzero by (H2).
\end{enumerate}
Then by Lemma~\ref{lem:bs-inversion}:
\begin{equation}\label{eq:specgap}
  E_{n+1}(h)-E_n(h)
  =\frac{2\pi h}{T(\lambda_n)}+\mathcal{O}(h^2)
  \ge Ch, \quad C=\frac{\pi}{T(\lambda_n)}>0.
\end{equation}
In particular $\mathrm{dist}(E_n,\sigma\setminus\{E_n\})\ge Ch$.
\end{theorem}

\begin{remark}[Gatekeeper role]
Theorem~\ref{thm:specgap} ensures branch separation $\delta_\xi>0$,
prerequisite for all microlocal and CFU arguments.
\end{remark}

\section{Branch Selection and Uniform Ellipticity}
\label{sec:branch}
%% ============================================================

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
\textit{(i) Spectral positivity.}
$|E_n(h)-\lambda_n|=\mathcal{O}(h)$
$\Rightarrow E_n(h)-V\ge c_\varepsilon^{\rm geo}/2$ on $K_\varepsilon$.
Define $\delta_\xi:=\sqrt{c_\varepsilon^{\rm geo}/2}$; then $p\ge\delta_\xi>0$.

\medskip
\textit{(ii) Branch separation.}
$\mathrm{WF}_h(\psi_n)\subset\{\xi^2+V=E_n\}$ (\cite{Zworski2012} Ch.~12);
spectral gap excludes mode mixing;
$\chi\psi_n=\Pi^+(\chi\psi_n)+\mathcal{O}(h^\infty)$,
$\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\end{proposition}

\begin{remark}[CFU on each component]
\label{rem:cfu-components}
$K_\varepsilon^\pm:=K_\varepsilon\cap\{\pm(x-x_0(h))>0\}$.
On $K_\varepsilon^+$: $I>0$, $\phi=(\tfrac32 I)^{2/3}$.
On $K_\varepsilon^-$: $I<0$, $\phi=-(\tfrac32|I|)^{2/3}$.
CFU \cite{CFU1957} separately;
matching via \textbf{thm:CFU-geom} (Remark~\ref{rem:cfu-geom-nontrivial}).
\end{remark}

\section{Hard Core (C3): Olver $C^1$-Stability}
\label{sec:olver-C1}
%% ============================================================

\begin{lemma}[Uniform ellipticity, \Cond]
\label{lem:uniform-ellipticity}
$p\ge\delta_\xi>0$, $p_0\ge c_\varepsilon:=\sqrt{c_\varepsilon^{\rm geo}}>0$,
$p+p_0\ge\delta_\xi+c_\varepsilon>0$, uniform in $h$.
\end{lemma}

\begin{lemma}[Energy comparison, \Cond]
\label{lem:energy-comparison}
$|p-p_0|=|E_n-\lambda_n|/(p+p_0)\le|E_n-\lambda_n|/(\delta_\xi+c_\varepsilon)=:A_\varepsilon=\mathcal{O}(h)$.
\end{lemma}

\begin{lemma}[Integral stability, \Cond]
\label{lem:integral-stability}
$|I-I_0|\le\|p\|_{C^0}|x_0(h)-x_0^{(0)}|+|K_\varepsilon|A_\varepsilon=:\tilde A_\varepsilon=\mathcal{O}(h)$.
\end{lemma}

\begin{lemma}[Non-degeneracy, \Cond]
\label{lem:nondeg}
$|I|,|I_0|\ge c_\varepsilon':=\delta_\xi\varepsilon/2>0$;
$|\phi|\ge c_\varepsilon''$; $|\phi_0|\ge d_\varepsilon>0$.
\end{lemma}

\begin{lemma}[$C^0$-Airy stability, \Cond]
\label{lem:C0-airy}
$\|\phi-\phi_0\|_{C^0}\le L_\varepsilon\tilde A_\varepsilon=\mathcal{O}(h)$;
$L_\varepsilon=\tfrac23(\tfrac32)^{2/3}(c_\varepsilon')^{-1/3}$.
\end{lemma}

\begin{lemma}[Derivative commuting, \Cond]
\label{lem:deriv-commuting}
On $K_\varepsilon^\pm$:
$\phi'=p/\phi$ and $\phi_0'=p_0/\phi_0$
(Definition~\ref{def:pswf-p0}).
$\|\phi'-\phi_0'\|_{C^0}=\mathcal{O}(h)$ by quotient estimate.

\medskip
\noindent\emph{Why exchange is valid.}
$\phi'=p/\phi$ is an algebraic formula; $C^0$-control of $\phi$
directly controls $\phi'$.
(Warning: $h\sin(x/h)\to 0$ in $C^0$
but $\cos(x/h)\not\to 0$; that failure does not occur here.)
\end{lemma}

\begin{lemma}[Turning-point perturbation, \Cond]
\label{lem:tp-perturbation}
$|x_0(h)-x_0^{(0)}|=(E_n(h)-\lambda_n)/V'(x_0^{(0)})+\mathcal{O}(h^2)=\mathcal{O}(h)$
(IFT + Theorem~\ref{thm:eigenbranch} + (F1)).
\end{lemma}

\begin{corollary}[$C^1$-stability of $\phi$]
\label{cor:phi-C1}
$\inf_{K_\varepsilon}|\phi'(\cdot;h)|\ge\tfrac12\inf|\phi_0'|>0$;
$\phi_0'=p_0/\phi_0\ge c_\varepsilon/d_\varepsilon>0$.
\end{corollary}

\section{CFU Jacobian in 1D}
\label{sec:cfu-jacobian}
%% ============================================================

\begin{lemma}[CFU Jacobian, \Cond]
\label{lem:cfu-jacobian-1d}
In normalized CFU coordinates $\Phi=\phi\theta-\tfrac13\theta^3$:
$\partial^2_{x\theta}\Phi=\phi'$,
hence $|\det(dt/d\beta)|=|\phi'|\ge\delta_1>0$ (Corollary~\ref{cor:phi-C1}).
Other coordinates: bounded bounded-away-from-zero factor;
(A2) unaffected.
\end{lemma}

\begin{theorem}[PSWF A$_2$-stability]
\label{thm:xvii-main}
