\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI uniformly
as $h:=c^{-1}\to 0$.

\medskip
\noindent\textbf{Recall (Assumption~A2).}
(A1) $c_3(h)\ge\delta_0>0$;
(A2) $|\det(dt/d\beta)|\ge\delta_1>0$;
(A3) unique fold.
In 1D: unique turning point with $\partial_\xi\sigma=0$,
$\partial^2_\xi\sigma\ne 0$.

\medskip
\noindent\textbf{Main contribution.}
Lemmas~\ref{lem:uniform-ellipticity}--\ref{lem:phi-C1}
establish uniform $C^1$-control of the Olver phase without ODE or Wasow.
\end{abstract}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF]
$\psi_n$ satisfy $(h^2\mathcal{L}_{h^{-1}})\psi_n=E_n(h)\psi_n$,
$V(x)=x^2/(1-x^2)$, $E_n(h):=\chi_n h^2$.
\end{definition}

\begin{remark}[Shape of $V$ and unique turning points]
\label{rem:V-shape}
$V(x)=x^2/(1-x^2)$ satisfies:
\begin{enumerate}[{\rm(i)}]
\item $V(0)=0$, $V(x)\to+\infty$ as $x\to\pm 1$, $V(-x)=V(x)$.
\item \textit{Strict monotonicity on each half-interval:}
  $V'(x)=2x/(1-x^2)^2$, which is
  $>0$ on $(0,1)$ and $<0$ on $(-1,0)$.
\item \textit{Existence and uniqueness of turning point:}
  For any $\lambda_n>0$ and any half-interval $(0,1-\delta_{\rm bdy})$,
  the equation $V(x)=\lambda_n$ has exactly one solution
  provided $\lambda_n < V(1-\delta_{\rm bdy})$,
  which is guaranteed by Assumption~\ref{ass:boundary-away}.
\item For $\mathrm{supp}\,\chi$ crossing $0$,
  there are two turning points on opposite sides;
  each is a separate A$_2$-fold.
\end{enumerate}
\end{remark}

\begin{remark}[Geometric positivity of $\lambda_n - V$ on $K_\varepsilon$]
\label{rem:geo-positivity}
Let $x_0^{(0)}$ be the limiting turning point on the relevant half-interval.
$K_\varepsilon:=\mathrm{supp}\,\chi\setminus U_\varepsilon$
excludes an $\varepsilon$-neighborhood of $x_0^{(0)}$.
By strict monotonicity of $V$ (Remark~\ref{rem:V-shape}(ii)):
\[
  c_\varepsilon^{\rm geo} := \inf_{x\in K_\varepsilon}(\lambda_n-V(x)) > 0.
\]
This is a purely geometric constant, independent of $h$.
\textbf{All constants $c_\varepsilon^{\rm geo},\delta_\xi,c_\varepsilon',
c_\varepsilon'',d_\varepsilon,L_\varepsilon$ are independent of $h$.
All bounds hold on $K_\varepsilon$ unless stated otherwise.}
\end{remark}

\begin{remark}[Spectral convergence and turning-point shift]
\label{rem:eigenbranch}
$E_n(h)\to\lambda_n\in(0,1)$ smoothly (SL analyticity; \Cond).
$V(x_0(h))=E_n(h)$, $V'(x_0^{(0)})\ne 0$
$\Rightarrow$ $x_0(h)\to x_0^{(0)}$ smoothly (IFT; \Cond).
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$
independently of $h$.
Implies: $V,V',V''\in L^\infty$;
boundary layers excluded;
$\lambda_n < V(1-\delta_{\rm bdy})$ (turning point exists).
\end{assumption}

\section{Spectral Gap}
\label{sec:specgap}
%% ============================================================

\begin{lemma}[Spectral gap, \Cond]
\label{lem:specgap}
Assume:
\begin{enumerate}[{\rm(H1)}]
\item \textit{(Classical integrability.)}
  The Hamiltonian $H(x,\xi)=\xi^2+V(x)$ is completely integrable
  on the energy surface $\{H=\lambda_n\}$.
\item \textit{(Period nondegeneration.)}
  The classical period
  $T(E):=\oint_{H=E}\frac{dx}{p(x;E)}$
  satisfies $0<T(\lambda_n)<\infty$.
\item \textit{(Smooth Bohr--Sommerfeld invertibility.)}
  The Bohr--Sommerfeld quantization
  $S(E_n(h)):=\oint_{H=E_n}p\,dx=(n+\tfrac{1}{2})\pi h+\mathcal{O}(h^2)$
  is smoothly invertible in $E$ near $\lambda_n$
  (i.e., $S'(\lambda_n)=T(\lambda_n)/2\ne 0$).
\end{enumerate}
Then:
\begin{equation}\label{eq:specgap}
  E_{n+1}(h)-E_n(h)
  = \frac{2\pi h}{T(\lambda_n)} + \mathcal{O}(h^2)
  \ge Ch > 0
\end{equation}
for $C=\pi/T(\lambda_n)>0$ and all $h\le h_0$.
In particular,
$\mathrm{dist}(E_n(h),\sigma(\mathcal{L}_h)\setminus\{E_n(h)\})\ge Ch$.

\medskip
\noindent\emph{Verification for PSWF.}
(H1): 1D, hence automatically integrable.
(H2): $T(\lambda_n)=\int_{x_-}^{x_+}(\lambda_n-V)^{-1/2}\,dx$;
  finite since $V$ has only simple zeros at $x_\pm$
  (square-root integrable singularity);
  positive since the classically allowed interval is nonempty.
(H3): $S'(\lambda_n)=T(\lambda_n)/2\ne 0$ by (H2);
  see \cite{Zworski2012} Ch.~9.
\end{lemma}

\section{Branch Selection}
\label{sec:branch}
%% ============================================================

\begin{proposition}[Branch selection and uniform ellipticity, \Cond]
\label{prop:branch-selection}
Fix $\varepsilon>0$.

\medskip
\noindent\textit{(i) Spectral positivity.}
$E_n(h)\to\lambda_n$ (Remark~\ref{rem:eigenbranch}),
so for $h\le h_0$:
\[
  \inf_{x\in K_\varepsilon}(E_n(h)-V(x))
  \ge c_\varepsilon^{\rm geo} - |E_n(h)-\lambda_n|
  \ge \tfrac{1}{2}c_\varepsilon^{\rm geo} > 0.
\]
Define $\delta_\xi(\varepsilon):=\sqrt{c_\varepsilon^{\rm geo}/2}$;
then $p(x;h)\ge\delta_\xi>0$ on $K_\varepsilon$ for $h\le h_0$.

\medskip
\noindent\textit{(ii) Branch separation.}
$\mathrm{WF}_h(\psi_n)\subset\{\xi^2+V=E_n(h)\}$
(\cite{Zworski2012} Ch.~12).
On $K_\varepsilon$: $\Lambda_h^\pm\subset\{\pm\xi\ge\delta_\xi\}$
(uniform gap $2\delta_\xi$, ensured by (i)
and the spectral gap Lemma~\ref{lem:specgap} to exclude mode mixing).
Microlocal partition of unity yields $\Pi^+=\mathrm{Op}_h(\sigma^+)$:
(a) $\chi\psi_n=\Pi^+(\chi\psi_n)+\mathcal{O}(h^\infty)$;
(b) $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\textit{Caveat:} holds only on $K_\varepsilon$.
\end{proposition}

$K_\varepsilon^\pm:=K_\varepsilon\cap\{\pm(x-x_0(h))>0\}$.

\end{remark}

\section{The Differentiated Olver Estimate (Decomposed)}
\label{sec:diff-olver}
%% ============================================================

\begin{lemma}[Uniform ellipticity away from turning point, \Cond]
\label{lem:uniform-ellipticity}
On $K_\varepsilon$, both $p(x;h)$ and $p_0(x)$ are
uniformly bounded away from zero:
\[
  p(x;h) \ge \delta_\xi(\varepsilon) > 0,
  \qquad
  p_0(x) = \sqrt{\lambda_n - V(x)} \ge c_\varepsilon := \sqrt{c_\varepsilon^{\rm geo}} > 0.
\]
Consequently $p(x;h)+p_0(x) \ge \delta_\xi + c_\varepsilon > 0$,
uniform in $h$.
(Geometric part: Remark~\ref{rem:geo-positivity};
spectral part: Proposition~\ref{prop:branch-selection}(i).)
\end{lemma}

\begin{lemma}[Monotonicity of $I$, \Rig]
\label{lem:I-monotone}
$p>0$ on $K_\varepsilon$, so $I(x;h)=\int_{x_0(h)}^x p\,ds$ is strictly monotone,
$\mathrm{sgn}(I)=\mathrm{sgn}(x-x_0(h))$,
$|I|\ge\delta_\xi|x-x_0(h)|$.
\end{lemma}

\begin{lemma}[Energy comparison, \Cond]
\label{lem:energy-comparison}
By Lemma~\ref{lem:uniform-ellipticity}:
\[
  |p(x;h)-p_0(x)|
  = \frac{|E_n(h)-\lambda_n|}{p+p_0}
  \le \frac{|E_n-\lambda_n|}{\delta_\xi+c_\varepsilon}
  =: A_\varepsilon(h) = \mathcal{O}(h).
\]
(Lipschitz of $\sqrt{\cdot}$ fails at $p_0=0$;
the denominator bound from Lemma~\ref{lem:uniform-ellipticity}
is the key step.)
\end{lemma}

\begin{lemma}[Integral stability, \Cond]
\label{lem:integral-stability}
\[
  |I(x;h)-I_0(x)|
  \le \|p\|_{C^0}|x_0(h)-x_0^{(0)}|
  + |K_\varepsilon|A_\varepsilon(h)
  =: \tilde A_\varepsilon(h) = \mathcal{O}(h).
\]
\end{lemma}

\begin{lemma}[Non-degeneracy, \Cond]
\label{lem:nondeg}
By Lemma~\ref{lem:I-monotone} and $|x-x_0(h)|\ge\varepsilon/2$:
\[
  |I|,|I_0| \ge c_\varepsilon' := \delta_\xi\varepsilon/2 > 0;
  \quad
  |\phi| \ge c_\varepsilon'' := (\tfrac{3}{4}\delta_\xi\varepsilon)^{2/3};
  \quad
  |\phi_0| \ge d_\varepsilon := (\tfrac{3}{4}c_\varepsilon\varepsilon)^{2/3} > 0.
\]
\end{lemma}

\begin{lemma}[$C^0$-stability of $\phi$, \Cond]
\label{lem:phi-C0}
On each $K_\varepsilon^\pm$ (Remark~\ref{rem:cfu-components}),
$I$ has fixed sign,
so $I\mapsto(\tfrac{3}{2}I)^{2/3}$ is smooth.
Since $|I|,|I_0|\ge c_\varepsilon'>0$, this map is Lipschitz
with constant $L_\varepsilon=\tfrac{2}{3}(\tfrac{3}{2})^{2/3}(c_\varepsilon')^{-1/3}$.
Hence:
\[
  B_\varepsilon(h) := \|\phi-\phi_0\|_{C^0} \le L_\varepsilon\tilde A_\varepsilon(h) = \mathcal{O}(h).
\]
\end{lemma}

\begin{lemma}[$C^1$-stability of $\phi$, \Cond]
\label{lem:phi-C1}
On each $K_\varepsilon^\pm$:
$\phi'=p/\phi$, $\phi_0'=p_0/\phi_0$
($|\phi|\ge c_\varepsilon''>0$, $|\phi_0|\ge d_\varepsilon>0$).
\[
  |\phi'-\phi_0'|
  \le \frac{A_\varepsilon}{c_\varepsilon''}
  + \frac{p_0^{\max}B_\varepsilon}{c_\varepsilon''d_\varepsilon}
  \to 0
\]
uniformly.
Since $\phi_0'\ge c_\varepsilon/d_\varepsilon>0$:
$\inf_{K_\varepsilon}|\phi'(\cdot;h)|\ge\tfrac{1}{2}\inf|\phi_0'|>0$.
\end{lemma}

\section{CFU Jacobian in 1D}
\label{sec:cfu-jacobian}
%% ============================================================

\begin{lemma}[CFU Jacobian reduction in 1D, \Cond]
\label{lem:cfu-jacobian-1d}
In the 1D A$_2$ setting with generating function
$\Phi(x,\theta)=\phi(x;h)\theta-\tfrac{1}{3}\theta^3$:
$\partial^2_{x\theta}\Phi=\phi'(x;h)$,
hence $|\det(dt/d\beta)|=|\phi'|\ge\delta_1>0$ (Lemma~\ref{lem:phi-C1}).
\end{lemma}

\begin{remark}[Normalization of CFU coordinates]
\label{rem:cfu-coords}
The form $\Phi=\phi\theta-\tfrac{1}{3}\theta^3$ holds in the normalized
CFU phase coordinates produced by \textbf{thm:CFU-geom}.
In other coordinates an additional bounded and
bounded-away-from-zero factor appears,
not affecting uniform positivity of (A2).
\end{remark}

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
Under Assumption~\ref{ass:boundary-away},
the smooth-eigenbranch hypothesis,
and hypotheses (H1)--(H3) of Lemma~\ref{lem:specgap},
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly on $K_\varepsilon$ as $h\to 0$.
