\title[A$_2$-Stability for PSWF Amplitudes]{%
  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
(conditions (A1)--(A3)) uniformly as $h:=c^{-1}\to 0$.

\medskip
\noindent\textbf{Recall (Assumption~A2).}
A Lagrangian $\Lambda_h$ satisfies Assumption~A2 if:
(A1) the cubic coefficient of the A$_2$-normal form is
  $c_3(h)\ge\delta_0>0$ uniformly;
(A2) $|\det(dt/d\beta)|\ge\delta_1>0$ uniformly;
(A3) unique fold on $\mathrm{supp}\,\chi$.
In 1D: a unique turning point with $\partial_\xi\sigma=0$,
$\partial^2_\xi\sigma\ne 0$ (A$_2$-type).

\medskip
\noindent\textbf{Main technical contribution.}
Lemmas~\ref{lem:energy-comparison}--\ref{lem:phi-C1}
establish uniform $C^1$-control of the Olver phase
\[
  \phi(x;h)=\mathrm{sgn}(x{-}x_0(h))\cdot
  \Bigl(\tfrac{3}{2}\int_{x_0(h)}^x p(s;h)\,ds\Bigr)^{2/3},
  \quad p:=\sqrt{E_n(h)-V(x)}.
\]
No ODE, no Gr\"onwall, no Wasow expansion.
\end{abstract}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF]
$\psi_n$ satisfy $(h^2\mathcal{L}_{h^{-1}})\psi_n=E_n(h)\psi_n$
with $V(x)=x^2/(1-x^2)$, $E_n(h):=\chi_n h^2$.
Turning points: $V(x_\pm(h))=E_n(h)$.
\end{definition}

\begin{remark}[Shape of $V$ and unique turning point]
\label{rem:V-shape}
$V(x)=x^2/(1-x^2)$ satisfies:
\begin{enumerate}[{\rm(i)}]
\item $V(0)=0$, $V(x)\to+\infty$ as $x\to\pm 1$.
\item $V$ is \emph{strictly increasing} on $(0,1)$
  and \emph{strictly decreasing} on $(-1,0)$
  (hence symmetric: $V(-x)=V(x)$).
\item For any $\lambda_n>0$,
  the equation $V(x)=\lambda_n$ has exactly one solution
  in $(0,1-\delta_{\rm bdy})$ and exactly one in $(-1+\delta_{\rm bdy},0)$,
  provided $\lambda_n<V(1-\delta_{\rm bdy})$.
  The latter condition is guaranteed by Assumption~\ref{ass:boundary-away}
  and the fact that $\lambda_n\in(0,1)$ while
  $V(1-\delta_{\rm bdy})>1$ for small $\delta_{\rm bdy}$.
\item If $\mathrm{supp}\,\chi$ is contained in one half-interval
  $(0,1-\delta_{\rm bdy})$ or $(-1+\delta_{\rm bdy},0)$,
  there is a unique turning point on $\mathrm{supp}\,\chi$.
  For the general case (supp $\chi$ crossing 0),
  the two turning points are on opposite sides of $0$
  and are treated as separate A$_2$-folds.
\end{enumerate}
\end{remark}

\begin{remark}[Geometric and spectral positivity of $p$]
\label{rem:p-positivity}
Positivity $p(x;h)>0$ on $K_\varepsilon$ has two components:

\medskip
\noindent\textit{(i) Geometric part (\Rig).}
Let $x_0^{(0)}$ be the limiting turning point on the relevant half-interval.
$K_\varepsilon=\mathrm{supp}\,\chi\setminus U_\varepsilon$ excludes a
neighborhood of $x_0^{(0)}$, so by strict monotonicity of $V$
(Remark~\ref{rem:V-shape}(ii)):
\[
  \inf_{x\in K_\varepsilon}|\lambda_n-V(x)|\ge c_\varepsilon^{\rm geo}>0.
\]
Since $\lambda_n-V(x)>0$ on the classically allowed region,
this gives $\lambda_n-V(x)\ge c_\varepsilon^{\rm geo}$ on $K_\varepsilon$.

\medskip
\noindent\textit{(ii) Spectral part (\Cond).}
$E_n(h)\to\lambda_n$ smoothly, so for $h\le h_0$:
$E_n(h)-V(x)\ge\tfrac{1}{2}c_\varepsilon^{\rm geo}>0$.
Define $\delta_\xi(\varepsilon):=\sqrt{c_\varepsilon^{\rm geo}/2}$;
then $p(x;h)\ge\delta_\xi>0$ on $K_\varepsilon$ for $h\le h_0$.

\medskip
\textbf{All constants
$c_\varepsilon^{\rm geo},\delta_\xi,c_\varepsilon,c_\varepsilon',c_\varepsilon'',d_\varepsilon,L_\varepsilon$
are independent of $h$.}
\end{remark}

\begin{remark}[Turning-point shift]
\label{rem:eigenbranch}
$E_n(h)\to\lambda_n$ smoothly;
$V(x_0(h))=E_n(h)$, $V'(x_0^{(0)})\ne 0$ (strict monotonicity on the
half-interval) $\Rightarrow$ $x_0(h)\to x_0^{(0)}$ smoothly (IFT).
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$
independently of $h$.
Consequences: $V,V',V''\in L^\infty$; boundary layers excluded;
$\lambda_n<V(1-\delta_{\rm bdy})$ (ensures turning point exists).
\end{assumption}

\section{Olver and the CFU Gap}
\label{sec:olver}
%% ============================================================

Olver \cite{Olver1974} Ch.~10--11:
$\psi_n\sim h^{-1/4}a\,\mathrm{Ai}(h^{-2/3}\phi)+\mathcal{O}(h^\infty)$,
uniform on $\mathrm{supp}\,\chi$.

\begin{remark}[Olver--CFU gap]
Olver does not provide Lagrangian/FIO structure or uniform $C^1$-control of $\phi$.
Warning: $f_h=x+h\sin(x/h)\to x$ uniformly but $f_h'$ does not.
\end{remark}

\section{Branch Selection}
\label{sec:branch}
%% ============================================================

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
Fix $\varepsilon>0$.
$\mathrm{WF}_h(\psi_n)\subset\{\xi^2+V=E_n(h)\}=\Lambda_h^+\cup\Lambda_h^-$
(\cite{Zworski2012} Ch.~12).
On $K_\varepsilon$: $\Lambda_h^\pm\subset\{\pm\xi\ge\delta_\xi\}$.
Microlocal partition of unity yields $\Pi^+=\mathrm{Op}_h(\sigma^+)$ with:
(i) $\chi\psi_n=\Pi^+(\chi\psi_n)+\mathcal{O}(h^\infty)$;
(ii) $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\textit{Caveat:} holds only on $K_\varepsilon$.
\end{proposition}

%% ============================================================
\section{Global Geometry: (P1) and (P3)}
\label{sec:phi-conditions}
%% ============================================================

\begin{remark}[(P1)]
$\xi=p>0$ on $K_\varepsilon$ $\Rightarrow$ $\phi$ global generating function. \Cond.
\end{remark}

\begin{remark}[(P3): unique turning point; spectral gap]
\label{rem:p3}
\textit{Unique turning point (\Rig).}
On each half-interval,
$V$ is strictly monotone (Remark~\ref{rem:V-shape}(ii)--(iii)),
so $E_n(h)-V(x)=0$ has exactly one solution $x_0(h)$
on $\mathrm{supp}\,\chi\cap(0,1-\delta_{\rm bdy})$
(resp.\ on the left half;
Assumption~\ref{ass:boundary-away} ensures $\lambda_n<V(1-\delta_{\rm bdy})$).

\medskip
\noindent\textit{Spectral gap (\Cond).}
\begin{equation}\label{eq:specgap}
  \mathrm{dist}(E_n(h),\sigma(\mathcal{L}_h)\setminus\{E_n(h)\})\ge Ch.
\end{equation}
The quantitative bound $E_{n+1}(h)-E_n(h)\sim Ch$ follows from
the \emph{Bohr--Sommerfeld quantization rule} for $\mathcal{L}_h$:
\[
  \int_{x_-(h)}^{x_+(h)} p(x;h)\,dx = (n+\tfrac{1}{2})\pi h + \mathcal{O}(h^2),
\]
which gives $E_{n+1}-E_n = 2\pi h/T(E_n)+\mathcal{O}(h^2)$
where $T(E_n)=\int_{x_-}^{x_+}p^{-1}\,dx$ is the classical period;
see \cite{Zworski2012} Ch.~9.
\end{remark}

\end{lemma}

\section{CFU Construction on $K_\varepsilon^\pm$ (Separate)}
\label{sec:cfu-components}
%% ============================================================

\begin{remark}[CFU on each component]
\label{rem:cfu-components}
$K_\varepsilon$ has two components
$K_\varepsilon^\pm:=K_\varepsilon\cap\{\pm(x-x_0(h))>0\}$.
On $K_\varepsilon^+$: $I>0$, $\phi=(\tfrac{3}{2}I)^{2/3}>0$, smooth.
On $K_\varepsilon^-$: $I<0$, $\phi=-(\tfrac{3}{2}|I|)^{2/3}<0$, smooth.
The CFU construction \cite{CFU1957} is performed
\emph{separately on each component};
no global branch of $z^{2/3}$ across the turning point is needed.
The two branches are matched at $x_0(h)$ by
\textbf{thm:CFU-geom} (Paper~XVI).
\end{remark}

\section{The Differentiated Olver Estimate (Decomposed)}
\label{sec:diff-olver}
%% ============================================================

\begin{lemma}[Monotonicity of $I$, \Rig]
\label{lem:I-monotone}
$p>0$ on $K_\varepsilon$, so $I(x;h)=\int_{x_0(h)}^x p\,ds$ is strictly monotone,
$\mathrm{sgn}(I)=\mathrm{sgn}(x-x_0(h))$,
$|I|\ge\delta_\xi|x-x_0(h)|$.
\end{lemma}

\begin{lemma}[Energy comparison, \Cond]
\label{lem:energy-comparison}
Since $p,p_0\ge\delta_\xi,c_\varepsilon>0$
(Remark~\ref{rem:p-positivity}):
\[
  |p-p_0|=\frac{|E_n(h)-\lambda_n|}{p+p_0}
  \le\frac{|E_n-\lambda_n|}{\delta_\xi+c_\varepsilon}
  =:A_\varepsilon(h)=\mathcal{O}(h).
\]
(Lipschitz of $\sqrt{\cdot}$ fails at $p_0=0$;
rationalization via $p+p_0\ge\delta_\xi+c_\varepsilon$ is the correct argument.)
\end{lemma}

\begin{lemma}[Integral stability, \Cond]
\label{lem:integral-stability}
\[
  |I(x;h)-I_0(x)|\le\|p\|_{C^0}|x_0(h)-x_0^{(0)}|
  +|K_\varepsilon|A_\varepsilon(h)=:\tilde A_\varepsilon(h)=\mathcal{O}(h).
\]
\end{lemma}

\begin{lemma}[Non-degeneracy, \Cond]
\label{lem:nondeg}
By Lemma~\ref{lem:I-monotone} and $|x-x_0(h)|\ge\varepsilon/2$:
\[
  |I|,|I_0|\ge c_\varepsilon':=\delta_\xi\varepsilon/2>0;
  \quad
  |\phi|\ge c_\varepsilon'':=(\tfrac{3}{4}\delta_\xi\varepsilon)^{2/3};
  \quad
  |\phi_0|\ge d_\varepsilon:=(\tfrac{3}{4}c_\varepsilon\varepsilon)^{2/3}>0.
\]
\end{lemma}

\begin{lemma}[$C^0$-stability of $\phi$, \Cond]
\label{lem:phi-C0}
On each $K_\varepsilon^\pm$, $I$ has fixed sign
(Remark~\ref{rem:cfu-components}),
so $I\mapsto(\tfrac{3}{2}I)^{2/3}$ is smooth.
Since $|I|,|I_0|\ge c_\varepsilon'>0$ (Lemma~\ref{lem:nondeg}),
this map is Lipschitz on $[c_\varepsilon',\infty)$
with constant
$L_\varepsilon=\tfrac{2}{3}(\tfrac{3}{2})^{2/3}(c_\varepsilon')^{-1/3}$.
Therefore:
\[
  B_\varepsilon(h):=\|\phi-\phi_0\|_{C^0(K_\varepsilon)}
  \le L_\varepsilon\tilde A_\varepsilon(h)=\mathcal{O}(h).
\]
\end{lemma}

\begin{lemma}[$C^1$-stability of $\phi$, \Cond]
\label{lem:phi-C1}
On each $K_\varepsilon^\pm$:
$\phi'=p/\phi$, $\phi_0'=p_0/\phi_0$
(well-defined: $|\phi|\ge c_\varepsilon''>0$, $|\phi_0|\ge d_\varepsilon>0$).
\[
  |\phi'-\phi_0'|
  \le\frac{A_\varepsilon}{c_\varepsilon''}
  +\frac{p_0^{\max}B_\varepsilon}{c_\varepsilon''d_\varepsilon}\to 0
\]
uniformly (Lemmas~\ref{lem:energy-comparison},~\ref{lem:phi-C0}).
Since $\phi_0'\ge c_\varepsilon/d_\varepsilon>0$:
$\inf_{K_\varepsilon}|\phi'(\cdot;h)|\ge\tfrac{1}{2}\inf|\phi_0'|>0$.
\end{lemma}

\section{CFU Jacobian in 1D}
\label{sec:cfu-jacobian}
%% ============================================================

\begin{lemma}[CFU Jacobian reduction in 1D, \Cond]
\label{lem:cfu-jacobian-1d}
In the 1D A$_2$ setting with
generating function $\Phi(x,\theta)=\phi(x;h)\theta-\tfrac{1}{3}\theta^3$,
the mixed Hessian is
$\partial^2_{x\theta}\Phi=\phi'(x;h)$.
Hence $|\det(dt/d\beta)|=|\phi'(x;h)|\ge\delta_1>0$ on $K_\varepsilon$
(Lemma~\ref{lem:phi-C1}).
\end{lemma}

\begin{remark}[Normalization of CFU coordinates]
\label{rem:cfu-coords}
The identification $|\det(dt/d\beta)|=|\phi'|$ holds in the
\emph{normalized CFU phase coordinates} in which the generating function
takes the Airy form $\Phi=\phi\theta-\tfrac{1}{3}\theta^3$.
This normalization is provided by \textbf{thm:CFU-geom} of Paper~XVI,
which produces precisely these coordinates on each $K_\varepsilon^\pm$.
In other coordinate systems the Jacobian acquires an additional
bounded and bounded-away-from-zero factor coming from the
coordinate change, which does not affect the uniform positivity of (A2).
\end{remark}

\section{Target Theorem}
\label{sec:target}
%% ============================================================

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
Under Assumption~\ref{ass:boundary-away},
smooth-eigenbranch hypothesis, and~\eqref{eq:specgap},
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly on $K_\varepsilon$ as $h\to 0$.
\end{theorem}

\begin{remark}[Dependency chain]
\[
  \bigl[\lambda_n{-}V\ge c^{\rm geo},\;
  E_n\to\lambda_n,\;
  V'\ne 0,\;
  \text{\eqref{eq:specgap}},\;
  \mathrm{WF}_h\subset\Lambda_h^\pm\bigr]
  \Rightarrow
  \bigl[\text{Lems.}~\ref{lem:energy-comparison}\text{--}\ref{lem:phi-C1}\bigr]
  \Rightarrow

  \Rightarrow
  \text{Prop.}~\ref{prop:a2-from-lemmas}
  \Rightarrow
  \text{Thm.}~\ref{thm:xvii-main}.

\end{remark}
