  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 uniformly as $h:=c^{-1}\to 0$.

\medskip
\noindent\textbf{Assumption~A2.}
(A1) $c_3(h)\ge\delta_0>0$;
(A2) $|\det(dt/d\beta)|\ge\delta_1>0$;
(A3) unique non-degenerate fold ($V'(x_0)\ne 0$; standard A$_2$ condition).

\medskip
\noindent\textbf{Three hard cores.}
\begin{enumerate}[\rm(C1)]
\item Spectral gap: $E_{n+1}(h)-E_n(h)\ge Ch$
  (Theorem~\ref{thm:specgap} + Lemma~\ref{lem:wkb-validity}).
\item Smooth eigenbranch: $h\mapsto E_n(h)$ real-analytic
  (Theorem~\ref{thm:eigenbranch}).
\item Olver $C^1$-stability: $\inf_{K_\varepsilon}|\phi'(\cdot;h)|\ge\delta>0$
  (Lemmas~\ref{lem:C0-airy}--\ref{lem:tp-perturbation}).
\end{enumerate}
\end{abstract}

\begin{remark}[CFU-geom: global gluing and Maslov]
\label{rem:cfu-geom-nontrivial}
In 1D with a single non-degenerate fold (Hypothesis~\ref{hyp:a2-fold}):
(i) Maslov index $+1$ across $x_0(h)$ (\cite{Zworski2012} Ch.~12).
(ii) No phase mismatch: $\phi$ smooth with fixed sign on each $K_\varepsilon^\pm$.
(iii) \emph{Phase fixation:}
  the overall phase freedom (a constant shift) is fixed by the
  \emph{local Airy normal form normalization} provided by \textbf{thm:CFU-geom};
  this is not a canonical choice in general, but is
  determined unambiguously by the normalization condition
  $\Phi(x,\theta)=\phi(x;h)\theta-\tfrac13\theta^3$
  on each $K_\varepsilon^\pm$.
  CFU-gluing is therefore well-defined up to the sign convention
  fixed by thm:CFU-geom.
(iv) Multiple turning points: each treated separately
  (Remark~\ref{rem:V-shape}(iv)).
\end{remark}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF, classical and semiclassical momentum]
\label{def:pswf-p0}
$\psi_n$ satisfy $(h^2\mathcal{L}_{h^{-1}})\psi_n=E_n(h)\psi_n$,
$V(x)=x^2/(1-x^2)$, $E_n(h):=\chi_n h^2$.

\noindent\textbf{Limiting energy:}
$\lambda_n:=\lim_{h\to 0}E_n(h)$ (Theorem~\ref{thm:eigenbranch}).

\noindent\textbf{Classical momentum:}
\begin{equation}\label{eq:p0-def}
  p_0(x):=\sqrt{\lambda_n-V(x)},\quad x\in\{\lambda_n>V(x)\}.
\end{equation}

\noindent\textbf{Semiclassical momentum:}
$p(x;h):=\sqrt{E_n(h)-V(x)}$ for $x\in K_\varepsilon$, $h\le h_0$.
\end{definition}

\begin{hypothesis}[Non-degenerate fold: standard A$_2$ condition]
\label{hyp:a2-fold}
The limiting turning point $x_0^{(0)}$ satisfies:
\begin{enumerate}[{\rm(F1)}]
\item $V'(x_0^{(0)})\ne 0$ (non-degenerate fold; IFT applicable).
  \emph{Verification:} $V'(x)=2x/(1-x^2)^2\ne 0$
  for $x\in(0,1-\delta_{\rm bdy})$. \Rig.
\end{enumerate}
\begin{remark}
(F1) is the \emph{standard} A$_2$ fold condition in the
Hamilton-Jacobi / CFU sense \cite{Zworski2012}:
a non-degenerate turning point means $V'(x_0)\ne 0$,
not $V''(x_0)\ne 0$.
The positivity $V''(x_0^{(0)})=2(1+3x_0^{(0)2})/(1-x_0^{(0)2})^3>0$
holds for PSWF (\Rig) and is recorded here for reference,
but is \emph{not} imposed as a hypothesis since it follows from
(F1) and the analytic structure of $V$, and is not required
for A$_2$ classification.
\end{remark}
\end{hypothesis}

\begin{remark}[Shape of $V$]
\label{rem:V-shape}
\begin{enumerate}[{\rm(i)}]
\item $V(0)=0$, $V\to+\infty$ as $x\to\pm 1$, $V(-x)=V(x)$.
\item $V'=2x/(1-x^2)^2>0$ on $(0,1)$; $<0$ on $(-1,0)$. \Rig.
\item Unique turning point per half-interval for $\lambda_n<V(1-\delta_{\rm bdy})$. \Rig.
\item Multiple turning points: each a separate A$_2$-fold. \Rig.
\end{enumerate}
\end{remark}

\begin{remark}[Geometric positivity]
\label{rem:geo-positivity}
$c_\varepsilon^{\rm geo}:=\inf_{K_\varepsilon}(\lambda_n-V)>0$. \Rig.
\textbf{All constants are $h$-independent.}
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$,
indep.\ of $h$; $V,V',V''\in L^\infty$; $\lambda_n<V(1-\delta_{\rm bdy})$.
\end{assumption}

\section{Hard Core (C2): Smooth Eigenbranch}
\label{sec:eigenbranch}
%% ============================================================

\begin{theorem}[Smooth eigenbranch, \Cond]
\label{thm:eigenbranch}
Assume:
\begin{enumerate}[{\rm(E1)}]
\item \textit{(Fixed domain.)} $\mathcal{D}=H^2(-1,1)\cap H^1_0(-1,1)$,
  $h$-independent. \Rig\ ($V\in L^\infty$; boundary cond.\ independent of $h$).
\item \textit{(Kato--Rellich: relative boundedness.)}
  Write $h^2\mathcal{L}_{h^{-1}}=\mathcal{L}_0+hA_1+h^2A_2$
  where $\mathcal{L}_0=-d^2/dx^2$ and
  $A_j$ are lower-order differential operators with $L^\infty$-coefficients.
  Each $A_j$ is $\mathcal{L}_0$-relatively bounded with relative bound $0$
  (Kato--Rellich; \cite{Kato1966} Ch.~V.\S4),
  i.e.\ $\|A_j u\|\le a\|\mathcal{L}_0 u\|+b\|u\|$ with $a<1$.
  \emph{Verification:} $A_1,A_2$ are multiplication operators
  (bounded on $L^2$ by Assumption~\ref{ass:boundary-away}),
  hence $\mathcal{L}_0$-relatively bounded with $a=0$.
\item \textit{(Norm-resolvent analyticity.)}
  By (E1)--(E2) and the resolvent identity:
  $(\mathcal{L}_h-z)^{-1}=(\mathcal{L}_0-z)^{-1}[I+(hA_1+h^2A_2)(\mathcal{L}_0-z)^{-1}]^{-1}$;
  the Neumann series converges in $\mathcal{B}(L^2)$
  for $|h|$ small and $z\notin\sigma(\mathcal{L}_0)$,
  and is analytic in $h$. \Cond.
\item \textit{(Simple eigenvalue.)} $E_n(h_0)$ simple: SL oscillation. \Rig.
\end{enumerate}
Hence $h^2\mathcal{L}_{h^{-1}}$ is a Kato analytic family of type~(A);
$h\mapsto E_n(h)$ is real-analytic \cite{Kato1966} Ch.~VII.\S2;
$E_n(h)\to\lambda_n$;
$x_0(h)$ real-analytic by IFT + (F1).
\end{theorem}

\begin{remark}[Why relative boundedness closes (E2)]
The key point is that $A_j$ are \emph{multiplication operators}
(bounded, not merely relatively bounded with $a<1$),
so the Neumann series expansion of the resolvent converges
uniformly in $\mathcal{B}(L^2)$ for all $|h|\le h_0$,
not just for $h$ near a fixed $h_0$.
This is stronger than what polynomial coefficients alone give
(which would only yield strong resolvent analyticity).
\end{remark}

\section{Hard Core (C1): Spectral Gap}
\label{sec:specgap}
%% ============================================================

\begin{lemma}[Uniform WKB quantization validity, \Cond]
\label{lem:wkb-validity}
For $V(x)=x^2/(1-x^2)$ on $(-1,1)$ with
$\mathrm{supp}\,\chi\subset(-1+\delta_{\rm bdy},1-\delta_{\rm bdy})$:
\begin{enumerate}[{\rm(W1)}]
\item \textit{(Single well.)} $V$ has a unique minimum at $x=0$;
  $E_n(h)\in(V(0),V(1-\delta_{\rm bdy}))=(0,V(1-\delta_{\rm bdy}))$.
  The classically allowed interval $(x_-(h),x_+(h))$ is connected
  and stays away from $\pm 1$. \Rig\ (Assumption~\ref{ass:boundary-away}).
\item \textit{(No tunneling correction.)} In the single-well regime,
  tunneling is exponentially small: $\mathcal{O}(e^{-c/h})$
  (standard WKB; \cite{Zworski2012} Ch.~6);
  does not affect $\mathcal{O}(h^2)$ error in Bohr--Sommerfeld.
\item \textit{(WKB uniformity.)} Olver's theory
  (\cite{Olver1974} Ch.~10--11) gives uniform WKB expansion
  on $\mathrm{supp}\,\chi$;
  in particular:
  $S(E_n(h)):=\int_{x_-(h)}^{x_+(h)}\sqrt{E_n-V}\,dx
  = (n+\tfrac12)\pi h + \mathcal{O}(h^2)$
  uniformly for $n$ in a bounded range.
\end{enumerate}
\end{lemma}

\begin{lemma}[Bohr--Sommerfeld inversion, \Cond]
\label{lem:bs-inversion}
Under Theorem~\ref{thm:specgap}(H2)--(H3)
and Lemma~\ref{lem:wkb-validity}(W3):
\begin{enumerate}[{\rm(i)}]
\item $S'(E)=T(E)/2>0$: strict monotonicity near $\lambda_n$.
\item $S$ is a local $C^\infty$ diffeomorphism.
\item \textit{(MVT with uniform denominator.)} For
  $\xi\in(E_n,E_{n+1})\to\lambda_n$ as $h\to 0$:
  $S'(\xi)\ge T(\lambda_n)/4>0$ for $h\le h_0$ (by (i) and continuity); hence:
\[
  E_{n+1}-E_n
  =\frac{S(E_{n+1})-S(E_n)}{S'(\xi)}
  =\frac{\pi h+\mathcal{O}(h^2)}{T(\lambda_n)/2+\mathcal{O}(h)}
  =\frac{2\pi h}{T(\lambda_n)}+\mathcal{O}(h^2).
\]
\end{enumerate}
\end{lemma}

\begin{theorem}[Spectral gap, \Cond]
\label{thm:specgap}
Assume:
\begin{enumerate}[{\rm(H1)}]
\item 1D integrability: automatic. \Rig.
\item $0<T(\lambda_n)<\infty$: square-root singularity integrable at simple zeros. \Cond.
\item $\partial_E T(\lambda_n)=-\tfrac12\oint(\lambda_n-V)^{-3/2}\,dx\ne 0$:
  finite and nonzero by (H2). \Cond.
\end{enumerate}
By Lemmas~\ref{lem:wkb-validity} and~\ref{lem:bs-inversion}:
\begin{equation}\label{eq:specgap}
  E_{n+1}(h)-E_n(h)
  =\frac{2\pi h}{T(\lambda_n)}+\mathcal{O}(h^2)\ge Ch,
  \quad C=\frac{\pi}{T(\lambda_n)}>0.
\end{equation}
$\mathrm{dist}(E_n,\sigma\setminus\{E_n\})\ge Ch$.
\end{theorem}

\section{Branch Selection and Uniform Ellipticity}
\label{sec:branch}
%% ============================================================

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
\textit{(i)} $|E_n-\lambda_n|=\mathcal{O}(h)$
$\Rightarrow p\ge\delta_\xi:=\sqrt{c_\varepsilon^{\rm geo}/2}>0$ on $K_\varepsilon$.

\textit{(ii)} $\mathrm{WF}_h(\psi_n)\subset\{\xi^2+V=E_n\}$
(\cite{Zworski2012} Ch.~12;
canonical relation: eigenfunction $\to$ Lagrangian distribution,
stable under $h\to 0$ by spectral gap);
spectral gap excludes mode mixing;
$\chi\psi_n=\Pi^+(\chi\psi_n)+\mathcal{O}(h^\infty)$,
$\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\end{proposition}

\begin{remark}[Microlocal bridge: eigenfunction to FIO]
\label{rem:micro-bridge}
The inclusion $\mathrm{WF}_h(\psi_n)\subset\Lambda_h$
follows from ellipticity of $h^2\mathcal{L}_{h^{-1}}-E_n(h)$
outside $\Lambda_h$ (standard pseudodiff.\ argument;
\cite{Zworski2012} Thm.~12.1).
Stability under $h\to 0$ uses Theorem~\ref{thm:eigenbranch}
(smooth eigenbranch) and Theorem~\ref{thm:specgap}
(spectral gap prevents mode mixing).
This closes the ``PDE$\to$FIO'' bridge:
$\psi_n$ is genuinely a Lagrangian distribution,
not merely asserted to be one.
\end{remark}

\section{Hard Core (C3): Olver $C^1$-Stability}
\label{sec:olver-C1}
%% ============================================================

\begin{lemma}[Uniform ellipticity, \Cond]
\label{lem:uniform-ellipticity}
$p\ge\delta_\xi$, $p_0\ge c_\varepsilon:=\sqrt{c_\varepsilon^{\rm geo}}$,
$p+p_0\ge\delta_\xi+c_\varepsilon>0$, uniform in $h$.
\end{lemma}

\begin{lemma}[Energy comparison, \Cond]
\label{lem:energy-comparison}
$|p-p_0|=|E_n-\lambda_n|/(p+p_0)\le|E_n-\lambda_n|/(\delta_\xi+c_\varepsilon)=:A_\varepsilon=\mathcal{O}(h)$.
\end{lemma}

\begin{lemma}[Integral stability, \Cond]
\label{lem:integral-stability}
$|I-I_0|\le\|p\|_{C^0}|x_0(h)-x_0^{(0)}|+|K_\varepsilon|A_\varepsilon=:\tilde A_\varepsilon=\mathcal{O}(h)$.
\end{lemma}

\begin{lemma}[Non-degeneracy, \Cond]
\label{lem:nondeg}
$|I|,|I_0|\ge c_\varepsilon':=\delta_\xi\varepsilon/2>0$;
$|\phi|\ge c_\varepsilon''$; $|\phi_0|\ge d_\varepsilon>0$.
\end{lemma}

\begin{lemma}[$C^0$-Airy stability, \Cond]
\label{lem:C0-airy}
$\|\phi-\phi_0\|_{C^0}\le L_\varepsilon\tilde A_\varepsilon=\mathcal{O}(h)$;
$L_\varepsilon=\tfrac23(\tfrac32)^{2/3}(c_\varepsilon')^{-1/3}$.
\end{lemma}

\begin{lemma}[Derivative commuting, \Cond]
\label{lem:deriv-commuting}
On $K_\varepsilon^\pm$:
$\phi'=p/\phi$, $\phi_0'=p_0/\phi_0$.
$\|\phi'-\phi_0'\|_{C^0}=\mathcal{O}(h)$ by quotient estimate.
\emph{Exchange valid:} $\phi'=p/\phi$ is algebraic;
$C^0$-control of $\phi$ directly gives $C^0$-control of $\phi'$.
(Warning: $h\sin(x/h)\to 0$ in $C^0$ but $\cos(x/h)\not\to 0$
does not apply here.)
\end{lemma}

\begin{lemma}[Turning-point perturbation, \Cond]
\label{lem:tp-perturbation}
$|x_0(h)-x_0^{(0)}|=(E_n-\lambda_n)/V'(x_0^{(0)})+\mathcal{O}(h^2)=\mathcal{O}(h)$.
\end{lemma}

\begin{corollary}[$C^1$-stability]
\label{cor:phi-C1}
$\inf_{K_\varepsilon}|\phi'(\cdot;h)|\ge\tfrac12\inf|\phi_0'|>0$;
$\phi_0'=p_0/\phi_0\ge c_\varepsilon/d_\varepsilon>0$.
\end{corollary}

\begin{lemma}[CFU Jacobian, \Cond]
\label{lem:cfu-jacobian-1d}

\end{lemma}
