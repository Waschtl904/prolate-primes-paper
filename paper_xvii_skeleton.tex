  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI uniformly
as $h:=c^{-1}\to 0$.

\medskip
\noindent\textbf{Recall (Assumption~A2).}
(A1) $c_3(h)\ge\delta_0>0$;
(A2) $|\det(dt/d\beta)|\ge\delta_1>0$;
(A3) unique fold.

\medskip
\noindent\textbf{Three hard cores.}
All of A2-stability reduces to three nontrivial analytic inputs:
\begin{enumerate}[\rm(C1)]
\item \textit{Spectral gap:}
  $E_{n+1}(h)-E_n(h)\ge Ch>0$
  (Theorem~\ref{thm:specgap}).
\item \textit{Smooth eigenbranch:}
  $h\mapsto E_n(h)$ is real-analytic
  (Theorem~\ref{thm:eigenbranch}).
\item \textit{Olver $C^1$-stability:}
  $\inf_{K_\varepsilon}|\phi'(\cdot;h)|\ge\delta>0$
  (Lemmas~\ref{lem:C0-airy}--\ref{lem:tp-perturbation}).
\end{enumerate}
Geometry, microlocal bookkeeping, and CFU structure
are consequences once (C1)--(C3) are in hand.
\end{abstract}

\label{sec:input}
%% ============================================================

\begin{itemize}
\item \textbf{thm:CFU-geom} (global gluing theorem):
  on each $K_\varepsilon^\pm$, an A$_2$-fold produces
  normalized Airy phase coordinates
  $\Phi(x,\theta)=\phi(x;h)\theta-\tfrac13\theta^3$
  via a smooth canonical transformation;
  the two branches are glued across $x_0(h)$
  with consistent Maslov index and trivial phase mismatch
  (non-trivial: see Remark~\ref{rem:cfu-geom-nontrivial}).
\item \textbf{thm:CFU-symbol}:
  $I^{-1/4,0}(\Lambda_0,\Lambda_1)\to S^{-1/2}_{1,0}(h)$, uniform.
\item \textbf{Assumption~A2}: (A1)--(A3) as above.
\item \textbf{hyp:rigidity}: verified here for PSWF.
\end{itemize}

\begin{remark}[CFU-geom as global gluing theorem]
\label{rem:cfu-geom-nontrivial}
\textbf{thm:CFU-geom} is not a local statement.
It requires:
(i) consistent orientation of the Maslov bundle across $x_0(h)$;
(ii) absence of phase mismatch between the $K_\varepsilon^+$ and $K_\varepsilon^-$ branches;
(iii) global trivialization of the Lagrangian $\Lambda_h$
  (no Maslov obstruction in 1D: $\pi_1(\mathrm{Lag}(1))=\mathbb{Z}$,
  but 1D turning-point structure fixes Maslov index to $+1$).
In 1D with a single non-degenerate turning point,
(i)--(iii) all hold automatically;
see \cite{Zworski2012} Ch.~12 and \cite{MelroseU1979}.
For multiple turning points, each is treated separately
(Remark~\ref{rem:V-shape}(iv)).
\end{remark}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF]
$\psi_n$ satisfy $(h^2\mathcal{L}_{h^{-1}})\psi_n=E_n(h)\psi_n$,
$V(x)=x^2/(1-x^2)$, $E_n(h):=\chi_n h^2$.
\end{definition}

\begin{remark}[Shape of $V$ and unique turning points]
\label{rem:V-shape}
\begin{enumerate}[{\rm(i)}]
\item $V(0)=0$, $V\to+\infty$ as $x\to\pm 1$, $V(-x)=V(x)$.
\item \textit{Explicit derivative:}
  $V'(x)=2x/(1-x^2)^2>0$ on $(0,1)$,
  $<0$ on $(-1,0)$. \Rig.
\item \textit{Unique turning point per half-interval:}
  $V(x)=\lambda_n$ has exactly one solution in $(0,1-\delta_{\rm bdy})$
  provided $\lambda_n<V(1-\delta_{\rm bdy})$
  (Assumption~\ref{ass:boundary-away}). \Rig.
\item For $\mathrm{supp}\,\chi$ crossing $0$:
  two turning points on opposite sides; each a separate A$_2$-fold. \Rig.
\end{enumerate}
\end{remark}

\begin{remark}[Geometric positivity of $\lambda_n-V$ on $K_\varepsilon$]
\label{rem:geo-positivity}
$K_\varepsilon:=\mathrm{supp}\,\chi\setminus U_\varepsilon$
excludes an $\varepsilon$-neighborhood of $x_0^{(0)}$.
By strict monotonicity:
$c_\varepsilon^{\rm geo}:=\inf_{K_\varepsilon}(\lambda_n-V(x))>0$. \Rig.
\textbf{All constants $c_\varepsilon^{\rm geo},\delta_\xi,c_\varepsilon',
c_\varepsilon'',d_\varepsilon,L_\varepsilon$ are $h$-independent.}
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$,
independently of $h$.
Implies: $V,V',V''\in L^\infty$; $\lambda_n<V(1-\delta_{\rm bdy})$.
\end{assumption}

\section{Hard Core (C2): Smooth Eigenbranch}
\label{sec:eigenbranch}
%% ============================================================

\begin{theorem}[Smooth eigenbranch, \Cond]
\label{thm:eigenbranch}
Assume:
\begin{enumerate}[{\rm(E1)}]
\item \textit{(Analytic family.)}
  $\mathcal{L}_h$ is a self-adjoint analytic family of type~(A)
  in the sense of Kato \cite{Kato1966}
  for $h\in(0,h_0)$.
\item \textit{(Simple eigenvalue.)}  
  $E_n(h_0)$ is a simple eigenvalue of $\mathcal{L}_{h_0}$
  (guaranteed by Sturm--Liouville theory: all eigenvalues
  of the 1D SL operator are simple).
\end{enumerate}
Then $h\mapsto E_n(h)$ is real-analytic on $(0,h_0)$,
with $E_n(h)\to\lambda_n:=E_n(0)$ as $h\to 0$,
and $h\mapsto x_0(h)$ (the turning point) is also real-analytic
by the IFT applied to $V(x_0(h))=E_n(h)$ using $V'(x_0^{(0)})\ne 0$
(Remark~\ref{rem:V-shape}(ii)).

\medskip
\noindent\emph{Verification for PSWF.}
(E1): $h\mapsto h^2\mathcal{L}_{h^{-1}}$ is an analytic family
(the symbol is polynomial in $h$; see \cite{Slepian1961}).
(E2): all eigenvalues of the PSWF SL operator are simple
by classical Sturm--Liouville oscillation theory.
\end{theorem}

%% ============================================================
\section{Hard Core (C1): Spectral Gap}
\label{sec:specgap}
%% ============================================================

\begin{theorem}[Spectral gap, \Cond]
\label{thm:specgap}
Assume:
\begin{enumerate}[{\rm(H1)}]
\item \textit{(Classical integrability.)} 1D: automatic. \Rig.
\item \textit{(Period finiteness and positivity.)}
  $T(E):=\oint_{H=E}p^{-1}\,dx$
  satisfies $0<T(\lambda_n)<\infty$.
  For PSWF: $V-\lambda_n$ has simple zeros at $x_\pm$,
  so the square-root singularity is integrable;
  $T(\lambda_n)>0$ since the classically allowed interval is nonempty.
\item \textit{(Period nondegeneracy.)} 
  $\partial_E T(E)|_{E=\lambda_n}\ne 0$,
  i.e.\ the map $E\mapsto T(E)$ is locally injective at $\lambda_n$.
  For PSWF: $T'(\lambda_n)=\partial_E\oint_{H=E}p^{-1}\,dx$;
  this equals $-\tfrac12\oint_{H=\lambda_n}|p|^{-3}\,dx\ne 0$
  (negative, finite, nonzero by (H2)).
\item \textit{(BS smooth invertibility.)}
  The Bohr--Sommerfeld action
  $S(E):=\oint_{H=E}p\,dx$
  satisfies $S'(\lambda_n)=T(\lambda_n)/2\ne 0$
  (immediate from (H2)).
\end{enumerate}
Then for $h\le h_0$:
\begin{equation}\label{eq:specgap}
  E_{n+1}(h)-E_n(h)
  = \frac{2\pi h}{T(\lambda_n)}+\mathcal{O}(h^2)
  \ge Ch,
  \quad C=\frac{\pi}{T(\lambda_n)}>0.
\end{equation}
In particular,
$\mathrm{dist}(E_n(h),\sigma(\mathcal{L}_h)\setminus\{E_n(h)\})\ge Ch$.

\medskip
\noindent\emph{Proof sketch.}
$S(E_{n+1})-S(E_n)=\pi h+\mathcal{O}(h^2)$
(consecutive Bohr--Sommerfeld levels);
$S'(\lambda_n)=T(\lambda_n)/2$
$\Rightarrow E_{n+1}-E_n=(2\pi h/T(\lambda_n))+\mathcal{O}(h^2)$
by the mean value theorem and (H3).
See \cite{Zworski2012} Ch.~9.
\end{theorem}

\begin{remark}[Role of spectral gap]
Theorem~\ref{thm:specgap} is the \emph{gatekeeper lemma}:
it ensures branch separation $\delta_\xi>0$
(Proposition~\ref{prop:branch-selection}),
which is prerequisite for all subsequent
microlocal and CFU arguments.
\end{remark}

\section{Branch Selection and Uniform Ellipticity}
\label{sec:branch}
%% ============================================================

\begin{proposition}[Branch selection and uniform ellipticity, \Cond]
\label{prop:branch-selection}

\medskip
\noindent\textit{(i) Spectral positivity.}
By Theorem~\ref{thm:eigenbranch}: $|E_n(h)-\lambda_n|=\mathcal{O}(h)$.
Hence for $h\le h_0$:
$E_n(h)-V(x)\ge c_\varepsilon^{\rm geo}/2>0$ on $K_\varepsilon$.
Define $\delta_\xi:=\sqrt{c_\varepsilon^{\rm geo}/2}$;
then $p(x;h)\ge\delta_\xi>0$.

\medskip
\noindent\textit{(ii) Branch separation.}
$\mathrm{WF}_h(\psi_n)\subset\{\xi^2+V=E_n(h)\}$
(\cite{Zworski2012} Ch.~12).
The spectral gap (Theorem~\ref{thm:specgap})
excludes mode mixing with $E_{n\pm 1}$.
On $K_\varepsilon$: $\Lambda_h^\pm\subset\{\pm\xi\ge\delta_\xi\}$.
Microlocal partition of unity yields
$\Pi^+$ with $\chi\psi_n=\Pi^+(\chi\psi_n)+\mathcal{O}(h^\infty)$
and $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\end{proposition}

\begin{remark}[CFU on each component separately]
\label{rem:cfu-components}
$K_\varepsilon^\pm:=K_\varepsilon\cap\{\pm(x-x_0(h))>0\}$.
On $K_\varepsilon^+$: $I>0$, $\phi=(\tfrac32 I)^{2/3}$.
On $K_\varepsilon^-$: $I<0$, $\phi=-(\tfrac32|I|)^{2/3}$.

Matching at $x_0(h)$: \textbf{thm:CFU-geom},
which handles Maslov index and phase gluing
(Remark~\ref{rem:cfu-geom-nontrivial}).
\end{remark}

\section{Hard Core (C3): Olver $C^1$-Stability}
\label{sec:olver-C1}
%% ============================================================

\begin{lemma}[$C^0$-Airy stability, \Cond]
\label{lem:C0-airy}
\textit{Statement.}
The Olver phase satisfies
$\|\phi(\cdot;h)-\phi_0\|_{C^0(K_\varepsilon)}=\mathcal{O}(h)$.

\medskip
\noindent\textit{Proof route.}
Uniform ellipticity (Lemma~\ref{lem:uniform-ellipticity}):
$p,p_0\ge\delta_\xi,c_\varepsilon>0$.
Energy comparison (Lemma~\ref{lem:energy-comparison}):
$\|p-p_0\|_{C^0}=A_\varepsilon(h)=\mathcal{O}(h)$.
Integral stability (Lemma~\ref{lem:integral-stability}):
$\|I-I_0\|_{C^0}=\tilde A_\varepsilon(h)=\mathcal{O}(h)$.
Lipschitz of $t\mapsto(\tfrac32 t)^{2/3}$ on $[c_\varepsilon',\infty)$
(non-degeneracy Lemma~\ref{lem:nondeg}):
$\|\phi-\phi_0\|_{C^0}\le L_\varepsilon\tilde A_\varepsilon(h)$.
\end{lemma}

\begin{lemma}[Derivative commuting estimate, \Cond]
\label{lem:deriv-commuting}
\textit{Statement.}
On $K_\varepsilon^\pm$: $\phi'(x;h)=p(x;h)/\phi(x;h)$
and the map $h\mapsto\phi'(\cdot;h)$ is uniformly controlled:
$\|\phi'-\phi_0'\|_{C^0(K_\varepsilon)}=\mathcal{O}(h)$.

\medskip
\noindent\textit{Key point (why derivative commuting is not automatic).}
An asymptotic expansion $\phi=\phi_0+h\phi_1+\cdots$
does not automatically give $\phi'=\phi_0'+h\phi_1'+\cdots$ uniformly.
(Warning example: $h\sin(x/h)\to 0$ in $C^0$ but not in $C^1$.)
Here the derivative exchange is justified because:
$\phi'=p/\phi$ is given by an \emph{explicit algebraic formula}
(no asymptotic differentiation needed);
$p$ and $\phi$ are both uniformly bounded away from zero on $K_\varepsilon$;
and Lemma~\ref{lem:C0-airy} gives $C^0$-control of $\phi$,
so the quotient is $C^1$-stable.
\end{lemma}

\begin{lemma}[Turning-point perturbation, \Cond]
\label{lem:tp-perturbation}
\textit{Statement.}
The turning point $x_0(h)$ satisfies
$|x_0(h)-x_0^{(0)}|=\mathcal{O}(h)$
and contributes $\mathcal{O}(h)$ to all integral estimates.

\medskip
\noindent\textit{Proof route.}
$V(x_0(h))=E_n(h)=\lambda_n+\mathcal{O}(h)$ (Theorem~\ref{thm:eigenbranch});
$V'(x_0^{(0)})\ne 0$ (Remark~\ref{rem:V-shape}(ii));
IFT gives $x_0(h)-x_0^{(0)}=(E_n(h)-\lambda_n)/V'(x_0^{(0)})+\mathcal{O}(h^2)=\mathcal{O}(h)$.
The contribution to $\tilde A_\varepsilon$ is $\|p\|_{C^0}\cdot\mathcal{O}(h)=\mathcal{O}(h)$.
\end{lemma}

\begin{corollary}[$C^1$-stability of $\phi$, \Cond]
\label{cor:phi-C1}
Combining Lemmas~\ref{lem:C0-airy}--\ref{lem:tp-perturbation}:
\[
  \inf_{h\le h_0,\,x\in K_\varepsilon}|\phi'(x;h)|
  \ge \tfrac12\inf_{K_\varepsilon}|\phi_0'| > 0.
\]
\end{corollary}

\section{Auxiliary Lemmas}
\label{sec:aux}

\begin{lemma}[Uniform ellipticity, \Cond]
\label{lem:uniform-ellipticity}
$p\ge\delta_\xi>0$,
$p_0\ge c_\varepsilon:=\sqrt{c_\varepsilon^{\rm geo}}>0$,
$p+p_0\ge\delta_\xi+c_\varepsilon>0$, all uniform in $h$.
\end{lemma}

\begin{lemma}[Energy comparison, \Cond]
\label{lem:energy-comparison}
$|p-p_0|=|E_n-\lambda_n|/(p+p_0)
\le|E_n-\lambda_n|/(\delta_\xi+c_\varepsilon)
=:A_\varepsilon(h)=\mathcal{O}(h)$.
\end{lemma}

\begin{lemma}[Integral stability, \Cond]
\label{lem:integral-stability}
$|I-I_0|\le\|p\|_{C^0}|x_0(h)-x_0^{(0)}|+|K_\varepsilon|A_\varepsilon
=:\tilde A_\varepsilon(h)=\mathcal{O}(h)$.
\end{lemma}

\begin{lemma}[Non-degeneracy, \Cond]
\label{lem:nondeg}
$|I|,|I_0|\ge c_\varepsilon':=\delta_\xi\varepsilon/2>0$;
$|\phi|\ge c_\varepsilon''$; $|\phi_0|\ge d_\varepsilon>0$.
\end{lemma}

\section{CFU Jacobian in 1D}
\label{sec:cfu-jacobian}
%% ============================================================

\begin{lemma}[CFU Jacobian in 1D, \Cond]
\label{lem:cfu-jacobian-1d}
In normalized CFU coordinates
$\Phi(x,\theta)=\phi\theta-\tfrac13\theta^3$:
$\partial^2_{x\theta}\Phi=\phi'$,
hence $|\det(dt/d\beta)|=|\phi'|\ge\delta_1>0$
(Corollary~\ref{cor:phi-C1}).
Other coordinate systems contribute a bounded,
bounded-away-from-zero Jacobian factor
(Remark~\ref{rem:cfu-geom-nontrivial}),
not affecting (A2).
\end{lemma}

\section{Target Theorem}
\label{sec:target}

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
Under Assumption~\ref{ass:boundary-away}
and hard cores (C1)--(C3)
(Theorems~\ref{thm:eigenbranch},~\ref{thm:specgap}
and Lemmas~\ref{lem:C0-airy}--\ref{lem:tp-perturbation}),
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly on $K_\varepsilon$ as $h\to 0$.
\end{theorem}

\[
  \underbrace{\text{(C2) Eigenbranch}}_{\text{Thm.}~\ref{thm:eigenbranch}}
  +
  \underbrace{\text{(C1) Spec.\ gap}}_{\text{Thm.}~\ref{thm:specgap}}
  \;\Rightarrow\;
  \text{Prop.}~\ref{prop:branch-selection}
  \;\Rightarrow\;
  \underbrace{\text{(C3) } C^1\text{-chain}}_{\text{Lems.}~\ref{lem:C0-airy}\text{--}\ref{lem:tp-perturbation}}
  \;\Rightarrow\;
  \text{Lem.}~\ref{lem:cfu-jacobian-1d}
  \;\Rightarrow\;
  \text{Thm.}~\ref{thm:xvii-main}.
\]
\end{remark}
