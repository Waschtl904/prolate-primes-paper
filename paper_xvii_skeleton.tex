\title[A$_2$-Stability for PSWF Amplitudes]{%
  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
(conditions (A1)--(A3)) uniformly as $h := c^{-1} \to 0$.

\medskip
\noindent\textbf{Main technical contribution.}
Lemma~\ref{lem:diff-olver} establishes uniform $C^1$-control
of the Olver phase $\phi(x;h)$ via the integral representation
\[
  \phi(x;h)
  = \mathrm{sgn}(x-x_0(h))\cdot
    \Bigl(\tfrac{3}{2}\int_{x_0(h)}^x p(s;h)\,ds\Bigr)^{2/3},
  \quad
  p(x;h) := \sqrt{E_n(h)-V(x)},
\]
reducing all $C^1$-control to uniform control of $p(x;h)$
and its integral on $K_\varepsilon$.
No ODE, no Gr\"onwall, no Wasow expansion.

\medskip
\noindent\textbf{Meta-remark.}
CFU stability depends only on:
energy convergence $E_n(h)\to\lambda_n$;
$L^\infty$-control of $V$ on $\mathrm{supp}\,\chi$;
spectral isolation;
and distance from the turning point.
This is general well beyond PSWF.
\end{abstract}

Used as black boxes:

\item \textbf{thm:CFU-symbol}:
  $I^{-1/4,0}(\Lambda_0,\Lambda_1)\to S^{-1/2}_{1,0}(h)$, uniform.
\item \textbf{Assumption~A2}:
  (A1) $c_3(h)\ge\delta_0>0$;
  (A2) $|\det(dt/d\beta)|\ge\delta_1>0$;
  (A3) unique fold on $\mathrm{supp}\,\chi$.
\item \textbf{hyp:rigidity}: Airy class $\Rightarrow$ A$_2$-fold uniform.
\end{itemize}

\begin{remark}[Airy $\not\Rightarrow$ CFU]
Airy normal form is local;
CFU admissibility is global and uniform in $h$.
The gap is closed by Proposition~\ref{prop:branch-selection},
Assumption~\ref{ass:boundary-away}, and Lemma~\ref{lem:diff-olver}.
\end{remark}

%% ============================================================
\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF]
$\psi_n$ satisfy $(h^2\mathcal{L}_{h^{-1}})\psi_n = E_n(h)\psi_n$
with $V(x)=x^2/(1-x^2)$, $E_n(h):=\chi_n h^2$.
Turning points: $V(x_\pm(h))=E_n(h)$.
\end{definition}

\begin{remark}[Eigenbranch, momentum, turning point]
\label{rem:eigenbranch}
$E_n(h)\to\lambda_n\in(0,1)$ smoothly;
$p_0(x):=\sqrt{\lambda_n-V(x)}$.

\medskip
\noindent\textit{Uniform positivity on $K_\varepsilon$.}
For every $\varepsilon>0$, on
$K_\varepsilon:=\mathrm{supp}\,\chi\setminus U_\varepsilon$:
\[
  \inf_{x\in K_\varepsilon}p_0(x)\ge c_\varepsilon>0,
  \qquad
  \inf_{h\le h_0,\,x\in K_\varepsilon}p(x;h)\ge\delta_\xi(\varepsilon)>0.
\]
All constants $c_\varepsilon,\delta_\xi(\varepsilon),c_\varepsilon',c_\varepsilon''$
below are \emph{independent of $h$}
(they depend on $\varepsilon$, $V$, $\lambda_n$ only).

\medskip
\noindent\textit{Turning-point shift (IFT).}
$V(x_0(h))=E_n(h)$, $V'(x_0^{(0)})\ne 0$
$\Rightarrow$ $x_0(h)\to x_0^{(0)}$ smoothly.
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$,
fixed independently of $h$.
Consequences: $V,V',V''\in L^\infty$ uniformly;
boundary-layer effects excluded.
\end{assumption}

\section{Olver's Uniform Asymptotic Scheme}
\label{sec:olver}
%% ============================================================

Olver \cite{Olver1974} (Ch.~10--11):
$\psi_n\sim h^{-1/4}a(x;h)\,\mathrm{Ai}(h^{-2/3}\phi)+\mathcal{O}(h^\infty)$,
uniform on $\mathrm{supp}\,\chi$.

\begin{remark}[Olver--CFU gap]
\label{rem:olver-gap}
Olver does not provide:
Lagrangian/FIO structure; global geometry of $\Lambda_h$;
uniform transversality; or uniform $C^1$-control of $\phi$.
Warning: $f_h(x)=x+h\sin(x/h)\to x$ uniformly
but $f_h'=1+\cos(x/h)$ does not.
\end{remark}

%% ============================================================
\section{Branch Selection}
\label{sec:branch}
%% ============================================================

\begin{remark}[Two-branch structure]
$\Lambda_h=\Lambda_h^+\cup\Lambda_h^-$,
$\Lambda_h^\pm=\{\xi=\pm p(x;h)\}$;
(P2) fails for $\Lambda_h$ as a whole.
\end{remark}

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
Fix $\varepsilon>0$; let $\delta_\xi(\varepsilon)>0$ be as in
Remark~\ref{rem:eigenbranch}.
On $K_\varepsilon$, the two Lagrangian branches are
microlocally separated:
$\Lambda_h^+\subset\{\xi\ge\delta_\xi\}$ and
$\Lambda_h^-\subset\{\xi\le -\delta_\xi\}$.
Since $\mathrm{WF}_h(\psi_n)\subset\Lambda_h^+\cup\Lambda_h^-$
and the two components are separated by a uniform
gap $2\delta_\xi(\varepsilon)$ in $\xi$,
a microlocal partition of unity in phase space
yields $\Pi^+=\mathrm{Op}_h(\sigma^+)$ with
$\sigma^+\in C_c^\infty(T^*(-1,1))$,
$\mathrm{supp}\,\sigma^+\subset\{\xi\ge\delta_\xi/2\}$,
such that:
\begin{enumerate}[{\rm(i)}]
\item
  $\chi\psi_n = \Pi^+(\chi\psi_n)+\mathcal{O}(h^\infty)$ in $L^2$,
  uniform in $h$.
\item
  $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\end{enumerate}
The remainder is $\mathcal{O}(h^\infty)$ by the
$h$-pseudodifferential Egorov theorem \cite{Zworski2012}.
\textit{Caveat:} branch separation holds only on $K_\varepsilon$;
it breaks down in $U_\varepsilon$ (near the turning point).
\end{proposition}

%% ============================================================
\section{Global Geometry: (P1) and (P3)}
\label{sec:phi-conditions}
%% ============================================================

\begin{remark}[(P1): global generating function]
On $\Lambda_h^+\cap\pi^{-1}(K_\varepsilon)$,
$\xi=p(x;h)>0$ is single-valued,
so $\phi(x;h)$ is a global generating function.
Status: \Cond.
\end{remark}

\begin{remark}[(P3): no secondary caustics]
\label{rem:p3}
On $\mathrm{supp}\,\chi$,
the equation $E_n(h)-V(x)=0$
has exactly one solution $x_0(h)$.
This is a purely geometric statement:
it follows from the \emph{strict monotonicity of $V$}
on $\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$
(where $V(x)=x^2/(1-x^2)$ is strictly increasing for $x>0$
and strictly decreasing for $x<0$,
with unique zero at $x=0$).
No mode-interaction or spectral-gap argument is needed
for this conclusion:
\textit{a single turning point is an immediate consequence
of the shape of $V$.}

\medskip
\noindent\textit{Spectral gap (separate condition).}
To exclude frequency overlap between different eigenbranches
within the symbol support of $\Pi^+$, one requires:
\begin{equation}\label{eq:specgap}
  \exists\,C>0:\quad
  \mathrm{dist}(E_n(h),\,\sigma(\mathcal{L}_h)\setminus\{E_n(h)\})
  \ge Ch.
\end{equation}
This follows from SL eigenvalue asymptotics
($E_{n+1}(h)-E_n(h)\sim Ch$; \cite{Olver1974} Ch.~12).
Status: \Cond.
\end{remark}

\section{The Olver--FIO Lift}
\label{sec:lift}
%% ============================================================

\begin{lemma}[Microlocal Structure, \Cond]
\label{lem:olver-fio-lift-micro}
Under Assumption~\ref{ass:boundary-away}
and the smooth-eigenbranch hypothesis:
(i) $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$
with phase $\phi(x;h)$;
(ii) $\Lambda_h^+$ has a single A$_2$-fold at $(x_0,0)$
with $c_3(h)\ge\delta_0>0$ uniformly.
\end{lemma}

\begin{lemma}[Uniform Projection, \Cond]
\label{lem:uniform-projection}
Under Lemma~\ref{lem:olver-fio-lift-micro},
Proposition~\ref{prop:branch-selection},
(P1), (P3), and Lemma~\ref{lem:diff-olver}:
$\inf_{h,x\in K_\varepsilon}|\phi'(x;h)|\ge\delta>0$
and $|\det d\pi_h^+|\ge\delta_2>0$ on $K_\varepsilon$.
\end{lemma}

%% ============================================================
\section{The Differentiated Olver Estimate}
\label{sec:diff-olver}
%% ============================================================

\begin{lemma}[Monotonicity of $I$, \Rig]
\label{lem:I-monotone}
On $K_\varepsilon$, $p(x;h)>0$ uniformly,
so $x\mapsto I(x;h)=\int_{x_0(h)}^x p(s;h)\,ds$
is strictly monotone.
Consequently:
$\mathrm{sgn}(I(x;h))=\mathrm{sgn}(x-x_0(h))$,
and $|I(x;h)|\ge\delta_\xi(\varepsilon)\cdot|x-x_0(h)|$.
\end{lemma}

\begin{remark}[Sign-constancy and smoothness of $\phi$]
\label{rem:sign-constancy}
$K_\varepsilon$ has two connected components:
$K_\varepsilon^- = K_\varepsilon\cap\{x<x_0(h)\}$ and
$K_\varepsilon^+ = K_\varepsilon\cap\{x>x_0(h)\}$.
On each component, $\mathrm{sgn}(x-x_0(h))$ is constant,
so $\phi(x;h)=(\tfrac{3}{2}|I(x;h)|)^{2/3}$
(with fixed sign) is smooth.
Differentiation of $\phi^3 = (\tfrac{3}{2})^3 I^2$ yields
$\phi'(x;h)=p(x;h)/\phi(x;h)$ on each component.
\end{remark}

\begin{lemma}[Differentiated Olver Estimate, \Cond]
\label{lem:diff-olver}
\emph{(Main technical lemma.)}\;
Fix $\varepsilon>0$; let $K_\varepsilon^\pm$ be the two
components of $K_\varepsilon$ (Remark~\ref{rem:sign-constancy}).
All constants below are independent of $h$.
On $K_\varepsilon$:

\begin{enumerate}[{\rm(A)}]
\item \textit{(Uniform convergence of $p$.)}
  Since $p\ge\delta_\xi>0$ and $p_0\ge c_\varepsilon>0$
  (Remark~\ref{rem:eigenbranch}), rationalize:
  \[
    |p(x;h)-p_0(x)|
    = \frac{|E_n(h)-\lambda_n|}{p(x;h)+p_0(x)}
    \le\frac{|E_n(h)-\lambda_n|}{\delta_\xi+c_\varepsilon}
    =: A_\varepsilon(h)=\mathcal{O}(h).
  \]
  \emph{(The Lipschitz route via $\sqrt{\,\cdot\,}$ fails at $p_0=0$;
  rationalization is the correct argument.)}

\item \textit{(Uniform convergence of $I$.)}
  Split the error into shift and integrand terms:
  \[
    |I(x;h)-I_0(x)|
    \le\|p\|_{C^0}\cdot|x_0(h)-x_0^{(0)}|
    +|K_\varepsilon|\cdot A_\varepsilon(h)
    =\mathcal{O}(h),
  \]
  using $x_0(h)\to x_0^{(0)}$ smoothly (IFT;
  Remark~\ref{rem:eigenbranch}) and (A).

\item \textit{(Denominator bounded away from zero.)}
  By Lemma~\ref{lem:I-monotone} and
  $|x-x_0(h)|\ge\varepsilon/2$ on $K_\varepsilon$
  for $h\le h_0$:
  \[
    |I(x;h)|\ge\delta_\xi\cdot\tfrac{\varepsilon}{2}=:c_\varepsilon'>0,
    \quad
    |\phi(x;h)|\ge(\tfrac{3}{4}\delta_\xi\varepsilon)^{2/3}=:c_\varepsilon''>0.
  \]
  In particular $I_0(x)\ne 0$, hence $\phi_0(x)\ne 0$,
  on $K_\varepsilon$.

\item \textit{(Uniform $C^1$-control.)}
  By Remark~\ref{rem:sign-constancy},
  on each component of $K_\varepsilon$:
  $\phi'(x;h)=p(x;h)/\phi(x;h)$
  and $\phi_0'(x)=p_0(x)/\phi_0(x)$
  (both well-defined since $|\phi|,|\phi_0|\ge c_\varepsilon''>0$).
  Therefore:
  \[
    |\phi'-\phi_0'|
    \le\frac{|p-p_0|}{|\phi|}
    +p_0\cdot\frac{|\phi-\phi_0|}{|\phi|\cdot|\phi_0|}
    \le\frac{A_\varepsilon(h)}{c_\varepsilon''}
    +\frac{p_0^{\max}\cdot B_\varepsilon(h)}{(c_\varepsilon'')^2}
    \to 0,
  \]
  where $B_\varepsilon(h):=\|\phi(\cdot;h)-\phi_0\|_{C^0}=\mathcal{O}(h)$
  follows from (B).
  Since $\phi_0'=p_0/\phi_0\ge c_\varepsilon/c_\varepsilon''>0$ on $K_\varepsilon$:
  \[
    \inf_{h\le h_0,\,x\in K_\varepsilon}|\phi'(x;h)|
    \ge\tfrac{1}{2}\inf_{K_\varepsilon}|\phi_0'|>0.
  \]
\end{enumerate}
\end{lemma}

\begin{remark}[Robustness]
\label{rem:robustness}
The proof uses only:
(i) smooth energy convergence $E_n(h)\to\lambda_n$;
(ii) $V\in L^\infty$ (Assumption~\ref{ass:boundary-away});
(iii) rationalization via $p,p_0\ge c_\varepsilon>0$ (the key step);
(iv) monotonicity of $I$ (Lemma~\ref{lem:I-monotone});
(v) IFT for $x_0(h)$;
(vi) sign-constancy of $x-x_0(h)$ on $K_\varepsilon^\pm$
     (Remark~\ref{rem:sign-constancy}).
No ODE, no Gr\"onwall, no Wasow expansion.
The argument applies to any semiclassical Schr\"odinger
operator with a simple turning point,
$V\in L^\infty$, and smooth energy convergence.
\end{remark}

\section{Derivation of Assumption~A2}
\label{sec:a2-derivation}
%% ============================================================

\begin{proposition}[A2, \Cond]
\label{prop:a2-from-lemmas}
Under all preceding lemmas and conditions:
$|\det(dt/d\beta)|\ge\delta_1>0$ uniformly.
\end{proposition}

\begin{proof}[Proof sketch]
Restrict to $\Lambda_h^+$ (Proposition~\ref{prop:branch-selection});
(P1) $\Rightarrow$ $\phi$ is a global generating function on $K_\varepsilon$;
(P3) $\Rightarrow$ no secondary caustics.
The CFU Jacobian is controlled by $|\phi'(x;h)|$;
Lemma~\ref{lem:diff-olver}(D) gives $|\phi'|\ge\delta>0$ uniformly.
\end{proof}

\section{Target Theorem}
\label{sec:target}
%% ============================================================

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
Under Assumption~\ref{ass:boundary-away},
the smooth-eigenbranch hypothesis,
and the spectral gap condition~\eqref{eq:specgap},
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly as $h\to 0$,
and \textup{hyp:rigidity} (XVI) holds.
\end{theorem}

\begin{remark}[Dependency chain]
\[
  \bigl[E_n\to\lambda_n,\;V\in L^\infty,\;
  V'(x_0^{(0)})\ne 0,\;\text{\eqref{eq:specgap}}\bigr]
  \Rightarrow
  \bigl[\text{Lem.}~\ref{lem:diff-olver},\;
  \text{Prop.}~\ref{prop:branch-selection}\bigr]
  \Rightarrow
  \text{Lem.}~\ref{lem:uniform-projection}
  \Rightarrow
  \text{Prop.}~\ref{prop:a2-from-lemmas}
  \Rightarrow
  \text{Thm.}~\ref{thm:xvii-main}.
\]
\end{remark}

\begin{remark}[Scope]
Paper~XVII does not address the full inverse microlocal conjecture,
global PSWF bounds, or concentration-of-energy.
\end{remark}
