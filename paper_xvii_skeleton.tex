\title[A$_2$-Stability for PSWF Amplitudes]{%
  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
(conditions (A1)--(A3)) uniformly as $h := c^{-1} \to 0$.

\medskip
\noindent\textbf{Main technical contribution.}
Lemma~\ref{lem:diff-olver} establishes uniform $C^1$-control
of the Olver phase $\phi(x;h)$ via the integral representation
\[
  \phi(x;h)
  = \mathrm{sgn}(x-x_0(h))\cdot
    \Bigl(\tfrac{3}{2}\int_{x_0(h)}^x p(s;h)\,ds\Bigr)^{2/3},
  \quad
  p(x;h) := \sqrt{E_n(h)-V(x)},
\]
reducing all $C^1$-control to uniform control of $p(x;h)$
and its integral on $K_\varepsilon$.
No ODE, no Gr\"onwall, no Wasow expansion.

\medskip
\noindent\textbf{Meta-remark.}
CFU stability depends only on:
energy convergence $E_n(h)\to\lambda_n$;
$L^\infty$-control of $V$;
spectral isolation;
and distance from the turning point.
This is general well beyond PSWF.
\end{abstract}

\end{remark}

%% ============================================================
\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF]
$\psi_n$ satisfy $(h^2\mathcal{L}_{h^{-1}})\psi_n = E_n(h)\psi_n$
with $V(x)=x^2/(1-x^2)$, $E_n(h):=\chi_n h^2$.
Turning points: $V(x_\pm(h))=E_n(h)$.
\end{definition}

\begin{remark}[Eigenbranch, momentum, turning point]
\label{rem:eigenbranch}
$E_n(h)\to\lambda_n\in(0,1)$ smoothly;
$p_0(x):=\sqrt{\lambda_n-V(x)}$.

\medskip
\noindent\textit{Uniform positivity on $K_\varepsilon$.}
For every $\varepsilon>0$, on
$K_\varepsilon:=\mathrm{supp}\,\chi\setminus U_\varepsilon$
(where $U_\varepsilon$ is an open $\varepsilon$-neighborhood of $x_0^{(0)}$;
all bounds below are on $K_\varepsilon$ unless stated otherwise):
\[
  \inf_{x\in K_\varepsilon}p_0(x)\ge c_\varepsilon>0,
  \qquad
  \inf_{h\le h_0,\,x\in K_\varepsilon}p(x;h)\ge\delta_\xi(\varepsilon)>0.
\]
\textbf{All constants $c_\varepsilon,\delta_\xi,c_\varepsilon',c_\varepsilon'',d_\varepsilon,L_\varepsilon$
are independent of $h$} (depending only on $\varepsilon,V,\lambda_n$).

\medskip
\noindent\textit{Turning-point shift (IFT).}
$V(x_0(h))=E_n(h)$, $V'(x_0^{(0)})\ne 0$
$\Rightarrow$ $x_0(h)\to x_0^{(0)}$ smoothly.
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$,
fixed independently of $h$.
Consequences: $V,V',V''\in L^\infty$ uniformly;
boundary-layer effects excluded.
\end{assumption}

\section{Olver's Uniform Asymptotic Scheme}
\label{sec:olver}
%% ============================================================

Olver \cite{Olver1974} (Ch.~10--11):
$\psi_n\sim h^{-1/4}a(x;h)\,\mathrm{Ai}(h^{-2/3}\phi)+\mathcal{O}(h^\infty)$,
uniform on $\mathrm{supp}\,\chi$.

\begin{remark}[Olver--CFU gap]
\label{rem:olver-gap}
Olver does not provide:
Lagrangian/FIO structure; global geometry of $\Lambda_h$;
uniform transversality; or uniform $C^1$-control of $\phi$.
Warning: $f_h(x)=x+h\sin(x/h)\to x$ uniformly
but $f_h'=1+\cos(x/h)$ does not.
\end{remark}

%% ============================================================
\section{Branch Selection}
\label{sec:branch}
%% ============================================================

\begin{remark}[Two-branch structure]
$\Lambda_h=\Lambda_h^+\cup\Lambda_h^-$,
$\Lambda_h^\pm=\{\xi=\pm p(x;h)\}$;
(P2) fails for $\Lambda_h$ as a whole.
\end{remark}

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
Fix $\varepsilon>0$; let $\delta_\xi(\varepsilon)>0$ be as above.
Since $\psi_n$ is an eigenfunction of $h^2\mathcal{L}_{h^{-1}}$
with eigenvalue $E_n(h)$, semiclassical propagation gives:
\[
  \mathrm{WF}_h(\psi_n)
  \subset \{(x,\xi):\xi^2+V(x)=E_n(h)\}
  = \Lambda_h^+\cup\Lambda_h^-
\]
(microlocal support on the characteristic set;
\cite{Zworski2012} Ch.~12).
On $K_\varepsilon$: $\Lambda_h^+\subset\{\xi\ge\delta_\xi\}$,
$\Lambda_h^-\subset\{\xi\le-\delta_\xi\}$ (uniform gap $2\delta_\xi$).
A microlocal partition of unity in phase space yields
$\Pi^+=\mathrm{Op}_h(\sigma^+)$,
$\mathrm{supp}\,\sigma^+\subset\{\xi\ge\delta_\xi/2\}$, such that:
\begin{enumerate}[{\rm(i)}]
\item
  $\chi\psi_n=\Pi^+(\chi\psi_n)+\mathcal{O}(h^\infty)$ in $L^2$.
\item
  $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\end{enumerate}
\textit{Caveat:} branch separation holds only on $K_\varepsilon$.
\end{proposition}

%% ============================================================
\section{Global Geometry: (P1) and (P3)}
\label{sec:phi-conditions}
%% ============================================================

\begin{remark}[(P1): global generating function]
On $\Lambda_h^+\cap\pi^{-1}(K_\varepsilon)$,
$\xi=p>0$ single-valued $\Rightarrow$ $\phi$ global generating function.
\Cond.
\end{remark}

\begin{remark}[(P3): no secondary caustics]
\label{rem:p3}
\textit{Unique turning point (\Rig).}
$V(x)=x^2/(1-x^2)$ is strictly monotone on
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$,
so $E_n(h)-V(x)=0$ has exactly one solution $x_0(h)$.
Geometric statement; no spectral input needed.

\medskip
\noindent\textit{Spectral gap (\Cond, for frequency-overlap exclusion).}
\begin{equation}\label{eq:specgap}
  \exists\,C>0:\quad
  \mathrm{dist}(E_n(h),\sigma(\mathcal{L}_h)\setminus\{E_n(h)\})
  \ge Ch.
\end{equation}
Follows from SL eigenvalue asymptotics \cite{Olver1974} Ch.~12.
\end{remark}

\begin{lemma}[Microlocal Structure, \Cond]
\label{lem:olver-fio-lift-micro}

(i) $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$;
(ii) $\Lambda_h^+$ has a single A$_2$-fold at $(x_0,0)$
with $c_3(h)\ge\delta_0>0$ uniformly.
\end{lemma}

\begin{lemma}[Uniform Projection, \Cond]
\label{lem:uniform-projection}
Under Lemma~\ref{lem:olver-fio-lift-micro},
Proposition~\ref{prop:branch-selection},
(P1), (P3), Lemma~\ref{lem:diff-olver}:
$\inf_{h,x\in K_\varepsilon}|\phi'(x;h)|\ge\delta>0$
and $|\det d\pi_h^+|\ge\delta_2>0$ on $K_\varepsilon$.
\end{lemma}

%% ============================================================
\section{The Differentiated Olver Estimate}
\label{sec:diff-olver}
%% ============================================================

\begin{lemma}[Monotonicity of $I$, \Rig]
\label{lem:I-monotone}
On $K_\varepsilon$, $p(x;h)>0$ uniformly.
Hence $x\mapsto I(x;h)=\int_{x_0(h)}^x p(s;h)\,ds$
is strictly monotone,
$\mathrm{sgn}(I)=\mathrm{sgn}(x-x_0(h))$,
and $|I(x;h)|\ge\delta_\xi\cdot|x-x_0(h)|$.
\end{lemma}

\begin{remark}[Sign-constancy and smoothness of $\phi$]
\label{rem:sign-constancy}
$K_\varepsilon$ has two connected components:
$K_\varepsilon^\pm:=K_\varepsilon\cap\{\pm(x-x_0(h))>0\}$.
On each, $\mathrm{sgn}(x-x_0(h))$ is constant,
so $I(x;h)$ has fixed sign and
$\phi(x;h)=(\tfrac{3}{2}I(x;h))^{2/3}$
(no absolute value needed) is smooth.
Differentiating $\phi^3=(\tfrac{3}{2})^3 I^2$ gives
$\phi'(x;h)=p(x;h)/\phi(x;h)$ on each component.
\end{remark}

\begin{lemma}[Differentiated Olver Estimate, \Cond]
\label{lem:diff-olver}
\emph{(Main technical lemma.)}\;
Fix $\varepsilon>0$.
\textbf{All constants are independent of $h$ and
all bounds hold on $K_\varepsilon$.}

\begin{enumerate}[{\rm(A)}]
\item \textit{(Uniform convergence of $p$.)}
  Since $p\ge\delta_\xi>0$ and $p_0\ge c_\varepsilon>0$:
  \[
    |p(x;h)-p_0(x)|
    = \frac{|E_n(h)-\lambda_n|}{p(x;h)+p_0(x)}
    \le\frac{|E_n(h)-\lambda_n|}{\delta_\xi+c_\varepsilon}
    =: A_\varepsilon(h)=\mathcal{O}(h).
  \]
  (Lipschitz via $\sqrt{\,\cdot\,}$ fails at $p_0=0$;
  rationalization is the correct argument here.)

\item \textit{(Uniform convergence of $I$.)}
  \[
    |I(x;h)-I_0(x)|
    \le\|p\|_{C^0}\cdot|x_0(h)-x_0^{(0)}|
    +|K_\varepsilon|\cdot A_\varepsilon(h)
    =\mathcal{O}(h).
  \]

\item \textit{(Lower bounds on $|I|$ and $|\phi|$.)}
  By Lemma~\ref{lem:I-monotone} and
  $|x-x_0(h)|\ge\varepsilon/2$ on $K_\varepsilon$:
  \[
    |I(x;h)|\ge\delta_\xi\cdot\tfrac{\varepsilon}{2}=:c_\varepsilon'>0,
    \qquad
    |\phi(x;h)|\ge\bigl(\tfrac{3}{4}\delta_\xi\varepsilon\bigr)^{2/3}=:c_\varepsilon''>0.
  \]
  Similarly $|I_0(x)|\ge c_\varepsilon\cdot\tfrac{\varepsilon}{2}>0$,
  hence $|\phi_0(x)|\ge(\tfrac{3}{4}c_\varepsilon\varepsilon)^{2/3}=:d_\varepsilon>0$.

\item \textit{(Uniform $C^0$-control of $\phi$, via Lipschitz of
  $I\mapsto(\tfrac{3}{2}I)^{2/3}$.)}
  On each component $K_\varepsilon^\pm$ (Remark~\ref{rem:sign-constancy}),
  $I(x;h)$ has fixed sign,
  so the map $I\mapsto(\tfrac{3}{2}I)^{2/3}$ is smooth.
  Since $|I(x;h)|,|I_0(x)|\ge c_\varepsilon'>0$,
  this map is Lipschitz on $[c_\varepsilon',\infty)$
  (or $(-\infty,-c_\varepsilon']$) with constant
  \[
    L_\varepsilon
    = \tfrac{2}{3}\bigl(\tfrac{3}{2}\bigr)^{2/3}(c_\varepsilon')^{-1/3}.
  \]
  Therefore:
  \begin{equation}\label{eq:phi-C0}
    B_\varepsilon(h)
    :=\|\phi(\cdot;h)-\phi_0\|_{C^0(K_\varepsilon)}
    \le L_\varepsilon\cdot\|I(\cdot;h)-I_0\|_{C^0}
    =\mathcal{O}(h).
  \end{equation}

\item \textit{(Uniform $C^1$-control.)}
  By Remark~\ref{rem:sign-constancy}:
  $\phi'=p/\phi$ and $\phi_0'=p_0/\phi_0$
  (both well-defined: $|\phi|\ge c_\varepsilon''>0$,
  $|\phi_0|\ge d_\varepsilon>0$).
  \[
    |\phi'-\phi_0'|
    \le\frac{|p-p_0|}{|\phi|}
    +p_0\cdot\frac{|\phi-\phi_0|}{|\phi|\cdot|\phi_0|}
    \le\frac{A_\varepsilon(h)}{c_\varepsilon''}
    +\frac{p_0^{\max}\cdot B_\varepsilon(h)}{c_\varepsilon''\cdot d_\varepsilon}
    \to 0
  \]
  uniformly on $K_\varepsilon$, by (A) and~\eqref{eq:phi-C0}.
  Since $\phi_0'=p_0/\phi_0\ge c_\varepsilon/d_\varepsilon>0$:
  \[
    \inf_{h\le h_0,\,x\in K_\varepsilon}|\phi'(x;h)|
    \ge\tfrac{1}{2}\inf_{K_\varepsilon}|\phi_0'|>0.
  \]
\end{enumerate}
\end{lemma}

\begin{remark}[Robustness]
\label{rem:robustness}
The proof uses only (all on $K_\varepsilon$):
(i) smooth energy convergence $E_n(h)\to\lambda_n$;
(ii) $V\in L^\infty$;
(iii) rationalization of $|p-p_0|$ (key: $p,p_0\ge c_\varepsilon>0$);
(iv) monotonicity of $I$ (Lemma~\ref{lem:I-monotone});
(v) IFT for $x_0(h)$;
(vi) sign-constancy on $K_\varepsilon^\pm$ (Remark~\ref{rem:sign-constancy});
(vii) Lipschitz of $I\mapsto(\tfrac{3}{2}I)^{2/3}$
  on $[c_\varepsilon',\infty)$ with constant
  $L_\varepsilon=\tfrac{2}{3}(\tfrac{3}{2})^{2/3}(c_\varepsilon')^{-1/3}$.
No ODE, no Gr\"onwall, no Wasow expansion.
Applies to any semiclassical Schr\"odinger operator with
simple turning point, $V\in L^\infty$, smooth energy convergence.
\end{remark}

\section{Derivation of Assumption~A2}
\label{sec:a2-derivation}
%% ============================================================

\begin{proposition}[A2, \Cond]
\label{prop:a2-from-lemmas}
Under all preceding lemmas and conditions:
$|\det(dt/d\beta)|\ge\delta_1>0$ uniformly on $K_\varepsilon$.
\end{proposition}

\begin{proof}[Proof sketch]
Restrict to $\Lambda_h^+$;
(P1) $\Rightarrow$ $\phi$ global generating function on $K_\varepsilon$;
(P3) $\Rightarrow$ no secondary caustics;
Lemma~\ref{lem:diff-olver}(E) $\Rightarrow$ $|\phi'|\ge\delta>0$ on $K_\varepsilon$.
\end{proof}

\section{Target Theorem}
\label{sec:target}
%% ============================================================

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
Under Assumption~\ref{ass:boundary-away},
the smooth-eigenbranch hypothesis,
and the spectral gap~\eqref{eq:specgap},
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly on $K_\varepsilon$ as $h\to 0$,
and \textup{hyp:rigidity} (XVI) holds.
\end{theorem}

\begin{remark}[Dependency chain]
\[
  \bigl[E_n\to\lambda_n,\;V\in L^\infty,\;
  V'(x_0^{(0)})\ne 0,\;\text{\eqref{eq:specgap}},\;
  \mathrm{WF}_h(\psi_n)\subset\{\xi^2+V=E_n\}\bigr]
  \Rightarrow
  \bigl[\text{Lem.}~\ref{lem:diff-olver},\;
  \text{Prop.}~\ref{prop:branch-selection}\bigr]
  \Rightarrow
  \text{Lem.}~\ref{lem:uniform-projection}
  \Rightarrow
  \text{Prop.}~\ref{prop:a2-from-lemmas}
  \Rightarrow
  \text{Thm.}~\ref{thm:xvii-main}.
\]
\end{remark}
