  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.
It fixes the logical structure and identifies the critical bottlenecks
before full LaTeX elaboration.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI
for PSWF amplitudes:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
(conditions (A1)--(A3)) uniformly as $h := c^{-1} \to 0$.

\medskip
\noindent\textbf{Central new objects.}
The \emph{Olver--FIO Lift} is split into two logically independent lemmas:
\begin{itemize}
\item Lemma~\ref{lem:olver-fio-lift-micro}
  (\emph{Microlocal Structure}):
  Lagrangian structure and A$_2$-fold from Olver $+$ Sturm--Liouville.
  Status: \Cond.
\item Lemma~\ref{lem:uniform-projection}
  (\emph{Uniform Projection}):
  the canonical projection $\pi_h\colon \Lambda_h \to T^*(-1,1)$
  is uniformly transversal in $h$.
  Status: \Open\ (genuine analytic contribution of Paper~XVII).
\end{itemize}
The Jacobian bound (Assumption~(A2) of XVI) follows from
Lemma~\ref{lem:uniform-projection} together with
conditions (P1)--(P3) on the global geometry of $\Lambda_h$
(Section~\ref{sec:phi-conditions}).
Conditions (P2)--(P3) are \emph{independent geometric conditions}
not implied by Sturm--Liouville theory alone.

\medskip
\noindent\textbf{Strategy.}
Primary: Olver turning-point analysis (Ch.~10--11 of \cite{Olver1974}).
Secondary: FIO reinterpretation of Olver expansions.
Auxiliary: Sturm--Liouville spectral theory for global consistency.
\end{abstract}

\section{Input from Paper~XVI}
\label{sec:input}
%% ============================================================

The following are used as \emph{black boxes} (not reproved here):

\begin{itemize}
\item \textbf{thm:CFU-geom} (XVI):
  A$_2$-fold phase $\Rightarrow$ Airy normal form
  via local canonical transformation $\kappa$.
\item \textbf{thm:CFU-symbol} (XVI):
  $I^{-1/4,0}(\Lambda_0,\Lambda_1) \to S^{-1/2}_{1,0}(h)$
  under $\kappa$, uniform in $h$.
\item \textbf{Assumption~A2} (XVI):
  \begin{enumerate}[{\rm(A1)}]
  \item Single nondegenerate fold:
    $c_3(h) \ge \delta_0 > 0$ uniformly.
  \item CFU admissibility:
    $|\det(dt/d\beta)| \ge \delta_1 > 0$ uniformly in $h$.
  \item Unique fold on $\mathrm{supp}\,\chi$.
  \end{enumerate}
\item \textbf{hyp:rigidity} (XVI, conjectural):
  Airy class $\Rightarrow$ A$_2$-fold uniform in $h$
  (the inverse direction, verified here for PSWF).
\end{itemize}

\begin{remark}[What Airy structure does \emph{not} imply]
\label{rem:airy-not-cfu}
Airy normal form (i.e.\ a local change of variables reducing the ODE phase
to $\tfrac{2}{3}(-t)^{3/2}$) is a \emph{local, pointwise} statement.
CFU admissibility (Assumption~(A2)) is a \emph{global, uniform} statement
about the canonical projection map $\pi_h$.
The implication
\[
  \text{Airy structure} \;\Longrightarrow\; \text{CFU admissibility}
\]
is \emph{not} FIO-invariant, not Olver-implied, and not Sturm--Liouville-implied.
It requires the additional input of
Lemma~\ref{lem:uniform-projection} and conditions (P1)--(P3) below.
\end{remark}

\section{PSWF Setup}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF as semiclassical eigenfamily]\label{def:pswf}
The prolate spheroidal wave functions $\psi_n(x;c)$,
$x \in [-1,1]$, are eigenfunctions of
\[
  \mathcal{L}_c := -\frac{d}{dx}
  \bigl[(1-x^2)\frac{d}{dx}\bigr]
  + c^2 x^2,
\]
with eigenvalues $\chi_n(c)$.
Setting $h := c^{-1}$, the rescaled operator is
\[
  h^2 \mathcal{L}_{h^{-1}}
  = -h^2\frac{d}{dx}\bigl[(1-x^2)\frac{d}{dx}\bigr]
  + x^2,
\]
a semiclassical Schr\"odinger operator with effective potential
$V(x) = x^2/(1-x^2)$.
The turning points $x_\pm(h)$, where $V(x_\pm) = \chi_n h^2$,
are the loci of expected A$_2$-fold singularities.
\end{definition}

\begin{remark}[Eigenvalue branch]
\label{rem:eigenbranch}
For fixed mode $n$ and $h \to 0$,
one has $\chi_n(c) \sim c^2\lambda_n$ with $\lambda_n \in (0,1)$,
so the classical turning points converge:
$x_\pm(h) \to x_\pm^{(0)} \in (-1,1)$ as $h\to 0$.
The map $h \mapsto \chi_n(h^{-1})$ is smooth and
strictly monotone on $(0,h_0]$
(\Cond; follows from analyticity of
Sturm--Liouville eigenvalues \cite{Olver1974}).
\end{remark}

\section{Olver's Uniform Asymptotic Scheme}
\label{sec:olver}
%% ============================================================

\subsection{What Olver provides}

Olver \cite{Olver1974} (Ch.~10--11) provides, for a
second-order ODE with a simple turning point:
\begin{enumerate}[(i)]
\item A \emph{uniform asymptotic expansion} in $h$ matching
  the WKB and Airy regimes.
\item \emph{Uniform error bounds} on compact subsets of $(-1,1)$
  avoiding the endpoints $\pm 1$.
\item A local phase substitution $x \mapsto t(x;h)$ reducing
  the ODE phase to $\tfrac{2}{3}(-t)^{3/2}$ (Airy scaling).
\end{enumerate}
In particular, the PSWF admits the uniform asymptotic form
\[
  \psi_n(x;h)
  \;\sim\;
  h^{-1/4}\,a(x;h)\,
  \mathrm{Ai}\bigl(h^{-2/3}\,\phi(x;h)\bigr)
  + \mathcal{O}(h^\infty),
\]
where $\phi(x;h) \to \phi_0(x)$ as $h\to 0$ and $\phi_0(x_0)=0$
at the turning point $x_0$.

\subsection{What Olver does \emph{not} provide}
\label{ssec:olver-gap}

\begin{remark}[The Olver--CFU gap]
\label{rem:olver-gap}
Olver's analysis guarantees existence of $\phi(x;h)$ and $a(x;h)$,
pointwise and compact-uniform bounds, and the local Airy normal form.
It does \emph{not} provide:
\begin{itemize}
\item a microlocal canonical relation or Lagrangian distribution framework;
\item FIO-class membership of the amplitude;
\item the global geometry of $\Lambda_h$ (self-intersection, branch count);
\item the uniform transversality of $\pi_h\colon\Lambda_h\to T^*(-1,1)$;
\item the Jacobian bound $|\det(dt/d\beta)| \ge \delta_1 > 0$ uniform in $h$;
\item uniform $C^1$-control of $\phi(x;h)$ as $h\to 0$.
\end{itemize}
Bridging these gaps is the content of
Lemmas \ref{lem:olver-fio-lift-micro}--\ref{lem:uniform-projection}
and conditions (P1)--(P3).
\end{remark}

\subsection{Lemma 4.1a: Microlocal Structure}
%% ------------------------------------------------------------

\begin{lemma}[Olver--FIO Lift: Microlocal Structure, \Cond]
\label{lem:olver-fio-lift-micro}
Let $\psi_n(\,\cdot\,;h)$ be the PSWF family
(Definition~\ref{def:pswf}).
Under the eigenvalue-branch hypothesis of
Remark~\ref{rem:eigenbranch}:
\begin{enumerate}[{\rm(i)}]
\item \textit{(Lagrangian distribution.)}
  $\psi_n(\,\cdot\,;h)$ is microlocally equivalent,
  modulo $\mathcal{O}(h^\infty)$ uniformly in $h$,
  to a Lagrangian distribution
  $u_h \in I^{-1/4,0}(\Lambda_h)$
  with phase function $\phi(x;h)$ from the Olver reduction.
\item \textit{(A$_2$-fold.)}
  $\Lambda_h$ carries a single A$_2$-fold
  at the turning point $(x_0,\xi_0(h))$,
  with cubic coefficient
  $c_3(h) = \tfrac{1}{6}\Theta'''(0;h) \ge \delta_0 > 0$
  uniformly in $h$
  (Assumptions~(A1) and (A3) of XVI).
\end{enumerate}
\end{lemma}

\begin{remark}[Proof strategy]
\begin{enumerate}[{\rm(a)}]
\item Part~(i): the Olver series defines a Lagrangian distribution
  via H\"ormander \cite{Hormander1983} / Zworski \cite{Zworski2012};
  $h$-uniformity of the symbol class requires
  Olver's compact-uniform error bounds.
  Status: \Cond.
\item Part~(ii): one simple turning point per mode by
  Sturm--Liouville theory; A$_2$-type confirmed by Olver's local analysis;
  uniformity of $c_3(h)$ from smooth eigenbranch.
  Status: \Cond.
\end{enumerate}
\end{remark}

%% ------------------------------------------------------------
\subsection{Global Geometry of $\Lambda_h$: Conditions (P1)--(P3)}
\label{sec:phi-conditions}
%% ------------------------------------------------------------

\begin{remark}[Why SL theory does not control global $\Lambda_h$ geometry]
\label{rem:sl-not-global}
Sturm--Liouville theory establishes:
\begin{itemize}
\item the \emph{number} and \emph{simplicity} of turning points;
\item the \emph{local} non-degeneracy of the phase at each turning point.
\end{itemize}
It does \emph{not} control:
\begin{itemize}
\item whether $\Lambda_h$ is self-intersection-free globally;
\item whether $\pi_h$ is single-valued (one branch) over $\mathrm{supp}\,\chi$;
\item whether secondary caustics develop under $h \to 0$.
\end{itemize}
Even with a single turning point, the semiclassical WKB picture admits
multiple Lagrangian branches (incoming/outgoing) meeting at the fold;
global single-valuedness of $\pi_h$ is a separate geometric condition.
This is the classical phenomenon of WKB multi-branch projection and
cannot be excluded by spectral theory alone.
\end{remark}

\begin{remark}[Three required conditions on $\Lambda_h$]
\label{rem:phi-conditions}
The reduction from $\inf|\phi'(x;h)| \ge \delta$ to the
Jacobian bound $|\det(dt/d\beta)| \ge \delta_1$ is valid
\emph{only} under the following three conditions,
which must be verified \emph{independently} for the PSWF
(not derivable from Olver or SL alone):
\begin{enumerate}[{\rm(P1)}]
\item \textit{(Global generating function.)}
  $\phi(x;h)$ is a global generating function for $\Lambda_h$
  on $\mathrm{supp}\,\chi$;
  equivalently, $\Lambda_h$ has no self-intersections
  over $\mathrm{supp}\,\chi$.
  Status: \Cond\ (follows from global form of Olver phase on
  the compact region strictly inside $(-1,1)$).
\item \textit{(Single-branch projection.)}
  The projection $\pi_h\colon \Lambda_h \to T^*(-1,1)$
  is single-valued (one branch) over $\mathrm{supp}\,\chi$.
  This is \emph{not} implied by the existence of a unique
  turning point: a single fold still produces two WKB branches
  (incoming/outgoing), and restriction to one branch requires
  an explicit microlocal cutoff or energy argument.
  Status: \Open.
\item \textit{(No caustic interaction.)}
  The single A$_2$-fold of $\Lambda_h$ is the
  \emph{only} critical point of $\pi_h$ over $\mathrm{supp}\,\chi$;
  no secondary caustics arise from endpoint effects or
  mode interactions as $h \to 0$.
  Status: \Open.
\end{enumerate}
\end{remark}

%% ------------------------------------------------------------
\subsection{Lemma 4.1b: Uniform Projection}
%% ------------------------------------------------------------

\begin{lemma}[Uniform Projection Lemma, \Open]
\label{lem:uniform-projection}
Assume Lemma~\ref{lem:olver-fio-lift-micro}
and conditions \textup{(P1)--(P3)}
of Remark~\ref{rem:phi-conditions}.
Then the canonical projection map
\[
  \pi_h \;\colon\; \Lambda_h \;\longrightarrow\; T^*(-1,1),
  \quad h \in (0,h_0],
\]
is uniformly transversal away from the unique A$_2$-fold:
\[
  \inf_{h \in (0,h_0],\;(x,\xi)\in \Lambda_h\setminus U_\varepsilon}
  |\det\,d\pi_h(x,\xi)| \;\ge\; \delta_2 > 0.
\]
Moreover, under conditions \textup{(P1)--(P3)}, this is equivalent to
\[
  \inf_{h\in(0,h_0],\;x\in\mathrm{supp}\,\chi}
  |\phi'(x;h)| \;\ge\; \delta > 0.
\]
\end{lemma}

\begin{remark}[Proof strategy]
\label{rem:proj-strategy}
\begin{enumerate}[{\rm(a)}]
\item \textit{(Non-vanishing at turning point.)}
  The Olver phase equation satisfies
  $\phi(x;h)^2\phi'(x;h)^2 = (x-x_0(h))^2 V_1(x;h)$
  with $V_1$ smooth and nonvanishing.
  Differentiating at $x = x_0$, using $\phi(x_0;h)=0$
  and the nondegeneracy of the A$_2$-fold
  (Lemma~\ref{lem:olver-fio-lift-micro}(ii)),
  gives $\phi'(x_0;h) \ne 0$ for all $h \in (0,h_0]$.
  Status: \Cond.
\item \textit{(Uniform $C^1$-control and lower bound.)}
  The argument requires \emph{uniform $C^1$-control}
  of $\phi(x;h)$ as $h \to 0$, not merely pointwise convergence:
  \begin{itemize}
  \item \emph{Required:}
    $\phi(\,\cdot\,;h) \to \phi_0$ in $C^1(\mathrm{supp}\,\chi)$
    uniformly as $h\to 0$.
    This is \emph{strictly stronger} than Olver's pointwise
    and compact-uniform bounds on $\phi$ itself;
    it requires uniform bounds on $\partial_x\phi(x;h)$,
    i.e.\ that $\phi'(x;h)$ does not develop
    rapid oscillations or ``flattening'' near $h=0$.
    Classical counterexample: $f_h(x) = x + h\sin(x/h)$
    converges uniformly but has oscillating derivative,
    so pointwise convergence of $\phi \to \phi_0$ alone
    does not give $\phi' \to \phi_0'$ uniformly.
  \item \emph{Candidate argument:}
    Differentiate the Olver phase equation with respect to $x$;
    the resulting ODE for $\phi'(x;h)$ inherits
    the smooth-eigenbranch property (Remark~\ref{rem:eigenbranch})
    and satisfies uniform estimates on compact subsets of $(-1,1)$.
    Combined with non-vanishing (step~(a)), the map
    $(x,h) \mapsto \phi'(x;h)$ is continuous on
    $\mathrm{supp}\,\chi \times [0,h_0]$
    (including $h=0$) and nonvanishing throughout;
    by compactness, $\inf > 0$.
  \item \emph{What must be proved explicitly:}
    the $C^1$-convergence claim is \emph{not}
    a consequence of Olver \cite{Olver1974} and requires
    a new differentiated-Olver estimate
    (or a reference to a strengthened version, e.g.
    Wasow \cite{Wasow1965} or a modern treatment).
  \end{itemize}
  Status: \Open\ (core analytic contribution).
\end{enumerate}
\end{remark}

\section{Derivation of Assumption~A2}
\label{sec:a2-derivation}
%% ============================================================

\begin{proposition}[A2 from 4.1a $+$ (P1--P3) $+$ 4.1b, \Cond]
\label{prop:a2-from-lemmas}
Assume Lemmas~\ref{lem:olver-fio-lift-micro}
and~\ref{lem:uniform-projection},
and conditions \textup{(P1)--(P3)}.
Then
\[
  |\det(dt/d\beta)| \;\ge\; \delta_1 > 0
  \quad \text{uniformly in } h \in (0,h_0],
\]
i.e.\ Assumption~\textup{(A2)} of Paper~XVI holds.
\end{proposition}

\begin{proof}[Proof sketch]
By Lemma~\ref{lem:olver-fio-lift-micro}(i),
the PSWF is a Lagrangian distribution with phase $\phi(x;h)$.
Condition (P1) ensures $\phi$ is a global generating function;
(P2) ensures the projection is single-branch;
(P3) excludes secondary caustic interactions.
Under (P1)--(P3), the CFU projection $\beta \mapsto t(\beta,u;h)$
is locally equivalent to $\pi_h|_{\Lambda_h}$,
and the Jacobian is controlled below by $|\phi'(x;h)|$.
Lemma~\ref{lem:uniform-projection}, whose proof requires
uniform $C^1$-control of $\phi$ (Remark~\ref{rem:proj-strategy}(b)),
then gives $|\phi'| \ge \delta$, completing the bound.
\end{proof}

\section{Critical Bottlenecks}
\label{sec:bottlenecks}
%% ============================================================

\subsection{Open core~1: uniform $C^1$-control of $\phi$}

The primary open problem is:
\[
  \phi(\,\cdot\,;h) \to \phi_0 \text{ in } C^1(\mathrm{supp}\,\chi)
  \quad \text{uniformly as } h \to 0.
\]
This requires a \emph{differentiated Olver estimate}
not present in \cite{Olver1974}.
Status: \Open.

\subsection{Open core~2: global Lagrangian geometry (P2)--(P3)}

The single-branch and no-secondary-caustic conditions
(P2)--(P3) are \emph{independent geometric conditions}
not implied by Sturm--Liouville theory.
A rigorous argument must control:
\begin{itemize}
\item the WKB branch selection (microlocal cutoff to incoming/outgoing);
\item the absence of endpoint-induced secondary folds as $h \to 0$.
\end{itemize}
Status: \Open.

\subsection{Conditioned items}

(P1) global generating function: \Cond\ (Olver phase, compact support).
(A1) cubic coefficient $c_3(h) \ge \delta_0$: \Cond\ (Olver $+$ eigenbranch).
(A3) unique fold: \Rig\ (Sturm--Liouville).

\section{Target Theorem of Paper~XVII}
\label{sec:target}
%% ============================================================

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
\emph{(Target; contingent on
Lemma~\ref{lem:uniform-projection} and conditions (P1)--(P3).)}
The PSWF amplitude $\psi_n(\,\cdot\,;h)$
satisfies Assumption~A2 of Paper~XVI uniformly as $h \to 0$.
Consequently, the CFU reduction of Paper~XVI applies to PSWF,
and \textup{hyp:rigidity} holds for the PSWF family.
\end{theorem}

\begin{remark}[Logical dependency chain]
\label{rem:chain}
The proof chain is:
\[
  \underbrace{\text{Olver} + \text{SL}}_{{\rm[Rig/Cond]}}
  \;\Rightarrow\;
  \underbrace{\text{Lem.~4.1a}}_{{\rm[Cond]}}
  \;+\;
  \underbrace{(\mathrm{P1})}_{{\rm[Cond]}}
  \;+\;
  \underbrace{(\mathrm{P2}),(\mathrm{P3})}_{{\rm[Open]}}
  \;+\;
  \underbrace{C^1\text{-control}}_{{\rm[Open]}}
  \;\Rightarrow\;
  \underbrace{\text{Lem.~4.1b}}_{{\rm[Open]}}
  \;\Rightarrow\;
  \underbrace{\text{Prop.~5.1}}_{{\rm[Cond]}}
  \;\Rightarrow\;
  \underbrace{\text{Thm.~\ref{thm:xvii-main}}}_{{\rm[Cond]}}.
\]
There are \emph{three} independent open inputs:
(P2) WKB branch selection,
(P3) no secondary caustics,
and uniform $C^1$-control of $\phi'$.
The genuine contribution of Paper~XVII is to supply all three.
\end{remark}

\begin{remark}[Scope]
Paper~XVII does not prove the full inverse microlocal conjecture;
does not give global (non-microlocal) bounds on PSWF;
does not address PSWF completeness or concentration-of-energy.
Scope: verify Assumption~A2 of XVI for PSWF
via Olver $+$ FIO reinterpretation.
\end{remark}
