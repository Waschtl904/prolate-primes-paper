\title[A$_2$-Stability for PSWF Amplitudes]{%
  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.
It fixes the logical structure and identifies the critical bottlenecks
before full LaTeX elaboration.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI
for PSWF amplitudes:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
(conditions (A1)--(A3)) uniformly as $h := c^{-1} \to 0$.

\medskip
\noindent\textbf{Central new objects.}
The \emph{Olver--FIO Lift} is split into two logically independent lemmas:
\begin{itemize}
\item Lemma~\ref{lem:olver-fio-lift-micro}
  (\emph{Microlocal Structure}):
  Lagrangian structure and A$_2$-fold from Olver $+$ Sturm--Liouville.
  Status: \Cond.
\item Lemma~\ref{lem:uniform-projection}
  (\emph{Uniform Projection}):
  the canonical projection $\pi_h^+\colon \Lambda_h^+ \to T^*(-1,1)$
  on the selected positive-momentum branch
  is uniformly transversal in $h$.
  Status: \Open.
\end{itemize}
The branch selection $\Lambda_h \to \Lambda_h^+$
(Proposition~\ref{prop:branch-selection})
and the Differentiated Olver Estimate
(Lemma~\ref{lem:diff-olver})
are the two main technical contributions of Paper~XVII.
The Jacobian bound (Assumption~(A2) of XVI) follows from
these together with conditions (P1) and (P3)
(Section~\ref{sec:phi-conditions}).

\medskip
\noindent\textbf{Strategy.}
Primary: Olver turning-point analysis (Ch.~10--11 of \cite{Olver1974}).
Secondary: FIO reinterpretation of Olver expansions.
Auxiliary: Sturm--Liouville spectral theory; boundary-away cutoff.
\end{abstract}

\section{Input from Paper~XVI}

The following are used as \emph{black boxes} (not reproved here):

\begin{itemize}
\item \textbf{thm:CFU-geom} (XVI):
  A$_2$-fold phase $\Rightarrow$ Airy normal form
  via local canonical transformation $\kappa$.
\item \textbf{thm:CFU-symbol} (XVI):
  $I^{-1/4,0}(\Lambda_0,\Lambda_1) \to S^{-1/2}_{1,0}(h)$
  under $\kappa$, uniform in $h$.
\item \textbf{Assumption~A2} (XVI):
  \begin{enumerate}[{\rm(A1)}]
  \item Single nondegenerate fold:
    $c_3(h) \ge \delta_0 > 0$ uniformly.
  \item CFU admissibility:
    $|\det(dt/d\beta)| \ge \delta_1 > 0$ uniformly in $h$.
  \item Unique fold on $\mathrm{supp}\,\chi$.
  \end{enumerate}
\item \textbf{hyp:rigidity} (XVI, conjectural):
  Airy class $\Rightarrow$ A$_2$-fold uniform in $h$
  (the inverse direction, verified here for PSWF).
\end{itemize}

\begin{remark}[What Airy structure does \emph{not} imply]
\label{rem:airy-not-cfu}
Airy normal form is a \emph{local, pointwise} statement.
CFU admissibility is a \emph{global, uniform} statement about $\pi_h$.
The implication Airy $\Rightarrow$ CFU admissibility
is not FIO-invariant, not Olver-implied, not SL-implied.
It requires the branch selection of
Proposition~\ref{prop:branch-selection},
the boundary-away condition of
Assumption~\ref{ass:boundary-away},
and the differentiated Olver estimate of
Lemma~\ref{lem:diff-olver}.
\end{remark}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF as semiclassical eigenfamily]\label{def:pswf}
The prolate spheroidal wave functions $\psi_n(x;c)$,
$x \in [-1,1]$, are eigenfunctions of
\[
  \mathcal{L}_c := -\frac{d}{dx}
  \bigl[(1-x^2)\frac{d}{dx}\bigr]
  + c^2 x^2
\]
with eigenvalues $\chi_n(c)$.
Setting $h := c^{-1}$, the rescaled operator is
\[
  h^2 \mathcal{L}_{h^{-1}}
  = -h^2\frac{d}{dx}\bigl[(1-x^2)\frac{d}{dx}\bigr]
  + x^2,
\]
a semiclassical Schr\"odinger operator with effective potential
$V(x) = x^2/(1-x^2)$.
The two turning points $x_\pm(h)$, where $V(x_\pm) = \chi_n h^2$,
are the loci of expected A$_2$-fold singularities.
\end{definition}

\begin{remark}[Eigenvalue branch]
\label{rem:eigenbranch}
For fixed mode $n$ and $h \to 0$,
$\chi_n(c) \sim c^2\lambda_n$ with $\lambda_n \in (0,1)$,
so $x_\pm(h) \to x_\pm^{(0)} \in (-1,1)$ as $h \to 0$.
The map $h \mapsto \chi_n(h^{-1})$ is smooth and strictly monotone
on $(0,h_0]$ (\Cond; analyticity of SL eigenvalues).
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
Fix $\delta_{\rm bdy} > 0$.
All constructions are localized to
\[
  \mathrm{supp}\,\chi \;\subset\; [-1+\delta_{\rm bdy},\,1-\delta_{\rm bdy}],
\]
where $\chi \in C_c^\infty(-1,1)$ is a fixed smooth cutoff
independent of $h$.
On this set the potential $V(x) = x^2/(1-x^2)$ satisfies
$V,V',V'' \in L^\infty$ uniformly, ruling out
boundary-layer effects and potential growth $V \to \infty$
as $x \to \pm 1$.
All statements labeled \Cond\ or \Open\ are relative to this assumption.
\end{assumption}

\section{Olver's Uniform Asymptotic Scheme}
\label{sec:olver}
%% ============================================================

\subsection{What Olver provides}

Olver \cite{Olver1974} (Ch.~10--11) provides:
\begin{enumerate}[(i)]
\item A uniform asymptotic expansion matching WKB and Airy regimes.
\item Uniform error bounds on compact subsets of $(-1,1)$
  (guaranteed by Assumption~\ref{ass:boundary-away}).
\item A phase substitution $x \mapsto t(x;h)$ reducing
  the ODE phase to $\tfrac{2}{3}(-t)^{3/2}$.
\end{enumerate}
In particular:
\[
  \psi_n(x;h)
  \;\sim\;
  h^{-1/4}\,a(x;h)\,
  \mathrm{Ai}\bigl(h^{-2/3}\,\phi(x;h)\bigr)
  + \mathcal{O}(h^\infty),
\]
where $\phi(x;h) \to \phi_0(x)$ as $h\to 0$ with $\phi_0(x_0)=0$.

\subsection{What Olver does \emph{not} provide}

\begin{remark}[The Olver--CFU gap]
\label{rem:olver-gap}
Olver does \emph{not} provide:
\begin{itemize}
\item a canonical relation or Lagrangian distribution framework;
\item FIO-class membership;
\item the global geometry of $\Lambda_h$ (branch count, self-intersections);
\item uniform transversality of $\pi_h$;
\item uniform $C^1$-control of $\phi(x;h)$ as $h \to 0$
  (pointwise convergence does not imply derivative convergence;
  classical counterexample: $f_h(x) = x + h\sin(x/h)$).
\end{itemize}
\end{remark}

%% ============================================================
\section{Branch Selection}
\label{sec:branch}
%% ============================================================

\begin{remark}[WKB two-branch structure]
\label{rem:two-branch}
In the classically allowed region $V(x) < \chi_n h^2$,
the semiclassical Lagrangian is generically
\emph{two-sheeted}:
\[
  \Lambda_h \;=\; \Lambda_h^+ \cup \Lambda_h^-,
  \qquad
  \Lambda_h^\pm := \{(x,\xi)\in T^*(-1,1) : \xi = \pm p(x;h)\},
\]
where $p(x;h) := \sqrt{\chi_n h^2 - V(x)} > 0$.
Both branches project onto the same base interval;
$\pi_h\colon\Lambda_h \to T^*(-1,1)$ is \emph{not injective}.
At the turning point $x_0$, the two branches meet:
$\Lambda_h^+ \cap \Lambda_h^- = \{(x_0, 0)\}$, the A$_2$-fold.
Thus condition~(P2) (single-branch projection) is \emph{false}
for $\Lambda_h$ as a whole, and must be established by
an explicit construction selecting one branch.
\end{remark}

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
There exists a microlocal projector
$\Pi^+\colon L^2(-1,1) \to L^2(-1,1)$,
microlocally supported on $\{\xi > 0\}$,
such that:
\begin{enumerate}[{\rm(i)}]
\item $\chi\,\psi_n(\,\cdot\,;h) = \Pi^+\bigl(\chi\,\psi_n\bigr)
  + \mathcal{O}(h^\infty)$ in $L^2$, uniformly in $h$.
\item $\Pi^+(\chi\,\psi_n)$ is a Lagrangian distribution
  associated solely to $\Lambda_h^+$,
  the positive-momentum branch.
\end{enumerate}
Consequently, $\pi_h^+\colon\Lambda_h^+ \to T^*(-1,1)$
is single-sheeted, and condition~\textup{(P2)} holds
for $\Lambda_h^+$.
\end{proposition}

\begin{remark}[Proof strategy]
The projector $\Pi^+$ is constructed as a
semiclassical pseudodifferential operator
$\Pi^+ = \mathrm{Op}_h(\sigma^+)$
where $\sigma^+(x,\xi) \in C_c^\infty(T^*(-1,1))$
is supported on $\{\xi \ge \delta_{\xi}/2\}$
for some $\delta_{\xi} > 0$
(determined by the infimum of $p(x;h)$ on $\mathrm{supp}\,\chi$,
which is strictly positive by
Assumption~\ref{ass:boundary-away}
and the eigenvalue bound $\chi_n h^2 > V(x)$
on $\mathrm{supp}\,\chi$).
The $\mathcal{O}(h^\infty)$ error in~(i) follows from
standard $h$-pseudodifferential calculus \cite{Zworski2012}.
Status: \Cond.
\end{remark}

%% ============================================================
\section{Global Geometry: Conditions (P1) and (P3)}
\label{sec:phi-conditions}
%% ============================================================

\begin{remark}[Status of conditions after branch selection]
\label{rem:p-status}
After Proposition~\ref{prop:branch-selection},
condition~(P2) is established by construction.
Two conditions remain:
\begin{enumerate}[{\rm(P1)}]
\item \textit{(Global generating function.)}
  $\phi(x;h)$ is a global generating function for
  $\Lambda_h^+$ on $\mathrm{supp}\,\chi$;
  $\Lambda_h^+$ has no self-intersections there.
  The single-branch structure (Proposition~\ref{prop:branch-selection})
  and the strict positivity $\xi = p(x;h) > 0$ on $\mathrm{supp}\,\chi$
  (by Assumption~\ref{ass:boundary-away})
  ensure this.
  Status: \Cond.
\item[{\rm(P3)}] \textit{(No secondary caustics.)}
  The A$_2$-fold at $x_0$ is the only critical point of
  $\pi_h^+\colon\Lambda_h^+ \to T^*(-1,1)$
  over $\mathrm{supp}\,\chi$.
  This is non-trivial: the endpoints $x = \pm 1$ carry
  $V(x) \to \infty$, creating potential boundary-layer
  turning-point effects.
  However, Assumption~\ref{ass:boundary-away} places
  $\mathrm{supp}\,\chi \subset [-1+\delta_{\rm bdy}, 1-\delta_{\rm bdy}]$,
  so the boundary singularities of $V$ are excluded.
  On this compact set, $\Lambda_h^+$ is smooth and the only
  zero of $\partial_\xi \pi_h^+$ is at the turning point $x_0$
  (where $p(x_0;h) = 0$, which lies in $\mathrm{supp}\,\chi$
  by assumption).
  Mode interactions as $h \to 0$ are excluded by the strict
  spectral gap between modes (SL eigenvalue separation).
  Status: \Cond\ (contingent on explicit eigenvalue separation
  estimate; required for Paper~XVII).
\end{enumerate}
\end{remark}

\section{The Olver--FIO Lift: Split Formulation}
\label{sec:lift}
%% ============================================================

\subsection{Lemma 4.1a: Microlocal Structure}

\begin{lemma}[Olver--FIO Lift: Microlocal Structure, \Cond]
\label{lem:olver-fio-lift-micro}
Under Assumption~\ref{ass:boundary-away}
and the eigenvalue-branch hypothesis:
\begin{enumerate}[{\rm(i)}]
\item $\Pi^+(\chi\,\psi_n(\,\cdot\,;h)) \in I^{-1/4,0}(\Lambda_h^+)$
  microlocally, with phase $\phi(x;h)$ from the Olver reduction.
\item $\Lambda_h^+$ carries a single A$_2$-fold at $(x_0,0)$
  with $c_3(h) \ge \delta_0 > 0$ uniformly.
\end{enumerate}
\end{lemma}

\begin{remark}[Proof strategy]
(i) FIO identification via H\"ormander/Zworski;
$h$-uniform symbol class from Olver's compact-uniform bounds.
(ii) Unique simple turning point (SL); A$_2$-type from Olver local analysis;
uniformity of $c_3(h)$ from smooth eigenbranch.
Status: \Cond.
\end{remark}

\subsection{Lemma 4.1b: Uniform Projection}

\begin{lemma}[Uniform Projection Lemma, \Open]
\label{lem:uniform-projection}
Assume Lemma~\ref{lem:olver-fio-lift-micro},
Proposition~\ref{prop:branch-selection},
conditions \textup{(P1)} and \textup{(P3)},
and Lemma~\ref{lem:diff-olver} below.
Then:
\[
  \inf_{h\in(0,h_0],\;x\in\mathrm{supp}\,\chi\setminus U_\varepsilon}
  |\phi'(x;h)| \;\ge\; \delta > 0,
\]
where $U_\varepsilon$ is an $\varepsilon$-neighborhood of $x_0$.
Consequently,
\[
  \inf_{h\in(0,h_0]}
  |\det\,d\pi_h^+(x,\xi)| \;\ge\; \delta_2 > 0
\]
on $\Lambda_h^+ \setminus U_\varepsilon$.
\end{lemma}

\begin{remark}[Proof strategy]
\begin{enumerate}[{\rm(a)}]
\item \emph{Non-vanishing.}
  At $x \ne x_0$: $\phi'(x;h) = p(x;h)/\phi(x;h)$
  (from the Olver phase equation
  $\phi(x;h)^2\phi'(x;h)^2 = p(x;h)^2$);
  both $p > 0$ and $\phi \ne 0$ on $\mathrm{supp}\,\chi \setminus U_\varepsilon$.
  Status: \Cond.
\item \emph{Uniform lower bound.}
  Requires uniform $C^1$-control of $\phi$
  (Lemma~\ref{lem:diff-olver}).
  Status: \Open\ (reduces to Lemma~\ref{lem:diff-olver}).
\end{enumerate}
\end{remark}

%% ============================================================
\section{The Differentiated Olver Estimate}
\label{sec:diff-olver}
%% ============================================================

\begin{lemma}[Differentiated Olver Estimate, \Open]
\label{lem:diff-olver}
\emph{(Main technical lemma of Paper~XVII.)}
Under Assumption~\ref{ass:boundary-away}
and the smooth-eigenbranch hypothesis
(Remark~\ref{rem:eigenbranch}):
\begin{enumerate}[{\rm(i)}]
\item \textit{($C^0$-uniform convergence.)}
  $\phi(x;h) \to \phi_0(x)$ in $C^0(\mathrm{supp}\,\chi)$
  as $h \to 0$,
  with $\|\phi(\,\cdot\,;h) - \phi_0\|_{C^0} = \mathcal{O}(h)$.
  This is contained (implicitly) in Olver \cite{Olver1974}.
\item \textit{(Uniform $C^1$-control.)}
  $\partial_x\phi(x;h) \to \phi_0'(x)$
  uniformly on $\mathrm{supp}\,\chi$ as $h \to 0$:
  \[
    \|\partial_x\phi(\,\cdot\,;h) - \phi_0'\|_{C^0(\mathrm{supp}\,\chi)}
    \;\le\; C\,h,
  \]
  with $C > 0$ independent of $h$.
  Since $\phi_0'(x) = p_0(x)/\phi_0(x) \ne 0$
  on $\mathrm{supp}\,\chi \setminus U_\varepsilon$,
  it follows that
  $\inf_{h,x}|\partial_x\phi(x;h)| \ge \delta > 0$.
\item \textit{(Uniform $C^1$-bounds on $\phi$ itself.)}
  $\|\partial_x\phi(\,\cdot\,;h)\|_{C^0} \le C$
  uniformly in $h$.
\end{enumerate}
\end{lemma}

\begin{remark}[Proof strategy]
\label{rem:diff-olver-proof}
\begin{enumerate}[{\rm(a)}]
\item \emph{Differentiated Olver equation.}
  Differentiate the Olver phase equation
  $\phi^2(\phi')^2 = p^2 = \chi_n h^2 - V(x)$
  with respect to $x$:
  \[
    2\phi(\phi')^3 + 2\phi^2\phi'\phi''
    \;=\; -V'(x).
  \]
  This is a first-order ODE for $\phi'$ given $\phi$,
  with right-hand side $-V'(x)/[2\phi(\phi')]$.
\item \emph{Grönwall control.}
  On $\mathrm{supp}\,\chi \setminus U_\varepsilon$,
  both $|\phi| \ge c_\varepsilon > 0$ (away from turning point)
  and $|\phi'|$ are bounded below (by step~(a) of
  Remark above), so the right-hand side is
  $\mathcal{O}(1)$ uniformly in $h$.
  A Gr\"onwall argument on the differentiated equation
  then yields uniform $C^0$-bounds on $\partial_x\phi$.
\item \emph{$h\to 0$ limit.}
  Since $V(x)$ and $\chi_n(h^{-1})h^2 \to \chi_n^{(0)}$
  are smooth in $x$ and converge as $h\to 0$,
  the right-hand side of the differentiated equation
  converges uniformly to that of the $h=0$ equation.
  Hence $\partial_x\phi(x;h) \to \phi_0'(x)$
  uniformly on $\mathrm{supp}\,\chi$.
\item \emph{What is new vs.\ Olver.}
  Olver \cite{Olver1974} provides $C^0$-control of $\phi$
  but not of $\partial_x\phi$.
  Part~(ii) of the lemma is a new estimate.
  A strengthened treatment appears in
  Wasow \cite{Wasow1965} for related ODE expansions;
  adaptation to the PSWF context is required.
\end{enumerate}
Status: \Open\ (main technical contribution).
\end{remark}

\section{Derivation of Assumption~A2}
\label{sec:a2-derivation}
%% ============================================================

\begin{proposition}[A2 from all inputs, \Cond]
\label{prop:a2-from-lemmas}
Assume
Lemma~\ref{lem:olver-fio-lift-micro},
Proposition~\ref{prop:branch-selection},
conditions \textup{(P1)} and \textup{(P3)},
and Lemma~\ref{lem:diff-olver}.
Then:
\[
  |\det(dt/d\beta)| \;\ge\; \delta_1 > 0
  \quad\text{uniformly in } h\in(0,h_0].
\]
\end{proposition}

\begin{proof}[Proof sketch]
By Proposition~\ref{prop:branch-selection},
restrict to the single-branch Lagrangian $\Lambda_h^+$.
By (P1), $\phi$ is a global generating function;
by (P3), there are no secondary caustics on $\mathrm{supp}\,\chi$.
The CFU change $\beta \mapsto t(\beta,u;h)$ is locally
equivalent to $\pi_h^+|_{\Lambda_h^+}$, so
$|\det(dt/d\beta)|$ is controlled by $|\phi'(x;h)|$.
Lemma~\ref{lem:diff-olver}(ii) gives
$|\phi'(x;h)| \ge \delta > 0$ uniformly,
completing the argument.
\end{proof}

\section{Target Theorem}
\label{sec:target}
%% ============================================================

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
\emph{(Contingent on Lemma~\ref{lem:diff-olver}.)}
Under Assumption~\ref{ass:boundary-away},
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly as $h \to 0$.
Consequently, the CFU reduction of Paper~XVI
applies to the PSWF family and
\textup{hyp:rigidity} holds for PSWF.
\end{theorem}

\begin{remark}[Logical dependency chain]
\label{rem:chain}
\[
  \underbrace{\text{Olver}+\text{SL}+\text{Ass.}\,\ref{ass:boundary-away}}_{{\rm[Rig/Cond]}}
  \;\Rightarrow\;
  \underbrace{\text{Lem.~4.1a}}_{{\rm[Cond]}}
  \;+\;
  \underbrace{\text{Prop.~\ref{prop:branch-selection}+}(\mathrm{P1})(\mathrm{P3})}_{{\rm[Cond]}}
  \;+\;
  \underbrace{\text{Lem.~\ref{lem:diff-olver}}}_{{\rm[Open]}}
  \;\Rightarrow\;
  \underbrace{\text{Lem.~4.1b}}_{{\rm[Open\to Cond]}}
  \;\Rightarrow\;
  \underbrace{\text{Prop.~\ref{prop:a2-from-lemmas}}}_{{\rm[Cond]}}
  \;\Rightarrow\;
  \underbrace{\text{Thm.~\ref{thm:xvii-main}}}_{{\rm[Cond]}}.
\]

(Lemma~\ref{lem:diff-olver}(ii)--(iii)).
\end{remark}

\begin{remark}[Scope]

conjecture; does not give global (non-microlocal) bounds;
