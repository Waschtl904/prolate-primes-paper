\title[A$_2$-Stability for PSWF Amplitudes]{%
  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
(conditions (A1)--(A3)) uniformly as $h := c^{-1} \to 0$.

\medskip
\noindent\textbf{Recall (Assumption~A2).}
A Lagrangian $\Lambda_h$ satisfies Assumption~A2 if:
(A1) the cubic coefficient of the A$_2$-normal form is
  $c_3(h)\ge\delta_0>0$ uniformly;
(A2) the CFU Jacobian satisfies $|\det(dt/d\beta)|\ge\delta_1>0$ uniformly;
(A3) there is a unique fold on $\mathrm{supp}\,\chi$.
In 1D the fold condition means: a unique turning point where
$\partial_\xi\sigma=0$ and $\partial^2_\xi\sigma\ne 0$ (A$_2$-type).

\medskip
\noindent\textbf{Main technical contribution.}
Lemmas~\ref{lem:energy-comparison}--\ref{lem:phi-C1}
establish uniform $C^1$-control of the Olver phase $\phi(x;h)$
via the integral representation
\[
  \phi(x;h)=\mathrm{sgn}(x-x_0(h))\cdot
  \Bigl(\tfrac{3}{2}\int_{x_0(h)}^x p(s;h)\,ds\Bigr)^{2/3},
  \quad p:=\sqrt{E_n(h)-V(x)}.
\]
No ODE, no Gr\"onwall, no Wasow expansion.
\end{abstract}

Airy is local; CFU admissibility is global and uniform in $h$.
The gap is closed by
Proposition~\ref{prop:branch-selection},

and Lemmas~\ref{lem:energy-comparison}--\ref{lem:cfu-jacobian-1d}.
\end{remark}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF]
$\psi_n$ satisfy $(h^2\mathcal{L}_{h^{-1}})\psi_n=E_n(h)\psi_n$
with $V(x)=x^2/(1-x^2)$, $E_n(h):=\chi_n h^2$.
Turning points: $V(x_\pm(h))=E_n(h)$.
\end{definition}

\begin{remark}[Geometric and spectral positivity of $p$]
\label{rem:p-positivity}
The positivity $p(x;h)>0$ on $K_\varepsilon$ has two independent components:

\medskip
\noindent\textit{(i) Geometric part (\Rig).}
Since $V$ is continuous and $K_\varepsilon=\mathrm{supp}\,\chi\setminus U_\varepsilon$
excludes an $\varepsilon$-neighborhood of the limiting turning point $x_0^{(0)}$:
\[
  \inf_{x\in K_\varepsilon}(\lambda_n-V(x))\ge c_\varepsilon^{\rm geo}>0.
\]
This depends only on $\lambda_n$, $V$, and $\varepsilon$, not on $h$.

\medskip
\noindent\textit{(ii) Spectral part (\Cond).}
$E_n(h)\to\lambda_n$ smoothly (SL analyticity), so for $h\le h_0$:
\[
  \inf_{h\le h_0,\,x\in K_\varepsilon}(E_n(h)-V(x))
  \ge c_\varepsilon^{\rm geo}-|E_n(h)-\lambda_n|
  \ge\tfrac{1}{2}c_\varepsilon^{\rm geo}>0.
\]
Define $\delta_\xi(\varepsilon):=\sqrt{c_\varepsilon^{\rm geo}/2}>0$;
then $p(x;h)=\sqrt{E_n(h)-V(x)}\ge\delta_\xi>0$ on $K_\varepsilon$
for all $h\le h_0$.

\medskip
\textbf{All constants below are independent of $h$
(depending only on $\varepsilon,V,\lambda_n$)
and all bounds hold on $K_\varepsilon$ unless stated otherwise.}
\end{remark}

\begin{remark}[Turning-point shift]
\label{rem:eigenbranch}
$E_n(h)\to\lambda_n\in(0,1)$ smoothly;
$x_0(h)\to x_0^{(0)}$ smoothly by IFT
($V(x_0(h))=E_n(h)$, $V'(x_0^{(0)})\ne 0$).
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
$\mathrm{supp}\,\chi\subset[-1+\delta_{\rm bdy},1-\delta_{\rm bdy}]$
independently of $h$.
Consequences: $V,V',V''\in L^\infty$; boundary layers excluded.
\end{assumption}

\section{Olver's Scheme and the CFU Gap}
\label{sec:olver}
%% ============================================================

Olver \cite{Olver1974} Ch.~10--11:
$\psi_n\sim h^{-1/4}a\,\mathrm{Ai}(h^{-2/3}\phi)+\mathcal{O}(h^\infty)$,
uniform on $\mathrm{supp}\,\chi$.

\begin{remark}[Olver--CFU gap]
\label{rem:olver-gap}
Olver does not provide Lagrangian/FIO structure,
global geometry of $\Lambda_h$, or uniform $C^1$-control of $\phi$.
Classical warning: $f_h=x+h\sin(x/h)\to x$ uniformly
but $f_h'$ does not.
\end{remark}

\section{Branch Selection}
\label{sec:branch}
%% ============================================================

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
Fix $\varepsilon>0$.
Since $\psi_n$ is an eigenfunction,
semiclassical propagation gives
$\mathrm{WF}_h(\psi_n)\subset\{\xi^2+V=E_n(h)\}=\Lambda_h^+\cup\Lambda_h^-$
(\cite{Zworski2012} Ch.~12).
On $K_\varepsilon$: $\Lambda_h^\pm\subset\{\pm\xi\ge\delta_\xi\}$ (uniform gap $2\delta_\xi$).
A microlocal partition of unity yields $\Pi^+=\mathrm{Op}_h(\sigma^+)$ such that:
(i) $\chi\psi_n=\Pi^+(\chi\psi_n)+\mathcal{O}(h^\infty)$ in $L^2$;
(ii) $\Pi^+(\chi\psi_n)\in I^{-1/4,0}(\Lambda_h^+)$.
\textit{Caveat:} branch separation holds only on $K_\varepsilon$.
\end{proposition}

\section{Global Geometry: (P1) and (P3)}
\label{sec:phi-conditions}
%% ============================================================

\begin{remark}[(P1)]
$\xi=p>0$ single-valued on $K_\varepsilon\Rightarrow\phi$ global generating function. \Cond.
\end{remark}

\begin{remark}[(P3): one turning point; spectral gap]
\label{rem:p3}
\textit{Unique turning point (\Rig, geometric).}
$V$ is strictly monotone on $\mathrm{supp}\,\chi$,
so $E_n(h)-V(x)=0$ has exactly one solution $x_0(h)$.

\medskip
\noindent\textit{Spectral gap (\Cond, for frequency-overlap exclusion).}
\begin{equation}\label{eq:specgap}
  \mathrm{dist}(E_n(h),\sigma(\mathcal{L}_h)\setminus\{E_n(h)\})\ge Ch.
\end{equation}
Follows from SL asymptotics \cite{Olver1974} Ch.~12.
\end{remark}

%% ============================================================
\section{CFU Construction on $K_\varepsilon^\pm$ (Separate)}
\label{sec:cfu-components}
%% ============================================================

\begin{remark}[CFU on each component]
\label{rem:cfu-components}
$K_\varepsilon$ has two connected components:
$K_\varepsilon^\pm:=K_\varepsilon\cap\{\pm(x-x_0(h))>0\}$.
On $K_\varepsilon^+$: $I(x;h)>0$,
so $\phi=(\tfrac{3}{2}I)^{2/3}$ is positive and smooth.
On $K_\varepsilon^-$: $I(x;h)<0$,
so $\phi=-(\tfrac{3}{2}|I|)^{2/3}$ is negative and smooth.
The CFU construction of \cite{CFU1957} is performed
\emph{separately on each component};
no global branch of $z^{2/3}$ over the turning point is needed.
The two branches are matched at $x_0(h)$ by the Airy transition,
which is handled by Paper~XVI's thm:CFU-geom.
\end{remark}

%% ============================================================
\section{The Differentiated Olver Estimate (Decomposed)}
\label{sec:diff-olver}
%% ============================================================

\begin{lemma}[Monotonicity of $I$, \Rig]
\label{lem:I-monotone}
$p>0$ on $K_\varepsilon$, so $I(x;h)=\int_{x_0(h)}^x p\,ds$ is strictly monotone,
$\mathrm{sgn}(I)=\mathrm{sgn}(x-x_0(h))$,
$|I(x;h)|\ge\delta_\xi\cdot|x-x_0(h)|$.
\end{lemma}

\begin{lemma}[Energy comparison, \Cond]
\label{lem:energy-comparison}
Since $p,p_0\ge\delta_\xi,c_\varepsilon>0$:
\[
  |p(x;h)-p_0(x)|
  =\frac{|E_n(h)-\lambda_n|}{p+p_0}
  \le\frac{|E_n(h)-\lambda_n|}{\delta_\xi+c_\varepsilon}
  =:A_\varepsilon(h)=\mathcal{O}(h).
\]
(Lipschitz of $\sqrt{\,\cdot\,}$ fails at $p_0=0$;
rationalization is the correct argument.)
\end{lemma}

\begin{lemma}[Integral stability, \Cond]
\label{lem:integral-stability}
\[
  |I(x;h)-I_0(x)|
  \le\|p\|_{C^0}\cdot|x_0(h)-x_0^{(0)}|
  +|K_\varepsilon|\cdot A_\varepsilon(h)
  =:\tilde A_\varepsilon(h)=\mathcal{O}(h),
\]
using Remark~\ref{rem:eigenbranch} and Lemma~\ref{lem:energy-comparison}.
\end{lemma}

\begin{lemma}[Non-degeneracy of $I$ and $\phi$, \Cond]
\label{lem:nondeg}
By Lemma~\ref{lem:I-monotone} and $|x-x_0(h)|\ge\varepsilon/2$ on $K_\varepsilon$:
\[
  |I(x;h)|,|I_0(x)|\ge c_\varepsilon':=\delta_\xi\varepsilon/2>0;
  \quad
  |\phi(x;h)|\ge c_\varepsilon'':=(\tfrac{3}{4}\delta_\xi\varepsilon)^{2/3};
  \quad
  |\phi_0(x)|\ge d_\varepsilon:=(\tfrac{3}{4}c_\varepsilon\varepsilon)^{2/3}>0.
\]
\end{lemma}

\begin{lemma}[$C^0$-stability of $\phi$, \Cond]
\label{lem:phi-C0}
On each component $K_\varepsilon^\pm$ (Remark~\ref{rem:cfu-components}),
$I(x;h)$ has fixed sign,
so $I\mapsto(\tfrac{3}{2}I)^{2/3}$ is smooth.
Since $|I|,|I_0|\ge c_\varepsilon'>0$ (Lemma~\ref{lem:nondeg}),
this map is Lipschitz on $[c_\varepsilon',\infty)$
(resp.\ $(-\infty,-c_\varepsilon']$)
with constant
\[
  L_\varepsilon=\tfrac{2}{3}\bigl(\tfrac{3}{2}\bigr)^{2/3}(c_\varepsilon')^{-1/3}.
\]
Therefore:
\[
  B_\varepsilon(h):=\|\phi(\cdot;h)-\phi_0\|_{C^0(K_\varepsilon)}
  \le L_\varepsilon\cdot\tilde A_\varepsilon(h)=\mathcal{O}(h).
\]
\end{lemma}

\begin{lemma}[$C^1$-stability of $\phi$, \Cond]
\label{lem:phi-C1}
On each $K_\varepsilon^\pm$,
differentiating $\phi^3=(\tfrac{3}{2})^3 I^2$ gives
$\phi'=p/\phi$ and $\phi_0'=p_0/\phi_0$
(well-defined: $|\phi|\ge c_\varepsilon''>0$, $|\phi_0|\ge d_\varepsilon>0$).
\[
  |\phi'-\phi_0'|
  \le\frac{A_\varepsilon(h)}{c_\varepsilon''}
  +\frac{p_0^{\max}\cdot B_\varepsilon(h)}{c_\varepsilon''\cdot d_\varepsilon}
  \to 0
\]
uniformly (Lemmas~\ref{lem:energy-comparison} and \ref{lem:phi-C0}).
Since $\phi_0'=p_0/\phi_0\ge c_\varepsilon/d_\varepsilon>0$:
\[
  \inf_{h\le h_0,\,x\in K_\varepsilon}|\phi'(x;h)|
  \ge\tfrac{1}{2}\inf_{K_\varepsilon}|\phi_0'|>0.
\]
\end{lemma}

\section{CFU Jacobian in 1D}
\label{sec:cfu-jacobian}
%% ============================================================

\begin{lemma}[CFU Jacobian reduction in 1D, \Cond]
\label{lem:cfu-jacobian-1d}
In the 1D A$_2$ setting,
the CFU Jacobian $|\det(dt/d\beta)|$
is the absolute Jacobian of the map
$(x,\xi)\mapsto(\phi(x;h),p(x;h)/\phi(x;h))$
restricted to $\Lambda_h^+$.
On $\Lambda_h^+$ with parametrisation $x\mapsto(x,p(x;h))$:
\[
  |\det(dt/d\beta)|
  = \left|\frac{d}{dx}\phi(x;h)\right|
  = |\phi'(x;h)|.
\]
Lemma~\ref{lem:phi-C1} gives
$|\phi'(x;h)|\ge\delta>0$ uniformly on $K_\varepsilon$,
hence $|\det(dt/d\beta)|\ge\delta_1:=\delta>0$.
\end{lemma}

\begin{remark}[Why 1D reduces to $|\phi'|$]
\label{rem:jacobian-reduction}
In 1D, the Lagrangian $\Lambda_h^+$ is a curve
parametrised by $x$; the projection $\pi:T^*\mathbb{R}\to\mathbb{R}$
restricts to a map $x\mapsto\phi(x;h)$.
The CFU Jacobian (determinant of the mixed Hessian
$\partial^2_{x\theta}\Phi$ of the generating function $\Phi$)
reduces to $\phi'(x;h)$ because $\Phi(x,\theta)=\phi(x;h)\theta-\tfrac{1}{3}\theta^3$
and $\partial^2_{x\theta}\Phi=\phi'(x;h)$.
\end{remark}

\begin{proposition}[Assumption~A2, \Cond]
\label{prop:a2-from-lemmas}

(A3) unique fold: Remark~\ref{rem:p3} (monotone $V$).
\end{proposition}

\section{Target Theorem}
\label{sec:target}
%% ============================================================

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
Under Assumption~\ref{ass:boundary-away},
the smooth-eigenbranch hypothesis, and~\eqref{eq:specgap},
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly on $K_\varepsilon$ as $h\to 0$.
\end{theorem}

\begin{remark}[Dependency chain]
\[
  \bigl[\lambda_n{-}V\ge c^{\rm geo}_{\varepsilon},\;
  E_n\to\lambda_n,\;
  V'(x_0^{(0)})\ne 0,\;
  \text{\eqref{eq:specgap}},\;
  \mathrm{WF}_h\subset\Lambda_h^\pm\bigr]
  \;\Rightarrow\;
  \bigl[\text{Lems.}~\ref{lem:energy-comparison}\text{--}\ref{lem:phi-C1}\bigr]
  \;\Rightarrow\;
  \text{Lem.}~\ref{lem:cfu-jacobian-1d}
  \;\Rightarrow\;
  \text{Prop.}~\ref{prop:a2-from-lemmas}
  \;\Rightarrow\;
  \text{Thm.}~\ref{thm:xvii-main}.
\]
\end{remark}

Paper~XVII does not address the full inverse microlocal conjecture,
global PSWF bounds, or concentration-of-energy.
\end{remark}
