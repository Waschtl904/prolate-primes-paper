\title[A$_2$-Stability for PSWF Amplitudes]{%
  A$_2$-Stability for Prolate Spheroidal Wave Function Amplitudes\\
  in the Semiclassical Limit:\\
  Proof Architecture Skeleton}

\author{[Author]}

\begin{abstract}
This is the \emph{Level~0--1 proof architecture skeleton} for Paper~XVII.
It fixes the logical structure and identifies the critical bottlenecks
before full LaTeX elaboration.

\medskip
\noindent\textbf{Objective.}
Verify Hypothesis~\textup{hyp:rigidity} from Paper~XVI
for PSWF amplitudes:
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
(conditions (A1)--(A3)) uniformly as $h := c^{-1} \to 0$.

\medskip
\noindent\textbf{Central new objects.}
The \emph{Olver--FIO Lift} is split into two logically independent lemmas:
\begin{itemize}
\item Lemma~\ref{lem:olver-fio-lift-micro}
  (\emph{Microlocal Structure}):
  Lagrangian structure and A$_2$-fold from Olver $+$ Sturm--Liouville.
  Status: \Cond.
\item Lemma~\ref{lem:uniform-projection}
  (\emph{Uniform Projection}):
  the projection $\pi_h^+\colon\Lambda_h^+\to T^*(-1,1)$
  on the positive-momentum branch is uniformly transversal.
  Status: \Cond\ (follows from Lemma~\ref{lem:diff-olver}).
\end{itemize}
The \emph{Differentiated Olver Estimate}
(Lemma~\ref{lem:diff-olver}) is the main technical contribution.
Its proof proceeds via the \emph{integral representation}
\[
  \phi(x;h)
  = \mathrm{sgn}(x-x_0(h))\cdot
    \Bigl(\tfrac{3}{2}\int_{x_0(h)}^x p(s;h)\,ds\Bigr)^{2/3},
  \quad
  p(x;h) := \sqrt{E_n(h)-V(x)},
\]
reducing all $C^1$-control of $\phi$ to
uniform control of $p(x;h)$ and its integral.
No ODE, no Gr\"onwall inequality, and no Wasow-type expansion is needed.

\medskip
\noindent\textbf{Meta-remark.}
The proof shows that CFU stability depends only on
energy convergence $E_n(h)\to\lambda_n$,
$L^\infty$-control of $V$ on $\mathrm{supp}\,\chi$,
and the distance from the turning point.
This is verifiably general beyond the PSWF setting.

\medskip
\noindent\textbf{Strategy.}
Primary: Olver turning-point analysis (Ch.~10--11 of \cite{Olver1974}).
Secondary: FIO reinterpretation of Olver expansions.
Auxiliary: Sturm--Liouville spectral theory; boundary-away cutoff.
\end{abstract}

Lemma~\ref{lem:diff-olver}.
\end{remark}

\section{PSWF Setup and Standing Assumptions}
\label{sec:pswf}
%% ============================================================

\begin{definition}[PSWF as semiclassical eigenfamily]\label{def:pswf}
$\psi_n(x;c)$ are eigenfunctions of
$\mathcal{L}_c = -\tfrac{d}{dx}[(1-x^2)\tfrac{d}{dx}] + c^2x^2$.
With $h = c^{-1}$:
\[
  h^2\mathcal{L}_{h^{-1}}
  = -h^2\tfrac{d}{dx}[(1-x^2)\tfrac{d}{dx}] + x^2,
\]
a semiclassical Schr\"odinger operator with effective potential
$V(x) = x^2/(1-x^2)$.
The turning points $x_\pm(h)$ solve $V(x_\pm) = E_n(h)$,
where $E_n(h) := \chi_n(h^{-1})\cdot h^2$.
\end{definition}

\begin{remark}[Eigenvalue branch, momentum limit, turning-point smoothness]
\label{rem:eigenbranch}
$E_n(h) \to \lambda_n \in (0,1)$ smoothly and monotonically as $h\to 0$.
Define the limiting momentum $p_0(x) := \sqrt{\lambda_n - V(x)}$.

\medskip
\noindent\textit{Uniform positivity of $p_0$ on $K_\varepsilon$.}
For every $\varepsilon > 0$, on
$K_\varepsilon := \mathrm{supp}\,\chi\setminus U_\varepsilon$
(where $U_\varepsilon$ excludes an $\varepsilon$-neighborhood of $x_0^{(0)}$
and Assumption~\ref{ass:boundary-away} excludes $x\to\pm 1$):
\[
  \inf_{x\in K_\varepsilon} p_0(x) \;\ge\; c_\varepsilon > 0.
\]

\medskip
\noindent\textit{Smooth dependence of turning point $x_0(h)$.}
The limiting turning point $x_0^{(0)}$ satisfies $V(x_0^{(0)}) = \lambda_n$.
Since $V'(x_0^{(0)}) \ne 0$ (V is strictly increasing near $x_0^{(0)}$),
the implicit function theorem applies to $V(x_0(h)) = E_n(h)$:
the map $h \mapsto x_0(h)$ is smooth on $(0,h_0]$
and $x_0(h) \to x_0^{(0)}$ as $h\to 0$.
\end{remark}

\begin{assumption}[Boundary-away cutoff]
\label{ass:boundary-away}
Fix $\delta_{\rm bdy}>0$.
All constructions are localized to
$\mathrm{supp}\,\chi \subset [-1+\delta_{\rm bdy},\,1-\delta_{\rm bdy}]$,
where $\chi\in C_c^\infty(-1,1)$ is fixed independently of $h$.
On this set: $V, V', V'' \in L^\infty$ uniformly,
ruling out boundary-layer effects and $V \to \infty$.
All statements below are relative to this assumption.
\end{assumption}

\section{Olver's Uniform Asymptotic Scheme}
\label{sec:olver}
%% ============================================================

\subsection{What Olver provides}

Olver \cite{Olver1974} (Ch.~10--11) provides uniform
asymptotics matching WKB and Airy regimes:
\[
  \psi_n(x;h)
  \sim h^{-1/4}\,a(x;h)\,
  \mathrm{Ai}(h^{-2/3}\phi(x;h))
  + \mathcal{O}(h^\infty),
\]
with uniform error bounds on $\mathrm{supp}\,\chi$
(guaranteed by Assumption~\ref{ass:boundary-away}).

\subsection{What Olver does not provide}

\begin{remark}[Olver--CFU gap]
\label{rem:olver-gap}
Olver does not provide:
a Lagrangian distribution framework;
FIO-class membership;
global geometry of $\Lambda_h$;
uniform transversality of $\pi_h$;
or uniform $C^1$-control of $\phi(x;h)$.
Classical warning:
$f_h(x) = x + h\sin(x/h)$ converges uniformly
but $f_h'(x) = 1 + \cos(x/h)$ does not.
\end{remark}

\section{Branch Selection}
\label{sec:branch}
%% ============================================================

\begin{remark}[WKB two-branch structure]
\label{rem:two-branch}
In the classically allowed region $E_n(h) > V(x)$:
\[
  \Lambda_h = \Lambda_h^+ \cup \Lambda_h^-,
  \quad
  \Lambda_h^\pm = \{\xi = \pm p(x;h)\}.
\]
Both branches project to the same base interval;
$\pi_h$ is not injective on $\Lambda_h$.
Condition (P2) is false for $\Lambda_h$ as a whole
and must be achieved by explicit construction.
\end{remark}

\begin{proposition}[Branch selection, \Cond]
\label{prop:branch-selection}
Let
$\delta_\xi := \tfrac{1}{2}\inf_{h\in(0,h_0],\,x\in\mathrm{supp}\,\chi} p(x;h) > 0$
(uniform positivity holds by Assumption~\ref{ass:boundary-away}
and $E_n(h) > V(x)$ on $\mathrm{supp}\,\chi$ for $h\le h_0$).
There exists a semiclassical pseudodifferential operator
$\Pi^+ = \mathrm{Op}_h(\sigma^+)$
with symbol $\sigma^+\in C_c^\infty(T^*(-1,1))$
supported on $\{\xi \ge \delta_\xi/2\}$,
such that:
\begin{enumerate}[{\rm(i)}]
\item $\chi\,\psi_n(\cdot;h) = \Pi^+(\chi\psi_n) + \mathcal{O}(h^\infty)$
  in $L^2$, uniformly in $h$.
\item $\Pi^+(\chi\psi_n)$ is a Lagrangian distribution
  on $\Lambda_h^+$ alone.
\end{enumerate}
Consequently $\pi_h^+\colon\Lambda_h^+\to T^*(-1,1)$ is single-sheeted.
\end{proposition}

\begin{remark}[Proof: ellipticity of the symbol away from $\xi=0$]
\label{rem:branch-proof}
The key input is that the two branches
$\Lambda_h^\pm$ are \emph{microlocally separated}:
$\Lambda_h^+\subset\{\xi\ge\delta_\xi\}$
and $\Lambda_h^-\subset\{\xi\le -\delta_\xi\}$,
with $\delta_\xi > 0$ uniform in $h$ by the positivity of $p$.
The symbol $\sigma^+$ is elliptic on $\{\xi\ge\delta_\xi\}$
and vanishes on $\{\xi\le 0\}$;
no frequency overlap between $\Lambda_h^+$ and $\Lambda_h^-$
occurs on $\mathrm{supp}\,\chi$.
The $\mathcal{O}(h^\infty)$ error follows from the
standard $h$-pseudodifferential Egorov theorem \cite{Zworski2012}.
Status: \Cond.
\end{remark}

\section{The Olver--FIO Lift}

\subsection{Lemma 4.1a: Microlocal Structure}

\begin{lemma}[Microlocal Structure, \Cond]
\label{lem:olver-fio-lift-micro}
Under Assumption~\ref{ass:boundary-away}
and the smooth-eigenbranch hypothesis:
\begin{enumerate}[{\rm(i)}]
\item $\Pi^+(\chi\psi_n(\cdot;h))\in I^{-1/4,0}(\Lambda_h^+)$
  with phase $\phi(x;h)$.
\item $\Lambda_h^+$ carries a single A$_2$-fold at $(x_0,0)$
  with $c_3(h)\ge\delta_0>0$ uniformly.
\end{enumerate}
\end{lemma}

\subsection{Lemma 4.1b: Uniform Projection}

\begin{lemma}[Uniform Projection, \Cond]
\label{lem:uniform-projection}
Assume Lemma~\ref{lem:olver-fio-lift-micro},
Proposition~\ref{prop:branch-selection},
(P1), (P3), and Lemma~\ref{lem:diff-olver}.
Then, on $K_\varepsilon$:
\[
  \inf_{h\in(0,h_0],\,x\in K_\varepsilon}|\phi'(x;h)| \ge \delta > 0,
\]
hence $|\det\,d\pi_h^+|\ge\delta_2>0$ uniformly.
\end{lemma}

\begin{remark}[Reduction to Lemma~\ref{lem:diff-olver}]
On $K_\varepsilon$: $\phi'(x;h) = p(x;h)/\phi(x;h)$.
Numerator $p > \delta_\xi > 0$ uniformly
(Proposition~\ref{prop:branch-selection}).
Denominator $|\phi| \ge c_\varepsilon'' > 0$
(Lemma~\ref{lem:diff-olver}(C)).
Uniform convergence $\phi'\to\phi_0'$ from Lemma~\ref{lem:diff-olver}(D).
Status: \Cond.
\end{remark}

%% ============================================================
\section{The Differentiated Olver Estimate}
\label{sec:diff-olver}
%% ============================================================

\begin{lemma}[Differentiated Olver Estimate, \Cond]
\label{lem:diff-olver}
\emph{(Main technical lemma of Paper~XVII.)}
Assume Assumption~\ref{ass:boundary-away}
and the smooth-eigenbranch hypothesis (Remark~\ref{rem:eigenbranch}).
Fix $\varepsilon > 0$ and set
$K_\varepsilon := \mathrm{supp}\,\chi\setminus U_\varepsilon$.
Define:
\[
  p(x;h) := \sqrt{E_n(h)-V(x)},
  \qquad
  I(x;h) := \int_{x_0(h)}^x p(s;h)\,ds,
\]
\[
  \phi(x;h) := \mathrm{sgn}(x-x_0(h))\cdot
  \Bigl(\tfrac{3}{2}\,I(x;h)\Bigr)^{2/3},
\]
with the positive branch of $\sqrt{\,\cdot\,}$
(consistent with Proposition~\ref{prop:branch-selection}).
Then:

\begin{enumerate}[{\rm(A)}]
\item \textit{(Uniform convergence of $p$, via rationalization.)}
  Let $\delta_\xi > 0$ be as in
  Proposition~\ref{prop:branch-selection}.
  Since $p(x;h) \ge \delta_\xi > 0$ and
  $p_0(x) \ge c_\varepsilon > 0$ on $K_\varepsilon$
  (Remark~\ref{rem:eigenbranch}),
  we may rationalize:
  \[
    |p(x;h) - p_0(x)|
    = \frac{|p(x;h)^2 - p_0(x)^2|}{p(x;h) + p_0(x)}
    = \frac{|E_n(h) - \lambda_n|}{p(x;h) + p_0(x)}
    \;\le\; \frac{|E_n(h)-\lambda_n|}{2\delta_\xi}.
  \]
  Since $E_n(h) \to \lambda_n$ as $h\to 0$:
  \[
    \|p(\cdot;h) - p_0\|_{C^0(K_\varepsilon)}
    \;\le\; \frac{|E_n(h)-\lambda_n|}{2\delta_\xi}
    \;=\; \mathcal{O}(h).
  \]
  \emph{Note:} The rationalization uses positivity of both $p$ and $p_0$;
the Lipschitz argument via $\sqrt{\,\cdot\,}$ would fail near $p_0 = 0$.

\item \textit{(Uniform convergence of $I$, with turning-point shift.)}
  Since $x_0(h)\to x_0^{(0)}$ smoothly
  (by the implicit function theorem:
  $V'(x_0^{(0)}) \ne 0$, so $V(x_0(h)) = E_n(h)$
  determines $x_0(h)$ as a smooth function of $E_n(h)$;
  see Remark~\ref{rem:eigenbranch}):
  \begin{align*}
    |I(x;h) - I_0(x)|
    &\le \int_{\min(x_0(h),x_0^{(0)})}^{\max(x_0(h),x_0^{(0)})}
         p(s;h)\,ds
         + \int_{x_0^{(0)}}^x |p(s;h)-p_0(s)|\,ds \\
    &\le \|p(\cdot;h)\|_{C^0}\cdot|x_0(h)-x_0^{(0)}|
         + |K_\varepsilon|\cdot\|p(\cdot;h)-p_0\|_{C^0(K_\varepsilon)} \\
    &= \mathcal{O}(h).
  \end{align*}

\item \textit{(Denominator bounded away from zero,
  via linear growth of $I_0$.)}
  Since $p_0 > 0$ on $\mathrm{supp}\,\chi$
  (uniformly positive; Remark~\ref{rem:eigenbranch}),
  the integral $I_0(x) = \int_{x_0^{(0)}}^x p_0(s)\,ds$
  grows linearly away from the turning point:
  \[
    |I_0(x)| \ge c_\varepsilon \cdot |x - x_0^{(0)}|
    \ge c_\varepsilon \cdot \varepsilon
    \qquad \text{for } x\in K_\varepsilon.
  \]
  By (B), $|I(x;h) - I_0(x)| = \mathcal{O}(h)$,
  so for $h\le h_0$ small enough:
  $|I(x;h)| \ge \tfrac{1}{2}c_\varepsilon\varepsilon =: c_\varepsilon' > 0$.
  Hence
  $|\phi(x;h)| = (\tfrac{3}{2}|I(x;h)|)^{2/3}
  \ge (\tfrac{3}{4}c_\varepsilon\varepsilon)^{2/3} =: c_\varepsilon'' > 0$
  uniformly.

\item \textit{(Uniform $C^1$-control via quotient estimate.)}
  On $K_\varepsilon$:
  $\phi'(x;h) = p(x;h)/\phi(x;h)$ and
  $\phi_0'(x) = p_0(x)/\phi_0(x)$.
  Therefore:
  \[
    |\phi'(x;h) - \phi_0'(x)|
    \le
    \frac{|p(x;h)-p_0(x)|}{|\phi(x;h)|}
    +
    p_0(x)\cdot
    \frac{|\phi(x;h)-\phi_0(x)|}{|\phi(x;h)|\cdot|\phi_0(x)|}.
  \]
  First term $\le \mathcal{O}(h)/c_\varepsilon'' \to 0$ by (A),(C).
  Second term $\to 0$ by (B),(C).
  Hence
  $\sup_{K_\varepsilon}|\phi'(\cdot;h)-\phi_0'| \to 0$.
  Since $\phi_0' = p_0/\phi_0 \ge c_\varepsilon/c_\varepsilon'' > 0$
  on $K_\varepsilon$:
  \[
    \inf_{h\in(0,h_0],\,x\in K_\varepsilon}|\phi'(x;h)|
    \;\ge\; \tfrac{1}{2}\inf_{K_\varepsilon}|\phi_0'| > 0.
  \]
\end{enumerate}
\end{lemma}

\begin{remark}[Robustness and generality]
\label{rem:robustness}
The proof uses only:
\begin{enumerate}[(i)]
\item $E_n(h)\to\lambda_n$ smoothly (eigenbranch);
\item $V\in L^\infty$ on $\mathrm{supp}\,\chi$ (Assumption~\ref{ass:boundary-away});
\item Rationalization of $|p-p_0|$ via uniform positivity
  $p,p_0\ge c_\varepsilon>0$ on $K_\varepsilon$ --- this is the
  key step that replaces the Lipschitz argument;
\item Monotonicity of $I(x;h)$ in $x$ (since $p > 0$),
  giving linear growth $|I_0(x)|\ge c_\varepsilon|x-x_0^{(0)}|$;
\item Smooth turning-point shift $x_0(h)\to x_0^{(0)}$
  via the implicit function theorem.
\end{enumerate}
No ODE, no Gr\"onwall inequality, no Wasow-type expansion.
The argument is general: it applies to any semiclassical
Schr\"odinger operator with a simple turning point,
provided $V\in L^\infty$ and the energy converges smoothly.
\end{remark}

\section{Derivation of Assumption~A2}
\label{sec:a2-derivation}
%% ============================================================

\begin{proposition}[A2 from all inputs, \Cond]
\label{prop:a2-from-lemmas}
Under Lemma~\ref{lem:olver-fio-lift-micro},
Proposition~\ref{prop:branch-selection},
(P1), (P3), and Lemma~\ref{lem:diff-olver}:
\[
  |\det(dt/d\beta)| \ge \delta_1 > 0
  \quad\text{uniformly in } h\in(0,h_0].
\]
\end{proposition}

\begin{proof}[Proof sketch]
Restrict to $\Lambda_h^+$ (Proposition~\ref{prop:branch-selection}).
By (P1), $\phi$ is a global generating function;
by (P3), no secondary caustics on $\mathrm{supp}\,\chi$.
The CFU map $\beta\mapsto t(\beta,u;h)$ is locally
equivalent to $\pi_h^+|_{\Lambda_h^+}$,
so $|\det(dt/d\beta)|$ is controlled by $|\phi'(x;h)|$.
Lemma~\ref{lem:diff-olver}(D) gives
$|\phi'(x;h)|\ge\delta>0$ uniformly.
\end{proof}

\section{Target Theorem}

\begin{theorem}[PSWF A$_2$-stability, \Cond]
\label{thm:xvii-main}
Under Assumption~\ref{ass:boundary-away}
and the smooth-eigenbranch hypothesis,
the PSWF amplitude satisfies Assumption~A2 of Paper~XVI
uniformly as $h\to 0$.
Consequently, the CFU reduction of Paper~XVI
applies to the PSWF family and
\textup{hyp:rigidity} (XVI) holds.
\end{theorem}
